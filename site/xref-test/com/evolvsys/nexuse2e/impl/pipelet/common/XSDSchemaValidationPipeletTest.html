24      @Test
25      void testValidationWithSchemaAsParameter() throws NexusException {
26          XSDSchemaValidationPipelet xsdSchemaValidationPipelet = new XSDSchemaValidationPipelet();
27          TestMessageContext testMessageContext = new TestMessageContext();
28  
29          Parameters parameters = new Parameters();
30          parameters.put(XSDSchemaValidationPipelet.PARAM_SCHEMA_FILE_CONTENT, getSchema());
31  
32          xsdSchemaValidationPipelet.setParameters(parameters);
33          ExecutionContext executionContext = getExecutionContext(testMessageContext);
34  
35          addValidPayload(executionContext);
36          xsdSchemaValidationPipelet.execute(executionContext);
37      }
38  
39      @Test
40      void testValidationWithSchemaAsParameterFailing() {
41          XSDSchemaValidationPipelet xsdSchemaValidationPipelet = new XSDSchemaValidationPipelet();
42          TestMessageContext testMessageContext = new TestMessageContext();
43  
44          Parameters parameters = new Parameters();
45          parameters.put(XSDSchemaValidationPipelet.PARAM_SCHEMA_FILE_CONTENT, getSchema());
46  
47          xsdSchemaValidationPipelet.setParameters(parameters);
48          ExecutionContext executionContext = getExecutionContext(testMessageContext);
49  
50          addInvalidPayload(executionContext);
51  
52          assertThatThrownBy(() -> xsdSchemaValidationPipelet.execute(executionContext))
53                  .isInstanceOf(NexusException.class)
54                  .hasMessageContaining("Payload validation failed");
55      }

57      @Test
58      void testValidationInvalidButDisabled() {
59          XSDSchemaValidationPipelet xsdSchemaValidationPipelet = new XSDSchemaValidationPipelet();
60          TestMessageContext testMessageContext = new TestMessageContext();
61  
62          Parameters parameters = new Parameters();
63          parameters.put(XSDSchemaValidationPipelet.PARAM_SCHEMA_FILE_CONTENT, getSchema());
64          parameters.put(XSDSchemaValidationPipelet.PARAM_DISABLE_VALIDATION, Boolean.TRUE);
65  
66          xsdSchemaValidationPipelet.setParameters(parameters);
67          ExecutionContext executionContext = getExecutionContext(testMessageContext);
68  
69          addInvalidPayload(executionContext);
70  
71          assertDoesNotThrow(() -> xsdSchemaValidationPipelet.execute(executionContext));
72      }

74      private static ExecutionContext getExecutionContext(TestMessageContext testMessageContext) {
75          ExecutionContext executionContext = testMessageContext.createExecutionContext(MessageType.NORMAL);
76          executionContext.setMessage(new Message());
77          executionContext.getMessage().setId("originalId");
78          executionContext.getMessage().setConversationId("originalId");
79          return executionContext;
80      }
81  
82      private static void addValidPayload(ExecutionContext executionContext) {
83          ContextPayload payload = new ContextPayload();
84          payload.setContent(PayloadContent.of(
85                  """
86                                                       <?xml version="1.0" encoding="UTF-8" ?>
87                                                       <individual>
88                                                           <name>Sample</name>
89                                                           <address>
90                                                               <zip>00001</zip>
91                                                               <city>New York</city>
92                                                               <street>Main Str.</street>
93                                                           </address>
94                                                       </individual>
95                                                       """
96                          .getBytes(StandardCharsets.UTF_8)));
97          payload.setContentType(MediaType.TEXT_PLAIN_VALUE);
98          payload.setSize(payload.getContent().getSize());
99          payload.setEncoding(StandardCharsets.UTF_8.name());
100         executionContext.getMessage().setPayloads(List.of(payload));
101     }
102
