28  class PayloadMetadataToMessagePipeletTest {
29      @MethodSource("copyAllArguments")
30      @ParameterizedTest
31      void copyAllTest(String category, Map<String, List<String>> categoriesCopied) throws NexusException {
32          PayloadMetadataToMessagePipelet pipelet = new PayloadMetadataToMessagePipelet();
33          ComponentTestUtils.applyParameters(
34                  pipelet, Map.of("copy_all", true, "category", category, "metadata_fields", List.of("foo", "baz")));
35  
36          MetaData metaData = new MetaData();
37          metaData.add("manual_message", true);
38          metaData.add("first_category", "foo", "bar");
39          metaData.add("first_category", "baz", "qux");
40          metaData.add("second_category", "bar", "foo");
41  
42          ExecutionContext messageContext = createContext(metaData);
43  
44          pipelet.execute(messageContext);
45  
46          MetaData messageMetaData = messageContext.getMessage().getMetaData();
47  
48          assertThat(messageMetaData.categories()).containsExactlyInAnyOrderElementsOf(categoriesCopied.keySet());
49          for (Map.Entry<String, List<String>> expectedCategory : categoriesCopied.entrySet()) {
50              assertEquals(
51                      expectedCategory.getValue().size(),
52                      messageMetaData.keys(expectedCategory.getKey()).size());
53  
54              assertThat(messageMetaData.keys(expectedCategory.getKey()))
55                      .containsExactlyInAnyOrderElementsOf(expectedCategory.getValue());
56          }
57      }
58  
59      @MethodSource("copyArguments")
60      @ParameterizedTest
61      void copyTest(String category, List<String> metadataFields, List<String> expectedKeys) throws NexusException {
62          PayloadMetadataToMessagePipelet pipelet = new PayloadMetadataToMessagePipelet();
63          ComponentTestUtils.applyParameters(
64                  pipelet, Map.of("copy_all", false, "category", category, "metadata_fields", metadataFields));
65  
66          MetaData metaData = new MetaData();
67          metaData.add("manual_message", true);
68          metaData.add("first_category", "foo", "bar");
69          metaData.add("first_category", "baz", "qux");
70          metaData.add("second_category", "bar", "foo");
71  
72          ExecutionContext messageContext = createContext(metaData);
73  
74          pipelet.execute(messageContext);
75  
76          MetaData messageMetaData = messageContext.getMessage().getMetaData();
77  
78          if (CollectionUtils.isNotEmpty(expectedKeys)) {
79              assertThat(messageMetaData.keys(category)).containsExactlyInAnyOrderElementsOf(expectedKeys);
80          }
81  
82          for (String expectedKey : expectedKeys) {
83              assertThat(messageMetaData.get(expectedKey)).isEqualTo(metaData.get(expectedKey));
84          }
85      }

119     private ExecutionContext createContext(MetaData metaData) {
120         ContextPayload contextPayload = new ContextPayload();
121         contextPayload.setMetaData(metaData);
122 
123         Message message = new Message();
124         message.setPayloads(List.of(contextPayload));
125 
126         TestMessageContext messageContext = new TestMessageContext();
127         ExecutionContext executionContext = messageContext.createExecutionContext(MessageType.NORMAL);
128         executionContext.setMessage(message);
129         return executionContext;
130     }
