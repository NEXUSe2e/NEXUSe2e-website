243         Parameters parameters = new Parameters();
244         parameters.put(
245                 JsonToMetaDataPipelet.PARAM_META_DATA_KEYS,
246                 List.of(new DefaultKeyValue<>("id", "id"), new DefaultKeyValue<>("eventId", "eventId")));
247 
248         JsonToMetaDataPipelet jsonToMetaDataPipelet = mock(JsonToMetaDataPipelet.class);
249         when(jsonToMetaDataPipelet.getParameters()).thenReturn(parameters);
250 
251         JsonToMetaDataTraversal traversal = JsonToMetaDataTraversal.forInstance(jsonToMetaDataPipelet);
252         try (InputStream inputStream = getClass().getResourceAsStream("/json_to_metadata/nested-object-array.json")) {
253             ContextPayload contextPayload = new ContextPayload();
254             contextPayload.setContent(PayloadContent.of(inputStream));
255 
256             MetaData metaData = traversal.resolve(contextPayload);
257 
258             assertThat(metaData)
259                     .isEqualTo(new MetaData()
260                             .set("id", List.of("S1234567890"))
261                             .set(
262                                     "eventId",
263                                     List.of(
264                                             "eb3d5b4e-cf14-4824-9af8-2afc79675a72",
265                                             "450a3026-0fa3-413f-b04c-1ad1319c9917")));
