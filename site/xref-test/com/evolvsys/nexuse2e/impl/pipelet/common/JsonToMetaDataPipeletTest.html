37      @ParameterizedTest
38      @MethodSource
39      void execute(
40              String description,
41              String jsonPath,
42              List<KeyValue<String, String>> fieldToKeys,
43              Map<String, List<String>> expected)
44              throws IOException, NexusException {
45          ComponentTestUtils.applyParameters(pipelet, Map.of(JsonToMetaDataPipelet.PARAM_META_DATA_KEYS, fieldToKeys));
46  
47          TestMessageContext testMessageContext = new TestMessageContext();
48          Message message = new Message();
49          message.getMetaData()
50                  .set(
51                          "originalPayload",
52                          """
53                               {
54                                   "foo": "bar"
55                               }
56                               """);
57          try (InputStream inputStream = getClass().getResourceAsStream(jsonPath)) {
58              message.setPayloads(List.of(createPayload(inputStream)));
59  
60              ExecutionContext executionContext = testMessageContext.createExecutionContext(MessageType.NORMAL);
61              executionContext.setMessage(message);
62  
63              pipelet.execute(executionContext);
64          }
65  
66          // assert that the originalPayload metadata is still set
67          assertThat(message.getMetaData().get("originalPayload"))
68                  .isEqualTo(
69                          List.of(
70                                  """
71                                             {
72                                                 "foo": "bar"
73                                             }
74                                             """));
75          Collection<String> expectedKeys =
76                  fieldToKeys.stream().map(KeyValue::getValue).toList();
77          for (String expectedKey : expectedKeys) {
78              assertThat(message.getMetaData())
79                      .extracting(metaData -> metaData.get(expectedKey))
80                      .isEqualTo(expected.get(expectedKey));
81          }
82      }

249     @ValueSource(booleans = {true, false})
250     @ParameterizedTest
251     void execute_failOnError(boolean failOnError) throws IOException, NexusException {
252         ComponentTestUtils.applyParameters(
253                 pipelet,
254                 Map.of(
255                         JsonToMetaDataPipelet.PARAM_META_DATA_KEYS,
256                         List.of(new DefaultKeyValue<>("", "boolean")),
257                         JsonToMetaDataPipelet.PARAM_FAIL_ON_ERROR,
258                         failOnError));
259 
260         TestMessageContext testMessageContext = new TestMessageContext();
261         Message message = new Message();
262         try (InputStream invalidStream =
263                         getClass().getResourceAsStream("/json_to_metadata/invalid/invalid_syntax.json");
264                 InputStream validStream = getClass().getResourceAsStream("/json_to_metadata/root-boolean-array.json")) {
265             message.setPayloads(List.of(createPayload(invalidStream), createPayload(validStream)));
266 
267             ExecutionContext executionContext = testMessageContext.createExecutionContext(MessageType.NORMAL);
268             executionContext.setMessage(message);
269 
270             if (failOnError) {
271                 assertThatThrownBy(() -> pipelet.execute(executionContext))
272                         .isInstanceOf(NexusException.class)
273                         .hasMessage(
274                                 "NexusError(code=null, message=Could not read json field values from payload null, messageKey=null, extensions={})");
275             } else {
276                 pipelet.execute(executionContext);
277                 assertThat(message.getMetaData())
278                         .isEqualTo(new MetaData().set("boolean", List.of(true, false, true, true)));
279             }
280         }
281     }
