23  class PersistentTimestampPipeletTest {
24      @Test
25      void execute() throws NexusException {
26          PersistentTimestampPipelet pipelet = new PersistentTimestampPipelet();
27  
28          String category = "test_category";
29          String key = "test_key";
30  
31          Map<String, Object> parameters = new HashMap<>(Map.of("category", category, "key", key));
32          ComponentTestUtils.applyParameters(pipelet, parameters);
33  
34          TestMessageContext messageContext = new TestMessageContext();
35  
36          ZonedDateTime mockedZonedDateTime = ZonedDateTime.now();
37  
38          try (MockedStatic<ZonedDateTime> mockedStaticZonedDateTime =
39                  mockStatic(ZonedDateTime.class, InvocationOnMock::callRealMethod)) {
40              mockedStaticZonedDateTime.when(ZonedDateTime::now).thenReturn(mockedZonedDateTime);
41  
42              pipelet.execute(messageContext.createExecutionContext(MessageType.NORMAL));
43  
44              assertThat(messageContext.getPersistentKeyValue().getAsDatetime(category, key))
45                      .hasValue(mockedZonedDateTime);
46          }
47      }
48  }
