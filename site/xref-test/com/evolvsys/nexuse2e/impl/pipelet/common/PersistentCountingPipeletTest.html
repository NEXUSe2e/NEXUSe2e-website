20  class PersistentCountingPipeletTest {
21  
22      @ParameterizedTest
23      @CsvSource({",,1,2,3", "5,1,6,7,8", "2,,3,4,5", ",2,2,4,6", ",-5,-5,-10,-15", "3,4,7,11,15"})
24      void execute(Long initial, Long increment, Long expectedFirst, Long expectedSecond, Long expectedThird)
25              throws NexusException {
26          PersistentCountingPipelet pipelet = new PersistentCountingPipelet();
27  
28          Map<String, Object> parameters = new HashMap<>(Map.of("category", "my category", "key", "my key"));
29          if (initial != null) {
30              parameters.put("initial_value", initial);
31          }
32          if (increment != null) {
33              parameters.put("increment", increment);
34          }
35          ComponentTestUtils.applyParameters(pipelet, parameters);
36  
37          TestMessageContext testMessageContext = new TestMessageContext();
38  
39          pipelet.execute(testMessageContext.createExecutionContext(MessageType.NORMAL));
40  
41          assertThat(testMessageContext.getPersistentKeyValue().getAsLong("my category", "my key"))
42                  .describedAs("First value")
43                  .hasValue(expectedFirst);
44  
45          pipelet.execute(testMessageContext.createExecutionContext(MessageType.NORMAL));
46  
47          assertThat(testMessageContext.getPersistentKeyValue().getAsLong("my category", "my key"))
48                  .describedAs("Second value")
49                  .hasValue(expectedSecond);
50  
51          pipelet.execute(testMessageContext.createExecutionContext(MessageType.NORMAL));
52  
53          assertThat(testMessageContext.getPersistentKeyValue().getAsLong("my category", "my key"))
54                  .describedAs("Third value")
55                  .hasValue(expectedThird);
56      }
57  }
