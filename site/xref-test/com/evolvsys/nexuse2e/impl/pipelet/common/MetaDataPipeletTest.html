20  /**
21   * Test für {@link MetaDataPipelet}
22   *
23   * @author SvenBarden
24   * @since 05.07.2024
25   */
26  class MetaDataPipeletTest {
27  
28      @ParameterizedTest
29      @NullAndEmptySource
30      @ValueSource(strings = {"category", "cat 2"})
31      void execute(String category) throws NexusException {
32          MetaDataPipelet metaDataPipelet = new MetaDataPipelet();
33  
34          ComponentTestUtils.applyParameters(
35                  metaDataPipelet,
36                  EntryStream.<String, Object>empty()
37                          .append("category", category)
38                          .append(
39                                  "meta_datas",
40                                  List.of(
41                                          new DefaultKeyValue<>("key1", "value"),
42                                          new DefaultKeyValue<>("key2", "foo"),
43                                          new DefaultKeyValue<>("key3", "bar"),
44                                          new DefaultKeyValue<>("key3", "ex")))
45                          .filterValues(Objects::nonNull)
46                          .toMap());
47  
48          TestMessageContext testMessageContext = new TestMessageContext();
49          Message message = new Message();
50  
51          message.getMetaData().set("key2", "hello");
52  
53          testMessageContext.setSourceMessage(message);
54          ExecutionContext executionContext = testMessageContext.createExecutionContext(MessageType.NORMAL);
55  
56          ExecutionContext result = metaDataPipelet.execute(executionContext);
57  
58          assertThat(result).isSameAs(executionContext);
59  
60          if (StringUtils.isBlank(category)) {
61              assertThat(message.getMetaData().get("key1")).isEqualTo(List.of("value"));
62              assertThat(message.getMetaData().get("key2")).isEqualTo(List.of("hello", "foo"));
63              assertThat(message.getMetaData().get("key3")).isEqualTo(List.of("bar", "ex"));
64          } else {
65              assertThat(message.getMetaData().get(category, "key1")).isEqualTo(List.of("value"));
66              assertThat(message.getMetaData().get(category, "key2")).isEqualTo(List.of("foo"));
67              assertThat(message.getMetaData().get(category, "key3")).isEqualTo(List.of("bar", "ex"));
68  
69              assertThat(message.getMetaData().get("key2")).isEqualTo(List.of("hello"));
70          }
71      }
72  }
