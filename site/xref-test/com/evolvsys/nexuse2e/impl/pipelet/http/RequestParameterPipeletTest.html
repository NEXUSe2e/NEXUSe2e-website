19      @Test
20      void executeTest() throws NexusException {
21          Parameters parameters = new Parameters();
22          parameters.put(PARAM_CHOREOGRAPHY_ID_ENABLED, true);
23          parameters.put(PARAM_CHOREOGRAPHY_ID, "ChoreographyId");
24          parameters.put(PARAM_ACTION_ID_ENABLED, true);
25          parameters.put(PARAM_ACTION_ID, "ActionId");
26          parameters.put(PARAM_SOURCE_PARTNER_ID_ENABLED, true);
27          parameters.put(PARAM_SOURCE_PARTNER_ID, "SourcePartnerId");
28          parameters.put(PARAM_TARGET_PARTNER_ID_ENABLED, true);
29          parameters.put(PARAM_TARGET_PARTNER_ID, "TargetPartnerId");
30          parameters.put(PARAM_CONVERSATION_ID_ENABLED, true);
31          parameters.put(PARAM_CONVERSATION_ID, "ConversationId");
32          parameters.put(PARAM_MESSAGE_ID_ENABLED, true);
33          parameters.put(PARAM_MESSAGE_ID, "MessageId");
34  
35          RequestParameterPipelet pipelet = new RequestParameterPipelet();
36          pipelet.setParameters(parameters);
37  
38          TestMessageContext testMessageContext = new TestMessageContext();
39          ExecutionContext executionContext = testMessageContext.createExecutionContext(MessageType.NORMAL);
40          executionContext.setMessage(new Message());
41  
42          executionContext.setChoreographyId("dummy_ChoreographyId");
43          executionContext.setActionId("dummy_ActionId");
44          executionContext.setSourcePartnerId("dummy_SourcePartnerId");
45          executionContext.setTargetPartnerId("dummy_TargetPartnerId");
46          executionContext.getMessage().setConversationId("dummy_ConversationId");
47          executionContext.getMessage().setId("dummy_MessageId");
48  
49          pipelet.execute(executionContext);
50  
51          assertEquals("dummy_ChoreographyId", getRequestParameterValue(executionContext, "ChoreographyId"));
52          assertEquals("dummy_ActionId", getRequestParameterValue(executionContext, "ActionId"));
53          assertEquals("dummy_SourcePartnerId", getRequestParameterValue(executionContext, "SourcePartnerId"));
54          assertEquals("dummy_TargetPartnerId", getRequestParameterValue(executionContext, "TargetPartnerId"));
55          assertEquals("dummy_ConversationId", getRequestParameterValue(executionContext, "ConversationId"));
56          assertEquals("dummy_MessageId", getRequestParameterValue(executionContext, "MessageId"));
57  
58          verifyNoInteractions(testMessageContext.messageIdGenerator, testMessageContext.conversationIdGenerator);
59      }

142     private String getRequestParameterValue(ExecutionContext executionContext, String key) {
143         return executionContext.getMetaData().getFirstOrNull(HttpConstants.PREFIX_OUTBOUND_REQUEST_PARAMETER, key);
