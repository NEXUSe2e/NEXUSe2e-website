56      @ParameterizedTest
57      @ValueSource(strings = {"Existing-Dummy-Content-Id", "Existing-Dummy-Filename", "NOT_SET"})
58      void testBusinessMessage(String contentIdBase) throws NexusException, IOException {
59          ExecutionContext executionContext = getTestExecutionContext(MessageType.NORMAL);
60  
61          ContextPayload payload = executionContext.getMessage().getPayloads().getFirst();
62          if ("Existing-Dummy-Content-Id".equals(contentIdBase)) {
63              payload.setContentId("Existing-Dummy-Content-Id");
64          } else if ("Existing-Dummy-Filename".equals(contentIdBase)) {
65              payload.getMetaData().set(FileConstants.FILE_NAME, "Existing-Dummy-Filename");
66          }
67  
68          EbxmlHeaderSerializerPipelet ebxmlHeaderSerializerPipelet = new EbxmlHeaderSerializerPipelet();
69          ebxmlHeaderSerializerPipelet.execute(executionContext);
70  
71          String actualEbxml = executionContext
72                  .getMessage()
73                  .getMetaData()
74                  .getFirstOrThrows(EbxmlConstants.HEADER_DATA)
75                  .getContent()
76                  .getString(StandardCharsets.UTF_8);
77  
78          ZonedDateTime expectedCreatedDate = executionContext.getMessage().getOriginalCreated();
79  
80          String expectedXml = IOUtils.resourceToString(
81                          "/ebxml/serializer_pipelet_test/expected_normal.txt", StandardCharsets.UTF_8)
82                  .replace("{timestamp}", ebxmlTimestampFormatter.getTimestamp(expectedCreatedDate))
83                  .replace("{cpaId}", "Invoice");
84  
85          switch (contentIdBase) {
86              case "Existing-Dummy-Content-Id" ->
87                  expectedXml = expectedXml.replace("{contentId}", "Existing-Dummy-Content-Id");
88              case "Existing-Dummy-Filename" ->
89                  expectedXml = expectedXml.replace("{contentId}", "Existing-Dummy-Filename__body_1");
90              case "NOT_SET" ->
91                  expectedXml = expectedXml.replace("{contentId}", "d9e7a17d-1d4b-40b6-b933-cdbc3d2562a9__body1");
92              default -> throw new RuntimeException("invalid value " + contentIdBase);
93          }
94  
95          assertEquals(expectedXml, actualEbxml);
96      }

98      @ParameterizedTest
99      @ValueSource(strings = {"CHOREOGRAPHY_ID", "URI", "NOT_SET"})
100     void testCPAIdSchemes(String cpaIdScheme) throws NexusException, IOException {
101         ExecutionContext executionContext = getTestExecutionContext(MessageType.NORMAL);
102 
103         EbxmlHeaderSerializerPipelet ebxmlHeaderSerializerPipelet = new EbxmlHeaderSerializerPipelet();
104 
105         if (!"NOT_SET".equals(cpaIdScheme)) {
106             Parameters parameters = new Parameters();
107             parameters.put(EbxmlConstants.CPA_ID_SCHEME.getName(), cpaIdScheme);
108             ebxmlHeaderSerializerPipelet.setParameters(parameters);
109         }
110 
111         ebxmlHeaderSerializerPipelet.execute(executionContext);
112 
113         String actualEbxml = executionContext
114                 .getMessage()
115                 .getMetaData()
116                 .getFirstOrThrows(EbxmlConstants.HEADER_DATA)
117                 .getContent()
118                 .getString(StandardCharsets.UTF_8);
119 
120         ZonedDateTime expectedCreatedDate = executionContext.getMessage().getOriginalCreated();
121 
122         String expectedXml = IOUtils.resourceToString(
123                         "/ebxml/serializer_pipelet_test/expected_normal.txt", StandardCharsets.UTF_8)
124                 .replace("{timestamp}", ebxmlTimestampFormatter.getTimestamp(expectedCreatedDate))
125                 .replace("{contentId}", "d9e7a17d-1d4b-40b6-b933-cdbc3d2562a9__body1");
126 
127         if ("CHOREOGRAPHY_ID".equals(cpaIdScheme) || "NOT_SET".equals(cpaIdScheme)) {
128             expectedXml = expectedXml.replace("{cpaId}", "Invoice");
129         } else if ("URI".equals(cpaIdScheme)) {
130             expectedXml = expectedXml.replace("{cpaId}", "uri://091589686/0053141170000/Invoice");
131         }
132 
133         assertEquals(expectedXml, actualEbxml);
134     }

159     private ExecutionContext getTestExecutionContext(MessageType messageType) {
160         ExecutionContext executionContext = ContextFactory.createExecutionContext(testMessageContext, messageType);
161 
162         executionContext.setSourcePartnerId("091589686");
163         executionContext.setTargetPartnerId("0053141170000");
164         executionContext.setChoreographyId("Invoice");
165 
166         if (MessageType.NORMAL.equals(messageType)) {
167             executionContext.setActionId("SendFile");
168             executionContext.setMessage(getNormalMessage());
169         } else if (MessageType.ACKNOWLEDGMENT.equals(messageType)) {
170             executionContext.setActionId("Acknowledgment");
171             executionContext.setSourceMessage(getSourceMessage());
172             executionContext.setAcknowledgmentMessage(getAcknowledgmentMessage());
173         }
174 
175         ContextConnection contextConnection =
176                 ContextConnection.builder().reliable(true).build();
177         executionContext.setRoute(
178                 ContextRoute.builder().targetConnection(contextConnection).build());
179 
180         return executionContext;
181     }
