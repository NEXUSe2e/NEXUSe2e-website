23   * Test for {@link SoapUnpackagerPipelet}.
24   *
25   * @author SvenBarden
26   * @since 25.07.2025
27   */
28  @SuppressWarnings("PMD.TestClassWithoutTestCases") // carthesian test false positive
29  class SoapUnpackagerPipeletTest {
30      private final SoapUnpackagerPipelet pipelet = new SoapUnpackagerPipelet();
31      private final TestMessageContext testMessageContext = new TestMessageContext();
32  
33      @BeforeEach
34      void setUp() {
35          testMessageContext.setSourceMessage(new Message());
36      }
37  
38      @CartesianTest
39      void execute(
40              @CartesianTest.Values(strings = {"utf-8", "iso-8859-1"}) String sourceEncoding,
41              @CartesianTest.Values(strings = {"utf-8", "iso-8859-1", ""}) String targetEncoding)
42              throws NexusException, IOException {
43  
44          PayloadContent payloadContent =
45                  PayloadContent.of(IOUtils.resourceToByteArray("/soap/soap_inbound_message_" + sourceEncoding + ".xml"));
46          testMessageContext.setSourceData(ContextPayload.builder()
47                  .contentType("text/xml")
48                  .encoding(sourceEncoding)
49                  .content(payloadContent)
50                  .size(payloadContent.getSize())
51                  .build());
52  
53          ComponentTestUtils.applyParameters(pipelet, Map.of("encoding", targetEncoding));
54          ExecutionContext execute = pipelet.execute(testMessageContext.createExecutionContext(MessageType.NORMAL));
55  
56          if (StringUtils.isBlank(targetEncoding)) {
57              targetEncoding = sourceEncoding;
58          }
59  
60          assertThat(execute).isNotNull();
61          assertThat(testMessageContext.getSourceMessage().getPayloads()).hasSize(1);
62          assertThat(testMessageContext
63                          .getSourceMessage()
64                          .getPayloads()
65                          .getFirst()
66                          .getEncoding())
67                  .isEqualTo(targetEncoding);
68          XmlAssert.assertThat(testMessageContext
69                          .getSourceMessage()
70                          .getPayloads()
71                          .getFirst()
72                          .getContent()
73                          .getString(Charset.forName(targetEncoding)))
74                  .and(IOUtils.resourceToString(
75                          "/soap/soap_payload_" + targetEncoding + ".xml", Charset.forName(targetEncoding)))
76                  .ignoreWhitespace()
77                  .areSimilar();
78      }
79  }
