23  class SecureInboundRulePipeletTest {
24  
25      @ParameterizedTest
26      @ValueSource(booleans = {true, false})
27      void execute(boolean secure) throws NexusException {
28          SecureInboundRulePipelet secureInboundRulePipelet = new SecureInboundRulePipelet();
29  
30          ComponentTestUtils.applyParameters(secureInboundRulePipelet, Map.of());
31          ExecutionContext executionContext = new TestMessageContext().createExecutionContext(MessageType.NORMAL);
32          executionContext.getMetaData().set(ConnectionConstants.SECURE, secure);
33          executionContext.setMessage(new Message());
34  
35          if (secure) {
36              ExecutionContext result = secureInboundRulePipelet.execute(executionContext);
37              assertThat(result).isSameAs(executionContext);
38          } else {
39              assertThatThrownBy(() -> secureInboundRulePipelet.execute(executionContext))
40                      .isInstanceOf(NexusException.class)
41                      .hasMessageContaining("Secure inbound required");
42          }
43      }
44  }
