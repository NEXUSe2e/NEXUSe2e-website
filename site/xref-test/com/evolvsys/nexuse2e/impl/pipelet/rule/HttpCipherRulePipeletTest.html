20  /**
21   * Test für {@link HttpCipherRulePipelet}
22   *
23   * @author SvenBarden
24   * @since 02.07.2024
25   */
26  class HttpCipherRulePipeletTest {
27  
28      @ParameterizedTest
29      @CsvSource({
30          "CIPHER1,'cipher1,cipher2'",
31          "CIPHER1,CIPHER1",
32          "CIPHERX,",
33          "CIPHERX,'tlsv1.2,cipherX'",
34      })
35      void execute_success(String cipher, @ConvertWith(StringToListConverter.class) List<String> validCiphers)
36              throws NexusException {
37          HttpCipherRulePipelet httpCipherRulePipelet = new HttpCipherRulePipelet();
38  
39          ComponentTestUtils.applyParameters(httpCipherRulePipelet, Map.of("ciphers", validCiphers));
40          ExecutionContext executionContext = new TestMessageContext().createExecutionContext(MessageType.NORMAL);
41          executionContext.getMetaData().set(HttpConstants.CIPHERS, cipher);
42          executionContext.setMessage(new Message());
43  
44          ExecutionContext result = httpCipherRulePipelet.execute(executionContext);
45  
46          assertThat(result).isSameAs(executionContext);
47      }
48  
49      @ParameterizedTest
50      @CsvSource({
51          ",,No http cipher found in message context",
52          ",'cipher1,cipher2',No http cipher found in message context",
53          "CIPHER1,CIPHER0,Invalid inbound http cipher",
54          ",,No http cipher found in message context",
55          "CIPHER2,'cipher1,cipher3',Invalid inbound http cipher",
56      })
57      void execute_fail(
58              String cipher, @ConvertWith(StringToListConverter.class) List<String> validCiphers, String message) {
59          HttpCipherRulePipelet httpCipherRulePipelet = new HttpCipherRulePipelet();
60  
61          ComponentTestUtils.applyParameters(httpCipherRulePipelet, Map.of("ciphers", validCiphers));
62          ExecutionContext executionContext = new TestMessageContext().createExecutionContext(MessageType.NORMAL);
63          executionContext.getMetaData().set(HttpConstants.CIPHERS, cipher);
64          executionContext.setMessage(new Message());
65  
66          assertThatThrownBy(() -> httpCipherRulePipelet.execute(executionContext))
67                  .isInstanceOf(NexusException.class)
68                  .hasMessageContaining(message);
69      }
70  }
