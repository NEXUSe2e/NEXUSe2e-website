20  class UnzipTransformatorServiceTest {
21  
22      private ContextPayload toPayload(byte[] payload) {
23          PayloadContent payloadContent = PayloadContent.of(payload);
24          return ContextPayload.builder()
25                  .content(payloadContent)
26                  .size(payloadContent.getSize())
27                  .contentType("application/zip")
28                  .build();
29      }
30  
31      private UnzipTransformatorService createService() {
32          UnzipTransformatorService unzipTransformatorService = new UnzipTransformatorService();
33          ComponentTestUtils.applyParameters(
34                  unzipTransformatorService, Map.of("password", "password", "encoding_hint", "cp1252"));
35  
36          return unzipTransformatorService;
37      }
38  
39      @Test
40      void transform_empty() throws NexusException {
41  
42          UnzipTransformatorService unzipTransformatorService = createService();
43  
44          List<ContextPayload> transformed = unzipTransformatorService.transform(List.of());
45  
46          assertThat(transformed).isEmpty();
47  
48          transformed = unzipTransformatorService.transform(null);
49  
50          assertThat(transformed).isEmpty();
51      }
52  
53      @Test
54      void transform_success() throws NexusException, IOException {
55  
56          byte[] payload = IOUtils.resourceToByteArray(
57                  "/csv_splitter/bonuspunkte_20241105_010026_very_small.zip"); // bonuspunkte_20241105_010026_very_small.zip 5735fe1c-a0ba-4c1b-92e1-793b4f0153ff_payload-0.zip
58  
59          ContextPayload contextPayload = toPayload(payload);
60  
61          UnzipTransformatorService unzipTransformatorService = createService();
62  
63          List<ContextPayload> transformed = unzipTransformatorService.transform(List.of(contextPayload));
64  
65          assertThat(transformed).hasSize(1);
66          ContextPayload transformedPayload = transformed.getFirst();
67  
68          assertThat(transformedPayload.getSize()).isEqualTo(1693L);
69  
70          assertThat(transformedPayload.getEncoding()).isEqualToIgnoringCase("cp1252");
71          String payloadString = new String(
72                  transformedPayload.getContent().getByteArray(), Charset.forName(transformedPayload.getEncoding()));
73          String expectedPayload1 = IOUtils.resourceToString(
74                  "/csv_splitter/bonuspunkte_20241105_010026_very_small.csv", Charsets.toCharset("cp1252"));
75          assertThat(payloadString).isEqualTo(expectedPayload1);
76      }
77  }
