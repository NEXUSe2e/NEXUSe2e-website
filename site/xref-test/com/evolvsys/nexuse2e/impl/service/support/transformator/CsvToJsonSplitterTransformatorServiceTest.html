30      private ContextPayload toPayload(byte[] payload, String encoding) {
31          PayloadContent payloadContent = PayloadContent.of(payload);
32          return ContextPayload.builder()
33                  .encoding(encoding)
34                  .content(payloadContent)
35                  .size(payloadContent.getSize())
36                  .contentType("text/csv")
37                  .build();
38      }

105     @ParameterizedTest
106     @CsvSource({"empty", "','", ";", "/"})
107     void transform_seperator(String seperator) throws NexusException, IOException {
108         byte[] payload;
109         if ("empty".equals(seperator)) {
110             payload = getDummyPayload1(false, ";");
111         } else {
112             payload = getDummyPayload1(false, seperator);
113         }
114 
115         ContextPayload contextPayload = toPayload(payload, "UTF-8");
116         Map<String, Object> parameters = getParameterMapForDummyData();
117         if (!"empty".equals(seperator)) {
118             parameters.put("separator_char", seperator);
119         }
120         parameters.put("has_header", false);
121         CsvToJsonSplitterTransformatorService csvToJsonSplitterTransformatorService = createService(parameters);
122         List<ContextPayload> transformed = csvToJsonSplitterTransformatorService.transform(List.of(contextPayload));
123 
124         assertThat(transformed).hasSize(4);
125 
126         assertThat(transformed)
127                 .extracting(ContextPayload::getContentType)
128                 .isEqualTo(List.of("application/json", "application/json", "application/json", "application/json"));
129         assertThat(transformed)
130                 .extracting(ContextPayload::getEncoding)
131                 .isEqualTo(List.of("UTF-8", "UTF-8", "UTF-8", "UTF-8"));
132         assertThat(transformed).extracting(ContextPayload::getSize).isEqualTo(List.of(94L, 94L, 95L, 94L));
133 
134         assertThat(transformed.get(0).getContent().getString(StandardCharsets.UTF_8))
135                 .contains("12340");
136         assertThat(transformed.get(1).getContent().getString(StandardCharsets.UTF_8))
137                 .contains("12341");
138         assertThat(transformed.get(2).getContent().getString(StandardCharsets.UTF_8))
139                 .contains("12339");
140         assertThat(transformed.get(3).getContent().getString(StandardCharsets.UTF_8))
141                 .contains("12342");
142     }
