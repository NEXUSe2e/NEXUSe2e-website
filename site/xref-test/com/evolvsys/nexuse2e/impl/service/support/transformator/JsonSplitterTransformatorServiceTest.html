32      private ContextPayload toPayload(String payload, String encoding) {
33          PayloadContent payloadContent = PayloadContent.of(payload.getBytes(Charset.forName(encoding)));
34          return ContextPayload.builder()
35                  .encoding(encoding)
36                  .content(payloadContent)
37                  .size(payloadContent.getSize())
38                  .contentType("application/json")
39                  .build();
40      }
41  
42      private JsonSplitterTransformatorService createService() {
43          JsonSplitterTransformatorService jsonSplitterTransformatorService = new JsonSplitterTransformatorService();
44          BeanFactory beanFactory = mock(BeanFactory.class);
45          when(beanFactory.getBean(ObjectMapper.class)).thenReturn(new ObjectMapper());
46          jsonSplitterTransformatorService.setBeanFactory(beanFactory);
47          return jsonSplitterTransformatorService;
48      }

63      @CartesianTest
64      void transform_success(
65              @CartesianTest.Values(strings = {"utf-8", "iso-8859-1"}) String encoding,
66              @CartesianTest.Values(booleans = {true, false}) boolean addRaw)
67              throws NexusException {
68          String payload = "[{\"foo\":\"bar\"},{\"foo\":\"xuz\"}]";
69          ContextPayload contextPayload = toPayload(payload, encoding);
70  
71          JsonSplitterTransformatorService jsonSplitterTransformatorService = createService();
72  
73          ComponentTestUtils.applyParameters(jsonSplitterTransformatorService, Map.of("add_raw", addRaw));
74  
75          List<ContextPayload> transformed = jsonSplitterTransformatorService.transform(List.of(contextPayload));
76  
77          if (addRaw) {
78              assertThat(transformed).hasSize(3);
79              assertThat(transformed)
80                      .extracting(ContextPayload::getContentType)
81                      .isEqualTo(List.of("application/json", "application/json", "application/json"));
82              assertThat(transformed)
83                      .extracting(ContextPayload::getEncoding)
84                      .isEqualTo(List.of(encoding, encoding, encoding));
85              assertThat(transformed).extracting(ContextPayload::getSize).isEqualTo(List.of(29L, 13L, 13L));
86              assertThat(transformed)
87                      .extracting(ContextPayload::getContent)
88                      .extracting(payloadContent -> payloadContent.getString(Charset.forName(encoding)))
89                      .isEqualTo(List.of(payload, "{\"foo\":\"bar\"}", "{\"foo\":\"xuz\"}"));
90          } else {
91              assertThat(transformed).hasSize(2);
92              assertThat(transformed)
93                      .extracting(ContextPayload::getContentType)
94                      .isEqualTo(List.of("application/json", "application/json"));
95              assertThat(transformed).extracting(ContextPayload::getEncoding).isEqualTo(List.of(encoding, encoding));
96              assertThat(transformed).extracting(ContextPayload::getSize).isEqualTo(List.of(13L, 13L));
97              assertThat(transformed)
98                      .extracting(ContextPayload::getContent)
99                      .extracting(payloadContent -> payloadContent.getString(Charset.forName(encoding)))
100                     .isEqualTo(List.of("{\"foo\":\"bar\"}", "{\"foo\":\"xuz\"}"));
101         }
102     }

104     @ParameterizedTest
105     @ValueSource(strings = {"/metadata", "", "   "})
106     void transform_nested_success(String metadataPointer) throws NexusException {
107         String payload =
108                 "{\"metadata\":{\"foo\":\"bar\", \"baz\":\"qux\"},\"elements\":[{\"foo\":\"bar\"},{\"foo\":\"xuz\"}]}";
109         ContextPayload contextPayload = toPayload(payload, "utf-8");
110 
111         JsonSplitterTransformatorService jsonSplitterTransformatorService = createService();
112         ComponentTestUtils.applyParameters(
113                 jsonSplitterTransformatorService,
114                 Map.of("array_pointer", "/elements", "metadata_pointer", metadataPointer));
115 
116         List<ContextPayload> transformed = jsonSplitterTransformatorService.transform(List.of(contextPayload));
117 
118         assertThat(transformed).hasSize(2);
119         assertThat(transformed)
120                 .extracting(ContextPayload::getContentType)
121                 .isEqualTo(List.of("application/json", "application/json"));
122         assertThat(transformed).extracting(ContextPayload::getEncoding).isEqualTo(List.of("utf-8", "utf-8"));
123         assertThat(transformed).extracting(ContextPayload::getSize).isEqualTo(List.of(13L, 13L));
124         assertThat(transformed)
125                 .extracting(ContextPayload::getContent)
126                 .extracting(payloadContent -> payloadContent.getString(StandardCharsets.UTF_8))
127                 .isEqualTo(List.of("{\"foo\":\"bar\"}", "{\"foo\":\"xuz\"}"));
128 
129         if (StringUtils.isNotBlank(metadataPointer)) {
130             assertThat(transformed)
131                     .extracting(p -> p.getMetaData().get(JsonSplitterTransformatorService.METADATA_CATEGORY, "foo"))
132                     .isEqualTo(List.of(List.of("\"bar\""), List.of("\"bar\"")));
133             assertThat(transformed)
134                     .extracting(p -> p.getMetaData().get(JsonSplitterTransformatorService.METADATA_CATEGORY, "baz"))
135                     .isEqualTo(List.of(List.of("\"qux\""), List.of("\"qux\"")));
136         }
137     }

195     @Test
196     void transform_nested_object_success() throws Exception {
197         String payload =
198                 "{\"array\":[{\"element\":{\"id\":4711,\"name\":\"element 1\"}},{\"element\":{\"id\":4712,\"name\":\"element 2\"}}]}";
199         ContextPayload contextPayload = toPayload(payload, StandardCharsets.UTF_8.name());
200 
201         JsonSplitterTransformatorService jsonSplitterTransformatorService = createService();
202         ComponentTestUtils.applyParameters(
203                 jsonSplitterTransformatorService, Map.of("array_pointer", "/array", "content_pointer", "/element"));
204 
205         List<ContextPayload> transformed = jsonSplitterTransformatorService.transform(List.of(contextPayload));
206 
207         assertThat(transformed).hasSize(2);
208 
209         ContextPayload firstContextPayload = transformed.getFirst();
210         assertThat(firstContextPayload)
211                 .extracting(ContextPayload::getContentType)
212                 .isEqualTo(MediaType.APPLICATION_JSON_VALUE);
213         assertThat(firstContextPayload)
214                 .extracting(ContextPayload::getContent)
215                 .extracting(payloadContent -> payloadContent.getString(StandardCharsets.UTF_8))
216                 .isEqualTo("{\"id\":4711,\"name\":\"element 1\"}");
217 
218         ContextPayload secondContextPayload = transformed.getLast();
219         assertThat(secondContextPayload)
220                 .extracting(ContextPayload::getContentType)
221                 .isEqualTo(MediaType.APPLICATION_JSON_VALUE);
222         assertThat(secondContextPayload)
223                 .extracting(ContextPayload::getContent)
224                 .extracting(payloadContent -> payloadContent.getString(StandardCharsets.UTF_8))
225                 .isEqualTo("{\"id\":4712,\"name\":\"element 2\"}");
226     }
