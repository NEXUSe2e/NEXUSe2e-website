19  class JsonFieldRenderTypeTest {
20  
21      JsonFactory jsonFactory = new JsonFactory();
22      ByteArrayOutputStream byteArrayOutputStream;

24  
25      @ParameterizedTest
26      @ValueSource(strings = {"10", "20.6", "0.23423522", "0"})
27      @NullSource
28      void test_number_success(String value) throws IOException {
29  
30          initObject();
31          JsonFieldRenderType.NUMBER.render(jsonGenerator, "number_value", value);
32          String result = finishObject();
33  
34          assertEquals("{\"number_value\":" + value + "}", result);
35      }
36  
37      @ParameterizedTest
38      @CsvSource(
39              value = {"10;10", "20,6;20.6", "0,23423522;0.23423522", "0;0", "null;null"},
40              delimiter = ';',
41              nullValues = "null")
42      void test_number_de_success(String value, String expected) throws IOException {
43  
44          initObject();
45          JsonFieldRenderType.NUMBER_DE.render(jsonGenerator, "number_value", value);
46          String result = finishObject();
47  
48          assertEquals("{\"number_value\":" + expected + "}", result);
49      }

83      @ParameterizedTest
84      @ValueSource(strings = {"true", "false", "abc", ""})
85      @NullSource
86      void test_boolean(String value) throws IOException {
87  
88          initObject();
89          JsonFieldRenderType.BOOLEAN.render(jsonGenerator, "number_value", value);
90          String result = finishObject();
91          String expected = "false";
92          if ("true".equals(value)) {
93              expected = "true";
94          }
95          assertEquals("{\"number_value\":" + expected + "}", result);
96      }
97  
98      private void initObject() throws IOException {
99  
100         byteArrayOutputStream = new ByteArrayOutputStream();
101         jsonGenerator = jsonFactory.createGenerator(byteArrayOutputStream, JsonEncoding.UTF8);
102         jsonGenerator.writeStartObject();
103     }
104 
105     private String finishObject() throws IOException {
106         jsonGenerator.writeEndObject();
107         jsonGenerator.close();
108         return byteArrayOutputStream.toString(StandardCharsets.UTF_8);
109     }
