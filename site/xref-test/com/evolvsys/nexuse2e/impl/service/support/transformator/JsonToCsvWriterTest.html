17  /**
18   * Tests for the {@link JsonToCsvWriter}.
19   *
20   * @author Julian Drees
21   * @since 09.09.2025
22   */
23  class JsonToCsvWriterTest {
24      @ParameterizedTest(name = "[{index}] - includeHeaders: {0}")
25      @CsvSource({
26          "true,44,/json_to_csv/target/target-with-headers.csv",
27          "false,44,/json_to_csv/target/target-without-headers.csv",
28          "true,59,/json_to_csv/target/target-with-headers-separator.csv",
29          "false,59,/json_to_csv/target/target-without-headers-separator.csv"
30      })
31      void write(boolean includeHeaders, int columnSeparator, String targetCsv) throws IOException {
32          List<KeyValue<String, String>> fieldColumnMappings = List.of(
33                  new DefaultKeyValue<>("index", "i"),
34                  new DefaultKeyValue<>("eventId", "eventIdInCsv"),
35                  new DefaultKeyValue<>("active", "active"));
36  
37          JsonToCsvWriter jsonToCsvWriter = new JsonToCsvWriter(
38                  fieldColumnMappings, includeHeaders, StandardCharsets.UTF_8, (char) columnSeparator);
39  
40          List<Map<String, Optional<Object>>> records = List.of(
41                  Map.of(
42                          "index", Optional.of(0),
43                          "eventId", Optional.empty(),
44                          "active", Optional.of(true)),
45                  Map.of(
46                          "index", Optional.of(1),
47                          "eventId", Optional.of("81efe022-4f7a-40fd-b651-30dacca56ed0"),
48                          "active", Optional.of(false)),
49                  Map.of(
50                          "index", Optional.empty(),
51                          "eventId", Optional.empty(),
52                          "active", Optional.of(false)),
53                  Map.of("index", Optional.of(3)),
54                  Map.of());
55  
56          PayloadContent csvContent = jsonToCsvWriter.write(records.iterator());
57  
58          String csv = csvContent.getString(StandardCharsets.UTF_8);
59  
60          assertThat(csv).isEqualTo(IOUtils.resourceToString(targetCsv, StandardCharsets.UTF_8));
61      }
62  }
