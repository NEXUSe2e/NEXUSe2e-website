20  class AzureKeyVaultKeyValueSupportServiceTest {
21  
22      @CartesianTest
23      void getKeyValueProvider_secretCredentials(
24              @CartesianTest.Values(strings = {"https://nexuse2e.vault.azure.net/", "https://foo.vault.azure.net/"})
25                      String url,
26              @CartesianTest.Values(strings = {"service 1", "service 2"}) String serviceId,
27              @CartesianTest.Values(strings = {"clientId", "clientId 2"}) String clientId,
28              @CartesianTest.Values(strings = {"secret", "bar"}) String clientSecret,
29              @CartesianTest.Values(
30                              strings = {"9f4f5767-019f-481f-b5cf-e46cc397be0a", "2f4f5767-019f-481f-b5cf-e46cc397be0a"})
31                      String tenantId) {
32          AzureKeyVaultKeyValueSupportService service = new AzureKeyVaultKeyValueSupportService();
33          service.setServiceId(serviceId);
34  
35          ComponentTestUtils.applyParameters(
36                  service,
37                  Map.of("url", url, "client_id", clientId, "client_secret", clientSecret, "tenant_id", tenantId));
38  
39          KeyValueProvider keyValueProvider = service.getKeyValueProvider();
40  
41          assertThat(keyValueProvider).isInstanceOf(AzureKeyVaultKeyValueSupportService.AzureKeyValueProvider.class);
42          assertThat(keyValueProvider.getName()).isEqualTo(serviceId);
43          AzureKeyVaultKeyValueSupportService.AzureKeyValueProvider azureKeyValueProvider =
44                  (AzureKeyVaultKeyValueSupportService.AzureKeyValueProvider) keyValueProvider;
45          assertThat(azureKeyValueProvider.getSecretClient().getVaultUrl()).isEqualTo(url);
46      }
