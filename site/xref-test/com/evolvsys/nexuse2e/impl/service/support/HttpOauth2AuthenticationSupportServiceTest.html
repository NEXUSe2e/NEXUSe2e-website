43      @RegisterExtension
44      static final WireMockExtension OAUTH2_SERVER = WireMockExtension.newInstance()
45              .options(wireMockConfig()
46                      .httpsPort(8181)
47                      .httpDisabled(true)
48                      .trustStorePath(resourcePath("/certs/certs-dev2/truststore.jks"))
49                      .trustStorePassword("nexuse2e")
50                      .keystorePath(resourcePath("/certs/certs-dev2/certificate.p12"))
51                      .keystorePassword("nexuse2e")
52                      .keyManagerPassword("nexuse2e"))
53              .failOnUnmatchedRequests(true)
54              .build();

56      @RegisterExtension
57      static final WireMockExtension TARGET_SERVER = WireMockExtension.newInstance()
58              .options(wireMockConfig()
59                      .httpsPort(8282)
60                      .httpDisabled(true)
61                      .trustStorePath(resourcePath("/certs/certs-remote1/truststore.jks"))
62                      .trustStorePassword("nexuse2e")
63                      .keystorePath(resourcePath("/certs/certs-remote1/certificate.p12"))
64                      .keystorePassword("nexuse2e")

66              .failOnUnmatchedRequests(true)
67              .build();

72      @SneakyThrows
73      private static String resourcePath(String resource) {
74          String path = IOUtils.resourceToURL(resource).getPath();
75  
76          if (path.startsWith("/") && path.contains(":")) {
77              path = path.substring(1);
78          }
79          return path;
80      }

82      @ParameterizedTest
83      @ValueSource(booleans = {true, false})
84      void basic_auth_success(boolean reactive) throws NexusException {
85          OAUTH2_SERVER.stubFor(
86                  post(urlEqualTo("/oauth2/basic-auth"))
87                          .withBasicAuth("basic_auth_oauth2~client_id", "basic_auth_oauth2~client_secret")
88                          .willReturn(
89                                  okJson(
90                                          """
91                                                                   {
92                                                                     "access_token":"MTQ0NjJkZmQ5OTM2NDE1ZTZjNGZmZjI3",
93                                                                     "token_type":"Bearer",
94                                                                     "expires_in":3600,
95                                                                     "refresh_token":"IwOGYzYTlmM2YxOTQ5MGE3YmNmMDFkNTVk",
96                                                                     "scope":"create"
97                                                                   }
98                                                                   """)));
99          TARGET_SERVER.stubFor(post(urlEqualTo("/p1")));
100 
101         TestOauth2SourceService service = serviceFactory.get("trigger service");
102         service.reactive = reactive;
103         service.execute();
104 
105         OAUTH2_SERVER.verify(postRequestedFor(urlEqualTo("/oauth2/basic-auth"))
106                 .withBasicAuth(new BasicCredentials("basic_auth_oauth2~client_id", "basic_auth_oauth2~client_secret"))
107                 .withRequestBody(equalTo("grant_type=client_credentials&scope=scope1+scope2")));
108         TARGET_SERVER.verify(postRequestedFor(urlEqualTo("/p1"))
109                 .withHeader("Authorization", equalTo("Bearer MTQ0NjJkZmQ5OTM2NDE1ZTZjNGZmZjI3")));
110     }
111 
112     @ParameterizedTest
113     @ValueSource(booleans = {true, false})
114     void basic_auth_fail(boolean reactive) throws NexusException {
115         OAUTH2_SERVER.stubFor(post(urlEqualTo("/oauth2/basic-auth"))
116                 .withBasicAuth("basic_auth_oauth2~client_id", "basic_auth_oauth2~client_secret")
117                 .willReturn(aResponse().withStatus(400).withBody("Wrong oauth request")));
118         TARGET_SERVER.stubFor(post(urlEqualTo("/p1")));
119 
120         TestOauth2SourceService service = serviceFactory.get("trigger service");
121         service.reactive = reactive;
122         service.execute();
123 
124         OAUTH2_SERVER.verify(postRequestedFor(urlEqualTo("/oauth2/basic-auth"))
125                 .withBasicAuth(new BasicCredentials("basic_auth_oauth2~client_id", "basic_auth_oauth2~client_secret"))
126                 .withRequestBody(equalTo("grant_type=client_credentials&scope=scope1+scope2")));
127         TARGET_SERVER.verify(0, anyRequestedFor(anyUrl()));
128     }

130     public static class TestOauth2SourceService extends SingleContextNexusService {
131         private boolean reactive;
132 
133         @Override
134         protected MessageContext buildMessageContext() throws NexusException {
135             MessageContext messageContext = contextFactory.createNew();
136             Message message = new Message();
137             message.setType(MessageType.NORMAL);
138             byte[] bytes = "body content".getBytes(StandardCharsets.UTF_8);
139             message.getPayloads()
140                     .add(ContextPayload.builder()
141                             .contentId("body")
142                             .encoding("utf-8")
143                             .contentType("text/plain")
144                             .size(bytes.length)
145                             .content(PayloadContent.of(bytes))
146                             .build());
147             messageContext.setSourceMessage(message);
148             messageContext.setChoreographyId("choreography");
149             messageContext.setActionId("action1");
150             messageContext.setTargetPartnerId(reactive ? "partner2" : "partner1");
151             messageContext.setSourcePartnerId("local");
152             return messageContext;
153         }
