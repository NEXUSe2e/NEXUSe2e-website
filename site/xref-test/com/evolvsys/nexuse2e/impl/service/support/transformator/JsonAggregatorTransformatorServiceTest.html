169     @ParameterizedTest(name = "{0}")
170     @MethodSource
171     void transform(
172             String label, Map<String, Object> params, List<String> expectedFiles, List<MetaData> expectedMetaDatas)
173             throws NexusException, IOException {
174         ZonedDateTime now = ZonedDateTime.of(2025, 4, 8, 12, 0, 0, 0, ZoneId.of("UTC"));
175 
176         try (MockedStatic<ZonedDateTime> nowMock = mockStatic(ZonedDateTime.class, InvocationOnMock::callRealMethod)) {
177             nowMock.when(ZonedDateTime::now).thenReturn(now);
178 
179             KeyValueTestInitializer.init();
180             ComponentTestUtils.applyParameters(jsonAggregatorTransformatorService, params);
181         }
182 
183         PayloadContent payloadContent1 =
184                 PayloadContent.of(getClass().getResourceAsStream("/json_aggregator/source1.json"));
185         PayloadContent payloadContent2 =
186                 PayloadContent.of(getClass().getResourceAsStream("/json_aggregator/source2.json"));
187 
188         List<ContextPayload> transform = jsonAggregatorTransformatorService.transform(
189                 List.of(toPayload(payloadContent1, "1"), toPayload(payloadContent2, "2")));
190 
191         List<String> jsonOutput = transform.stream()
192                 .map(contextPayload -> contextPayload.getContent().getString(StandardCharsets.UTF_8))
193                 .toList();
194 
195         assertThat(jsonOutput).hasSize(expectedFiles.size());
196         for (int i = 0; i < expectedFiles.size(); i++) {
197             JsonAssert jsonAssert = JsonAssertions.assertThatJson(jsonOutput.get(i));
198             jsonAssert.isEqualTo(IOUtils.resourceToString(
199                     "/json_aggregator/" + expectedFiles.get(i) + ".json", StandardCharsets.UTF_8));
200 
201             MetaData expectedMetaData = expectedMetaDatas.get(i);
202             assertThat(transform.get(i).getMetaData()).isEqualTo(expectedMetaData);
203         }
204     }

206     private static ContextPayload toPayload(PayloadContent payloadContent, String id) {
207         return ContextPayload.builder()
208                 .content(payloadContent)
209                 .size(payloadContent.getSize())
210                 .contentType("application/json")
211                 .encoding("UTF-8")
212                 .contentId(id)
213                 .metaData(new MetaData().set("myId", id).set("test-source", "generated"))
214                 .build();
215     }
