143     private void startSftpServerWithKeyAuth() throws IOException, NoSuchAlgorithmException, InvalidKeySpecException {
144 
145         LOGGER.info("Starting SFTP server with key authentication");
146         sshd = SshServer.setUpDefaultServer();
147         VirtualFileSystemFactory virtualFileSystemFactory = new VirtualFileSystemFactory(testFolder);
148         sshd.setFileSystemFactory(virtualFileSystemFactory);
149         sshd.setPort(PORT.intValue());
150         sshd.setSubsystemFactories(List.of(new SftpSubsystemFactory()));
151         sshd.setCommandFactory(new ScpCommandFactory());
152         sshd.setKeyPairProvider(
153                 new SimpleGeneratorHostKeyProvider(Files.createFile(testFolder.resolve("hostkey.ser"))));
154 
155         Security.addProvider(new BouncyCastleProvider());
156 
157         AsymmetricKeyParameter asymmetricKeyParameter = OpenSSHPublicKeyUtil.parsePublicKey(
158                 Base64.getDecoder().decode(KEY_PUBLIC_KEY.split(" ")[1]));
159         OpenSSHPublicKeySpec openSSHPublicKeySpec =
160                 new OpenSSHPublicKeySpec(OpenSSHPublicKeyUtil.encodePublicKey(asymmetricKeyParameter));
161         PublicKey publicKey = KeyFactory.getInstance(RSA).generatePublic(openSSHPublicKeySpec);
162 
163         List<PublicKey> keys = List.of(publicKey);
164         sshd.setPublickeyAuthenticator(new KeySetPublickeyAuthenticator("dummy", keys));
165 
166         sshd.start();
167         LOGGER.info("In-memory SFTP server started");
168     }

170     private ExecutionContext getExecutionContext() throws IOException {
171         ExecutionContext executionContext = new TestMessageContext().createExecutionContext(MessageType.NORMAL);
172         executionContext.setSourceMessage(new Message());
173         executionContext.getMessage().setId("some_id_1234");
174         executionContext.setSourcePartnerId("source_partner_id");
175         URL resource = getClass().getResource("/payload/png_payload.PNG");
176         String contentType = new Tika().detect(resource);
177         byte[] bytes = resource.openStream().readAllBytes();
178         PayloadContent payloadContent = PayloadContent.of(bytes);
179 
180         ContextPayload contextPayload = ContextPayload.builder()
181                 .contentType(contentType)
182                 .content(payloadContent)
183                 .size(payloadContent.getSize())
184                 .build();
185 
186         executionContext.getMessage().setPayloads(List.of(contextPayload));
187 
188         return executionContext;
