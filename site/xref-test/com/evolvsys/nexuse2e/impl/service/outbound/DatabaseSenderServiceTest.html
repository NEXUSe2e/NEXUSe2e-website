30  @SpringBootTest(
31          webEnvironment = SpringBootTest.WebEnvironment.RANDOM_PORT,
32          properties = {
33              "spring.datasource.url=jdbc:h2:mem:databaseSenderServiceTest;IGNORECASE=TRUE",

36  class DatabaseSenderServiceTest {
37      @Autowired
38      private ServiceFactory serviceFactory;
39  
40      @BeforeEach
41      void createTempTable() throws NexusException {
42          // create temporary table to save payload of current test
43          DatabaseConnectionSupportService supportService = serviceFactory.get("databaseConnectionSupportService");
44  
45          String sql =
46                  """
47                          create table NXS_store_temp_data (
48                              id bigint not null,
49                              data_ varchar(max) not null,
50                              primary key (id)
51                          )
52                          """;
53  
54          NamedParameterJdbcTemplate namedParameterJdbcTemplate = supportService.getNamedParameterJdbcTemplate();
55          namedParameterJdbcTemplate.getJdbcTemplate().execute(sql);
56      }
57  
58      @AfterEach
59      void deleteTempTable() throws NexusException {
60          // delete temporary table after test execution
61          DatabaseConnectionSupportService supportService = serviceFactory.get("databaseConnectionSupportService");
62  
63          NamedParameterJdbcTemplate namedParameterJdbcTemplate = supportService.getNamedParameterJdbcTemplate();
64          namedParameterJdbcTemplate.getJdbcTemplate().update("drop table NXS_store_temp_data");
65      }
66  
67      @Test
68      void executeInternalTest() throws NexusException {
69          ExecutionContext executionContext = getExecutionContext();
70  
71          DatabaseSenderContext databaseSenderContext = DatabaseSenderContext.builder()
72                  .tableName("NXS_store_temp_data")
73                  .keyValue("id", new IncrementPrimaryKeyIdDbValue("NXS_store_temp_data", "id"))
74                  .keyValue("data_", "test data")
75                  .build();
76  
77          executionContext
78                  .getMetaData()
79                  .set(DatabaseSenderService.CONTEXT_CATEGORY, DatabaseSenderService.CONTEXT_KEY, databaseSenderContext);
80  
81          NexusOutflowService senderService = serviceFactory.get("databaseSenderService");
82          senderService.executeInternal(executionContext);
83  
84          // verify result
85          DatabaseConnectionSupportService supportService =
86                  senderService.getParameters().value("database_connection_support_service");
87  
88          String sql = "select data_ from NXS_store_temp_data";
89  
90          NamedParameterJdbcTemplate namedParameterJdbcTemplate = supportService.getNamedParameterJdbcTemplate();
91  
92          List<String> tableData =
93                  namedParameterJdbcTemplate.queryForList(sql, new MapSqlParameterSource(), String.class);
94          assertEquals(1, tableData.size());
95          assertEquals("test data", tableData.getFirst());
96      }
97  
98      @Test
99      void executeInternalTest_invalidSupportService() {
100         ExecutionContext executionContext = getExecutionContext();
101 
102         NexusOutflowService senderService = serviceFactory.get("databaseSenderService_invalidSupportService");
103         NexusException nexusException =
104                 assertThrows(NexusException.class, () -> senderService.executeInternal(executionContext));
105 
106         NexusError nexusError = nexusException.getError();
107 
108         assertEquals("The configured support service can not be used for this service", nexusError.getMessage());
109         assertSame(
110                 DatabaseConnectionSupportService.class.getSimpleName(),
111                 nexusError.getExtensions().get("type"));
112     }
113 
114     @Test
