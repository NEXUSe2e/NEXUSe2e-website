28   * <p>Test für den {@link HttpPollingService}.</p>

30   * @author Florian Drees

33  @SpringBootTest(
34          webEnvironment = SpringBootTest.WebEnvironment.RANDOM_PORT,
35          properties = {
36              "spring.datasource.url=jdbc:h2:mem:pollingServiceTest;IGNORECASE=TRUE",
37              "nexuse2e.core.default-config=classpath:config/polling_service_test.xml"
38          })
39  @WireMockTest(httpPort = 8181)
40  @Slf4j
41  class HttpPollingServiceTest {
42      @Autowired
43      private ServiceFactory serviceFactory;
44  
45      @Test
46      void buildMessageContextTest() throws NexusException, IOException {
47          String expectedResult = IOUtils.resourceToString("/payload/ISO_8859_1.json", StandardCharsets.ISO_8859_1);
48          stubFor(get(urlEqualTo("/data?param1=value1&param2=value2"))
49                  .willReturn(aResponse()
50                          .withStatus(200)
51                          .withHeader(
52                                  "Content-Type",
53                                  new MediaType("application", "json", StandardCharsets.ISO_8859_1).toString())
54                          .withBody(IOUtils.resourceToByteArray("/payload/ISO_8859_1.json")))); // byte array required to
55          // prevent implicit utf-8
56          // conversion
57  
58          HttpPollingService httpPollingService = serviceFactory.get("httpPollingService");
59  
60          List<MessageContext> messageContexts = httpPollingService.buildMessageContexts();
61          assertThat(messageContexts).isNotNull().hasSize(3);
62  
63          assertThat(messageContexts)
64                  .extracting(MessageContext::getChoreographyId)
65                  .isEqualTo(List.of("Polling", "Polling", "Polling"));
66          assertThat(messageContexts)
67                  .extracting(MessageContext::getActionId)
68                  .isEqualTo(List.of("PollingAction", "PollingAction", "PollingAction"));
69          assertThat(messageContexts)
70                  .extracting(MessageContext::getTargetPartnerId)
71                  .isEqualTo(List.of("partner2", "partner2", "partner2"));
72          assertThat(messageContexts)
73                  .extracting(MessageContext::getSourceData)
74                  .extracting(List::size)
75                  .isEqualTo(List.of(1, 1, 1));
76          assertThat(messageContexts)
77                  .extracting(MessageContext::getMetaData)
78                  .extracting(metaData -> metaData.get(HttpConstants.PREFIX_REQUEST_HEADER, "Content-Type")
79                          .getFirst())
80                  .isEqualTo(List.of(
81                          "application/json;charset=ISO-8859-1",
82                          "application/json;charset=ISO-8859-1",
83                          "application/json;charset=ISO-8859-1"));
84          assertThat(messageContexts.getFirst())
85                  .extracting(MessageContext::getSourceMessage)
86                  .extracting(Message::getMetaData)
87                  .extracting(metaData -> metaData.getFirstOrNull(TransformatorSupportService.META_DATA_RAW))
88                  .isEqualTo(Boolean.TRUE);
89          assertThat(messageContexts.getLast())
90                  .extracting(MessageContext::getSourceMessage)
91                  .extracting(Message::getMetaData)
92                  .extracting(metaData -> metaData.getFirstOrNull(TransformatorSupportService.META_DATA_RAW))
93                  .isNull();
94          assertThat(messageContexts.stream()
95                          .map(MessageContext::getMetaData)
96                          .map(metaData -> metaData.getFirstOrNull(
97                                  TransactionConstants.CATEGORY, TransactionConstants.TRANSACTION_ID))
98                          .collect(Collectors.toSet()))
99                  .hasSize(1);
100         assertThat(messageContexts.stream()
101                         .map(MessageContext::getMetaData)
102                         .map(metaData -> metaData.getFirstOrNull(
103                                 TransactionConstants.CATEGORY, TransactionConstants.TRANSACTION_TIMESTAMP))
104                         .collect(Collectors.toSet()))
105                 .hasSize(1);
106 
107         assertThat(messageContexts)
108                 .extracting(MessageContext::getSourceData)
109                 .flatExtracting(contextPayloads -> contextPayloads)
110                 .extracting(ContextPayload::getContentType)
111                 .isEqualTo(List.of("application/json", "application/json", "application/json"));
112         assertThat(messageContexts)
113                 .extracting(MessageContext::getSourceData)
114                 .flatExtracting(contextPayloads -> contextPayloads)
115                 .extracting(ContextPayload::getEncoding)
116                 .isEqualTo(List.of(
117                         StandardCharsets.ISO_8859_1.toString(),
118                         StandardCharsets.ISO_8859_1.toString(),
119                         StandardCharsets.ISO_8859_1.toString()));
120         assertThat(messageContexts)
121                 .extracting(MessageContext::getSourceData)
122                 .flatExtracting(contextPayloads -> contextPayloads)
123                 .extracting(payload -> payload.getMetaData()
124                         .getFirstOrDefault(TransformatorSupportService.META_DATA_RAW, Boolean.FALSE))
125                 .isEqualTo(List.of(Boolean.TRUE, Boolean.FALSE, Boolean.FALSE));
126         assertThat(messageContexts)
127                 .extracting(MessageContext::getSourceData)
128                 .flatExtracting(Function.identity())
129                 .extracting(ContextPayload::getContent)
130                 .extracting(payloadContent -> payloadContent.getString(StandardCharsets.ISO_8859_1))
131                 .isEqualTo(List.of(expectedResult, "{\"foo\":\"Über\"}", "{\"foo\":\"qux\"}"));
132     }
