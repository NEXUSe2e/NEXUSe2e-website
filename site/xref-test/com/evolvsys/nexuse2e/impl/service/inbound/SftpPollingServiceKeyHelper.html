214     public static PublicKey getRsaPublicKey(String algorithm, String sshPublicKey)
215             throws IOException, InvalidKeySpecException, NoSuchAlgorithmException {
216         AsymmetricKeyParameter asymmetricKeyParameter = OpenSSHPublicKeyUtil.parsePublicKey(
217                 Base64.getDecoder().decode(sshPublicKey.split(" ")[1]));
218         OpenSSHPublicKeySpec openSSHPublicKeySpec =
219                 new OpenSSHPublicKeySpec(OpenSSHPublicKeyUtil.encodePublicKey(asymmetricKeyParameter));
220         return KeyFactory.getInstance(algorithm).generatePublic(openSSHPublicKeySpec);
221     }
