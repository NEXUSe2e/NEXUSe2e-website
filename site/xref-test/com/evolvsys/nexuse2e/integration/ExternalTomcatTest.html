39      private static void sslProviderBuilder(SslProvider.SslContextSpec sslContextSpec, SslContext sslContext) {
40          SslProvider.Builder builder = sslContextSpec.sslContext(sslContext);
41          builder.handlerConfigurator(handler -> {
42              SSLEngine engine = handler.engine();
43              SSLParameters sslParameters = engine.getSSLParameters();
44              SSLParameters params = new SSLParameters(sslParameters.getCipherSuites(), sslParameters.getProtocols());
45              // With java update null value is no longer sufficient to skip host name verification
46              params.setEndpointIdentificationAlgorithm("");
47              engine.setSSLParameters(params);
48          });
49      }

51      @ParameterizedTest
52      @CsvSource({
53          "TLSv1,partner1,false",
54          "TLSv1.1,partner1,false",
55          "TLSv1.2,partner1,false",
56          "TLSv1.3,partner1,true",
57          "TLSv1,partner2,false",

59          "TLSv1.2,partner2,true",
60          "TLSv1.3,partner2,false",

66          "TLSv1.1,partner4,true",
67          "TLSv1.2,partner4,true",
68          "TLSv1.3,partner4,true"
69      })
70      void test(String protocol, String partner, boolean valid)
71              throws KeyStoreException, CertificateException, IOException, NoSuchAlgorithmException {
72          Security.setProperty(
73                  "jdk.tls.disabledAlgorithms",
74                  "SSLv3, DTLSv1.0, RC4, DES, MD5withRSA, DH keySize < 256, EC keySize < 224, 3DES_EDE_CBC, anon, NULL, ECDH");
75  
76          KeyStore trustStore = KeyStore.getInstance("JKS");
77          trustStore.load(getClass().getResourceAsStream("/certs/truststore.jks"), "nexuse2e".toCharArray());
78  
79          TrustManagerFactory trustManagerFactory =
80                  TrustManagerFactory.getInstance(TrustManagerFactory.getDefaultAlgorithm());
81          trustManagerFactory.init(trustStore);
82  
83          SslContext sslContext = SslContextBuilder.forClient()
84                  .protocols(protocol)
85                  .trustManager(trustManagerFactory)
86                  .build();
87  
88          HttpClient httpClient =
89                  HttpClient.create().secure(sslContextSpec -> sslProviderBuilder(sslContextSpec, sslContext));
90  
91          WebClient.Builder webClientBuilder = WebClient.builder()
92                  .baseUrl("https://localhost:" + 8443 + "/service/")
93                  .clientConnector(new ReactorClientHttpConnector(httpClient));
94  
95          try {
96              ResponseEntity<String> response = webClientBuilder
97                      .build()
98                      .post()
99                      .uri("ebxml")
100                     .contentType(MediaType.MULTIPART_RELATED)
101                     .bodyValue(IOUtils.resourceToString("/ssl_cipher/" + partner + ".txt", StandardCharsets.UTF_8))
102                     .retrieve()
103                     .toEntity(String.class)
104                     .block();
105 
106             LOGGER.info("got response {}", response);
107             if (valid) {
108                 LOGGER.info("successful");
109             } else {
110                 fail("expected exception");
111             }
112         } catch (Exception e) {
113             if (valid) {
114                 fail("no exception expected");
115             } else {
116                 LOGGER.info("got exception", e);
117             }
118         }
119     }
