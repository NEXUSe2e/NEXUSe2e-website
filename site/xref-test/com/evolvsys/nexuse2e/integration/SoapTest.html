35  @SpringBootTest(
36          webEnvironment = SpringBootTest.WebEnvironment.RANDOM_PORT,
37          properties = {
38              "spring.datasource.url=jdbc:h2:mem:soapTest",
39              "nexuse2e.core.default-config=classpath:soap/nexus_config.xml",
40          })
41  @WireMockTest(httpPort = 8181)
42  class SoapTest {
43      private RestClient.Builder restClientBuilder;
44  
45      @LocalServerPort
46      private int port;
47  
48      @BeforeEach
49      void setUp() {
50          restClientBuilder = RestClient.builder().baseUrl("http://localhost:" + port + "/service/");
51      }
52  
53      @ParameterizedTest
54      @ValueSource(strings = {"utf-8", "iso-8859-1"})
55      void test(String charset)
56              throws ParserConfigurationException, IOException, SAXException, WSSecurityException, TransformerException {
57          Document document =
58                  SoapTestUtils.getDocument("/soap/soap_inbound_message_" + charset + ".xml", Charset.forName(charset));
59  
60          WSSecHeader secHeader = new WSSecHeader(document);
61          secHeader.insertSecurityHeader();
62          WSSecUsernameToken builder = new WSSecUsernameToken(secHeader);
63          builder.addNonce();
64          builder.addCreated();
65          builder.setUserInfo("soap-user", "soap-secret");
66          document = builder.build();
67  
68          String payload = SoapTestUtils.prettyDocumentToString(document, Charset.forName(charset));
69  
70          ResponseEntity<String> responseEntity = restClientBuilder
71                  .build()
72                  .post()
73                  .uri("soap")
74                  .contentType(new MediaType(MediaType.TEXT_XML, Charset.forName(charset)))
75                  .body(payload)
76                  .retrieve()
77                  .toEntity(String.class);
78  
79          assertThat(responseEntity.getStatusCode()).isEqualTo(HttpStatus.OK);
80          XmlAssert.assertThat(responseEntity.getBody())
81                  .and(IOUtils.resourceToString("/soap/soap_response_utf-8.xml", StandardCharsets.UTF_8))
82                  .ignoreWhitespace()
83                  .areSimilar();
84      }
85  }
