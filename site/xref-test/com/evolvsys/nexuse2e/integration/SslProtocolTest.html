107     void protocolVersion(String protocol, String partner, boolean valid)
108             throws IOException, KeyStoreException, CertificateException, NoSuchAlgorithmException, NexusException {
109         KeyStore trustStore = KeyStore.getInstance("JKS");
110         trustStore.load(getClass().getResourceAsStream("/certs/truststore.jks"), "nexuse2e".toCharArray());
111 
112         TrustManagerFactory trustManagerFactory =
113                 TrustManagerFactory.getInstance(TrustManagerFactory.getDefaultAlgorithm());
114         trustManagerFactory.init(trustStore);
115 
116         WebClientContext webClientContext = nexusRestClientFactory
117                 .createContextBuilder()
118                 .ssl(true)
119                 .trustStore(trustStore)
120                 .protocols(List.of(protocol))
121                 .build();
122         RestClient restClient = nexusRestClientFactory
123                 .createClientBuilder(webClientContext)
124                 .baseUrl("https://localhost:" + port + "/service/")
125                 .build();
126 
127         try {
128             ResponseEntity<String> response = restClient
129                     .post()
130                     .uri("ebxml")
131                     .contentType(MediaType.MULTIPART_RELATED)
132                     .body(IOUtils.resourceToString("/ssl_cipher/" + partner + ".txt", StandardCharsets.UTF_8))
133                     .exchange(
134                             (clientRequest, clientResponse) -> ResponseEntity.status(clientResponse.getStatusCode())
135                                     .body(IOUtils.toString(clientResponse.getBody(), StandardCharsets.UTF_8)),
136                             true);
137 
138             assertThat(response).isNotNull();
139             if (valid) {
140                 assertThat(response.getStatusCode()).isSameAs(HttpStatus.OK);
141                 assertThat(response.getBody()).isEqualTo("");
142                 LOGGER.info("successful");
143             } else {
144                 assertThat(response.getStatusCode()).isSameAs(HttpStatus.BAD_REQUEST);
145                 assertThat(response.getBody())
146                         .contains("Invalid inbound http protocol")
147                         .contains("protocol: " + protocol)
148                         .contains("validProtocols: " + VALID_PARTNER_PROTOCOLS.get(partner))
149                         .contains("choreographyId: choreoTest1")
150                         .contains("actionId: action1")
151                         .contains("sourcePartnerId: " + partner)
152                         .contains("targetPartnerId: local")
153                         .contains("messageId: start_msg");
154             }
155         } catch (Exception e) {
156             fail("no exception expected but got " + e.getMessage(), e);
157         }
158     }
