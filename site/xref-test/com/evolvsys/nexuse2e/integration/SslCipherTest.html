91      private static void sslProviderBuilder(SslProvider.SslContextSpec sslContextSpec, SslContext sslContext) {
92          SslProvider.Builder builder = sslContextSpec.sslContext(sslContext);
93          builder.handlerConfigurator(handler -> {
94              SSLEngine engine = handler.engine();
95              SSLParameters sslParameters = engine.getSSLParameters();
96              SSLParameters params = new SSLParameters(sslParameters.getCipherSuites(), sslParameters.getProtocols());
97              // With java update null value is no longer sufficient to skip host name verification
98              params.setEndpointIdentificationAlgorithm("");
99              engine.setSSLParameters(params);
100         });
101     }
102 
103     private static List<Arguments> cipher() {
104         List<Arguments> arguments = new ArrayList<>();
105         for (int i = 1; i <= 4; i++) {
106             String partner = "partner" + i;
107             for (String cipher : ALL_CIPHERS) {
108                 List<String> validCiphers = VALID_PARTNER_CIPHERS.get(partner);
109                 arguments.add(Arguments.of(partner, cipher, validCiphers.isEmpty() || validCiphers.contains(cipher)));
110             }
111         }
112 
113         return arguments;
114     }
115 
116     @ParameterizedTest
117     @MethodSource
118     void cipher(String partner, String cipher, boolean valid)
119             throws IOException, KeyStoreException, CertificateException, NoSuchAlgorithmException {
120         KeyStore trustStore = KeyStore.getInstance("JKS");
121         trustStore.load(getClass().getResourceAsStream("/certs/truststore.jks"), "nexuse2e".toCharArray());
122 
123         TrustManagerFactory trustManagerFactory =
124                 TrustManagerFactory.getInstance(TrustManagerFactory.getDefaultAlgorithm());
125         trustManagerFactory.init(trustStore);
126 
127         SslContext sslContext = SslContextBuilder.forClient()
128                 .ciphers(List.of(cipher))
129                 .trustManager(trustManagerFactory)
130                 .build();
131 
132         HttpClient httpClient =
133                 HttpClient.create().secure(sslContextSpec -> sslProviderBuilder(sslContextSpec, sslContext));
134 
135         WebClient.Builder webClientBuilder = WebClient.builder()
136                 .baseUrl("https://localhost:" + port + "/service/")
137                 .clientConnector(new ReactorClientHttpConnector(httpClient));
138 
139         try {
140             ResponseEntity<String> response = webClientBuilder
141                     .build()
142                     .post()
143                     .uri("ebxml")
144                     .contentType(MediaType.MULTIPART_RELATED)
145                     .bodyValue(IOUtils.resourceToString("/ssl_cipher/" + partner + ".txt", StandardCharsets.UTF_8))
146                     .exchangeToMono(clientResponse -> clientResponse.toEntity(String.class))
147                     .block();
148 
149             assertThat(response).isNotNull();
150             if (valid) {
151                 assertThat(response.getStatusCode()).isSameAs(HttpStatus.OK);
152                 assertThat(response.getBody()).isNull();
153                 LOGGER.info("successful");
154             } else {
155                 assertThat(response.getStatusCode()).isSameAs(HttpStatus.BAD_REQUEST);
156                 assertThat(response.getBody())
157                         .contains("Invalid inbound http cipher")
158                         .contains("cipher: " + cipher)
159                         .contains("validCiphers: " + VALID_PARTNER_CIPHERS.get(partner))
160                         .contains("choreographyId: choreoTest1")
161                         .contains("actionId: action1")
162                         .contains("sourcePartnerId: " + partner)
163                         .contains("targetPartnerId: local")
164                         .contains("messageId: start_msg");
165             }
166         } catch (Exception e) {
167             LOGGER.error("no exception expected", e);
168             fail("no exception expected but got " + e.getMessage());
169         }
170     }
