11  import org.junit.jupiter.api.DynamicTest;
12  import org.junit.jupiter.api.TestFactory;
13  
14  /**
15   * Tests if all components have a description
16   *
17   * @author SvenBarden
18   * @since 03.07.2024
19   */
20  @Slf4j
21  class ComponentDescriptionTest {
22      private final ComponentRegistry componentRegistry = new ComponentRegistry(new NEXUSe2eConfigurationProperties());
23  
24      @TestFactory
25      List<DynamicTest> verifyDescription() throws Exception {
26          componentRegistry.afterPropertiesSet();
27          List<DynamicTest> tests = new ArrayList<>();
28          for (RegisteredNexusComponent registeredComponent : componentRegistry.getRegisteredComponents()) {
29              if (registeredComponent
30                      .getLoadedClass()
31                      .getProtectionDomain()
32                      .getCodeSource()
33                      .getLocation()
34                      .getPath()
35                      .contains("target/test-classes/")) {
36                  LOGGER.info("skip description check for test component {}", registeredComponent.getComponentId());
37              } else {
38                  tests.add(DynamicTest.dynamicTest(
39                          registeredComponent.getComponentType() + ": " + registeredComponent.getComponentId(),
40                          () -> assertThat(registeredComponent.getDescription())
41                                  .describedAs("Description should not be blank")
42                                  .isNotEmpty()));
43              }
44          }
45          return tests;
46      }
47  }
