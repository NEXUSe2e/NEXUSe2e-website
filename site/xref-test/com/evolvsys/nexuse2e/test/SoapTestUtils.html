31  @UtilityClass
32  public class SoapTestUtils {
33      public static final DocumentBuilderFactory FACTORY = DocumentBuilderFactory.newInstance();
34  
35      static {
36          FACTORY.setNamespaceAware(true);
37      }
38  
39      @SuppressFBWarnings("XXE_DOCUMENT")
40      public static Document getDocument(String resourcePath)
41              throws ParserConfigurationException, IOException, SAXException {
42          try (InputStream resourceAsStream = SoapTestUtils.class.getResourceAsStream(resourcePath)) {
43              return FACTORY.newDocumentBuilder().parse(resourceAsStream);
44          }
45      }
46  
47      @SuppressFBWarnings("XXE_DOCUMENT")
48      public static Document getDocument(String resourcePath, Charset charset)
49              throws ParserConfigurationException, IOException, SAXException {
50          try (InputStream resourceAsStream = SoapTestUtils.class.getResourceAsStream(resourcePath)) {
51              InputSource inputSource = new InputSource(resourceAsStream);
52              inputSource.setEncoding(charset.name());
53              return FACTORY.newDocumentBuilder().parse(inputSource);
54          }
55      }
56  
57      public static String prettyDocumentToString(Document doc, Charset charset)
58              throws IOException, TransformerException {
59          try (ByteArrayOutputStream baos = new ByteArrayOutputStream()) {
60              elementToStream(doc.getDocumentElement(), baos, charset);
61              return baos.toString(charset);
62          }
63      }
64  
65      public static String prettyDocumentToString(Document doc) throws IOException, TransformerException {
66          return prettyDocumentToString(doc, StandardCharsets.UTF_8);
67      }
68  
69      public static void elementToStream(Element element, OutputStream out, Charset charset) throws TransformerException {
70          DOMSource source = new DOMSource(element);
71          StreamResult result = new StreamResult(out);
72  
73          TransformerFactory transFactory = TransformerFactory.newInstance();
74          transFactory.setFeature(XMLConstants.FEATURE_SECURE_PROCESSING, true);
75          try {
76              transFactory.setAttribute(XMLConstants.ACCESS_EXTERNAL_DTD, "");
77              transFactory.setAttribute(XMLConstants.ACCESS_EXTERNAL_STYLESHEET, "");
78          } catch (IllegalArgumentException ex) {
79              // ignore
80          }
81  
82          Transformer transformer = transFactory.newTransformer();
83          transformer.setOutputProperty(OutputKeys.ENCODING, charset.name());
84          transformer.transform(source, result);
85      }
