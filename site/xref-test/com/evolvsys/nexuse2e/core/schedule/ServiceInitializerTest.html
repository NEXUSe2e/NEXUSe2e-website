18   * <p>Test für den {@link ServiceInitializer}.</p>

23  class ServiceInitializerTest {
24      private final ServiceRepository serviceRepository = mock(ServiceRepository.class);
25      private final ServiceFactory serviceFactory = mock(ServiceFactory.class);
26      private final ServiceInitializer serviceInitializer = new ServiceInitializer(serviceRepository, serviceFactory);
27  
28      @Test
29      void afterPropertiesSetTest() {
30          List<ServiceEntity> serviceEntities = List.of(
31                  new ServiceEntity().setServiceId("serviceEntity1"), new ServiceEntity().setServiceId("serviceEntity2"));
32  
33          InflowService service1 = mock(InflowService.class);
34          InflowService service2 = mock(InflowService.class);
35  
36          when(serviceRepository.findByEnabledTrue()).thenReturn(serviceEntities);
37          when(serviceFactory.get("serviceEntity1")).thenReturn(service1);
38          when(serviceFactory.get("serviceEntity2")).thenReturn(service2);
39          doThrow(new NexusRuntimeException(NexusError.builder().build()))
40                  .when(service1)
41                  .start();
42  
43          serviceInitializer.afterPropertiesSet();
44  
45          ArgumentCaptor<ServiceEntity> serviceCaptor = ArgumentCaptor.forClass(ServiceEntity.class);
46  
47          verify(serviceRepository).findByEnabledTrue();
48          verify(serviceFactory).get("serviceEntity1");
49          verify(service1).start();
50          verify(serviceFactory).get("serviceEntity2");
51          verify(service2).start();
52          verify(serviceRepository).save(serviceCaptor.capture());
53          assertThat(serviceCaptor.getValue().isEnabled()).isFalse();
54          assertThat(serviceCaptor.getValue().getServiceId()).isEqualTo("serviceEntity1");
55          verifyNoMoreInteractions(serviceRepository, serviceFactory, service1, service2);
56      }
57  
58      @Test
59      void afterPropertiesSet_noResultsTest() {
60          when(serviceRepository.findByEnabledTrue()).thenReturn(Collections.emptyList());
61  
62          serviceInitializer.afterPropertiesSet();
63  
64          verify(serviceRepository).findByEnabledTrue();
65          verifyNoMoreInteractions(serviceRepository);
66          verifyNoInteractions(serviceFactory);
67      }
68  }
