19  /**
20   * Test für {@link SchedulingServiceCleanupJob}
21   *
22   * @author SvenBarden
23   * @since 27.03.2024
24   */
25  class SchedulingServiceCleanupJobTest {
26      private final ServiceRepository serviceRepository = mock(ServiceRepository.class);
27      private final ComponentRegistry componentRegistry = mock(ComponentRegistry.class);
28      private final SchedulingServiceCleanupJob schedulingServiceCleanupJob =
29              new SchedulingServiceCleanupJob(serviceRepository, componentRegistry);
30  
31      @SuppressWarnings("unchecked")
32      @Test
33      void executeInternal() throws SchedulerException {
34          JobExecutionContext jobExecutionContext = mock(JobExecutionContext.class);
35          Scheduler scheduler = mock(Scheduler.class);
36          ServiceEntity foundService = new ServiceEntity();
37          foundService.setEnabled(true);
38          foundService.setComponentId("other");
39  
40          ServiceEntity foundService2 = new ServiceEntity();
41          foundService2.setEnabled(true);
42          foundService2.setComponentId("componentInvalid");
43  
44          ServiceEntity foundService3 = new ServiceEntity();
45          foundService3.setEnabled(true);
46          foundService3.setComponentId("componentValid");
47  
48          when(jobExecutionContext.getScheduler()).thenReturn(scheduler);
49          when(scheduler.getJobKeys(any()))
50                  .thenReturn(Set.of(
51                          new JobKey("invalidformat", SchedulingService.QUARTZ_GROUP_ID),
52                          new JobKey("component_AS_notFound", SchedulingService.QUARTZ_GROUP_ID),
53                          new JobKey("componentNotMatch_AS_found", SchedulingService.QUARTZ_GROUP_ID),
54                          new JobKey("componentInvalid_AS_found2", SchedulingService.QUARTZ_GROUP_ID),
55                          new JobKey("componentValid_AS_found3", SchedulingService.QUARTZ_GROUP_ID)));
56  
57          when(serviceRepository.findByServiceId("found")).thenReturn(Optional.of(foundService));
58          when(serviceRepository.findByServiceId("found2")).thenReturn(Optional.of(foundService2));
59          when(serviceRepository.findByServiceId("found3")).thenReturn(Optional.of(foundService3));
60          when(componentRegistry.exists("componentValid")).thenReturn(Boolean.TRUE);
61  
62          schedulingServiceCleanupJob.executeInternal(jobExecutionContext);
63  
64          ArgumentCaptor<GroupMatcher<JobKey>> groupMatcherArgumentCaptor = ArgumentCaptor.forClass(GroupMatcher.class);
65  
66          verify(jobExecutionContext).getScheduler();
67          verify(scheduler).getJobKeys(groupMatcherArgumentCaptor.capture());
68          verify(scheduler).deleteJob(new JobKey("invalidformat", SchedulingService.QUARTZ_GROUP_ID));
69          verify(scheduler).deleteJob(new JobKey("component_AS_notFound", SchedulingService.QUARTZ_GROUP_ID));
70          verify(scheduler).deleteJob(new JobKey("componentNotMatch_AS_found", SchedulingService.QUARTZ_GROUP_ID));
71          verify(scheduler).deleteJob(new JobKey("componentInvalid_AS_found2", SchedulingService.QUARTZ_GROUP_ID));
72  
73          verify(serviceRepository).findByServiceId("notFound");
74          verify(serviceRepository).findByServiceId("found");
75          verify(serviceRepository).findByServiceId("found2");
76          verify(serviceRepository).findByServiceId("found3");
77          verify(serviceRepository).save(foundService);
78          verify(serviceRepository).save(foundService2);
79          verify(componentRegistry).exists("componentInvalid");
80          verify(componentRegistry).exists("componentValid");
81          verifyNoMoreInteractions(jobExecutionContext, scheduler, serviceRepository, componentRegistry);
82  
83          assertThat(groupMatcherArgumentCaptor.getValue().getCompareToValue())
84                  .isEqualTo(SchedulingService.QUARTZ_GROUP_ID);
85          assertThat(foundService.isEnabled()).isFalse();
86          assertThat(foundService2.isEnabled()).isFalse();
87      }
88  }
