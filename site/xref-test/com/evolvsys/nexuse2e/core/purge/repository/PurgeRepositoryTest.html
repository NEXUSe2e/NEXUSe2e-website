18   *

22  @SuppressWarnings("DataFlowIssue")
23  abstract class PurgeRepositoryTest {
24      @Autowired
25      protected PurgeRepository purgeRepository;
26  
27      @Autowired
28      protected JPQLQueryFactory queryFactory;
29  
30      protected long countConversations() {
31          return queryFactory
32                  .select(QConversationEntity.conversationEntity.count())
33                  .from(QConversationEntity.conversationEntity)
34                  .fetchOne();
35      }
36  
37      protected long countMessages() {
38          return queryFactory
39                  .select(QMessageEntity.messageEntity.count())
40                  .from(QMessageEntity.messageEntity)
41                  .fetchOne();
42      }
43  
44      protected long countBinaries() {
45          return queryFactory
46                  .select(QBinaryEntity.binaryEntity.count())
47                  .from(QBinaryEntity.binaryEntity)
48                  .fetchOne();
49      }
50  
51      protected long countPayloads() {
52          return queryFactory
53                  .select(QPayloadEntity.payloadEntity.count())
54                  .from(QPayloadEntity.payloadEntity)
55                  .fetchOne();
56      }
57  
58      protected long countMetaDatas() {
59          return queryFactory
60                  .select(QMetaDataEntity.metaDataEntity.count())
61                  .from(QMetaDataEntity.metaDataEntity)
62                  .fetchOne();
63      }
64  
65      protected long countLogs() {
66          return queryFactory
67                  .select(QLogEntity.logEntity.count())
68                  .from(QLogEntity.logEntity)
69                  .fetchOne();
70      }
71  
72      protected long countManualActions() {
73          return queryFactory
74                  .select(QManualActionEntity.manualActionEntity.count())
75                  .from(QManualActionEntity.manualActionEntity)
76                  .fetchOne();
77      }
78  
79      protected long countMessageReferences() {
80          return queryFactory
81                  .select(QMessageEntity.messageEntity.count())
82                  .from(QMessageActionManualActionEntity.messageActionManualActionEntity)
83                  .join(
84                          QMessageActionManualActionEntity.messageActionManualActionEntity.messages,
85                          QMessageEntity.messageEntity)
86                  .fetchCount();
87      }
88  
89      protected void verifySetup() {
90          // verify base setup
91          assertThat(countConversations())
92                  .describedAs("Conversation count start setup")
93                  .isEqualTo(3);
94          assertThat(countMessages()).describedAs("Message count start setup").isEqualTo(10);
95          assertThat(countPayloads()).describedAs("Payloads count start setup").isEqualTo(9);
96          assertThat(countBinaries()).describedAs("Binaries count start setup").isEqualTo(9);
97          assertThat(countMetaDatas()).describedAs("MetaData count start setup").isEqualTo(14);
98          assertThat(countLogs()).describedAs("Log count start setup").isEqualTo(9);
99          assertThat(countManualActions())
100                 .describedAs("ManualAction count start setup")
101                 .isEqualTo(8);
102         assertThat(countMessageReferences())
103                 .describedAs("ManualAction message reference count start setup")
104                 .isEqualTo(12);
105     }
106 }
