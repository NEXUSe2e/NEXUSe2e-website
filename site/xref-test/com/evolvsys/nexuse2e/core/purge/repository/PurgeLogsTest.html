21      @Test
22      void purgeLogs() {
23          verifySetup();
24  
25          PurgeResult purgeResult = purgeRepository.purgeLogs(
26                  LogFilter.builder().level("debug").level("trace").build());
27  
28          assertThat(purgeResult)
29                  .isEqualTo(PurgeResult.builder()
30                          .deletedMessages(0)
31                          .deletedConversations(0)
32                          .deletedPayloads(0)
33                          .deletedBinaries(0)
34                          .deletedMetaDatas(0)
35                          .deletedLogs(5)
36                          .deletedManualActions(0)
37                          .deletedManualActionsPartly(0)
38                          .build());
39  
40          assertThat(countConversations()).describedAs("Conversation count").isEqualTo(3);
41          assertThat(countMessages()).describedAs("Message count").isEqualTo(10);
42          assertThat(countPayloads()).describedAs("Payloads count").isEqualTo(9);
43          assertThat(countBinaries()).describedAs("Binaries count").isEqualTo(9);
44          assertThat(countMetaDatas()).describedAs("MetaData count").isEqualTo(14);
45          assertThat(countLogs()).describedAs("Log count").isEqualTo(4);
46          assertThat(countManualActions()).describedAs("ManualAction count").isEqualTo(8);
47          assertThat(countMessageReferences())
48                  .describedAs("Message references count")
49                  .isEqualTo(12);
50      }
