20      @Test
21      void purgeLogs() {
22          verifySetup();
23  
24          PurgeResult purgeResult = purgeRepository.purgeAllLogs();
25  
26          assertThat(purgeResult)
27                  .isEqualTo(PurgeResult.builder()
28                          .deletedMessages(0)
29                          .deletedConversations(0)
30                          .deletedPayloads(0)
31                          .deletedBinaries(0)
32                          .deletedMetaDatas(0)
33                          .deletedLogs(9)
34                          .deletedManualActions(0)
35                          .deletedManualActionsPartly(0)
36                          .build());
37  
38          assertThat(countConversations()).describedAs("Conversation count").isEqualTo(3);
39          assertThat(countMessages()).describedAs("Message count").isEqualTo(10);
40          assertThat(countPayloads()).describedAs("Payloads count").isEqualTo(9);
41          assertThat(countBinaries()).describedAs("Binaries count").isEqualTo(9);
42          assertThat(countMetaDatas()).describedAs("MetaData count").isEqualTo(14);
43          assertThat(countLogs()).describedAs("Log count").isEqualTo(0);
44          assertThat(countManualActions()).describedAs("ManualAction count").isEqualTo(8);
45          assertThat(countMessageReferences())
46                  .describedAs("Message references count")
47                  .isEqualTo(12);
48      }
