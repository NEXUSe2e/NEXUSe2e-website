55      @Test
56      void update_valid() {
57          PartnerConnectionDto partnerConnectionDto = new PartnerConnectionDto();
58          partnerConnectionDto.setId(44L);
59          PartnerConnectionDto expected = new PartnerConnectionDto()
60                  .setConnectionId("my connection")
61                  .setPartnerId(55L)
62                  .setAsync(true)
63                  .setReliable(true)
64                  .setMaxRetries(3);
65          expected.getWriteMetaData()
66                  .setInputProperties(Set.of("async", "reliable", "maxRetries", "connectionId", "partnerId"));
67  
68          when(partnerConnectionService.save((PartnerConnectionDto) any())).thenReturn(partnerConnectionDto);
69  
70          GraphQlTester.Response response = graphQlTester
71                  .documentName("partner-connection/partnerConnection_update")
72                  .execute();
73          response.errors().verify();
74  
75          response.path("partnerConnection_update.id").entity(Long.class).isEqualTo(44L);
76          response.path("partnerConnection_update.connectionId").pathDoesNotExist();
77  
78          ArgumentCaptor<PartnerConnectionDto> partnerConnectionDtoArgumentCaptor =
79                  ArgumentCaptor.forClass(PartnerConnectionDto.class);
80          verify(partnerConnectionService).save(partnerConnectionDtoArgumentCaptor.capture());
81          verifyNoMoreInteractions(partnerConnectionService);
82          verifyNoInteractions(dataLoaderService);
83  
84          assertThat(partnerConnectionDtoArgumentCaptor.getValue()).isEqualTo(expected);
85      }

138     @Test
139     void getById() {
140         PartnerConnectionDto partnerConnectionDto = new PartnerConnectionDto().setConnectionId("connection id");
141         PartnerDto partnerDto = new PartnerDto().setPartnerId("partner");
142         ServiceDto serviceDto = new ServiceDto().setServiceId("service");
143         CertificateContainerDto certificateContainerDto = new CertificateContainerDto().setName("container");
144         List<Long> requeueIntervals = List.of(50L, 100L);
145         List<ParameterDto> parameterDtos =
146                 List.of(new ParameterDto().setName("param 1"), new ParameterDto().setName("param 2"));
147 
148         when(partnerConnectionService.getById(44L)).thenReturn(Optional.of(partnerConnectionDto));
149         when(dataLoaderService.load(
150                         any(), eq(PartnerBatchLoader.class), same(partnerConnectionDto), eq("partnerId"), any()))
151                 .thenReturn(CompletableFuture.completedFuture(partnerDto));
152         when(dataLoaderService.load(
153                         any(), eq(ServiceBatchLoader.class), same(partnerConnectionDto), eq("outflowServiceId"), any()))
154                 .thenReturn(CompletableFuture.completedFuture(serviceDto));
155         when(dataLoaderService.load(
156                         any(),
157                         eq(CertificateContainerBatchLoader.class),
158                         same(partnerConnectionDto),
159                         eq("certificateContainerId"),
160                         any()))
161                 .thenReturn(CompletableFuture.completedFuture(certificateContainerDto));
162         when(dataLoaderService.load(
163                         any(), eq(PartnerConnectionParameterBatchLoader.class), same(partnerConnectionDto), any()))
164                 .thenReturn(CompletableFuture.completedFuture(new LinkedHashSet<>(parameterDtos)));
165         when(dataLoaderService.load(
166                         any(),
167                         eq(PartnerConnectionRequeueIntervalsBatchLoader.class),
168                         same(partnerConnectionDto),
169                         eq("requeueIntervals"),
170                         any()))
171                 .thenReturn(CompletableFuture.completedFuture(requeueIntervals));
172 
173         GraphQlTester.Response response = graphQlTester
174                 .documentName("partner-connection/partnerConnection_getById")
175                 .execute();
176         response.errors().verify();
177 
178         response.path("partnerConnection_getById.id").pathDoesNotExist();
179         response.path("partnerConnection_getById.connectionId")
180                 .entity(String.class)
181                 .isEqualTo("connection id");
182         response.path("partnerConnection_getById.partner.partnerId")
183                 .entity(String.class)
184                 .isEqualTo("partner");
185         response.path("partnerConnection_getById.outflowService.serviceId")
186                 .entity(String.class)
187                 .isEqualTo("service");
188         response.path("partnerConnection_getById.certificateContainer.name")
189                 .entity(String.class)
190                 .isEqualTo("container");
191         response.path("partnerConnection_getById.parameters")
192                 .entityList(ParameterDto.class)
193                 .isEqualTo(parameterDtos);
194         response.path("partnerConnection_getById.requeueIntervals")
195                 .entityList(Long.class)
196                 .isEqualTo(requeueIntervals);
197 
198         verify(partnerConnectionService).getById(44L);
199         verify(dataLoaderService)
200                 .load(any(), eq(PartnerBatchLoader.class), same(partnerConnectionDto), eq("partnerId"), any());
201         verify(dataLoaderService)
202                 .load(any(), eq(ServiceBatchLoader.class), same(partnerConnectionDto), eq("outflowServiceId"), any());
203         verify(dataLoaderService)
204                 .load(
205                         any(),
206                         eq(CertificateContainerBatchLoader.class),
207                         same(partnerConnectionDto),
208                         eq("certificateContainerId"),
209                         any());
210         verify(dataLoaderService)
211                 .load(any(), eq(PartnerConnectionParameterBatchLoader.class), same(partnerConnectionDto), any());
212         verify(dataLoaderService)
213                 .load(
214                         any(),
215                         eq(PartnerConnectionRequeueIntervalsBatchLoader.class),
216                         same(partnerConnectionDto),
217                         eq("requeueIntervals"),
218                         any());
219         verifyNoMoreInteractions(partnerConnectionService, dataLoaderService);
220     }
