23  class PipelineRemovePipeletActionTest {
24      private final PipelineRepository pipelineRepository = mock(PipelineRepository.class);
25      private final PipelineRemovePipeletAction pipelineRemovePipeletAction =
26              new PipelineRemovePipeletAction(pipelineRepository);
27  
28      @Test
29      void execute() {
30          PipelineEntity pipelineEntity = new PipelineEntity();
31          PipeletEntity pipeletEntity10 = new PipeletEntity();
32          pipeletEntity10.setId(10L);
33          PipeletEntity pipeletEntity20 = new PipeletEntity();
34          pipeletEntity20.setId(20L);
35          PipeletEntity pipeletEntity30 = new PipeletEntity();
36          pipeletEntity30.setId(30L);
37          pipelineEntity.getPipelets().add(pipeletEntity10);
38          pipelineEntity.getPipelets().add(pipeletEntity20);
39          pipelineEntity.getPipelets().add(pipeletEntity30);
40          PipelineEntity savedPipelineEntity = new PipelineEntity();
41          when(pipelineRepository.save(pipelineEntity)).thenReturn(savedPipelineEntity);
42  
43          PipelineEntity result = pipelineRemovePipeletAction.execute(
44                  pipelineEntity, null, new ProcessParameters(PipelineRemovePipeletAction.PIPELET_ID.of(20L)));
45  
46          verify(pipelineRepository).save(pipelineEntity);
47          verifyNoMoreInteractions(pipelineRepository);
48          assertThat(pipelineEntity.getPipelets()).containsExactly(pipeletEntity10, pipeletEntity30);
49          assertThat(result).isSameAs(savedPipelineEntity);
50          assertThat(pipeletEntity10.getOrder()).isEqualTo(0);
51          assertThat(pipeletEntity30.getOrder()).isEqualTo(1);
52      }
53  
54      private static List<Arguments> conditions() {
55          return List.of(
56                  Arguments.of(new ProcessParameters(), Boolean.FALSE),
57                  Arguments.of(new ProcessParameters(PipelineRemovePipeletAction.PIPELET_ID.of(1L)), Boolean.TRUE));
58      }
59  
60      @ParameterizedTest
61      @MethodSource
62      void conditions(ProcessParameters parameters, boolean valid) {
63          PipelineEntity pipelineEntity = new PipelineEntity();
64          ProcessActionConditions conditions = pipelineRemovePipeletAction.conditions(pipelineEntity, parameters);
65  
66          verifyNoInteractions(pipelineRepository);
67  
68          assertThat(conditions.getConditions().size()).isEqualTo(1);
69          if (valid) {
70              assertThat(conditions.check()).isEmpty();
71          } else {
72              assertThat(conditions.check()).isNotEmpty();
73          }
74      }
75  }
