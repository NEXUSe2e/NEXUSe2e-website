25  class PipelineRearrangePipeletsActionTest {
26      private final PipelineRepository pipelineRepository = mock(PipelineRepository.class);
27      private final PipelineRearrangePipeletsAction pipelineRearrangePipeletsAction =
28              new PipelineRearrangePipeletsAction(pipelineRepository);
29  
30      @Test
31      void execute() {
32          PipelineEntity pipelineEntity = new PipelineEntity();
33          PipeletEntity pipeletEntity10 = new PipeletEntity();
34          pipeletEntity10.setId(10L);
35          pipeletEntity10.setOrder(0);
36          PipeletEntity pipeletEntity20 = new PipeletEntity();
37          pipeletEntity20.setId(20L);
38          pipeletEntity20.setOrder(1);
39          pipelineEntity.getPipelets().add(pipeletEntity10);
40          pipelineEntity.getPipelets().add(pipeletEntity20);
41          PipelineEntity savedPipelineEntity = new PipelineEntity();
42          when(pipelineRepository.save(pipelineEntity)).thenReturn(savedPipelineEntity);
43  
44          PipelineEntity result = pipelineRearrangePipeletsAction.execute(
45                  pipelineEntity,
46                  null,
47                  new ProcessParameters(PipelineRearrangePipeletsAction.PIPELET_IDS.of(List.of(20L, 10L))));
48  
49          verify(pipelineRepository).save(pipelineEntity);
50          verifyNoMoreInteractions(pipelineRepository);
51          assertThat(pipelineEntity.getPipelets()).containsExactly(pipeletEntity10, pipeletEntity20);
52          assertThat(result).isSameAs(savedPipelineEntity);
53          assertThat(pipeletEntity10.getOrder()).isEqualTo(1);
54          assertThat(pipeletEntity20.getOrder()).isEqualTo(0);
55      }
56  
57      private static List<Arguments> conditions() {
58          return List.of(
59                  Arguments.of(new ProcessParameters(), Collections.emptyList(), Boolean.FALSE),
60                  Arguments.of(
61                          new ProcessParameters(PipelineRearrangePipeletsAction.PIPELET_IDS.of(List.of(10L))),
62                          List.of(),
63                          Boolean.FALSE),
64                  Arguments.of(
65                          new ProcessParameters(PipelineRearrangePipeletsAction.PIPELET_IDS.of(List.of(10L))),
66                          List.of(10L, 20L),
67                          Boolean.FALSE),
68                  Arguments.of(
69                          new ProcessParameters(PipelineRearrangePipeletsAction.PIPELET_IDS.of(List.of(20L, 10L))),
70                          List.of(10L, 20L),
71                          Boolean.TRUE));
72      }
73  
74      @ParameterizedTest
75      @MethodSource
76      void conditions(ProcessParameters parameters, SequencedCollection<Long> oldIds, boolean valid) {
77          PipelineEntity pipelineEntity = new PipelineEntity();
78          int i = 0;
79          for (Long oldId : oldIds) {
80              PipeletEntity pipeletEntity = new PipeletEntity();
81              pipeletEntity.setId(oldId);
82              pipeletEntity.setOrder(i++);
83              pipelineEntity.getPipelets().add(pipeletEntity);
84          }
85          ProcessActionConditions conditions = pipelineRearrangePipeletsAction.conditions(pipelineEntity, parameters);
86  
87          verifyNoInteractions(pipelineRepository);
88  
89          assertThat(conditions.getConditions().size()).isEqualTo(2);
90          if (valid) {
91              assertThat(conditions.check()).isEmpty();
92          } else {
93              assertThat(conditions.check()).isNotEmpty();
94          }
95      }
96  }
