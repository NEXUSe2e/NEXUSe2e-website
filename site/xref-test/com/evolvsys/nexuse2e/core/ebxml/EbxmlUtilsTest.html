13  
14  /**
15   * Test für {@link EbxmlUtils}
16   *
17   * @author SvenBarden
18   * @since 30.04.2024
19   */
20  class EbxmlUtilsTest {
21  
22      @ParameterizedTest
23      @ValueSource(
24              strings = {
25                  "simple",
26                  "with/slash",
27                  "with/multiple/slash",
28                  "with(brackets]or{something",
29                  "with\\backslash:and;colon,semi",
30                  "message@with<at>and?other",
31                  "message and whitespace\t=test"
32              })
33      void getMultipartRelatedContentType_context(String messageId) {
34          ExecutionContext executionContext = new TestMessageContext().createExecutionContext(MessageType.NORMAL);
35          Message message = new Message();
36          message.setId(messageId);
37          executionContext.setMessage(message);
38  
39          MediaType multipartRelatedContentType = EbxmlUtils.getMultipartRelatedContentType(executionContext);
40  
41          assertThat(multipartRelatedContentType.getType()).isEqualTo("multipart");
42          assertThat(multipartRelatedContentType.getSubtype()).isEqualTo("related");
43          assertThat(multipartRelatedContentType.getParameter("type")).isEqualTo("\"text/xml\"");
44          assertThat(multipartRelatedContentType.getParameter("boundary")).isEqualTo("\"MIME_boundary\"");
45          assertThat(multipartRelatedContentType.getParameter("start"))
46                  .isEqualTo("\"" + messageId + EbxmlConstants.PROTOCOL + "-Header\"");
47      }
48  
49      @CartesianTest
50      void getMultipartRelatedContentType_stringString(
51              @CartesianTest.Values(
52                              strings = {
53                                  "simple",
54                                  "with/slash",
55                                  "with/multiple/slash",
56                                  "with(brackets]or{something",
57                                  "with\\backslash:and;colon,semi",
58                                  "message@with<at>and?other",
59                                  "message and whitespace\t=test"
60                              })
61                      String messageId,
62              @CartesianTest.Values(strings = {"http", "https", "smtp", "ftp", "ftps", "sftp"}) String protocol) {
63          MediaType multipartRelatedContentType = EbxmlUtils.getMultipartRelatedContentType(messageId, protocol);
64  
65          assertThat(multipartRelatedContentType.getType()).isEqualTo("multipart");
66          assertThat(multipartRelatedContentType.getSubtype()).isEqualTo("related");
67          assertThat(multipartRelatedContentType.getParameter("type")).isEqualTo("\"text/xml\"");
68          assertThat(multipartRelatedContentType.getParameter("boundary")).isEqualTo("\"MIME_boundary\"");
69          assertThat(multipartRelatedContentType.getParameter("start"))
70                  .isEqualTo("\"" + messageId + protocol + "-Header\"");
71      }
72  }
