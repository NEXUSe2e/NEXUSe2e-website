3   import com.fasterxml.jackson.annotation.JsonProperty;
4   import com.github.tomakehurst.wiremock.common.Strings;
5   import com.github.tomakehurst.wiremock.matching.MatchResult;
6   import com.github.tomakehurst.wiremock.matching.StringValuePattern;
7   import java.util.Objects;
8   
9   /**
10   * @author SvenBarden
11   * @since 30.01.2024
12   */
13  public class EqualIgnoreEndingPattern extends StringValuePattern {
14  
15      public EqualIgnoreEndingPattern(@JsonProperty("equalIgnoreEnding") String expectedValue) {
16          super(expectedValue);
17      }
18  
19      public String getEqualIgnoreEnding() {
20          return expectedValue;
21      }
22  
23      @Override
24      protected boolean isNullValuePermitted() {
25          return true;
26      }
27  
28      @Override
29      public MatchResult match(String value) {
30          return new MatchResult() {
31              @Override
32              public boolean isExactMatch() {
33                  return value != null
34                          && expectedValue != null
35                          && Objects.equals(expectedValue.replace("\r", ""), value.replace("\r", ""));
36              }
37  
38              @Override
39              public double getDistance() {
40                  return Strings.normalisedLevenshteinDistance(expectedValue, value);
41              }
42          };
43      }
44  }
