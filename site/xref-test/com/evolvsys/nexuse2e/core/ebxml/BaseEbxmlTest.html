159     private static void sslProviderBuilder(SslProvider.SslContextSpec sslContextSpec, SslContext sslContext) {
160         SslProvider.Builder builder = sslContextSpec.sslContext(sslContext);
161         builder.handlerConfigurator(handler -> {
162             SSLEngine engine = handler.engine();
163             SSLParameters sslParameters = engine.getSSLParameters();
164             SSLParameters params = new SSLParameters(sslParameters.getCipherSuites(), sslParameters.getProtocols());
165             // With java update null value is no longer sufficient to skip host name verification
166             params.setEndpointIdentificationAlgorithm("");
167             engine.setSSLParameters(params);
168         });
169     }

171     void assertPayload(PayloadEntity payloadEntity) {
172         PayloadConfig payloadConfig = getPayloadConfig();
173         assertEquals(payloadConfig.getEncoding(), payloadEntity.getEncoding());
174         assertEquals(payloadConfig.getContentType(), payloadEntity.getContentType());
175         assertEquals(payloadConfig.getSize().toBytes(), payloadEntity.getSize());
176         assertEquals(1, payloadEntity.getPosition());
177         try {
178             byte[] expectedContent = IOUtils.resourceToByteArray(payloadConfig.getResourcePath());
179             byte[] content = IOUtils.toByteArray(
180                     payloadEntity.getContent().getContent().getBinaryStream(), payloadEntity.getSize());
181 
182             assertArrayEquals(expectedContent, content);
183         } catch (SQLException | IOException e) {
184             throw new RuntimeException(e);
185         }
186     }

311     private void verifyMessage_start() throws IOException {
312         MessageEntity inboundMessageEntity =
313                 messageRepository.findByMessageId(START_MESSAGE_ID).orElseThrow();
314 
315         String outboundCreated =
316                 EbxmlTimestampFormatter.getInstance().getTimestamp(inboundMessageEntity.getOriginalCreated());
317         String expectedOutbound = getStringContent("start_outbound.txt");
318 
319         expectedOutbound = expectedOutbound.replace("${timestamp}", outboundCreated);
320 
321         verify(postRequestedFor(urlEqualTo("/p2"))
322                 .withHeader(
323                         "content-type",
324                         equalTo("multipart/related;boundary=\"MIME_boundary\";start=\"start_msg/with_slash and"
325                                 + " whitespaceebxml-Header\";type=\"text/xml\";charset=UTF-8"))
326                 .withRequestBody(new EqualIgnoreEndingPattern(expectedOutbound))
327                 .withBasicAuth(new BasicCredentials("p1_user", "my secret")));
328 
329         verifyInboundMessageEntity_start(inboundMessageEntity);
330 
331         verifyConversationEntity_start(inboundMessageEntity.getConversation());
332     }

374     @Transactional
375     @ParameterizedTest
376     @ValueSource(booleans = {true, false})
377     void sendInboundMessage_startAction(boolean first) throws IOException {
378         stubFor(post(urlEqualTo("/p2"))
379                 .withBasicAuth("p1_user", "my secret")
380                 .willReturn(aResponse().withBody("some response"))); // outbound to 2. partner
381         stubFor(post(urlEqualTo("/p1"))
382                 .withBasicAuth("p1_user", "my secret")
383                 .willReturn(aResponse().withBody("ack"))); // outbound ack to 1. partner
384 
385         ResponseEntity<String> response = sendEbxml("start_inbound.txt");
386 
387         await().atMost(Duration.ofSeconds(5)).pollDelay(Duration.ofSeconds(1)).until(() -> messageRepository
388                 .findByMessageId(START_OUTBOUND_ACK_ID)
389                 .isPresent());
390 
391         assertEquals(2, messageCounter);
392 
393         if (first) {
394             verifyMessage_start();
395         }
396         verifyOutboundAck_start();
397 
398         assertNotNull(response);
399         assertEquals(HttpStatus.OK, response.getStatusCode());
400         assertEquals(0, response.getHeaders().getContentLength());
401         assertNull(response.getBody());
402     }
