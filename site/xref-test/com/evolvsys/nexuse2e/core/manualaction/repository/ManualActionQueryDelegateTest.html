21  class ManualActionQueryDelegateTest {
22  
23      @Test
24      void filter() {
25          ManualActionFilter filter = ManualActionFilter.builder()
26                  .types(EnumSet.of(ManualActionType.MESSAGE_ACTION, ManualActionType.PURGE))
27                  .to(ZonedDateTime.of(2024, 2, 3, 11, 33, 55, 0, ZoneId.of("UTC")))
28                  .from(ZonedDateTime.of(2024, 1, 1, 0, 0, 0, 0, ZoneId.of("Europe/Berlin")))
29                  .userId(123L)
30                  .messageId(444L)
31                  .build();
32  
33          assertAllNotNull(filter);
34  
35          Predicate predicate = ManualActionQueryDelegate.filter(QManualActionEntity.manualActionEntity, filter);
36  
37          assertThat(predicate.toString())
38                  .isEqualTo(
39                          """
40                                     manualActionEntity.user.id = 123
41                                      && manualActionEntity.type in [MESSAGE_ACTION, PURGE]
42                                      && manualActionEntity.created >= 2024-01-01T00:00+01:00[Europe/Berlin]
43                                      && manualActionEntity.created <= 2024-02-03T11:33:55Z[UTC]
44                                      && any(manualActionEntity.messages).id = 444"""
45                                  .replace("\n", ""));
46      }
