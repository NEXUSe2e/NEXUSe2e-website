3   import static org.assertj.core.api.Assertions.assertThat;
4   import static org.assertj.core.api.Assertions.assertThatThrownBy;
5   import static org.mockito.Mockito.*;
6   
7   import com.evolvsys.nexuse2e.core.exception.NexusRuntimeException;
8   import com.evolvsys.nexuse2e.core.purge.model.PurgeResult;
9   import com.fasterxml.jackson.core.JsonProcessingException;
10  import com.fasterxml.jackson.databind.ObjectMapper;
11  import org.junit.jupiter.api.Test;
12  
13  /**
14   * Test for {@link PurgeResultConverter}
15   *
16   * @author SvenBarden
17   * @since 12.08.2024
18   */
19  class PurgeResultConverterTest {
20  
21      private final ObjectMapper objectMapper = mock(ObjectMapper.class);
22      private final PurgeResultConverter converter = new PurgeResultConverter(objectMapper);
23  
24      @Test
25      void convertToDatabaseColumn() throws JsonProcessingException {
26          PurgeResult purgeResult = PurgeResult.builder().build();
27          String expectedJson = "{\"key\":\"value\"}";
28          when(objectMapper.writeValueAsString(purgeResult)).thenReturn(expectedJson);
29  
30          String result = converter.convertToDatabaseColumn(purgeResult);
31  
32          assertThat(result).isEqualTo(expectedJson);
33          verify(objectMapper).writeValueAsString(purgeResult);
34          verifyNoMoreInteractions(objectMapper);
35      }
36  
37      @Test
38      void convertToDatabaseColumn_null() {
39          String result = converter.convertToDatabaseColumn(null);
40  
41          assertThat(result).isNull();
42      }
43  
44      @Test
45      void convertToDatabaseColumn_throwsException() throws JsonProcessingException {
46          PurgeResult purgeResult = PurgeResult.builder().build();
47          when(objectMapper.writeValueAsString(purgeResult)).thenThrow(new JsonProcessingException("error") {});
48  
49          assertThatThrownBy(() -> converter.convertToDatabaseColumn(purgeResult))
50                  .isInstanceOf(NexusRuntimeException.class)
51                  .hasMessageContaining("Could not serialize purge result");
52          verify(objectMapper).writeValueAsString(purgeResult);
53          verifyNoMoreInteractions(objectMapper);
54      }
55  
56      @Test
57      void convertToEntityAttribute() throws JsonProcessingException {
58          String json = "{\"key\":\"value\"}";
59          PurgeResult expectedPurgeResult = PurgeResult.builder().build();
60          when(objectMapper.readValue(json, PurgeResult.class)).thenReturn(expectedPurgeResult);
61  
62          PurgeResult result = converter.convertToEntityAttribute(json);
63  
64          assertThat(result).isEqualTo(expectedPurgeResult);
65          verify(objectMapper).readValue(json, PurgeResult.class);
66          verifyNoMoreInteractions(objectMapper);
67      }
68  
69      @Test
70      void convertToEntityAttribute_null() {
71          PurgeResult result = converter.convertToEntityAttribute(null);
72  
73          assertThat(result).isNull();
74      }
75  
76      @Test
77      void convertToEntityAttribute_throwsException() throws JsonProcessingException {
78          String json = "{\"key\":\"value\"}";
79          when(objectMapper.readValue(json, PurgeResult.class)).thenThrow(new JsonProcessingException("error") {});
80  
81          assertThatThrownBy(() -> converter.convertToEntityAttribute(json))
82                  .isInstanceOf(NexusRuntimeException.class)
83                  .hasMessageContaining("Could not deserialize purge result");
84          verify(objectMapper).readValue(json, PurgeResult.class);
85          verifyNoMoreInteractions(objectMapper);
86      }
87  }
