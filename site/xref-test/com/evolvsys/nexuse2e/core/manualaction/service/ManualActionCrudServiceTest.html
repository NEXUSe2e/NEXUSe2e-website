27  class ManualActionCrudServiceTest {
28      private final ManualActionRepository manualActionRepository = mock(ManualActionRepository.class);
29      private final ManualActionMapper manualActionMapper = mock(ManualActionMapper.class);
30      private final CurrentUserService currentUserService = mock(CurrentUserService.class);
31      private final SecurityQueryService securityQueryService = mock(SecurityQueryService.class);
32      private final ManualActionCrudService manualActionCrudService = new ManualActionCrudService(
33              manualActionRepository, manualActionMapper,
34              currentUserService, securityQueryService);
35  
36      @Test
37      void list() {
38          ManualActionFilter manualActionFilter = ManualActionFilter.builder().build();
39          PageRequest pageRequest = PageRequest.of(10, 2);
40          CurrentUserData currentUserData = CurrentUserData.builder().build();
41          ManualActionEntity manualActionEntity1 = new ManualActionEntity1();
42          ManualActionEntity manualActionEntity2 = new ManualActionEntity2();
43          ManualActionDto manualActionDto1 = new ManualActionDto1();
44          ManualActionDto manualActionDto2 = new ManualActionDto2();
45          Page<ManualActionEntity> page = new PageImpl<>(List.of(manualActionEntity1, manualActionEntity2));
46  
47          BooleanExpression expression = mock(BooleanExpression.class);
48  
49          when(currentUserService.getCurrentUserData()).thenReturn(currentUserData);
50          when(securityQueryService.getPredicate(ManualActionDto.class, currentUserData))
51                  .thenReturn(expression);
52          when(manualActionRepository.findActionsToRead(pageRequest, manualActionFilter, expression))
53                  .thenReturn(page);
54          when(manualActionMapper.entityToDto(manualActionEntity1, currentUserData))
55                  .thenReturn(manualActionDto1);
56          when(manualActionMapper.entityToDto(manualActionEntity2, currentUserData))
57                  .thenReturn(manualActionDto2);
58  
59          Page<ManualActionDto> result = manualActionCrudService.list(pageRequest, manualActionFilter);
60  
61          verify(currentUserService, times(3)).getCurrentUserData();
62          verify(securityQueryService).getPredicate(ManualActionDto.class, currentUserData);
63          verify(manualActionRepository).findActionsToRead(pageRequest, manualActionFilter, expression);
64          verify(manualActionMapper).entityToDto(manualActionEntity1, currentUserData);
65          verify(manualActionMapper).entityToDto(manualActionEntity2, currentUserData);
66  
67          verifyNoMoreInteractions(currentUserService, securityQueryService, manualActionRepository, manualActionMapper);
68  
69          assertThat(result.getContent()).containsExactly(manualActionDto1, manualActionDto2);
70      }
71  
72      private static final class ManualActionEntity1 extends ManualActionEntity {}
73  
74      private static final class ManualActionEntity2 extends ManualActionEntity {}
75  
76      private static final class ManualActionDto1 extends ManualActionDto {}
77  
78      private static final class ManualActionDto2 extends ManualActionDto {}
79  }
