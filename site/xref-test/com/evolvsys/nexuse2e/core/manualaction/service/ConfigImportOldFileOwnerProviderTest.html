13  /**
14   * Test class for {@link ConfigImportOldFileOwnerProvider}.
15   *
16   * @author SvenBarden
17   * @since 09.08.2024
18   */
19  class ConfigImportOldFileOwnerProviderTest {
20      private final ConfigImportManualActionCrudService configImportCrudService =
21              mock(ConfigImportManualActionCrudService.class);
22      private final ConfigImportOldFileOwnerProvider configImportOldFileOwnerProvider =
23              new ConfigImportOldFileOwnerProvider(configImportCrudService);
24  
25      @Test
26      void getFiles() {
27          ConfigImportManualActionDto manualActionDto = new ConfigImportManualActionDto();
28          FileDto oldConfig = new FileDto();
29          oldConfig.setFileName("old");
30          FileDto newConfig = new FileDto();
31          newConfig.setFileName("new");
32          manualActionDto.setOldConfig(oldConfig);
33          manualActionDto.setNewConfig(newConfig);
34  
35          Collection<FileDto> files = configImportOldFileOwnerProvider.getFiles(manualActionDto);
36  
37          assertThat(files).containsExactly(oldConfig);
38          assertThat(files).doesNotContain(newConfig);
39  
40          verifyNoInteractions(configImportCrudService);
41      }
42  
43      @ParameterizedTest
44      @ValueSource(longs = {10, 20})
45      void getOwnerDto(Long ownerId) {
46          ConfigImportManualActionDto manualActionDto = new ConfigImportManualActionDto();
47  
48          when(configImportCrudService.getOrThrow(ownerId)).thenReturn(manualActionDto);
49  
50          ConfigImportManualActionDto result = configImportOldFileOwnerProvider.getOwnerDto(ownerId);
51  
52          assertThat(result).isSameAs(manualActionDto);
53  
54          verify(configImportCrudService).getOrThrow(ownerId);
55          verifyNoMoreInteractions(configImportCrudService);
56      }
57  
58      @Test
59      void getType() {
60          assertThat(configImportOldFileOwnerProvider.getType()).isSameAs(ConfigImportOldFileOwnerProvider.OLD_CONFIG);
61      }
62  }
