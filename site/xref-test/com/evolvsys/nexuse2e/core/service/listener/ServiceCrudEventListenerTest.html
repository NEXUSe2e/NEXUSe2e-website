20  class ServiceCrudEventListenerTest {
21      private final ServiceFactory serviceFactory = mock(ServiceFactory.class);
22      private final ServiceCrudEventListener serviceCrudEventListener = new ServiceCrudEventListener(serviceFactory);
23  
24      @Test
25      void preDeleteInternal() {
26          ServiceEntity serviceEntity = new ServiceEntity();
27          Service service = mock(OutflowService.class);
28  
29          when(serviceFactory.get(serviceEntity)).thenReturn(service);
30  
31          serviceCrudEventListener.preDeleteInternal(serviceEntity);
32  
33          verify(serviceFactory).get(serviceEntity);
34          verify(service).stop();
35  
36          verifyNoMoreInteractions(serviceFactory, service);
37      }
38  
39      @ParameterizedTest
40      @NullSource
41      @ValueSource(longs = {1L, 2L})
42      void preSaveInternal(Long id) {
43          ServiceEntity serviceEntity = new ServiceEntity();
44          serviceEntity.setId(id);
45  
46          if (id == null) {
47              serviceCrudEventListener.preSaveInternal(serviceEntity);
48  
49              verifyNoInteractions(serviceFactory);
50          } else {
51              Service service = mock(OutflowService.class);
52  
53              when(serviceFactory.get(serviceEntity)).thenReturn(service);
54  
55              serviceCrudEventListener.preSaveInternal(serviceEntity);
56  
57              verify(serviceFactory).get(serviceEntity);
58              verify(service).stop();
59  
60              verifyNoMoreInteractions(serviceFactory, service);
61          }
62      }
63  
64      @ParameterizedTest
65      @ValueSource(booleans = {true, false})
66      void postSaveInternal(boolean isEnabled) {
67          ServiceEntity serviceEntity = new ServiceEntity();
68          serviceEntity.setEnabled(isEnabled);
69  
70          if (isEnabled) {

72

76  
77              verify(serviceFactory).get(serviceEntity);

79  
80              verifyNoMoreInteractions(serviceFactory, service);
81          } else {
82              serviceCrudEventListener.postSaveInternal(serviceEntity);
83  
84              verifyNoInteractions(serviceFactory);
85          }
86      }
