24  class ServiceRunOnceValidCheckTest {
25      private final ComponentRegistry componentRegistry = mock(ComponentRegistry.class);
26      private final ServiceRunOnceValidCheck serviceRunOnceValidCheck = new ServiceRunOnceValidCheck(componentRegistry);
27      private final ServiceRunOnceValidCheck serviceRunOnceValidCheckMock =
28              mock(ServiceRunOnceValidCheck.class, withSettings().spiedInstance(serviceRunOnceValidCheck));
29  
30      @ParameterizedTest
31      @MethodSource("provideClassesAndExpectedResults")
32      void isValidTest(Class<?> loadedClass, boolean expected) {
33          RegisteredNexusComponent component =
34                  RegisteredNexusComponent.builder().loadedClass(loadedClass).build();
35  
36          String componentName = loadedClass.getSimpleName();
37  
38          when(componentRegistry.getComponent(componentName)).thenReturn(Optional.of(component));
39  
40          boolean result = serviceRunOnceValidCheck.isValid(componentName);
41          assertEquals(expected, result);
42  
43          verify(componentRegistry).getComponent(componentName);
44          verifyNoMoreInteractions(componentRegistry);
45      }
46  
47      @ParameterizedTest
48      @ValueSource(booleans = {true, false})
49      void isValidTest_withDto(boolean checkResult) {
50          ServiceDto dto = new ServiceDto();
51          dto.setComponentId("TestRunnableService");
52  
53          when(serviceRunOnceValidCheckMock.isValid(dto)).thenCallRealMethod();
54          when(serviceRunOnceValidCheckMock.isValid("TestRunnableService")).thenReturn(checkResult);
55  
56          assertEquals(checkResult, serviceRunOnceValidCheckMock.isValid(dto));
57  
58          verify(serviceRunOnceValidCheckMock).isValid(dto);
59          verify(serviceRunOnceValidCheckMock).isValid("TestRunnableService");
60          verifyNoMoreInteractions(serviceRunOnceValidCheckMock);
61          verifyNoInteractions(componentRegistry);
62      }
63  
64      @ParameterizedTest
65      @ValueSource(booleans = {true, false})
66      void isValidTest_withEntity(boolean checkResult) {
67          ServiceEntity entity = new ServiceEntity();
68          entity.setComponentId("TestRunnableService");
69  
70          when(serviceRunOnceValidCheckMock.isValid(entity)).thenCallRealMethod();
71          when(serviceRunOnceValidCheckMock.isValid("TestRunnableService")).thenReturn(checkResult);
72  
73          assertEquals(checkResult, serviceRunOnceValidCheckMock.isValid(entity));
74  
75          verify(serviceRunOnceValidCheckMock).isValid(entity);
76          verify(serviceRunOnceValidCheckMock).isValid("TestRunnableService");
77          verifyNoMoreInteractions(serviceRunOnceValidCheckMock);
78          verifyNoInteractions(componentRegistry);
79      }
80  
81      private static Stream<Arguments> provideClassesAndExpectedResults() {
82          return Stream.of(
83                  Arguments.of(TestRunnableService.class, true), Arguments.of(TestNotRunnableService.class, false));
84      }
