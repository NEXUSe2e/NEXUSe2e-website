153     private static X509Certificate createMockCert(CertState state, boolean selfSigned)
154             throws CertificateNotYetValidException, CertificateExpiredException {
155         X509Certificate cert = mock(X509Certificate.class);
156         if (state == CertState.NOT_YET_VALID) {
157             doThrow(new CertificateNotYetValidException("Invalid certificate"))
158                     .when(cert)
159                     .checkValidity();
160         } else if (state == CertState.EXPIRED) {
161             doThrow(new CertificateExpiredException("Invalid certificate"))
162                     .when(cert)
163                     .checkValidity();
164         }
165 
166         when(cert.getSubjectX500Principal()).thenReturn(new X500Principal("CN=Test" + ++CERT_COUNT));
167         if (selfSigned) {
168             when(cert.getIssuerX500Principal()).thenReturn(new X500Principal("CN=Test" + CERT_COUNT));
169         } else {
170             when(cert.getIssuerX500Principal()).thenReturn(new X500Principal("CN=Test" + ++CERT_COUNT));
171         }
172 
173         return cert;
174     }

276     @ParameterizedTest(name = "{index} - {0}")
277     @MethodSource
278     void getTrustedCertificates_partner_withChain(String label, List<PartnerCert> partnerCerts) throws NexusException {
279         List<X509Certificate> caCerts = List.of(mock(X509Certificate.class), mock(X509Certificate.class));
280 
281         try (MockedStatic<CertificateHelper> certificateHelperMockedStatic = mockStatic(CertificateHelper.class)) {
282             List<X509Certificate> partnerCertList = new ArrayList<>(partnerCerts.size());
283             List<X509Certificate> expected = new ArrayList<>();
284             for (PartnerCert partnerCert : partnerCerts) {
285                 partnerCertList.add(partnerCert.getCertificate());
286                 certificateHelperMockedStatic
287                         .when(() -> CertificateHelper.buildChain(partnerCert.getCertificate(), caCerts))
288                         .thenReturn(partnerCert.getChain());
289                 if (partnerCert.isChainValid()) {
290                     expected.add(partnerCert.getCertificate());
291                 }
292             }
293 
294             WebClientContext webClientContext = WebClientContext.builder()
295                     .validateCertChain(true)
296                     .trustedCerts(caCerts)
297                     .partnerCertificates(partnerCertList)
298                     .build();
299 
300             if (expected.isEmpty()) {
301                 assertThatThrownBy(() -> HttpContextSslUtils.getTrustedCertificates(webClientContext))
302                         .isInstanceOf(NexusException.class)
303                         .hasMessageContaining("No trusted partner certificates with valid chain found");
304             } else {
305                 List<X509Certificate> result = HttpContextSslUtils.getTrustedCertificates(webClientContext);
306                 assertThat(result).isEqualTo(expected);
307                 assertThat(result).isNotEqualTo(caCerts);
308             }
309         }
310     }
