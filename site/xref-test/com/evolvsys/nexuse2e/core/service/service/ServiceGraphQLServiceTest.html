56      @Test
57      void update_valid() {
58          when(componentRegistry.exists("component 1")).thenReturn(true);
59  
60          ServiceDto serviceDto = new ServiceDto();
61          serviceDto.setId(40L);
62  
63          ServiceDto expected = new ServiceDto()
64                  .setServiceId("my service 1")
65                  .setComponentId("component 1")
66                  .setEnabled(true);
67          expected.getWriteMetaData().setInputProperties(Set.of("serviceId", "componentId", "enabled"));
68  
69          when(serviceService.save((ServiceDto) any())).thenReturn(serviceDto);
70  
71          GraphQlTester.Response response =
72                  graphQlTester.documentName("service/service_update").execute();
73          response.errors().verify();
74  
75          response.path("service_update.id").entity(Long.class).isEqualTo(40L);
76          response.path("service_update.componentId").pathDoesNotExist();
77  
78          ArgumentCaptor<ServiceDto> serviceDtoArgumentCaptor = ArgumentCaptor.forClass(ServiceDto.class);
79          verify(serviceService).save(serviceDtoArgumentCaptor.capture());
80          verifyNoMoreInteractions(serviceService);
81          verifyNoInteractions(dataLoaderService);
82  
83          assertThat(serviceDtoArgumentCaptor.getValue()).isEqualTo(expected);
84      }

218     @Test
219     void getById() {
220         ServiceDto serviceDto = new ServiceDto().setServiceId("my service");
221         PipelineDto errorPipeline = new PipelineDto().setDescription("error pipeline");
222         PipelineDto returnPipeline = new PipelineDto().setDescription("return pipeline");
223         PipelineDto resolverPipeline = new PipelineDto().setDescription("resolver pipeline");
224 
225         List<ParameterDto> parameterDtos =
226                 List.of(new ParameterDto().setName("param 1"), new ParameterDto().setName("param 2"));
227 
228         when(serviceService.getById(60L)).thenReturn(Optional.of(serviceDto));
229         when(dataLoaderService.load(
230                         any(), eq(PipelineBatchLoader.class), same(serviceDto), eq("errorPipelineId"), any()))
231                 .thenReturn(CompletableFuture.completedFuture(errorPipeline));
232         when(dataLoaderService.load(
233                         any(), eq(PipelineBatchLoader.class), same(serviceDto), eq("returnPipelineId"), any()))
234                 .thenReturn(CompletableFuture.completedFuture(returnPipeline));
235         when(dataLoaderService.load(
236                         any(), eq(PipelineBatchLoader.class), same(serviceDto), eq("resolverPipelineId"), any()))
237                 .thenReturn(CompletableFuture.completedFuture(resolverPipeline));
238         when(dataLoaderService.load(any(), eq(ServiceParameterBatchLoader.class), same(serviceDto), any()))
239                 .thenReturn(CompletableFuture.completedFuture(new LinkedHashSet<>(parameterDtos)));
240 
241         GraphQlTester.Response response =
242                 graphQlTester.documentName("service/service_getById").execute();
243         response.errors().verify();
244 
245         response.path("service_getById.id").pathDoesNotExist();
246         response.path("service_getById.serviceId").entity(String.class).isEqualTo("my service");
247         response.path("service_getById.errorPipeline.description")
248                 .entity(String.class)
249                 .isEqualTo(errorPipeline.getDescription());
250         response.path("service_getById.parameters")
251                 .entityList(ParameterDto.class)
252                 .isEqualTo(parameterDtos);
253 
254         verify(serviceService).getById(60L);
255         verify(dataLoaderService)
256                 .load(any(), eq(PipelineBatchLoader.class), same(serviceDto), eq("errorPipelineId"), any());
257         verify(dataLoaderService)
258                 .load(any(), eq(PipelineBatchLoader.class), same(serviceDto), eq("returnPipelineId"), any());
259         verify(dataLoaderService)
260                 .load(any(), eq(PipelineBatchLoader.class), same(serviceDto), eq("resolverPipelineId"), any());
261         verify(dataLoaderService).load(any(), eq(ServiceParameterBatchLoader.class), same(serviceDto), any());
262         verifyNoMoreInteractions(serviceService, dataLoaderService);
263     }
