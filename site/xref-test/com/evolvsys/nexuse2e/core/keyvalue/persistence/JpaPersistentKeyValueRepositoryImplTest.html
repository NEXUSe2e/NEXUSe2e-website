18  @SuppressWarnings("PMD.TestClassWithoutTestCases") // pmd does not know carthesian tests
19  class JpaPersistentKeyValueRepositoryImplTest {
20      private final KeyValuePropertyRepository keyValuePropertyRepository = mock(KeyValuePropertyRepository.class);
21      private final JpaPersistentKeyValueRepositoryImpl jpaPersistentKeyValueRepository =
22              new JpaPersistentKeyValueRepositoryImpl(keyValuePropertyRepository);
23  
24      @CartesianTest
25      void save_notFound(
26              @CartesianTest.Values(strings = {"cat1", "cat2"}) String category,
27              @CartesianTest.Values(strings = {"key1", "key2"}) String key,
28              @CartesianTest.Values(strings = {"value1", "value2"}) String value) {
29          jpaPersistentKeyValueRepository.save(category, key, value);
30  
31          ArgumentCaptor<KeyValuePropertyEntity> argumentCaptor = ArgumentCaptor.forClass(KeyValuePropertyEntity.class);
32  
33          verify(keyValuePropertyRepository).findByCategoryAndKey(category, key);
34          verify(keyValuePropertyRepository).save(argumentCaptor.capture());
35          verifyNoMoreInteractions(keyValuePropertyRepository);
36  
37          assertThat(argumentCaptor.getValue().getValue()).isEqualTo(value);
38      }
39  
40      @CartesianTest
41      void save_found(
42              @CartesianTest.Values(strings = {"cat1", "cat2"}) String category,
43              @CartesianTest.Values(strings = {"key1", "key2"}) String key,
44              @CartesianTest.Values(strings = {"value1", "value2"}) String value) {
45          KeyValuePropertyEntity entity = new KeyValuePropertyEntity();
46          when(keyValuePropertyRepository.findByCategoryAndKey(category, key)).thenReturn(Optional.of(entity));
47  
48          jpaPersistentKeyValueRepository.save(category, key, value);
49  
50          verify(keyValuePropertyRepository).findByCategoryAndKey(category, key);
51          verify(keyValuePropertyRepository).save(same(entity));
52          verifyNoMoreInteractions(keyValuePropertyRepository);
53  
54          assertThat(entity.getValue()).isEqualTo(value);
55      }
56  
57      @CartesianTest
58      void get_notFound(
59              @CartesianTest.Values(strings = {"cat1", "cat2"}) String category,
60              @CartesianTest.Values(strings = {"key1", "key2"}) String key) {
61          assertThat(jpaPersistentKeyValueRepository.get(category, key)).isEmpty();
62          verify(keyValuePropertyRepository).findByCategoryAndKey(category, key);
63          verifyNoMoreInteractions(keyValuePropertyRepository);
64      }
65  
66      @CartesianTest
67      void get_found(
68              @CartesianTest.Values(strings = {"cat1", "cat2"}) String category,
69              @CartesianTest.Values(strings = {"key1", "key2"}) String key,
70              @CartesianTest.Values(strings = {"value1", "value2"}) String value) {
71          KeyValuePropertyEntity entity = new KeyValuePropertyEntity();
72          entity.setValue(value);
73          when(keyValuePropertyRepository.findByCategoryAndKey(category, key)).thenReturn(Optional.of(entity));
74  
75          assertThat(jpaPersistentKeyValueRepository.get(category, key)).hasValue(value);
76          verify(keyValuePropertyRepository).findByCategoryAndKey(category, key);
77          verifyNoMoreInteractions(keyValuePropertyRepository);
78      }
