3   import java.util.Map;
4   import lombok.experimental.UtilityClass;
5   import lombok.extern.slf4j.Slf4j;
6   import org.apache.commons.text.lookup.StringLookup;
7   import org.apache.commons.text.lookup.StringLookupFactory;
8   
9   /**
10   * Sets up the {@link KeyValueLookup} for testing.
11   *
12   * @author SvenBarden
13   * @since 11.04.2025
14   */
15  @UtilityClass
16  @Slf4j
17  public class KeyValueTestInitializer {
18  
19      public static void init() {
20          init(Map.of());
21      }
22  
23      public static void init(Map<String, StringLookup> lookupMap) {
24          LOGGER.trace("initializing KeyValueLookup with {}", lookupMap);
25          StringLookup lookup =
26                  StringLookupFactory.INSTANCE.interpolatorStringLookup(lookupMap, new DefaultKeyValueProvider(), false);
27  
28          KeyValueLookup.setStringLookup(lookup);
29      }
30  
31      public static void reset() {
32          KeyValueLookup.setStringLookup(null);
33      }
34  }
