58      @Test
59      void xml() {
60          RhinoScript rhinoScript = new RhinoScript();
61  
62          byte[] content = EBXML_HEADER.getBytes(StandardCharsets.UTF_8);
63  
64          ContextPayload payload = new ContextPayload();
65          payload.setContent(PayloadContent.of(content));
66          payload.setSize(content.length);
67          payload.setEncoding("utf-8");
68  
69          Message message = new Message();
70          MessageContext messageContext = new TestMessageContext();
71          messageContext.setSourceMessage(message);
72          ExecutionContext context = ContextFactory.createExecutionContext(messageContext, MessageType.NORMAL);
73          message.getMetaData().set("header", payload);
74  
75          rhinoScript.evaluate(
76                  """
77                                       const headerPayload = context.message.metaData.getFirstOrNull('header');
78                                       if(headerPayload) {
79                                           var headerContent = asString(headerPayload);
80                                           //remove first line
81                                           headerContent = headerContent.replace(/<\\?[^>]*>/g,'').trim();
82                                           logInfo('got header payload:' +  headerContent);
83  
84                                           const parser = new XML(headerContent);
85  
86                                           const data = parser.*::Header.*::MessageHeader.*::From.*;
87                                           const partnerId = data.toString();
88  
89                                           logInfo('test: ' + data.toString());
90  
91                                           context.message.metaData.set('result', partnerId);
92                                       }
93                                       """,
94                  String.class,
95                  new RhinoScript.Parameter("context", context));
96  
97          assertEquals("DEV_OLD", message.getMetaData().getFirstOrNull("result"));
98      }

100     @Test
101     void evaluate() {
102 
103         RhinoScript rhinoScript = new RhinoScript();
104 
105         byte[] payload = "hello world".getBytes(StandardCharsets.UTF_8);
106 
107         ContextPayload payload1 = new ContextPayload();
108         payload1.setContent(PayloadContent.of(payload));
109         payload1.setSize(payload.length);
110         payload1.setEncoding("utf-8");
111 
112         Message message = new Message();
113         message.setPayloads(List.of(payload1));
114         MessageContext messageContext = new TestMessageContext();
115         messageContext.setSourceMessage(message);
116         ExecutionContext context = ContextFactory.createExecutionContext(messageContext, MessageType.NORMAL);
117         context.getMetaData().set("meta1", "meta value");
118 
119         rhinoScript.evaluate(
120                 """
121                                      asString(context.message.payloads[0].content, context.message.payloads[0].encoding);
122 
123                                      var content = asString(context.message.payloads[0]);
124 
125                                      var metaValue = context.metaData.getFirstOrNull('meta1');
126 
127                                      logInfo('processing content ' + content);
128 
129                                      var result = content + '\\n' + metaValue;
130 
131                                      setContent(context.message.payloads[0], result);
132                                      """,
133                 String.class,
134                 new RhinoScript.Parameter("context", context));
135 
136         assertEquals(
137                 """
138                              hello world
139                              meta value""",
140                 payload1.getContent().getString(StandardCharsets.UTF_8));
