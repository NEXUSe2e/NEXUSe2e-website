24  class CertificateQueryDelegateTest {
25      @Test
26      void filterTest() {
27          CertificateFilter filter = CertificateFilter.builder()
28                  .types(EnumSet.allOf(CertificateType.class))
29                  .valid(BooleanFilterType.TRUE)
30                  .active(BooleanFilterType.TRUE)
31                  .sha1Fingerprint("a421bd3928af3283")
32                  .sha256Fingerprint("62ab263810381ab382cd5f")
33                  .commonName("commonName_value")
34                  .organisation("organisation_value")
35                  .organisationUnit("organisationUnit_value")
36                  .country("country_value")
37                  .state("state_value")
38                  .location("location_value")
39                  .email("email_value")
40                  .issuer("issuer_value")
41                  .selfSigned(BooleanFilterType.FALSE)
42                  .build();
43          assertAllNotNull(filter);
44  
45          String expectedPredicate =
46                  """
47                          certificateEntity.info.validFrom < 2024-04-11T15:30:20Z[GMT]
48                           && certificateEntity.info.validUntil > 2024-04-11T15:30:20Z[GMT]
49                           && certificateEntity.type in [LOCAL, CA, PARTNER]
50                           && certificateEntity.active = true
51                           && contains(certificateEntity.info.sha1Fingerprint,a421bd3928af3283)
52                           && contains(certificateEntity.info.sha256Fingerprint,62ab263810381ab382cd5f)
53                           && contains(certificateEntity.info.commonName,commonName_value)
54                           && contains(certificateEntity.info.organisation,organisation_value)
55                           && contains(certificateEntity.info.organisationUnit,organisationUnit_value)
56                           && contains(certificateEntity.info.country,country_value)
57                           && contains(certificateEntity.info.state,state_value)
58                           && contains(certificateEntity.info.location,location_value)
59                           && contains(certificateEntity.info.email,email_value)
60                           && contains(certificateEntity.info.issuer,issuer_value)
61                           && certificateEntity.info.selfSigned = false
62                          """
63                          .replace("\n", "");
64  
65          ZonedDateTime mockedNow = ZonedDateTime.of(2024, 4, 11, 15, 30, 20, 0, ZoneId.of("GMT"));
66  
67          try (MockedStatic<ZonedDateTime> mockedStatic =
68                  mockStatic(ZonedDateTime.class, InvocationOnMock::callRealMethod)) {
69              mockedStatic.when(ZonedDateTime::now).thenReturn(mockedNow);
70  
71              assertEquals(
72                      expectedPredicate,
73                      QCertificateEntity.certificateEntity.filter(filter).toString());
74          }
75      }
76  }
