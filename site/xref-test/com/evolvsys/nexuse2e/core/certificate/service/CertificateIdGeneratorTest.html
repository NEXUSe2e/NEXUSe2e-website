34                  getClass().getResourceAsStream(TestCertificateConstants.CERTIFICATE_PATH));
35  
36          String id = certificateIdGenerator.createCertificateId(x509Certificate).orElseThrow();
37  
38          assertEquals("dev1.local", id);
39      }
40  
41      @ParameterizedTest
42      @ValueSource(booleans = {true, false})
43      void createCertificateId(boolean replaceNonPrintable) throws NexusException {
44          X509Certificate x509Certificate = CertificateHelper.getCertificate(
45                  getClass().getResourceAsStream(TestCertificateConstants.CERTIFICATE_PATH));
46  
47          String id = certificateIdGenerator
48                  .createCertificateId(x509Certificate, replaceNonPrintable)
