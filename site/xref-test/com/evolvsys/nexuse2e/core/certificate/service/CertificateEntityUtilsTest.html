27  @Slf4j
28  class CertificateEntityUtilsTest {
29      private final CertificateRepository certificateRepository = mock(CertificateRepository.class);
30      private final CertificateEntityUtils certificateEntityUtils = new CertificateEntityUtils(certificateRepository);
31      private final CertificateEntityUtils certificateEntityUtilsMock =
32              mock(CertificateEntityUtils.class, withSettings().spiedInstance(certificateEntityUtils));
33  
34      @ParameterizedTest
35      @CsvSource({"LOCAL,true", "CA,false", "PARTNER,false"})
36      void isPrivateCertTest(CertificateType certificateType, boolean expected) {
37          CertificateEntity certificateEntity = new CertificateEntity();
38          certificateEntity.setType(certificateType);
39          when(certificateRepository.findById(4711L)).thenReturn(Optional.of(certificateEntity));
40          assertEquals(expected, certificateEntityUtils.isPrivateCert(4711L));
41      }
42  
43      @Test
44      void getPrivateKeyTest() throws NexusException {
45          CertificateEntity certificateEntity = new CertificateEntity();
46          certificateEntity.setType(CertificateType.LOCAL);
47          certificateEntity.setPassword("nexuse2e");
48  
49          when(certificateRepository.findById(4711L)).thenReturn(Optional.of(certificateEntity));
50  
51          KeyStore keyStore = mock(KeyStore.class);
52          PrivateKey privateKey = mock(PrivateKey.class);
53  
54          try (MockedStatic<CertificateKeystoreUtils> mockedStaticCertificateKeystoreUtils =
55                  mockStatic(CertificateKeystoreUtils.class, InvocationOnMock::callRealMethod)) {
56              mockedStaticCertificateKeystoreUtils
57                      .when(() -> CertificateKeystoreUtils.toKeyStore(certificateEntity))
58                      .thenReturn(keyStore);
59              mockedStaticCertificateKeystoreUtils
60                      .when(() -> CertificateKeystoreUtils.getPrivateKey(keyStore, "nexuse2e"))
61                      .thenReturn(privateKey);
62  
63              assertSame(privateKey, certificateEntityUtils.getPrivateKey(4711L));
64  
65              mockedStaticCertificateKeystoreUtils.verify(() -> CertificateKeystoreUtils.toKeyStore(certificateEntity));
66              mockedStaticCertificateKeystoreUtils.verify(
67                      () -> CertificateKeystoreUtils.getPrivateKey(keyStore, "nexuse2e"), times(2));
68          }
69  
70          verify(certificateRepository, times(2)).findById(4711L);
71          verifyNoMoreInteractions(certificateRepository);
72      }
73  
74      @Test
75      void getPrivateKeyTest_noKeystore() throws NexusException {
76          CertificateEntity certificateEntity = new CertificateEntity();
77          certificateEntity.setId(4711L);
78          certificateEntity.setName("keystoreJune2024");
79  
80          when(certificateEntityUtilsMock.getPrivateKey(4711L)).thenCallRealMethod();
81          when(certificateRepository.findById(4711L)).thenReturn(Optional.of(certificateEntity));
82          when(certificateEntityUtilsMock.isPrivateCert(4711L)).thenReturn(false);
83  
84          NexusException thrownException =
85                  assertThrows(NexusException.class, () -> certificateEntityUtilsMock.getPrivateKey(4711L));
86          NexusError nexusError = thrownException.getError();
87  
88          assertEquals("Unable to load private key as the entity is not a keystore", nexusError.getMessage());
89          assertEquals(4711L, nexusError.getExtensions().get("id"));
90          assertEquals("keystoreJune2024", nexusError.getExtensions().get("name"));
91  
92          verify(certificateEntityUtilsMock).getPrivateKey(4711L);
93          verify(certificateRepository).findById(4711L);
94          verify(certificateEntityUtilsMock).isPrivateCert(4711L);
95          verifyNoMoreInteractions(certificateEntityUtilsMock, certificateRepository);
96      }
97  }
