36  @Slf4j
37  class CertificatePemWriterTest {
38      @BeforeEach
39      void beforeEach() {
40          if (Security.getProvider("BC") == null) {
41              LOGGER.info("adding bouncy castle provider");
42              Security.addProvider(new BouncyCastleProvider());
43          }
44      }
45  
46      @Test
47      void toCertificateTest() throws NexusException, IOException {
48          X509Certificate x509Certificate = CertificateHelper.getCertificate(
49                  getClass().getResourceAsStream(TestCertificateConstants.CERTIFICATE_PATH));
50  
51          String expectedContent =
52                  IOUtils.resourceToString(TestCertificateConstants.CERTIFICATE_PATH, StandardCharsets.UTF_8);
53          assertEquals(expectedContent, CertificatePemWriter.toCertificate(x509Certificate));
54      }
55  
56      @Test
57      void toCertificateChainTest() throws NexusException, IOException {
58          X509Certificate caCertificate =
59                  CertificateHelper.getCertificate(getClass().getResourceAsStream(TestCertificateConstants.CA_PATH));
60          X509Certificate leafCertificate = CertificateHelper.getCertificate(
61                  getClass().getResourceAsStream(TestCertificateConstants.CERTIFICATE_PATH));
62  
63          String expectedContent =
64                  IOUtils.resourceToString("/certs/certificate_chain_expected.pem", StandardCharsets.UTF_8);
65          assertEquals(expectedContent, CertificatePemWriter.toCertificateChain(List.of(leafCertificate, caCertificate)));
66      }
67  
68      @ParameterizedTest
69      @CsvSource({"nexuse2e,ENCRYPTED PRIVATE KEY", ",PRIVATE KEY"})
70      void toEncryptedPrivateKeyTest(String password, String expectedType) throws NexusException, IOException {
71  
72          KeyStore keyStore = CertificateKeystoreUtils.toKeystore(
73                  Objects.requireNonNull(getClass().getResourceAsStream("/certs/certs-dev2/certificate.p12")),
74                  TestCertificateConstants.PASSWORD);
75          PrivateKey privateKey = CertificateKeystoreUtils.getPrivateKey(keyStore, TestCertificateConstants.PASSWORD);
76  
77          String privateKeyContent = CertificatePemWriter.toEncryptedPrivateKey(privateKey, password);
78          assertThat(privateKeyContent).startsWith("-----BEGIN " + expectedType);
79  
80          if ("ENCRYPTED PRIVATE KEY".equals(expectedType)) {
81              LOGGER.info("Decrypting private key of given pem content");
82              try (PEMParser pemParser = new PEMParser(new StringReader(privateKeyContent))) {
83                  Object object = pemParser.readObject();
84                  PEMDecryptorProvider decryptorProvider =
85                          new JcePEMDecryptorProviderBuilder().build(password.toCharArray());
86                  JcaPEMKeyConverter keyConverter = new JcaPEMKeyConverter();
87  
88                  if (object instanceof PEMEncryptedKeyPair pemEncryptedKeyPair) {
89                      PrivateKey decryptedPrivateKey = keyConverter
90                              .getKeyPair(pemEncryptedKeyPair.decryptKeyPair(decryptorProvider))
91                              .getPrivate();
92                      assertEquals(privateKey.getEncoded(), decryptedPrivateKey.getEncoded());
93                  }
94              }
95          }
96      }
