31  class ClientCertificateKeystoreBinaryTypeTest {
32      private final CertificateEntityUtils certificateEntityUtils = mock(CertificateEntityUtils.class);
33      private final CertificateChainLoader certificateChainLoader = mock(CertificateChainLoader.class);
34      private final CertificateInfoMapper certificateInfoMapper = mock(CertificateInfoMapper.class);
35      private final CertificateRepository certificateRepository = mock(CertificateRepository.class);
36      private final ClientCertificateKeystoreBinaryType clientCertificateKeystoreBinaryType =
37              new ClientCertificateKeystoreBinaryType(
38                      certificateEntityUtils, certificateChainLoader, certificateInfoMapper, certificateRepository);
39  
40      @Test
41      void nameTest() {
42          assertSame(ClientCertificateKeystoreBinaryType.NAME, clientCertificateKeystoreBinaryType.name());
43      }
44  
45      @Test
46      void getTest() throws NexusException, IOException {
47          X509Certificate certificate1 = mock(X509Certificate.class);
48          String encodedContent1 =
49                  """
50                          -----BEGIN CERTIFICATE-----
51                          encoded content of certificate 1
52                          -----END CERTIFICATE-----
53                          """;
54  
55          X509Certificate certificate2 = mock(X509Certificate.class);
56          String encodedContent2 =
57                  """
58                          -----BEGIN CERTIFICATE-----
59                          encoded content of certificate 2
60                          -----END CERTIFICATE-----
61                          """;
62  
63          List<X509Certificate> certificateChain = List.of(certificate1, certificate2);
64  
65          PrivateKey privateKey = mock(PrivateKey.class);
66          String privateKeyEncodedContent =
67                  """
68                          -----BEGIN PRIVATE KEY-----
69                          encoded private key
70                          -----END PRIVATE KEY-----
71                          """;
72  
73          CertificateInfo certificateInfo = new CertificateInfo();
74          certificateInfo.setCommonName("NEXUSe2e-leaf");
75  
76          CertificateEntity certificateEntity = new CertificateEntity();
77          certificateEntity.setName("keystoreJune2024");
78          certificateEntity.setPassword("nexuse2e");
79  
80          when(certificateEntityUtils.isPrivateCert(4711L)).thenReturn(true);
81          when(certificateChainLoader.getX509CertificateChain(4711L)).thenReturn(certificateChain);
82          when(certificateInfoMapper.certificateToInfo(certificate1)).thenReturn(certificateInfo);
83          when(certificateRepository.findById(4711L)).thenReturn(Optional.of(certificateEntity));
84          when(certificateEntityUtils.getPrivateKey(4711L)).thenReturn(privateKey);
85  
86          LocalDateTime mockedLocalDateTime = LocalDateTime.of(2024, 6, 5, 10, 30, 0);
87  
88          try (MockedStatic<LocalDateTime> mockedStaticLocalDateTime =
89                          mockStatic(LocalDateTime.class, InvocationOnMock::callRealMethod);
90                  MockedStatic<CertificatePemWriter> mockedStaticCertificatePemWriter =
91                          mockStatic(CertificatePemWriter.class, InvocationOnMock::callRealMethod)) {
92              mockedStaticLocalDateTime.when(LocalDateTime::now).thenReturn(mockedLocalDateTime);
93              mockedStaticCertificatePemWriter
94                      .when(() -> CertificatePemWriter.toEncryptedPrivateKey(privateKey, "nexuse2e"))
95                      .thenReturn(privateKeyEncodedContent);
96              mockedStaticCertificatePemWriter
97                      .when(() -> CertificatePemWriter.toCertificate(certificate1))
98                      .thenReturn(encodedContent1);
99              mockedStaticCertificatePemWriter
100                     .when(() -> CertificatePemWriter.toCertificate(certificate2))
101                     .thenReturn(encodedContent2);
102 
103             BinaryResult binaryResult =
104                     clientCertificateKeystoreBinaryType.get("4711").orElseThrow();
105 
106             assertEquals(
107                     "NEXUSe2e_privateKey_withChain_keystoreJune2024_2024-06-05T10_30.zip", binaryResult.getFileName());
108             assertEquals("application/zip", binaryResult.getMimeType());
109 
110             verify(certificateEntityUtils).isPrivateCert(4711L);
111             verify(certificateRepository).findById(4711L);
112             verify(certificateChainLoader).getX509CertificateChain(4711L);
113             verify(certificateInfoMapper).certificateToInfo(certificate1);
114             verify(certificateEntityUtils).getPrivateKey(4711L);
115 
116             mockedStaticLocalDateTime.verify(LocalDateTime::now, times(2));
117             mockedStaticCertificatePemWriter.verify(() -> CertificatePemWriter.toCertificateChain(certificateChain));
118             mockedStaticCertificatePemWriter.verify(
119                     () -> CertificatePemWriter.toEncryptedPrivateKey(privateKey, "nexuse2e"));
120 
121             try (ZipInputStream zipInputStream =
122                     new ZipInputStream(binaryResult.getResource().getInputStream())) {
123                 ZipEntry privateKeyZipEntry = zipInputStream.getNextEntry();
124                 String actualPrivateKeyEncodedContent =
125                         new String(zipInputStream.readAllBytes(), StandardCharsets.UTF_8);
126                 assertNotNull(privateKeyZipEntry);
127                 assertEquals("NEXUSe2e-leaf_private_key.pem", privateKeyZipEntry.getName());
128                 assertEquals(privateKeyEncodedContent.length(), privateKeyZipEntry.getSize());
129                 assertEquals(privateKeyEncodedContent, actualPrivateKeyEncodedContent);
130 
131                 ZipEntry certificateChainZipEntry = zipInputStream.getNextEntry();
132                 String expectedCertificateChainEncodedContent =
133                         """
134                                 -----BEGIN CERTIFICATE-----
135                                 encoded content of certificate 1
136                                 -----END CERTIFICATE-----
137                                 -----BEGIN CERTIFICATE-----
138                                 encoded content of certificate 2
139                                 -----END CERTIFICATE-----
140                                 """;
141                 String actualCertificateChainEncodedContent =
142                         new String(zipInputStream.readAllBytes(), StandardCharsets.UTF_8);
143                 assertNotNull(certificateChainZipEntry);
144                 assertEquals("NEXUSe2e-leaf_certificate_chain.pem", certificateChainZipEntry.getName());
145                 assertEquals(expectedCertificateChainEncodedContent.length(), certificateChainZipEntry.getSize());
146                 assertEquals(expectedCertificateChainEncodedContent, actualCertificateChainEncodedContent);
147             }
148         }
149     }
