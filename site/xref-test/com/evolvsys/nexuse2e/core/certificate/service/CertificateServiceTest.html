103         Blob blob = mock(Blob.class);

107         caEntity1.setInfo(new CertificateInfo());

109         fileEntity.setFileContent(new FileContentEntity());
110         fileEntity.getFileContent().setContent(blob);
111 
112         InputStream content = mock(InputStream.class);
113         X509Certificate certificate = mock(X509Certificate.class);
114 
115         when(blob.getBinaryStream()).thenReturn(content);
116         certificateHelperMockedStatic
117                 .when(() -> CertificateHelper.getCertificate(content))
118                 .thenReturn(certificate);
119         when(certificate.getEncoded()).thenReturn(encoded.getBytes(StandardCharsets.UTF_8));
120 
121         return caEntity1;
122     }

124     private String getSha1(String content) {
125         SHA1.Digest digest = new SHA1.Digest();
126 
127         byte[] bytes = content.getBytes(StandardCharsets.UTF_8);
128 
129         digest.update(bytes, 0, bytes.length);
130 
131         return new String(Hex.encode(digest.engineDigest()), StandardCharsets.UTF_8);
132     }

134     @Test
135     void uploadLocal() throws NexusException, SQLException, KeyStoreException, CertificateEncodingException {
136         FileEntity fileEntity = prepareP12FileEntity();
137         CertificateEntity certificateEntity = new CertificateEntity();
138         certificateEntity.setInfo(new CertificateInfo());
139         certificateEntity.getInfo().setSha1Fingerprint(TestCertificateConstants.PRIVATE_FINGERPRINT_SHA1);
140         CertificateEntity caEntity1 = prepareCaEntity("print 1");
141         CertificateEntity caEntity2 = prepareCaEntity("print 2");
142         CertificateContainerEntity certificateContainerEntity = new CertificateContainerEntity();
143         certificateContainerEntity.setType(CertificateContainerType.PRIVATE_KEY);
144         CertificateEntity savedEntity = new CertificateEntity();
145         savedEntity.setId(65L);
146 
147         CertificateEntity newCaEntity1 = new CertificateEntity();
148         CertificateEntity newCaEntity2 = new CertificateEntity();
149 
150         KeystoreUpload keystoreUpload = KeystoreUpload.builder()
151                 .password(TestCertificateConstants.PASSWORD)
152                 .fileId(15L)
153                 .certificateContainerId(50L)
154                 .build();
155 
156         when(certificateRepository.findByType(CertificateType.CA)).thenReturn(List.of(caEntity1, caEntity2));
157         when(certificateEntityFactory.createFromPKCS12(any(), any())).thenReturn(certificateEntity);
158         when(certificateEntityFactory.createMissingCaEntities(any(), any()))
159                 .thenReturn(List.of(newCaEntity1, newCaEntity2));
160         when(certificateContainerRepository.findById(50L)).thenReturn(Optional.of(certificateContainerEntity));
161         when(certificateRepository.existsByCertificateContainerAndInfo_sha1Fingerprint(
162                         certificateContainerEntity, TestCertificateConstants.PRIVATE_FINGERPRINT_SHA1))
163                 .thenReturn(false);
164         when(certificateRepository.save(certificateEntity)).thenReturn(savedEntity);
165 
166         long id = certificateService.uploadPrivateKey(keystoreUpload);
167 
168         ArgumentCaptor<KeyStore> keyStoreCaptor = ArgumentCaptor.forClass(KeyStore.class);
169 
170         verify(certificateEntityFactory).createFromPKCS12(keyStoreCaptor.capture(), same(keystoreUpload));
171 
172         KeyStore value = keyStoreCaptor.getValue();
173         X509Certificate privateKey = (X509Certificate) value.getCertificate(TestCertificateConstants.ALIAS);
174         assertNotNull(privateKey);
175         assertEquals("X.509", privateKey.getType());
176         assertEquals(TestCertificateConstants.PRIVATE_FINGERPRINT_SHA1, CertificateUtil.getSha1Fingerprint(privateKey));
177         assertEquals(65, id);
178 
179         verify(fileRepository).getFileEntityById(15L);
180         verify(fileRepository).delete(fileEntity);
181         verify(certificateRepository).findByType(CertificateType.CA);
182         verify(certificateEntityFactory).createMissingCaEntities(Set.of(getSha1("print 1"), getSha1("print 2")), value);
183         verify(certificateRepository).save(certificateEntity);
184         verify(certificateRepository).saveAll(List.of(newCaEntity1, newCaEntity2));
185         verify(certificateContainerRepository).findById(50L);
186         verify(certificateRepository)
187                 .existsByCertificateContainerAndInfo_sha1Fingerprint(
188                         certificateContainerEntity, TestCertificateConstants.PRIVATE_FINGERPRINT_SHA1);
189         verifyNoMoreInteractions(
190                 fileRepository, certificateEntityFactory, certificateRepository, certificateContainerRepository);
191     }

193     @Test
194     void uploadPartner() throws NexusException, SQLException {
195         CertificateContainerEntity certificateContainerEntity = new CertificateContainerEntity();
196         certificateContainerEntity.setType(CertificateContainerType.PUBLIC_KEY);
197         certificateContainerEntity.setPartner(new PartnerEntity());
198         certificateContainerEntity.getPartner().setPartnerType(PartnerType.EXTERN);
199         FileEntity fileEntity = prepareCertFileEntity();
200         CertificateEntity savedEntity = new CertificateEntity();
201         savedEntity.setId(55L);
202 
203         CertificateUpload certificateUpload = CertificateUpload.builder()
204                 .name("mycert")
205                 .certificateContainerId(50L)
206                 .fileId(15L)
207                 .build();
208 
209         X509Certificate x509Certificate = mock(X509Certificate.class);
210         CertificateInfo certificateInfo = new CertificateInfo();
211         certificateInfo.setSha1Fingerprint("finger");
212 
213         when(certificateContainerRepository.findById(50L)).thenReturn(Optional.of(certificateContainerEntity));
214         when(certificateInfoMapper.certificateToInfo(x509Certificate)).thenReturn(certificateInfo);
215         when(certificateRepository.save(any())).thenReturn(savedEntity);
216 
217         certificateHelperMockedStatic
218                 .when(() -> CertificateHelper.getCertificate(any()))
219                 .thenReturn(x509Certificate);
220 
221         long id = certificateService.uploadPartner(certificateUpload);
222 
223         ArgumentCaptor<CertificateEntity> savedCertEntityCaptor = ArgumentCaptor.forClass(CertificateEntity.class);
224 
225         assertEquals(FileStatus.SAVED, fileEntity.getStatus());
226 
227         verify(certificateRepository).save(savedCertEntityCaptor.capture());
228 
229         assertSame(fileEntity, savedCertEntityCaptor.getValue().getFile());
230         assertSame(CertificateType.PARTNER, savedCertEntityCaptor.getValue().getType());
231         assertSame(certificateContainerEntity, savedCertEntityCaptor.getValue().getCertificateContainer());
232         assertEquals("mycert", savedCertEntityCaptor.getValue().getName());
233         assertEquals(55, id);
234 
235         verify(certificateContainerRepository).findById(50L);
236         verify(fileRepository).getFileEntityById(15L);
237         verify(fileRepository).save(fileEntity);
238         verify(certificateInfoMapper).certificateToInfo(x509Certificate);
239         verify(certificateRepository)
240                 .existsByCertificateContainerAndInfo_sha1Fingerprint(certificateContainerEntity, "finger");
241         verifyNoMoreInteractions(
242                 fileRepository,
243                 certificateEntityFactory,
244                 certificateRepository,
245                 certificateContainerRepository,
246                 certificateInfoMapper);
247 
248         certificateHelperMockedStatic.verify(() -> CertificateHelper.getCertificate(
249                 fileEntity.getFileContent().getContent().getBinaryStream()));
250         certificateHelperMockedStatic.verifyNoMoreInteractions();
251     }
