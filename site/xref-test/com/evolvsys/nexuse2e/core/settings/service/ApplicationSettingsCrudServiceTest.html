28  class ApplicationSettingsCrudServiceTest {
29      private final ApplicationsSettingsMapper mapper = mock(ApplicationsSettingsMapper.class);
30      private final ApplicationSettingsRepository repository = mock(ApplicationSettingsRepository.class);
31      private final CurrentUserService currentUserService = mock(CurrentUserService.class);
32      private final NEXUSe2eConfigurationProperties configurationProperties = new NEXUSe2eConfigurationProperties();
33      private final ApplicationSettingsService applicationSettingsService = mock(ApplicationSettingsService.class);
34      private final ApplicationSettingsCrudService applicationSettingsCrudService = new ApplicationSettingsCrudService(
35              mapper, repository, currentUserService, configurationProperties, applicationSettingsService);
36  
37      private static List<Arguments> listTimeZones() {
38          return List.of(
39                  Arguments.of(null, ZoneId.getAvailableZoneIds()),
40                  Arguments.of(Set.of(), ZoneId.getAvailableZoneIds()),
41                  Arguments.of(Set.of("Europe/Berlin", "Etc/UTC"), Set.of("Europe/Berlin", "Etc/UTC")),
42                  Arguments.of(Set.of("Europe/Berlin", "Etc/UTC", "wrong"), Set.of("Europe/Berlin", "Etc/UTC")));
43      }
44  
45      @ParameterizedTest
46      @MethodSource
47      void listTimeZones(Set<String> config, Set<String> expected) {
48          configurationProperties.setTimeZones(config);
49  
50          Set<String> result = applicationSettingsCrudService.listTimeZones();
51  
52          assertThat(result).isEqualTo(expected);
53      }
54  
55      @Test
56      void getSystemSettings() {
57          ApplicationSettingsEntity applicationSettingsEntity = new ApplicationSettingsEntity();
58          ApplicationSettingsDto applicationSettingsDto = new ApplicationSettingsDto();
59          when(applicationSettingsService.getSystemSettings()).thenReturn(Optional.of(applicationSettingsEntity));
60          when(mapper.entityToDto(same(applicationSettingsEntity), any())).thenReturn(applicationSettingsDto);
61  
62          ApplicationSettingsDto result =
63                  applicationSettingsCrudService.getSystemSettings().orElseThrow();
64  
65          verify(applicationSettingsService).getSystemSettings();
66          verifyNoMoreInteractions(applicationSettingsService);
67          assertThat(result).isSameAs(applicationSettingsDto);
68      }
69  
70      @Test
71      void createEntity() {
72          ApplicationSettingsEntity entity = new ApplicationSettingsEntity();
73          TenantEntity tenantEntity = new TenantEntity();
74  
75          when(applicationSettingsService.getDefault()).thenReturn(entity);
76          when(currentUserService.getActiveTenant()).thenReturn(tenantEntity);
77  
78          ApplicationSettingsEntity result = applicationSettingsCrudService.createEntity();
79  
80          assertThat(result).isSameAs(entity);
81          assertThat(result.getTenant()).isSameAs(tenantEntity);
82      }
