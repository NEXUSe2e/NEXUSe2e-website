55      @ParameterizedTest
56      @ValueSource(booleans = {true, false})
57      void update_valid(boolean validNextActions) {
58          String operationName = validNextActions ? "withValidNextActions" : "default";
59          String path = validNextActions ? "action_updateWithValidNextActions" : "action_update";
60  
61          ActionDto actionDto = new ActionDto();
62          actionDto.setId(30L);
63          ActionDto expected = new ActionDto()
64                  .setActionId("my action")
65                  .setValidStartAction(true)
66                  .setValidTerminationAction(true)
67                  .setChoreographyId(40L);
68          expected.getWriteMetaData()
69                  .setInputProperties(Set.of("actionId", "validStartAction", "validTerminationAction", "choreographyId"));
70  
71          when(actionService.save((ActionDto) any())).thenReturn(actionDto);
72          when(actionService.save(any(), any())).thenReturn(actionDto);
73  
74          GraphQlTester.Response response = graphQlTester
75                  .documentName("action/action_update")
76                  .operationName(operationName)
77                  .execute();
78          response.errors().verify();
79  
80          response.path(path + ".id").entity(Long.class).isEqualTo(30L);
81          response.path(path + ".actionId").pathDoesNotExist();
82  
83          ArgumentCaptor<ActionDto> actionDtoArgumentCaptor = ArgumentCaptor.forClass(ActionDto.class);
84          if (validNextActions) {
85              ArgumentCaptor<List<ActionDto>> listCaptor = ArgumentCaptor.forClass(List.class);
86              verify(actionService).save(actionDtoArgumentCaptor.capture(), listCaptor.capture());
87  
88              assertThat(listCaptor.getValue())
89                      .isEqualTo(List.of(
90                              new ActionDto()
91                                      .setActionId("action 1")
92                                      .setValidStartAction(true)
93                                      .setValidTerminationAction(true)
94                                      .setChoreographyId(40L),
95                              new ActionDto()
96                                      .setActionId("action 2")
97                                      .setValidStartAction(true)
98                                      .setValidTerminationAction(true)
99                                      .setChoreographyId(40L)));
100         } else {
101             verify(actionService).save(actionDtoArgumentCaptor.capture());
102         }
103 
104         verifyNoMoreInteractions(actionService);
105         verifyNoInteractions(dataLoaderService);
106 
107         assertThat(actionDtoArgumentCaptor.getValue()).isEqualTo(expected);
108     }

241     @Test
242     void getById() {
243         ActionDto actionDto = new ActionDto().setActionId("my action");
244         List<ActionDto> validNextActions =
245                 List.of(new ActionDto().setActionId("action 1"), new ActionDto().setActionId("action 2"));
246         List<RouteDto> routes = List.of(new RouteDto().setRouteId("route 1"), new RouteDto().setRouteId("route 2"));
247         PipelineDto forwardPipeline = new PipelineDto().setPipelineId("forward");
248         PipelineDto returnPipeline = new PipelineDto().setPipelineId("return");
249         PipelineDto acknowledgmentPipeline = new PipelineDto().setPipelineId("acknowledgment");
250 
251         when(actionService.getById(50L)).thenReturn(Optional.of(actionDto));
252         when(dataLoaderService.load(any(), eq(ValidNextActionsBatchLoader.class), same(actionDto), any()))
253                 .thenReturn(CompletableFuture.completedFuture(new LinkedHashSet<>(validNextActions)));
254         when(dataLoaderService.loadRelation(any(), eq(ActionRouteBatchLoader.class), same(actionDto)))
255                 .thenReturn(CompletableFuture.completedFuture(new LinkedHashSet<>(routes)));
256         when(dataLoaderService.load(
257                         any(), eq(PipelineBatchLoader.class), same(actionDto), eq("forwardPipelineId"), any()))
258                 .thenReturn(CompletableFuture.completedFuture(forwardPipeline));
259         when(dataLoaderService.load(
260                         any(), eq(PipelineBatchLoader.class), same(actionDto), eq("returnPipelineId"), any()))
261                 .thenReturn(CompletableFuture.completedFuture(returnPipeline));
262         when(dataLoaderService.load(
263                         any(), eq(PipelineBatchLoader.class), same(actionDto), eq("acknowledgmentPipelineId"), any()))
264                 .thenReturn(CompletableFuture.completedFuture(acknowledgmentPipeline));
265 
266         GraphQlTester.Response response =
267                 graphQlTester.documentName("action/action_getById").execute();
268         response.errors().verify();
269 
270         response.path("action_getById.id").pathDoesNotExist();
271         response.path("action_getById.actionId").entity(String.class).isEqualTo("my action");
272         response.path("action_getById.validNextActions")
273                 .entityList(ActionDto.class)
274                 .isEqualTo(validNextActions);
275         response.path("action_getById.routes").entityList(RouteDto.class).isEqualTo(routes);
276         response.path("action_getById.forwardPipeline")
277                 .entity(PipelineDto.class)
278                 .isEqualTo(forwardPipeline);
279         response.path("action_getById.returnPipeline").entity(PipelineDto.class).isEqualTo(returnPipeline);
280         response.path("action_getById.acknowledgmentPipeline")
281                 .entity(PipelineDto.class)
282                 .isEqualTo(acknowledgmentPipeline);
283 
284         verify(actionService).getById(50L);
285         verify(dataLoaderService).load(any(), eq(ValidNextActionsBatchLoader.class), same(actionDto), any());
286         verify(dataLoaderService).loadRelation(any(), eq(ActionRouteBatchLoader.class), same(actionDto));
287         verify(dataLoaderService)
288                 .load(any(), eq(PipelineBatchLoader.class), same(actionDto), eq("forwardPipelineId"), any());
289         verify(dataLoaderService)
290                 .load(any(), eq(PipelineBatchLoader.class), same(actionDto), eq("returnPipelineId"), any());
291         verify(dataLoaderService)
292                 .load(any(), eq(PipelineBatchLoader.class), same(actionDto), eq("acknowledgmentPipelineId"), any());
293         verifyNoMoreInteractions(actionService, dataLoaderService);
294     }
