19  class IpUtilsTest {
20      @ParameterizedTest(name = "[{index}] Ip address {0} validation expected to be {2}")
21      @CsvSource({
22          "192.168.2.30,TRUE",
23          "192.168.0.30,FALSE",
24          "127.0.0.1,TRUE",
25          "::1,TRUE",
26          "2638:adb2:1f2c:ffdc::,TRUE",
27          "26a8:adb2:1f2c:318a::,FALSE",
28          "0:0:0:0:0:0:0:1,TRUE"
29      })
30      void isValidTest(String ipAddress, boolean expected) {
31          List<String> validIpv4AddressRanges = List.of("192.168.2.0/24", "127.0.0.1");
32          List<String> validIpv6AddressRanges = List.of("2638:adb2:1f2c::/48", "::1/128");
33  
34          assertEquals(expected, IpUtils.isValid(ipAddress, validIpv4AddressRanges, validIpv6AddressRanges));
35      }
36  
37      @Test
38      void isValidTest_emptyCollection() {
39          List<String> mergedIpAddressRanges = Collections.emptyList();
40          assertTrue(IpUtils.isValid("192.168.2.30", mergedIpAddressRanges));
41      }
42  
43      @Test
44      void isValidTest_emptyCollections() {
45          List<String> validIpv4AddressRanges = Collections.emptyList();
46          List<String> validIpv6AddressRanges = Collections.emptyList();
47  
48          assertTrue(IpUtils.isValid("192.168.2.30", validIpv4AddressRanges, validIpv6AddressRanges));
49      }
50  }
