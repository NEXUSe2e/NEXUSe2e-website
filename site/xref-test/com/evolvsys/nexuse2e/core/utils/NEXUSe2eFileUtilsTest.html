23  class NEXUSe2eFileUtilsTest {
24  
25      @Test
26      void getFiles_classpath_relative() {
27          assertThrows(IllegalArgumentException.class, () -> NEXUSe2eFileUtils.getFiles("classpath:fileutils"));
28      }
29  
30      @Test
31      void getFiles_classpath() {
32          Collection<File> files = NEXUSe2eFileUtils.getFiles("classpath:/fileutils");
33  
34          assertThat(
35                  files.stream()
36                          .map(file -> file.getAbsolutePath().replace("\\", "/"))
37                          .toList(),
38                  containsInAnyOrder(
39                          endsWith("fileutils/main_dir1.txt"),
40                          endsWith("fileutils/main_dir2.txt"),
41                          endsWith("fileutils/subdir/sub_dir1.txt"),
42                          endsWith("fileutils/subdir2/sub_dir1.txt"),
43                          endsWith("fileutils/subdir2/sub_dir2.txt")));
44      }
45  
46      @Test
47      void getFiles_filepath() {
48          URL resource = NEXUSe2eFileUtils.class.getResource("/fileutils");
49          File fileDir = FileUtils.toFile(resource);
50          Collection<File> files = NEXUSe2eFileUtils.getFiles("file:" + fileDir.getAbsolutePath());
51  
52          assertThat(
53                  files.stream()
54                          .map(file -> file.getAbsolutePath().replace("\\", "/"))
55                          .toList(),
56                  containsInAnyOrder(
57                          endsWith("fileutils/main_dir1.txt"),
58                          endsWith("fileutils/main_dir2.txt"),
59                          endsWith("fileutils/subdir/sub_dir1.txt"),
60                          endsWith("fileutils/subdir2/sub_dir1.txt"),
61                          endsWith("fileutils/subdir2/sub_dir2.txt")));
62      }
63  
64      @Test
65      void getFiles_default() {
66          URL resource = NEXUSe2eFileUtils.class.getResource("/fileutils");
67          File fileDir = FileUtils.toFile(resource);
68          Collection<File> files = NEXUSe2eFileUtils.getFiles(fileDir.getAbsolutePath());
69  
70          assertThat(
71                  files.stream()
72                          .map(file -> file.getAbsolutePath().replace("\\", "/"))
73                          .toList(),
74                  containsInAnyOrder(
75                          endsWith("fileutils/main_dir1.txt"),
76                          endsWith("fileutils/main_dir2.txt"),
77                          endsWith("fileutils/subdir/sub_dir1.txt"),
78                          endsWith("fileutils/subdir2/sub_dir1.txt"),
79                          endsWith("fileutils/subdir2/sub_dir2.txt")));
80      }
81  
82      @ParameterizedTest
83      @CsvSource({
84          "my_file.txt, L, my_file.txt",
85          "my file with %$ specials.other, j, my file with jj specials.other",
86          "test-file should/not&%4$.§\"work, _, test-file should_not__4_.__work"
87      })
88      void cleanFileName(String fileName, String replace, String expected) {
89          String result = NEXUSe2eFileUtils.cleanFileName(fileName, replace);
90          Assertions.assertThat(result).isEqualTo(expected);
91      }
