49      void mergeTest(
50              String oldName,
51              String oldEmail,
52              String newName,
53              String newEmail,
54              boolean expectedMerge,
55              String expectedMergedOldEmail,
56              String expectedMergedNewEmail) {
57          UserConfig oldConfig = createUserConfig(oldName, oldEmail, "oldConfig");
58          UserConfig newConfig = createUserConfig(newName, newEmail, "newConfig");
59  
60          List<UserConfig> configs = userConfigHandler.merge(List.of(oldConfig), List.of(newConfig));
61  
62          String expectedMergedOldConfigDescription;
63          if (expectedMerge) {
64              assertThat(configs).isNotNull().hasSize(1);
65              expectedMergedOldConfigDescription = "newConfig";
66          } else {
67              assertThat(configs).isNotNull().hasSize(2);
68              expectedMergedOldConfigDescription = "oldConfig";
69  
70              Optional<UserConfig> mergedNewConfig = configs.stream()
71                      .filter(config -> "newConfig".equals(config.getDescription()))
72                      .findFirst();
73              assertThat(mergedNewConfig.map(UserConfig::getEmail))
74                      .isEqualTo(Optional.ofNullable(expectedMergedNewEmail));
75          }
76  
77          Optional<UserConfig> mergedOldConfig = configs.stream()
78                  .filter(config -> expectedMergedOldConfigDescription.equals(config.getDescription()))
79                  .findFirst();
80          assertThat(mergedOldConfig.map(UserConfig::getEmail)).isEqualTo(Optional.ofNullable(expectedMergedOldEmail));
81  
82          verifyNoInteractions(customUserRepository, tenantRoleRepository, roleRepository, tenantRepository);
83      }

106     void sortTest(String name1, String email1, String name2, String email2, String expectedFirstDescription) {
107         UserConfig config1 = createUserConfig(name1, email1, "config1");
108         UserConfig config2 = createUserConfig(name2, email2, "config2");
109 
110         NexusConfig nexusConfig = new NexusConfig();
111         List<UserConfig> userConfigs = new ArrayList<>();
112         userConfigs.add(config1);
113         userConfigs.add(config2);
114         nexusConfig.setUsers(userConfigs);
115 
116         userConfigHandler.sort(nexusConfig);
117 
118         assertThat(nexusConfig.getUsers().size()).isEqualTo(2);
119         assertThat(nexusConfig.getUsers().getFirst().getDescription()).isEqualTo(expectedFirstDescription);
120 
121         verifyNoInteractions(customUserRepository, tenantRoleRepository, roleRepository, tenantRepository);
122     }
