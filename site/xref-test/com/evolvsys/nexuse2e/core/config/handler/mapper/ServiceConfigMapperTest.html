19  class ServiceConfigMapperTest {
20      private final PipelineConfigMapper pipelineConfigMapper = mock(PipelineConfigMapper.class);
21      private final ParameterConfigMapper parameterConfigMapper = mock(ParameterConfigMapper.class);
22      private final ServiceRepository serviceRepository = mock(ServiceRepository.class);
23      private final ServiceUtils serviceUtils = mock(ServiceUtils.class);
24      private final ServiceConfigMapper serviceConfigMapper =
25              new ServiceConfigMapperImpl(pipelineConfigMapper, parameterConfigMapper);
26  
27      @BeforeEach
28      void config() {
29          serviceConfigMapper.config(parameterConfigMapper, serviceRepository, serviceUtils);
30      }
31  
32      @Test
33      void nameToEntityTest() {
34          ServiceEntity serviceEntity = new ServiceEntity();
35          when(serviceRepository.findByServiceId("nameToEntity")).thenReturn(Optional.of(serviceEntity));
36  
37          assertSame(serviceEntity, serviceConfigMapper.idToEntity("nameToEntity"));
38  
39          verify(serviceRepository).findByServiceId("nameToEntity");
40          verifyNoMoreInteractions(serviceRepository);
41      }
42  
43      @Test
44      void nameToEntityNullTest() {
45          when(serviceRepository.findByServiceId("nameToEntity")).thenReturn(Optional.empty());
46          assertNull(serviceConfigMapper.idToEntity("nameToEntity"));
47          verify(serviceRepository).findByServiceId("nameToEntity");
48          verifyNoMoreInteractions(serviceRepository);
49      }
50  
51      @Test
52      void entityToNameTest() {
53          ServiceEntity serviceEntity = new ServiceEntity();
54          serviceEntity.setServiceId("entityToName");
55          assertEquals("entityToName", serviceConfigMapper.entityToId(serviceEntity));
56      }
57  
58      @Test
59      void entityToNameNullTest() {
60          assertNull(serviceConfigMapper.entityToId(null));
61      }
