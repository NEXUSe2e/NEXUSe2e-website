232         verify(pipelineConfigHandler).sort(same(export));

234         verifyTypeCall();
235 
236         List<ConfigExportContext> allValues = contextCaptor.getAllValues();

243     @Test
244     void merge() {
245         NexusConfig oldConfig = createConfig();
246         NexusConfig newConfig = new NexusConfig();
247         KeyValuePropertyConfig keyValuePropertyConfig = new KeyValuePropertyConfig();
248         ClientKeyContainerConfig clientKeyContainerConfig = new ClientKeyContainerConfig();
249         ServiceConfig serviceConfig = new ServiceConfig();
250         PartnerConfig partnerConfig = new PartnerConfig();
251         RouteConfig routeConfig = new RouteConfig();
252         CaCertConfig caCertConfig = new CaCertConfig();
253         UserConfig userConfig = new UserConfig();
254         ChoreographyConfig choreographyConfig = new ChoreographyConfig();
255         PipelineConfig pipelineConfig = new PipelineConfig();
256         List<KeyValuePropertyConfig> mergedKeyValueProperties = List.of(keyValuePropertyConfig);
257         List<ClientKeyContainerConfig> mergedClientKeys = List.of(clientKeyContainerConfig);
258         List<ServiceConfig> mergedServices = List.of(serviceConfig);
259         List<PartnerConfig> mergedPartners = List.of(partnerConfig);
260         List<RouteConfig> mergedRoutes = List.of(routeConfig);
261         List<CaCertConfig> mergedCaCertificates = List.of(caCertConfig);
262         List<UserConfig> mergedUsers = List.of(userConfig);
263         List<ChoreographyConfig> mergedChoreographies = List.of(choreographyConfig);
264         List<PipelineConfig> mergedPipelines = List.of(pipelineConfig);
265 
266         when(keyValuePropertyConfigHandler.merge(
267                         same(oldConfig.getKeyValueProperties()), same(newConfig.getKeyValueProperties())))
268                 .thenReturn(mergedKeyValueProperties);
269         when(clientKeyConfigHandler.merge(
270                         same(oldConfig.getClientKeyContainers()), same(newConfig.getClientKeyContainers())))
271                 .thenReturn(mergedClientKeys);
272         when(serviceConfigHandler.merge(same(oldConfig.getServices()), same(newConfig.getServices())))
273                 .thenReturn(mergedServices);
274         when(partnerConfigHandler.merge(same(oldConfig.getPartners()), same(newConfig.getPartners())))
275                 .thenReturn(mergedPartners);
276         when(routeConfigHandler.merge(same(oldConfig.getRoutes()), same(newConfig.getRoutes())))
277                 .thenReturn(mergedRoutes);
278         when(caCertificateConfigHandler.merge(same(oldConfig.getCaCertificates()), same(newConfig.getCaCertificates())))
279                 .thenReturn(mergedCaCertificates);
280         when(userConfigHandler.merge(same(oldConfig.getUsers()), same(newConfig.getUsers())))
281                 .thenReturn(mergedUsers);
282         when(choreographyConfigHandler.merge(same(oldConfig.getChoreographies()), same(newConfig.getChoreographies())))
283                 .thenReturn(mergedChoreographies);
284         when(pipelineConfigHandler.merge(same(oldConfig.getPipelines()), same(newConfig.getPipelines())))
285                 .thenReturn(mergedPipelines);
286 
287         NexusConfig merged = configHandlerService.merge(oldConfig, newConfig);
288 
289         verify(keyValuePropertyConfigHandler)
290                 .merge(same(oldConfig.getKeyValueProperties()), same(newConfig.getKeyValueProperties()));
291         verify(clientKeyConfigHandler)
292                 .merge(same(oldConfig.getClientKeyContainers()), same(newConfig.getClientKeyContainers()));
293         verify(serviceConfigHandler).merge(same(oldConfig.getServices()), same(newConfig.getServices()));
294         verify(partnerConfigHandler).merge(same(oldConfig.getPartners()), same(newConfig.getPartners()));
295         verify(routeConfigHandler).merge(same(oldConfig.getRoutes()), same(newConfig.getRoutes()));
296         verify(caCertificateConfigHandler)
297                 .merge(same(oldConfig.getCaCertificates()), same(newConfig.getCaCertificates()));
298         verify(userConfigHandler).merge(same(oldConfig.getUsers()), same(newConfig.getUsers()));
299         verify(choreographyConfigHandler)
300                 .merge(same(oldConfig.getChoreographies()), same(newConfig.getChoreographies()));
301         verify(pipelineConfigHandler).merge(same(oldConfig.getPipelines()), same(newConfig.getPipelines()));
302 
303         verifyTypeCall();
304 
305         assertThat(merged.getKeyValueProperties()).isEqualTo(mergedKeyValueProperties);
306         assertThat(merged.getClientKeyContainers()).isEqualTo(mergedClientKeys);
307         assertThat(merged.getServices()).isEqualTo(mergedServices);
308         assertThat(merged.getPartners()).isEqualTo(mergedPartners);
309         assertThat(merged.getRoutes()).isEqualTo(mergedRoutes);
310         assertThat(merged.getCaCertificates()).isEqualTo(mergedCaCertificates);
311         assertThat(merged.getUsers()).isEqualTo(mergedUsers);
312         assertThat(merged.getChoreographies()).isEqualTo(mergedChoreographies);
313         assertThat(merged.getPipelines()).isEqualTo(mergedPipelines);
314     }
