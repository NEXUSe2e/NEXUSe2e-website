13  /**
14   * Test for {@link ConfigImportEventHandler}
15   *
16   * @author SvenBarden
17   * @since 09.09.2024
18   */
19  class ConfigImportEventHandlerTest {
20      private final PurgeService cleanupService = mock(PurgeService.class);
21      private final ConfigImportManualActionService configImportManualActionService =
22              mock(ConfigImportManualActionService.class);
23      private final KeyValueService keyValueService = mock(KeyValueService.class);
24      private final ConfigImportEventHandler configImportEventHandler =
25              new ConfigImportEventHandler(cleanupService, configImportManualActionService, keyValueService);
26  
27      @ParameterizedTest
28      @ValueSource(booleans = {true, false})
29      void preImport(boolean overwrite) {
30          configImportEventHandler.preImport(null, overwrite);
31  
32          if (overwrite) {
33              verify(cleanupService).purgeAllConversations();
34          }
35          verifyNoMoreInteractions(cleanupService, configImportManualActionService, keyValueService);
36      }
37  
38      @ParameterizedTest
39      @ValueSource(booleans = {true, false})
40      void postImport(boolean overwrite) throws NexusException {
41          NexusConfig oldConfig = new NexusConfig();
42          NexusConfig newConfig = new NexusConfig();
43  
44          configImportEventHandler.postImport(oldConfig, newConfig, overwrite);
45  
46          if (overwrite) {
47              verify(configImportManualActionService).overwrite(oldConfig, newConfig);
48          } else {
49              verify(configImportManualActionService).merge(oldConfig, newConfig);
50          }
51          verify(keyValueService).update();
52          verifyNoMoreInteractions(configImportManualActionService, keyValueService, cleanupService);
53      }
54  }
