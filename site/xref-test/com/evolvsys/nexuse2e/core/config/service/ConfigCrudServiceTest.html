29  class ConfigCrudServiceTest {
30      private final ConfigRepository configRepository = mock(ConfigRepository.class);
31      private final ConfigParserService configParserService = mock(ConfigParserService.class);
32      private final LockService lockService = mock(LockService.class);
33      private final ConfigCrudService configCrudService =
34              new ConfigCrudService(configRepository, configParserService, lockService);
35  
36      @ParameterizedTest
37      @ValueSource(longs = {15, 20})
38      void count(long value) {
39          when(configRepository.count()).thenReturn(value);
40  
41          long count = configCrudService.count();
42  
43          verify(configRepository).count();
44          verifyNoMoreInteractions(configRepository, configParserService, lockService);
45  
46          assertThat(count).isEqualTo(value);
47      }
48  
49      @ParameterizedTest
50      @ValueSource(strings = {"bar", "foo"})
51      void get(String s) throws JsonProcessingException {
52          NexusConfig expected = new NexusConfig();
53          when(configParserService.parse(s)).thenReturn(expected);
54  
55          NexusConfig nexusConfig = configCrudService.get(s);
56  
57          verify(configParserService).parse(s);
58          verifyNoMoreInteractions(configRepository, configParserService, lockService);
59          assertThat(nexusConfig).isSameAs(expected);
60      }
61  
62      @Test
63      void save() throws JsonProcessingException {
64          AtomicInteger saveCounter = new AtomicInteger(0);
65  
66          NexusConfig nexusConfig = new NexusConfig();
67  
68          ReentrantLock lock = new ReentrantLock();
69          when(lockService.obtain("nexuse2e-config-update")).thenReturn(lock);
70          when(configParserService.configToString(nexusConfig)).thenAnswer(invocation -> "content-" + saveCounter.get());
71  
72          doAnswer(new AnswersWithDelay(Duration.ofSeconds(2).toMillis(), invocation -> null))
73                  .when(configParserService)
74                  .update(nexusConfig);
75          doAnswer(invocation -> {
76                      saveCounter.incrementAndGet();
77                      return invocation.getArgument(0);
78                  })
79                  .when(configRepository)
80                  .save(any(ConfigEntity.class));
81  
82          CompletableFuture<Void> call1 = CompletableFuture.runAsync(() -> {
83              try {
84                  configCrudService.save(nexusConfig);
85              } catch (NexusException e) {
86                  throw new RuntimeException(e);
87              }
88          });
89          CompletableFuture<Void> call2 = CompletableFuture.runAsync(() -> {
90              try {
91                  configCrudService.save(nexusConfig);
92              } catch (NexusException e) {
93                  throw new RuntimeException(e);
94              }
95          });
96  
97          await().atLeast(Duration.ofSeconds(3))
98                  .atMost(Duration.ofSeconds(5))
99                  .until(() -> call1.isDone() && call2.isDone());
100 
101         ArgumentCaptor<ConfigEntity> configEntityArgumentCaptor = ArgumentCaptor.forClass(ConfigEntity.class);
102 
103         verify(lockService, times(2)).obtain("nexuse2e-config-update");
104         verify(configParserService, times(2)).update(nexusConfig);
105         verify(configParserService, times(2)).configToString(nexusConfig);
106         verify(configRepository, times(2)).save(configEntityArgumentCaptor.capture());
107 
108         List<ConfigEntity> allValues = configEntityArgumentCaptor.getAllValues();
109         for (int i = 0; i < allValues.size(); i++) {
110             ConfigEntity value = allValues.get(i);
111             assertThat(value.getContent()).isEqualTo("content-" + i);
112             verify(configRepository).save(value);
113         }
114 
115         verifyNoMoreInteractions(configRepository, configParserService, lockService);
116 
117         assertThat(call1.isDone()).isTrue();
118         assertThat(call2.isDone()).isTrue();
119     }
