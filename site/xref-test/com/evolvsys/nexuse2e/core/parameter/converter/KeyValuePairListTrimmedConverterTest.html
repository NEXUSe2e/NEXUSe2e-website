217     private static final class StringToListConverter implements ArgumentConverter {
218 
219         @Override
220         public Object convert(Object source, ParameterContext context) {
221             Object result;
222             if (source == null) {
223                 result = null;
224             } else if (source instanceof String string) {
225                 result = Arrays.stream(string.split("\\s*,\\s*")).toList();
226             } else {
227                 throw new ArgumentConversionException("Only strings can be converted to lists");
228             }
229             return result;
230         }
231     }
