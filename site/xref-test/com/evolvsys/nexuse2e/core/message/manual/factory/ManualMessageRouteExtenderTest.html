17  /**
18   * Test for {@link ManualMessageRouteExtender}
19   *
20   * @author SvenBarden
21   * @since 25.07.2024
22   */
23  @SuppressWarnings("PMD.TestClassWithoutTestCases") // pmd does not know carthesian tests
24  class ManualMessageRouteExtenderTest {
25      private final ContextRouteFactory contextRouteFactory = mock(ContextRouteFactory.class);
26      private final ManualMessageRouteExtender manualMessageRouteExtender =
27              new ManualMessageRouteExtender(contextRouteFactory);
28  
29      @CartesianTest
30      void extend(
31              @CartesianTest.Values(longs = {50, 70}) long choreographyId,
32              @CartesianTest.Values(longs = {10, 20}) long actionId,
33              @CartesianTest.Values(strings = {"source 1", "source 2"}) String sourcePartnerId,
34              @CartesianTest.Values(strings = {"target 1", "target 2"}) String targetPartnerId,
35              @CartesianTest.Values(booleans = {true, false}) boolean success)
36              throws NexusException {
37          TestMessageContext testMessageContext = new TestMessageContext();
38          testMessageContext.setSourcePartnerId(sourcePartnerId);
39          testMessageContext.setTargetPartnerId(targetPartnerId);
40          ManualMessageDto manualMessageDto = new ManualMessageDto();
41          manualMessageDto.setChoreographyId(choreographyId);
42          manualMessageDto.setActionId(actionId);
43  
44          if (success) {
45              ContextRoute route = ContextRoute.builder().build();
46              when(contextRouteFactory.getRoute(new RouteKey(choreographyId, actionId, sourcePartnerId, targetPartnerId)))
47                      .thenReturn(Optional.of(route));
48  
49              manualMessageRouteExtender.extend(testMessageContext, manualMessageDto);
50  
51              assertThat(testMessageContext.getRoute()).isSameAs(route);
52          } else {
53              assertThatThrownBy(() -> manualMessageRouteExtender.extend(testMessageContext, manualMessageDto))
54                      .isInstanceOf(NexusRuntimeException.class)
55                      .hasMessage("no route found");
56          }
57  
58          verify(contextRouteFactory).getRoute(new RouteKey(choreographyId, actionId, sourcePartnerId, targetPartnerId));
59          verifyNoMoreInteractions(contextRouteFactory);
60      }
61  }
