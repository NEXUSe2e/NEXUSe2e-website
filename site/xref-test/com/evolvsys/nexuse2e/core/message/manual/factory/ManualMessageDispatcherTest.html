22  /**
23   * Test for {@link ManualMessageDispatcher}
24   *
25   * @author SvenBarden
26   * @since 25.07.2024
27   */
28  class ManualMessageDispatcherTest {
29      private final OutboundMessageDispatcher outboundMessageDispatcher = mock(OutboundMessageDispatcher.class);
30      private final MessageEventService messageEventService = mock(MessageEventService.class);
31      private final MessageResolverValidator messageResolverValidator = mock(MessageResolverValidator.class);
32      private final ContextFactory contextFactory = mock(ContextFactory.class);
33      private final MDCService mdcService = mock(MDCService.class);
34      private final MDCIdGenerator mdcIdGenerator = mock(MDCIdGenerator.class);
35      private final ManualMessageDispatcher manualMessageDispatcher = new ManualMessageDispatcher(
36              outboundMessageDispatcher,
37              messageEventService,
38              messageResolverValidator,
39              contextFactory,
40              mdcService,
41              mdcIdGenerator);
42  
43      @Test
44      void send_success() throws NexusException {
45          TestMessageContext testMessageContext = new TestMessageContext();
46          testMessageContext.setSourceMessage(new Message());
47  
48          String oldId = "old id";
49          when(mdcService.getId()).thenReturn(oldId);
50          when(mdcIdGenerator.newId()).thenReturn("new id");
51  
52          ManualMessageResponseDto result = manualMessageDispatcher.send(testMessageContext);
53  
54          ArgumentCaptor<ExecutionContext> contextArgumentCaptor = ArgumentCaptor.forClass(ExecutionContext.class);
55          verify(outboundMessageDispatcher).dispatch(contextArgumentCaptor.capture());
56          verify(mdcIdGenerator).newId();
57          verify(mdcService).getId();
58          verify(mdcService).setId("new id");
59          verify(mdcService).setId(oldId);
60          assertThat(contextArgumentCaptor.getValue().getMessageType()).isSameAs(MessageType.NORMAL);
61          assertThat(result)
62                  .isEqualTo(ManualMessageResponseDto.builder()
63                          .status(ManualMessageStatus.SUCCESS)
64                          .message("success")
65                          .build());
66          verifyNoMoreInteractions(outboundMessageDispatcher, mdcService, mdcIdGenerator);
67      }
68  
69      @Test
70      void send_fail() throws NexusException {
71          TestMessageContext testMessageContext = new TestMessageContext();
72          testMessageContext.setSourceMessage(new Message());
73  
74          String oldId = "old id";
75          when(mdcService.getId()).thenReturn(oldId);
76          when(mdcIdGenerator.newId()).thenReturn("new id");
77  
78          doThrow(new IllegalArgumentException("test error"))
79                  .when(outboundMessageDispatcher)
80                  .dispatch(any());
81  
82          ManualMessageResponseDto result = manualMessageDispatcher.send(testMessageContext);
83  
84          assertThat(result)
85                  .isEqualTo(ManualMessageResponseDto.builder()
86                          .status(ManualMessageStatus.ERROR)
87                          .message("test error")
88                          .messageAggregated("{type: 'IllegalArgumentException', message: 'test error'}")
89                          .build());
90          verify(mdcIdGenerator).newId();
91          verify(mdcService).getId();
92          verify(mdcService).setId("new id");
93          verify(mdcService).setId(oldId);
94          verify(outboundMessageDispatcher).dispatch(any());
95          verifyNoMoreInteractions(outboundMessageDispatcher, mdcService, mdcIdGenerator);
96      }
97  }
