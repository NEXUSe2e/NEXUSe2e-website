25  @SuppressWarnings("PMD.TestClassWithoutTestCases") // pmd does not know carthesian tests
26  class ManualMessagePartnerExtenderTest {
27      private final PartnerRepository partnerRepository = mock(PartnerRepository.class);
28      private final ManualMessagePartnerExtender manualMessagePartnerExtender =
29              new ManualMessagePartnerExtender(partnerRepository);
30  
31      private static ArgumentSets extend() {
32          return ArgumentSets.argumentsForFirstParameter(null, 10L, 20L).argumentsForNextParameter(null, 40L, 50L);
33      }
34  
35      private void whenPartnerNotFound(Long id) {
36          if (id == null) {
37              when(partnerRepository.getDefaultLocalIdentityName(PartnerType.INTERN))
38                      .thenReturn(Optional.empty());
39          } else {
40              when(partnerRepository.findById(id)).thenReturn(Optional.empty());
41          }
42      }
43  
44      private void whenPartnerFound(Long id, String partnerId) {
45          if (id == null) {
46              when(partnerRepository.getDefaultLocalIdentityName(PartnerType.INTERN))
47                      .thenReturn(Optional.of("default"));
48          } else {
49              PartnerEntity partnerEntity = new PartnerEntity();
50              partnerEntity.setPartnerId(partnerId);
51              when(partnerRepository.findById(id)).thenReturn(Optional.of(partnerEntity));
52          }
53      }
54  
55      @CartesianTest
56      @CartesianTest.MethodFactory("extend")
57      void extend_sourceNotFound(Long sourcePartnerId, Long targetPartnerId) {
58          TestMessageContext testMessageContext = new TestMessageContext();
59          ManualMessageDto manualMessageDto = new ManualMessageDto();
60          manualMessageDto.setSourcePartnerId(sourcePartnerId);
61          manualMessageDto.setTargetPartnerId(targetPartnerId);
62  
63          if (sourcePartnerId == null) {
64              assertThatThrownBy(() -> manualMessagePartnerExtender.extend(testMessageContext, manualMessageDto))
65                      .isInstanceOf(NexusRuntimeException.class)
66                      .hasMessage("no default local identity found");
67              verify(partnerRepository).getDefaultLocalIdentityName(PartnerType.INTERN);
68          } else {
69              assertThatThrownBy(() -> manualMessagePartnerExtender.extend(testMessageContext, manualMessageDto))
70                      .isInstanceOf(NexusRuntimeException.class)
71                      .hasMessage("partner not found");
72              verify(partnerRepository).findById(sourcePartnerId);
73          }
74          verifyNoMoreInteractions(partnerRepository);
75      }
76  
77      @CartesianTest
78      @CartesianTest.MethodFactory("extend")
79      void extend_targetNotFound(Long sourcePartnerId, Long targetPartnerId) {
80          TestMessageContext testMessageContext = new TestMessageContext();
81          ManualMessageDto manualMessageDto = new ManualMessageDto();
82          manualMessageDto.setSourcePartnerId(sourcePartnerId);
83          manualMessageDto.setTargetPartnerId(targetPartnerId);
84  
85          if (sourcePartnerId == null) {
86              if (targetPartnerId == null) {
87                  when(partnerRepository.getDefaultLocalIdentityName(PartnerType.INTERN))
88                          .thenReturn(Optional.of("source partner"))
89                          .thenReturn(Optional.empty());
90              } else {
91                  whenPartnerFound(null, "source partner");
92                  whenPartnerNotFound(targetPartnerId);
93              }
94          } else {
95              whenPartnerFound(sourcePartnerId, "source partner");
96              whenPartnerNotFound(targetPartnerId);
97          }
98  
99          if (targetPartnerId == null) {
100             assertThatThrownBy(() -> manualMessagePartnerExtender.extend(testMessageContext, manualMessageDto))
101                     .isInstanceOf(NexusRuntimeException.class)
102                     .hasMessage("no default local identity found");
103 
104             if (sourcePartnerId == null) {
105                 verify(partnerRepository, times(2)).getDefaultLocalIdentityName(PartnerType.INTERN);
106             } else {
107                 verify(partnerRepository).findById(sourcePartnerId);
108                 verify(partnerRepository).getDefaultLocalIdentityName(PartnerType.INTERN);
109             }
110         } else {
111             assertThatThrownBy(() -> manualMessagePartnerExtender.extend(testMessageContext, manualMessageDto))
112                     .isInstanceOf(NexusRuntimeException.class)
113                     .hasMessage("partner not found");
114 
115             if (sourcePartnerId == null) {
116                 verify(partnerRepository).getDefaultLocalIdentityName(PartnerType.INTERN);
117             } else {
118                 verify(partnerRepository).findById(sourcePartnerId);
119             }
120             verify(partnerRepository).findById(targetPartnerId);
121         }
122         verifyNoMoreInteractions(partnerRepository);
123     }
124 
125     @CartesianTest
126     @CartesianTest.MethodFactory("extend")
127     void extend_success(Long sourcePartnerId, Long targetPartnerId) throws NexusException {
128         TestMessageContext testMessageContext = new TestMessageContext();
129         ManualMessageDto manualMessageDto = new ManualMessageDto();
130         manualMessageDto.setSourcePartnerId(sourcePartnerId);
131         manualMessageDto.setTargetPartnerId(targetPartnerId);
132 
133         whenPartnerFound(sourcePartnerId, "source partner");
134         whenPartnerFound(targetPartnerId, "target partner");
135 
136         manualMessagePartnerExtender.extend(testMessageContext, manualMessageDto);
137 
138         if (ObjectUtils.allNull(sourcePartnerId, targetPartnerId)) {
139             verify(partnerRepository, times(2)).getDefaultLocalIdentityName(PartnerType.INTERN);
140         } else if (ObjectUtils.allNotNull(sourcePartnerId, targetPartnerId) == false) {
141             verify(partnerRepository).getDefaultLocalIdentityName(PartnerType.INTERN);
142         }
143 
144         if (sourcePartnerId == null) {
145             assertThat(testMessageContext.getSourcePartnerId()).isEqualTo("default");
146         } else {
147             verify(partnerRepository).findById(sourcePartnerId);
148             assertThat(testMessageContext.getSourcePartnerId()).isEqualTo("source partner");
149         }
150 
151         if (targetPartnerId == null) {
152             assertThat(testMessageContext.getTargetPartnerId()).isEqualTo("default");
153         } else {
154             verify(partnerRepository).findById(targetPartnerId);
155             assertThat(testMessageContext.getTargetPartnerId()).isEqualTo("target partner");
156         }
157 
158         verifyNoMoreInteractions(partnerRepository);
159     }
