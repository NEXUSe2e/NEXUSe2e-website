32  class ManualMessageServiceTest {
33      private final NEXUSe2eConfigurationProperties nexuSe2eConfigurationProperties =

39              nexuSe2eConfigurationProperties, currentUserService,
40              manualMessageFactory, manualMessageDispatcher);

42      @Test

44          when(nexuSe2eConfigurationProperties.getDefaultManualMessageFiles()).thenReturn("classpath:fileutils");

46          List<String> result = manualMessageService.getDefaultFiles();

48          verify(nexuSe2eConfigurationProperties).getDefaultManualMessageFiles();
49          verifyNoMoreInteractions(nexuSe2eConfigurationProperties);
50          verifyNoInteractions(currentUserService, manualMessageFactory, manualMessageDispatcher);

52          assertThat(result).containsExactly("main_dir1.txt", "main_dir2.txt");
53      }

78      @Test
79      void send_noPermission() {
80          doThrow(new GeneralErrorException("no permission to send manual messages"))
81                  .when(currentUserService)
82                  .checkSystemPermissions("NEXUS.SEND_MANUAL_MESSAGE");
83  
84          assertThatThrownBy(() -> manualMessageService.send(new ManualMessageDto()))
85                  .isInstanceOf(GeneralErrorException.class)
86                  .hasMessage("no permission to send manual messages");
87  
88          verify(currentUserService).checkSystemPermissions("NEXUS.SEND_MANUAL_MESSAGE");
89          verifyNoMoreInteractions(currentUserService);
90          verifyNoInteractions(nexuSe2eConfigurationProperties, manualMessageFactory, manualMessageDispatcher);
91      }

93      @Test
94      void send() throws SQLException, IOException, NexusException {
95          ManualMessageDto manualMessageDto = new ManualMessageDto();
96          manualMessageDto.setRepeat(4);
97  
98          TestMessageContext testMessageContext1 = new TestMessageContext();
99          TestMessageContext testMessageContext3 = new TestMessageContext();
100         TestMessageContext testMessageContext4 = new TestMessageContext();
101 
102         ManualMessageResponseDto manualMessageResponseDto1 =
103                 ManualMessageResponseDto.builder().build();
104         ManualMessageResponseDto manualMessageResponseDto2 =
105                 ManualMessageResponseDto.builder().build();
106         ManualMessageResponseDto manualMessageResponseDto4 =
107                 ManualMessageResponseDto.builder().build();
108 
109         when(manualMessageFactory.create(manualMessageDto))
110                 .thenReturn(testMessageContext1)
111                 .thenThrow(new NexusRuntimeException(
112                         NexusError.builder().message("Create error").build(),
113                         new IllegalArgumentException("Root error")))
114                 .thenReturn(testMessageContext3, testMessageContext4);
115         when(manualMessageDispatcher.send(testMessageContext1)).thenReturn(manualMessageResponseDto1);
116         when(manualMessageDispatcher.send(testMessageContext3)).thenThrow(new IllegalStateException("Sending error"));
117         when(manualMessageDispatcher.send(testMessageContext4)).thenReturn(manualMessageResponseDto4);
118 
119         ManualMessageResponseListDto result = manualMessageService.send(manualMessageDto);
120 
121         verify(currentUserService).checkSystemPermissions("NEXUS.SEND_MANUAL_MESSAGE");
122         verify(manualMessageFactory, times(4)).create(manualMessageDto);
123         verify(manualMessageDispatcher).send(testMessageContext1);
124         verify(manualMessageDispatcher).send(testMessageContext3);
125         verify(manualMessageDispatcher).send(testMessageContext4);
126         verifyNoMoreInteractions(currentUserService, manualMessageFactory, manualMessageDispatcher);
127 
128         assertThat(result.getStatus()).isSameAs(ManualMessageStatus.FAILURE);
129         assertThat(result.getMessage()).isBlank();
130         assertThat(result.getMessageResponses()).hasSize(4);
131         assertThat(result.getMessageResponses().getFirst()).isSameAs(manualMessageResponseDto1);
132 
133         assertThat(result.getMessageResponses().getLast()).isSameAs(manualMessageResponseDto4);
134         assertThat(result.getMessageResponses().get(1)).isNotSameAs(manualMessageResponseDto2);
135         assertThat(result.getMessageResponses().get(1))
136                 .isEqualTo(ManualMessageResponseDto.builder()
137                         .status(ManualMessageStatus.ERROR)
138                         .message("Create error")
139                         .messageAggregated(
140                                 "{type: 'NexusRuntimeException', message: 'Create error'}, {type: 'IllegalArgumentException', message: 'Root error'}")
141                         .build());
142         assertThat(result.getMessageResponses().get(2))
143                 .isEqualTo(ManualMessageResponseDto.builder()
144                         .status(ManualMessageStatus.ERROR)
145                         .message("Sending error")
146                         .messageAggregated("{type: 'IllegalStateException', message: 'Sending error'}")
147                         .build());
148     }
