13  /**
14   * <p>Test for the {@link MessageDataService}.</p>
15   *
16   * @author Florian Drees
17   * @since 24.09.2024
18   */
19  class MessageDataServiceTest {
20      private final MessageRepository messageRepository = mock(MessageRepository.class);
21      private final ApplicationSettingsService applicationSettingsService = mock(ApplicationSettingsService.class);
22      private final MessageDataService messageDataService =
23              new MessageDataService(messageRepository, applicationSettingsService);
24  
25      @ParameterizedTest
26      @ValueSource(booleans = {true, false})
27      void exists(boolean exists) {
28          MessageFilter messageFilter = MessageFilter.builder().build();
29  
30          ApplicationSettingsEntity applicationSettingsEntity = new ApplicationSettingsEntity();
31          applicationSettingsEntity.setTimeZone("UTC");
32  
33          when(applicationSettingsService.getSystemSettingsOrDefault()).thenReturn(applicationSettingsEntity);
34          when(messageRepository.exists(messageFilter, "UTC")).thenReturn(exists);
35  
36          assertThat(messageDataService.exists(messageFilter))
37                  .describedAs("result is equal to the return value of the repository method")
38                  .isEqualTo(exists);
39  
40          verify(applicationSettingsService).getSystemSettingsOrDefault();
41          verify(messageRepository).exists(messageFilter, "UTC");
42          verifyNoMoreInteractions(applicationSettingsService, messageRepository);
43      }
44  }
