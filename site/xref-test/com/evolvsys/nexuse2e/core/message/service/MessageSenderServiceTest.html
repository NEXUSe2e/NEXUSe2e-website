23  class MessageSenderServiceTest {
24      private final ContextFactory contextFactory = mock(ContextFactory.class);
25      private final OutboundMessageDispatcher outboundMessageDispatcher = mock(OutboundMessageDispatcher.class);
26      private final MessageSenderService messageSenderService =
27              new MessageSenderService(contextFactory, outboundMessageDispatcher);
28  
29      private static List<MessageContext> resend_normal() {
30          return List.of(
31                  new TestMessageContext().setSourceMessage(new Message()),
32                  new TestMessageContext()
33                          .setSourceMessage(new Message())
34                          .setAcknowledgmentMessage(Message.builder().id("ack").build()));
35      }
36  
37      @ParameterizedTest
38      @MethodSource
39      void resend_normal(MessageContext messageContext) throws NexusException {
40          when(contextFactory.createForResend("msg1")).thenReturn(messageContext);
41  
42          messageSenderService.resend("msg1");
43  
44          verify(outboundMessageDispatcher).dispatch(messageContext.createExecutionContext(MessageType.NORMAL));
45      }
46  
47      @Test
48      void resend_ack() throws NexusException {
49          TestMessageContext messageContext = new TestMessageContext();
50          messageContext
51                  .setSourceMessage(new Message())
52                  .setAcknowledgmentMessage(Message.builder().id("msg1").build());
53  
54          when(contextFactory.createForResend("msg1")).thenReturn(messageContext);
55  
56          messageSenderService.resend("msg1");
57  
58          verify(outboundMessageDispatcher).dispatch(messageContext.createExecutionContext(MessageType.ACKNOWLEDGMENT));
59      }
60  }
