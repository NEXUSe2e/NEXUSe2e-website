24  class FileSystemContentTest {

26      @Test
27      void createFile() throws IOException {
28          Path path = mock(Path.class);
29          File expected = mock(File.class);
30  
31          when(path.toFile()).thenReturn(expected);
32  
33          try (MockedStatic<Files> filesMock = mockStatic(Files.class)) {
34              filesMock
35                      .when(() -> Files.createTempFile(TEMP_FILE_PREFIX, TEMP_FILE_SUFFIX))
36                      .thenReturn(path);
37  
38              File file = FileSystemContent.createFile();
39  
40              assertSame(expected, file);
41              filesMock.verify(() -> Files.createTempFile(TEMP_FILE_PREFIX, TEMP_FILE_SUFFIX));
42              filesMock.verifyNoMoreInteractions();
43          }
44      }
45  
46      @Test
47      void of_file() throws IOException {
48          File file = FileUtils.toFile(getClass().getResource("/fileutils/main_dir1.txt"));
49  
50          assertNotNull(file);
51  
52          FileSystemContent fileSystemContent = FileSystemContent.of(file);
53  
54          try (InputStream is1 = fileSystemContent.getInputStream();
55                  InputStream is2 = fileSystemContent.getInputStream()) {
56              assertNotSame(is1, is2);
57              assertEquals(file.length(), fileSystemContent.getSize());
58              assertArrayEquals(Files.readAllBytes(file.toPath()), fileSystemContent.getByteArray());
59              assertArrayEquals(Files.readAllBytes(file.toPath()), is1.readAllBytes());
60          }
61      }
62  
63      @Test
64      void of_byteArray() throws IOException {
65          byte[] content = {5, 3, 5};
66  
67          FileSystemContent fileSystemContent = FileSystemContent.of(content);
68  
69          try (InputStream is1 = fileSystemContent.getInputStream();
70                  InputStream is2 = fileSystemContent.getInputStream()) {
71              assertNotSame(is1, is2);
72              assertEquals(content.length, fileSystemContent.getSize());
73              assertArrayEquals(content, fileSystemContent.getByteArray());
74              assertArrayEquals(content, is1.readAllBytes());
75          }
76      }
