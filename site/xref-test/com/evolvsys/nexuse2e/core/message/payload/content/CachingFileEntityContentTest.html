22  class CachingFileEntityContentTest {
23      private final BinaryRepository binaryRepository = mock(BinaryRepository.class);
24  
25      private CachingFileEntityContent createEntity(int size, Long id) {
26          PayloadEntity payloadEntity = new PayloadEntity();
27          payloadEntity.setSize(size);
28          payloadEntity.setContent(new BinaryEntity());
29          payloadEntity.getContent().setId(id);
30  
31          return CachingFileEntityContent.of(payloadEntity, binaryRepository);
32      }
33  
34      private void whenBinaryRepo(Long id, InputStream inputStream, InputStream... additionalInputStreams)
35              throws SQLException {
36          Blob blob = mock(Blob.class);
37          BinaryEntity binaryEntity = new BinaryEntity();
38          binaryEntity.setContent(blob);
39  
40          when(binaryRepository.findById(id)).thenReturn(Optional.of(binaryEntity));
41          when(blob.getBinaryStream()).thenReturn(inputStream, additionalInputStreams);
42      }
43  
44      @Test
45      void getByteArray() throws SQLException {
46          CachingFileEntityContent content = createEntity(10, 15L);
47          InputStream inputStream1 = new NullInputStream();
48          InputStream inputStream2 = new NullInputStream();
49  
50          whenBinaryRepo(15L, inputStream1, inputStream2);
51  
52          byte[] byteArray1 = content.getByteArray();
53          byte[] byteArray2 = content.getByteArray();
54  
55          assertSame(byteArray1, byteArray2);
56          assertInstanceOf(PayloadEntityContent.class, content);
57          assertEquals(10, content.getSize());
58          verify(binaryRepository).findById(15L);
59          verifyNoMoreInteractions(binaryRepository);
60      }
61  
62      @Test
63      void getInputStream() throws SQLException {
64          CachingFileEntityContent content = createEntity(10, 15L);
65          InputStream inputStream1 = new NullInputStream();
66          InputStream inputStream2 = new NullInputStream();
67  
68          whenBinaryRepo(15L, inputStream1, inputStream2);
69  
70          InputStream result1 = content.getInputStream();
71          InputStream result2 = content.getInputStream();
72  
73          assertNotSame(result1, result2);
74          assertInstanceOf(PayloadEntityContent.class, content);
75          assertEquals(10, content.getSize());
76          verify(binaryRepository).findById(15L);
77          verifyNoMoreInteractions(binaryRepository);
78      }
79  }
