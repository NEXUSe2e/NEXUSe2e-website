53      @Test
54      void entityToDtoTest() {
55          MessageEntity messageEntity = createMessageEntity();
56  
57          InterceptingMappingContext interceptingMappingContext =
58                  InterceptingMappingContext.builder().skipIntercepting(true).build();
59  
60          MessageDto messageDto = messageMapper.entityToDto(messageEntity, interceptingMappingContext);
61  
62          assertEquals(100L, messageDto.getId());
63          assertEquals("31dbc679-2e74-4fe6-9485-ba3e14e9ad95", messageDto.getMessageId());
64          assertEquals(MessageStatus.RESOLVED, messageDto.getStatus());
65          assertEquals(20L, messageDto.getActionId());
66          assertEquals(MessageType.NORMAL, messageDto.getType());
67          assertEquals(3, messageDto.getRetries());
68          assertEquals(ZonedDateTime.of(2023, 12, 27, 12, 35, 0, 0, ZoneId.of("UTC")), messageDto.getSent());
69          assertNull(messageDto.getConversationId());
70      }
71  
72      private static MessageEntity createMessageEntity() {
73          ActionEntity actionEntity = new ActionEntity();
74          actionEntity.setId(20L);
75          MessageEntity messageEntity = new MessageEntity();
76          messageEntity.setId(100L);
77          messageEntity.setMessageId("31dbc679-2e74-4fe6-9485-ba3e14e9ad95");
78          messageEntity.setStatus(MessageStatus.RESOLVED);
79          messageEntity.setAction(actionEntity);
80          messageEntity.setType(MessageType.NORMAL);
81          messageEntity.setRetries(3);
82          messageEntity.setSent(ZonedDateTime.of(2023, 12, 27, 12, 35, 0, 0, ZoneId.of("UTC")));
83          return messageEntity;
84      }

111     void updateEntityTest() {
112         ActionEntity actionEntity = new ActionEntity();
113         MessageDto messageDto = new MessageDto();
114         messageDto.setId(100L);
115         messageDto.setMessageId("fe45b141-f495-48fe-8f7b-ac6f0ef5a47c");
116         messageDto.setStatus(MessageStatus.RESOLVED);
117         messageDto.setActionId(20L);
118         messageDto.setType(MessageType.NORMAL);
119         messageDto.setRetries(3);
120         messageDto.setSent(ZonedDateTime.of(2023, 12, 28, 16, 10, 0, 0, ZoneId.of("UTC")));
121 
122         MessageEntity messageEntity = new MessageEntity();
123         messageEntity.setMessageId("671c4e07-2514-479c-85ed-5950d2baa998");
124         messageEntity.setStatus(MessageStatus.FAILED);
125 
126         InterceptingMappingContext interceptingMappingContext =
127                 InterceptingMappingContext.builder().skipIntercepting(true).build();
128 
129         when(actionMapper.idToEntity(eq(20L), any())).thenReturn(actionEntity);
130 
131         messageMapper.updateEntity(messageDto, messageEntity, interceptingMappingContext);
132 
133         assertNull(messageEntity.getId());
134         assertEquals("fe45b141-f495-48fe-8f7b-ac6f0ef5a47c", messageEntity.getMessageId());
135         assertEquals(MessageStatus.RESOLVED, messageEntity.getStatus());
136         assertSame(actionEntity, messageEntity.getAction());
137         assertEquals(MessageType.NORMAL, messageEntity.getType());
138         assertEquals(3, messageEntity.getRetries());
139         assertEquals(ZonedDateTime.of(2023, 12, 28, 16, 10, 0, 0, ZoneId.of("UTC")), messageEntity.getSent());
