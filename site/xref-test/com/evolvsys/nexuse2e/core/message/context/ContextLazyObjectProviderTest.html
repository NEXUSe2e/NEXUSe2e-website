14  /**
15   * Test für {@link ContextLazyObjectProvider}
16   *
17   * @author SvenBarden
18   * @since 24.01.2024
19   */
20  class ContextLazyObjectProviderTest {
21      private static final Action ACTION = Action.builder().build();
22      private static final ContextPartner PARTNER = ContextPartner.builder().build();
23      private static final ContextChoreography CHOREOGRAPHY =
24              ContextChoreography.builder().build();
25      private final ContextLazyObjectProvider contextLazyObjectProvider = new ContextLazyObjectProvider(
26              List.of(new TestActionFactory(), new TestPartnerFactory(), new TestChoreographyFactory()));
27  
28      private static List<ActionKey> getAction_notFound() {
29          return List.of(new ActionKey("ch", "ac1"), new ActionKey("ch1", "ac"), new ActionKey("ch1", "ac1"));
30      }
31  
32      @Test
33      void getAction_found() {
34          assertSame(ACTION, contextLazyObjectProvider.getAction(new ActionKey("ch", "ac")));
35      }
36  
37      @ParameterizedTest
38      @MethodSource
39      void getAction_notFound(ActionKey actionKey) {
40          assertThat(contextLazyObjectProvider.getAction(actionKey)).isNull();
41      }
42  
43      @Test
44      void getAction_null() {
45          assertNull(contextLazyObjectProvider.getAction(null));
46      }
47  
48      @Test
49      void getPartner_found() {
50          assertSame(PARTNER, contextLazyObjectProvider.getPartner("found"));
51      }
52  
53      @Test
54      void getPartner_notFound() {
55          assertThat(contextLazyObjectProvider.getPartner("not found")).isNull();
56      }
57  
58      @Test
59      void getPartner_null() {
60          assertNull(contextLazyObjectProvider.getPartner(null));
61      }
62  
63      @Test
64      void getChoreography_found() {
65          assertSame(CHOREOGRAPHY, contextLazyObjectProvider.getChoreography("chor"));
66      }
67  
68      @Test
69      void getChoreography_notFound() {
70          assertThat(contextLazyObjectProvider.getChoreography("c")).isNull();
71      }
72  
73      @Test
74      void getChoreography_null() {
75          assertNull(contextLazyObjectProvider.getChoreography(null));
76      }
77  
78      @Test
79      void getConversation_null() {
80          assertNull(contextLazyObjectProvider.getConversation(null));
81      }
82  
83      @Test
84      void getConversation_exception() {
85          assertThrows(IllegalArgumentException.class, () -> contextLazyObjectProvider.getConversation("test"));
86      }
87  
88      private static final class TestActionFactory implements ContextObjectFactory<ActionKey, Action> {
89  
90          @Override
91          public Action create(ActionKey key) {
92              return new ActionKey("ch", "ac").equals(key) ? ACTION : null;
93          }
94      }
95  
96      private static final class TestPartnerFactory implements ContextObjectFactory<String, ContextPartner> {
97  
98          @Override
99          public ContextPartner create(String key) {
100             return "found".equals(key) ? PARTNER : null;
101         }
102     }
103 
104     private static final class TestChoreographyFactory implements ContextObjectFactory<String, ContextChoreography> {
105 
106         @Override
107         public ContextChoreography create(String key) {
108             return "chor".equals(key) ? CHOREOGRAPHY : null;
109         }
110     }
