18  /**
19   * Test für {@link MessageRequeueAction}
20   *
21   * @author SvenBarden
22   * @since 22.05.2024
23   */
24  class MessageRequeueActionTest {
25      private final MessageRepository messageRepository = mock(MessageRepository.class);
26      private final MessageSenderService messageSenderService = mock(MessageSenderService.class);
27      private final MessageActionManualActionService messageActionManualActionService =
28              mock(MessageActionManualActionService.class);
29      private final MessageRequeueAction messageRequeueAction =
30              new MessageRequeueAction(messageRepository, messageSenderService, messageActionManualActionService);
31  
32      @ParameterizedTest
33      @ValueSource(longs = {10, 20})
34      @NullSource
35      void execute(Long actionId) throws NexusException {
36          MessageEntity messageEntity = new MessageEntity();
37          messageEntity.setRetries(5);
38          messageEntity.setLastTry(ZonedDateTime.now());
39  
40          MessageEntity saved = new MessageEntity();
41          saved.setMessageId("my id");
42  
43          when(messageRepository.save(messageEntity)).thenReturn(saved);
44          when(messageActionManualActionService.requeue(actionId, saved)).thenReturn(99L);
45  
46          ProcessActionChain chain = mock(ProcessActionChain.class);
47  
48          MessageEntity result = messageRequeueAction.execute(
49                  messageEntity, chain, new ProcessParameters(MessageRequeueAction.MANUAL_ACTION_ID.of(actionId)));
50  
51          verify(messageRepository).save(messageEntity);
52          verify(messageSenderService).resend("my id");
53          verify(messageActionManualActionService).requeue(actionId, saved);
54          verify(chain).addOutput(MessageRequeueAction.MANUAL_ACTION_ID.of(99L));
55          verifyNoMoreInteractions(messageRepository, messageSenderService, messageActionManualActionService, chain);
56  
57          assertThat(messageEntity.getRetries()).isEqualTo(0);
58          assertThat(messageEntity.getLastTry()).isNull();
59          assertThat(result).isSameAs(saved);
60      }
61  }
