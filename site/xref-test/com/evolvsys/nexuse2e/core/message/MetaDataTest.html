27      @Test
28      void copy() {
29          MetaData metaData = createMetaData();
30          MetaData copy = metaData.copy();
31  
32          assertThat(copy).isEqualTo(metaData);
33          assertThat(copy).isNotSameAs(metaData);
34  
35          metaData.add("copy-key", "value");
36  
37          assertThat(copy).isNotEqualTo(metaData);
38  
39          assertThat(metaData.get("copy-key")).isEqualTo(List.of("value"));
40          assertThat(copy.get("copy-key")).isEqualTo(List.of());
41  
42          assertThat(metaData.keys()).contains("copy-key");
43          assertThat(copy.keys()).doesNotContain("copy-key");
44      }

112     <T> void get(@Nullable String category, String key, @Nullable List<T> expectedValue) {
113         MetaData metaData = createMetaData();
114 
115         String randomDefault = RandomStringUtils.secure().nextAlphanumeric(20);
116 
117         if (expectedValue == null) {
118             Constant<Object> constant = Constant.ofType(key, Object.class);
119             if (category == null) {
120                 assertThat(metaData.getAll(key)).isEqualTo(List.of());
121                 assertThat(metaData.getAll(constant)).isEqualTo(List.of());
122                 assertThat(metaData.getAllFirst(key)).isEmpty();
123                 assertThat(metaData.getAllFirst(constant)).isEmpty();
124                 assertThatThrownBy(() -> metaData.getAllFirstOrThrows(key))
125                         .hasMessageContaining("No metaData with key " + key + " found");
126                 assertThatThrownBy(() -> metaData.getAllFirstOrThrows(constant))
127                         .hasMessageContaining("No metaData with key " + key + " found");
128                 assertThat((T) metaData.getAllFirstOrNull(key)).isNull();
129                 assertThat((T) metaData.getAllFirstOrNull(constant)).isNull();
130                 assertThat(metaData.getAllFirstOrDefault(key, randomDefault)).isEqualTo(randomDefault);
131                 assertThat(metaData.getAllFirstOrDefault(constant, randomDefault))
132                         .isEqualTo(randomDefault);
133             } else if (MetaData.DEFAULT_CATEGORY.equals(category)) {
134                 assertThat(metaData.get(key)).isEqualTo(List.of());
135                 assertThat(metaData.get(constant)).isEqualTo(List.of());
136                 assertThat(metaData.getFirst(key)).isEmpty();
137                 assertThat(metaData.getFirst(constant)).isEmpty();
138                 assertThatThrownBy(() -> metaData.getFirstOrThrows(key))
139                         .hasMessageContaining("No metaData with key " + key + " found");
140                 assertThatThrownBy(() -> metaData.getFirstOrThrows(constant))
141                         .hasMessageContaining("No metaData with key " + key + " found");
142                 assertThat((T) metaData.getFirstOrNull(key)).isNull();
143                 assertThat((T) metaData.getFirstOrNull(constant)).isNull();
144                 assertThat(metaData.getFirstOrDefault(key, randomDefault)).isEqualTo(randomDefault);
145                 assertThat(metaData.getFirstOrDefault(constant, randomDefault)).isEqualTo(randomDefault);
146             } else {
147                 assertThat(metaData.get(category, key)).isEqualTo(List.of());
148                 assertThat(metaData.get(category, constant)).isEqualTo(List.of());
149                 assertThat(metaData.getFirst(category, key)).isEmpty();
150                 assertThat(metaData.getFirst(category, constant)).isEmpty();
151                 assertThatThrownBy(() -> metaData.getFirstOrThrows(category, key))
152                         .hasMessageContaining("No metaData with key " + key + " in category " + category + " found");
153                 assertThatThrownBy(() -> metaData.getFirstOrThrows(category, constant))
154                         .hasMessageContaining("No metaData with key " + key + " in category " + category + " found");
155                 assertThat((T) metaData.getFirstOrNull(category, key)).isNull();
156                 assertThat((T) metaData.getFirstOrNull(category, constant)).isNull();
157                 assertThat(metaData.getFirstOrDefault(category, key, randomDefault))
158                         .isEqualTo(randomDefault);
159                 assertThat(metaData.getFirstOrDefault(category, constant, randomDefault))
160                         .isEqualTo(randomDefault);
161             }
162         } else {
163             T first = expectedValue.getFirst();
164             Constant<T> constant = Constant.ofType(key, Object.class);
165             Constant<String> stringConstant = Constant.ofString(key);
166             if (category == null) {
167                 assertThat(metaData.getAll(key)).isEqualTo(expectedValue);
168                 assertThat(metaData.getAll(constant)).isEqualTo(expectedValue);
169                 assertThat(metaData.getAll(key, first.getClass())).isEqualTo(expectedValue);
170                 assertThat(metaData.getAllFirst(key)).isNotEmpty().get().isEqualTo(first);
171                 assertThat(metaData.getAllFirst(constant)).isNotEmpty().get().isEqualTo(first);
172                 assertThat(metaData.getAllFirst(key, first.getClass()))
173                         .isNotEmpty()
174                         .get()
175                         .isEqualTo(first);
176                 assertThat((T) metaData.getAllFirstOrThrows(key)).isEqualTo(first);
177                 assertThat(metaData.getAllFirstOrThrows(constant)).isEqualTo(first);
178                 assertThat((T) metaData.getAllFirstOrThrows(key, first.getClass()))
179                         .isEqualTo(first);
180                 assertThat((T) metaData.getAllFirstOrNull(key)).isEqualTo(first);
181                 assertThat(metaData.getAllFirstOrNull(constant)).isEqualTo(first);
182                 assertThat((T) metaData.getAllFirstOrNull(key, first.getClass()))
183                         .isEqualTo(first);
184                 assertThat((T) metaData.getAllFirstOrDefault(key, randomDefault))
185                         .isEqualTo(first);
186                 assertThat((T) metaData.getAllFirstOrDefault(stringConstant, randomDefault))
187                         .isEqualTo(first);
188             } else if (MetaData.DEFAULT_CATEGORY.equals(category)) {
189                 assertThat(metaData.get(key)).isEqualTo(expectedValue);
190                 assertThat(metaData.get(constant)).isEqualTo(expectedValue);
191                 assertThat(metaData.get(key, first.getClass())).isEqualTo(expectedValue);
192                 assertThat(metaData.getFirst(key)).isNotEmpty().get().isEqualTo(first);
193                 assertThat(metaData.getFirst(constant)).isNotEmpty().get().isEqualTo(first);
194                 assertThat(metaData.getFirst(key, getClass()))
195                         .isNotEmpty()
196                         .get()
197                         .isEqualTo(first);
198                 assertThat((T) metaData.getFirstOrThrows(key)).isEqualTo(first);
199                 assertThat(metaData.getFirstOrThrows(constant)).isEqualTo(first);
200                 assertThat((T) metaData.getFirstOrThrows(key, first.getClass())).isEqualTo(first);
201                 assertThat((T) metaData.getFirstOrNull(key)).isEqualTo(first);
202                 assertThat(metaData.getFirstOrNull(constant)).isEqualTo(first);
203                 assertThat((T) metaData.getFirstOrNull(key, first.getClass())).isEqualTo(first);
204                 assertThat((T) metaData.getFirstOrDefault(key, randomDefault)).isEqualTo(first);
205                 assertThat((T) metaData.getFirstOrDefault(stringConstant, randomDefault))
206                         .isEqualTo(first);
207             } else {
208                 assertThat(metaData.get(category, key)).isEqualTo(expectedValue);
209                 assertThat(metaData.get(category, constant)).isEqualTo(expectedValue);
210                 assertThat(metaData.get(category, key, first.getClass())).isEqualTo(expectedValue);
211                 assertThat(metaData.getFirst(category, key)).isNotEmpty().get().isEqualTo(first);
212                 assertThat(metaData.getFirst(category, constant))
213                         .isNotEmpty()
214                         .get()
215                         .isEqualTo(first);
216                 assertThat(metaData.getFirst(category, key, first.getClass()))
217                         .isNotEmpty()
218                         .get()
219                         .isEqualTo(first);
220                 assertThat((T) metaData.getFirstOrThrows(category, key)).isEqualTo(first);
221                 assertThat(metaData.getFirstOrThrows(category, constant)).isEqualTo(first);
222                 assertThat((T) metaData.getFirstOrThrows(category, key, first.getClass()))
223                         .isEqualTo(first);
224                 assertThat((T) metaData.getFirstOrNull(category, key)).isEqualTo(first);
225                 assertThat(metaData.getFirstOrNull(category, constant)).isEqualTo(first);
226                 assertThat((T) metaData.getFirstOrNull(category, key, first.getClass()))
227                         .isEqualTo(first);
228                 assertThat((T) metaData.getFirstOrDefault(category, key, randomDefault))
229                         .isEqualTo(first);
230                 assertThat((T) metaData.getFirstOrDefault(category, stringConstant, randomDefault))
231                         .isEqualTo(first);
232             }
233         }
234 
235         assertThat(metaData.categories())
236                 .isEqualTo(Set.of(MetaData.DEFAULT_CATEGORY, "category1", "category2", "category3", "category4"));
237     }
