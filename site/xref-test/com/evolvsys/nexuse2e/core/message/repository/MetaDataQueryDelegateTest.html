30          assertAllNotNull(filter);

36                          """

161     @Test
162     void filterByValues_mixedType() {
163         List<Object> values = List.of(
164                 "1234",
165                 Boolean.TRUE,
166                 3456L,
167                 4567L,
168                 new BigDecimal("5678.90"),
169                 ZonedDateTime.of(2025, 1, 1, 13, 13, 13, 0, ZoneId.of("UTC")),
170                 7890);
171 
172         String expectedPredicate =
173                 """
174                         metaDataEntity.name = key
175                          && (metaDataEntity.stringValue = 1234
176                          || metaDataEntity.longValue in [3456, 4567, 7890]
177                          || metaDataEntity.booleanValue = true
178                          || metaDataEntity.bigDecimalValue = 5678.90
179                          || metaDataEntity.dateTimeValue = 2025-01-01T13:13:13Z[UTC]
180                         )
181                         """
182                         .replace("\n", "");
183         String predicate =
184                 QMetaDataEntity.metaDataEntity.filterByValues("key", values).toString();
185 
186         assertEquals(expectedPredicate, predicate);
187     }
