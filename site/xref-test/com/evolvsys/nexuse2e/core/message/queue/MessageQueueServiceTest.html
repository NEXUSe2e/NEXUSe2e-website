27   * @author SvenBarden
28   * @since 05.03.2024

30  @Slf4j
31  class MessageQueueServiceTest {
32      private final NEXUSe2eConfigurationProperties configurationProperties = new NEXUSe2eConfigurationProperties();
33      private final MessageRepository messageRepository = mock(MessageRepository.class);
34      private final MessageSenderService messageSenderService = mock(MessageSenderService.class);
35      private final Executor messageExecutor = Executors.newFixedThreadPool(3);
36      private final MessageQueueService messageQueueService =
37              new MessageQueueService(configurationProperties, messageRepository, messageSenderService, messageExecutor);
38  
39      @ParameterizedTest
40      @ValueSource(booleans = {true, false})
41      void run(boolean singleThread) throws NexusException {
42          configurationProperties.getMessageQueue().setSingleThread(singleThread);
43  
44          MessageEntity msg1 = MessageEntity.builder().messageId("msg1").build();
45          MessageEntity msg2 = MessageEntity.builder().messageId("msg2").build();
46  
47          List<MessageEntity> messageEntities = List.of(msg1, msg2);
48  
49          AtomicInteger count = new AtomicInteger(0);
50  
51          when(messageRepository.findBy((Predicate) any(), any())).thenReturn(messageEntities);
52          doAnswer(invocationOnMock -> {
53                      LOGGER.debug("calling resend");
54                      Thread.sleep(Duration.ofSeconds(1));
55                      LOGGER.info("incrementing count {}", count.incrementAndGet());
56                      if ("msg1".equals(invocationOnMock.getArgument(0))) {
57                          throw new RuntimeException("test");
58                      }
59                      return null;
60                  })
61                  .when(messageSenderService)
62                  .resend(any());
63  
64          messageQueueService.run();
65  
66          if (singleThread == false) {
67              assertThat(count.intValue()).isEqualTo(0);
68              await().atLeast(Duration.ofMillis(800))
69                      .atMost(Duration.ofMillis(1500))
70                      .untilAtomic(count, equalTo(2));
71          }
72          assertThat(count.intValue()).isEqualTo(2);
73  
74          verify(messageRepository).findBy((Predicate) any(), any());
75  
76          verify(messageSenderService).resend("msg1");
77          verify(messageSenderService).resend("msg2");
78  
79          verifyNoMoreInteractions(messageRepository, messageSenderService);
80      }
81  }
