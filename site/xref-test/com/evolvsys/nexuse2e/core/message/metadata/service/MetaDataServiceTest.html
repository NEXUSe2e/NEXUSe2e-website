44      @CartesianTest
45      void getDuplicatesByKey(
46              @CartesianTest.Values(strings = {"default", "", "category1"}) String category,
47              @CartesianTest.Enum(MetaDataLinkType.class) MetaDataLinkType linkType) {
48          String expectedCategory = StringUtils.defaultIfBlank(category, MetaData.DEFAULT_CATEGORY);
49          MetaData metaData = new MetaData()
50                  .set(expectedCategory, "eventId", List.of("eventId1", "eventId2"))
51                  .set(expectedCategory, "active", List.of(false))
52                  .set("transactionId", List.of(UUID.randomUUID().toString()));
53  
54          when(metaDataRepository.getDuplicatesByKey(
55                          Map.of(
56                                  "eventId", List.of("eventId1", "eventId2"),
57                                  "active", List.of(false)),
58                          expectedCategory,
59                          "choreographyId",
60                          linkType))
61                  .thenReturn(Map.of(
62                          "eventId", List.of(createMetaData(MetaDataType.STRING, "eventId1")),
63                          "active", List.of(createMetaData(MetaDataType.BOOLEAN, false))));
64  
65          Map<String, List<String>> duplicatesByKey = metaDataService.getDuplicatesByKey(
66                  metaData, category, Set.of("eventId", "active"), "choreographyId", linkType);
67  
68          verify(metaDataRepository)
69                  .getDuplicatesByKey(
70                          Map.of(
71                                  "eventId", List.of("eventId1", "eventId2"),
72                                  "active", List.of(false)),
73                          expectedCategory,
74                          "choreographyId",
75                          linkType);
76          verifyNoMoreInteractions(metaDataRepository);
77          assertThat(duplicatesByKey)
78                  .isEqualTo(Map.of(
79                          "eventId", List.of("eventId1"),
80                          "active", List.of("false")));
81      }

165     private MetaDataEntity createMetaData(MetaDataType metaDataType, Object value) {
166         MetaDataEntity metaData = new MetaDataEntity();
167         metaData.setType(metaDataType);
168         switch (metaDataType) {
169             case LONG -> metaData.setLongValue((Long) value);
170             case INTEGER -> metaData.setLongValue(Long.valueOf((Integer) value));
171             case DOUBLE -> metaData.setBigDecimalValue((BigDecimal) value);
172             case STRING -> metaData.setStringValue((String) value);
173             case BOOLEAN -> metaData.setBooleanValue((Boolean) value);
174             case DATETIME -> metaData.setDateTimeValue((ZonedDateTime) value);
175             case PAYLOAD -> metaData.setBinaryValue(new PayloadEntity());
176             case null, default -> fail("Unsupported type: " + metaDataType);
177         }
178         return metaData;
179     }
