18  class BigDecimalMetaDataConverterTest {
19      private final BigDecimalMetaDataConverter bigDecimalMetaDataConverter = new BigDecimalMetaDataConverter();
20  
21      @Test
22      void type() {
23          assertThat(bigDecimalMetaDataConverter.type()).isSameAs(MetaDataType.DOUBLE);
24      }
25  
26      @ParameterizedTest
27      @ValueSource(strings = {"15", "10.5", "10.11"})
28      void updateEntityInternal(String value) {
29          MetaDataEntity metaDataEntity = new MetaDataEntity();
30  
31          bigDecimalMetaDataConverter.updateEntityInternal(new BigDecimal(value), metaDataEntity);
32  
33          assertThat(metaDataEntity.getBigDecimalValue()).isEqualTo(value);
34      }
35  
36      @ParameterizedTest
37      @ValueSource(strings = {"15", "10.5", "10.11"})
38      void getValueInternal(String value) {
39          MetaDataEntity metaDataEntity = new MetaDataEntity();
40          metaDataEntity.setBigDecimalValue(new BigDecimal(value));
41  
42          BigDecimal result = bigDecimalMetaDataConverter.getValueInternal(metaDataEntity);
43  
44          assertThat(result).isEqualTo(value);
45      }
46  }
