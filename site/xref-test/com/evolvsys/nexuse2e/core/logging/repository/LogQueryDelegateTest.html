24  class LogQueryDelegateTest {
25  
26      @Test
27      void filter() {
28          LogFilter filter = LogFilter.builder()
29                  .mdcId("mdc")
30                  .actionId("act")
31                  .choreographyId("chor")
32                  .conversationId("con")
33                  .level("lvl")
34                  .level("lvl2")
35                  .logger("log")
36                  .messageId("msg-id")
37                  .message("msg")
38                  .createdFrom(ZonedDateTime.of(2024, 11, 10, 22, 33, 44, 0, ZoneOffset.UTC))
39                  .createdTill(ZonedDateTime.of(2025, 12, 1, 12, 34, 44, 0, ZoneOffset.UTC))
40                  .type(FilterType.WITHOUT_MESSAGE_CONTEXT)
41                  .createdRange(DateRangeFilterUtils.Range.YESTERDAY)
42                  .build();
43  
44          assertAllNotNull(filter);
45  
46          try (MockedStatic<DateRangeFilterUtils> dateRangeFilterUtilsMockedStatic =
47                  mockStatic(DateRangeFilterUtils.class)) {
48              Predicate predicate = Expressions.stringPath("utils").eq("test");
49              dateRangeFilterUtilsMockedStatic
50                      .when(() -> DateRangeFilterUtils.filter(
51                              QLogEntity.logEntity.created,
52                              "Asia/Hebron",
53                              DateRangeFilterUtils.Range.YESTERDAY,
54                              filter.getCreatedFrom(),
55                              filter.getCreatedTill()))
56                      .thenReturn(predicate);
57  
58              Predicate result = LogQueryDelegate.filter(QLogEntity.logEntity, filter, "Asia/Hebron");
59  
60              dateRangeFilterUtilsMockedStatic.verify(() -> DateRangeFilterUtils.filter(
61                      QLogEntity.logEntity.created,
62                      "Asia/Hebron",
63                      DateRangeFilterUtils.Range.YESTERDAY,
64                      filter.getCreatedFrom(),
65                      filter.getCreatedTill()));
66  
67              String s =
68                      """
69                              contains(logEntity.logger,log)
70                              && logEntity.choreographyId = chor
71                              && logEntity.actionId = act
72                              && logEntity.conversationId = con
73                              && logEntity.messageId = msg-id
74                              && logEntity.mdcId = mdc
75                              && contains(logEntity.message,msg)
76                              && logEntity.level in [lvl, lvl2]
77                              && logEntity.conversationId is null
78                              && utils = test
79                              """
80                              .replace("\n", " ")
81                              .trim();
82  
83              assertEquals(s, result.toString());
84          }
85      }
