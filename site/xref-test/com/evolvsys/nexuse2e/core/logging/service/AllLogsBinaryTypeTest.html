25  class AllLogsBinaryTypeTest {
26      private final NEXUSe2eConfigurationProperties configurationProperties = new NEXUSe2eConfigurationProperties();
27      private final AllLogsBinaryType allLogsBinaryType = new AllLogsBinaryType(configurationProperties);
28  
29      @Test
30      void name() {
31          assertThat(allLogsBinaryType.name()).isSameAs(AllLogsBinaryType.NAME);
32      }
33  
34      @Test
35      void get() throws IOException {
36          configurationProperties.setLogFileDirectory(new ClassPathResource("/logs"));
37  
38          LocalDateTime localDateTime = LocalDateTime.of(2024, 2, 11, 12, 15, 8);
39  
40          try (MockedStatic<LocalDateTime> localDateTimeMock =
41                  mockStatic(LocalDateTime.class, InvocationOnMock::callRealMethod)) {
42              localDateTimeMock.when(LocalDateTime::now).thenReturn(localDateTime);
43  
44              BinaryResult binaryResult = allLogsBinaryType.get("xxx").orElseThrow();
45  
46              assertThat(binaryResult.getMimeType()).isEqualTo("application/zip");
47              assertThat(binaryResult.getFileName()).isEqualTo("NEXUSe2e_logs_2024-02-11T12:15:08.zip");
48              assertThat(binaryResult.getSize()).isEqualTo(433);
49  
50              try (ZipInputStream zipInputStream =
51                      new ZipInputStream(binaryResult.getResource().getInputStream())) {
52                  ZipEntry entry1 = zipInputStream.getNextEntry();
53                  ZipEntry entry2 = zipInputStream.getNextEntry();
54                  ZipEntry entry3 = zipInputStream.getNextEntry();
55                  ZipEntry entry4 = zipInputStream.getNextEntry();
56  
57                  assertThat(entry1).isNotNull();
58                  assertThat(entry2).isNotNull();
59                  assertThat(entry3).isNotNull();
60                  assertThat(entry4).isNull();
61  
62                  List<String> names =
63                          Stream.of(entry1, entry2, entry3).map(ZipEntry::getName).toList();
64                  assertThat(names).containsExactlyInAnyOrder("log.xml", "log1.txt", "log2.txt");
65  
66                  List<Long> sizes =
67                          Stream.of(entry1, entry2, entry3).map(ZipEntry::getSize).toList();
68                  assertThat(sizes).containsExactlyInAnyOrder(64L, 12L, 12L);
69              }
70          }
71      }
72  
73      @Test
74      void get_emptyDir() {
75          configurationProperties.setLogFileDirectory(null);
76  
77          assertThat(allLogsBinaryType.get("any")).isEmpty();
78      }
79  }
