20  class LoggingCleanupServiceTest {
21      private final LoggingService loggingService = mock(LoggingService.class);
22      private final LogRepository logRepository = mock(LogRepository.class);
23      private final MessageRepository messageRepository = mock(MessageRepository.class);
24      private final LoggingCleanupService loggingCleanupService =
25              new LoggingCleanupService(loggingService, logRepository, messageRepository);
26  
27      @Test
28      void updateLogsTest() {
29          MessageContext messageContext = new TestMessageContext();
30          messageContext.setSourceMessage(new Message());
31          messageContext.getSourceMessage().setId("sourceMessageId");
32  
33          when(messageRepository.existsByMessageId("sourceMessageId")).thenReturn(true);
34  
35          loggingCleanupService.updateLogs(messageContext, "mdcId");
36  
37          verify(loggingService).persistQueue();
38          verify(messageRepository).existsByMessageId("sourceMessageId");
39          verifyNoMoreInteractions(messageRepository, loggingService);
40          verifyNoInteractions(logRepository);
41      }
42  
43      @ParameterizedTest
44      @ValueSource(booleans = {true, false})
45      void updateLogsTest_errorMessage(boolean messageExists) {
46          MessageContext messageContext = new TestMessageContext();
47          messageContext.setSourceMessage(new Message());
48          messageContext.getSourceMessage().setId("sourceMessageId");
49          messageContext.setErrorMessage(new Message());
50          messageContext.getErrorMessage().setId("errorMessageId");
51  
52          when(messageRepository.existsByMessageId("errorMessageId")).thenReturn(messageExists);
53          when(messageRepository.existsByMessageId("sourceMessageId")).thenReturn(messageExists);
54  
55          loggingCleanupService.updateLogs(messageContext, "mdcId");
56  
57          verify(loggingService).persistQueue();
58          verify(messageRepository).existsByMessageId("sourceMessageId");
59          verify(messageRepository).existsByMessageId("errorMessageId");
60          verifyNoMoreInteractions(loggingService, messageRepository);
61          if (messageExists) {
62              verifyNoInteractions(logRepository);
63          } else {
64              verify(logRepository).removeMessageIdByMdcId("mdcId", "sourceMessageId");
65              verify(logRepository).removeMessageIdByMdcId("mdcId", "errorMessageId");
66              verifyNoMoreInteractions(loggingService, logRepository);
67          }
68      }
69  
70      @Test
71      void updateLogsTest_noMessageId() {
72          MessageContext messageContext = new TestMessageContext();
73          messageContext.setSourceMessage(new Message());
74  
75          loggingCleanupService.updateLogs(messageContext, "mdcId");
76  
77          verify(loggingService).persistQueue();
78          verifyNoMoreInteractions(loggingService);
79          verifyNoInteractions(messageRepository, logRepository);
80      }
81  }
