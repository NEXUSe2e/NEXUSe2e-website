141     void doAppend(String logger, String level, boolean shouldLog) throws Exception {
142         customEmailLogbackAppender.afterPropertiesSet();
143         customEmailLogbackAppender.start();
144 
145         LoggingEvent loggingEvent = new LoggingEvent(
146                 "fq", (Logger) LoggerFactory.getLogger(logger), Level.valueOf(level), "log", null, null);
147         customEmailLogbackAppender.doAppend(loggingEvent);
148 
149         if (shouldLog) {
150             verify(springTemplateEngine).process(eq("error"), any());
151             verifyNoMoreInteractions(springTemplateEngine);
152         } else {
153             verifyNoInteractions(springTemplateEngine);
154         }
155 
156         // very rarely we get 3 executions in pipeline for whatever reason :(
157         verify(logFilterCacheService, atLeast(2)).getLogFilters(LogFilterType.EMAIL);
158         verifyNoMoreInteractions(logFilterCacheService);
159         MessageLogContextHolder.clearContext();
160     }

162     @Test
163     void afterPropertiesSet() throws Exception {
164         LoggerContext loggerContext = new LoggerContext();
165 
166         try (MockedStatic<LoggerFactory> loggerFactoryMockedStatic = mockStatic(LoggerFactory.class)) {
167             loggerFactoryMockedStatic.when(LoggerFactory::getILoggerFactory).thenReturn(loggerContext);
168 
169             Appender<ILoggingEvent> appender = loggerContext
170                     .getLogger(Logger.ROOT_LOGGER_NAME)
171                     .getAppender(CustomEmailLogbackAppender.class.getSimpleName());
172 
173             assertNull(appender);
174 
175             customEmailLogbackAppender.afterPropertiesSet();
176 
177             verify(logFilterCacheService).getLogFilters(LogFilterType.EMAIL);
178             verifyNoMoreInteractions(logFilterCacheService);

180             appender = loggerContext

182                     .getAppender(CustomEmailLogbackAppender.class.getSimpleName());
