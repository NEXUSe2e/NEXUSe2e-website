29  class DashboardQueryDelegateTest {
30  
31      private static List<Arguments> filter() {
32          return List.of(
33                  Arguments.of(
34                          (BiFunction<DashboardFilter, String, Predicate>) (dashboardFilter, s) ->
35                                  DashboardQueryDelegate.filter(QMessageEntity.messageEntity, dashboardFilter, s),
36                          QMessageEntity.messageEntity.created),
37                  Arguments.of(
38                          (BiFunction<DashboardFilter, String, Predicate>)
39                                  (dashboardFilter, s) -> DashboardQueryDelegate.filter(
40                                          QConversationEntity.conversationEntity, dashboardFilter, s),
41                          QConversationEntity.conversationEntity.created));
42      }
43  
44      @ParameterizedTest
45      @MethodSource
46      void filter(BiFunction<DashboardFilter, String, Predicate> function, DateTimePath<ZonedDateTime> dateTimePath) {
47          DashboardFilter dashboardFilter = DashboardFilter.builder()
48                  .from(ZonedDateTime.of(2024, 1, 1, 12, 30, 0, 0, ZoneId.of("GMT")))
49                  .till(ZonedDateTime.of(2024, 3, 31, 18, 10, 0, 0, ZoneId.of("GMT")))
50                  .range(DateRangeFilterUtils.Range.YESTERDAY)
51                  .build();
52          assertAllNotNull(dashboardFilter);
53  
54          String expectedPredicate = "utils = test";
55  
56          try (MockedStatic<DateRangeFilterUtils> dateRangeFilterUtilsMockedStatic =
57                  mockStatic(DateRangeFilterUtils.class)) {
58              Predicate predicate = Expressions.stringPath("utils").eq("test");
59              dateRangeFilterUtilsMockedStatic
60                      .when(() -> DateRangeFilterUtils.filter(
61                              dateTimePath,
62                              "Asia/Hebron",
63                              DateRangeFilterUtils.Range.YESTERDAY,
64                              dashboardFilter.getFrom(),
65                              dashboardFilter.getTill()))
66                      .thenReturn(predicate);
67  
68              Predicate result = function.apply(dashboardFilter, "Asia/Hebron");
69  
70              dateRangeFilterUtilsMockedStatic.verify(() -> DateRangeFilterUtils.filter(
71                      dateTimePath,
72                      "Asia/Hebron",
73                      DateRangeFilterUtils.Range.YESTERDAY,
74                      dashboardFilter.getFrom(),
75                      dashboardFilter.getTill()));
76  
77              assertEquals(expectedPredicate, result.toString());
78          }
79      }
