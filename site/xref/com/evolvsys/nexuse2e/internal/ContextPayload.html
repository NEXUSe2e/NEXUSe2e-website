9   import java.io.InputStream;
10  import lombok.*;
11  
12  /**
13   * @author SvenBarden
14   * @since 27.10.2023
15   */
16  @Data
17  @NoArgsConstructor
18  @AllArgsConstructor
19  @Builder(toBuilder = true)
20  public class ContextPayload {
21      private Long id;
22  
23      private String contentId;
24  
25      @ToString.Exclude
26      private PayloadContent content;
27  
28      private String contentType;
29      private long size;
30      private String encoding;
31      private int sequenceNumber;
32  
33      @Builder.Default
34      private MetaData metaData = new MetaData();
35  
36      /**
37       * Creates a copy of the ContextPayload. The content is copied to ensure that the original content is not modified. The metaData is also copied to
38       * maintain the original state.
39       *
40       * @return a new ContextPayload instance with copied content
41       * @throws NexusException if there is an error during the content copy process
42       * @see MetaData#copy()
43       */
44      public ContextPayload copy() throws NexusException {
45          PayloadContent contentCopy;
46          try (PayloadContentOutputStream outputStream = PayloadContent.createOutputStream();
47                  InputStream inputStream = content.getInputStream()) {
48              inputStream.transferTo(outputStream);
49              contentCopy = outputStream.finish();
50          } catch (IOException e) {
51              throw new NexusException(
52                      NexusError.builder()
53                              .message("Could not copy content for payload")
54                              .extension("contentId", contentId)
55                              .extension("id", id)
56                              .build(),
57                      e);
58          }
59          return toBuilder()
60                  .id(null)
61                  .content(contentCopy)
62                  .metaData(metaData.copy())
63                  .build();
64      }
65  }
