11  import lombok.extern.slf4j.Slf4j;
12  import org.apache.commons.io.IOUtils;
13  import org.springframework.core.io.ClassPathResource;
14  import org.springframework.stereotype.Component;
15  
16  /**
17   * @author SvenBarden
18   * @since 15.04.2024
19   */
20  @Component
21  @Slf4j
22  public class ChangelogRepository {
23      private static final String BASE_PATH = "/nexuse2e/changelogs/";
24  
25      public ChangelogDto get() throws IOException, NexusException {
26          ChangelogDto.ChangelogDtoBuilder builder = ChangelogDto.builder()
27                  .content(IOUtils.resourceToString("/nexuse2e/CHANGELOG.md", StandardCharsets.UTF_8));
28  
29          for (ExtendedResource changeLogDirResource :
30                  ResourceUtils.list("classpath:/nexuse2e/changelogs/**/version-summary.md")) {
31              addChangelogFromFiles(changeLogDirResource, builder);
32          }
33          return builder.build();
34      }
35  
36      private static void addChangelogFromFiles(ExtendedResource versionSummary, ChangelogDto.ChangelogDtoBuilder builder)
37              throws IOException {
38          String version = ResourceUtils.getParent(versionSummary.getFullPath());
39          LOGGER.trace("adding changelog for version {}", version);
40          ClassPathResource releaseDate = new ClassPathResource(BASE_PATH + version + "/release-date.txt");
41          if (versionSummary.exists() && releaseDate.exists()) {
42              builder.changelog(
43                      version,
44                      SingleChangelogDto.builder()
45                              .version(version)
46                              .content(IOUtils.toString(versionSummary.getInputStream(), StandardCharsets.UTF_8))
47                              .releaseDate(LocalDate.parse(
48                                      IOUtils.toString(releaseDate.getInputStream(), StandardCharsets.UTF_8)))
49                              .build());
50          }
51      }
52  }
