19   * @author SvenBarden
20   * @since 01.03.2024
21   */
22  @Slf4j
23  @RequiredArgsConstructor
24  public class ManualMessageResolver extends MessageResolver {
25      private final MessageType messageType;
26      private final MessageResolverValidator messageResolverValidator;
27      private final MessageEventService messageEventService;
28      private final ContextFactory contextFactory;
29  
30      @Override
31      public MessageContext resolve(MessageContext messageContext) throws NexusException {
32  
33          LOGGER.debug("running manual message resolver");
34  
35          ExecutionContext executionContext = messageContext.createExecutionContext(messageType);
36  
37          Message message = executionContext.getMessage();
38  
39          messageEventService.trigger(executionContext, MessageEventType.RESOLVING);
40  
41          messageResolverValidator.validate(
42                  messageContext, Default.class, MessageContextValidationGroups.ManualMessage.class);
43  
44          LOGGER.debug(
45                  "processed message with {} payloads with messageType {}",
46                  message.getPayloads().size(),
47                  messageType);
48  
49          if (contextFactory.exists(message.getId())) {
50              throw new NexusException(NexusError.builder()
51                      .message("Cannot resend same message with manual message operation")
52                      .extension("messageId", message.getId())
53                      .build());
54          } else {
55              LOGGER.debug("message is new, triggering resolved");
56              message.setProcessed(ZonedDateTime.now());
57  
58              messageEventService.trigger(executionContext, MessageEventType.RESOLVED);
59          }
60  
61          return messageContext;
62      }
63  }
