13  /**
14   * @author SvenBarden

16   */
17  @Component
18  @RequiredArgsConstructor
19  @Slf4j
20  public class MessageResolverFactory {
21      private final MessageResolverValidator messageResolverValidator;
22      private final PipelineFactory pipelineFactory;
23      private final MessageEventService messageEventService;
24      private final ContextFactory contextFactory;
25  
26      public MessageResolver create(@Nullable PipelineEntity pipeline, MessageType messageType) {
27          return switch (messageType) {
28              case ERROR -> new ErrorMessageResolver(pipeline == null ? null : pipelineFactory.get(pipeline));
29              case RETURN -> new ReturnMessageResolver(pipeline == null ? null : pipelineFactory.get(pipeline));
30              default ->
31                  new NormalMessageResolver(
32                          pipeline == null ? null : pipelineFactory.get(pipeline),
33                          messageType,
34                          messageResolverValidator,
35                          messageEventService,
36                          contextFactory);
37          };
38      }
39  }
