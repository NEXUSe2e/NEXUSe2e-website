10  import jakarta.validation.ConstraintValidator;

12  import java.util.EnumSet;
13  import java.util.Set;
14  import lombok.RequiredArgsConstructor;
15  import lombok.extern.slf4j.Slf4j;
16  import org.apache.commons.collections4.keyvalue.DefaultKeyValue;
17  import org.apache.commons.lang3.StringUtils;
18  import org.springframework.stereotype.Component;

20  /**
21   * @author SvenBarden
22   * @since 18.04.2024
23   */
24  @Component
25  @RequiredArgsConstructor
26  @Slf4j
27  class ValidAcknowledgmentConstraintValidator implements ConstraintValidator<ValidAcknowledgment, MessageContext> {
28      private static final Set<MessageStatus> VALID_MESSAGE_STATUS =
29              EnumSet.of(MessageStatus.FAILED_MISSING_ACKNOWLEDGMENT, MessageStatus.WAITING_FOR_ACKNOWLEDGMENT);
30  
31      private final MessageRepository messageRepository;
32  
33      private void emptyReferenceId(ConstraintValidatorContext context, Message message) {
34          context.disableDefaultConstraintViolation();
35          BeanValidationUtils.addMessageParameters(
36                  context,
37                  new DefaultKeyValue<>("acknowledgmentId", message.getId()),
38                  new DefaultKeyValue<>("conversationId", message.getConversationId()));
39          context.buildConstraintViolationWithTemplate(
40                          "{com.evolvsys.nexuse2e.core.resolver.validation.ValidAcknowledgment.missingReferenceId.message}")
41                  .addConstraintViolation();
42      }
43  
44      private void noReferenceMessage(ConstraintValidatorContext context, Message message) {
45          context.disableDefaultConstraintViolation();
46          BeanValidationUtils.addMessageParameters(
47                  context,
48                  new DefaultKeyValue<>("acknowledgmentId", message.getId()),
49                  new DefaultKeyValue<>("referenceMessageId", message.getReferenceMessageId()),
50                  new DefaultKeyValue<>("conversationId", message.getConversationId()));
51          context.buildConstraintViolationWithTemplate(
52                          "{com.evolvsys.nexuse2e.core.resolver.validation.ValidAcknowledgment.noReferenceMessage.message}")
53                  .addConstraintViolation();
54      }
55  
56      private void invalidState(ConstraintValidatorContext context, Message message, MessageStatus messageStatus) {
57          context.disableDefaultConstraintViolation();
58          BeanValidationUtils.addMessageParameters(
59                  context,
60                  new DefaultKeyValue<>("acknowledgmentId", message.getId()),
61                  new DefaultKeyValue<>("referenceMessageId", message.getReferenceMessageId()),
62                  new DefaultKeyValue<>("conversationId", message.getConversationId()),
63                  new DefaultKeyValue<>("messageStatus", messageStatus));
64          context.buildConstraintViolationWithTemplate(
65                          "{com.evolvsys.nexuse2e.core.resolver.validation.ValidAcknowledgment.invalidStatus.message}")
66                  .addConstraintViolation();
67      }
68  
69      @Override
70      public boolean isValid(MessageContext value, ConstraintValidatorContext context) {
71          Message message = value.getSourceMessage();
72          boolean valid;
73          if (message.getType() == MessageType.ACKNOWLEDGMENT) {
74              String referenceMessageId = message.getReferenceMessageId();
75              if (StringUtils.isBlank(referenceMessageId)) {
76                  valid = false;
77                  LOGGER.debug("acknowledgment {} has no referenceId", message.getId());
78                  emptyReferenceId(context, message);
79              } else {
80                  MessageEntity messageEntity =
81                          messageRepository.findByMessageId(referenceMessageId).orElse(null);
82                  if (messageEntity == null) {
83                      valid = false;
84                      LOGGER.debug(
85                              "reference {} for acknowledgment {} does not exist",
86                              message.getReferenceMessageId(),
87                              message.getId());
88                      noReferenceMessage(context, message);
89                  } else {
90                      if (VALID_MESSAGE_STATUS.contains(messageEntity.getStatus())) {
91                          valid = true;
92                          LOGGER.debug(
93                                  "got acknowledgment {} for message {} in state {}",
94                                  message.getId(),
95                                  message.getReferenceMessageId(),
96                                  messageEntity.getStatus());
97                      } else {
98                          valid = false;
99                          LOGGER.warn(
100                                 "got acknowledgment {} for message {}, but message was in state {}, valid states are {}",
101                                 message.getId(),
102                                 message.getReferenceMessageId(),
103                                 messageEntity.getStatus(),
104                                 VALID_MESSAGE_STATUS);
105                         invalidState(context, message, messageEntity.getStatus());
106                     }
107                 }
108             }
109         } else {
110             valid = true;
111         }
112         return valid;
113     }
114 }
