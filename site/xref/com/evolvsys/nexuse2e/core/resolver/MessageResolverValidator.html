17   * @author SvenBarden

20  @RequiredArgsConstructor
21  @Component
22  public class MessageResolverValidator {
23      private final ValidatorFactory validatorFactory;
24  
25      public void validate(MessageContext messageContext, Class<?>... groups) throws NexusException {
26          Set<ConstraintViolation<MessageContext>> violations =
27                  validatorFactory.getValidator().validate(messageContext, groups);
28  
29          List<ConstraintViolation<MessageContext>> violationList = violations.stream()
30                  .sorted((o1, o2) -> {
31                      Integer order1 = (Integer) AnnotationUtils.getValue(
32                              o1.getConstraintDescriptor().getAnnotation(), "order");
33                      Integer order2 = (Integer) AnnotationUtils.getValue(
34                              o2.getConstraintDescriptor().getAnnotation(), "order");
35  
36                      return ObjectUtils.defaultIfNull(order1, Integer.MAX_VALUE)
37                              .compareTo(ObjectUtils.defaultIfNull(order2, Integer.MAX_VALUE));
38                  })
39                  .toList();
40  
41          // AnnotationUtils.getValue()
42          List<Map<String, Object>> constraintViolations = new ArrayList<>(violationList.size());
43  
44          ConstraintViolation<MessageContext> first = null;
45          for (ConstraintViolation<MessageContext> violation : violationList) {
46              if (first == null) {
47                  first = violation;
48              }
49              Map<String, Object> map = LinkedHashMap.newLinkedHashMap(3);
50              map.put("message", violation.getMessage());
51              map.put("key", violation.getMessageTemplate());
52              map.put("path", violation.getPropertyPath().toString());
53              constraintViolations.add(map);
54          }
55  
56          if (first != null) {
57              NexusError nexusError = NexusError.builder()
58                      .message(first.getMessage())
59                      .code("RESOLVER_VALIDATION")
60                      .extension("key", first.getMessageTemplate())
61                      .extension("path", first.getPropertyPath().toString())
62                      .extension("violations", constraintViolations.toString())
63                      .build();
64  
65              throw new NexusException(nexusError);
66          }
67      }
68  
69      public void validate(MessageContext messageContext) throws NexusException {
70          validate(messageContext, Default.class, MessageContextValidationGroups.Resolver.class);
71      }
72  }
