17  public final class ManualActionQueryDelegate {
18      private ManualActionQueryDelegate() {}
19  
20      @QueryDelegate(ManualActionEntity.class)
21      public static Predicate filter(
22              QManualActionEntity manualActionEntity, @Nullable ManualActionFilter manualActionFilter) {
23          BooleanBuilder builder = new BooleanBuilder();
24          if (manualActionFilter != null) {
25              if (manualActionFilter.getUserId() != null) {
26                  builder.and(manualActionEntity.user.id.eq(manualActionFilter.getUserId()));
27              }
28              if (CollectionUtils.isNotEmpty(manualActionFilter.getTypes())) {
29                  builder.and(manualActionEntity.type.in(manualActionFilter.getTypes()));
30              }
31              if (manualActionFilter.getFrom() != null) {
32                  builder.and(manualActionEntity.created.goe(manualActionFilter.getFrom()));
33              }
34              if (manualActionFilter.getTo() != null) {
35                  builder.and(manualActionEntity.created.loe(manualActionFilter.getTo()));
36              }
37              if (manualActionFilter.getMessageId() != null) {
38                  builder.and(manualActionEntity
39                          .as(QMessageActionManualActionEntity.class)
40                          .messages
41                          .any()
42                          .id
43                          .in(manualActionFilter.getMessageId()));
44              }
45          }
46  
47          return builder.getValue();
48      }
49  }
