5   import com.evolvsys.app.core.mapper.DefaultMapperConfiguration;
6   import com.evolvsys.app.core.mapper.interceptor.InterceptingMappingContext;

8   import com.evolvsys.nexuse2e.core.manualaction.model.ConfigImportManualActionDto;
9   import com.evolvsys.nexuse2e.core.manualaction.model.ConfigImportManualActionEntity;
10  import com.evolvsys.nexuse2e.core.manualaction.repository.ConfigImportManualActionRepository;
11  import com.evolvsys.nexuse2e.core.manualaction.service.ConfigImportNewFileOwnerProvider;
12  import com.evolvsys.nexuse2e.core.manualaction.service.ConfigImportOldFileOwnerProvider;
13  import org.mapstruct.*;
14  import org.springframework.beans.factory.annotation.Autowired;
15  
16  /**
17   * @author SvenBarden
18   * @since 05.08.2024

20  @Mapper(
21          config = DefaultMapperConfiguration.class,
22          uses = {UserMapper.class})
23  public abstract class ConfigImportManualActionMapper
24          extends BaseDtoEntityMapper<ConfigImportManualActionDto, ConfigImportManualActionEntity> {
25      private ConfigImportManualActionRepository configImportManualActionRepository;
26      private FileMapper fileMapper;
27  
28      @Autowired
29      protected void config(
30              ConfigImportManualActionRepository configImportManualActionRepository, FileMapper fileMapper) {
31          this.configImportManualActionRepository = configImportManualActionRepository;
32          this.fileMapper = fileMapper;
33      }
34  
35      @Override
36      public ConfigImportManualActionEntity idToEntity(
37              Long id, @Context InterceptingMappingContext interceptingMappingContext) {
38          return idToEntity(id, interceptingMappingContext, configImportManualActionRepository::findById);
39      }
40  
41      @InheritConfiguration
42      @ManualActionToDto
43      @Mapping(target = "oldConfig", ignore = true)
44      @Mapping(target = "newConfig", ignore = true)
45      @Override
46      protected abstract ConfigImportManualActionDto entityToDto(
47              ConfigImportManualActionEntity entity, @Context InterceptingMappingContext interceptingMappingContext);
48  
49      @InheritConfiguration
50      @ManualActionToEntity
51      @Mapping(target = "oldConfig", ignore = true)
52      @Mapping(target = "newConfig", ignore = true)
53      @Override
54      protected abstract void updateEntity(
55              ConfigImportManualActionDto dto,
56              @MappingTarget ConfigImportManualActionEntity entity,
57              @Context InterceptingMappingContext interceptingMappingContext);
58  
59      @AfterMapping
60      protected void fileToDto(ConfigImportManualActionEntity entity, @MappingTarget ConfigImportManualActionDto dto) {
61          if (entity.getOldConfig() != null) {
62              dto.setOldConfig(fileMapper.entityToDto(
63                      entity.getOldConfig(), ConfigImportOldFileOwnerProvider.OLD_CONFIG, entity.getId()));
64          }
65          if (entity.getNewConfig() != null) {
66              dto.setNewConfig(fileMapper.entityToDto(
67                      entity.getNewConfig(), ConfigImportNewFileOwnerProvider.NEW_CONFIG, entity.getId()));
68          }
69      }
70  
71      @AfterMapping
72      protected void fileToEntity(ConfigImportManualActionDto dto, @MappingTarget ConfigImportManualActionEntity entity) {
73          fileMapper.dtoToEntity(dto.getOldConfig(), entity::getOldConfig, entity::setOldConfig);
74          fileMapper.dtoToEntity(dto.getNewConfig(), entity::getNewConfig, entity::setNewConfig);
75      }
76  }
