29  @Service
30  @RequiredArgsConstructor
31  @Slf4j
32  public class ConfigImportManualActionService {
33      private final ConfigImportManualActionRepository configImportManualActionRepository;
34      private final ConfigParserService configParserService;
35      private final FileFactory fileFactory;
36      private final SystemConfigurationProperties systemConfigurationProperties;
37      private final CurrentUserService currentUserService;
38  
39      private FileEntity configToFileEntity(UserEntity userEntity, NexusConfig config, String fileNameSuffix)
40              throws NexusException {
41          ByteArrayOutputStream outputStream = new ByteArrayOutputStream();
42          try {
43              configParserService.print(outputStream, config);
44          } catch (IOException e) {
45              throw new NexusException(
46                      NexusError.builder()
47                              .message("Could not create output stream for config")
48                              .extension("suffix", fileNameSuffix)
49                              .build(),
50                      e);
51          }
52  
53          StringBuilder fileName = new StringBuilder("NEXUSe2e_config_").append(fileNameSuffix);
54          String environmentName = systemConfigurationProperties.getEnvironmentName();
55          if (StringUtils.isNotBlank(environmentName)) {
56              fileName.append('_').append(environmentName);
57          }
58          fileName.append('_')
59                  .append(DateTimeFormatter.ISO_OFFSET_DATE_TIME.format(
60                          ZonedDateTime.now().withNano(0)))
61                  .append(".xml");
62  
63          FileEntity fileEntity = fileFactory.create(
64                  fileName.toString().replaceAll("[^A-Za-z0-9.\\-_+]", "_"),
65                  "text/xml",
66                  outputStream.toByteArray(),
67                  userEntity);
68          fileEntity.setStatus(FileStatus.SAVED);
69          return fileEntity;
70      }
71  
72      private void mergeInternal(UserEntity currentUser, NexusConfig oldConfig, NexusConfig newConfig)
73              throws NexusException {
74          ConfigImportManualActionEntity manualActionEntity = new ConfigImportManualActionEntity();
75          manualActionEntity.setOldConfig(configToFileEntity(currentUser, oldConfig, "old"));
76          manualActionEntity.setNewConfig(configToFileEntity(currentUser, newConfig, "new"));
77          manualActionEntity.setOverwrite(false);
78          configImportManualActionRepository.save(manualActionEntity);
79          LOGGER.info("{} {} merged config", currentUser.getFirstName(), currentUser.getLastName());
80      }
81  
82      @Transactional(rollbackFor = NexusException.class)
83      public void merge(NexusConfig oldConfig, NexusConfig newConfig) throws NexusException {
84          try {
85              UserEntity currentUser = currentUserService.getCurrentUser();
86              mergeInternal(currentUser, oldConfig, newConfig);
87          } catch (IllegalStateException e) {
88              // no user currently logged in, e.g. during startup or internal calls
89          }
90      }
91  
92      private void overwriteInternal(UserEntity currentUser, NexusConfig oldConfig, NexusConfig newConfig)
93              throws NexusException {
94          ConfigImportManualActionEntity manualActionEntity = new ConfigImportManualActionEntity();
95          manualActionEntity.setOldConfig(configToFileEntity(currentUser, oldConfig, "old"));
96          manualActionEntity.setNewConfig(configToFileEntity(currentUser, newConfig, "new"));
97          manualActionEntity.setOverwrite(true);
98          configImportManualActionRepository.save(manualActionEntity);
99          LOGGER.info("{} {} overwrote config", currentUser.getFirstName(), currentUser.getLastName());
100     }
101 
102     @Transactional(rollbackFor = NexusException.class)
103     public void overwrite(NexusConfig oldConfig, NexusConfig newConfig) throws NexusException {
104         try {
105             UserEntity currentUser = currentUserService.getCurrentUser();
106             overwriteInternal(currentUser, oldConfig, newConfig);
107         } catch (IllegalStateException e) {
108             // no user currently logged in, e.g. during startup or internal calls
109         }
110     }
