16  /**
17   * @author SvenBarden
18   * @since 05.08.2024
19   */
20  @Service
21  @RequiredArgsConstructor
22  @Slf4j
23  public class MessageActionManualActionService {
24      private final MessageActionManualActionRepository manualActionRepository;
25      private final CurrentUserService currentUserService;
26  
27      private Long saveEntity(Long manualActionId, MessageEntity messageEntity, MessageActionManualActionType type) {
28          MessageActionManualActionEntity manualAction;
29          if (manualActionId == null) {
30              manualAction = new MessageActionManualActionEntity();
31          } else {
32              manualAction = manualActionRepository
33                      .findById(manualActionId)
34                      .orElseThrow(() -> new NexusRuntimeException(NexusError.builder()
35                              .message("Manual action not found")
36                              .extension("manualActionId", manualActionId)
37                              .build()));
38          }
39          manualAction.setActionType(type);
40          manualAction.getMessages().add(messageEntity);
41          manualAction = manualActionRepository.save(manualAction);
42  
43          return manualAction.getId();
44      }
45  
46      public Long requeue(Long manualActionId, MessageEntity messageEntity) {
47          manualActionId = saveEntity(manualActionId, messageEntity, MessageActionManualActionType.REQUEUE);
48          LoggingUtils.forMessageWrapper(LOGGER, ForMessageLoggingContext.forEntity(messageEntity))
49                  .info(
50                          "{} {} requeued message {}",
51                          currentUserService.getCurrentUser().getFirstName(),
52                          currentUserService.getCurrentUser().getLastName(),
53                          messageEntity.getMessageId());
54          return manualActionId;
55      }
56  
57      public Long stop(Long manualActionId, MessageEntity messageEntity) {
58          manualActionId = saveEntity(manualActionId, messageEntity, MessageActionManualActionType.STOP);
59          LoggingUtils.forMessageWrapper(LOGGER, ForMessageLoggingContext.forEntity(messageEntity))
60                  .info(
61                          "{} {} stopped message {}",
62                          currentUserService.getCurrentUser().getFirstName(),
63                          currentUserService.getCurrentUser().getLastName(),
64                          messageEntity.getMessageId());
65          return manualActionId;
66      }
67  }
