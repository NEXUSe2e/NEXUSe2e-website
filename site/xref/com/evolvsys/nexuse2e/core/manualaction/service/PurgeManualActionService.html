20   * @author SvenBarden

23  @Service
24  @RequiredArgsConstructor
25  @Slf4j
26  public class PurgeManualActionService {
27      private final PurgeManualActionRepository purgeManualActionRepository;
28      private final CurrentUserService currentUserService;
29      private final ObjectMapper objectMapper;
30  
31      private void purgeConversationsInternal(
32              UserEntity currentUser, ConversationFilter conversationFilter, PurgeResult purgeResult)
33              throws NexusException {
34          PurgeManualActionEntity purgeManualActionEntity = new PurgeManualActionEntity();
35          purgeManualActionEntity.setPurgeType(PurgeManualActionType.CONVERSATIONS);
36          purgeManualActionEntity.setResult(purgeResult);
37          if (conversationFilter != null) {
38              try {
39                  purgeManualActionEntity.setFilter(objectMapper.writeValueAsString(conversationFilter));
40              } catch (JsonProcessingException e) {
41                  throw new NexusException(
42                          NexusError.builder()
43                                  .message("Could not deserialize conversation filter")
44                                  .extension("filter", conversationFilter)
45                                  .build(),
46                          e);
47              }
48          }
49          purgeManualActionRepository.save(purgeManualActionEntity);
50          LOGGER.info(
51                  "{} {} purged conversations: {} with filter {}",
52                  currentUser.getFirstName(),
53                  currentUser.getLastName(),
54                  purgeResult,
55                  conversationFilter);
56      }
57  
58      public void purgeConversations(ConversationFilter conversationFilter, PurgeResult purgeResult)
59              throws NexusException {
60          try {
61              UserEntity currentUser = currentUserService.getCurrentUser();
62              purgeConversationsInternal(currentUser, conversationFilter, purgeResult);
63          } catch (IllegalStateException e) {
64              // no user currently logged in, e.g. during startup or internal calls
65          }
66      }
67

73          if (logFilter != null) {
74              try {
75                  purgeManualActionEntity.setFilter(objectMapper.writeValueAsString(logFilter));

78                          NexusError.builder()
79                                  .message("Could not deserialize log filter")
80                                  .extension("filter", logFilter)
81                                  .build(),
82                          e);
83              }
84          }
85          purgeManualActionRepository.save(purgeManualActionEntity);

87                  "{} {} purged logs: {} with filter {}",
88                  currentUser.getFirstName(),
89                  currentUser.getLastName(),
90                  purgeResult,
91                  logFilter);
92      }
93  
94      public void purgeLogs(LogFilter logFilter, PurgeResult purgeResult) throws NexusException {
95          try {
96              UserEntity currentUser = currentUserService.getCurrentUser();

98          } catch (IllegalStateException e) {

100         }
101     }
