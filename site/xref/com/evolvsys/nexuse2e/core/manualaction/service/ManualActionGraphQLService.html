24  @GraphQLService
25  @RequiredArgsConstructor
26  public class ManualActionGraphQLService {
27      private final ManualActionCrudService manualActionService;
28      private final DataLoaderService dataLoaderService;
29  
30      @GraphQLQuery(name = "manualAction_list")
31      @GraphQLNonNull
32      public Page<@GraphQLNonNull ManualActionDto> list(Pageable pageable, @Valid ManualActionFilter manualActionFilter) {
33          return manualActionService.list(pageable, manualActionFilter);
34      }
35  
36      @GraphQLQuery
37      public CompletableFuture<UserDto> user(
38              @GraphQLContext ManualActionDto manualActionDto,
39              @GraphQLEnvironment ResolutionEnvironment resolutionEnvironment) {
40          return dataLoaderService.load(
41                  resolutionEnvironment, UserBatchLoader.class, manualActionDto, "userId", ManualActionDto::getUserId);
42      }
43  }
