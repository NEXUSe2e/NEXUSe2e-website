8   import java.util.Map;
9   import java.util.Set;
10  import java.util.concurrent.CompletionStage;

13  import org.springframework.stereotype.Component;
14  
15  /**
16   * @author SvenBarden
17   * @since 12.08.2024
18   */
19  @Component
20  @RequiredArgsConstructor
21  public class MessageCountMessageActionManualActionBatchLoader implements GraphQLBatchLoader<Long, Long> {
22      private final JPQLQueryFactory queryFactory;
23  
24      @Override
25      public CompletionStage<Map<Long, Long>> load(Set<Long> keys, BatchLoaderEnvironment environment) {
26          return BatchLoaderUtils.load(keys, this::loadInternal);
27      }
28  
29      private Map<Long, Long> loadInternal(Set<Long> keys) {
30          return queryFactory
31                  .select(
32                          QMessageActionManualActionEntity.messageActionManualActionEntity.id,
33                          QMessageActionManualActionEntity.messageActionManualActionEntity
34                                  .messages
35                                  .size()
36                                  .longValue())
37                  .from(QMessageActionManualActionEntity.messageActionManualActionEntity)
38                  .where(QMessageActionManualActionEntity.messageActionManualActionEntity.id.in(keys))
39                  .transform(GroupBy.groupBy(QMessageActionManualActionEntity.messageActionManualActionEntity.id)
40                          .as(QMessageActionManualActionEntity.messageActionManualActionEntity
41                                  .messages
42                                  .size()
43                                  .longValue()));
44      }
45  }
