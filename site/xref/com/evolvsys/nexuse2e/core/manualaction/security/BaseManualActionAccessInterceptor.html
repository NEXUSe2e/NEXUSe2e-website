3   import com.evolvsys.app.core.auth.model.CurrentUserData;
4   import com.evolvsys.app.core.security.service.AccessInterceptor;
5   import com.evolvsys.nexuse2e.core.manualaction.model.ManualActionDto;
6   import com.evolvsys.nexuse2e.core.model.dto.NexusTenantRelation;
7   import com.querydsl.core.types.dsl.BooleanExpression;
8   import com.querydsl.core.types.dsl.Expressions;
9   
10  /**
11   * @author SvenBarden
12   * @since 05.08.2024
13   */
14  public abstract class BaseManualActionAccessInterceptor<T extends ManualActionDto> implements AccessInterceptor<T> {
15      public static final String READ_MANUAL_ACTION = "NEXUS.READ_MANUAL_ACTION";
16  
17      @Override
18      public BooleanExpression getPredicate(CurrentUserData currentUserData) {
19          boolean contains = currentUserData
20                  .getPermissionsForTenant(NexusTenantRelation.SYSTEM_TENANT_ID)
21                  .contains(READ_MANUAL_ACTION);
22          BooleanExpression booleanExpression;
23          if (contains) {
24              booleanExpression = null;
25          } else {
26              booleanExpression = Expressions.TRUE.isFalse();
27          }
28          return booleanExpression;
29      }
30  }
