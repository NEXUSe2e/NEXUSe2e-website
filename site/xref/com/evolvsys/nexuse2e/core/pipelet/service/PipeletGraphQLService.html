13  import java.util.Set;

19  /**
20   * @author Florian Drees
21   * @since 04.12.2023
22   */
23  @GraphQLService
24  @RequiredArgsConstructor
25  @Slf4j
26  public class PipeletGraphQLService {
27      private final PipeletService pipeletService;
28      private final DataLoaderService dataLoaderService;
29  
30      @GraphQLMutation(name = "pipelet_update", description = "save parameters of a pipelet")
31      @GraphQLNonNull
32      public PipeletDto update(@GraphQLNonNull @Valid PipeletDto pipeletDto) {
33          return ConfigGraphQLService.notifySubscribers(pipeletService.save(pipeletDto));
34      }
35  
36      @GraphQLQuery(name = "pipelet_getById", description = "Gets a pipelet")
37      public Optional<PipeletDto> getById(long pipeletId) {
38          return pipeletService.getById(pipeletId);
39      }
40  
41      @GraphQLQuery(description = "list of parameters of the provided pipelet")
42      public CompletableFuture<Set<@GraphQLNonNull ParameterDto>> parameters(
43              @GraphQLContext PipeletDto pipeletDto, @GraphQLEnvironment ResolutionEnvironment resolutionEnvironment) {
44          LOGGER.trace("getting parameters for pipelet {}", pipeletDto);
45          return dataLoaderService.load(
46                  resolutionEnvironment, PipeletParameterBatchLoader.class, pipeletDto, Function.identity());
47      }
48  }
