14  import org.springframework.lang.Nullable;
15  import org.springframework.stereotype.Service;
16  
17  /**
18   * @author Florian Drees
19   * @since 03.06.2024
20   */
21  @Service
22  @RequiredArgsConstructor
23  @Slf4j
24  public class LoggingCleanupService {
25      private final LoggingService loggingService;
26      private final LogRepository logRepository;
27      private final MessageRepository messageRepository;
28  
29      private void removeMessageReference(String mdcId, String messageId) {
30          try {
31              int updateLogEntries = logRepository.removeMessageIdByMdcId(mdcId, messageId);
32              LoggingUtils.globalWrapper(LOGGER)
33                      .trace(
34                              "Removed message reference '{}' from {} log entries (mdcId: {})",
35                              messageId,
36                              updateLogEntries,
37                              mdcId);
38          } catch (RuntimeException e) {
39              LOGGER.error(
40                      "Error removing log messages for mdcId {} and messageId {}: {}", mdcId, messageId, e.getMessage());
41          }
42      }
43  
44      private void updateLogs(@Nullable Message message, @NonNull String mdcId) {
45          String messageId = Optional.ofNullable(message).map(Message::getId).orElse(null);
46          if (StringUtils.isNotEmpty(messageId)) {
47              if (!messageRepository.existsByMessageId(messageId)) {
48                  removeMessageReference(mdcId, messageId);
49              }
50          }
51      }
52  
53      public void updateLogs(@NonNull MessageContext messageContext, @NonNull String mdcId) {
54          Validate.notNull(messageContext, "messageContext must not be null");
55          Validate.notNull(mdcId, "mdcId must not be null");
56          LoggingUtils.globalWrapper(LOGGER).trace("cleaning logs for mdcId {}", mdcId);
57          loggingService.persistQueue();
58          updateLogs(messageContext.getSourceMessage(), mdcId);
59          updateLogs(messageContext.getErrorMessage(), mdcId);
60          updateLogs(messageContext.getReturnMessage(), mdcId);
61          updateLogs(messageContext.getAcknowledgmentMessage(), mdcId);
62      }
63  }
