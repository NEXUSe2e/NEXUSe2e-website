22  @Service
23  @RequiredArgsConstructor
24  public class LogFileService {
25      private final NEXUSe2eConfigurationProperties configurationProperties;
26  
27      public List<BinaryObjectDto> list() throws NexusException {
28          List<BinaryObjectDto> files = new ArrayList<>();
29  
30          Resource directory = configurationProperties.getLogFileDirectory();
31          if (directory != null) {
32              try {
33                  File directoryFile = directory.getFile();
34                  if (directoryFile.exists() && directoryFile.isDirectory()) {
35                      addFiles(directoryFile, files);
36                  }
37              } catch (IOException e) {
38                  throw new NexusException(
39                          NexusError.builder()
40                                  .message("Could not load log file list")
41                                  .build(),
42                          e);
43              }
44          }
45  
46          return files;
47      }
48  
49      private static void addFiles(File directoryFile, Collection<BinaryObjectDto> files) throws IOException {
50          File[] fileArray = directoryFile.listFiles();
51          if (fileArray != null) {
52              Tika tika = new Tika();
53              for (File file : fileArray) {
54                  if (file.isFile()) {
55                      files.add(BinaryObjectDto.builder()
56                              .fileName(file.getName())
57                              .size(file.length())
58                              .mimeType(tika.detect(file))
59                              .path(BinaryType.getPath(SingleLogBinaryType.NAME, file.getName()))
60                              .build());
61                  }
62              }
63          }
64      }
65  }
