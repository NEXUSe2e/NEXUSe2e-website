1   package com.evolvsys.nexuse2e.core.logging.service;
2   
3   import com.evolvsys.nexuse2e.core.logging.LogFilterCacheContainer;
4   import com.evolvsys.nexuse2e.core.logging.mapper.LogFilterCacheMapper;
5   import com.evolvsys.nexuse2e.core.logging.model.LogFilterType;
6   import com.evolvsys.nexuse2e.core.logging.repository.LogFilterRepository;
7   import lombok.RequiredArgsConstructor;
8   import lombok.extern.slf4j.Slf4j;
9   import org.springframework.cache.annotation.Cacheable;
10  import org.springframework.stereotype.Service;
11  
12  /**
13   * @author SvenBarden
14   * @since 14.03.2024
15   */
16  @Service
17  @RequiredArgsConstructor
18  @Slf4j
19  public class LogFilterCacheService {
20      private final LogFilterRepository logFilterRepository;
21      private final LogFilterCacheMapper logFilterCacheMapper;
22  
23      @Cacheable("logFilters")
24      public LogFilterCacheContainer getLogFilters(LogFilterType type) {
25          LOGGER.debug("load log filters for {}", type);
26          return new LogFilterCacheContainer(logFilterRepository.getByType(type).stream()
27                  .map(logFilterCacheMapper::entityToCache)
28                  .toList());
29      }
30  }
