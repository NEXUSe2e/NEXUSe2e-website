21  public final class LogQueryDelegate {
22      private LogQueryDelegate() {}
23  
24      private static void filterStringContains(BooleanBuilder builder, String filter, StringPath stringPath) {
25          if (StringUtils.isNotBlank(filter)) {
26              builder.and(stringPath.contains(filter));
27          }
28      }
29  
30      private static void filterStringEquals(BooleanBuilder builder, String filter, StringPath stringPath) {
31          if (StringUtils.isNotBlank(filter)) {
32              builder.and(stringPath.eq(filter));
33          }
34      }
35  
36      private static void filterStringIn(BooleanBuilder builder, Set<String> filters, StringPath stringPath) {
37          if (CollectionUtils.isNotEmpty(filters)) {
38              builder.and(stringPath.in(filters));
39          }
40      }
41  
42      @QueryDelegate(LogEntity.class)
43      public static Predicate filter(QLogEntity logEntity, @Nullable LogFilter logFilter, String timeZone) {
44          BooleanBuilder builder = new BooleanBuilder();
45  
46          if (logFilter != null) {
47              filterStringContains(builder, logFilter.getLogger(), logEntity.logger);
48              filterStringEquals(builder, logFilter.getChoreographyId(), logEntity.choreographyId);
49              filterStringEquals(builder, logFilter.getActionId(), logEntity.actionId);
50              filterStringEquals(builder, logFilter.getConversationId(), logEntity.conversationId);
51              filterStringEquals(builder, logFilter.getMessageId(), logEntity.messageId);
52              filterStringEquals(builder, logFilter.getMdcId(), logEntity.mdcId);
53              filterStringContains(builder, logFilter.getMessage(), logEntity.message);
54              filterStringIn(builder, logFilter.getLevels(), logEntity.level);
55              if (logFilter.getType() != null && logFilter.getType() != FilterType.ALL) {
56                  switch (logFilter.getType()) {
57                      case WITH_MESSAGE_CONTEXT -> builder.and(logEntity.conversationId.isNotNull());
58                      case WITHOUT_MESSAGE_CONTEXT -> builder.and(logEntity.conversationId.isNull());
59                      default -> {
60                          // do nothing, hello spotbugs
61                      }
62                  }
63              }
64              Predicate predicate = DateRangeFilterUtils.filter(
65                      logEntity.created,
66                      timeZone,
67                      logFilter.getCreatedRange(),
68                      logFilter.getCreatedFrom(),
69                      logFilter.getCreatedTill());
70              if (predicate != null) {
71                  builder.and(predicate);
72              }
73          }
74  
75          return builder.getValue();
76      }
77  }
