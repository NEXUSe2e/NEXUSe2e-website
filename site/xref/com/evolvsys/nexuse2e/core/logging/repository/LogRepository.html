11  import org.springframework.data.jpa.repository.Query;
12  import org.springframework.data.querydsl.QuerydslPredicateExecutor;
13  import org.springframework.lang.Nullable;
14  import org.springframework.transaction.annotation.Transactional;
15  
16  /**
17   * @author SvenBarden
18   * @since 04.01.2024
19   */
20  public interface LogRepository extends JpaRepository<LogEntity, Long>, QuerydslPredicateExecutor<LogEntity> {
21  
22      default Page<LogEntity> findAllToRead(
23              Pageable pageable, @Nullable LogFilter filter, String timeZone, BooleanExpression securityPredicate) {
24          return findAll(securityPredicate.and(QLogEntity.logEntity.filter(filter, timeZone)), pageable);
25      }
26  
27      @Transactional
28      @Modifying
29      @Query(
30              "update LogEntity l set l.messageId = :messageId where l.mdcId = :mdcId and l.inMessageContext = true and (l.messageId is null or l.messageId != :messageId)")
31      int updateMessageIdByMdcId(String messageId, String mdcId);
32  
33      @Transactional
34      @Modifying
35      @Query(
36              "update LogEntity l set l.messageId = NULL, l.conversationId = NULL where l.mdcId = :mdcId and l.messageId = :messageId")
37      int removeMessageIdByMdcId(String mdcId, String messageId);
38  }
