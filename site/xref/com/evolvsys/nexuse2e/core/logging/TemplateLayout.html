3   import ch.qos.logback.classic.spi.ILoggingEvent;
4   import ch.qos.logback.classic.spi.ThrowableProxyUtil;
5   import ch.qos.logback.core.LayoutBase;
6   import lombok.RequiredArgsConstructor;
7   import org.thymeleaf.context.Context;
8   import org.thymeleaf.spring6.SpringTemplateEngine;
9   
10  /**
11   * @author SvenBarden
12   * @since 19.03.2024
13   */
14  @RequiredArgsConstructor
15  public class TemplateLayout extends LayoutBase<ILoggingEvent> {
16      private final SpringTemplateEngine springTemplateEngine;
17  
18      @Override
19      public String getContentType() {
20          return "text/html";
21      }
22  
23      @Override
24      public String doLayout(ILoggingEvent event) {
25          Context context = new Context();
26  
27          String layout;
28          try {
29              if (event instanceof LoggingEventWrapper loggingEventWrapper) {
30                  context.setVariable("messageContext", loggingEventWrapper.getMessageContext());
31              }
32              if (event.getThrowableProxy() != null) {
33                  context.setVariable("stacktrace", ThrowableProxyUtil.asString(event.getThrowableProxy()));
34              }
35              context.setVariable("logMessage", event.getFormattedMessage());
36              layout = springTemplateEngine.process("error", context);
37          } catch (Throwable e) {
38              layout = "Could not generate email template: " + e.getMessage();
39          }
40          return layout;
41      }
42  }
