16  @Mapper(config = DefaultMapperConfiguration.class)
17  public abstract class LogMapper extends BaseDtoEntityMapper<LogDto, LogEntity> {
18      private LogRepository logRepository;
19  
20      @Autowired
21      protected void config(LogRepository logRepository) {
22          this.logRepository = logRepository;
23      }
24  
25      @Override
26      public LogEntity idToEntity(Long id, InterceptingMappingContext interceptingMappingContext) {
27          return idToEntity(id, interceptingMappingContext, logRepository::findById);
28      }
29  
30      @Override
31      @InheritConfiguration
32      protected abstract LogDto entityToDto(
33              LogEntity entity, @Context InterceptingMappingContext interceptingMappingContext);
34  
35      @Mapping(target = "lastUpdated", ignore = true)
36      @Mapping(target = "created", ignore = true)
37      @Mapping(target = "inMessageContext", ignore = true)
38      @Override
39      @InheritConfiguration
40      protected abstract void updateEntity(
41              LogDto dto,
42              @MappingTarget LogEntity entity,
43              @Context InterceptingMappingContext interceptingMappingContext);
44  }
