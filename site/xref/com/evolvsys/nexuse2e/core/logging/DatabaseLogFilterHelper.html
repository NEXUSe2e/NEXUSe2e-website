18  /**
19   * @author SvenBarden
20   * @since 19.03.2024
21   */
22  @RequiredArgsConstructor
23  @Slf4j
24  public class DatabaseLogFilterHelper {
25      private final LogFilterCacheService logFilterCacheService;
26      private final Appender<ILoggingEvent> appender;
27      private final LogFilterType logFilterType;
28      private final Filter<ILoggingEvent> defaultFilter;
29  
30      private int filterHash = -1;
31      private final Lock filterLock = new ReentrantLock();
32  
33      private DatabaseLogFilter createFilter(LogFilterCache logFilter) {
34          DatabaseLogFilter databaseLogFilter = new DatabaseLogFilter();
35          databaseLogFilter.setLogger(logFilter.getLogger());
36          databaseLogFilter.setLevel(Level.toLevel(logFilter.getLevel().name()));
37          databaseLogFilter.setMessageStatuses(new ArrayList<>(ListUtils.emptyIfNull(logFilter.getMessageStatuses())));
38          databaseLogFilter.setErrorPattern(logFilter.getErrorPattern());
39          databaseLogFilter.start();
40          return databaseLogFilter;
41      }
42  
43      @SuppressFBWarnings(value = "USFW_UNSYNCHRONIZED_SINGLETON_FIELD_WRITES", justification = "we use a lock")
44      public void updateFilters() {
45          LogFilterCacheContainer logFilters = logFilterCacheService.getLogFilters(logFilterType);
46          int newHash = logFilters.hashCode();
47          if (newHash != filterHash) {
48              if (filterLock.tryLock()) {
49                  try {
50                      filterHash = logFilters.hashCode();
51                      appender.clearAllFilters();
52                      for (LogFilterCache logFilter : logFilters.getLogFilters()) {
53                          try {
54                              appender.addFilter(createFilter(logFilter));
55                          } catch (Throwable e) {
56                              LOGGER.error("could not add custom filter {}", logFilter, e);
57                          }
58                      }
59                      appender.addFilter(defaultFilter);
60                  } finally {
61                      filterLock.unlock();
62                  }
63              } else {
64                  // filter update is already running in another thread
65              }
66          }
67      }
68  }
