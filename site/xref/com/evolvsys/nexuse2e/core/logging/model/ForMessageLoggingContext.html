11  @Value
12  @Builder
13  public class ForMessageLoggingContext {
14      public static final String KEY_NAME = "ForMessageLoggingContext";
15      String messageId;
16      String conversationId;
17      String choreographyId;
18      String actionId;
19  
20      public static ForMessageLoggingContext forEntity(MessageEntity messageEntity) {
21          return ForMessageLoggingContext.builder()
22                  .messageId(messageEntity.getMessageId())
23                  .conversationId(messageEntity.getConversation().getConversationId())
24                  .choreographyId(
25                          messageEntity.getConversation().getChoreography().getChoreographyId())
26                  .actionId(messageEntity.getConversation().getCurrentAction().getActionId())
27                  .build();
28      }
29  }
