3   import com.evolvsys.nexuse2e.core.config.NEXUSe2eConfigurationProperties;
4   import java.security.Security;
5   import java.util.Arrays;
6   import java.util.Set;
7   import java.util.stream.Collectors;
8   import lombok.RequiredArgsConstructor;
9   import lombok.extern.slf4j.Slf4j;
10  import org.springframework.beans.factory.InitializingBean;
11  import org.springframework.stereotype.Component;
12  
13  /**
14   * @author SvenBarden
15   * @since 29.02.2024
16   */
17  @Component
18  @RequiredArgsConstructor
19  @Slf4j
20  class SystemInitializer implements InitializingBean {
21      private final NEXUSe2eConfigurationProperties configurationProperties;
22  
23      @Override
24      public void afterPropertiesSet() throws Exception {
25          if (configurationProperties.isEnableTLSv1() || configurationProperties.isEnableTLSv1_1()) {
26              String defaultProperties = Security.getProperty("jdk.tls.disabledAlgorithms");
27  
28              Set<String> values =
29                      Arrays.stream(defaultProperties.split("\\s*,\\s*")).collect(Collectors.toSet());
30  
31              if (configurationProperties.isEnableTLSv1()) {
32                  values.remove("TLSv1");
33              }
34              if (configurationProperties.isEnableTLSv1_1()) {
35                  values.remove("TLSv1.1");
36              }
37  
38              String newValue = String.join(", ", values);
39  
40              LOGGER.info("changed disabledAlgorithms for non standalone tomcat {} to {}", defaultProperties, newValue);
41  
42              Security.setProperty("jdk.tls.disabledAlgorithms", newValue);
43          }
44      }
45  }
