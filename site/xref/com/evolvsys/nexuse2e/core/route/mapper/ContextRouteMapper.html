14   * @author SvenBarden

17  @Mapper(config = DefaultMapperConfiguration.class)
18  public abstract class ContextRouteMapper {
19      private CertificateContainerKeyMapper certificateContainerKeyMapper;
20  
21      @Autowired
22      protected void config(CertificateContainerKeyMapper certificateContainerKeyMapper) {
23          this.certificateContainerKeyMapper = certificateContainerKeyMapper;
24      }
25  
26      @Mapping(target = "sourcePartnerTrustCertificates", ignore = true)
27      @Mapping(target = "targetPartnerTrustCertificates", ignore = true)
28      @Mapping(target = "sourceConnectionTrustCertificates", ignore = true)
29      @Mapping(target = "targetConnectionTrustCertificates", ignore = true)
30      @Mapping(target = "targetLocalKey", ignore = true)
31      @Mapping(target = "targetConnection", ignore = true)
32      @Mapping(target = "sourceLocalKey", ignore = true)
33      @Mapping(target = "sourceConnection", ignore = true)
34      @Mapping(target = "inflowPipeline", ignore = true)
35      public abstract ContextRoute entityToContext(RouteEntity routeEntity);
36  
37      protected Lazy<ContextKeyStore> containerIdToLazyKeystore(Long id) {
38          return Lazy.of(() -> certificateContainerKeyMapper.idToKeyStore(id));
39      }
40  }
