17  public final class RouteQueryDelegate {
18      private RouteQueryDelegate() {}
19  
20      @QueryDelegate(RouteEntity.class)
21      public static Predicate forPartner(QRouteEntity routeEntity, String firstPartnerId, String secondPartnerId) {
22          return Expressions.anyOf(
23                  Expressions.allOf(
24                          routeEntity.firstPartner.partnerId.eq(firstPartnerId),
25                          routeEntity.secondPartner.partnerId.eq(secondPartnerId)),
26                  Expressions.allOf(
27                          routeEntity.firstPartner.partnerId.eq(secondPartnerId),
28                          routeEntity.secondPartner.partnerId.eq(firstPartnerId)));
29      }
30  
31      @QueryDelegate(RouteEntity.class)
32      public static Predicate filter(QRouteEntity qRouteEntity, @Nullable RouteFilter routeFilter) {
33          BooleanBuilder builder = new BooleanBuilder();
34          if (routeFilter != null) {
35              if (StringUtils.isNotEmpty(routeFilter.getRouteId())) {
36                  builder.and(qRouteEntity.routeId.contains(routeFilter.getRouteId()));
37              }
38              if (StringUtils.isNotEmpty(routeFilter.getDescription())) {
39                  builder.and(qRouteEntity.description.contains(routeFilter.getDescription()));
40              }
41              if (StringUtils.isNotEmpty(routeFilter.getPartnerId())) {
42                  builder.and(Expressions.anyOf(
43                          qRouteEntity.firstPartner.partnerId.contains(routeFilter.getPartnerId()),
44                          qRouteEntity.secondPartner.partnerId.contains(routeFilter.getPartnerId())));
45              }
46          }
47          return builder.getValue();
48      }
49  }
