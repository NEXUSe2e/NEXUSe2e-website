12  import java.util.List;

14  import org.springframework.lang.Nullable;
15  import org.springframework.stereotype.Service;
16  
17  /**
18   * @author Florian Drees
19   * @since 29.12.2023
20   */
21  @Service
22  @Slf4j
23  public class RouteService extends BaseService<RouteDto, RouteEntity, RouteRepository> {
24      private final ListService listService;
25  
26      public RouteService(
27              RouteMapper mapper,
28              RouteRepository repository,
29              CurrentUserService currentUserService,
30              ListService listService) {
31          super(mapper, repository, currentUserService);
32          this.listService = listService;
33      }
34  
35      @Override
36      protected RouteEntity createEntity() {
37          return new RouteEntity();
38      }
39  
40      public List<RouteDto> list(@Nullable RouteFilter routeFilter) {
41          return listService.list(routeFilter, RouteDto.class, repository::findAll).stream()
42                  .sorted(Comparator.comparing(RouteEntity::getRouteId))
43                  .map(this::mapEntityToDto)
44                  .toList();
45      }
46  }
