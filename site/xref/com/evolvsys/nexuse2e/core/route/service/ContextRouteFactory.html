36  @Component
37  @RequiredArgsConstructor
38  public class ContextRouteFactory implements ContextObjectFactory<RouteKey, ContextRoute> {
39      private final RouteRepository routeRepository;
40      private final ContextConnectionMapper contextConnectionMapper;
41      private final CertificateContainerKeyMapper certificateContainerKeyMapper;
42      private final PipelineFactory pipelineFactory;
43      private final CertificateContainerRepository certificateContainerRepository;
44  
45      private Long getSaveId(IdEntity idEntity) {
46          return idEntity == null ? null : idEntity.getId();
47      }
48  
49      private Long getContainerIdFromConnection(PartnerConnectionEntity partnerConnectionEntity) {
50          return partnerConnectionEntity == null || partnerConnectionEntity.getCertificateContainer() == null
51                  ? null
52                  : partnerConnectionEntity.getCertificateContainer().getId();
53      }
54  
55      private List<Long> getAllContainerIdsFromPartner(PartnerEntity partnerEntity) {
56          return partnerEntity == null
57                  ? List.of()
58                  : certificateContainerRepository.findByPartner(partnerEntity).stream()
59                          .map(CertificateContainerEntity::getId)
60                          .toList();
61      }
62  
63      private List<X509Certificate> containerIdsToCerts(Collection<Long> containerIds) {
64          return StreamEx.of(containerIds)
65                  .map(certificateContainerKeyMapper::idToCertificates)
66                  .toFlatList(Function.identity());
67      }
68  
69      private Pipeline getPipeline(PipelineEntity pipelineEntity) {
70          return pipelineEntity == null ? null : pipelineFactory.get(pipelineEntity);
71      }
72  
73      private ContextRoute createContextRoute(
74              RouteEntity routeEntity,
75              Function<RouteEntity, PartnerEntity> sourcePartner,
76              Function<RouteEntity, PartnerEntity> targetPartner,
77              Function<RouteEntity, PartnerConnectionEntity> sourceConnection,
78              Function<RouteEntity, PartnerConnectionEntity> targetConnection,
79              Function<RouteEntity, CertificateContainerEntity> sourceKeyContainer,
80              Function<RouteEntity, CertificateContainerEntity> targetKeyContainer,
81              Function<RouteEntity, PipelineEntity> inflowPipeline) {
82  
83          return ContextRoute.builder()
84                  .id(routeEntity.getId())
85                  .routeId(routeEntity.getRouteId())
86                  .sourceConnection(contextConnectionMapper.entityToConnection(sourceConnection.apply(routeEntity)))
87                  .sourceLocalKey(Lazy.of(() ->
88                          certificateContainerKeyMapper.idToKeyStore(getSaveId(sourceKeyContainer.apply(routeEntity)))))
89                  .sourceConnectionTrustCertificates(Lazy.of(() -> certificateContainerKeyMapper.idToCertificates(
90                          getContainerIdFromConnection(sourceConnection.apply(routeEntity)))))
91                  .sourcePartnerTrustCertificates(Lazy.of(
92                          () -> containerIdsToCerts(getAllContainerIdsFromPartner(sourcePartner.apply(routeEntity)))))
93                  .targetConnection(contextConnectionMapper.entityToConnection(targetConnection.apply(routeEntity)))
94                  .targetLocalKey(Lazy.of(() ->
95                          certificateContainerKeyMapper.idToKeyStore(getSaveId(targetKeyContainer.apply(routeEntity)))))
96                  .targetConnectionTrustCertificates(Lazy.of(() -> certificateContainerKeyMapper.idToCertificates(
97                          getContainerIdFromConnection(targetConnection.apply(routeEntity)))))
98                  .targetPartnerTrustCertificates(Lazy.of(
99                          () -> containerIdsToCerts(getAllContainerIdsFromPartner(targetPartner.apply(routeEntity)))))
100                 .allowSamePartnerType(routeEntity.isAllowSamePartnerType())
101                 .inflowPipeline(getPipeline(inflowPipeline.apply(routeEntity)))
102                 .build();
103     }
104 
105     private ContextRoute toRoute(String sourcePartner, RouteEntity routeEntity) {
106         ContextRoute contextRoute;
107         if (routeEntity.getFirstPartner().getPartnerId().equals(sourcePartner)) {
108             contextRoute = createContextRoute(
109                     routeEntity,
110                     RouteEntity::getFirstPartner,
111                     RouteEntity::getSecondPartner,
112                     RouteEntity::getFirstPartnerConnection,
113                     RouteEntity::getSecondPartnerConnection,
114                     RouteEntity::getFirstNexusCertificateContainer,
115                     RouteEntity::getSecondNexusCertificateContainer,
116                     RouteEntity::getFirstInflowPipeline);
117         } else if (routeEntity.getSecondPartner().getPartnerId().equals(sourcePartner)) {
118             contextRoute = createContextRoute(
119                     routeEntity,
120                     RouteEntity::getSecondPartner,
121                     RouteEntity::getFirstPartner,
122                     RouteEntity::getSecondPartnerConnection,
123                     RouteEntity::getFirstPartnerConnection,
124                     RouteEntity::getSecondNexusCertificateContainer,
125                     RouteEntity::getFirstNexusCertificateContainer,
126                     RouteEntity::getSecondInflowPipeline);
127         } else {
128             throw new NexusRuntimeException(NexusError.builder()
129                     .message("sourcePartner is neither first nor second partner in route")
130                     .extension("routeId", routeEntity.getId())
131                     .extension("sourcePartner", sourcePartner)
132                     .extension("firstPartnerId", routeEntity.getFirstPartner().getPartnerId())
133                     .extension("secondPartnerId", routeEntity.getSecondPartner().getPartnerId())
134                     .build());
135         }
136         return contextRoute;
137     }

139     @Transactional(readOnly = true)
140     public Optional<ContextRoute> getRoute(RouteKey key) {
141         Optional<ContextRoute> result;
142         if (StringUtils.isAnyBlank(key.getSourcePartnerId(), key.getTargetPartnerId())
143                 || key.getChoreographyId() == null
144                 || key.getActionId() == null) {
145             result = Optional.empty();
146         } else {
147             result = routeRepository
148                     .findForAction(
149                             key.getSourcePartnerId(),
150                             key.getTargetPartnerId(),
151                             key.getChoreographyId(),
152                             key.getActionId())
153                     .or(() -> routeRepository.findForChoreography(
154                             key.getSourcePartnerId(), key.getTargetPartnerId(), key.getChoreographyId()))
155                     .map(routeEntity -> toRoute(key.getSourcePartnerId(), routeEntity));
156         }
157         return result;
158     }
159 
160     @Override
161     @Transactional(readOnly = true)
162     public ContextRoute create(RouteKey key) {
163         ContextRoute route;
164         if (StringUtils.isNoneBlank(key.getSourcePartnerId(), key.getTargetPartnerId())
165                 || key.getChoreographyId() == null
166                 || key.getActionId() == null) {
167             route = getRoute(key).orElse(null);
168         } else {
169             route = null;
170         }
171         return route;
172     }
173 
174     @Transactional(readOnly = true)
175     public ContextRoute create(Long routeId, String sourcePartnerId) {
176         return routeRepository
177                 .findById(routeId)
178                 .map(routeEntity -> toRoute(sourcePartnerId, routeEntity))
179                 .orElse(null);
180     }
