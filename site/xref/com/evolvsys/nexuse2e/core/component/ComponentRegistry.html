40      private static String getI18n(String bundle, String key, Locale locale) {
41          try {
42              return XMLResourceBundleUtils.getBundle(bundle, ObjectUtils.defaultIfNull(locale, getDefaultLocale()))
43                      .getString(key);
44          } catch (MissingResourceException e) {
45              LOGGER.debug("could not find resource key for {} in bundle {}", key, bundle);
46              return "";
47          }
48      }
49  
50      private String getI18nText(AnnotationParameterValueList parameterValues, String componentName, String baseName) {
51  
52          String text = Optional.ofNullable(parameterValues.getValue(baseName))
53                  .map(Object::toString)
54                  .orElse(null);
55  
56          if (StringUtils.isBlank(text)) {
57              String key = parameterValues.getValue(baseName + "Key").toString();
58  
59              if (StringUtils.isBlank(key)) {
60                  key = String.format("component.%s.%s", componentName, baseName);
61              }
62  
63              String bundle = StringUtils.defaultIfBlank(
64                      parameterValues.getValue("bundle").toString(), getDefaultBundle());
65  
66              text = getI18n(bundle, key, getDefaultLocale());
67          }
68  
69          return text;
70      }

72      public Optional<Class<?>> get(String componentId) {
73          return Optional.ofNullable(componentMap.get(componentId)).map(RegisteredNexusComponent::getLoadedClass);
74      }
75  
76      public Optional<RegisteredNexusComponent> getComponent(String componentId) {
77          return Optional.ofNullable(componentMap.get(componentId));
78      }
79  
80      public void register(RegisteredNexusComponent nexusComponent) {
81          Validate.notNull(nexusComponent.getComponentId(), "id must not be null");
82          if (componentMap.containsKey(nexusComponent.getComponentId())) {
83              throw new IllegalArgumentException(
84                      "component with id " + nexusComponent.getComponentId() + " is already registered");
85          }
86          LOGGER.debug(
87                  "registering component {} of type {}",
88                  nexusComponent.getComponentId(),
89                  nexusComponent.getComponentType());
90          componentMap.put(nexusComponent.getComponentId(), nexusComponent);
91      }
92  
93      public List<RegisteredNexusComponent> getRegisteredComponents() {
94          return componentMap.values().stream().toList();
95      }
96  
97      public List<RegisteredNexusComponent> getRegisteredComponents(ComponentType componentType) {
98          return componentMap.values().stream()
99                  .filter(component -> component.getComponentType() == componentType)
100                 .toList();
101     }
102 
103     public boolean exists(String componentName) {
104         return componentMap.keySet().stream()
105                 .anyMatch(registeredComponentName -> registeredComponentName.equals(componentName));
106     }

108     @Override
109     public void afterPropertiesSet() throws Exception {
110         LOGGER.debug("scanning for components");
111         Set<String> additionalComponentPackages = nexuSe2eConfigurationProperties.getAdditionalComponentPackages();
112         Set<String> componentPackages = new HashSet<>();
113         componentPackages.add("com.evolvsys.nexuse2e");
114         if (additionalComponentPackages != null) {
115             componentPackages.addAll(additionalComponentPackages);
116         }
117         try (ScanResult scanResult = new ClassGraph()
118                 .enableAllInfo()
119                 .acceptPackages(componentPackages.toArray(new String[0]))
120                 .scan()) {
121             ClassInfoList classInfos = scanResult.getClassesWithAnnotation(NexusComponent.class);
122             for (ClassInfo classInfo : classInfos) {
123 
124                 String className = classInfo.getName();
125 
126                 if (classInfo.isAbstract()) {
127                     LOGGER.trace("skip abstract class {}", className);
128                     continue;
129                 }
130 
131                 AnnotationParameterValueList parameterValues =
132                         classInfo.getAnnotationInfo(NexusComponent.class).getParameterValues();
133 
134                 String componentId = parameterValues.getValue("value").toString();
135                 if ("".equals(componentId)) {
136                     componentId = classInfo.getSimpleName();
137                 }
138 
139                 Optional<ComponentType> componentType = getComponentType(classInfo);
140                 if (componentType.isEmpty()) {
141                     LOGGER.error("Unable to resolve type for component of class '{}'", classInfo.getSimpleName());
142                     continue;
143                 }
144 
145                 LOGGER.trace("identified component {} as {}", componentId, componentType.orElseThrow());
146 
147                 String description = getI18nText(parameterValues, componentId, "description");
148                 String helpText = getI18nText(parameterValues, componentId, "helpText");
149 
150                 Class<?> loadedClass = classInfo.loadClass();
151 
152                 RegisteredNexusComponent.RegisteredNexusComponentBuilder registeredNexusComponentBuilder =
153                         RegisteredNexusComponent.builder()
154                                 .componentId(componentId)
155                                 .componentType(componentType.orElseThrow())
156                                 .loadedClass(loadedClass)
157                                 .description(description)
158                                 .helpText(helpText);
159 
160                 if (Configurable.class.isAssignableFrom(loadedClass)) {
161                     try {
162                         Configurable configurable = (Configurable)
163                                 classInfo.loadClass().getConstructor().newInstance();
164 
165                         registeredNexusComponentBuilder.parameterDefinitions(ObjectUtils.defaultIfNull(
166                                 configurable.parameterDefinitions(), Collections.emptyList()));
167                     } catch (NoSuchMethodException e) {
168                         LOGGER.error(
169                                 "Class '{}' could not be instantiated as a Configurable", loadedClass.getSimpleName());
170                     }
171                 }
172 
173                 RegisteredNexusComponent registeredNexusComponent = registeredNexusComponentBuilder.build();
174 
175                 LOGGER.trace("scanned component {}", registeredNexusComponent);
176 
177                 register(registeredNexusComponent);
178             }
179         }
180         LOGGER.debug("finished component scanning");
181     }

183     private Optional<ComponentType> getComponentType(ClassInfo classInfo) {
184         if (classInfo.implementsInterface(InflowService.class)) {
185             return Optional.of(ComponentType.INFLOW_SERVICE);
186         }
187         if (classInfo.implementsInterface(OutflowService.class)) {
188             return Optional.of(ComponentType.OUTFLOW_SERVICE);
189         }
190         if (classInfo.extendsSuperclass(Pipelet.class)) {
191             return Optional.of(ComponentType.PIPELET);
192         }
193         if (classInfo.implementsInterface(SupportService.class)) {
194             return Optional.of(ComponentType.SUPPORT_SERVICE);
195         }
196         return Optional.empty();
197     }
