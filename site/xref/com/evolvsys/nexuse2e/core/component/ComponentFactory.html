28  @RequiredArgsConstructor
29  @Component
30  @Slf4j
31  public class ComponentFactory implements BeanFactoryAware {
32      private final ComponentRegistry componentRegistry;
33      private final ParameterConverterRegistry parameterConverterUtils;
34      private final ComponentPostCreationService componentPostCreationService;
35      private BeanFactory beanFactory;
36  
37      @SuppressWarnings("unchecked")
38      public <T> T create(String name, @Nullable Collection<ParameterEntity> parameterEntities) {
39          Class<T> componentType = (Class<T>) getComponentType(name);
40  
41          return create(componentType, parameterEntities);
42      }
43  
44      private Class<?> getComponentType(String name) {
45          return componentRegistry
46                  .get(name)
47                  .orElseThrow(() -> new IllegalArgumentException("no component with name " + name + " found"));
48      }
49  
50      @SuppressFBWarnings("EXS_EXCEPTION_SOFTENING_NO_CONSTRAINTS")
51      private <T> T create(Class<T> type, @Nullable Collection<ParameterEntity> parameterEntities) {
52          T component;
53          try {
54              component = type.getConstructor().newInstance();
55              if (component instanceof Configurable configurable) {
56                  configurable.setParameters(buildParameters(
57                          configurable, ObjectUtils.defaultIfNull(parameterEntities, Collections.emptyList())));
58              }
59              componentPostCreationService.processComponentPostCreation(component);
60          } catch (InstantiationException
61                  | IllegalAccessException
62                  | InvocationTargetException
63                  | NoSuchMethodException e) {
64              throw new ComponentCreationException(type, e);
65          }
66          return component;
67      }
68  
69      private Object getParameterValue(ParameterDefinition parameterDefinition, ParameterEntity parameterEntity) {
70          if (ParameterType.SPRING_BEAN == parameterDefinition.getType()) {
71              return beanFactory.getBean(parameterEntity.getStringValue());
72          } else {
73              return parameterConverterUtils
74                      .getTypeConverter(parameterDefinition.getType())
75                      .toContextValue(parameterEntity);
76          }
77      }
78  
79      public Parameters buildParameters(Configurable configurable, Collection<ParameterEntity> parameterEntities) {
80          return buildParameters(configurable, parameterEntities, true);
81      }
82  
83      public Parameters buildParameters(
84              Configurable configurable, Collection<ParameterEntity> parameterEntities, boolean addDefault) {
85          Parameters parameters = new Parameters();
86  
87          if (CollectionUtils.isNotEmpty(configurable.parameterDefinitions())) {
88              for (ParameterDefinition parameterDefinition : configurable.parameterDefinitions()) {
89                  Optional<ParameterEntity> optionalEntity = parameterEntities.stream()
90                          .filter(parameterEntity -> parameterEntity.getName().equals(parameterDefinition.getName()))
91                          .findFirst();
92  
93                  if (optionalEntity.isPresent()) {
94                      Object value = getParameterValue(parameterDefinition, optionalEntity.orElseThrow());
95  
96                      LOGGER.trace(
97                              "add parameter {} with value {} to component {}",
98                              parameterDefinition.getName(),
99                              value,
100                             configurable);
101                     parameters.put(parameterDefinition.getName(), value);
102                 } else if (addDefault) {
103                     Object value = parameterDefinition.getDefaultValue();
104 
105                     LOGGER.trace(
106                             "add default parameter {} with value {} to component {}",
107                             parameterDefinition.getName(),
108                             value,
109                             configurable);
110                     parameters.put(parameterDefinition.getName(), value);
111                 }
112             }
113         }
114 
115         return parameters;
116     }
117 
118     @Override
119     public void setBeanFactory(BeanFactory beanFactory) {
120         synchronized (this) {
121             this.beanFactory = beanFactory;
122         }
123     }
124 }
