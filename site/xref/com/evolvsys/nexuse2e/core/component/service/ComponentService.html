26  @Service
27  @RequiredArgsConstructor
28  @Slf4j
29  public class ComponentService {
30      private final ComponentRegistry componentRegistry;
31      private final ComponentMapper componentMapper;
32      private final ParameterMapper parameterMapper;
33      private final CurrentUserService currentUserService;
34  
35      public List<ComponentDto> list(@Nullable ComponentFilter componentFilter) {
36          return currentUserService.hasAnySystemPermission(BaseConfigAccessInterceptor.READ_CONFIG)
37                  ? componentRegistry.getRegisteredComponents().stream()
38                          .filter(c -> ComponentFilterUtils.filter(c, componentFilter))
39                          .sorted(Comparator.comparing(RegisteredNexusComponent::getComponentId))
40                          .map(componentMapper::componentToDto)
41                          .toList()
42                  : Collections.emptyList();
43      }
44  
45      public List<ParameterDto> getParameterDefinitions(@NonNull String component) {
46          return componentRegistry
47                  .getComponent(component)
48                  .map(registeredComponent -> CollectionUtils.emptyIfNull(registeredComponent.getParameterDefinitions()))
49                  .orElse(Collections.emptyList())
50                  .stream()
51                  .map(definition -> parameterMapper.definitionToDto(definition, currentUserService.getCurrentUserData()))
52                  .toList();
53      }
54  }
