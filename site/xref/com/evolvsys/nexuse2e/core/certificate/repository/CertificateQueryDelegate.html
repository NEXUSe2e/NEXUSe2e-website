21  public final class CertificateQueryDelegate {
22      private CertificateQueryDelegate() {}
23  
24      @QueryDelegate(CertificateEntity.class)
25      public static Predicate filter(QCertificateEntity certificateEntity, CertificateFilter certificateFilter) {
26          BooleanBuilder builder = new BooleanBuilder();
27  
28          if (certificateFilter != null) {
29              filterInternal(certificateEntity, certificateFilter, builder);
30          }
31          return builder.getValue();
32      }
33  
34      private static void filterInternal(
35              QCertificateEntity certificateEntity, CertificateFilter certificateFilter, BooleanBuilder builder) {
36          QCertificateInfo certificateInfo = QCertificateEntity.certificateEntity.info;
37  
38          if (certificateFilter.getValid() != null) {
39              ZonedDateTime now = ZonedDateTime.now();
40              if (certificateFilter.getValid() == BooleanFilterType.TRUE) {
41                  builder.and(certificateInfo.validFrom.before(now).and(certificateInfo.validUntil.after(now)));
42              } else if (certificateFilter.getValid() == BooleanFilterType.FALSE) {
43                  builder.and(certificateInfo.validFrom.after(now).or(certificateInfo.validUntil.before(now)));
44              }
45          }
46          if (CollectionUtils.isNotEmpty(certificateFilter.getTypes())) {
47              builder.and(certificateEntity.type.in(certificateFilter.getTypes()));
48          }
49          if (certificateFilter.getActive() == BooleanFilterType.TRUE) {
50              builder.and(certificateEntity.active.isTrue());
51          } else if (certificateFilter.getActive() == BooleanFilterType.FALSE) {
52              builder.and(certificateEntity.active.isFalse());
53          }
54  
55          filterStringValue(builder, certificateInfo.sha1Fingerprint, certificateFilter.getSha1Fingerprint());
56          filterStringValue(builder, certificateInfo.sha256Fingerprint, certificateFilter.getSha256Fingerprint());
57          filterStringValue(builder, certificateInfo.commonName, certificateFilter.getCommonName());
58          filterStringValue(builder, certificateInfo.organisation, certificateFilter.getOrganisation());
59          filterStringValue(builder, certificateInfo.organisationUnit, certificateFilter.getOrganisationUnit());
60          filterStringValue(builder, certificateInfo.country, certificateFilter.getCountry());
61          filterStringValue(builder, certificateInfo.state, certificateFilter.getState());
62          filterStringValue(builder, certificateInfo.location, certificateFilter.getLocation());
63          filterStringValue(builder, certificateInfo.email, certificateFilter.getEmail());
64          filterStringValue(builder, certificateInfo.issuer, certificateFilter.getIssuer());
65  
66          if (certificateFilter.getSelfSigned() != null) {
67              Predicate selfSignedPredicate =
68                      BooleanQueryDelegateUtils.filter(certificateInfo.selfSigned, certificateFilter.getSelfSigned());
69              if (selfSignedPredicate != null) {
70                  builder.and(selfSignedPredicate);
71              }
72          }
73      }
74  
75      private static void filterStringValue(BooleanBuilder builder, StringPath filterPath, String value) {
76          if (StringUtils.isNotEmpty(value)) {
77              builder.and(filterPath.contains(value));
78          }
79      }
80  }
