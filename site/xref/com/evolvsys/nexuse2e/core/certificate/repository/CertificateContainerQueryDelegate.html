6   import com.querydsl.core.BooleanBuilder;
7   import com.querydsl.core.annotations.QueryDelegate;
8   import com.querydsl.core.types.Predicate;
9   import org.apache.commons.collections4.CollectionUtils;
10  import org.apache.commons.lang3.StringUtils;
11  
12  /**
13   * @author SvenBarden
14   * @since 07.02.2024
15   */
16  public final class CertificateContainerQueryDelegate {
17      private CertificateContainerQueryDelegate() {}
18  
19      @QueryDelegate(CertificateContainerEntity.class)
20      public static Predicate filter(
21              QCertificateContainerEntity certificateContainerEntity,
22              CertificateContainerFilter certificateContainerFilter) {
23          BooleanBuilder builder = new BooleanBuilder();
24          if (certificateContainerFilter != null) {
25              if (certificateContainerFilter.getPartnerId() != null) {
26                  builder.and(certificateContainerEntity.partner.id.eq(certificateContainerFilter.getPartnerId()));
27              }
28              if (StringUtils.isNotBlank(certificateContainerFilter.getName())) {
29                  builder.and(certificateContainerEntity.name.contains(certificateContainerFilter.getName()));
30              }
31              if (CollectionUtils.isNotEmpty(certificateContainerFilter.getTypes())) {
32                  builder.and(certificateContainerEntity.type.in(certificateContainerFilter.getTypes()));
33              }
34          }
35  
36          return builder.getValue();
37      }
38  }
