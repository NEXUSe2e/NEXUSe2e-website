17   * @author Florian Drees

19   */
20  @Component
21  @RequiredArgsConstructor
22  @Slf4j
23  public class CertificateEntityUtils {
24      private final CertificateRepository certificateRepository;
25  
26      public boolean isPrivateCert(long certificateId) {
27          return certificateRepository
28                  .findById(certificateId)
29                  .map(entity -> entity.getType() == CertificateType.LOCAL)
30                  .orElseThrow(() -> new ObjectNotFoundException(CertificateDto.class, certificateId));
31      }
32  
33      @Transactional(readOnly = true)
34      public PrivateKey getPrivateKey(long certificateId) throws NexusException {
35          CertificateEntity certificateEntity = certificateRepository
36                  .findById(certificateId)
37                  .orElseThrow(() -> new ObjectNotFoundException(CertificateDto.class, certificateId));
38          LOGGER.debug(
39                  "Getting private key for certificate with id '{}' and name '{}'",
40                  certificateEntity.getId(),
41                  certificateEntity.getName());
42          if (!isPrivateCert(certificateId)) {
43              throw new NexusException(NexusError.builder()
44                      .message("Unable to load private key as the entity is not a keystore")
45                      .extension("id", certificateEntity.getId())
46                      .extension("name", certificateEntity.getName())
47                      .build());
48          }
49          return CertificateKeystoreUtils.getPrivateKey(
50                  CertificateKeystoreUtils.toKeyStore(certificateEntity), certificateEntity.getPassword());
51      }
52  }
