44      private List<Certificate> getCertificateChain(KeyStore keyStore) throws NexusException {
45          List<Certificate> certificates = new ArrayList<>();
46  
47          if (keyStore != null) {
48              try {
49                  Enumeration<String> aliases = keyStore.aliases();
50  
51                  while (aliases.hasMoreElements() && certificates.isEmpty()) {
52                      String alias = aliases.nextElement();
53                      Certificate[] certificateChain = keyStore.getCertificateChain(alias);
54  
55                      if (ArrayUtils.isNotEmpty(certificateChain)) {
56                          CollectionUtils.addAll(certificates, certificateChain);
57                      }
58                  }
59              } catch (KeyStoreException e) {
60                  throw new NexusException(
61                          NexusError.builder()
62                                  .message("error getting certificate chain from keystore")
63                                  .build(),
64                          e);
65              }
66          }
67  
68          return certificates;
69      }

71      private boolean isSelfSigned(X509Certificate certificate) {
72          X500Principal subjectX500Principal = certificate.getSubjectX500Principal();
73          X500Principal issuerX500Principal = certificate.getIssuerX500Principal();
74  
75          return subjectX500Principal.equals(issuerX500Principal);
76      }

78      private CertificateEntity fromX509(X509Certificate certificate, CertificateType certificateType)
79              throws NexusException {
80          CertificateEntity certificateEntity;
81  
82          try {
83              String id = certificateIdGenerator
84                      .createCertificateId(certificate)
85                      .orElseThrow(() -> new NexusException(NexusError.builder()
86                              .message("could not create id from certificate")
87                              .build()));
88  
89              certificateEntity = new CertificateEntity();
90              certificateEntity.setName(id);
91              certificateEntity.setType(certificateType);
92              certificateEntity.setFile(createFileEntity(id, typeToMimeType(certificateType), certificate.getEncoded()));
93              certificateEntity.setInfo(certificateInfoMapper.certificateToInfo(certificate));
94          } catch (CertificateEncodingException e) {
95              throw new NexusException(
96                      NexusError.builder().message("could not encode certificate").build(), e);
97          }
98  
99          return certificateEntity;
100     }

102     public static String typeToMimeType(CertificateType certificateType) {
103         return switch (certificateType) {
104             case CA, PARTNER -> "application/x-x509-ca-cert";
105             case LOCAL -> "application/x-pkcs12";
106         };
107     }

109     private FileEntity createFileEntity(String name, String mimeType, byte[] content) {
110         String fileName;
111         try {
112             fileName = name
113                     + TikaConfig.getDefaultConfig()
114                             .getMimeRepository()
115                             .forName(mimeType)
116                             .getExtension();
117         } catch (MimeTypeException e) {
118             LOGGER.error("could not get file extension for cert {} with type {}", name, mimeType, e);
119             fileName = name;
120         }
121         FileEntity fileEntity = fileFactory.create(fileName, mimeType, content);
122         fileEntity.setStatus(FileStatus.SAVED);
123         return fileEntity;
124     }

126     private void addOnlyNewCertificate(
127             X509Certificate x509Certificate,
128             Collection<String> installedFingerprints,
129             SequencedCollection<CertificateEntity> certificateEntities)
130             throws CertificateEncodingException, NexusException {
131         String sha1Fingerprint = CertificateUtil.getSha1Fingerprint(x509Certificate);
132 
133         if (installedFingerprints.contains(sha1Fingerprint)) {
134             LOGGER.trace("fingerprint {} is already installed", sha1Fingerprint);
135         } else {
136             CertificateEntity certificateEntity = fromX509(x509Certificate, CertificateType.CA);
137             certificateEntity.setEnabled(ZonedDateTime.now());
138             certificateEntity.setActive(true);
139             certificateEntities.add(certificateEntity);
140         }
141     }

143     public List<CertificateEntity> createMissingCaEntitiesFromCaKeyStore(
144             Collection<String> installedFingerprints, KeyStore keyStore)
145             throws NexusException, CertificateEncodingException {
146         List<CertificateEntity> certificateEntities = new ArrayList<>();
147 
148         try {
149             Enumeration<String> aliases = keyStore.aliases();
150 
151             while (aliases.hasMoreElements()) {
152                 String alias = aliases.nextElement();
153 
154                 X509Certificate x509Certificate = (X509Certificate) keyStore.getCertificate(alias);
155                 addOnlyNewCertificate(x509Certificate, installedFingerprints, certificateEntities);
156             }
157         } catch (KeyStoreException e) {
158             throw new NexusException(
159                     NexusError.builder()
160                             .message("Error loading aliases from keystore")
161                             .build(),
162                     e);
163         }
164 
165         LOGGER.trace("created {} certs from keystore", certificateEntities.size());
166 
167         return certificateEntities;
168     }

170     public List<CertificateEntity> createMissingCaEntities(Collection<String> installedFingerprints, KeyStore keyStore)
171             throws NexusException, CertificateEncodingException {
172         List<CertificateEntity> certificateEntities = new ArrayList<>();
173         List<Certificate> certificateChain = getCertificateChain(keyStore);
174 
175         LOGGER.trace("got {} certs from keystore", certificateChain.size());
176 
177         if (CollectionUtils.isNotEmpty(certificateChain)) {
178             for (int i = 0; i < certificateChain.size(); i++) {
179                 Certificate certificate = certificateChain.get(i);
180                 X509Certificate x509Certificate = (X509Certificate) certificate;
181 
182                 if (i > 0 || isSelfSigned(x509Certificate)) {
183                     addOnlyNewCertificate(x509Certificate, installedFingerprints, certificateEntities);
184                 }
185             }
186         }
187 
188         return certificateEntities;
189     }

191     public CertificateEntity createFromPKCS12(KeyStore keyStore, KeystoreUpload keystoreUpload) throws NexusException {
192         CertificateEntity certificateEntity = new CertificateEntity();
193 
194         PrivateKey privateKey = CertificateKeystoreUtils.getPrivateKey(keyStore, null);
195         List<Certificate> certificateChain = getCertificateChain(keyStore);
196 
197         if (CollectionUtils.isEmpty(certificateChain)) {
198             throw new NexusException(NexusError.builder()
199                     .message("cannot import certificate without certificate chain")
200                     .build());
201         }
202 
203         try {
204             KeyStore newKeystore = KeyStore.getInstance(
205                     CertificateConstants.DEFAULT_KEY_STORE_TYPE, CertificateConstants.DEFAULT_JCE_PROVIDER);
206             newKeystore.load(null, null);
207 
208             String id = certificateIdGenerator
209                     .createCertificateId((X509Certificate) certificateChain.getFirst())
210                     .orElseThrow(() -> new NexusException(NexusError.builder()
211                             .message("could not create id from certificate")
212                             .build()));
213 
214             newKeystore.setKeyEntry(
215                     id,
216                     privateKey,
217                     keystoreUpload.getPassword().toCharArray(),
218                     certificateChain.toArray(new Certificate[0]));
219             ByteArrayOutputStream outputStream = new ByteArrayOutputStream();
220             newKeystore.store(outputStream, keystoreUpload.getPassword().toCharArray());
221 
222             certificateEntity.setName(id);
223             certificateEntity.setType(CertificateType.LOCAL);
224             certificateEntity.setPassword(keystoreUpload.getPassword());
225             certificateEntity.setFile(
226                     createFileEntity(id, typeToMimeType(CertificateType.LOCAL), outputStream.toByteArray()));
227             certificateEntity.setInfo(
228                     certificateInfoMapper.certificateToInfo((X509Certificate) certificateChain.getFirst()));
229 
230         } catch (KeyStoreException
231                 | NoSuchProviderException
232                 | CertificateException
233                 | IOException
234                 | NoSuchAlgorithmException e) {
235             throw new NexusException(
236                     NexusError.builder()
237                             .message("could not create certificateEntity from keystore")
238                             .build(),
239                     e);
240         }
241 
242         return certificateEntity;
243     }
