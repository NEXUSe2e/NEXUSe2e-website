23  @UtilityClass
24  public class DatabaseUtils {
25      public static List<String> enquoteIdentifier(Statement statement, Collection<String> identifiers)
26              throws NexusException {
27          List<String> quotedIdentifiers = new ArrayList<>(identifiers.size());
28          for (String identifier : identifiers) {
29              quotedIdentifiers.add(enquoteIdentifier(statement, identifier));
30          }
31          return quotedIdentifiers;
32      }
33  
34      public static String enquoteIdentifier(@NonNull Statement statement, @NonNull String identifier)
35              throws NexusException {
36          Validate.notNull(identifier, "identifier must not be null");
37          try {
38              return statement.enquoteIdentifier(identifier.toUpperCase(), true);
39          } catch (SQLException e) {
40              throw new NexusException(
41                      NexusError.builder()
42                              .message("Could not enquote identifier")
43                              .extension("identifier", identifier)
44                              .build(),
45                      e);
46          }
47      }
48  
49      private static List<String> toNamedParameters(Collection<String> columnNames) {
50          return columnNames.stream().map(columnName -> ":" + columnName).toList();
51      }
52  
53      public static String buildNamedSqlInsert(
54              NamedParameterJdbcTemplate namedParameterJdbcTemplate, String tableName, Collection<String> columnNames)
55              throws NexusException {
56          DataSource dataSource = Optional.ofNullable(namedParameterJdbcTemplate)
57                  .map(NamedParameterJdbcTemplate::getJdbcTemplate)
58                  .map(JdbcAccessor::getDataSource)
59                  .orElseThrow(() -> new NexusException(NexusError.builder()
60                          .message("Could not get data source from named parameter jdbc template")
61                          .build()));
62          return buildNamedSqlInsert(dataSource, tableName, columnNames);
63      }
64  
65      public static String buildNamedSqlInsert(DataSource dataSource, String tableName, Collection<String> columnNames)
66              throws NexusException {
67          try (Connection connection = dataSource.getConnection();
68                  Statement statement = connection.createStatement()) {
69              return String.format(
70                      "INSERT INTO %s (%s) VALUES (%s)",
71                      enquoteIdentifier(statement, tableName),
72                      String.join(", ", enquoteIdentifier(statement, columnNames)),
73                      String.join(", ", toNamedParameters(columnNames)));
74          } catch (SQLException e) {
75              throw new NexusException(
76                      NexusError.builder()
77                              .message("Could not open database connection")
78                              .build(),
79                      e);
80          }
81      }
82  }
