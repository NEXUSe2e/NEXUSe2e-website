37      @Transactional
38      public PurgeResult purgeLogs(LogFilter logFilter) {
39          long deletedLogs = jpqlQueryFactory
40                  .delete(QLogEntity.logEntity)
41                  .where(QLogEntity.logEntity.filter(
42                          logFilter,
43                          applicationSettingsService.getSystemSettingsOrDefault().getTimeZone()))
44                  .execute();
45          LOGGER.debug("deleted {} filtered logs", deletedLogs);
46  
47          return PurgeResult.builder().deletedLogs(deletedLogs).build();
48      }

50      @Transactional
51      public PurgeResult purgeAllLogs() {
52          long deletedLogs = jpqlQueryFactory.delete(QLogEntity.logEntity).execute();
53          LOGGER.debug("deleted all {} logs", deletedLogs);
54  
55          return PurgeResult.builder().deletedLogs(deletedLogs).build();
56      }

58      @Transactional
59      public PurgeResult purgeAllConversations() {
60          long deletedManualActions = jpqlQueryFactory
61                  .delete(QMessageActionManualActionEntity.messageActionManualActionEntity)
62                  .execute();
63          LOGGER.debug("deleted all {} manual message actions", deletedManualActions);
64  
65          long deletedMessages =
66                  jpqlQueryFactory.delete(QMessageEntity.messageEntity).execute();
67          LOGGER.debug("deleted all {} messages", deletedMessages);
68  
69          // remove payload reference from metaData, so we can safely delete all payloads and do not need two metaData
70          // delete statements
71          long clearedMetaData = jpqlQueryFactory
72                  .update(QMetaDataEntity.metaDataEntity)
73                  .setNull(QMetaDataEntity.metaDataEntity.binaryValue)
74                  .where(QMetaDataEntity.metaDataEntity.binaryValue.isNotNull())
75                  .execute();
76          LOGGER.debug("removed payload reference from {} metaData", clearedMetaData);
77  
78          long deletedPayloads =
79                  jpqlQueryFactory.delete(QPayloadEntity.payloadEntity).execute();
80          LOGGER.debug("deleted all {} payloads", deletedPayloads);
81  
82          long deletedBinaries =
83                  jpqlQueryFactory.delete(QBinaryEntity.binaryEntity).execute();
84          LOGGER.debug("deleted all {} binary contents", deletedBinaries);
85  
86          long deletedMetaDatas =
87                  jpqlQueryFactory.delete(QMetaDataEntity.metaDataEntity).execute();
88          LOGGER.debug("deleted all {} metaDatas", deletedMetaDatas);
89  
90          long deletedConversations =
91                  jpqlQueryFactory.delete(QConversationEntity.conversationEntity).execute();
92          LOGGER.debug("deleted all {} conversations", deletedConversations);
93  
94          long deletedLogs = jpqlQueryFactory.delete(QLogEntity.logEntity).execute();
95          LOGGER.debug("deleted all conversation {} logs", deletedLogs);
96  
97          return PurgeResult.builder()
98                  .deletedMessages(deletedMessages)
99                  .deletedConversations(deletedConversations)
100                 .deletedMetaDatas(deletedMetaDatas)
101                 .deletedPayloads(deletedPayloads)
102                 .deletedBinaries(deletedBinaries)
103                 .deletedLogs(deletedLogs)
104                 .deletedManualActions(deletedManualActions)
105                 .build();
106     }

108     @Transactional
109     public PurgeResult purgeConversations(ConversationFilter conversationFilter) {
110         JPQLSubQuery<ConversationEntity> conversationSelect = JPAExpressions.selectFrom(
111                         QConversationEntity.conversationEntity)
112                 .where(QConversationEntity.conversationEntity.filter(
113                         conversationFilter,
114                         applicationSettingsService.getSystemSettingsOrDefault().getTimeZone()));
115 
116         JPQLSubQuery<PayloadEntity> messagePayloads = JPAExpressions.select(QPayloadEntity.payloadEntity)
117                 .from(QMessageEntity.messageEntity)
118                 .join(QMessageEntity.messageEntity.payloads, QPayloadEntity.payloadEntity);
119 
120         JPQLSubQuery<PayloadEntity> metaDataPayloads = JPAExpressions.select(QMetaDataEntity.metaDataEntity.binaryValue)
121                 .from(QMetaDataEntity.metaDataEntity)
122                 .where(QMetaDataEntity.metaDataEntity.binaryValue.isNotNull());
123 
124         JPQLSubQuery<MetaDataEntity> messageMetaDatas = JPAExpressions.select(QMetaDataEntity.metaDataEntity)
125                 .from(QMessageEntity.messageEntity)
126                 .join(QMessageEntity.messageEntity.metaDatas, QMetaDataEntity.metaDataEntity);
127         JPQLSubQuery<MetaDataEntity> payloadMetaDatas = JPAExpressions.select(QMetaDataEntity.metaDataEntity)
128                 .from(QPayloadEntity.payloadEntity)
129                 .join(QPayloadEntity.payloadEntity.metaDatas, QMetaDataEntity.metaDataEntity);
130 
131         // delete manual actions
132 
133         QMessageActionManualActionEntity mainAction = new QMessageActionManualActionEntity("main");
134 
135         QMessageActionManualActionEntity allActions = new QMessageActionManualActionEntity("all");
136         QMessageEntity allActionsMessages = new QMessageEntity("actions_all");
137         JPQLSubQuery<Long> allSubquery = JPAExpressions.select(allActionsMessages.count())
138                 .from(allActions)
139                 .join(allActions.messages, allActionsMessages)
140                 .where(allActions.eq(mainAction));
141 
142         QMessageActionManualActionEntity filteredActions = new QMessageActionManualActionEntity("filtered");
143         QMessageEntity filteredActionsMessages = new QMessageEntity("actions_filtered");
144         JPQLSubQuery<Long> filteredSubquery = JPAExpressions.select(filteredActionsMessages.count())
145                 .from(filteredActions)
146                 .join(filteredActions.messages, filteredActionsMessages)
147                 .on(filteredActionsMessages.conversation.in(conversationSelect))
148                 .where(filteredActions.eq(mainAction));
149 
150         long deletedManualActions = jpqlQueryFactory
151                 .delete(mainAction)
152                 .where(allSubquery.eq(filteredSubquery))
153                 .execute();
154         LOGGER.debug("deleted {} manual actions", deletedManualActions);
155 
156         // sadly we cannot directly delete entries from the join table with jpa queries, so we need to remove the
157         // references manually
158 
159         List<Tuple> manualActions = jpqlQueryFactory
160                 .select(
161                         QMessageActionManualActionEntity.messageActionManualActionEntity,
162                         QMessageEntity.messageEntity.id)
163                 .from(QMessageActionManualActionEntity.messageActionManualActionEntity)
164                 .join(
165                         QMessageActionManualActionEntity.messageActionManualActionEntity.messages,
166                         QMessageEntity.messageEntity)
167                 .on(QMessageEntity.messageEntity.conversation.in(conversationSelect))
168                 .fetch();
169         for (Tuple tuple : manualActions) {
170             Long messageEntityId = tuple.get(QMessageEntity.messageEntity.id);
171             MessageActionManualActionEntity messageActionManualActionEntity =
172                     tuple.get(QMessageActionManualActionEntity.messageActionManualActionEntity);
173             if (messageActionManualActionEntity != null) {
174                 messageActionManualActionEntity.getMessages().removeIf(message -> message.getId()
175                         .equals(messageEntityId));
176                 LOGGER.info(
177                         "manually removed message {} reference from manual action {}",
178                         messageEntityId,
179                         messageActionManualActionEntity.getId());
180                 entityManager.persist(messageActionManualActionEntity);
181                 // we need do flush the changes to the database, so the following delete statements can be executed
182                 entityManager.flush();
183             }
184         }
185         long deletedManualActionsPartly = manualActions.size();
186         LOGGER.debug("deleted {} manual actions references manually", manualActions.size());
187 
188         // delete messages
189         long deletedMessages = jpqlQueryFactory
190                 .delete(QMessageEntity.messageEntity)
191                 .where(QMessageEntity.messageEntity.conversation.in(conversationSelect))
192                 .execute();
193         LOGGER.debug("deleted {} messages", deletedMessages);
194 
195         // delete metaDatas without ref. Mapping table was removed by hibernate with delete messages statement, so we
196         // need to find metaDatas without reference in the database.
197         long deletedMetaDatas = jpqlQueryFactory
198                 .delete(QMetaDataEntity.metaDataEntity)
199                 .where(
200                         QMetaDataEntity.metaDataEntity.notIn(messageMetaDatas),
201                         QMetaDataEntity.metaDataEntity.notIn(payloadMetaDatas))
202                 .execute();
203         LOGGER.debug("deleted {} metaDatas", deletedMetaDatas);
204 
205         // delete payload without ref. Mapping table was removed by hibernate with delete messages and delete metaData
206         // statements, so we need to find payloads without reference in the database
207         long deletedPayloads = jpqlQueryFactory
208                 .delete(QPayloadEntity.payloadEntity)
209                 .where(
210                         QPayloadEntity.payloadEntity.notIn(messagePayloads),
211                         QPayloadEntity.payloadEntity.notIn(metaDataPayloads))
212                 .execute();
213         LOGGER.debug("deleted {} payloads", deletedPayloads);
214 
215         // delete metaDatas again, because they may be references from payloads
216         long deletedSecondMetaDatas = jpqlQueryFactory
217                 .delete(QMetaDataEntity.metaDataEntity)
218                 .where(
219                         QMetaDataEntity.metaDataEntity.notIn(messageMetaDatas),
220                         QMetaDataEntity.metaDataEntity.notIn(payloadMetaDatas))
221                 .execute();
222         LOGGER.debug("deleted {} metaDatas (second iteration)", deletedSecondMetaDatas);
223 
224         // delete binaries
225         long deletedBinaries = jpqlQueryFactory
226                 .delete(QBinaryEntity.binaryEntity)
227                 .where(QBinaryEntity.binaryEntity.notIn(JPAExpressions.select(QPayloadEntity.payloadEntity.content)
228                         .from(QPayloadEntity.payloadEntity)
229                         .where(QPayloadEntity.payloadEntity.content.isNotNull())))
230                 .execute();
231         LOGGER.debug("deleted {} binary contents", deletedBinaries);
232 
233         // delete conversations
234         long deletedConversations = jpqlQueryFactory
235                 .delete(QConversationEntity.conversationEntity)
236                 .where(QConversationEntity.conversationEntity.filter(
237                         conversationFilter,
238                         applicationSettingsService.getSystemSettingsOrDefault().getTimeZone()))
239                 .execute();
240         LOGGER.debug("deleted {} conversations", deletedConversations);
241 
242         long deletedLogs = jpqlQueryFactory
243                 .delete(QLogEntity.logEntity)
244                 .where(
245                         QLogEntity.logEntity.conversationId.isNotNull(),
246                         QLogEntity.logEntity.conversationId.notIn(
247                                 JPAExpressions.select(QConversationEntity.conversationEntity.conversationId)
248                                         .from(QConversationEntity.conversationEntity)))
249                 .execute();
250         LOGGER.debug("deleted filtered conversation {} logs", deletedLogs);
251 
252         return PurgeResult.builder()
253                 .deletedMessages(deletedMessages)
254                 .deletedConversations(deletedConversations)
255                 .deletedMetaDatas(deletedMetaDatas + deletedSecondMetaDatas)
256                 .deletedPayloads(deletedPayloads)
257                 .deletedBinaries(deletedBinaries)
258                 .deletedLogs(deletedLogs)
259                 .deletedManualActions(deletedManualActions)
260                 .deletedManualActionsPartly(deletedManualActionsPartly)
261                 .build();
262     }
