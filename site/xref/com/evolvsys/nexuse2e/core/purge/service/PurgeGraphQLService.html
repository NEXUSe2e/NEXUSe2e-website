1   package com.evolvsys.nexuse2e.core.purge.service;
2   
3   import com.evolvsys.app.core.graphql.annotation.GraphQLService;
4   import com.evolvsys.nexuse2e.core.exception.NexusException;
5   import com.evolvsys.nexuse2e.core.logging.model.LogFilter;
6   import com.evolvsys.nexuse2e.core.manualaction.service.PurgeManualActionService;
7   import com.evolvsys.nexuse2e.core.message.ConversationFilter;
8   import com.evolvsys.nexuse2e.core.purge.model.PurgeResult;
9   import io.leangen.graphql.annotations.GraphQLMutation;
10  import io.leangen.graphql.annotations.GraphQLNonNull;
11  import jakarta.validation.Valid;
12  import lombok.RequiredArgsConstructor;
13  
14  /**
15   * @author SvenBarden
16   * @since 06.02.2024
17   */
18  @GraphQLService
19  @RequiredArgsConstructor
20  public class PurgeGraphQLService {
21      private final PurgeService purgeService;
22      private final PurgeManualActionService purgeManualActionService;
23  
24      @GraphQLMutation(name = "purge_allLogs")
25      @GraphQLNonNull
26      public PurgeResult purgeAllLogs() throws NexusException {
27          PurgeResult purgeResult = purgeService.purgeAllLogs();
28          purgeManualActionService.purgeLogs(null, purgeResult);
29          return purgeResult;
30      }
31  
32      @GraphQLMutation(name = "purge_logs", description = "Purge logs with optional selection")
33      @GraphQLNonNull
34      public PurgeResult purgeLogs(@Valid LogFilter logFilter) throws NexusException {
35          PurgeResult purgeResult = purgeService.purgeLogs(logFilter);
36          purgeManualActionService.purgeLogs(logFilter, purgeResult);
37          return purgeResult;
38      }
39  
40      @GraphQLMutation(name = "purge_conversations")
41      @GraphQLNonNull
42      public PurgeResult purgeConversations(@GraphQLNonNull ConversationFilter conversationFilter) throws NexusException {
43          PurgeResult purgeResult = purgeService.purgeConversations(conversationFilter);
44          purgeManualActionService.purgeConversations(conversationFilter, purgeResult);
45          return purgeResult;
46      }
47  
48      @GraphQLMutation(name = "purge_allConversations")
49      @GraphQLNonNull
50      public PurgeResult purgeAllConversations() throws NexusException {
51          PurgeResult purgeResult = purgeService.purgeAllConversations();
52          purgeManualActionService.purgeConversations(null, purgeResult);
53          return purgeResult;
54      }
55  }
