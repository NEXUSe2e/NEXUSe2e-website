3   import java.util.LinkedHashMap;
4   import java.util.Map;
5   import java.util.Optional;
6   import lombok.Value;
7   
8   /**
9    * @author SvenBarden
10   * @since 18.06.2024
11   */
12  public class InMemoryPersistentKeyValueRepositoryImpl implements PersistentKeyValueRepository {
13      private final Map<Key, String> values = new LinkedHashMap<>();
14  
15      @Override
16      public void save(String category, String key, String value) {
17          values.put(new Key(category, key), value);
18      }
19  
20      @Override
21      public Optional<String> get(String category, String key) {
22          return Optional.ofNullable(values.get(new Key(category, key)));
23      }
24  
25      @Value
26      private static class Key {
27          String category;
28          String name;
29      }
30  }
