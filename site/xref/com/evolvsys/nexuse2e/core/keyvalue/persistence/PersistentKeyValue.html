3   import com.evolvsys.nexuse2e.core.exception.NexusError;
4   import com.evolvsys.nexuse2e.core.exception.NexusRuntimeException;
5   import java.time.ZonedDateTime;
6   import java.time.format.DateTimeFormatter;
7   import java.util.Optional;
8   import lombok.RequiredArgsConstructor;
9   import org.apache.commons.lang3.StringUtils;
10  
11  /**
12   * @author SvenBarden
13   * @since 18.06.2024
14   */
15  @RequiredArgsConstructor
16  public class PersistentKeyValue {
17      private final PersistentKeyValueRepository keyValueRepository;
18  
19      public void set(String category, String key, String value) {
20          if (StringUtils.isBlank(category)) {
21              throw new NexusRuntimeException(NexusError.builder()
22                      .message("Persistent Category Name must not be blank")
23                      .build());
24          }
25          if (StringUtils.isBlank(key)) {
26              throw new NexusRuntimeException(NexusError.builder()
27                      .message("Persistent Key must not be blank")
28                      .build());
29          }
30          keyValueRepository.save(category, key, value);
31      }
32  
33      public void setLong(String category, String key, Long value) {
34          set(category, key, value == null ? null : value.toString());
35      }
36  
37      public void setDateTime(String category, String key, ZonedDateTime dateTime) {
38          set(category, key, dateTime == null ? null : dateTime.format(DateTimeFormatter.ISO_ZONED_DATE_TIME));
39      }
40  
41      public Optional<String> get(String category, String key) {
42          return keyValueRepository.get(category, key);
43      }
44  
45      public Optional<Long> getAsLong(String category, String key) {
46          return get(category, key).map(Long::valueOf);
47      }
48  
49      public Optional<ZonedDateTime> getAsDatetime(String category, String key) {
50          return get(category, key).map(ZonedDateTime::parse);
51      }
52  }
