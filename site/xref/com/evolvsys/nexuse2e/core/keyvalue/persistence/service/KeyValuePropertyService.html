27  @Service
28  public class KeyValuePropertyService
29          extends BaseService<KeyValuePropertyDto, KeyValuePropertyEntity, KeyValuePropertyRepository> {
30      private final ListService listService;
31  
32      public KeyValuePropertyService(
33              KeyValuePropertyMapper mapper,
34              KeyValuePropertyRepository repository,
35              CurrentUserService currentUserService,
36              ListService listService) {
37          super(mapper, repository, currentUserService);
38          this.listService = listService;
39      }
40  
41      private StringLookup toLookup(Collection<KeyValuePropertyEntity> entities) {
42          return StringLookupFactory.INSTANCE.mapStringLookup(entities.stream()
43                  .collect(StreamUtils.toMapOfNullables(
44                          KeyValuePropertyEntity::getKey, KeyValuePropertyEntity::getValue)));
45      }
46  
47      public Map<String, StringLookup> loadLookups() {
48          List<KeyValuePropertyEntity> entities = getRepository().findAll();
49          return EntryStream.of(StreamEx.of(entities).groupingBy(KeyValuePropertyEntity::getCategory))
50                  .mapValues(this::toLookup)
51                  .toMap();
52      }
53  
54      @Override
55      protected KeyValuePropertyEntity createEntity() {
56          return new KeyValuePropertyEntity();
57      }
58  
59      public Page<KeyValuePropertyDto> list(Pageable pageable, KeyValuePropertyFilter filter) {
60          return listService.map(listService.list(pageable, filter, this), mapper);
61      }
62  }
