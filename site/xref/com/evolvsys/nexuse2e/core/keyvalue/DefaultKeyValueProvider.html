3   import java.time.Instant;
4   import java.time.LocalDate;
5   import java.time.ZonedDateTime;
6   import java.time.format.DateTimeFormatter;
7   import java.util.Map;
8   import java.util.function.Supplier;
9   import lombok.extern.slf4j.Slf4j;
10  import org.apache.commons.text.lookup.StringLookup;
11  
12  /**
13   * Implements
14   *
15   * @author SvenBarden
16   * @since 11.04.2025
17   */
18  @Slf4j
19  class DefaultKeyValueProvider implements StringLookup {
20      private static final Map<String, Supplier<String>> DEFAULT_LOOKUPS = Map.of(
21              "NOW",
22              () -> ZonedDateTime.now().format(DateTimeFormatter.ISO_DATE_TIME),
23              "NOW_MILLIS",
24              () -> String.valueOf(Instant.now().toEpochMilli()),
25              "NOW_DATE",
26              () -> LocalDate.now().format(DateTimeFormatter.ISO_DATE));
27  
28      @Override
29      public String lookup(String key) {
30          LOGGER.trace("KeyValueProvider lookup for key {}", key);
31          Supplier<String> supplier = DEFAULT_LOOKUPS.get(key);
32          String result;
33          if (supplier == null) {
34              LOGGER.warn("Unknown keyValue key '{}'", key);
35              result = "";
36          } else {
37              result = supplier.get();
38          }
39          return result;
40      }
41  }
