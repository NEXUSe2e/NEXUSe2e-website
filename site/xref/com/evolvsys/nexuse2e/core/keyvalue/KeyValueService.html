40      private final KeyValuePropertyService keyValuePropertyService;
41      private final ComponentRegistry componentRegistry;
42      private final ServiceRepository serviceRepository;
43      private final ServiceFactory serviceFactory;
44      private final List<String> componentIds = new ArrayList<>();
45      private final AtomicBoolean started = new AtomicBoolean(false);

47      StringLookup createStringLookup() {
48          Map<String, StringLookup> lookupMap = new LinkedHashMap<>(keyValuePropertyService.loadLookups());
49  
50          List<ServiceEntity> serviceEntities = serviceRepository.findBy(
51                  QServiceEntity.serviceEntity
52                          .componentId
53                          .in(componentIds)
54                          .and(QServiceEntity.serviceEntity.enabled.isTrue()),
55                  FluentQuery.FetchableFluentQuery::all);
56  
57          // since services may have dependencies that trigger errors we need to collect them and try again later
58          List<ServiceEntity> errorServices = new ArrayList<>();
59  
60          for (ServiceEntity serviceEntity : serviceEntities) {
61              try {
62                  addServiceToMap(serviceEntity, lookupMap);
63              } catch (Throwable e) {
64                  LOGGER.debug("initial error for key value provider service {}", serviceEntity.getServiceId(), e);
65                  errorServices.add(serviceEntity);
66              }
67          }
68  
69          int lastSize = 0;
70          while (CollectionUtils.size(errorServices) != lastSize) {
71              lastSize = errorServices.size();
72              KeyValueLookup.setStringLookup(StringLookupFactory.INSTANCE.interpolatorStringLookup(
73                      lookupMap, new DefaultKeyValueProvider(), false));
74              List<ServiceEntity> tempServices = new ArrayList<>(errorServices);
75              for (ServiceEntity errorService : tempServices) {
76                  try {
77                      addServiceToMap(errorService, lookupMap);
78                      errorServices.remove(errorService);
79                  } catch (Throwable e) {
80                      LOGGER.debug("additional error for key value provider service {}", errorService.getServiceId(), e);
81                  }
82              }
83          }
84  
85          for (ServiceEntity errorService : errorServices) {
86              LOGGER.error("error while reading key value service {}", errorService.getServiceId());
87          }
88  
89          return StringLookupFactory.INSTANCE.interpolatorStringLookup(lookupMap, new DefaultKeyValueProvider(), false);
90      }
91  
92      private void addServiceToMap(ServiceEntity serviceEntity, Map<String, StringLookup> lookupMap) {
93          KeyValueSupportService service = serviceFactory.get(serviceEntity);
94          KeyValueProvider keyValueProvider = service.getKeyValueProvider();
95          if (keyValueProvider == null) {
96              throw new NexusRuntimeException(NexusError.builder()
97                      .message(String.format("Service '%s' returned no keyValueProvider", serviceEntity.getServiceId()))
98                      .build());
99          } else {
100             lookupMap.put(keyValueProvider.getName(), keyValueProvider.getLookup());
101         }
102     }
103 
104     @Scheduled(fixedRate = 1, timeUnit = TimeUnit.MINUTES)
105     @Transactional(readOnly = true)
106     public void update() {
107         if (isRunning()) {
108             LOGGER.debug("updating keyValue map");
109             KeyValueLookup.setStringLookup(createStringLookup());
110         } else {
111             LOGGER.debug("skipping keyValue map update, not running");
112         }
113     }
114 
115     @Override
116     public void start() {
117         try {
118             if (started.compareAndSet(false, true)) {
119                 componentIds.clear();
120                 componentIds.addAll(componentRegistry.getRegisteredComponents(ComponentType.SUPPORT_SERVICE).stream()
121                         .filter(registeredNexusComponent -> KeyValueSupportService.class.isAssignableFrom(
122                                 registeredNexusComponent.getLoadedClass()))
123                         .map(RegisteredNexusComponent::getComponentId)
124                         .toList());
125             }
126         } catch (Exception e) {
127             throw new NexusRuntimeException(
128                     NexusError.builder()
129                             .message("Could not initialize key value service")
130                             .build(),
131                     e);
132         }
133     }
134 
135     @Override
136     public void stop() {
137         started.set(false);
138     }
139 
140     @Override
141     public boolean isRunning() {
142         return started.get();
143     }
144 
145     @Override
146     public int getPhase() {
147         return LifecycleConstants.KEY_VALUE_INITIALIZE;
148     }
