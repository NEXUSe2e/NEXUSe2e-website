17  
18  /**
19   * @author SvenBarden
20   * @since 11.10.2023
21   */
22  @Component
23  @RequiredArgsConstructor
24  @Slf4j
25  public class ServiceFactory {
26      private final ComponentFactory componentFactory;
27      private final ServiceRepository serviceRepository;
28      private final InboundMessageDispatcher inboundMessageDispatcher;
29      private final ContextFactory contextFactory;
30      private final MessageResolverFactory messageResolverFactory;
31  
32      @Transactional(readOnly = true)
33      public <T extends Service> T get(Long id) {
34          return get(serviceRepository.findById(id).orElseThrow(() -> new ObjectNotFoundException(ServiceDto.class, id)));
35      }
36  
37      @Transactional(readOnly = true)
38      public <T extends Service> T get(String serviceId) {
39          return get(serviceRepository
40                  .findByServiceId(serviceId)
41                  .orElseThrow(() -> new IllegalArgumentException("Service with id " + serviceId + " not found")));
42      }
43  
44      public <T extends Service> T get(ServiceEntity entity) {
45          T service = componentFactory.create(entity.getComponentId(), entity.getParameters());
46          LOGGER.debug(
47                  "creating service {} with component {} and implementation {}",
48                  entity.getServiceId(),
49                  entity.getComponentId(),
50                  service.getClass().getName());
51  
52          service.setServiceId(entity.getServiceId());
53          service.setComponentId(entity.getComponentId());
54          if (service instanceof BaseInflowNexusService nexusService) {
55              nexusService.setMessageResolver(
56                      messageResolverFactory.create(entity.getResolverPipeline(), MessageType.NORMAL));
57              nexusService.setReturnMessageResolver(
58                      messageResolverFactory.create(entity.getReturnPipeline(), MessageType.RETURN));
59              if (entity.getErrorPipeline() != null) {
60                  nexusService.setErrorMessageResolver(
61                          messageResolverFactory.create(entity.getErrorPipeline(), MessageType.ERROR));
62              }
63  
64              nexusService.setMessageDispatcher(inboundMessageDispatcher);
65              nexusService.setContextFactory(contextFactory);
66          }
67  
68          return service;
69      }
70  }
