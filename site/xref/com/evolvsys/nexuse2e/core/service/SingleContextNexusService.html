17      @Override
18      protected final List<MessageContext> buildMessageContexts() throws NexusException {
19          MessageContext messageContext = buildMessageContext();
20          List<MessageContext> contexts;
21          if (messageContext == null) {
22              contexts = Collections.emptyList();
23          } else {
24              contexts = List.of(messageContext);
25          }
26          return contexts;
27      }
