14  /**
15   * @author Florian Drees

17   */
18  @Component
19  @RequiredArgsConstructor
20  public class ServiceStopAction implements ProcessAction<ServiceEntity> {
21      private final ServiceFactory serviceFactory;
22      private final ServiceRepository serviceRepository;
23  
24      @Override
25      public ServiceEntity execute(ServiceEntity entity, ProcessActionChain chain, ProcessParameters processParameters) {
26          Service service = serviceFactory.get(entity);
27          service.stop();
28  
29          entity.setEnabled(false);
30          return serviceRepository.save(entity);
31      }
32  
33      @Override
34      public ProcessActionConditions conditions(ServiceEntity entity, ProcessParameters processParameters) {
35          return ProcessActionConditions.EMPTY;
36      }
37  }
