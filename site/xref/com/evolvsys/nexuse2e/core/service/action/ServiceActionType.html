16              .action(ServiceRunOnceAction.class)

18              .build()),
19      START(ProcessDefinition.<ServiceEntity, ServiceDto>builder()
20              .action(ServiceStartAction.class)
21              .isValidDto(dto -> !dto.isEnabled())
22              .isValidEntity(entity -> !entity.isEnabled())
23              .build()),
24      STOP(ProcessDefinition.<ServiceEntity, ServiceDto>builder()
25              .action(ServiceStopAction.class)
26              .isValidDto(ServiceDto::isEnabled)
27              .isValidEntity(ServiceEntity::isEnabled)
28              .build());
