1   package com.evolvsys.nexuse2e.core.service.service;
2   
3   import com.evolvsys.app.core.graphql.annotation.GraphQLService;
4   import com.evolvsys.app.core.graphql.dataloader.DataLoaderService;
5   import com.evolvsys.nexuse2e.core.config.service.ConfigGraphQLService;
6   import com.evolvsys.nexuse2e.core.parameter.model.ParameterDto;
7   import com.evolvsys.nexuse2e.core.pipeline.PipelineBatchLoader;
8   import com.evolvsys.nexuse2e.core.pipeline.model.PipelineDto;
9   import com.evolvsys.nexuse2e.core.service.ServiceFilter;
10  import com.evolvsys.nexuse2e.core.service.ServiceParameterBatchLoader;
11  import com.evolvsys.nexuse2e.core.service.model.ServiceDto;
12  import com.evolvsys.nexuse2e.core.service.validation.ValidServiceHasConnections;
13  import io.leangen.graphql.annotations.*;
14  import io.leangen.graphql.execution.ResolutionEnvironment;
15  import jakarta.validation.Valid;
16  import java.util.List;
17  import java.util.Optional;
18  import java.util.Set;
19  import java.util.concurrent.CompletableFuture;
20  import java.util.function.Function;
21  import lombok.RequiredArgsConstructor;
22  import lombok.extern.slf4j.Slf4j;
23  
24  /**
25   * <p>Service for (Nexus-) sender services.</p>
26   *
27   * @author Florian Drees
28   * @since 14.12.2023
29   */
30  @GraphQLService
31  @RequiredArgsConstructor
32  @Slf4j
33  public class ServiceGraphQLService {
34      private final ServiceService serviceService;
35      private final DataLoaderService dataLoaderService;
36  
37      @GraphQLMutation(name = "service_update", description = "Updates or saves a service")
38      @GraphQLNonNull
39      public ServiceDto update(@Valid ServiceDto serviceDto) {
40          return ConfigGraphQLService.notifySubscribers(serviceService.save(serviceDto));
41      }
42  
43      @GraphQLMutation(name = "service_delete", description = "Deletes a service")
44      public boolean delete(@ValidServiceHasConnections long serviceId) {
45          serviceService.delete(serviceId);
46          ConfigGraphQLService.notifySubscribers();
47          return true;
48      }
49  
50      @GraphQLMutation(name = "service_startService", description = "Start the service in the scheduler")
51      @GraphQLNonNull
52      public ServiceDto startService(long serviceId) {
53          return ConfigGraphQLService.notifySubscribers(serviceService.startService(serviceId));
54      }
55  
56      @GraphQLMutation(name = "service_stopService", description = "Remove the service from the scheduler")
57      @GraphQLNonNull
58      public ServiceDto stopService(long serviceId) {
59          return ConfigGraphQLService.notifySubscribers(serviceService.stopService(serviceId));
60      }
61  
62      @GraphQLMutation(name = "service_runOnce", description = "Runs the service once")
63      public boolean runServiceOnce(long serviceId) {
64          return serviceService.run(serviceId);
65      }
66  
67      @GraphQLQuery(name = "service_list", description = "List all services")
68      @GraphQLNonNull
69      public List<@GraphQLNonNull ServiceDto> list(ServiceFilter serviceFilter) {
70          return serviceService.list(serviceFilter);
71      }
72  
73      @GraphQLQuery(name = "service_getById", description = "Gets a service")
74      public Optional<ServiceDto> getById(long serviceId) {
75          return serviceService.getById(serviceId);
76      }
77  
78      @GraphQLQuery(description = "list all parameters of the provided service")
79      public CompletableFuture<Set<@GraphQLNonNull ParameterDto>> parameters(
80              @GraphQLContext ServiceDto serviceDto, @GraphQLEnvironment ResolutionEnvironment resolutionEnvironment) {
81          LOGGER.trace("getting parameters for service {}", serviceDto);
82          return dataLoaderService.load(
83                  resolutionEnvironment, ServiceParameterBatchLoader.class, serviceDto, Function.identity());
84      }
85  
86      @GraphQLQuery(description = "load the resolver pipeline for the provided service")
87      public CompletableFuture<PipelineDto> resolverPipeline(
88              @GraphQLContext ServiceDto serviceDto, @GraphQLEnvironment ResolutionEnvironment resolutionEnvironment) {
89          return dataLoaderService.load(
90                  resolutionEnvironment,
91                  PipelineBatchLoader.class,
92                  serviceDto,
93                  "resolverPipelineId",
94                  ServiceDto::getResolverPipelineId);
95      }
96  
97      @GraphQLQuery(description = "load the return pipeline for the provided service")
98      public CompletableFuture<PipelineDto> returnPipeline(
99              @GraphQLContext ServiceDto serviceDto, @GraphQLEnvironment ResolutionEnvironment resolutionEnvironment) {
100         return dataLoaderService.load(
101                 resolutionEnvironment,
102                 PipelineBatchLoader.class,
103                 serviceDto,
104                 "returnPipelineId",
105                 ServiceDto::getReturnPipelineId);
106     }
107 
108     @GraphQLQuery(description = "load the error pipeline for the provided service")
109     public CompletableFuture<PipelineDto> errorPipeline(
110             @GraphQLContext ServiceDto serviceDto, @GraphQLEnvironment ResolutionEnvironment resolutionEnvironment) {
111         return dataLoaderService.load(
112                 resolutionEnvironment,
113                 PipelineBatchLoader.class,
114                 serviceDto,
115                 "errorPipelineId",
116                 ServiceDto::getErrorPipelineId);
117     }
118 }
