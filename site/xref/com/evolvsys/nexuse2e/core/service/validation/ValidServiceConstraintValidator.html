11   * @author Florian Drees

14  @Component
15  @RequiredArgsConstructor
16  public class ValidServiceConstraintValidator implements ConstraintValidator<ValidService, ServiceDto> {
17      private final ServiceRepository serviceRepository;
18  
19      @Override
20      public boolean isValid(ServiceDto value, ConstraintValidatorContext context) {
21          return ServiceValidationUtils.validateServiceName(
22                  value.getId(),
23                  value.getServiceId(),
24                  serviceRepository::existsByServiceId,
25                  serviceRepository::existsByServiceIdAndIdNot);
26      }
27  }
