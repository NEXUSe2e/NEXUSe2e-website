3   import com.evolvsys.nexuse2e.core.component.ComponentRegistry;
4   import com.evolvsys.nexuse2e.core.component.RegisteredNexusComponent;
5   import lombok.RequiredArgsConstructor;
6   import lombok.extern.slf4j.Slf4j;
7   import org.apache.commons.lang3.Validate;
8   import org.springframework.lang.NonNull;
9   import org.springframework.stereotype.Component;
10  
11  /**
12   * @author Florian Drees
13   * @since 14.02.2024
14   */
15  @Component
16  @RequiredArgsConstructor
17  @Slf4j
18  public class ServiceUtils {
19      private final ComponentRegistry componentRegistry;
20  
21      public ServiceType getServiceType(@NonNull String componentId) {
22          Validate.notNull(componentId, "component id must not be null");
23          return componentRegistry
24                  .getComponent(componentId)
25                  .map(RegisteredNexusComponent::getComponentType)
26                  .map((componentType) -> switch (componentType) {
27                      case INFLOW_SERVICE -> ServiceType.INFLOW_SERVICE;
28                      case OUTFLOW_SERVICE -> ServiceType.OUTFLOW_SERVICE;
29                      case SUPPORT_SERVICE -> ServiceType.SUPPORT_SERVICE;
30                      default -> throw new IllegalArgumentException("Component type not supported");
31                  })
32                  .orElseThrow(() ->
33                          new IllegalArgumentException("Component with id " + componentId + " could not be found!"));
34      }
35  }
