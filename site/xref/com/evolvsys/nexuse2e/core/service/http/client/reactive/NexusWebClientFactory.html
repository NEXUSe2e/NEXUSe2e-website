14  import org.springframework.http.client.reactive.ReactorClientHttpConnector;
15  import org.springframework.stereotype.Component;
16  import org.springframework.web.reactive.function.client.WebClient;
17  import reactor.netty.http.client.HttpClient;
18  
19  /**
20   * @author SvenBarden
21   * @since 12.02.2024
22   */
23  @Component
24  @RequiredArgsConstructor
25  @Slf4j
26  public class NexusWebClientFactory {
27      @Delegate
28      private final HttpContextFactory httpContextFactory;
29  
30      private WebClient.Builder createSslWebClient(WebClientContext context)
31              throws NoSuchAlgorithmException, UnrecoverableKeyException, KeyStoreException, SSLException,
32                      NexusException {
33          WebClient.Builder builder = WebClient.builder();
34  
35          HttpClient httpClient = HttpClientFactory.createHttpClient(context);
36  
37          return builder.clientConnector(new ReactorClientHttpConnector(httpClient));
38      }
39  
40      public WebClient.Builder createWebClient(WebClientContext context) throws NexusException {
41          LOGGER.debug("create web client");
42          try {
43              return createSslWebClient(context);
44          } catch (NoSuchAlgorithmException | UnrecoverableKeyException | KeyStoreException | SSLException e) {
45              throw new NexusException(
46                      NexusError.builder()
47                              .message("could not create webClientBuilder")
48                              .build(),
49                      e);
50          }
51      }
52  }
