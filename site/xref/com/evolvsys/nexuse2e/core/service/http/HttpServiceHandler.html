31      private static final Pattern BASE_PATTERN = Pattern.compile("/service/([^/]+).*");
32      private final HttpServiceFactory httpServiceFactory;

34      private String serializeError(Error error) {
35          StringBuilder s = new StringBuilder();
36          s.append(error.getMessage());
37          for (Map.Entry<String, Object> entry : error.getExtensions().entrySet()) {
38              s.append('\n').append(entry.getKey()).append(": ").append(entry.getValue());
39          }
40          return s.toString();
41      }
42  
43      private void error(String message, HttpServletResponse response) {
44          try {
45              response.setStatus(HttpServletResponse.SC_BAD_REQUEST);
46              response.getOutputStream().write(message.getBytes(StandardCharsets.UTF_8));
47          } catch (IOException e) {
48              response.setStatus(HttpServletResponse.SC_INTERNAL_SERVER_ERROR);
49              LOGGER.error(
50                      "could not write message to response output stream: {}, {}", message, ErrorUtils.aggregate(e), e);
51          }
52      }
53  
54      private String getRequestInfo(HttpServletRequest httpServletRequest) {
55          StringBuilder requestInfo = new StringBuilder(24);
56          requestInfo
57                  .append("ip: ")
58                  .append(httpServletRequest.getRemoteAddr())
59                  .append(", method: ")
60                  .append(httpServletRequest.getMethod());
61          Map<String, String[]> parameterMap = httpServletRequest.getParameterMap();
62          if (MapUtils.isNotEmpty(parameterMap)) {
63              requestInfo
64                      .append(", params: ")
65                      .append(EntryStream.of(parameterMap)
66                              .map(stringEntry ->
67                                      stringEntry.getKey() + ": [" + StringUtils.join(stringEntry.getValue(), ", ") + "]")
68                              .joining(", "));
69          }
70  
71          return requestInfo.toString();
72      }

74      private void logAdditionalInfos(HttpServletRequest httpServletRequest) {
75          if (LOGGER.isTraceEnabled()) {
76              LOGGER.trace("client ip {}", httpServletRequest.getRemoteAddr());
77              LOGGER.trace("request method {}", httpServletRequest.getMethod());
78  
79              LOGGER.trace("request headers start");
80              Enumeration<String> headerNames = httpServletRequest.getHeaderNames();
81              if (headerNames != null) {
82                  while (headerNames.hasMoreElements()) {
83                      String headerName = headerNames.nextElement();
84                      Enumeration<String> values = httpServletRequest.getHeaders(headerName);
85                      if (values == null) {
86                          LOGGER.trace("header {}: ", headerNames);
87                      } else {
88                          while (values.hasMoreElements()) {
89                              LOGGER.trace("header {}: {}", headerName, values.nextElement());
90                          }
91                      }
92                  }
93              }
94              LOGGER.trace("request headers finished");
95  
96              LOGGER.trace("request parameters start");
97              Map<String, String[]> parameterMap = httpServletRequest.getParameterMap();
98              if (MapUtils.isNotEmpty(parameterMap)) {
99                  for (Map.Entry<String, String[]> entry : parameterMap.entrySet()) {
100                     String[] values = entry.getValue();
101                     if (ArrayUtils.isEmpty(values)) {
102                         LOGGER.trace("param {}: []", entry.getKey());
103                     } else {
104                         for (String value : values) {
105                             LOGGER.trace("param {}: {}", entry.getKey(), value);
106                         }
107                     }
108                 }
109             }
110             LOGGER.trace("request parameters finished");
111         }
112     }

114     @SuppressFBWarnings("EXS_EXCEPTION_SOFTENING_NO_CONSTRAINTS")
115     public void handle(HttpServletRequest httpServletRequest, HttpServletResponse httpServletResponse) {
116         String servletPath = httpServletRequest.getServletPath();
117 
118         LOGGER.trace("got request at {}", servletPath);
119 
120         logAdditionalInfos(httpServletRequest);
121 
122         String servicePath = BASE_PATTERN.matcher(servletPath).replaceAll("$1");
123 
124         LOGGER.debug("looking for http service with path {}", servicePath);
125 
126         try {
127             HttpReceiver service = httpServiceFactory.get(servicePath);
128             service.setHttpServletResponse(httpServletResponse);
129             service.setHttpServletRequest(httpServletRequest);
130 
131             try {
132                 service.execute();
133                 int status = httpServletResponse.getStatus();
134                 if (status >= 200 && status < 300) {
135                     LOGGER.debug("response status {} in 2xx range, request was successful", status);
136                 } else {
137                     LOGGER.warn(
138                             "response status {} not in 2xx range, request {} was unsuccessful",
139                             status,
140                             getRequestInfo(httpServletRequest));
141                 }
142             } catch (NexusException e) {
143                 error(serializeError(e.getError()), httpServletResponse);
144                 LOGGER.error(
145                         "error calling http receiver for path {} with request {}: {}",
146                         servicePath,
147                         getRequestInfo(httpServletRequest),
148                         ErrorUtils.aggregate(e),
149                         e);
150             } catch (Throwable e) {
151                 error("Error calling http receiver: " + e.getMessage(), httpServletResponse);
152                 LOGGER.error(
153                         "error calling http receiver for path {} with request {}: {}",
154                         servicePath,
155                         getRequestInfo(httpServletRequest),
156                         ErrorUtils.aggregate(e),
157                         e);
158             }
159         } catch (NexusException e) {
160             error(serializeError(e.getError()), httpServletResponse);
161             LOGGER.error(
162                     "error creating http receiver for path {} with request {}: {}",
163                     servicePath,
164                     getRequestInfo(httpServletRequest),
165                     ErrorUtils.aggregate(e),
166                     e);
167         } catch (Throwable e) {
168             error("Error creating http receiver: " + e.getMessage(), httpServletResponse);
169             LOGGER.error(
170                     "error creating http receiver for path {} with request {}: {}",
171                     servicePath,
172                     getRequestInfo(httpServletRequest),
173                     ErrorUtils.aggregate(e),
174                     e);
175         }
176     }
