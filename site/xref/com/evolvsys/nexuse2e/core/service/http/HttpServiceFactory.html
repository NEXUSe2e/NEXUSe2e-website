17   * @author SvenBarden

20  @Component
21  @RequiredArgsConstructor
22  class HttpServiceFactory {
23      private final ServiceRepository serviceRepository;
24      private final ComponentRegistry componentRegistry;
25      private final ServiceFactory serviceFactory;
26  
27      private boolean filterHttpService(ServiceEntity serviceEntity) {
28          return componentRegistry
29                  .getComponent(serviceEntity.getComponentId())
30                  .map(component -> HttpReceiver.class.isAssignableFrom(component.getLoadedClass())
31                          && BaseInflowNexusService.class.isAssignableFrom(component.getLoadedClass()))
32                  .orElse(Boolean.FALSE);
33      }
34  
35      public HttpReceiver get(String path) throws NexusException {
36          List<ServiceEntity> services = serviceRepository.getByTypeAndParameterValue(
37                  ServiceType.INFLOW_SERVICE, HttpReceiver.URL_PATH_PARAM_NAME, path);
38  
39          if (CollectionUtils.isEmpty(services)) {
40              throw new NexusException(NexusError.builder()
41                      .message("No service found for url path")
42                      .extension("path", path)
43                      .build());
44          }
45  
46          List<ServiceEntity> filteredServices =
47                  services.stream().filter(this::filterHttpService).toList();
48          if (CollectionUtils.isEmpty(filteredServices)) {
49              throw new NexusException(NexusError.builder()
50                      .message("Services found for url path missing HttpReceiver or NexusService dependency")
51                      .extension("path", path)
52                      .build());
53          }
54  
55          if (filteredServices.size() > 1) {
56              throw new NexusException(NexusError.builder()
57                      .message("Found more than one HttpReceiver for the path")
58                      .extension("path", path)
59                      .build());
60          }
61  
62          ServiceEntity serviceEntity = filteredServices.getFirst();
63  
64          return serviceFactory.get(serviceEntity.getId());
65      }
66  }
