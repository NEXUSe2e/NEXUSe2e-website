13  import lombok.experimental.UtilityClass;

15  import reactor.netty.tcp.SslProvider;
16  
17  /**
18   * @author SvenBarden
19   * @since 16.02.2024
20   */
21  @UtilityClass
22  public class HttpClientFactory {
23  
24      private static void sslProviderBuilder(
25              SslProvider.SslContextSpec sslContextSpec, SslContext sslContext, WebClientContext context) {
26          SslProvider.Builder builder = sslContextSpec.sslContext(sslContext);
27          if (context.isDisableHostNameVerification()) {
28              builder.handlerConfigurator(handler -> {
29                  SSLEngine engine = handler.engine();
30                  SSLParameters sslParameters = engine.getSSLParameters();
31                  SSLParameters params = new SSLParameters(sslParameters.getCipherSuites(), sslParameters.getProtocols());
32                  // With java update null value is no longer sufficient to skip host name verification
33                  params.setEndpointIdentificationAlgorithm("");
34                  engine.setSSLParameters(params);
35              });
36          }
37      }
38  
39      public static HttpClient createHttpClient(WebClientContext context)
40              throws SSLException, NexusException, UnrecoverableKeyException, NoSuchAlgorithmException,
41                      KeyStoreException {
42          HttpClient httpClient = HttpClient.create();
43  
44          if (context.isSsl()) {
45              SslContext sslContext = SslContextFactory.createSSLContext(context);
46              httpClient = httpClient.secure(sslContextSpec -> sslProviderBuilder(sslContextSpec, sslContext, context));
47          }
48  
49          if (context.getResponseTimeout() != null) {
50              httpClient.responseTimeout(context.getResponseTimeout());
51          }
52          if (context.getConnectionTimeout() != null) {
53              httpClient.option(ChannelOption.CONNECT_TIMEOUT_MILLIS, (int)
54                      context.getConnectionTimeout().toMillis());
55          }
56  
57          return httpClient;
58      }
59  }
