20  public interface MessageRepository
21          extends JpaRepository<MessageEntity, Long>, QuerydslPredicateExecutor<MessageEntity> {
22  
23      Optional<MessageEntity> findByMessageId(String messageId);
24  
25      default Page<MessageEntity> findAllToRead(
26              Pageable pageable,
27              @Nullable MessageFilter messageFilter,
28              String timeZone,
29              BooleanExpression securityPredicate) {
30          return findAll(securityPredicate.and(QMessageEntity.messageEntity.filter(messageFilter, timeZone)), pageable);
31      }
32  
33      Optional<MessageEntity> findByReferenceMessageAndTypeAndOutboundAcknowledgment(
34              MessageEntity referenceMessage, MessageType messageType, boolean outboundAcknowledgment);
35  
36      boolean existsByMessageId(String messageId);
37  
38      boolean existsByAction_Id(Long id);
39  
40      boolean existsBySourcePartner_IdOrTargetPartner_Id(Long id, Long id1);
41  
42      default boolean exists(@Nullable MessageFilter messageFilter, @NonNull String timeZone) {
43          return exists(QMessageEntity.messageEntity.filter(messageFilter, timeZone));
44      }
45  }
