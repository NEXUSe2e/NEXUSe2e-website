24  public final class MetaDataQueryDelegate {
25      private MetaDataQueryDelegate() {}
26  
27      @QueryDelegate(MetaDataEntity.class)
28      public static Predicate filter(QMetaDataEntity qMetaDataEntity, MetaDataFilter metaDataFilter) {
29          BooleanBuilder builder = new BooleanBuilder();
30          if (StringUtils.isNoneBlank(metaDataFilter.getKey(), metaDataFilter.getValue())) {
31              builder.and(
32                      qMetaDataEntity.name.eq(metaDataFilter.getKey()).and(filterValue(qMetaDataEntity, metaDataFilter)));
33          } else if (StringUtils.isNotEmpty(metaDataFilter.getKey())) {
34              builder.and(qMetaDataEntity.name.eq(metaDataFilter.getKey()));
35          } else if (StringUtils.isNotEmpty(metaDataFilter.getValue())) {
36              builder.and(filterValue(qMetaDataEntity, metaDataFilter));
37          }
38          return builder.getValue();
39      }
40  
41      private static Predicate filterValue(QMetaDataEntity qMetaDataEntity, MetaDataFilter metaDataFilter) {
42          BooleanBuilder builder = new BooleanBuilder();
43          builder.or(qMetaDataEntity.stringValue.eq(metaDataFilter.getValue()));
44  
45          try {
46              builder.or(qMetaDataEntity.longValue.eq(Long.valueOf(metaDataFilter.getValue())));
47          } catch (NumberFormatException e) {
48              // do nothing and ignore result. Filtering of field longValue is ignored
49          }
50  
51          try {
52              builder.or(qMetaDataEntity.dateTimeValue.eq(ZonedDateTime.parse(metaDataFilter.getValue())));
53          } catch (DateTimeParseException e) {
54              // do nothing and ignore result. Filtering of field dateTimeValue is ignored
55          }
56  
57          Boolean booleanValue = BooleanUtils.toBooleanObject(metaDataFilter.getValue());
58          if (booleanValue != null) {
59              builder.or(qMetaDataEntity.booleanValue.eq(booleanValue));
60          }
61          return builder.getValue();
62      }
63  
64      @QueryDelegate(MetaDataEntity.class)
65      public static Predicate filterByValues(
66              QMetaDataEntity qMetaDataEntity, @NonNull String metaDataKey, @NonNull List<Object> values) {
67          BooleanBuilder builder = new BooleanBuilder();
68  
69          List<String> stringValues = getTypedValues(values, String.class);
70          if (CollectionUtils.isNotEmpty(stringValues)) {
71              builder.or(qMetaDataEntity.stringValue.in(stringValues));
72          }
73  
74          List<Long> longValues = getLongValues(values);
75          if (CollectionUtils.isNotEmpty(longValues)) {
76              builder.or(qMetaDataEntity.longValue.in(longValues));
77          }
78  
79          List<Boolean> booleanValues = getTypedValues(values, Boolean.class);
80          if (CollectionUtils.isNotEmpty(booleanValues)) {
81              builder.or(qMetaDataEntity.booleanValue.in(booleanValues));
82          }
83  
84          List<BigDecimal> bigDecimalValues = getTypedValues(values, BigDecimal.class);
85          if (CollectionUtils.isNotEmpty(bigDecimalValues)) {
86              builder.or(qMetaDataEntity.bigDecimalValue.in(bigDecimalValues));
87          }
88  
89          List<ZonedDateTime> dateTimeValues = getTypedValues(values, ZonedDateTime.class);
90          if (CollectionUtils.isNotEmpty(dateTimeValues)) {
91              builder.or(qMetaDataEntity.dateTimeValue.in(dateTimeValues));
92          }
93  
94          BooleanBuilder wrapper = new BooleanBuilder(qMetaDataEntity.name.eq(metaDataKey));
95          if (builder.hasValue()) {
96              wrapper.and(builder.getValue());
97          }
98          return wrapper;
99      }
100 
101     @NonNull
102     private static <T> List<T> getTypedValues(@NonNull List<Object> values, @NonNull Class<T> target) {
103         return CollectionUtils.emptyIfNull(values).stream()
104                 .filter(object -> target.isAssignableFrom(object.getClass()))
105                 .map(target::cast)
106                 .toList();
107     }
108 
109     @NonNull
110     private static List<Long> getLongValues(@NonNull List<Object> values) {
111         Stream<Long> longValues = getTypedValues(values, Long.class).stream();
112         return StreamEx.of(longValues)
113                 .append(getTypedValues(values, Integer.class).stream()
114                         .mapToLong(Long::valueOf)
115                         .boxed())
116                 .toList();
117     }
