20  public final class ConversationQueryDelegate {
21      private ConversationQueryDelegate() {}
22  
23      @QueryDelegate(ConversationEntity.class)
24      public static Predicate filter(
25              QConversationEntity qConversationEntity, @Nullable ConversationFilter conversationFilter, String timeZone) {
26          BooleanBuilder builder = new BooleanBuilder();
27          if (conversationFilter != null) {
28              if (StringUtils.isNotEmpty(conversationFilter.getConversationId())) {
29                  builder.and(qConversationEntity.conversationId.contains(conversationFilter.getConversationId()));
30              }
31              if (CollectionUtils.isNotEmpty(conversationFilter.getChoreographyIds())) {
32                  builder.and(qConversationEntity.choreography.id.in(conversationFilter.getChoreographyIds()));
33              }
34              if (conversationFilter.getFinishedFrom() != null) {
35                  builder.and(qConversationEntity.finished.goe(conversationFilter.getFinishedFrom()));
36              }
37              if (conversationFilter.getFinishedTill() != null) {
38                  builder.and(qConversationEntity.finished.loe(conversationFilter.getFinishedTill()));
39              }
40              if (CollectionUtils.isNotEmpty(conversationFilter.getLastDirections())) {
41                  builder.and(qConversationEntity.lastDirection.in(conversationFilter.getLastDirections()));
42              }
43              if (CollectionUtils.isNotEmpty(conversationFilter.getStatuses())) {
44                  builder.and(qConversationEntity.status.in(conversationFilter.getStatuses()));
45              }
46              if (StringUtils.isNotBlank(conversationFilter.getLastSourcePartnerId())) {
47                  builder.and(qConversationEntity.lastSourcePartner.partnerId.contains(
48                          conversationFilter.getLastSourcePartnerId()));
49              }
50              if (StringUtils.isNotBlank(conversationFilter.getLastSourcePartnerDescription())) {
51                  builder.and(qConversationEntity.lastSourcePartner.description.contains(
52                          conversationFilter.getLastSourcePartnerDescription()));
53              }
54              if (StringUtils.isNotBlank(conversationFilter.getLastTargetPartnerId())) {
55                  builder.and(qConversationEntity.lastTargetPartner.partnerId.contains(
56                          conversationFilter.getLastTargetPartnerId()));
57              }
58              if (StringUtils.isNotBlank(conversationFilter.getLastTargetPartnerDescription())) {
59                  builder.and(qConversationEntity.lastTargetPartner.description.contains(
60                          conversationFilter.getLastTargetPartnerDescription()));
61              }
62              if (conversationFilter.getLastUpdatedSince() != null) {
63                  builder.and(qConversationEntity.lastUpdated.loe(
64                          ZonedDateTime.now().minus(conversationFilter.getLastUpdatedSince())));
65              }
66              Predicate predicate = DateRangeFilterUtils.filter(
67                      qConversationEntity.created,
68                      timeZone,
69                      conversationFilter.getCreatedRange(),
70                      conversationFilter.getCreatedFrom(),
71                      conversationFilter.getCreatedTill());
72              if (predicate != null) {
73                  builder.and(predicate);
74              }
75              if (conversationFilter.getLastPartnerId() != null) {
76                  builder.and(Expressions.anyOf(
77                          qConversationEntity.lastSourcePartner.id.eq(conversationFilter.getLastPartnerId()),
78                          qConversationEntity.lastTargetPartner.id.eq(conversationFilter.getLastPartnerId())));
79              }
80          }
81          return builder.getValue();
82      }
83  }
