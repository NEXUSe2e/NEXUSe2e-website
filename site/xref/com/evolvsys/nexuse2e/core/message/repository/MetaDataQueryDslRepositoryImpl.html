27   * QueryDsl-Repository for the message metadata.
28   *
29   * @author Julian Drees
30   * @since 29.08.2025
31   */
32  @Slf4j
33  @Repository
34  @RequiredArgsConstructor
35  class MetaDataQueryDslRepositoryImpl implements MetaDataQueryDslRepository {
36      private final JPQLQueryFactory jpqlQueryFactory;
37  
38      @NonNull
39      @Override
40      public Map<String, List<MetaDataEntity>> getDuplicatesByKey(
41              @NonNull Map<String, List<Object>> metaData,
42              @NonNull String category,
43              @Nullable String choreographyId,
44              @NonNull MetaDataLinkType linkType) {
45          QMessageEntity qMessageEntity = QMessageEntity.messageEntity;
46          QMetaDataEntity qMetaDataEntity = QMetaDataEntity.metaDataEntity;
47  
48          JPQLSubQuery<Long> subQuery = createSubQuery(metaData, linkType, category, metaData.size());
49          JPQLQuery<MetaDataEntity> query = jpqlQueryFactory
50                  .select(qMetaDataEntity)
51                  .from(qMessageEntity)
52                  .innerJoin(qMessageEntity.metaDatas, qMetaDataEntity)
53                  .where(
54                          qMessageEntity.id.in(subQuery),
55                          qMetaDataEntity.category.eq(category),
56                          createPredicate(metaData, qMetaDataEntity));
57  
58          if (StringUtils.isNotBlank(choreographyId)) {
59              LOGGER.debug("Searching for duplicates with choreography '{}'.", choreographyId);
60              QConversationEntity qConversationEntity = QConversationEntity.conversationEntity;
61              QChoreographyEntity qChoreographyEntity = QChoreographyEntity.choreographyEntity;
62              query = query.innerJoin(qMessageEntity.conversation, qConversationEntity)
63                      .innerJoin(qConversationEntity.choreography, qChoreographyEntity)
64                      .on(qChoreographyEntity.choreographyId.eq(choreographyId));
65          }
66          return query.transform(GroupBy.groupBy(qMetaDataEntity.name).as(GroupBy.list(qMetaDataEntity)));
67      }
68  
69      @NonNull
70      private JPQLSubQuery<Long> createSubQuery(
71              @NonNull Map<String, List<Object>> metaData,
72              @NonNull MetaDataLinkType linkType,
73              @Nullable String category,
74              long keys) {
75          QMessageEntity qMessageEntity = new QMessageEntity("innerMessageEntity");
76          QMetaDataEntity qMetaDataEntity = new QMetaDataEntity("innerMetaDataEntity");
77          JPQLSubQuery<Long> subQuery = JPAExpressions.select(qMessageEntity.id)
78                  .from(qMessageEntity)
79                  .innerJoin(qMessageEntity.metaDatas, qMetaDataEntity)
80                  .where(createPredicate(metaData, qMetaDataEntity), qMetaDataEntity.category.eq(category))
81                  .groupBy(qMessageEntity.id);
82          if (linkType == MetaDataLinkType.AND) {
83              subQuery = subQuery.having(qMetaDataEntity.name.countDistinct().eq(keys));
84          }
85          return subQuery;
86      }
87  
88      @NonNull
89      private Predicate createPredicate(
90              @NonNull Map<String, List<Object>> metaData, @NonNull QMetaDataEntity qMetaDataEntity) {
91          BooleanBuilder builder = new BooleanBuilder();
92          for (Map.Entry<String, List<Object>> entry : metaData.entrySet()) {
93              builder.or(qMetaDataEntity.filterByValues(entry.getKey(), entry.getValue()));
94          }
95          return Optional.ofNullable(builder.getValue())
96                  .orElseThrow(() -> new IllegalArgumentException(
97                          "Creating this predicate without metadata entries is not allowed."));
98      }
99  }
