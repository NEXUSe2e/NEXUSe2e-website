1   package com.evolvsys.nexuse2e.core.message.dispatcher;
2   
3   import com.evolvsys.nexuse2e.core.exception.NexusError;
4   import com.evolvsys.nexuse2e.core.exception.NexusException;
5   import com.evolvsys.nexuse2e.core.message.context.MessageContext;
6   import lombok.RequiredArgsConstructor;
7   import lombok.extern.slf4j.Slf4j;
8   import org.springframework.stereotype.Component;
9   
10  /**
11   * @author SvenBarden
12   * @since 21.01.2024
13   */
14  @Component
15  @RequiredArgsConstructor
16  @Slf4j
17  public class InboundMessageDispatcher {
18      private final InboundNormalDispatcher inboundNormalDispatcher;
19      private final InboundAcknowledgmentDispatcher inboundAcknowledgmentDispatcher;
20  
21      public MessageContext dispatch(MessageContext messageContext) throws NexusException {
22          LOGGER.debug("dispatching inbound message {}", messageContext);
23  
24          return switch (messageContext.getSourceMessage().getType()) {
25              case NORMAL -> inboundNormalDispatcher.dispatch(messageContext);
26              case ACKNOWLEDGMENT -> inboundAcknowledgmentDispatcher.dispatch(messageContext);
27              default ->
28                  throw new NexusException(NexusError.builder()
29                          .message("Invalid inbound message type")
30                          .code("INVALID_MESSAGE_TYPE")
31                          .extension(
32                                  "type",
33                                  messageContext.getSourceMessage().getType().name())
34                          .extension(
35                                  "messageId", messageContext.getSourceMessage().getId())
36                          .extension(
37                                  "conversationId",
38                                  messageContext.getSourceMessage().getConversationId())
39                          .build());
40          };
41      }
42  }
