15  /**
16   * @author SvenBarden

18   */
19  @Component
20  @RequiredArgsConstructor
21  @Slf4j
22  class ReturnDispatcher implements Dispatcher {
23      private final ActionFactory actionFactory;
24      private final ContextMessageService messageService;
25  
26      @Override
27      public MessageContext dispatch(MessageContext messageContext) throws NexusException {
28          String actionId = messageContext.getActionId();
29          String choreographyId = messageContext.getChoreographyId();
30  
31          Action action = actionFactory.getAction(choreographyId, actionId);
32  
33          LOGGER.debug(
34                  "running return pipeline for message {} in action {} from choreography {}",
35                  messageContext.getSourceMessage().getId(),
36                  action.getActionId(),
37                  action.getChoreographyId());
38          if (messageContext.getReturnMessage() != null) {
39              if (action.hasReturnPipeline()) {
40                  try {
41                      messageContext = action.getReturnPipeline()
42                              .execute(messageContext.createExecutionContext(MessageType.RETURN))
43                              .getMessageContext();
44                  } catch (NexusException e) {
45                      // TODO error handling
46                      LOGGER.error("error running return pipeline: {}", ErrorUtils.aggregate(e), e);
47                  }
48              }
49  
50              if (messageContext.getReturnMessage() != null) {
51                  if (messageContext.getReturnMessage().getId() == null) {
52                      messageContext.getReturnMessage().setId(UUID.randomUUID().toString());
53                      messageContext
54                              .getReturnMessage()
55                              .setConversationId(messageContext.getSourceMessage().getConversationId());
56                  }
57                  messageService.save(messageContext.getReturnMessage(), action);
58              }
59          } else {
60              LOGGER.trace("context has no return message, do not run return pipeline");
61          }
62          return messageContext;
63      }
64  }
