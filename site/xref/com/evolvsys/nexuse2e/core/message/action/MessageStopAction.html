8   import com.evolvsys.nexuse2e.core.model.entity.ConversationStatus;
9   import com.evolvsys.nexuse2e.core.model.entity.MessageEntity;
10  import java.util.List;
11  import lombok.RequiredArgsConstructor;
12  import lombok.extern.slf4j.Slf4j;
13  import org.springframework.stereotype.Component;
14  
15  /**
16   * @author SvenBarden
17   * @since 22.05.2024
18   */
19  @Component
20  @RequiredArgsConstructor
21  @Slf4j
22  public class MessageStopAction implements ProcessAction<MessageEntity> {
23      public static final ProcessParameter<Long> MANUAL_ACTION_ID = new ProcessParameter<>() {};
24      public static final List<MessageStatus> VALID_STATUSES = List.of(
25              MessageStatus.RESOLVED,
26              MessageStatus.ACTION_SUCCESSFUL,
27              MessageStatus.QUEUE_ACTION,
28              MessageStatus.QUEUE_SENDER,
29              MessageStatus.WAITING_FOR_ACKNOWLEDGMENT);
30      private final MessageRepository messageRepository;
31      private final ConversationRepository conversationRepository;
32      private final MessageActionManualActionService messageActionManualActionService;
33  
34      @Override
35      public MessageEntity execute(MessageEntity entity, ProcessActionChain chain, ProcessParameters processParameters) {
36  
37          entity.setStatus(MessageStatus.STOPPED);
38          entity.setNextRequeue(null);
39          entity = messageRepository.save(entity);
40  
41          entity.getConversation().setStatus(ConversationStatus.STOPPED);
42          conversationRepository.save(entity.getConversation());
43  
44          Long manualActionId = processParameters.get(MANUAL_ACTION_ID);
45          manualActionId = messageActionManualActionService.stop(manualActionId, entity);
46          chain.addOutput(MANUAL_ACTION_ID.of(manualActionId));
47  
48          return entity;
49      }
50  
51      @Override
52      public ProcessActionConditions conditions(MessageEntity entity, ProcessParameters processParameters) {
53          return ProcessActionConditions.builder()
54                  .condition(VALID_STATUSES.contains(entity.getStatus()), "Status is not valid for stopping")
55                  .build();
56      }
57  }
