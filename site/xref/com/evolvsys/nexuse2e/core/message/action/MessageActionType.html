13  @Getter
14  public enum MessageActionType implements ProcessActionType<MessageEntity, MessageDto> {
15      REQUEUE(ProcessDefinition.<MessageEntity, MessageDto>builder()
16              .action(MessageRequeueAction.class)
17              .requiredPermission("NEXUS.SEND_MESSAGE")
18              .build()),
19      STOP(ProcessDefinition.<MessageEntity, MessageDto>builder()
20              .action(MessageStopAction.class)
21              .requiredPermission("NEXUS.SEND_MESSAGE")
22              .isValidDto(messageDto -> MessageStopAction.VALID_STATUSES.contains(messageDto.getStatus()))
23              .isValidEntity(messageEntity -> MessageStopAction.VALID_STATUSES.contains(messageEntity.getStatus()))
24              .build());
25  
26      private final ProcessDefinition<MessageEntity, MessageDto> process;
27  
28      MessageActionType(ProcessDefinition<MessageEntity, MessageDto> process) {
