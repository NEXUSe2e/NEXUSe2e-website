14  import org.springframework.stereotype.Component;
15  
16  /**
17   * @author SvenBarden
18   * @since 22.05.2024
19   */
20  @Component
21  @RequiredArgsConstructor
22  @Slf4j
23  public class MessageRequeueAction implements ProcessAction<MessageEntity> {
24      public static final ProcessParameter<Long> MANUAL_ACTION_ID = new ProcessParameter<>() {};
25      private final MessageRepository messageRepository;
26      private final MessageSenderService messageSenderService;
27      private final MessageActionManualActionService messageActionManualActionService;
28  
29      @Override
30      @SuppressFBWarnings("EXS_EXCEPTION_SOFTENING_NO_CONSTRAINTS")
31      public MessageEntity execute(MessageEntity entity, ProcessActionChain chain, ProcessParameters processParameters) {
32          entity.setRetries(0);
33          entity.setLastTry(null);
34          entity = messageRepository.save(entity);
35  
36          Long manualActionId = processParameters.get(MANUAL_ACTION_ID);
37          manualActionId = messageActionManualActionService.requeue(manualActionId, entity);
38          chain.addOutput(MANUAL_ACTION_ID.of(manualActionId));
39          try {
40              messageSenderService.resend(entity.getMessageId());
41          } catch (NexusException e) {
42              throw new NexusRuntimeException(
43                      NexusError.builder()
44                              .message("Could not requeue message")
45                              .extension("messageId", entity.getMessageId())
46                              .build(),
47                      e);
48          }
49          return entity;
50      }
51  
52      @Override
53      public ProcessActionConditions conditions(MessageEntity entity, ProcessParameters processParameters) {
54          return ProcessActionConditions.EMPTY;
55      }
56  }
