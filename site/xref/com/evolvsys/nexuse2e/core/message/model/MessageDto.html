68      public Duration getInternalDuration() {
69          Duration result;
70          if (created == null || finished == null) {
71              result = null;
72          } else {
73              result = Duration.between(created, finished);
74          }
75          return result;
76      }
77  
78      public Duration getCompleteDuration() {
79          Duration result;
80          if (originalCreated == null || finished == null) {
81              result = null;
82          } else {
83              result = Duration.between(originalCreated, finished);
84          }
85          return result;
86      }
