3   import java.util.Objects;
4   import java.util.function.Function;
5   import lombok.RequiredArgsConstructor;
6   
7   /**
8    * @author SvenBarden
9    * @since 24.01.2024
10   */
11  @RequiredArgsConstructor
12  final class LazyObject<K, T> {
13      private final Function<K, T> get;
14      private K key;
15      private T value;
16  
17      public T getValue(K key) {
18          if (value == null || (Objects.equals(this.key, key) == false)) {
19              if (key == null) {
20                  value = null;
21              } else {
22                  value = get.apply(key);
23              }
24              this.key = key;
25          }
26          return value;
27      }
28  }
