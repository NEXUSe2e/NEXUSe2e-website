1   package com.evolvsys.nexuse2e.core.message.service;
2   
3   import com.evolvsys.nexuse2e.core.message.context.ContextObjectFactory;
4   import com.evolvsys.nexuse2e.core.message.mapper.ContextConversationMapper;
5   import com.evolvsys.nexuse2e.core.message.model.ContextConversation;
6   import com.evolvsys.nexuse2e.core.message.repository.ConversationRepository;
7   import lombok.RequiredArgsConstructor;
8   import org.springframework.stereotype.Component;
9   
10  /**
11   * @author SvenBarden
12   * @since 24.01.2024
13   */
14  @Component
15  @RequiredArgsConstructor
16  public class ContextConversationFactory implements ContextObjectFactory<String, ContextConversation> {
17      private final ConversationRepository conversationRepository;
18      private final ContextConversationMapper contextConversationMapper;
19  
20      @Override
21      public ContextConversation create(String key) {
22          return conversationRepository
23                  .findByConversationId(key)
24                  .map(contextConversationMapper::entityToConversation)
25                  .orElseThrow(() -> new IllegalArgumentException("no conversation with name " + key + " found"));
26      }
27  }
