27  @Component
28  @RequiredArgsConstructor
29  public class ConversationPartnersBatchLoader
30          implements GraphQLBatchLoader<ConversationDto, Set<ConversationPartnerDto>> {
31      private final JPQLQueryFactory queryFactory;
32  
33      @Override
34      public CompletionStage<Map<ConversationDto, Set<ConversationPartnerDto>>> load(
35              Set<ConversationDto> keys, BatchLoaderEnvironment environment) {
36          Set<ConversationDto> nonNullKeys =
37                  keys.stream().filter(Objects::nonNull).collect(Collectors.toSet());
38  
39          CompletionStage<Map<ConversationDto, Set<ConversationPartnerDto>>> result;
40          if (CollectionUtils.isEmpty(nonNullKeys)) {
41              result = CompletableFuture.completedStage(new HashMap<>());
42          } else {
43              result = CompletableFuture.supplyAsync(() -> loadInternal(nonNullKeys));
44          }
45          return result;
46      }
47  
48      private Map<ConversationDto, Set<ConversationPartnerDto>> loadInternal(Set<ConversationDto> keys) {
49          List<Long> idKeys = keys.stream().map(IdDto::getId).distinct().toList();
50  
51          return queryFactory
52                  .from(QMessageEntity.messageEntity)
53                  .where(QMessageEntity.messageEntity.conversation.id.in(idKeys))
54                  .groupBy(
55                          QMessageEntity.messageEntity.conversation.id,
56                          QMessageEntity.messageEntity.sourcePartner.id,
57                          QMessageEntity.messageEntity.targetPartner.id)
58                  .transform(new CustomTransformer(keys));
59      }
60  
61      @RequiredArgsConstructor
62      private static class CustomTransformer
63              implements ResultTransformer<Map<ConversationDto, Set<ConversationPartnerDto>>> {
64          private final Set<ConversationDto> conversationDtos;
65  
66          @Override
67          public Map<ConversationDto, Set<ConversationPartnerDto>> transform(FetchableQuery<?, ?> query) {
68              Map<ConversationDto, Set<ConversationPartnerDto>> result = new HashMap<>();
69  
70              Map<Long, ConversationDto> conversationMap =
71                      conversationDtos.stream().collect(Collectors.toMap(IdDto::getId, Function.identity()));
72  
73              try (CloseableIterator<Tuple> iterate = query.select(
74                              QMessageEntity.messageEntity.conversation.id,
75                              QMessageEntity.messageEntity.sourcePartner.id,
76                              QMessageEntity.messageEntity.targetPartner.id)
77                      .iterate()) {
78                  while (iterate.hasNext()) {
79                      Tuple tuple = iterate.next();
80                      Long conversationId = tuple.get(QMessageEntity.messageEntity.conversation.id);
81                      Long sourcePartnerId = tuple.get(QMessageEntity.messageEntity.sourcePartner.id);
82                      Long targetPartnerId = tuple.get(QMessageEntity.messageEntity.targetPartner.id);
83  
84                      ConversationDto conversationDto = conversationMap.get(conversationId);
85  
86                      Set<ConversationPartnerDto> conversationPartnerDtos =
87                              result.computeIfAbsent(conversationDto, aLong -> new HashSet<>());
88                      conversationPartnerDtos.add(new ConversationPartnerDto(sourcePartnerId));
89                      conversationPartnerDtos.add(new ConversationPartnerDto(targetPartnerId));
90                  }
91              }
92              return result;
93          }
94      }
95  }
