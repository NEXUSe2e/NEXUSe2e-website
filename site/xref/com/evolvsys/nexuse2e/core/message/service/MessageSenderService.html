12  import org.springframework.stereotype.Service;
13  
14  /**
15   * @author SvenBarden
16   * @since 02.02.2024
17   */
18  @Service
19  @RequiredArgsConstructor
20  @Slf4j
21  public class MessageSenderService {
22      private final ContextFactory contextFactory;
23      private final OutboundMessageDispatcher outboundMessageDispatcher;
24  
25      public void resend(String messageId) throws NexusException {
26          try {
27              MessageContext messageContext = contextFactory.createForResend(messageId);
28  
29              Validate.notNull(messageContext.getSourceMessage(), "messageContext.getSourceMessage() must not be null");
30  
31              MessageType messageType = MessageType.NORMAL;
32              if (messageContext.getAcknowledgmentMessage() != null
33                      && messageId.equals(
34                              messageContext.getAcknowledgmentMessage().getId())) {
35                  messageType = MessageType.ACKNOWLEDGMENT;
36              }
37  
38              LOGGER.debug("sending message {} as {}", messageId, messageType);
39  
40              outboundMessageDispatcher.dispatch(messageContext.createExecutionContext(messageType));
41          } finally {
42              MessageLogContextHolder.clearContext();
43          }
44      }
45  }
