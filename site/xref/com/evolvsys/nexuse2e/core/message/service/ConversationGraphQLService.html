33  @GraphQLService
34  @RequiredArgsConstructor
35  public class ConversationGraphQLService {
36      private final ConversationService conversationService;
37      private final DataLoaderService dataLoaderService;
38  
39      @GraphQLQuery(name = "conversation_list", description = "List all conversations")
40      @GraphQLNonNull
41      public Page<@GraphQLNonNull ConversationDto> list(
42              Pageable pageable, @Nullable ConversationFilter conversationFilter) {
43          return conversationService.list(pageable, conversationFilter);
44      }
45  
46      @GraphQLQuery(name = "conversation_getById", description = "Gets a conversation")
47      public Optional<ConversationDto> getById(long conversationId) {
48          return conversationService.getById(conversationId);
49      }
50  
51      @GraphQLQuery(name = "conversation_countAll")
52      public long countAll() {
53          return conversationService.countAll();
54      }
55  
56      @GraphQLQuery
57      public CompletableFuture<ChoreographyDto> choreography(
58              @GraphQLContext ConversationDto conversationDto,
59              @GraphQLEnvironment ResolutionEnvironment resolutionEnvironment) {
60          return dataLoaderService.load(
61                  resolutionEnvironment,
62                  ChoreographyBatchLoader.class,
63                  conversationDto,
64                  "choreographyId",
65                  ConversationDto::getChoreographyId);
66      }
67  
68      @GraphQLQuery
69      public CompletableFuture<ActionDto> currentAction(
70              @GraphQLContext ConversationDto conversationDto,
71              @GraphQLEnvironment ResolutionEnvironment resolutionEnvironment) {
72          return dataLoaderService.load(
73                  resolutionEnvironment,
74                  ActionBatchLoader.class,
75                  conversationDto,
76                  "currentActionId",
77                  ConversationDto::getCurrentActionId);
78      }
79  
80      @GraphQLQuery
81      public CompletableFuture<Set<@GraphQLNonNull MessageDto>> messages(
82              @GraphQLContext ConversationDto conversationDto,
83              @GraphQLEnvironment ResolutionEnvironment resolutionEnvironment) {
84          return dataLoaderService.load(
85                  resolutionEnvironment, ConversationMessageBatchLoader.class, conversationDto, ConversationDto::getId);
86      }
87  
88      @GraphQLQuery
89      public CompletableFuture<PartnerDto> lastSourcePartner(
90              @GraphQLContext ConversationDto conversationDto,
91              @GraphQLEnvironment ResolutionEnvironment resolutionEnvironment) {
92          return dataLoaderService.load(
93                  resolutionEnvironment,
94                  PartnerBatchLoader.class,
95                  conversationDto,
96                  "lastSourcePartnerId",
97                  ConversationDto::getLastSourcePartnerId);
98      }
99  
100     @GraphQLQuery
101     public CompletableFuture<PartnerDto> lastTargetPartner(
102             @GraphQLContext ConversationDto conversationDto,
103             @GraphQLEnvironment ResolutionEnvironment resolutionEnvironment) {
104         return dataLoaderService.load(
105                 resolutionEnvironment,
106                 PartnerBatchLoader.class,
107                 conversationDto,
108                 "lastTargetPartnerId",
109                 ConversationDto::getLastTargetPartnerId);
110     }
111 
112     @GraphQLQuery
113     public CompletableFuture<Set<@GraphQLNonNull ConversationPartnerDto>> partners(
114             @GraphQLContext ConversationDto conversationDto,
115             @GraphQLEnvironment ResolutionEnvironment resolutionEnvironment) {
116         return dataLoaderService.load(
117                 resolutionEnvironment, ConversationPartnersBatchLoader.class, conversationDto, Function.identity());
118     }
119 
120     @GraphQLQuery
121     public CompletableFuture<PartnerDto> partner(
122             @GraphQLContext ConversationPartnerDto conversationPartnerDto,
123             @GraphQLEnvironment ResolutionEnvironment resolutionEnvironment) {
124         return dataLoaderService.load(
125                 resolutionEnvironment,
126                 PartnerBatchLoader.class,
127                 conversationPartnerDto,
128                 ConversationPartnerDto::getPartnerId);
129     }
