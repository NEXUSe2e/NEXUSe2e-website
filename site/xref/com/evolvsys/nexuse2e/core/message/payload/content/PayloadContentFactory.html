21  @Component
22  @RequiredArgsConstructor
23  @Slf4j
24  @SuppressFBWarnings("EXS_EXCEPTION_SOFTENING_NO_CONSTRAINTS")
25  public class PayloadContentFactory implements InitializingBean {
26      private static DataSize IN_MEMORY_THRESHOLD = DataSize.ofKilobytes(10);
27      private final NEXUSe2eConfigurationProperties nexuSe2eConfigurationProperties;
28      private final BinaryRepository binaryRepository;
29  
30      private static boolean doInMemory(long size) {
31          return size < IN_MEMORY_THRESHOLD.toBytes();
32      }
33  
34      public PayloadContent createFromEntity(PayloadEntity payloadEntity) {
35          PayloadContent payloadContent;
36          if (doInMemory(payloadEntity.getSize())) {
37              LOGGER.trace(
38                      "create new caching entity payload content from entity {} and binaryId {} with size {}",
39                      payloadEntity.getId(),
40                      payloadEntity.getContent().getId(),
41                      DataSize.ofBytes(payloadEntity.getSize()));
42              payloadContent = CachingFileEntityContent.of(payloadEntity, binaryRepository);
43          } else {
44              LOGGER.trace(
45                      "create new entity payload content from entity {} and binaryId {} with size {}",
46                      payloadEntity.getId(),
47                      payloadEntity.getContent().getId(),
48                      payloadEntity.getSize());
49              payloadContent = PayloadEntityContent.of(payloadEntity, binaryRepository);
50          }
51          return payloadContent;
52      }
53  
54      static PayloadContent create(InputStream inputStream) {
55          PayloadContent payloadContent;
56          try (inputStream) {
57              try {
58                  LOGGER.trace("create new fileSystem payload content from input stream");
59                  payloadContent = FileSystemContent.of(inputStream);
60              } catch (IOException e) {
61                  throw new NexusRuntimeException(
62                          NexusError.builder()
63                                  .message("could not create file payload from inputStream")
64                                  .build(),
65                          e);
66              }
67          } catch (IOException e) {
68              throw new NexusRuntimeException(
69                      NexusError.builder()
70                              .message("could not close input stream for payloadContent")
71                              .build(),
72                      e);
73          }
74          return payloadContent;
75      }
76  
77      static PayloadContent create(InputStream inputStream, long length) {
78          PayloadContent payloadContent;
79          try (inputStream) {
80              if (doInMemory(length)) {
81                  try {
82                      LOGGER.trace(
83                              "create new inMemory payload content from input stream with length {}",
84                              DataSize.ofBytes(length));
85                      payloadContent = InMemoryContent.of(inputStream.readAllBytes());
86                  } catch (IOException e) {
87                      throw new NexusRuntimeException(
88                              NexusError.builder()
89                                      .message("could not create inMemory payload from inputStream")
90                                      .extension("size", length)
91                                      .build(),
92                              e);
93                  }
94              } else {
95                  try {
96                      LOGGER.trace(
97                              "create new fileSystem payload content from input stream with length {}",
98                              DataSize.ofBytes(length));
99                      payloadContent = FileSystemContent.of(inputStream);
100                 } catch (IOException e) {
101                     throw new NexusRuntimeException(
102                             NexusError.builder()
103                                     .message("could not create file payload from inputStream")
104                                     .extension("size", length)
105                                     .build(),
106                             e);
107                 }
108             }
109         } catch (IOException e) {
110             throw new NexusRuntimeException(
111                     NexusError.builder()
112                             .message("could not close input stream for payloadContent")
113                             .build(),
114                     e);
115         }
116         return payloadContent;
117     }
118 
119     static PayloadContent create(byte[] content) {
120         PayloadContent payloadContent;
121         if (doInMemory(content.length)) {
122             LOGGER.trace(
123                     "create new inMemory payload content from byte array with length {}",
124                     DataSize.ofBytes(content.length));
125             payloadContent = InMemoryContent.of(content);
126         } else {
127             LOGGER.trace(
128                     "create new fileSystem payload content from byte array with length {}",
129                     DataSize.ofBytes(content.length));
130             payloadContent = FileSystemContent.of(content);
131         }
132         return payloadContent;
133     }
134 
135     static void setInMemoryThreshold(DataSize dataSize) {
136         IN_MEMORY_THRESHOLD = dataSize;
137     }
138 
139     @Override
140     public void afterPropertiesSet() throws Exception {
141         setInMemoryThreshold(nexuSe2eConfigurationProperties.getInMemoryPayloadThreshold());
142     }
