11  import lombok.AccessLevel;

14  /**
15   * @author SvenBarden
16   * @since 22.02.2024
17   */
18  @RequiredArgsConstructor(access = AccessLevel.PROTECTED)
19  @SuppressFBWarnings({"EXS_EXCEPTION_SOFTENING_NO_CONSTRAINTS", "EXS_EXCEPTION_SOFTENING_NO_CHECKED"})
20  sealed class PayloadEntityContent implements PayloadContent permits CachingFileEntityContent {
21      private final Long binaryId;
22      private final long size;
23      private final BinaryRepository binaryRepository;
24  
25      static PayloadEntityContent of(PayloadEntity payloadEntity, BinaryRepository binaryRepository) {
26          return new PayloadEntityContent(payloadEntity.getContent().getId(), payloadEntity.getSize(), binaryRepository);
27      }
28  
29      private InputStream loadContent() {
30          try {
31              return binaryRepository
32                      .findById(binaryId)
33                      .orElseThrow(() -> new NexusRuntimeException(NexusError.builder()
34                              .message("binary file for payload could not be found")
35                              .extension("id", binaryId)
36                              .build()))
37                      .getContent()
38                      .getBinaryStream();
39          } catch (SQLException e) {
40              throw new NexusRuntimeException(
41                      NexusError.builder()
42                              .message("could not load payload binary entity")
43                              .extension("id", binaryId)
44                              .build(),
45                      e);
46          }
47      }
48  
49      @Override
50      public Long getId() {
51          return binaryId;
52      }
53  
54      @Override
55      public InputStream getInputStream() {
56          return loadContent();
57      }
58  
59      @Override
60      public byte[] getByteArray() {
61          try (InputStream inputStream = loadContent()) {
62              return inputStream.readAllBytes();
63          } catch (IOException e) {
64              throw new NexusRuntimeException(
65                      NexusError.builder()
66                              .message("could not get byte array from binary content")
67                              .extension("id", binaryId)
68                              .build(),
69                      e);
70          }
71      }
72  
73      @Override
74      public long getSize() {
75          return size;
76      }
77  }
