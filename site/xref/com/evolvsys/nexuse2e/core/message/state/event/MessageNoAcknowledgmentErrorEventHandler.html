20   * @author SvenBarden
21   * @since 02.04.2024
22   */
23  @Component
24  @RequiredArgsConstructor
25  @Slf4j
26  class MessageNoAcknowledgmentErrorEventHandler implements MessageEventHandler {
27      private final MessageRepository messageRepository;
28      private final ConversationRepository conversationRepository;
29  
30      @Override
31      @Transactional(rollbackFor = NexusException.class)
32      public void handle(ExecutionContext executionContext) throws NexusException {
33          Message message = executionContext.getMessage();
34  
35          MessageEntity messageEntity = messageRepository
36                  .findByMessageId(message.getId())
37                  .orElseThrow(() -> new NexusException(NexusError.builder().build()));
38  
39          messageEntity.setStatus(MessageStatus.FAILED_MISSING_ACKNOWLEDGMENT);
40          ConversationEntity conversationEntity = messageEntity.getConversation();
41          conversationEntity.setStatus(ConversationStatus.FAILED);
42  
43          LOGGER.trace("setting message {} to error status {}", message.getId(), messageEntity.getStatus());
44  
45          messageRepository.save(messageEntity);
46          conversationRepository.save(conversationEntity);
47      }
48  
49      @Override
50      public MessageEventType type() {
51          return MessageEventType.WAITING_FOR_ACK_ERROR;
52      }
53  }
