19  @Mapper(
20          config = DefaultMapperConfiguration.class,
21          uses = {ActionMapper.class, ChoreographyMapper.class, PartnerMapper.class})
22  public abstract class ConversationMapper extends BaseDtoEntityMapper<ConversationDto, ConversationEntity> {
23      private ConversationRepository conversationRepository;
24  
25      @Autowired
26      protected void config(ConversationRepository conversationRepository) {
27          this.conversationRepository = conversationRepository;
28      }
29  
30      @Override
31      public ConversationEntity idToEntity(Long id, InterceptingMappingContext interceptingMappingContext) {
32          return idToEntity(id, interceptingMappingContext, conversationRepository::findById);
33      }
34  
35      @Mapping(target = "lastTargetPartnerId", source = "lastTargetPartner.id")
36      @Mapping(target = "lastSourcePartnerId", source = "lastSourcePartner.id")
37      @Mapping(target = "currentActionId", source = "currentAction.id")
38      @Mapping(target = "choreographyId", source = "choreography.id")
39      @InheritConfiguration
40      @Override
41      protected abstract ConversationDto entityToDto(
42              ConversationEntity entity, @Context InterceptingMappingContext interceptingMappingContext);
43  
44      @Mapping(target = "lastTargetPartner", source = "lastTargetPartnerId")
45      @Mapping(target = "lastSourcePartner", source = "lastSourcePartnerId")
46      @Mapping(target = "currentAction", source = "currentActionId")
47      @Mapping(target = "choreography", source = "choreographyId")
48      @Mapping(target = "created", ignore = true)
49      @Mapping(target = "lastUpdated", ignore = true)
50      @InheritConfiguration
51      @Override
52      protected abstract void updateEntity(
53              ConversationDto dto,
54              @MappingTarget ConversationEntity entity,
55              @Context InterceptingMappingContext interceptingMappingContext);
56  }
