1   package com.evolvsys.nexuse2e.core.message.queue;
2   
3   import com.evolvsys.nexuse2e.core.utils.ErrorUtils;
4   import lombok.RequiredArgsConstructor;
5   import lombok.extern.slf4j.Slf4j;
6   import org.quartz.DisallowConcurrentExecution;
7   import org.quartz.JobExecutionContext;
8   import org.quartz.JobExecutionException;
9   import org.springframework.scheduling.quartz.QuartzJobBean;
10  
11  /**
12   * @author SvenBarden
13   * @since 05.03.2024
14   */
15  @DisallowConcurrentExecution
16  @RequiredArgsConstructor
17  @Slf4j
18  public class MessageQueueJob extends QuartzJobBean {
19      private final MessageQueueService messageQueueService;
20  
21      @Override
22      protected void executeInternal(JobExecutionContext context) throws JobExecutionException {
23          try {
24              LOGGER.trace(
25                      "executing message queue job starting at {}",
26                      context.getTrigger().getStartTime());
27              messageQueueService.run();
28              LOGGER.trace(
29                      "finished message queue, wait till {}", context.getTrigger().getNextFireTime());
30          } catch (Throwable e) {
31              LOGGER.error("error while running message queue job: {}", ErrorUtils.aggregate(e), e);
32          }
33      }
34  }
