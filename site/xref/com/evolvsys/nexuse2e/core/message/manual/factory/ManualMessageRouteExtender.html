16   * @author SvenBarden

19  @Component
20  @RequiredArgsConstructor
21  @Order(500)
22  class ManualMessageRouteExtender implements ManualMessageContextExtender {
23      private final ContextRouteFactory contextRouteFactory;
24  
25      @Override
26      public void extend(MessageContext messageContext, ManualMessageDto manualMessageDto) throws NexusException {
27          ContextRoute contextRoute = contextRouteFactory
28                  .getRoute(new RouteKey(
29                          manualMessageDto.getChoreographyId(),
30                          manualMessageDto.getActionId(),
31                          messageContext.getSourcePartnerId(),
32                          messageContext.getTargetPartnerId()))
33                  .orElseThrow(() -> new NexusRuntimeException(
34                          NexusError.builder().message("no route found").build()));
35          messageContext.setRoute(contextRoute);
36      }
37  }
