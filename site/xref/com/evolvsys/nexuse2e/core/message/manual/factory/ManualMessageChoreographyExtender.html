13  import org.springframework.stereotype.Component;
14  
15  /**
16   * @author SvenBarden
17   * @since 25.07.2024
18   */
19  @Component
20  @RequiredArgsConstructor
21  @Order(200)
22  class ManualMessageChoreographyExtender implements ManualMessageContextExtender {
23      private final ChoreographyRepository choreographyRepository;
24      private final ActionRepository actionRepository;
25  
26      @Override
27      public void extend(MessageContext messageContext, ManualMessageDto manualMessageDto) {
28          ChoreographyEntity choreographyEntity = choreographyRepository
29                  .findById(manualMessageDto.getChoreographyId())
30                  .orElseThrow(() -> new NexusRuntimeException(NexusError.builder()
31                          .message("Could not find choreography for manual message")
32                          .extension("id", manualMessageDto.getChoreographyId())
33                          .build()));
34          ActionEntity actionEntity = actionRepository
35                  .findById(manualMessageDto.getActionId())
36                  .orElseThrow(() -> new NexusRuntimeException(NexusError.builder()
37                          .message("Could not find action for manual message")
38                          .extension("id", manualMessageDto.getActionId())
39                          .build()));
40  
41          messageContext.setChoreographyId(choreographyEntity.getChoreographyId());
42          messageContext.setActionId(actionEntity.getActionId());
43      }
44  }
