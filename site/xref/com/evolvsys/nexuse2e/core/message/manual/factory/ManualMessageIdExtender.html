14  @Component
15  @Order(300)
16  class ManualMessageIdExtender implements ManualMessageContextExtender {
17      @Override
18      public void extend(MessageContext messageContext, ManualMessageDto manualMessageDto) throws NexusException {
19          if (StringUtils.isNotBlank(manualMessageDto.getMessageId())) {
20              messageContext.getSourceMessage().setId(manualMessageDto.getMessageId());
21          }
22          if (StringUtils.isNotBlank(manualMessageDto.getConversationId())) {
23              messageContext.getSourceMessage().setConversationId(manualMessageDto.getConversationId());
24          }
25      }
26  }
