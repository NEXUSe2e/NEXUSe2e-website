28  @Service
29  @RequiredArgsConstructor
30  @Slf4j
31  public class ManualMessageService {
32      private final NEXUSe2eConfigurationProperties configurationProperties;
33      private final CurrentUserService currentUserService;
34  
35      private final ManualMessageFactory manualMessageFactory;
36      private final ManualMessageDispatcher manualMessageDispatcher;
37  
38      @SuppressFBWarnings("EXS_EXCEPTION_SOFTENING_NO_CONSTRAINTS")
39      public List<String> getDefaultFiles() throws NexusException {
40          return ResourceUtils.list(configurationProperties.getDefaultManualMessageFiles()).stream()
41                  .filter(resource -> resource.getFilename() != null)
42                  .map(Resource::getFilename)
43                  .sorted()
44                  .toList();
45      }
46  
47      public ManualMessageResponseListDto send(ManualMessageDto manualMessageDto) {
48          currentUserService.checkSystemPermissions("NEXUS.SEND_MANUAL_MESSAGE");
49          LOGGER.debug("prepare sending manual message {}", manualMessageDto);
50  
51          ManualMessageResponseListDto.ManualMessageResponseListDtoBuilder responseListBuilder =
52                  ManualMessageResponseListDto.builder().status(ManualMessageStatus.SUCCESS);
53          for (int i = 1; i <= manualMessageDto.getRepeat(); i++) {
54              LOGGER.trace("sending message number {}", i);
55              try {
56                  MessageContext messageContext = manualMessageFactory.create(manualMessageDto);
57  
58                  ManualMessageResponseDto messageResponseDto = manualMessageDispatcher.send(messageContext);
59                  responseListBuilder.messageResponse(messageResponseDto);
60  
61                  if (messageResponseDto.getStatus() != ManualMessageStatus.SUCCESS) {
62                      responseListBuilder.status(ManualMessageStatus.FAILURE);
63                  }
64              } catch (Exception e) {
65                  ManualMessageResponseDto.ManualMessageResponseDtoBuilder responseBuilder =
66                          ManualMessageResponseDto.builder();
67                  responseBuilder.status(ManualMessageStatus.ERROR);
68                  if (e instanceof NexusErrorContainer errorContainer) {
69                      responseBuilder.message(errorContainer.getError().getMessage());
70                  } else {
71                      responseBuilder.message(e.getMessage());
72                  }
73                  responseBuilder.messageAggregated(ErrorUtils.aggregate(e));
74                  responseListBuilder.messageResponse(responseBuilder.build());
75                  responseListBuilder.status(ManualMessageStatus.FAILURE);
76              } finally {
77                  MessageLogContextHolder.clearContext();
78              }
79          }
80  
81          return responseListBuilder.build();
82      }
83  }
