19  /**
20   * @author SvenBarden
21   * @since 25.07.2024
22   */
23  @Component
24  @RequiredArgsConstructor
25  public class ManualMessageDispatcher {
26      private final OutboundMessageDispatcher outboundMessageDispatcher;
27      private final MessageEventService messageEventService;
28      private final MessageResolverValidator messageResolverValidator;
29      private final ContextFactory contextFactory;
30      private final MDCService mdcService;
31      private final MDCIdGenerator mdcIdGenerator;
32  
33      public ManualMessageResponseDto send(MessageContext messageContext) {
34          ManualMessageResponseDto.ManualMessageResponseDtoBuilder responseBuilder = ManualMessageResponseDto.builder();
35  
36          String originalMdcId = mdcService.getId();
37          try {
38              mdcService.setId(mdcIdGenerator.newId());
39  
40              ManualMessageResolver manualMessageResolver = new ManualMessageResolver(
41                      MessageType.NORMAL, messageResolverValidator, messageEventService, contextFactory);
42              manualMessageResolver.resolve(messageContext);
43  
44              outboundMessageDispatcher.dispatch(messageContext.createExecutionContext(MessageType.NORMAL));
45  
46              responseBuilder.status(ManualMessageStatus.SUCCESS);
47              responseBuilder.message("success");
48          } catch (Exception e) {
49              responseBuilder.status(ManualMessageStatus.ERROR);
50              if (e instanceof NexusErrorContainer errorContainer) {
51                  responseBuilder.message(errorContainer.getError().getMessage());
52              } else {
53                  responseBuilder.message(e.getMessage());
54              }
55              responseBuilder.messageAggregated(ErrorUtils.aggregate(e));
56          } finally {
57              if (originalMdcId != null) {
58                  mdcService.setId(originalMdcId);
59              }
60          }
61          return responseBuilder.build();
62      }
63  }
