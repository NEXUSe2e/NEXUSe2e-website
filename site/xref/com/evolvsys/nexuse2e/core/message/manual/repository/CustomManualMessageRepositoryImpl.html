22  @Repository
23  @RequiredArgsConstructor
24  class CustomManualMessageRepositoryImpl implements CustomManualMessageRepository {
25      private final JPQLQueryFactory queryFactory;
26  
27      @Override
28      public Collection<ManualMessageRoutingInfoDto> getRoutingInfos(Long sourcePartnerId, Long targetPartnerId) {
29          QChoreographyEntity choreographyEntity = QChoreographyEntity.choreographyEntity;
30          QActionEntity actionEntity = QActionEntity.actionEntity;
31          QRouteEntity routeEntity = QRouteEntity.routeEntity;
32  
33          List<ManualMessageRoutingInfo> choreographyResults = queryFactory
34                  .select(new QCustomManualMessageRepositoryImpl_ManualMessageRoutingInfo(
35                          choreographyEntity.id,
36                          choreographyEntity.choreographyId,
37                          actionEntity.id,
38                          actionEntity.actionId))
39                  .from(choreographyEntity)
40                  .join(choreographyEntity.routes, routeEntity)
41                  .on(getPartnerPredicate(routeEntity, sourcePartnerId, targetPartnerId))
42                  .join(choreographyEntity.actions, actionEntity)
43                  .fetch();
44  
45          List<ManualMessageRoutingInfo> actionResults = queryFactory
46                  .select(new QCustomManualMessageRepositoryImpl_ManualMessageRoutingInfo(
47                          choreographyEntity.id,
48                          choreographyEntity.choreographyId,
49                          actionEntity.id,
50                          actionEntity.actionId))
51                  .from(choreographyEntity)
52                  .join(choreographyEntity.actions, actionEntity)
53                  .join(actionEntity.routes, routeEntity)
54                  .on(getPartnerPredicate(routeEntity, sourcePartnerId, targetPartnerId))
55                  .fetch();
56  
57          return StreamEx.of(choreographyResults)
58                  .append(actionResults)
59                  .toMap(
60                          CustomManualMessageRepositoryImpl.ManualMessageRoutingInfo::choreographyId,
61                          CustomManualMessageRepositoryImpl.ManualMessageRoutingInfo::toDto,
62                          this::merge)
63                  .values();
64      }
65  
66      private BooleanExpression getPartnerPredicate(
67              QRouteEntity routeEntity, Long sourcePartnerId, Long targetPartnerId) {
68          BooleanExpression firstPartnerPredicate = Expressions.allOf(
69                  routeEntity.firstPartner.id.eq(sourcePartnerId), routeEntity.secondPartner.id.eq(targetPartnerId));
70          BooleanExpression secondPartnerPredicate = Expressions.allOf(
71                  routeEntity.firstPartner.id.eq(targetPartnerId), routeEntity.secondPartner.id.eq(sourcePartnerId));
72  
73          return Expressions.anyOf(firstPartnerPredicate, secondPartnerPredicate);
74      }
75  
76      private ManualMessageRoutingInfoDto merge(ManualMessageRoutingInfoDto left, ManualMessageRoutingInfoDto right) {
77          right.getActions().addAll(left.getActions());
78          return right;
79      }
80  
81      public record ManualMessageRoutingInfo(
82              Long choreographyId, String choreographyName, Long actionId, String actionName) {
83          @QueryProjection
84          public ManualMessageRoutingInfo {}
85  
86          public ManualMessageRoutingInfoDto toDto() {
87              ManualMessageRoutingInfoDto dto = new ManualMessageRoutingInfoDto();
88              dto.setId(choreographyId);
89              dto.setName(choreographyName);
90              ManualMessageActionInfoDto actionDto = new ManualMessageActionInfoDto();
91              actionDto.setId(actionId);
92              actionDto.setName(actionName);
93              dto.getActions().add(actionDto);
94              return dto;
95          }
96      }
97  }
