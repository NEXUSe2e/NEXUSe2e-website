22   * @author SvenBarden

25  @Component
26  @RequiredArgsConstructor
27  @Slf4j
28  class ManualMessagePayloadFactory {
29      private final FileRepository fileRepository;
30      private final NEXUSe2eConfigurationProperties configurationProperties;
31  
32      public ContextPayload buildContextPayload(Long fileId) throws SQLException {
33          FileEntity fileEntity = fileRepository
34                  .getFileEntityById(fileId)
35                  .orElseThrow(() -> new ObjectNotFoundException(FileDto.class, fileId));
36  
37          ContextPayload contextPayload = new ContextPayload();
38          contextPayload.setContent(
39                  PayloadContent.of(fileEntity.getFileContent().getContent().getBinaryStream(), fileEntity.getSize()));
40          contextPayload.setEncoding("utf-8");
41          contextPayload.setSize(fileEntity.getSize());
42          contextPayload.setContentType(fileEntity.getMimeType());
43          contextPayload.setContentId(fileEntity.getFileName());
44  
45          return contextPayload;
46      }
47  
48      public ContextPayload buildContextPayload(String fileName) throws IOException, NexusException {
49          LOGGER.debug("filter manual files for {}", fileName);
50          Resource manualMessageFile = ResourceUtils.list(configurationProperties.getDefaultManualMessageFiles()).stream()
51                  .filter(resource -> fileName.equals(resource.getFilename()))
52                  .findFirst()
53                  .orElse(null);
54          if (manualMessageFile == null) {
55              LOGGER.error(
56                      "file {} could not be found in directory {}",
57                      fileName,
58                      configurationProperties.getDefaultManualMessageFiles());
59              throw new GeneralErrorException("no manual message files found");
60          }
61  
62          long fileLength = manualMessageFile.contentLength();
63  
64          ContextPayload contextPayload = new ContextPayload();
65          contextPayload.setContent(PayloadContent.of(manualMessageFile.getInputStream(), fileLength));
66          contextPayload.setEncoding("utf-8");
67          contextPayload.setSize(fileLength);
68          contextPayload.setContentType(new Tika().detect(fileName));
69          contextPayload.setContentId(fileName);
70  
71          return contextPayload;
72      }
73  }
