7   import java.util.concurrent.CompletableFuture;
8   import java.util.concurrent.Executor;
9   import java.util.function.Supplier;
10  import lombok.RequiredArgsConstructor;
11  import lombok.SneakyThrows;
12  import lombok.extern.slf4j.Slf4j;
13  import org.springframework.stereotype.Component;
14  
15  /**
16   * @author SvenBarden
17   * @since 18.10.2023
18   */
19  @Component
20  @Slf4j
21  @RequiredArgsConstructor
22  public class MessageRunner {
23      @MessageExecutor
24      private final Executor messageExecutor;
25  
26      private Supplier<MessageContext> createSupplier(MessageContext messageContext, Supplier<MessageContext> delegate) {
27          return () -> {
28              // assign the message context from the parent thread to the new created thread
29              MessageLogContextHolder.getContext().setMessageContext(messageContext);
30              MessageContext newContext = delegate.get();
31              MessageLogContextHolder.clearContext();
32              return newContext;
33          };
34      }
35  
36      public CompletableFuture<MessageContext> createFuture(
37              MessageContext messageContext, Supplier<MessageContext> messageContextSupplier) {
38          LOGGER.debug("creating future for {}", messageContext);
39  
40          return CompletableFuture.supplyAsync(createSupplier(messageContext, messageContextSupplier), messageExecutor);
41      }
42  
43      public CompletableFuture<MessageContext> createWrappedFuture(
44              MessageContext messageContext, SupplierWrapper messageContextSupplier) {
45          LOGGER.debug("creating future for {}", messageContext);
46  
47          Supplier<MessageContext> supplier = new Supplier<>() {
48              @SneakyThrows
49              @Override
50              public MessageContext get() {
51                  return messageContextSupplier.get();
52              }
53          };
54  
55          return CompletableFuture.supplyAsync(createSupplier(messageContext, supplier), messageExecutor);
56      }
57  
58      @FunctionalInterface
59      public interface SupplierWrapper {
60          MessageContext get() throws NexusException;
61      }
62  }
