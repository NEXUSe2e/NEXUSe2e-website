10  import lombok.RequiredArgsConstructor;
11  import org.hibernate.engine.jdbc.BlobProxy;
12  import org.springframework.stereotype.Component;
13  
14  /**
15   * @author SvenBarden
16   * @since 31.01.2024
17   */
18  @Component
19  @RequiredArgsConstructor
20  class PayloadMetaDataConverter extends TypeSafeMetaDataConverter<ContextPayload> {
21      private final PayloadContentFactory payloadContentFactory;
22  
23      @Override
24      protected void updateEntityInternal(ContextPayload value, MetaDataEntity metaDataEntity) {
25          PayloadEntity payloadEntity = metaDataEntity.getBinaryValue();
26  
27          if (payloadEntity == null) {
28              metaDataEntity.setBinaryValue(payloadEntity = new PayloadEntity());
29          }
30  
31          BinaryEntity binaryEntity = payloadEntity.getContent();
32          if (binaryEntity == null) {
33              payloadEntity.setContent(binaryEntity = new BinaryEntity());
34          }
35  
36          binaryEntity.setContent(BlobProxy.generateProxy(value.getContent().getInputStream(), value.getSize()));
37  
38          payloadEntity.setSize(value.getSize());
39          payloadEntity.setContentType(value.getContentType());
40          payloadEntity.setEncoding(value.getEncoding());
41          payloadEntity.setContent(binaryEntity);
42      }
43  
44      @Override
45      @SuppressFBWarnings("EXS_EXCEPTION_SOFTENING_NO_CONSTRAINTS")
46      protected ContextPayload getValueInternal(MetaDataEntity metaDataEntity) {
47          ContextPayload payload = new ContextPayload();
48  
49          PayloadEntity payloadEntity = metaDataEntity.getBinaryValue();
50  
51          payload.setContentType(payloadEntity.getContentType());
52          payload.setSize(payloadEntity.getSize());
53          payload.setEncoding(payloadEntity.getEncoding());
54          payload.setContent(payloadContentFactory.createFromEntity(payloadEntity));
55  
56          return payload;
57      }
58  
59      @Override
60      public MetaDataType type() {
61          return MetaDataType.PAYLOAD;
62      }
63  }
