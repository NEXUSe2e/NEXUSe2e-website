1   package com.evolvsys.nexuse2e.core.message.metadata;

3   import com.evolvsys.nexuse2e.core.exception.NexusError;
4   import com.evolvsys.nexuse2e.core.exception.NexusRuntimeException;
5   import com.evolvsys.nexuse2e.core.model.MetaDataType;
6   import com.evolvsys.nexuse2e.core.model.entity.MetaDataEntity;
7   import java.util.List;
8   import java.util.Optional;
9   import lombok.RequiredArgsConstructor;
10  import lombok.extern.slf4j.Slf4j;
11  import org.apache.commons.lang3.Validate;
12  import org.springframework.stereotype.Service;
13  
14  /**
15   * @author SvenBarden
16   * @since 31.01.2024
17   */
18  @Service
19  @RequiredArgsConstructor
20  @Slf4j
21  public class MetaDataConverterService {
22      private final List<MetaDataConverter> metaDataConverters;
23  
24      private Optional<MetaDataConverter> getConverter(MetaDataType metaDataType) {
25          return metaDataConverters.stream()
26                  .filter(metaDataConverter -> metaDataConverter.type() == metaDataType)
27                  .findFirst();
28      }
29  
30      private Optional<MetaDataConverter> getConverter(Object value) {
31          return metaDataConverters.stream()
32                  .filter(metaDataConverter -> metaDataConverter.supports(value.getClass()))
33                  .findFirst();
34      }
35  
36      private void updateNullValue(MetaDataEntity metaDataEntity) {
37          metaDataEntity.setLongValue(null);
38          metaDataEntity.setStringValue(null);
39          metaDataEntity.setDateTimeValue(null);
40          metaDataEntity.setBinaryValue(null);
41          metaDataEntity.setBooleanValue(null);
42          metaDataEntity.setBigDecimalValue(null);
43      }
44  
45      public void updateEntity(Object value, MetaDataEntity metaDataEntity) {
46          Validate.notNull(value, "value must not be null");
47          Validate.notNull(metaDataEntity, "metaDataEntity must not be null");
48  
49          updateNullValue(metaDataEntity);
50          MetaDataConverter metaDataConverter = getConverter(value)
51                  .orElseThrow(() -> new NexusRuntimeException(NexusError.builder()
52                          .message("no metaDataConverter for class found")
53                          .extension("class", value.getClass().getName())
54                          .build()));
55          LOGGER.trace("converting {} with {}", value.getClass(), metaDataConverter);
56  
57          metaDataConverter.updateEntity(value, metaDataEntity);
58          metaDataEntity.setType(metaDataConverter.type());
59      }
60  
61      public Object getValue(MetaDataEntity metaDataEntity) {
62          return getConverter(metaDataEntity.getType())
63                  .map(metaDataConverter -> metaDataConverter.getValue(metaDataEntity))
64                  .orElseThrow(() -> new NexusRuntimeException(NexusError.builder()
65                          .message("no metaDataConverter found for type")
66                          .extension("type", metaDataEntity.getType().name())
67                          .build()));
68      }
69  }
