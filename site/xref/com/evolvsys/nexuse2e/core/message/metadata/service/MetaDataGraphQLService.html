20   * @author Florian Drees

23  @GraphQLService
24  @RequiredArgsConstructor
25  @Slf4j
26  public class MetaDataGraphQLService {
27      private final DataLoaderService dataLoaderService;
28      private final MetaDataService metaDataService;
29  
30      @GraphQLQuery
31      public CompletableFuture<PayloadDto> binaryValue(
32              @GraphQLContext MetaDataDto metaDataDto, @GraphQLEnvironment ResolutionEnvironment resolutionEnvironment) {
33          String messageId = metaDataDto.getParentMessageId();
34  
35          return dataLoaderService
36                  .load(resolutionEnvironment, PayloadBatchLoader.class, metaDataDto, MetaDataDto::getPayloadId)
37                  .thenApply(payloadDto -> updateDownloadName(messageId, metaDataDto, payloadDto));
38      }
39  
40      @GraphQLQuery(name = "metadata_keyList", description = "List all metadata key values")
41      @GraphQLNonNull
42      public MetaDataKeyListDto metaDataKeyList() {
43          MetaDataKeyListDto metaDataKeyListDto = new MetaDataKeyListDto();
44          metaDataKeyListDto.setMetaDataKeys(metaDataService.getMetaDataKeyList());
45          return metaDataKeyListDto;
46      }
47  
48      private static PayloadDto updateDownloadName(String messageId, MetaDataDto metaDataDto, PayloadDto payloadDto) {
49          if (payloadDto != null) {
50              payloadDto.setDownloadName(getDownloadName(messageId, metaDataDto, payloadDto));
51          }
52          return payloadDto;
53      }
54  
55      private static String getDownloadName(String messageId, MetaDataDto metaDataDto, PayloadDto payloadDto) {
56          String result;
57          if (messageId == null) {
58              result = metaDataDto.getName() + "_" + metaDataDto.getIndex();
59          } else {
60              result = messageId + "_" + metaDataDto.getName() + "_" + metaDataDto.getIndex();
61          }
62          try {
63              result += TikaConfig.getDefaultConfig()
64                      .getMimeRepository()
65                      .forName(payloadDto.getContentType())
66                      .getExtension();
67          } catch (MimeTypeException e) {
68              LOGGER.error("could not get extension for payload {} in message {}", payloadDto, messageId, e);
69          }
70          return result;
71      }
72  }
