31  @Slf4j
32  @Service
33  @RequiredArgsConstructor
34  public class ContextMessageService {
35      private final MessageRepository messageRepository;
36      private final ChoreographyRepository choreographyRepository;
37      private final ConversationRepository conversationRepository;
38      private final ActionRepository actionRepository;
39      private final LockService lockService;
40      private final ContextMessageMapper contextMessageMapper;
41  
42      private String getMessageLockKey(String messageId) {
43          return "message_" + messageId;
44      }
45  
46      private String getConversationLockKey(String conversationId) {
47          return "conversation_" + conversationId;
48      }
49  
50      private ActionEntity getActionEntity(Action action) {
51          return actionRepository
52                  .findByActionIdAndChoreography_ChoreographyId(action.getActionId(), action.getChoreographyId())
53                  .stream()
54                  .filter(actionEntity ->
55                          actionEntity.getChoreography().getChoreographyId().equals(action.getChoreographyId())
56                                  && actionEntity.getActionId().equals(action.getActionId()))
57                  .findFirst()
58                  .orElseThrow(() -> new NexusRuntimeException(NexusError.builder()
59                          .message("action not found")
60                          .extension("action", action.getActionId())
61                          .extension("choreography", action.getChoreographyId())
62                          .build()));
63      }
64  
65      private ConversationEntity createConversation(Message message, Action action) {
66          ConversationEntity conversationEntity = new ConversationEntity();
67          conversationEntity.setConversationId(message.getConversationId());
68          conversationEntity.setChoreography(
69                  choreographyRepository.findByChoreographyId(action.getChoreographyId()).stream()
70                          .filter(choreographyEntity ->
71                                  choreographyEntity.getChoreographyId().equals(action.getChoreographyId()))
72                          .findFirst()
73                          .orElseThrow(() -> new NexusRuntimeException(NexusError.builder()
74                                  .message("choreography not found")
75                                  .extension("choreographyId", action.getChoreographyId())
76                                  .build())));
77          conversationEntity.setCurrentAction(getActionEntity(action));
78          return conversationEntity;
79      }
80  
81      private MessageEntity create(Message message) {
82          LOGGER.trace("create new messageEntity with id {}", message.getId());
83          MessageEntity messageEntity = new MessageEntity();
84          messageEntity.setMessageId(message.getId());
85          messageEntity.setType(message.getType());
86          return messageEntity;
87      }
88  
89      private Message entityToMessage(MessageEntity messageEntity) {
90          return contextMessageMapper.entityToMessage(messageEntity);
91      }
92  
93      public Optional<Message> getMessage(String messageId) {
94          return messageRepository.findByMessageId(messageId).map(this::entityToMessage);
95      }
96  
97      @Transactional
98      public void sent(Message message) {
99          lockService.execute(getMessageLockKey(message.getId()), () -> {
100             MessageEntity messageEntity =
101                     messageRepository.findByMessageId(message.getId()).orElseGet(MessageEntity::new);
102             messageEntity.setMessageId(message.getId());
103             messageEntity.setStatus(MessageStatus.TRANSFERRED);
104             messageEntity.setSent(ZonedDateTime.now());
105             messageRepository.save(messageEntity);
106         });
107     }
108 
109     @Transactional
110     public void save(Message message, Action action) {
111         LOGGER.debug("saving message {}", message);
112 
113         String conversationId = message.getConversationId();
114         ConversationEntity conversationEntity = lockService.evaluate(getConversationLockKey(conversationId), () -> {
115             ConversationEntity internalConversationEntity = conversationRepository
116                     .findByConversationId(conversationId)
117                     .orElseGet(() -> createConversation(message, action));
118             LOGGER.trace("setting currentAction to {} in conversation {}", action.getActionId(), conversationId);
119             internalConversationEntity.setCurrentAction(getActionEntity(action));
120             return conversationRepository.save(internalConversationEntity);
121         });
122 
123         lockService.execute(getMessageLockKey(message.getId()), () -> {
124             MessageEntity messageEntity =
125                     messageRepository.findByMessageId(message.getId()).orElseGet(() -> create(message));
126             boolean newMessage = messageEntity.getId() == null;
127 
128             contextMessageMapper.updateEntity(message, messageEntity);
129 
130             messageEntity.setConversation(conversationEntity);
131 
132             messageEntity.setStatus(message.getMessageStatus());
133             if (messageEntity.getStatus() == MessageStatus.TRANSFERRED) {
134                 messageEntity.setSent(ZonedDateTime.now());
135             }
136 
137             if (newMessage) {
138                 LOGGER.trace("saving message payloads");
139                 messageEntity.getPayloads().clear();
140                 for (ContextPayload payload : message.getPayloads()) {
141                     BinaryEntity binaryEntity = new BinaryEntity();
142                     binaryEntity.setContent(
143                             BlobProxy.generateProxy(payload.getContent().getInputStream(), payload.getSize()));
144 
145                     PayloadEntity payloadEntity = new PayloadEntity();
146                     payloadEntity.setSize(payload.getSize());
147                     payloadEntity.setContentType(payload.getContentType());
148                     payloadEntity.setEncoding(payload.getEncoding());
149                     payloadEntity.setContent(binaryEntity);
150 
151                     messageEntity.getPayloads().add(payloadEntity);
152                 }
153             }
154 
155             messageRepository.save(messageEntity);
156         });
157     }
