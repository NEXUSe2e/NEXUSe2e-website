3   import com.evolvsys.nexuse2e.internal.ContextPayload;
4   import java.util.function.BiFunction;
5   import lombok.Getter;
6   import lombok.extern.slf4j.Slf4j;
7   import org.apache.commons.lang3.ObjectUtils;
8   import org.apache.tika.config.TikaConfig;
9   import org.apache.tika.mime.MimeTypeException;
10  
11  /**
12   * @author SvenBarden
13   * @since 24.04.2024
14   */
15  @Getter
16  @Slf4j
17  public enum PayloadPatternToken {
18      CONTENT_ID(
19              "{contentId}",
20              (s, payload) -> s.replace("{contentId}", ObjectUtils.defaultIfNull(payload.getContentId(), ""))),
21      SEQUENCE_NUMBER(
22              "{sequenceNumber}",
23              (s, payload) -> s.replace("{sequenceNumber}", String.valueOf(payload.getSequenceNumber()))),
24      EXTENSION("{extension}", (s, payload) -> {
25          String extension;
26          try {
27              extension = TikaConfig.getDefaultConfig()
28                      .getMimeRepository()
29                      .forName(payload.getContentType())
30                      .getExtension();
31          } catch (MimeTypeException e) {
32              LOGGER.warn(
33                      "could not detect extension for content type {} in payload {} {}",
34                      payload.getContentType(),
35                      payload.getContentId(),
36                      payload.getSequenceNumber(),
37                      e);
38              extension = ".dat";
39          }
40          return s.replace("{extension}", extension);
41      });
42  
43      private final String pattern;
44      private final BiFunction<String, ContextPayload, String> function;
45  
46      PayloadPatternToken(String pattern, BiFunction<String, ContextPayload, String> function) {
47          this.pattern = pattern;
48          this.function = function;
49      }
50  
51      public String applyPattern(String targetString, ContextPayload payload) {
52          return function.apply(targetString, payload);
53      }
54  }
