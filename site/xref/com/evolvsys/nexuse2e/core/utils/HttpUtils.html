4   import java.util.regex.Matcher;
5   import java.util.regex.Pattern;
6   import lombok.experimental.UtilityClass;
7   import lombok.extern.slf4j.Slf4j;
8   import org.apache.commons.lang3.StringUtils;
9   import org.springframework.http.HttpStatusCode;
10  
11  /**
12   * @author SvenBarden
13   * @since 30.04.2024
14   */
15  @UtilityClass
16  @Slf4j
17  public class HttpUtils {
18      private static final Pattern VALID_STATUS_PATTERN = Pattern.compile("[\\dx]{3}");
19      private static final Pattern CONTENT_TYPE_FIX_PATTERN =
20              Pattern.compile("(?!\\s+;)([A-Za-z0-9]+=)([^\"]+?)($|(;\\s+))");
21  
22      public static String fixContentType(String contentType) {
23          StringBuilder result = new StringBuilder();
24          if (StringUtils.isNotEmpty(contentType)) {
25              Matcher matcher = CONTENT_TYPE_FIX_PATTERN.matcher(contentType);
26              while (matcher.find()) {
27                  matcher.appendReplacement(result, matcher.group(1) + "\"" + matcher.group(2) + "\"" + matcher.group(3));
28              }
29              matcher.appendTail(result);
30          }
31          return result.toString();
32      }
33  
34      public static boolean validResponseStatus(HttpStatusCode httpStatusCode, List<String> validCodes) {
35          return validCodes.stream().anyMatch(code -> validResponseStatus(httpStatusCode, code));
36      }
37  
38      public static boolean validResponseStatus(HttpStatusCode httpStatusCode, String validStatus) {
39          boolean valid;
40          if (VALID_STATUS_PATTERN.matcher(validStatus).matches()) {
41              valid = Pattern.compile(validStatus.replace("x", "\\d"))
42                      .matcher(String.valueOf(httpStatusCode.value()))
43                      .matches();
44          } else {
45              valid = validStatus.equals(String.valueOf(httpStatusCode.value()));
46          }
47          return valid;
48      }
49  }
