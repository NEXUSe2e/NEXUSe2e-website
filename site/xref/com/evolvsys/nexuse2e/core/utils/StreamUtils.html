15  @UtilityClass

17  public class StreamUtils {
18  
19      public static <T, K, U> Collector<T, ?, Map<K, U>> toMapOfNullables(
20              Function<? super T, ? extends K> keyMapper, Function<? super T, ? extends U> valueMapper) {
21          return Collectors.collectingAndThen(Collectors.toList(), list -> {
22              Map<K, U> map = new LinkedHashMap<>();
23              list.forEach(item -> {
24                  K key = keyMapper.apply(item);
25                  U value = valueMapper.apply(item);
26                  if (map.containsKey(key)) {
27                      throw new IllegalStateException(String.format(
28                              "Duplicate key %s (attempted merging values %s and %s)", key, map.get(key), value));
29                  }
30                  map.put(key, value);
31              });
32              return map;
33          });
34      }
35  }
