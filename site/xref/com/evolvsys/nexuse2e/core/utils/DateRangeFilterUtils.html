15  @UtilityClass
16  public class DateRangeFilterUtils {
17  
18      @Nullable
19      public static Predicate filter(
20              DateTimePath<ZonedDateTime> dateTimePath,
21              String timezone,
22              @Nullable Range range,
23              @Nullable ZonedDateTime from,
24              @Nullable ZonedDateTime till) {
25          BooleanBuilder builder = new BooleanBuilder();
26          if (range != null) {
27              builder.and(range.rangeFunction.apply(dateTimePath, ZoneId.of(timezone)));
28          }
29  
30          if (from != null) {
31              builder.and(dateTimePath.goe(from));
32          }
33          if (till != null) {
34              builder.and(dateTimePath.loe(till));
35          }
36          return builder.getValue();
37      }
38  
39      public enum Range {
40          LAST_24_H((dateTimePath, zoneId) ->
41                  dateTimePath.goe(ZonedDateTime.now(zoneId).minusHours(24))),
42          TODAY((dateTimePath, zoneId) -> dateTimePath.goe(LocalDate.now(zoneId).atStartOfDay(zoneId))),
43          YESTERDAY((dateTimePath, zoneId) -> dateTimePath.between(
44                  LocalDate.now(zoneId).minusDays(1).atStartOfDay(zoneId),
45                  LocalDate.now(zoneId).minusDays(1).atTime(LocalTime.MAX).atZone(zoneId))),
46          THIS_WEEK((dateTimePath, zoneId) ->
47                  dateTimePath.goe(LocalDate.now(zoneId).atStartOfDay(zoneId).with(DayOfWeek.MONDAY))),
48          LAST_WEEK((dateTimePath, zoneId) -> dateTimePath.between(
49                  LocalDate.now(zoneId)
50                          .atStartOfDay(zoneId)
51                          .with(DayOfWeek.MONDAY)
52                          .minusWeeks(1),
53                  LocalDate.now(zoneId)
54                          .atTime(LocalTime.MAX)
55                          .atZone(zoneId)
56                          .with(DayOfWeek.SUNDAY)
57                          .minusWeeks(1))),
58          THIS_MONTH((dateTimePath, zoneId) ->
59                  dateTimePath.goe(LocalDate.now(zoneId).atStartOfDay(zoneId).withDayOfMonth(1))),
60          LAST_7_DAYS((dateTimePath, zoneId) ->
61                  dateTimePath.goe(LocalDate.now(zoneId).atStartOfDay(zoneId).minusDays(6))),
62          LAST_30_DAYS((dateTimePath, zoneId) ->
63                  dateTimePath.goe(LocalDate.now(zoneId).atStartOfDay(zoneId).minusDays(29)));
64  
65          private final BiFunction<DateTimePath<ZonedDateTime>, ZoneId, Predicate> rangeFunction;
66  
67          Range(BiFunction<DateTimePath<ZonedDateTime>, ZoneId, Predicate> rangeFunction) {
68              this.rangeFunction = rangeFunction;
69          }
70      }
71  }
