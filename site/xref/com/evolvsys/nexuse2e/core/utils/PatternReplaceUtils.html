3   import com.evolvsys.nexuse2e.core.message.context.ExecutionContext;
4   import com.evolvsys.nexuse2e.internal.ContextPayload;
5   import lombok.experimental.UtilityClass;

7   /**
8    * @author GuidoEsch
9    * @since 06.02.2024
10   */
11  @UtilityClass
12  public class PatternReplaceUtils {
13  
14      public static String replaceTokens(String targetValue, ExecutionContext executionContext, ContextPayload payload) {
15          if (targetValue == null) {
16              return null;
17          }
18          return replacePayloadTokens(replaceContextTokens(targetValue, executionContext), payload);
19      }
20  
21      public static String replaceContextTokens(String targetValue, ExecutionContext executionContext) {
22          String result = targetValue;
23          for (ContextPatternToken patternToken : ContextPatternToken.values()) {
24              result = patternToken.applyPattern(result, executionContext);
25          }
26          return result;
27      }
28  
29      public static String replacePayloadTokens(String targetValue, ContextPayload payload) {
30          String result = targetValue;
31          for (PayloadPatternToken patternToken : PayloadPatternToken.values()) {
32              result = patternToken.applyPattern(result, payload);
33          }
34          return result;
35      }
36  }
