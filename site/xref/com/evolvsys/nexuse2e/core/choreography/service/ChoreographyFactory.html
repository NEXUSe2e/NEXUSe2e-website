1   package com.evolvsys.nexuse2e.core.choreography.service;
2   
3   import com.evolvsys.nexuse2e.core.choreography.mapper.ContextChoreographyMapper;
4   import com.evolvsys.nexuse2e.core.choreography.model.ContextChoreography;
5   import com.evolvsys.nexuse2e.core.choreography.repository.ChoreographyRepository;
6   import com.evolvsys.nexuse2e.core.message.context.ContextObjectFactory;
7   import lombok.RequiredArgsConstructor;
8   import org.springframework.stereotype.Component;
9   
10  /**
11   * @author SvenBarden
12   * @since 24.01.2024
13   */
14  @Component
15  @RequiredArgsConstructor
16  public class ChoreographyFactory implements ContextObjectFactory<String, ContextChoreography> {
17      private final ChoreographyRepository choreographyRepository;
18      private final ContextChoreographyMapper contextChoreographyMapper;
19  
20      @Override
21      public ContextChoreography create(String choreographyId) {
22          return choreographyRepository.findByChoreographyId(choreographyId).stream()
23                  .filter(choreographyEntity ->
24                          choreographyEntity.getChoreographyId().equals(choreographyId))
25                  .findFirst()
26                  .map(contextChoreographyMapper::entityToChoreography)
27                  .orElseThrow(
28                          () -> new IllegalArgumentException("no choreography with name " + choreographyId + " found"));
29      }
30  }
