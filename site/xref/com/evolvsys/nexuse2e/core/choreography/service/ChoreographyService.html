25  @Service
26  public class ChoreographyService extends BaseService<ChoreographyDto, ChoreographyEntity, ChoreographyRepository> {
27      private final ProcessService processService;
28      private final ListService listService;
29  
30      public ChoreographyService(
31              ChoreographyMapper mapper,
32              ChoreographyRepository repository,
33              CurrentUserService currentUserService,
34              ProcessService processService,
35              ListService listService) {
36          super(mapper, repository, currentUserService);
37          this.processService = processService;
38          this.listService = listService;
39      }
40  
41      @Override
42      protected ChoreographyEntity createEntity() {
43          return new ChoreographyEntity();
44      }
45  
46      public List<ChoreographyDto> list() {
47          return listService.list(this).stream()
48                  .sorted(Comparator.comparing(ChoreographyEntity::getChoreographyId))
49                  .map(this::mapEntityToDto)
50                  .toList();
51      }
52  
53      public ChoreographyDto addRoute(long choreographyId, RouteDto routeDto) {
54          ChoreographyEntity choreographyEntity =
55                  findWithActionCheck(choreographyId, ChoreographyActionType.ADD_ROUTE.name());
56  
57          choreographyEntity =
58                  processService.execute(ProcessExecutionContext.<ChoreographyEntity, ChoreographyDto>builder()
59                          .target(choreographyEntity)
60                          .actionType(ChoreographyActionType.ADD_ROUTE)
61                          .parameterValue(ChoreographyAddRouteAction.ROUTE_PARAMETER.of(routeDto))
62                          .build());
63  
64          return mapEntityToDto(choreographyEntity);
65      }
66  
67      public ChoreographyDto removeRoute(long choreographyId, long routeId) {
68          ChoreographyEntity choreographyEntity =
69                  findWithActionCheck(choreographyId, ChoreographyActionType.REMOVE_ROUTE.name());
70  
71          choreographyEntity =
72                  processService.execute(ProcessExecutionContext.<ChoreographyEntity, ChoreographyDto>builder()
73                          .target(choreographyEntity)
74                          .actionType(ChoreographyActionType.REMOVE_ROUTE)
75                          .parameterValue(ChoreographyRemoveRouteAction.ROUTE_ID_PARAMETER.of(routeId))
76                          .build());
77  
78          return mapEntityToDto(choreographyEntity);
79      }
80  
81      public ChoreographyDto linkRoutes(long choreographyId, List<Long> routeIds) {
82          ChoreographyEntity choreographyEntity =
83                  findWithActionCheck(choreographyId, ChoreographyActionType.LINK_ROUTE.name());
84  
85          choreographyEntity =
86                  processService.execute(ProcessExecutionContext.<ChoreographyEntity, ChoreographyDto>builder()
87                          .target(choreographyEntity)
88                          .actionType(ChoreographyActionType.LINK_ROUTE)
89                          .parameterValue(ChoreographyLinkRouteAction.ROUTE_ID_PARAMETER.of(routeIds))
90                          .build());
91  
92          return mapEntityToDto(choreographyEntity);
93      }
94  }
