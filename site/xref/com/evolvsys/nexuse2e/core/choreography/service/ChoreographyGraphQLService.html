1   package com.evolvsys.nexuse2e.core.choreography.service;
2   
3   import com.evolvsys.app.core.graphql.annotation.GraphQLService;
4   import com.evolvsys.app.core.graphql.dataloader.DataLoaderService;
5   import com.evolvsys.nexuse2e.core.action.model.ActionDto;
6   import com.evolvsys.nexuse2e.core.choreography.model.ChoreographyDto;
7   import com.evolvsys.nexuse2e.core.choreography.repository.ChoreographyActionBatchLoader;
8   import com.evolvsys.nexuse2e.core.choreography.repository.ChoreographyRouteBatchLoader;
9   import com.evolvsys.nexuse2e.core.choreography.validation.ValidChoreographyHasActions;
10  import com.evolvsys.nexuse2e.core.choreography.validation.ValidChoreographyHasConversations;
11  import com.evolvsys.nexuse2e.core.config.service.ConfigGraphQLService;
12  import com.evolvsys.nexuse2e.core.route.model.RouteDto;
13  import io.leangen.graphql.annotations.*;
14  import io.leangen.graphql.execution.ResolutionEnvironment;
15  import jakarta.validation.Valid;
16  import java.util.List;
17  import java.util.Optional;
18  import java.util.Set;
19  import java.util.concurrent.CompletableFuture;
20  import lombok.RequiredArgsConstructor;
21  import lombok.extern.slf4j.Slf4j;
22  
23  /**
24   * @author Florian Drees
25   * @since 18.12.2023
26   */
27  @GraphQLService
28  @RequiredArgsConstructor
29  @Slf4j
30  public class ChoreographyGraphQLService {
31      private final ChoreographyService choreographyService;
32      private final DataLoaderService dataLoaderService;
33  
34      @GraphQLMutation(name = "choreography_update", description = "Updates an existing or creates a new choreography")
35      @GraphQLNonNull
36      public ChoreographyDto update(@GraphQLNonNull @Valid ChoreographyDto choreographyDto) {
37          return ConfigGraphQLService.notifySubscribers(choreographyService.save(choreographyDto));
38      }
39  
40      @GraphQLMutation(name = "choreography_delete", description = "Deletes a choreography")
41      public boolean delete(@ValidChoreographyHasConversations @ValidChoreographyHasActions long choreographyId) {
42          choreographyService.delete(choreographyId);
43          ConfigGraphQLService.notifySubscribers();
44          return true;
45      }
46  
47      @GraphQLMutation(name = "choreography_addRoute")
48      @GraphQLNonNull
49      public ChoreographyDto addRoute(long choreographyId, @GraphQLNonNull @Valid RouteDto route) {
50          return ConfigGraphQLService.notifySubscribers(choreographyService.addRoute(choreographyId, route));
51      }
52  
53      @GraphQLMutation(name = "choreography_linkRoutes", description = "Links existing routes to the choreography")
54      @GraphQLNonNull
55      public ChoreographyDto linkRoutes(long choreographyId, List<@GraphQLNonNull Long> routeIds) {
56          return ConfigGraphQLService.notifySubscribers(choreographyService.linkRoutes(choreographyId, routeIds));
57      }
58  
59      @GraphQLMutation(name = "choreography_removeRoute", description = "Removes the route from the choreography")
60      @GraphQLNonNull
61      public ChoreographyDto removeRoute(long choreographyId, long routeId) {
62          return ConfigGraphQLService.notifySubscribers(choreographyService.removeRoute(choreographyId, routeId));
63      }
64  
65      @GraphQLQuery(name = "choreography_list", description = "List all choreographies")
66      @GraphQLNonNull
67      public List<@GraphQLNonNull ChoreographyDto> list() {
68          return choreographyService.list();
69      }
70  
71      @GraphQLQuery(name = "choreography_getById", description = "Gets a choreography")
72      public Optional<ChoreographyDto> getById(long choreographyId) {
73          return choreographyService.getById(choreographyId);
74      }
75  
76      @GraphQLQuery(description = "list of actions of the provided choreography")
77      public CompletableFuture<Set<@GraphQLNonNull ActionDto>> actions(
78              @GraphQLContext ChoreographyDto choreographyDto,
79              @GraphQLEnvironment ResolutionEnvironment resolutionEnvironment) {
80          LOGGER.trace("getting actions for choreography {}", choreographyDto);
81          return dataLoaderService.loadRelation(
82                  resolutionEnvironment, ChoreographyActionBatchLoader.class, choreographyDto);
83      }
84  
85      @GraphQLQuery(description = "list of routes of the provided choreography")
86      public CompletableFuture<Set<@GraphQLNonNull RouteDto>> routes(
87              @GraphQLContext ChoreographyDto choreographyDto,
88              @GraphQLEnvironment ResolutionEnvironment resolutionEnvironment) {
89          return dataLoaderService.loadRelation(
90                  resolutionEnvironment, ChoreographyRouteBatchLoader.class, choreographyDto);
91      }
92  }
