1   package com.evolvsys.nexuse2e.core.schedule.provider;
2   
3   import lombok.RequiredArgsConstructor;
4   import org.quartz.CronExpression;
5   import org.quartz.CronScheduleBuilder;
6   
7   /**
8    * @author Florian Drees
9    * @since 30.01.2024
10   */
11  @RequiredArgsConstructor
12  public class CronScheduleProvider implements ScheduleProvider<CronScheduleBuilder> {
13      private final String cron;
14      private final MisfireInstruction misfireInstruction;
15  
16      public static CronScheduleProvider create(String cron) {
17          return create(cron, MisfireInstruction.NEXT_SCHEDULE);
18      }
19  
20      public static CronScheduleProvider create(String cron, MisfireInstruction misfireInstruction) {
21          if (CronExpression.isValidExpression(cron)) {
22              return new CronScheduleProvider(cron, misfireInstruction);
23          } else {
24              throw new IllegalArgumentException("The provided cron pattern is not valid: " + cron);
25          }
26      }
27  
28      @Override
29      public CronScheduleBuilder getScheduleBuilder() {
30          CronScheduleBuilder cronScheduleBuilder = CronScheduleBuilder.cronSchedule(cron);
31          return switch (misfireInstruction) {
32              case FIRE_ONCE -> cronScheduleBuilder.withMisfireHandlingInstructionFireAndProceed();
33              case NEXT_SCHEDULE -> cronScheduleBuilder.withMisfireHandlingInstructionDoNothing();
34          };
35      }
36  }
