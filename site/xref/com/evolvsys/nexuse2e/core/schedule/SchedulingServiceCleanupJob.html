10  import lombok.extern.slf4j.Slf4j;
11  import org.quartz.*;
12  import org.quartz.impl.matchers.GroupMatcher;
13  import org.springframework.scheduling.quartz.QuartzJobBean;
14  
15  /**
16   * @author SvenBarden
17   * @since 26.03.2024
18   */
19  @RequiredArgsConstructor
20  @Slf4j
21  @DisallowConcurrentExecution
22  public class SchedulingServiceCleanupJob extends QuartzJobBean {
23      private final ServiceRepository serviceRepository;
24      private final ComponentRegistry componentRegistry;
25  
26      @Override
27      @SuppressFBWarnings("CLI_CONSTANT_LIST_INDEX")
28      protected void executeInternal(JobExecutionContext context) throws JobExecutionException {
29          Scheduler scheduler = context.getScheduler();
30          try {
31              Set<JobKey> serviceJobKeys =
32                      scheduler.getJobKeys(GroupMatcher.jobGroupEquals(SchedulingService.QUARTZ_GROUP_ID));
33              for (JobKey serviceJobKey : serviceJobKeys) {
34                  String[] strings = extractServiceName(serviceJobKey.getName());
35                  if (strings.length == 2) {
36                      String componentId = strings[0];
37                      String serviceId = strings[1];
38  
39                      ServiceEntity serviceEntity =
40                              serviceRepository.findByServiceId(serviceId).orElse(null);
41                      if (serviceEntity == null) {
42                          LOGGER.error(
43                                  "service with id {} for job {} could not be found, job will be removed",
44                                  serviceId,
45                                  serviceJobKey);
46                          scheduler.deleteJob(serviceJobKey);
47                      } else {
48                          if (serviceEntity.getComponentId().equals(componentId)) {
49                              if (componentRegistry.exists(componentId)) {
50                                  LOGGER.trace("service {} for job {} is valid", serviceId, serviceJobKey);
51                              } else {
52                                  LOGGER.error(
53                                          "service {} has an invalid componentId {}. Job will be deleted and service will be disabled",
54                                          serviceId,
55                                          serviceJobKey);
56                                  serviceEntity.setEnabled(false);
57                                  serviceRepository.save(serviceEntity);
58                                  scheduler.deleteJob(serviceJobKey);
59                              }
60                          } else {
61                              LOGGER.error(
62                                      "component id for service {} does not match for job {}. Job will be deleted and service will be disabled",
63                                      serviceId,
64                                      serviceJobKey);
65                              serviceEntity.setEnabled(false);
66                              serviceRepository.save(serviceEntity);
67                              scheduler.deleteJob(serviceJobKey);
68                          }
69                      }
70                  } else {
71                      LOGGER.error("job key {} has invalid format for service jobs and will be removed", serviceJobKey);
72                      scheduler.deleteJob(serviceJobKey);
73                  }
74              }
75          } catch (SchedulerException e) {
76              LOGGER.error("scheduling error while cleaning up service jobs: {}", ErrorUtils.aggregate(e), e);
77          }
78      }
79  
80      private String[] extractServiceName(String jobName) {
81          return jobName.split("_AS_");
82      }
83  }
