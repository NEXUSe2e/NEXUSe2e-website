1   package com.evolvsys.nexuse2e.core.schedule;
2   
3   import org.quartz.*;
4   import org.springframework.context.annotation.Bean;
5   import org.springframework.context.annotation.Configuration;
6   
7   /**
8    * @author SvenBarden
9    * @since 26.03.2024
10   */
11  @Configuration
12  public class SchedulingJobConfiguration {
13      public static final JobKey SERVICE_CLEANUP_JOB_KEY = new JobKey("cleanup", "scheduling_service");
14      public static final TriggerKey SERVICE_CLEANUP_TRIGGER_KEY = new TriggerKey("cleanup", "scheduling_service");
15  
16      @Bean
17      JobDetail schedulingCleanupJob() {
18          return JobBuilder.newJob(SchedulingServiceCleanupJob.class)
19                  .withIdentity(SERVICE_CLEANUP_JOB_KEY)
20                  .storeDurably()
21                  .build();
22      }
23  
24      @Bean
25      Trigger schedulingCleanupTrigger() {
26          return TriggerBuilder.newTrigger()
27                  .withIdentity(SERVICE_CLEANUP_TRIGGER_KEY)
28                  .forJob(SERVICE_CLEANUP_JOB_KEY)
29                  .withSchedule(SimpleScheduleBuilder.repeatHourlyForever(1))
30                  .build();
31      }
32  }
