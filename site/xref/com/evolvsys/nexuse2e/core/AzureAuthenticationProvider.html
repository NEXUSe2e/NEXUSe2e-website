1   package com.evolvsys.nexuse2e.core;
2   
3   import com.azure.core.credential.TokenCredential;
4   import com.azure.identity.ClientSecretCredentialBuilder;
5   import com.azure.identity.DefaultAzureCredentialBuilder;
6   import com.evolvsys.nexuse2e.core.parameter.ParameterDefinition;
7   import com.evolvsys.nexuse2e.core.parameter.ParameterType;
8   import com.evolvsys.nexuse2e.core.parameter.Parameters;
9   import org.apache.commons.lang3.StringUtils;
10  
11  /**
12   * <p>Interface to provide azure authentication for Nexus services-</p>
13   *
14   * @author Florian Drees
15   * @since 02.07.2024
16   */
17  public interface AzureAuthenticationProvider {
18      String PARAM_CLIENT_ID = "client_id";
19      String PARAM_CLIENT_SECRET = "client_secret";
20      String PARAM_TENANT_ID = "tenant_id";
21  
22      default TokenCredential getTokenCredential(Parameters parameters) {
23          String clientId = parameters.value(PARAM_CLIENT_ID);
24          String clientSecret = parameters.value(PARAM_CLIENT_SECRET);
25          String tenantId = parameters.value(PARAM_TENANT_ID);
26          TokenCredential tokenCredential;
27          if (StringUtils.isNoneBlank(clientId, clientSecret, tenantId)) {
28              tokenCredential = new ClientSecretCredentialBuilder()
29                      .clientId(clientId)
30                      .clientSecret(clientSecret)
31                      .tenantId(tenantId)
32                      .build();
33          } else {
34              tokenCredential = new DefaultAzureCredentialBuilder().build();
35          }
36          return tokenCredential;
37      }
38  
39      default ParameterDefinition createClientIdParameter() {
40          return ParameterDefinition.builder()
41                  .name(PARAM_CLIENT_ID)
42                  .displayName("Client id")
43                  .description("The client id to authenticate against the keyvault")
44                  .type(ParameterType.TRIMMED_STRING)
45                  .build();
46      }
47  
48      default ParameterDefinition createClientSecretParameter() {
49          return ParameterDefinition.builder()
50                  .name(PARAM_CLIENT_SECRET)
51                  .displayName("Client secret")
52                  .description("The client secret to authenticate against the keyvault")
53                  .type(ParameterType.PASSWORD)
54                  .build();
55      }
56  
57      default ParameterDefinition createTenantIdParameter() {
58          return ParameterDefinition.builder()
59                  .name(PARAM_TENANT_ID)
60                  .displayName("Tenant id")
61                  .description(
62                          "The tenant id to authenticate against the keyvault. Only Required with clientId and clientSecret.")
63                  .type(ParameterType.TRIMMED_STRING)
64                  .build();
65      }
66  }
