23  @Component
24  @RequiredArgsConstructor
25  @Slf4j
26  @Order(5)
27  public class PipelineConfigHandler implements ConfigHandler<PipelineConfig> {
28      private final PipelineRepository pipelineRepository;
29      private final PipelineConfigMapper pipelineConfigMapper;
30      private final PipeletConfigMapper pipeletConfigMapper;
31  
32      private PipelineEntity create() {
33          return new PipelineEntity();
34      }
35  
36      private PipeletEntity create(PipeletConfig config, PipelineEntity pipelineEntity) {
37          LOGGER.trace("create pipelet {}", config);
38          PipeletEntity pipeletEntity = new PipeletEntity();
39          pipelineEntity.getPipelets().add(pipeletEntity);
40          return pipeletEntity;
41      }
42  
43      @Override
44      public void clear() {
45          pipelineRepository.deleteAll();
46      }
47  
48      @Override
49      @Transactional
50      public void postImportConfig(ConfigContext configContext, PipelineConfig config) {
51          importConfig(configContext, config);
52      }
53  
54      @Override
55      @Transactional
56      public void importConfig(ConfigContext configContext, PipelineConfig config) {
57          LOGGER.trace("handling pipeline with pipelineId {}", config.getPipelineId());
58  
59          PipelineEntity pipelineEntity =
60                  pipelineRepository.findByPipelineId(config.getPipelineId()).orElseGet(this::create);
61  
62          pipelineConfigMapper.updateEntity(config, pipelineEntity);
63  
64          pipelineEntity.getPipelets().clear();
65  
66          List<PipeletConfig> pipelets = config.getPipelets();
67          if (CollectionUtils.isNotEmpty(pipelets)) {
68              for (int i = 0; i < pipelets.size(); i++) {
69                  PipeletConfig pipelet = pipelets.get(i);
70                  LOGGER.trace("adding pipelet {}", pipelet.getComponentId());
71  
72                  PipeletEntity pipeletEntity = create(pipelet, pipelineEntity);
73                  pipeletEntity.setOrder(i);
74  
75                  pipeletConfigMapper.updateEntity(pipelet, pipeletEntity);
76              }
77          }
78  
79          pipelineRepository.save(pipelineEntity);
80      }
81  
82      @Override
83      public void exportConfig(ConfigExportContext configExportContext) {
84          pipelineRepository.findAll().stream()
85                  .map(pipelineConfigMapper::toConfig)
86                  .forEach(
87                          config -> configExportContext.getConfig().getPipelines().add(config));
88      }
89  
90      @Override
91      public List<PipelineConfig> merge(Collection<PipelineConfig> oldConfigs, Collection<PipelineConfig> newConfigs) {
92          return defaultMerge(oldConfigs, newConfigs, PipelineConfig::getPipelineId, pipelineConfigMapper::merge);
93      }
94  
95      @Override
96      public void sort(NexusConfig nexusConfig) {
97          nexusConfig.getPipelines().sort(Comparator.comparing(PipelineConfig::getPipelineId));
98          nexusConfig.getPipelines().forEach(this::sort);
99      }
100 
101     @Override
102     public void sort(PipelineConfig config) {
103         config.getPipelets().forEach(pipeletConfig -> {
104             pipeletConfig.getParameters().sort(Comparator.comparing(ParameterConfig::getName));
105         });
106     }
107 }
