3   import com.evolvsys.nexuse2e.core.config.model.ConnectionConfig;
4   import com.evolvsys.nexuse2e.core.model.entity.PartnerConnectionEntity;
5   import org.mapstruct.*;
6   import org.springframework.beans.factory.annotation.Autowired;
7   
8   /**
9    * @author SvenBarden
10   * @since 27.11.2023
11   */
12  @Mapper(
13          config = DefaultConfigMapperConfig.class,
14          uses = {ParameterConfigMapper.class, PartnerConfigMapper.class, ServiceConfigMapper.class})
15  public abstract class PartnerConnectionConfigMapper {
16      private ParameterConfigMapper parameterConfigMapper;
17  
18      @Autowired
19      protected void config(ParameterConfigMapper parameterConfigMapper) {
20          this.parameterConfigMapper = parameterConfigMapper;
21      }
22  
23      @Mapping(target = "certificateContainer", ignore = true)
24      @Mapping(target = "partner", ignore = true)
25      @Mapping(target = "parameters", ignore = true)
26      @InheritConfiguration
27      public abstract void updateEntity(
28              ConnectionConfig connectionConfig, @MappingTarget PartnerConnectionEntity partnerConnectionEntity);
29  
30      @Mapping(target = "certificateContainer", source = "certificateContainer.name")
31      public abstract ConnectionConfig toConfig(PartnerConnectionEntity partnerConnectionEntity);
32  
33      @AfterMapping
34      protected void afterUpdateEntity(
35              ConnectionConfig connectionConfig, @MappingTarget PartnerConnectionEntity partnerConnectionEntity) {
36          if (partnerConnectionEntity.getOutflowService() != null) {
37              parameterConfigMapper.updateEntities(
38                      connectionConfig.getParameters(),
39                      partnerConnectionEntity.getParameters(),
40                      partnerConnectionEntity.getOutflowService().getComponentId());
41          }
42      }
43  
44      public String toString(PartnerConnectionEntity partnerConnectionEntity) {
45          return partnerConnectionEntity == null ? null : partnerConnectionEntity.getConnectionId();
46      }
47  }
