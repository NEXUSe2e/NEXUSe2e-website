19  public interface ConfigHandler<T> {
20  
21      default Class<T> type() {
22          return GenericUtils.getGenericType(getClass(), ConfigHandler.class, 0);
23      }
24  
25      void clear();
26  
27      void importConfig(ConfigContext configContext, T config);
28  
29      default void postImportConfig(ConfigContext configContext, T config) {
30          // no default implementation
31      }
32  
33      void exportConfig(ConfigExportContext configExportContext);
34  
35      List<T> merge(Collection<T> oldConfigs, Collection<T> newConfigs);
36  
37      static <T, IT> List<T> defaultCollectionMerge(
38              Collection<T> oldConfigs,
39              Collection<T> newConfigs,
40              Function<T, IT> idFunction,
41              BiConsumer<T, T> mergeFunction,
42              BiFunction<IT, IT, Boolean> equalsFunction) {
43          List<T> mergedConfigs = new ArrayList<>(oldConfigs);
44  
45          for (T oldConfig : oldConfigs) {
46              newConfigs.stream()
47                      .filter(config -> equalsFunction.apply(idFunction.apply(config), idFunction.apply(oldConfig)))
48                      .findFirst()
49                      .ifPresent(newConfig -> mergeFunction.accept(newConfig, oldConfig));
50          }
51  
52          for (T newConfig : newConfigs) {
53              if (mergedConfigs.stream()
54                      .noneMatch(config -> equalsFunction.apply(idFunction.apply(config), idFunction.apply(newConfig)))) {
55                  mergedConfigs.add(newConfig);
56              }
57          }
58  
59          return mergedConfigs;
60      }
61  
62      default <IT> List<T> defaultMerge(
63              Collection<T> oldConfigs,
64              Collection<T> newConfigs,
65              Function<T, IT> idFunction,
66              BiConsumer<T, T> mergeFunction,
67              BiFunction<IT, IT, Boolean> equalsFunction) {
68          return defaultCollectionMerge(oldConfigs, newConfigs, idFunction, mergeFunction, equalsFunction);
69      }
70  
71      default List<T> defaultMerge(
72              Collection<T> oldConfigs,
73              Collection<T> newConfigs,
74              Function<T, String> idFunction,
75              BiConsumer<T, T> mergeFunction) {
76          return defaultCollectionMerge(oldConfigs, newConfigs, idFunction, mergeFunction, StringUtils::equalsIgnoreCase);
77      }
78  
79      default void sort(NexusConfig nexusConfig) {}
80  
81      default void sort(T config) {}
82  }
