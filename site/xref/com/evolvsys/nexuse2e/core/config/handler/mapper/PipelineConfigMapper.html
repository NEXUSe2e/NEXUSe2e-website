14  @Mapper(config = DefaultConfigMapperConfig.class, uses = PipeletConfigMapper.class)
15  public abstract class PipelineConfigMapper {
16      private PipelineRepository pipelineRepository;
17  
18      @Autowired
19      protected void config(PipelineRepository pipelineRepository) {
20          this.pipelineRepository = pipelineRepository;
21      }
22  
23      public PipelineEntity idToEntity(String pipelineId) {
24          return pipelineRepository.findByPipelineId(pipelineId).orElse(null);
25      }
26  
27      public String entityToString(PipelineEntity pipelineEntity) {
28          return pipelineEntity == null ? null : pipelineEntity.getPipelineId();
29      }
30  
31      @Mapping(target = "pipelets", ignore = true)
32      @Mapping(target = "description", source = "description", qualifiedByName = "toTrimmedString_pipeline")
33      @InheritConfiguration
34      public abstract void updateEntity(PipelineConfig pipelineConfig, @MappingTarget PipelineEntity pipelineEntity);
35  
36      @Named("toTrimmedString_pipeline")
37      protected String toTrimmedString(String stringToTrim) {
38          return Optional.ofNullable(stringToTrim).map(String::trim).orElse(stringToTrim);
39      }
40  
41      public String configToString(PipelineConfig pipelineConfig) {
42          return pipelineConfig == null ? null : pipelineConfig.getPipelineId();
43      }
44  
45      public PipelineConfig stringToConfig(String pipelineId) {
46          return pipelineRepository
47                  .findByPipelineId(pipelineId)
48                  .map(this::toConfig)
49                  .orElse(null);
50      }
51  
52      public abstract PipelineConfig toConfig(PipelineEntity pipelineEntity);
53  
54      public abstract void merge(PipelineConfig newConfig, @MappingTarget PipelineConfig oldConfig);
55  }
