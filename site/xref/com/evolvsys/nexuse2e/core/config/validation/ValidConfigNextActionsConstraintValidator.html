3   import com.evolvsys.nexuse2e.core.config.model.ActionConfig;
4   import jakarta.validation.ConstraintValidator;
5   import jakarta.validation.ConstraintValidatorContext;
6   import java.util.Objects;
7   import java.util.Set;
8   import lombok.extern.slf4j.Slf4j;
9   import org.apache.commons.collections4.CollectionUtils;
10  
11  /**
12   * @author SvenBarden
13   * @since 26.04.2024
14   */
15  @Slf4j
16  class ValidConfigNextActionsConstraintValidator implements ConstraintValidator<ValidConfigNextActions, ActionConfig> {
17      @Override
18      public boolean isValid(ActionConfig value, ConstraintValidatorContext context) {
19          boolean isValid;
20          if (ValidationContextHolder.exists()) {
21              Set<String> validNextActions = value.getValidNextActions();
22              if (CollectionUtils.isEmpty(validNextActions)) {
23                  isValid = true;
24              } else {
25                  isValid = ValidationContextHolder.getContext().getConfig().getChoreographies().stream()
26                          .filter(choreographyConfig ->
27                                  choreographyConfig.getActions().contains(value))
28                          .flatMap(choreographyConfig -> choreographyConfig.getActions().stream())
29                          .map(ActionConfig::getActionId)
30                          .filter(Objects::nonNull)
31                          .anyMatch(validNextActions::contains);
32              }
33          } else {
34              LOGGER.warn("no context found to validate ValidConfigNextActions for action {}", value);
35              isValid = true;
36          }
37          return isValid;
38      }
39  }
