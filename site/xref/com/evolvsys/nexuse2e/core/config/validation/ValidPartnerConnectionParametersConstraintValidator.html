12  import lombok.extern.slf4j.Slf4j;

14  import org.apache.commons.lang3.StringUtils;
15  import org.springframework.stereotype.Component;
16  
17  /**
18   * @author SvenBarden
19   * @since 25.04.2024
20   */
21  @Component
22  @RequiredArgsConstructor
23  @Slf4j
24  class ValidPartnerConnectionParametersConstraintValidator
25          implements ConstraintValidator<ValidPartnerConnectionParameters, ConnectionConfig> {
26      private final ComponentRegistry componentRegistry;
27  
28      @Override
29      public boolean isValid(ConnectionConfig value, ConstraintValidatorContext context) {
30          boolean isValid;
31          if (ValidationContextHolder.exists()) {
32              ValidationContext validationContext = ValidationContextHolder.getContext();
33              List<ParameterConfig> parameters = value.getParameters();
34              if (CollectionUtils.isEmpty(parameters)) {
35                  isValid = true;
36              } else {
37                  if (StringUtils.isBlank(value.getOutflowService())) {
38                      isValid = false;
39                  } else {
40                      List<String> parameterNames = validationContext.getConfig().getServices().stream()
41                              .filter(serviceConfig ->
42                                      Objects.equals(serviceConfig.getServiceId(), value.getOutflowService()))
43                              .flatMap(serviceConfig ->
44                                      componentRegistry.getComponent(serviceConfig.getComponentId()).stream())
45                              .flatMap(registeredNexusComponent ->
46                                      registeredNexusComponent.getParameterDefinitions().stream())
47                              .map(ParameterDefinition::getName)
48                              .toList();
49                      isValid = parameters.stream()
50                              .filter(parameterConfig -> StringUtils.isNotBlank(parameterConfig.getName()))
51                              .allMatch(parameterConfig -> parameterNames.contains(parameterConfig.getName()));
52                  }
53              }
54          } else {
55              LOGGER.warn("no context found to validate parameters for connection {}", value);
56              isValid = true;
57          }
58          return isValid;
59      }
60  }
