12  import org.springframework.stereotype.Service;

14  /**
15   * @author SvenBarden
16   * @since 21.11.2024
17   */
18  @Service
19  @RequiredArgsConstructor
20  @Slf4j
21  class ConfigHandlerService {
22      private static final List<Class<?>> ROOT_TYPES = List.of(
23              PartnerConfig.class,
24              PipelineConfig.class,
25              ChoreographyConfig.class,
26              RouteConfig.class,
27              ServiceConfig.class,
28              CaCertConfig.class,
29              ClientKeyContainerConfig.class,
30              UserConfig.class,
31              KeyValuePropertyConfig.class);
32      private final List<ConfigHandler<?>> configHandlers;
33  
34      private Optional<ConfigHandler<?>> findHandler(Class<?> configType) {
35          return configHandlers.stream()
36                  .filter(configHandler -> configHandler.type().equals(configType))
37                  .findFirst();
38      }
39  
40      @SuppressWarnings("unchecked")
41      private <T> void handle(ConfigContext configContext, T config) {
42          findHandler(config.getClass())
43                  .ifPresent(configHandler -> ((ConfigHandler<T>) configHandler).importConfig(configContext, config));
44      }
45  
46      @SuppressWarnings("unchecked")
47      private <T> void postHandle(ConfigContext configContext, T config) {
48          findHandler(config.getClass())
49                  .ifPresent(configHandler -> ((ConfigHandler<T>) configHandler).postImportConfig(configContext, config));
50      }
51  
52      private void handle(
53              ConfigContext configContext, NexusConfig nexusConfig, BiConsumer<ConfigContext, Object> handleFunction) {
54          for (PipelineConfig pipeline : nexusConfig.getPipelines()) {
55              handleFunction.accept(configContext, pipeline);
56          }
57          if (CollectionUtils.isNotEmpty(nexusConfig.getServices())) {
58              for (ServiceConfig service : nexusConfig.getServices()) {
59                  handleFunction.accept(configContext, service);
60              }
61          }
62          for (PartnerConfig partner : nexusConfig.getPartners()) {
63              handleFunction.accept(configContext, partner);
64          }
65          for (CaCertConfig caCertificate : nexusConfig.getCaCertificates()) {
66              handleFunction.accept(configContext, caCertificate);
67          }
68          for (ClientKeyContainerConfig clientKey : nexusConfig.getClientKeyContainers()) {
69              handleFunction.accept(configContext, clientKey);
70          }
71          for (RouteConfig route : nexusConfig.getRoutes()) {
72              handleFunction.accept(configContext, route);
73          }
74          for (ChoreographyConfig choreography : nexusConfig.getChoreographies()) {
75              handleFunction.accept(configContext, choreography);
76          }
77          for (UserConfig user : nexusConfig.getUsers()) {
78              handleFunction.accept(configContext, user);
79          }
80          for (KeyValuePropertyConfig keyValueProperty : nexusConfig.getKeyValueProperties()) {
81              handleFunction.accept(configContext, keyValueProperty);
82          }
83      }
84  
85      public void importConfig(ConfigContext configContext, NexusConfig nexusConfig) {
86          handle(configContext, nexusConfig, this::handle);
87      }
88  
89      public void postImportConfig(ConfigContext configContext, NexusConfig nexusConfig) {
90          handle(configContext, nexusConfig, this::postHandle);
91      }
92  
93      @SuppressWarnings("unchecked")
94      private <T> List<T> merge(Class<T> type, Collection<T> oldConfigs, Collection<T> newConfigs) {
95          return findHandler(type)
96                  .map(configHandler -> ((ConfigHandler<T>) configHandler).merge(oldConfigs, newConfigs))
97                  .orElseThrow();
98      }
99  
100     public NexusConfig export(NexusConfig nexusConfig) {
101         ConfigExportContext exportContext =
102                 ConfigExportContext.builder().config(nexusConfig).build();
103 
104         for (Class<?> rootType : ROOT_TYPES) {
105             findHandler(rootType).ifPresent(configHandler -> configHandler.exportConfig(exportContext));
106         }
107 
108         sortConfig(exportContext.getConfig());
109 
110         return exportContext.getConfig();
111     }
112 
113     private void sortConfig(NexusConfig nexusConfig) {
114         for (Class<?> rootType : ROOT_TYPES) {
115             findHandler(rootType).ifPresent(configHandler -> configHandler.sort(nexusConfig));
116         }
117     }
118 
119     public NexusConfig merge(NexusConfig oldConfig, NexusConfig newConfig) {
120         NexusConfig mergedConfig = new NexusConfig();
121         mergedConfig.getPartners().addAll(merge(PartnerConfig.class, oldConfig.getPartners(), newConfig.getPartners()));
122         mergedConfig
123                 .getPipelines()
124                 .addAll(merge(PipelineConfig.class, oldConfig.getPipelines(), newConfig.getPipelines()));
125         mergedConfig.getRoutes().addAll(merge(RouteConfig.class, oldConfig.getRoutes(), newConfig.getRoutes()));
126         mergedConfig
127                 .getChoreographies()
128                 .addAll(merge(ChoreographyConfig.class, oldConfig.getChoreographies(), newConfig.getChoreographies()));
129         mergedConfig.getServices().addAll(merge(ServiceConfig.class, oldConfig.getServices(), newConfig.getServices()));
130         mergedConfig
131                 .getCaCertificates()
132                 .addAll(merge(CaCertConfig.class, oldConfig.getCaCertificates(), newConfig.getCaCertificates()));
133         mergedConfig
134                 .getClientKeyContainers()
135                 .addAll(merge(
136                         ClientKeyContainerConfig.class,
137                         oldConfig.getClientKeyContainers(),
138                         newConfig.getClientKeyContainers()));
139         mergedConfig.getUsers().addAll(merge(UserConfig.class, oldConfig.getUsers(), newConfig.getUsers()));
140         mergedConfig
141                 .getKeyValueProperties()
142                 .addAll(merge(
143                         KeyValuePropertyConfig.class,
144                         oldConfig.getKeyValueProperties(),
145                         newConfig.getKeyValueProperties()));
146         return mergedConfig;
147     }
148 
149     public void clear() {
150         configHandlers.forEach(ConfigHandler::clear);
151     }
