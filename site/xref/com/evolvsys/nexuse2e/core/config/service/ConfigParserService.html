29  /**
30   * @author SvenBarden
31   * @since 26.10.2023
32   */
33  @Service
34  @RequiredArgsConstructor
35  @Slf4j
36  @DependsOn("parameterConverterRegistrar")
37  public class ConfigParserService implements InitializingBean, BeanFactoryAware {
38      private final XmlMapper xmlMapper = new XmlMapper();
39      private final ConfigHandlerService configHandlerService;
40      private final SystemConfigurationProperties systemConfigurationProperties;
41      private AutowireCapableBeanFactory beanFactory;
42  
43      // @Transactional
44      public void clear() {
45          configHandlerService.clear();
46      }
47  
48      public NexusConfig merge(NexusConfig oldConfig, NexusConfig newConfig) {
49          NexusConfig mergedConfig = configHandlerService.merge(oldConfig, newConfig);
50          mergedConfig.setVersion(systemConfigurationProperties.getVersion());
51          return mergedConfig;
52      }
53  
54      @Transactional(readOnly = true)
55      public NexusConfig exportConfig() {
56          NexusConfig nexusConfig = new NexusConfig();
57          nexusConfig.setVersion(systemConfigurationProperties.getVersion());
58          return configHandlerService.export(nexusConfig);
59      }
60  
61      public void print(OutputStream outputStream, NexusConfig nexusConfig) throws IOException {
62          xmlMapper.writeValue(outputStream, nexusConfig);
63      }
64  
65      public String configToString(NexusConfig nexusConfig) throws JsonProcessingException {
66          return xmlMapper.writeValueAsString(nexusConfig);
67      }
68  
69      public void update(NexusConfig nexusConfig) {
70          ConfigContext configContext =
71                  ConfigContext.builder().config(nexusConfig).build();
72          configHandlerService.importConfig(configContext, nexusConfig);
73          configHandlerService.postImportConfig(configContext, nexusConfig);
74      }
75  
76      public NexusConfig parse(String content) throws JsonProcessingException {
77          LOGGER.trace("parsing config from string");
78          return xmlMapper.readValue(content, NexusConfig.class);
79      }
80  
81      public NexusConfig parse(InputStream inputStream) throws IOException {
82          LOGGER.trace("parsing config from stream");
83          return xmlMapper.readValue(inputStream, NexusConfig.class);
84      }
85  
86      @Override
87      public void afterPropertiesSet() throws Exception {
88          xmlMapper.setHandlerInstantiator(new SpringHandlerInstantiator(beanFactory));
89          xmlMapper.enable(JsonGenerator.Feature.IGNORE_UNKNOWN);
90          xmlMapper.enable(JsonParser.Feature.IGNORE_UNDEFINED);
91          xmlMapper.disable(DeserializationFeature.FAIL_ON_UNKNOWN_PROPERTIES);
92          xmlMapper.getDeserializationContext().checkUnresolvedObjectId();
93          xmlMapper.enable(SerializationFeature.INDENT_OUTPUT);
94          xmlMapper.registerModule(new JakartaXmlBindAnnotationModule());
95          xmlMapper.setSerializationInclusion(JsonInclude.Include.NON_EMPTY);
96          xmlMapper.configure(ToXmlGenerator.Feature.WRITE_XML_DECLARATION, true);
97      }
98  
99      @Override
100     public void setBeanFactory(BeanFactory beanFactory) {
101         synchronized (this) {
102             this.beanFactory = (AutowireCapableBeanFactory) beanFactory;
103         }
104     }
105 }
