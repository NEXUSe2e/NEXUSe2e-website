24  @Service
25  @RequiredArgsConstructor
26  public class NEXUSe2eConfigService {
27      private static final String EXS_EXCEPTION_SOFTENING_NO_CONSTRAINTS = "EXS_EXCEPTION_SOFTENING_NO_CONSTRAINTS";
28      private static final String COULD_NOT_MERGE_CONFIG = "could not merge config";
29      private final ConfigParserService configParserService;
30      private final ConfigValidatorService configValidatorService;
31      private final ConfigCrudService configCrudService;
32      private final ConfigImportFileHelper configImportFileHelper;
33      private final ConfigImportEventHandler configImportEventHandler;
34  
35      public NexusConfig merge(NexusConfig oldConfig, NexusConfig newConfig) {
36          return configParserService.merge(oldConfig, newConfig);
37      }
38  
39      public NexusConfig getCurrentConfig() {
40          return configParserService.exportConfig();
41      }
42  
43      public void print(NexusConfig nexusConfig, OutputStream outputStream) throws IOException {
44          configParserService.print(outputStream, nexusConfig);
45      }
46  
47      public ConfigValidationResult validate(NexusConfig config) {
48          return configValidatorService.validate(config);
49      }
50  
51      @SuppressFBWarnings(EXS_EXCEPTION_SOFTENING_NO_CONSTRAINTS)
52      public ConfigValidationResult validate(long fileId) {
53          ConfigValidationResult validationResult;
54          try (InputStream binaryStream = configImportFileHelper.getContent(fileId)) {
55              NexusConfig nexusConfig = configParserService.parse(binaryStream);
56              validationResult = validate(nexusConfig);
57          } catch (Exception e) {
58              throw new GeneralErrorException(COULD_NOT_MERGE_CONFIG, e);
59          }
60          return validationResult;
61      }
62  
63      @SuppressWarnings("PMD.PrematureDeclaration") // oldConfig needs to be stored before we update the config
64      private NexusConfig importConfig(NexusConfig config, boolean overwrite) throws NexusException {
65          ConfigValidationResult validationResult = configValidatorService.validate(config);
66          if (validationResult.getFatalCount() > 0) {
67              throw new NexusException(NexusError.builder()
68                      .message("Config contains fatal errors and cannot be applied.")
69                      .extension("messages", String.join(", ", validationResult.getMessages()))
70                      .build());
71          }
72          NexusConfig oldConfig = getCurrentConfig();
73          configImportEventHandler.preImport(config, overwrite);
74          if (overwrite) {
75              configParserService.clear();
76          }
77          try {
78              configCrudService.save(config);
79          } catch (NexusException e) {
80              throw new NexusException(
81                      NexusError.builder().message(COULD_NOT_MERGE_CONFIG).build(), e);
82          }
83          NexusConfig currentConfig = getCurrentConfig();
84          configImportEventHandler.postImport(oldConfig, currentConfig, overwrite);
85          return currentConfig;
86      }
87  
88      public NexusConfig merge(NexusConfig config) throws NexusException {
89          return importConfig(config, false);
90      }
91  
92      public NexusConfig overwrite(NexusConfig config) throws NexusException {
93          return importConfig(config, true);
94      }
95  
96      public NexusConfig overwrite(long fileId) throws NexusException {
97          return overwrite(toConfig(fileId));
98      }
99  
100     public NexusConfigMerge merge(long fileId) {
101         NexusConfig currentConfig = getCurrentConfig();
102         NexusConfig nexusConfig = toConfig(fileId);
103         NexusConfig merge = merge(currentConfig, nexusConfig);
104 
105         return NexusConfigMerge.builder()
106                 .merged(merge)
107                 .upload(nexusConfig)
108                 .current(getCurrentConfig())
109                 .mergedValidationResult(validate(merge))
110                 .build();
111     }
112 
113     @SuppressFBWarnings(EXS_EXCEPTION_SOFTENING_NO_CONSTRAINTS)
114     public NexusConfig toConfig(long fileId) {
115         NexusConfig nexusConfig;
116         try (InputStream binaryStream = configImportFileHelper.getContent(fileId)) {
117             nexusConfig = configParserService.parse(binaryStream);
118         } catch (Exception e) {
119             throw new GeneralErrorException(COULD_NOT_MERGE_CONFIG, e);
120         }
121         return nexusConfig;
122     }
123 
124     @Transactional(rollbackFor = IOException.class)
125     public FileDto currentConfigToFile() throws IOException {
126         ByteArrayOutputStream outputStream = new ByteArrayOutputStream();
127         configParserService.print(outputStream, getCurrentConfig());
128 
129         return configImportFileHelper.create(outputStream);
130     }
