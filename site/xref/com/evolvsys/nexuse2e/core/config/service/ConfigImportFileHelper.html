23  @Component
24  @RequiredArgsConstructor
25  class ConfigImportFileHelper {
26      private final FileRepository fileRepository;
27      private final FileFactory fileFactory;
28      private final FileMapper fileMapper;
29      private final SystemConfigurationProperties systemConfigurationProperties;
30  
31      public InputStream getContent(long fileId) throws SQLException {
32          FileEntity fileEntity = fileRepository
33                  .getFileEntityById(fileId)
34                  .orElseThrow(() -> new ObjectNotFoundException(FileDto.class, fileId));
35          return fileEntity.getFileContent().getContent().getBinaryStream();
36      }
37  
38      public FileDto create(ByteArrayOutputStream outputStream) {
39          StringBuilder fileName = new StringBuilder("NEXUSe2e_config");
40          String environmentName = systemConfigurationProperties.getEnvironmentName();
41          if (StringUtils.isNotBlank(environmentName)) {
42              fileName.append('_').append(environmentName);
43          }
44          fileName.append('_')
45                  .append(DateTimeFormatter.ISO_OFFSET_DATE_TIME.format(
46                          ZonedDateTime.now().withNano(0)))
47                  .append(".xml");
48  
49          FileEntity fileEntity = fileFactory.create(
50                  fileName.toString().replaceAll("[^A-Za-z0-9.\\-_+]", "_"), "text/xml", outputStream.toByteArray());
51  
52          return fileMapper.entityToDto(fileEntity);
53      }
54  }
