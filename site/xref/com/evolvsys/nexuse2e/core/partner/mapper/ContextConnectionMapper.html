21  @Mapper(config = DefaultMapperConfiguration.class)
22  @Slf4j
23  public abstract class ContextConnectionMapper {
24      private PartnerConnectionRepository partnerConnectionRepository;
25  
26      @Autowired
27      protected void config(PartnerConnectionRepository partnerConnectionRepository) {
28          this.partnerConnectionRepository = partnerConnectionRepository;
29      }
30  
31      public PartnerConnectionEntity connectionToEntity(ContextConnection connection) {
32          return connection == null
33                  ? null
34                  : partnerConnectionRepository
35                          .findByConnectionIdAndPartner_PartnerId(connection.getConnectionId(), connection.getPartnerId())
36                          .orElse(null);
37      }
38  
39      @Mapping(target = "partnerId", source = "partner.partnerId")
40      @Mapping(target = "outflowService", source = "outflowService.serviceId")
41      public abstract ContextConnection entityToConnection(PartnerConnectionEntity partnerConnectionEntity);
42  
43      protected List<Duration> secondsToDuration(Collection<Long> collection) {
44          return CollectionUtils.isEmpty(collection)
45                  ? Collections.emptyList()
46                  : collection.stream().map(Duration::ofSeconds).toList();
47      }
48  
49      protected Set<String> stringToList(String s) {
50          return s == null
51                  ? Collections.emptySet()
52                  : Arrays.stream(s.split("\\s*,\\s*"))
53                          .filter(StringUtils::isNotBlank)
54                          .collect(Collectors.toSet());
55      }
56  }
