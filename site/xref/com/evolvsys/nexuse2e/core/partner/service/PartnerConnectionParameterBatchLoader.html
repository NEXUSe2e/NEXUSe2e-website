32  @Component
33  @RequiredArgsConstructor
34  @Slf4j
35  public class PartnerConnectionParameterBatchLoader
36          implements GraphQLBatchLoader<PartnerConnectionDto, Set<ParameterDto>> {
37      private final BatchLoaderRepository batchLoaderRepository;
38      private final ComponentRegistry componentRegistry;
39      private final ParameterBatchLoaderHelper parameterBatchLoaderHelper;
40      private final ServiceRepository serviceRepository;
41  
42      @Override
43      public CompletionStage<Map<PartnerConnectionDto, Set<ParameterDto>>> load(
44              Set<PartnerConnectionDto> keys, BatchLoaderEnvironment environment) {
45          BatchLoaderContext context = environment.getContext();
46          CurrentUserData currentUserData = context.getCurrentUserData();
47  
48          Set<PartnerConnectionDto> nonNullKeys =
49                  keys.stream().filter(Objects::nonNull).collect(Collectors.toSet());
50  
51          CompletionStage<Map<PartnerConnectionDto, Set<ParameterDto>>> result;
52          if (CollectionUtils.isEmpty(nonNullKeys)) {
53              result = CompletableFuture.completedStage(new HashMap<>());
54          } else {
55              result = CompletableFuture.supplyAsync(() -> loadInternal(nonNullKeys, currentUserData));
56          }
57          return result;
58      }
59  
60      private Map<PartnerConnectionDto, Set<ParameterDto>> loadInternal(
61              Set<PartnerConnectionDto> keys, CurrentUserData currentUserData) {
62          Map<PartnerConnectionDto, Set<ParameterDto>> map = new HashMap<>();
63  
64          List<Long> idKeys = keys.stream().map(IdDto::getId).toList();
65  
66          List<Long> outflowServiceIds = keys.stream()
67                  .map(PartnerConnectionDto::getOutflowServiceId)
68                  .distinct()
69                  .toList();
70  
71          Map<Long, String> outflowServiceComponentMap = serviceRepository.findAllById(outflowServiceIds).stream()
72                  .collect(Collectors.toMap(IdEntity::getId, ServiceEntity::getComponentId));
73  
74          Map<Long, Set<ParameterEntity>> entityMap = batchLoaderRepository.loadEntitiesByJoin(
75                  idKeys, QPartnerConnectionEntity.partnerConnectionEntity.parameters, false);
76  
77          Map<Long, List<String>> parameterStringLists = parameterBatchLoaderHelper.getStringList(entityMap);
78  
79          for (PartnerConnectionDto key : keys) {
80              String componentId = outflowServiceComponentMap.get(key.getOutflowServiceId());
81              RegisteredNexusComponent component =
82                      componentRegistry.getComponent(componentId).orElse(null);
83  
84              if (component != null) {
85                  Set<ParameterDto> parameterDtos = parameterBatchLoaderHelper.loadInternalById(
86                          component,
87                          currentUserData,
88                          entityMap.computeIfAbsent(key.getId(), id -> new HashSet<>()),
89                          parameterStringLists,
90                          false);
91                  map.put(key, parameterDtos);
92              }
93          }
94  
95          return map;
96      }
97  }
