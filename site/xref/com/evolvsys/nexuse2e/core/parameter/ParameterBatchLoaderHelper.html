30      private ParameterDto entityToDto(
31              ParameterEntity parameterEntity,
32              ParameterDefinition definition,
33              CurrentUserData currentUserData,
34              Map<Long, List<String>> parameterStringLists) {
35          List<String> strings = parameterStringLists.get(parameterEntity.getId());
36          if (strings == null) {
37              parameterEntity.setStringValues(Collections.emptyList());
38          } else {
39              parameterEntity.setStringValues(strings);
40          }
41          return parameterMapper.entityToDto(parameterEntity, definition, currentUserData);
42      }
43  
44      public Set<ParameterDto> loadInternalById(
45              RegisteredNexusComponent component,
46              CurrentUserData currentUserData,
47              Set<ParameterEntity> parameterEntities,
48              Map<Long, List<String>> parameterStringLists) {
49          return loadInternalById(component, currentUserData, parameterEntities, parameterStringLists, true);
50      }
51  
52      public Set<ParameterDto> loadInternalById(
53              RegisteredNexusComponent component,
54              CurrentUserData currentUserData,
55              Set<ParameterEntity> parameterEntities,
56              Map<Long, List<String>> parameterStringLists,
57              boolean addDefaults) {
58  
59          Set<ParameterDto> parameterDtos = new LinkedHashSet<>();
60          if (Configurable.class.isAssignableFrom(component.getLoadedClass())
61                  && CollectionUtils.isNotEmpty(component.getParameterDefinitions())) {
62  
63              for (ParameterDefinition definition : component.getParameterDefinitions()) {
64                  LOGGER.trace("Processing parameter '{}'", definition.getName());
65  
66                  ParameterDto parameterDto = parameterEntities.stream()
67                          .filter(entity -> entity.getName().equalsIgnoreCase(definition.getName()))
68                          .findFirst()
69                          .map(entity -> entityToDto(entity, definition, currentUserData, parameterStringLists))
70                          .orElseGet(() -> parameterMapper.definitionToDto(definition, currentUserData));
71  
72                  if (addDefaults || parameterDto.getId() != null) {
73                      parameterDtos.add(parameterDto);
74                  }
75              }
76          }
77          return parameterDtos;
78      }
79  
80      public Map<Long, List<String>> getStringList(Map<Long, Set<ParameterEntity>> entityMap) {
81          Set<Long> parameterIds = entityMap.values().stream()
82                  .flatMap(parameterEntities -> parameterEntities.stream()
83                          .filter(entity -> entity.getType() == ParameterType.STRING_LIST)
84                          .map(IdEntity::getId)
85                          .distinct())
86                  .collect(Collectors.toSet());
87          return batchLoaderRepository.loadList(parameterIds, QParameterEntity.parameterEntity.stringValues);
88      }
