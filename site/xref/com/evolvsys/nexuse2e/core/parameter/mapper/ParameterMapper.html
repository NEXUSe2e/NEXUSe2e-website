105     protected ParameterDto definitionToDto(
106             ParameterDefinition definition, @Context InterceptingMappingContext interceptingMappingContext) {

110     public void updateEntities(
111             Collection<ParameterDto> dtos,
112             @MappingTarget Collection<ParameterEntity> entities,
113             @Context InterceptingMappingContext interceptingMappingContext) {
114 
115         if (CollectionUtils.isEmpty(dtos)) {
116             entities.clear();
117         } else {
118             List<String> dtoParamNames =
119                     dtos.stream().map(ParameterDto::getName).toList();
120             entities.removeIf(parameterEntity -> dtoParamNames.contains(parameterEntity.getName()) == false);
121             for (ParameterDto parameterDto : dtos) {
122                 ParameterEntity parameterEntity = entities.stream()
123                         .filter(entity -> entity.getName().equalsIgnoreCase(parameterDto.getName()))
124                         .findFirst()
125                         .orElse(null);
126                 if (parameterEntity == null) {
127                     parameterEntity = new ParameterEntity();
128                     entities.add(parameterEntity);
129                 }
130                 updateEntity(parameterDto, parameterEntity, interceptingMappingContext);
131             }
132         }
133     }

140     @BeforeMapping
141     protected void beforeUpdateEntity(ParameterDto dto, @MappingTarget ParameterEntity entity) {
142         if (dto.getType() != null && entity.getType() != null && dto.getType() != entity.getType()) {
143             throw new GeneralErrorException("parameter type is not allowed to be changed");
144         }
145     }

147     @AfterMapping
148     protected void afterEntityToDto(
149             ParameterEntity entity,
150             @MappingTarget ParameterDto dto,
151             @Context ParameterDefinition definition,
152             @Context InterceptingMappingContext interceptingMappingContext) {
153         updateDto(definition, dto);
154     }
155 
156     @AfterMapping
157     protected void afterEntityToDto(ParameterEntity entity, @MappingTarget ParameterDto dto) {
158         if (entity != null && entity.getType() != null) {
159             TypeConverter<Object, Object> typeConverter = parameterConverterRegistry.getTypeConverter(entity.getType());
160             Object dtoValue = typeConverter.toDtoValue(entity);
161             LOGGER.trace(
162                     "parameter {}: set value {} of entity into dto [ afterEntityToDto(entity, dto) ]",
163                     dto.getName(),
164                     dtoValue);
165             typeConverter.getDtoSetter().apply(dto, dtoValue);
166         }
167     }
168 
169     @AfterMapping
170     protected void afterUpdateDto(ParameterDefinition definition, @MappingTarget ParameterDto dto) {
171 
172         TypeConverter<Object, Object> typeConverter = parameterConverterRegistry.getTypeConverter(definition.getType());
173 
174         Object defaultDtoValue = typeConverter.toDtoValue(
175                 typeConverter.getDefinitionDefaultGetter().apply(definition));
176         typeConverter.getDtoDefaultSetter().apply(dto, defaultDtoValue);
177 
178         if (dto.getId() == null) {
179             // entity does not exist so the default value is preset
180             LOGGER.trace(
181                     "parameter '{}': set default value {} for parameter type {} as dto value",
182                     dto.getName(),
183                     defaultDtoValue,
184                     dto.getType().name());
185             typeConverter.getDtoSetter().apply(dto, defaultDtoValue);
186         }
187 
188         if (StringUtils.isBlank(dto.getDisplayName())) {
189             dto.setDisplayName(StringUtils.capitalize(dto.getName()));
190         }
191     }
192 
193     @AfterMapping
194     protected void afterUpdateEntity(ParameterDto dto, @MappingTarget ParameterEntity entity) {
195         TypeConverter<Object, Object> typeConverter = parameterConverterRegistry.getTypeConverter(dto.getType());
196         typeConverter.updateFromDto(entity, dto);
197     }
