3   import com.evolvsys.nexuse2e.core.exception.NexusError;
4   import com.evolvsys.nexuse2e.core.exception.NexusRuntimeException;
5   import com.evolvsys.nexuse2e.core.parameter.ParameterType;
6   import com.evolvsys.nexuse2e.core.parameter.converter.TypeConverter;
7   import java.util.ArrayList;
8   import java.util.List;
9   import lombok.extern.slf4j.Slf4j;
10  import org.springframework.stereotype.Component;
11  import org.springframework.util.ClassUtils;
12  
13  /**
14   * @author Florian Drees
15   * @since 11.01.2024
16   */
17  @Component
18  @Slf4j
19  public class ParameterConverterRegistry {
20      private final List<TypeConverter<?, ?>> typeConverters = new ArrayList<>();
21  
22      @SuppressWarnings("unchecked")
23      public <EntityTypeT, DtoTypeT> TypeConverter<EntityTypeT, DtoTypeT> getTypeConverter(ParameterType type) {
24          return (TypeConverter<EntityTypeT, DtoTypeT>) typeConverters.stream()
25                  .filter(aTypeConverter ->
26                          ClassUtils.getUserClass(aTypeConverter).equals(type.getConverterClass()))
27                  .findAny()
28                  .orElseThrow(() -> new NexusRuntimeException(NexusError.builder()
29                          .message("Could not find converter for parameter type " + type)
30                          .extension("type", type)
31                          .build()));
32      }
33  
34      void register(TypeConverter<?, ?> typeConverter) {
35          typeConverters.add(typeConverter);
36      }
37  }
