4   import java.util.Optional;
5   import org.springframework.stereotype.Component;

7   /**
8    * @author Florian Drees

11  @Component
12  public class BooleanTypeConverter extends BaseStringDtoValueTypeConverter<Boolean> {
13  
14      @Override
15      public String toDtoValue(ParameterEntity parameterEntity) {
16          return parameterEntity.getBooleanValue() == null
17                  ? null
18                  : parameterEntity.getBooleanValue().toString();
19      }
20  
21      @Override
22      public Boolean toContextValue(ParameterEntity parameterEntity) {
23          return parameterEntity.getBooleanValue();
24      }
25  
26      @Override
27      public String toDtoValue(Boolean value) {
28          return Optional.ofNullable(value).map(Object::toString).orElse(null);
29      }
30  
31      @Override
32      public void updateFromDto(ParameterEntity parameterEntity, String dtoValue) {
33          parameterEntity.setBooleanValue(dtoValue == null ? null : Boolean.valueOf(dtoValue));
34      }
35  }
