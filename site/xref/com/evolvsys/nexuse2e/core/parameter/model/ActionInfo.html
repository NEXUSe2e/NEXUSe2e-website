1   package com.evolvsys.nexuse2e.core.parameter.model;
2   
3   import com.evolvsys.nexuse2e.core.exception.NexusError;
4   import com.evolvsys.nexuse2e.core.exception.NexusRuntimeException;
5   import lombok.Builder;
6   import lombok.Value;
7   
8   /**
9    * @author SvenBarden
10   * @since 02.04.2024
11   */
12  @Value
13  @Builder
14  public class ActionInfo {
15      public static final String SEPARATOR = "_~~_";
16      String choreographyId;
17      String actionId;
18  
19      public static ActionInfo of(String parse) {
20          String[] split = parse.split(SEPARATOR);
21          if (split.length != 2) {
22              throw new NexusRuntimeException(NexusError.builder()
23                      .message("ActionInfo string '" + parse + " 'was invalid")
24                      .extension("string", parse)
25                      .extension("separator", SEPARATOR)
26                      .build());
27          }
28          return new ActionInfo(split[0], split[1]);
29      }
30  
31      public String format() {
32          return choreographyId + SEPARATOR + actionId;
33      }
34  }
