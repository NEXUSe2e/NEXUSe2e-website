28  @RequiredArgsConstructor
29  @Slf4j
30  public abstract class ParameterBatchLoader
31          implements GraphQLBatchLoader<ParameterBatchLoaderKeyProvider, Set<ParameterDto>> {
32      private final ComponentRegistry componentRegistry;
33      private final BatchLoaderRepository batchLoaderRepository;
34      private final ParameterBatchLoaderHelper parameterBatchLoaderHelper;
35      private final SetPath<ParameterEntity, QParameterEntity> setPath;
36  
37      @Override
38      public CompletionStage<Map<ParameterBatchLoaderKeyProvider, Set<ParameterDto>>> load(
39              Set<ParameterBatchLoaderKeyProvider> keys, BatchLoaderEnvironment environment) {
40          BatchLoaderContext context = environment.getContext();
41          CurrentUserData currentUserData = context.getCurrentUserData();
42  
43          Set<ParameterBatchLoaderKeyProvider> notNullKeys =
44                  keys.stream().filter(Objects::nonNull).collect(Collectors.toSet());
45  
46          CompletionStage<Map<ParameterBatchLoaderKeyProvider, Set<ParameterDto>>> result;
47          if (CollectionUtils.isEmpty(notNullKeys)) {
48              result = CompletableFuture.completedStage(new HashMap<>());
49          } else {
50              result = CompletableFuture.supplyAsync(() -> loadInternal(notNullKeys, currentUserData));
51          }
52          return result;
53      }
54  
55      private Map<ParameterBatchLoaderKeyProvider, Set<ParameterDto>> loadInternal(
56              Set<ParameterBatchLoaderKeyProvider> keys, CurrentUserData currentUserData) {
57  
58          Map<ParameterBatchLoaderKeyProvider, Set<ParameterDto>> map = new HashMap<>();
59  
60          List<Long> idKeys =
61                  keys.stream().map(ParameterBatchLoaderKeyProvider::getId).toList();
62  
63          Map<Long, Set<ParameterEntity>> entityMap = batchLoaderRepository.loadEntitiesByJoin(idKeys, setPath, false);
64  
65          Map<Long, List<String>> parameterStringLists = parameterBatchLoaderHelper.getStringList(entityMap);
66  
67          for (ParameterBatchLoaderKeyProvider keyProvider : keys) {
68              Optional<RegisteredNexusComponent> optionalComponent =
69                      componentRegistry.getComponent(keyProvider.getComponentId());
70  
71              if (optionalComponent.isPresent()) {
72                  Set<ParameterDto> parameterDtos = parameterBatchLoaderHelper.loadInternalById(
73                          optionalComponent.orElseThrow(),
74                          currentUserData,
75                          entityMap.computeIfAbsent(keyProvider.getId(), id -> new HashSet<>()),
76                          parameterStringLists);
77                  map.put(keyProvider, parameterDtos);
78              }
79          }
80          return map;
81      }
82  }
