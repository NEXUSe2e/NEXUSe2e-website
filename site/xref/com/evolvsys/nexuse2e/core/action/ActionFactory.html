1   package com.evolvsys.nexuse2e.core.action;
2   
3   import com.evolvsys.nexuse2e.core.action.mapper.ContextActionMapper;
4   import com.evolvsys.nexuse2e.core.action.model.ActionEntity;
5   import com.evolvsys.nexuse2e.core.action.repository.ActionRepository;
6   import com.evolvsys.nexuse2e.core.message.context.ActionKey;
7   import com.evolvsys.nexuse2e.core.message.context.ContextObjectFactory;
8   import lombok.RequiredArgsConstructor;
9   import lombok.extern.slf4j.Slf4j;
10  import org.springframework.stereotype.Component;
11  import org.springframework.transaction.annotation.Transactional;
12  
13  /**
14   * @author SvenBarden
15   * @since 17.10.2023
16   */
17  @Component
18  @RequiredArgsConstructor
19  @Slf4j
20  public class ActionFactory implements ContextObjectFactory<ActionKey, Action> {
21      private final ActionRepository actionRepository;
22      private final ContextActionMapper contextActionMapper;
23  
24      private Action entityToAction(ActionEntity actionEntity) {
25          return contextActionMapper.entityToAction(actionEntity);
26      }
27  
28      @Transactional(readOnly = true)
29      public Action getAction(String choreographyId, String actionId) {
30          return actionRepository.findByActionIdAndChoreography_ChoreographyId(actionId, choreographyId).stream()
31                  .filter(actionEntity ->
32                          actionEntity.getChoreography().getChoreographyId().equals(choreographyId)
33                                  && actionEntity.getActionId().equals(actionId))
34                  .findFirst()
35                  .map(this::entityToAction)
36                  .orElseThrow(() -> new IllegalArgumentException(
37                          "no action for choreography " + choreographyId + " and action " + actionId));
38      }
39  
40      @Override
41      @Transactional(readOnly = true)
42      public Action create(ActionKey actionKey) {
43          return actionKey.getActionId() == null || actionKey.getChoreographyId() == null
44                  ? null
45                  : getAction(actionKey.getChoreographyId(), actionKey.getActionId());
46      }
47  }
