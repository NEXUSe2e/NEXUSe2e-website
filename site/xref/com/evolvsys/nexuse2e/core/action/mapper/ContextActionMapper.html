15   * @author SvenBarden

18  @Mapper(config = DefaultMapperConfiguration.class)
19  public abstract class ContextActionMapper {
20      private PipelineFactory pipelineFactory;
21      private ActionRepository actionRepository;
22  
23      @Autowired
24      protected void config(PipelineFactory pipelineFactory, ActionRepository actionRepository) {
25          this.pipelineFactory = pipelineFactory;
26          this.actionRepository = actionRepository;
27      }
28  
29      protected Pipeline stringToPipeline(String pipelineId) {
30          return pipelineId == null ? null : pipelineFactory.get(pipelineId);
31      }
32  
33      protected Pipeline pipelineToContext(PipelineEntity pipelineEntity) {
34          return pipelineEntity == null ? null : stringToPipeline(pipelineEntity.getPipelineId());
35      }
36  
37      protected String actionToName(ActionEntity actionEntity) {
38          return actionEntity == null ? null : actionEntity.getActionId();
39      }
40  
41      public ActionEntity actionToEntity(Action action) {
42          return namesToEntity(action.getChoreographyId(), action.getActionId());
43      }
44  
45      public ActionEntity namesToEntity(String choreographyId, String actionId) {
46          return actionRepository.findByActionIdAndChoreography_ChoreographyId(actionId, choreographyId).stream()
47                  .filter(actionEntity ->
48                          actionEntity.getChoreography().getChoreographyId().equals(choreographyId)
49                                  && actionEntity.getActionId().equals(actionId))
50                  .findFirst()
51                  .orElse(null);
52      }
53  
54      @Mapping(target = "validNextAction", ignore = true)
55      @Mapping(target = "choreographyId", source = "choreography.choreographyId")
56      @Mapping(target = "choreographyRefId", source = "choreography.id")
57      public abstract Action entityToAction(ActionEntity actionEntity);
58  }
