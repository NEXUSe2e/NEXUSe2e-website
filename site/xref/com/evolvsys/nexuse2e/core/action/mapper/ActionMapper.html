21  /**
22   * <p>Mapper für {@link ActionEntity} und {@link ActionDto}.</p>
23   *
24   * @author Florian Drees
25   * @since 19.12.2023
26   * @see BaseDtoEntityMapper
27   */
28  @Mapper(
29          config = DefaultMapperConfiguration.class,
30          uses = {BaseChoreographyMapper.class, PipelineMapper.class, RouteMapper.class})
31  public abstract class ActionMapper extends BaseDtoEntityMapper<ActionDto, ActionEntity> {
32      private ActionRepository actionRepository;
33  
34      @Autowired
35      protected void config(ActionRepository actionRepository) {
36          this.actionRepository = actionRepository;
37      }
38  
39      @Override
40      public ActionEntity idToEntity(Long id, @Context InterceptingMappingContext interceptingMappingContext) {
41          return idToEntity(id, interceptingMappingContext, actionRepository::findById);
42      }
43  
44      @Mapping(target = "returnPipelineId", source = "returnPipeline.id")
45      @Mapping(target = "forwardPipelineId", source = "forwardPipeline.id")
46      @Mapping(target = "acknowledgmentPipelineId", source = "acknowledgmentPipeline.id")
47      @Mapping(target = "choreographyId", source = "choreography.id")
48      @InheritConfiguration
49      @Override
50      protected abstract ActionDto entityToDto(
51              ActionEntity entity, @Context InterceptingMappingContext interceptingMappingContext);
52  
53      @Mapping(target = "routes", ignore = true)
54      @Mapping(target = "returnPipeline", source = "returnPipelineId")
55      @Mapping(target = "forwardPipeline", source = "forwardPipelineId")
56      @Mapping(target = "acknowledgmentPipeline", source = "acknowledgmentPipelineId")
57      @Mapping(target = "validNextActions", ignore = true)
58      @Mapping(target = "choreography", source = "choreographyId")
59      @InheritConfiguration
60      @Override
61      protected abstract void updateEntity(
62              ActionDto dto,
63              @MappingTarget ActionEntity entity,
64              @Context InterceptingMappingContext interceptingMappingContext);
65  
66      public void updateValidNextActions(
67              Collection<ActionDto> actionDtos, ActionEntity actionEntity, Collection<ActionEntity> choreographyActions) {
68  
69          actionEntity.getValidNextActions().clear();
70          if (CollectionUtils.isNotEmpty(actionDtos)) {
71              Map<String, ActionEntity> actionEntityMap = choreographyActions.stream()
72                      .collect(Collectors.toMap(ActionEntity::getActionId, Function.identity()));
73  
74              actionDtos.stream()
75                      .map(actionDto -> actionEntityMap.get(actionDto.getActionId()))
76                      .filter(Objects::nonNull)
77                      .forEach(nextAction -> actionEntity.getValidNextActions().add(nextAction));
78          }
79      }
80  }
