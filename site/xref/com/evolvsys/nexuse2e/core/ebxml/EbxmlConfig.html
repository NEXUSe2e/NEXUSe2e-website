1   package com.evolvsys.nexuse2e.core.ebxml;

3   import com.evolvsys.nexuse2e.core.config.NEXUSe2eConfigurationProperties;
4   import java.util.ArrayList;
5   import java.util.Collections;
6   import java.util.List;
7   import lombok.RequiredArgsConstructor;
8   import org.apache.commons.collections4.CollectionUtils;
9   import org.springframework.beans.factory.InitializingBean;
10  import org.springframework.stereotype.Component;
11  
12  /**
13   * @author SvenBarden
14   * @since 12.04.2024
15   */
16  @Component
17  @RequiredArgsConstructor
18  class EbxmlConfig implements InitializingBean {
19      private static final List<String> NON_BINARY_MIME_TYPES = new ArrayList<>();
20      private final NEXUSe2eConfigurationProperties configurationProperties;
21  
22      static {
23          CollectionUtils.addAll(
24                  NON_BINARY_MIME_TYPES,
25                  "text/plain",
26                  "text/html",
27                  "application/xhtml+xml",
28                  "text/xml",
29                  "application/xml",
30                  "text/csv");
31      }
32  
33      static List<String> getNonBinaryMimeTypes() {
34          return Collections.unmodifiableList(NON_BINARY_MIME_TYPES);
35      }
36  
37      @Override
38      public void afterPropertiesSet() throws Exception {
39          NEXUSe2eConfigurationProperties.Ebxml ebxml = configurationProperties.getEbxml();
40          if (ebxml != null && CollectionUtils.isNotEmpty(ebxml.getNonBinaryMimeTypes())) {
41              NON_BINARY_MIME_TYPES.addAll(ebxml.getNonBinaryMimeTypes());
42          }
43      }
44  }
