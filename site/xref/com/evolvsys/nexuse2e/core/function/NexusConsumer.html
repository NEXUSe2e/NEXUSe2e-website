1   package com.evolvsys.nexuse2e.core.function;
2   
3   import com.evolvsys.nexuse2e.core.exception.NexusException;
4   import java.util.Objects;
5   
6   /**
7    * Represents an operation that accepts a single input argument and returns no
8    * result. Unlike most other functional interfaces, {@code NexusConsumer} is expected
9    * to operate via side-effects.
10   *
11   * <p>This is a functional interface
12   * whose functional method is {@link #accept(Object)}.
13   *
14   * @param <T> the type of the input to the operation
15   * @author SvenBarden
16   * @since 20.09.2024
17   */
18  @FunctionalInterface
19  public interface NexusConsumer<T> {
20  
21      /**
22       * Performs this operation on the given argument.
23       *
24       * @param t the input argument
25       */
26      void accept(T t) throws NexusException;
27  
28      /**
29       * Returns a composed {@code NexusConsumer} that performs, in sequence, this
30       * operation followed by the {@code after} operation. If performing either
31       * operation throws an exception, it is relayed to the caller of the
32       * composed operation. If performing this operation throws an exception,
33       * the {@code after} operation will not be performed.
34       *
35       * @param after the operation to perform after this operation
36       * @return a composed {@code NexusConsumer} that performs in sequence this
37       *         operation followed by the {@code after} operation
38       * @throws NullPointerException if {@code after} is null
39       */
40      default NexusConsumer<T> andThen(NexusConsumer<? super T> after) {
41          Objects.requireNonNull(after);
42          return (T t) -> {
43              accept(t);
44              after.accept(t);
45          };
46      }
47  }
