1   package com.evolvsys.nexuse2e.core.pipeline.service;
2   
3   import com.evolvsys.app.core.graphql.annotation.GraphQLService;
4   import com.evolvsys.app.core.graphql.dataloader.DataLoaderService;
5   import com.evolvsys.nexuse2e.core.config.service.ConfigGraphQLService;
6   import com.evolvsys.nexuse2e.core.pipelet.model.PipeletDto;
7   import com.evolvsys.nexuse2e.core.pipelet.service.PipeletBatchLoader;
8   import com.evolvsys.nexuse2e.core.pipeline.PipelineFilter;
9   import com.evolvsys.nexuse2e.core.pipeline.model.PipelineDto;
10  import io.leangen.graphql.annotations.*;
11  import io.leangen.graphql.execution.ResolutionEnvironment;
12  import jakarta.validation.Valid;
13  import java.util.List;
14  import java.util.Optional;
15  import java.util.Set;
16  import java.util.concurrent.CompletableFuture;
17  import lombok.RequiredArgsConstructor;
18  import lombok.extern.slf4j.Slf4j;
19  
20  /**
21   * @author Florian Drees
22   * @since 20.11.2023
23   */
24  @GraphQLService
25  @RequiredArgsConstructor
26  @Slf4j
27  public class PipelineGraphQLService {
28      private final PipelineService pipelineService;
29      private final DataLoaderService dataLoaderService;
30  
31      @GraphQLQuery(name = "pipeline_list", description = "List all pipelines")
32      @GraphQLNonNull
33      public List<@GraphQLNonNull PipelineDto> list(PipelineFilter pipelineFilter) {
34          return pipelineService.list(pipelineFilter);
35      }
36  
37      @GraphQLQuery(name = "pipeline_getById", description = "Gets a pipeline")
38      public Optional<PipelineDto> getById(long pipelineId) {
39          return pipelineService.getById(pipelineId);
40      }
41  
42      @GraphQLMutation(name = "pipeline_update", description = "Updates an existing or creates a new pipeline")
43      @GraphQLNonNull
44      public PipelineDto update(@GraphQLNonNull @Valid PipelineDto pipelineDto) {
45          return ConfigGraphQLService.notifySubscribers(pipelineService.save(pipelineDto));
46      }
47  
48      @GraphQLMutation(name = "pipeline_updateWithPipelets")
49      @GraphQLNonNull
50      public PipelineDto update(@GraphQLNonNull @Valid PipelineDto pipeline, @Valid List<PipeletDto> pipelets) {
51          return ConfigGraphQLService.notifySubscribers(pipelineService.save(pipeline, pipelets));
52      }
53  
54      @GraphQLMutation(name = "pipeline_delete", description = "Delete a pipeline")
55      public boolean delete(long pipelineId) {
56          pipelineService.delete(pipelineId);
57          ConfigGraphQLService.notifySubscribers();
58          return true;
59      }
60  
61      @GraphQLQuery(description = "list of pipelets of the provided pipeline")
62      public CompletableFuture<Set<@GraphQLNonNull PipeletDto>> pipelets(
63              @GraphQLContext PipelineDto pipelineDto, @GraphQLEnvironment ResolutionEnvironment resolutionEnvironment) {
64          LOGGER.trace("getting pipelets for pipeline {}", pipelineDto);
65  
66          return dataLoaderService.loadRelation(resolutionEnvironment, PipeletBatchLoader.class, pipelineDto);
67      }
68  
69      @GraphQLMutation(name = "pipeline_addPipelet", description = "Adds a pipelet to an existing pipeline")
70      @GraphQLNonNull
71      public PipelineDto addPipelet(long pipelineId, @Valid @GraphQLNonNull PipeletDto pipeletDto) {
72          return ConfigGraphQLService.notifySubscribers(pipelineService.addPipelet(pipelineId, pipeletDto));
73      }
74  
75      @GraphQLMutation(name = "pipeline_removePipelet", description = "Removes a pipelet from an existing pipeline")
76      @GraphQLNonNull
77      public PipelineDto removePipelet(long pipelineId, long pipeletId) {
78          return ConfigGraphQLService.notifySubscribers(pipelineService.removePipelet(pipelineId, pipeletId));
79      }
80  
81      @GraphQLMutation(name = "pipeline_rearrange", description = "rearrange the order of pipelets")
82      public PipelineDto rearrangePipeline(long pipelineId, List<@GraphQLNonNull Long> pipeletIdsInOrder) {
83          return ConfigGraphQLService.notifySubscribers(pipelineService.rearrange(pipelineId, pipeletIdsInOrder));
84      }
85  }
