25   * @author Florian Drees

27   */
28  @Service
29  public class PipelineService extends BaseService<PipelineDto, PipelineEntity, PipelineRepository> {
30      private final PipelineMapper mapper;
31      private final ListService listService;
32      private final ProcessService processService;
33  
34      public PipelineService(
35              PipelineMapper mapper,
36              PipelineRepository repository,
37              CurrentUserService currentUserService,
38              ListService listService,
39              ProcessService processService) {
40          super(mapper, repository, currentUserService);
41          this.mapper = mapper;
42          this.listService = listService;
43          this.processService = processService;
44      }
45  
46      @Override
47      protected PipelineEntity createEntity() {
48          return new PipelineEntity();
49      }
50  
51      public List<PipelineDto> list(PipelineFilter pipelineFilter) {
52          return listService.list(pipelineFilter, PipelineDto.class, repository::findAll).stream()
53                  .sorted(Comparator.comparing(PipelineEntity::getPipelineId))
54                  .map(this::mapEntityToDto)
55                  .toList();
56      }
57  
58      @Transactional
59      public PipelineDto save(PipelineDto pipelineDto, List<PipeletDto> pipeletDtos) {
60          PipelineEntity entity = getEntity(pipelineDto);
61  
62          mapper.updatePipelets(pipeletDtos, entity, currentUserService.getCurrentUserData());
63  
64          entity = save(entity);
65  
66          return mapEntityToDto(entity);
67      }
68  
69      public PipelineDto addPipelet(long pipelineId, PipeletDto pipeletDto) {
70          Validate.notNull(pipeletDto, "pipelet dto must not be null");
71          Validate.notNull(pipeletDto.getComponentId(), "pipelet id must not be null");
72  
73          PipelineEntity pipelineEntity = findWithActionCheck(pipelineId, PipelineActionType.ADD_PIPELET.name());
74          pipelineEntity = processService.execute(ProcessExecutionContext.<PipelineEntity, PipelineDto>builder()
75                  .target(pipelineEntity)
76                  .actionType(PipelineActionType.ADD_PIPELET)
77                  .parameterValue(PipelineAddPipeletAction.PIPELET.of(pipeletDto))
78                  .build());
79  
80          return mapEntityToDto(pipelineEntity);
81      }
82  
83      public PipelineDto removePipelet(long pipelineId, long pipeletId) {
84          PipelineEntity pipelineEntity = findWithActionCheck(pipelineId, PipelineActionType.REMOVE_PIPELET.name());
85          pipelineEntity = processService.execute(ProcessExecutionContext.<PipelineEntity, PipelineDto>builder()
86                  .target(pipelineEntity)
87                  .actionType(PipelineActionType.REMOVE_PIPELET)
88                  .parameterValue(PipelineRemovePipeletAction.PIPELET_ID.of(pipeletId))
89                  .build());
90  
91          return mapEntityToDto(pipelineEntity);
92      }
93  
94      public PipelineDto rearrange(long pipelineId, List<Long> pipeletIdsInOrder) {
95          PipelineEntity pipelineEntity = findWithActionCheck(pipelineId, PipelineActionType.REARRANGE_PIPELETS.name());
96          pipelineEntity = processService.execute(ProcessExecutionContext.<PipelineEntity, PipelineDto>builder()
97                  .target(pipelineEntity)
98                  .actionType(PipelineActionType.REARRANGE_PIPELETS)
99                  .parameterValue(PipelineRearrangePipeletsAction.PIPELET_IDS.of(pipeletIdsInOrder))
100                 .build());
101 
102         return mapEntityToDto(pipelineEntity);
103     }
104 }
