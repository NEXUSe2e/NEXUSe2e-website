14  @Slf4j
15  public class HttpSenderService extends BaseHttpSenderService {
16  
17      @Override
18      public MessageContext executeInternal(ExecutionContext executionContext) throws NexusException {
19          LOGGER.trace("running http sender");
20  
21          validateParameters();
22  
23          NexusRestClientFactory restClientFactory =
24                  executionContext.getBeanFactory().getBean(NexusRestClientFactory.class);
25  
26          WebClientContext webClientContext =
27                  buildWebClientContext(executionContext, restClientFactory::createContextBuilder);
28          RestClient.Builder restClientBuilder = restClientFactory.createClientBuilder(webClientContext);
29  
30          MultiValueMap<String, String> requestParameters = getRequestParametersFromMetadata(executionContext);
31  
32          addAuthenticationSupport(httpAuthenticationSupportService ->
33                  httpAuthenticationSupportService.addAuthentication(restClientBuilder));
34  
35          RestClient.RequestBodySpec requestBodySpec = restClientBuilder
36                  .baseUrl((String) parameters.value(PARAM_URL))
37                  .build()
38                  .method(HttpMethod.valueOf(parameters.value(PARAM_METHOD)))
39                  .uri(uriBuilder -> uriBuilder.queryParams(requestParameters).build());
40          setContentType(requestBodySpec::contentType, executionContext);
41          setAdditionalHeaders(requestBodySpec::headers, executionContext);
42          addBody(
43                  requestBodySpec::body,
44                  requestBodySpec::header,
45                  inputStream -> requestBodySpec.body(outputStream -> StreamUtils.copy(inputStream, outputStream)),
46                  requestBodySpec::body,
47                  executionContext);
48  
49          return handleResponse(() -> requestBodySpec.retrieve().toEntity(String.class), executionContext);
50      }
51  }
