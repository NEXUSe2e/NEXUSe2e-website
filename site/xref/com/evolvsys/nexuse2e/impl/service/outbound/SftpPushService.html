24  /**
25   * Service to push files to an SFTP server.
26   * <p>
27   * This service allows for the configuration of SFTP parameters such as host, port, user credentials,
28   * and file patterns for saving files on sftp servers.
29   * </p>
30   */
31  @Slf4j
32  public class SftpPushService extends NexusOutflowService implements FileSaveAware, SftpParameterAware {
33  
34      @Override
35      public List<ParameterDefinition> parameterDefinitions() {
36  
37          List<ParameterDefinition> parameterDefinitions = new ArrayList<>();
38  
39          addSftpParameters(parameterDefinitions);
40  
41          parameterDefinitions.addAll(List.of(getCreateMissingDirectories(), getTargetFilename(), getTargetDirectory()));
42  
43          return parameterDefinitions;
44      }
45  
46      @Override
47      @SuppressWarnings("PMD.ExceptionAsFlowControl")
48      public MessageContext executeInternal(ExecutionContext executionContext) throws NexusException {
49          boolean enableDeprecatedAlgorithms = parameters.value(PARAM_ENABLE_DEPRECATED_ALGORITHMS);
50          String hostKeyData = parameters.value(PARAM_HOST_KEY_DATA);
51          String hostKeyAlgorithm = parameters.value(PARAM_HOST_KEY_ALGORITHM);
52          String host = parameters.value(PARAM_HOST);
53          long port = parameters.value(PARAM_PORT);
54          String user = parameters.value(PARAM_LOGIN_USERNAME);
55          Duration serverConnectTimeout = parameters.value(PARAM_SERVER_CONNECT_TIMEOUT);
56          String privateKeyData = parameters.value(PARAM_SSH_KEY_DATA);
57          String privateKeyPassword = parameters.value(PARAM_SSH_KEY_PASSWORD);
58          String password = parameters.value(PARAM_LOGIN_PASSWORD);
59          String filePattern = parameters.value(PARAM_TARGET_FILENAME);
60          String targetDirectory = parameters.value(PARAM_TARGET_DIRECTORY);
61          boolean createDirectory = parameters.value(PARAM_CREATE_DIRECTORY);
62          String serverPath = parameters.value(PARAM_SERVER_PATH);
63  
64          SftpClientConfig clientConfig = SftpClientConfig.builder()
65                  .host(host)
66                  .port((int) port)
67                  .user(user)
68                  .password(password)
69                  .privateKeyData(privateKeyData)
70                  .privateKeyPassword(privateKeyPassword)
71                  .serverRoot(serverPath)
72                  .enableDeprecatedAlgorithms(enableDeprecatedAlgorithms)
73                  .hostKeyAlgorithm(hostKeyAlgorithm)
74                  .hostKeyData(hostKeyData)
75                  .serverConnectTimeout(serverConnectTimeout)
76                  .createDirectory(createDirectory)
77                  .build();
78  
79          MessageContext messageContext = executionContext.getMessageContext();
80          Message message = executionContext.getMessage();
81  
82          try (SftpClientWrapper client = new SftpClientWrapper(clientConfig)) {
83  
84              if (client.rootNotExists()) {
85                  throw new NexusException(NexusError.builder()
86                          .message("Server root path does not exist, must be created manually")
87                          .build());
88              }
89  
90              for (ContextPayload payload : message.getPayloads()) {
91  
92                  String replacedTargetDirectory =
93                          PatternReplaceUtils.replaceTokens(targetDirectory, executionContext, payload);
94  
95                  replacedTargetDirectory = SftpUtils.cleanupAndValidateRelativePath(replacedTargetDirectory);
96  
97                  if (StringUtils.isNotBlank(replacedTargetDirectory)) {
98                      if (createDirectory && client.notExists(replacedTargetDirectory)) {
99                          client.mkDirs(replacedTargetDirectory);
100                     }
101                 }
102 
103                 String filename = PatternReplaceUtils.replaceTokens(filePattern, executionContext, payload);
104                 // first validate path for traversal issues
105                 SftpUtils.validateFilename(filename);
106                 // clean up special characters in filename
107                 filename = NEXUSe2eFileUtils.cleanFileName(filename, "_");
108                 LOGGER.trace(
109                         "applying filePattern {} to payload {} resulted in fileName {}",
110                         filePattern,
111                         payload,
112                         filename);
113 
114                 String filePath = replacedTargetDirectory + filename;
115                 client.persistFile(payload.getContent().getInputStream(), filePath);
116             }
117 
118         } catch (Exception e) {
119             throw new NexusException(
120                     NexusError.builder()
121                             .message("Error while connecting to SFTP server")
122                             .build(),
123                     e);
124         }
125 
126         return messageContext;
127     }
128 }
