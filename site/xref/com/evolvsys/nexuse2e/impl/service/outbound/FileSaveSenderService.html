23  @Slf4j
24  public class FileSaveSenderService extends NexusOutflowService implements FileSaveAware {
25  
26      @Override
27      public MessageContext executeInternal(ExecutionContext executionContext) throws NexusException {
28          LOGGER.trace("running file save sender.");
29          MessageContext messageContext = executionContext.getMessageContext();
30          Message message = executionContext.getMessage();
31  
32          if (StringUtils.isBlank(parameters.value(PARAM_TARGET_DIRECTORY))) {
33              throw new IllegalArgumentException("parameter 'Directory' must be provided to the sender service!");
34          }
35  
36          String directoryString = parameters.value(PARAM_TARGET_DIRECTORY);
37          LOGGER.trace("replacing directory pattern {}", directoryString);
38          directoryString = PatternReplaceUtils.replaceContextTokens(directoryString, executionContext);
39          directoryString = directoryString.replace('\\', '/');
40  
41          try {
42              String filePattern = parameters.value(PARAM_TARGET_FILENAME);
43              persistPayloads(
44                      executionContext, message, directoryString, filePattern, parameters.value(PARAM_CREATE_DIRECTORY));
45  
46          } catch (IOException e) {
47              throw new NexusException(
48                      NexusError.builder()
49                              .message("Error while writing file to disk")
50                              .extension("reason", e.getMessage())
51                              .build(),
52                      e);
53          }
54          return messageContext;
55      }
56  
57      @Override
58      public List<ParameterDefinition> parameterDefinitions() {
59          return List.of(getTargetDirectory(), getTargetFilename(), getCreateMissingDirectories());
60      }
61  }
