106     @Override
107     public MessageContext executeInternal(ExecutionContext executionContext) throws NexusException {
108         String endpoint = parameters.value(PARAM_ENDPOINT_URL);
109 
110         BlobServiceClientBuilder clientBuilder = new BlobServiceClientBuilder().endpoint(endpoint);
111 
112         AuthenticationMethod authenticationMethod =
113                 EnumUtils.getEnum(AuthenticationMethod.class, parameters.value(PARAM_AUTHENTICATION_METHOD));
114 
115         if (authenticationMethod == AuthenticationMethod.SAS) {
116             // since SAS authentication is done via the endpoint url, we log a different message here without exposing
117             // the url
118             LOGGER.trace("connecting to blob storage with authentication method {}", authenticationMethod);
119         } else {
120             LOGGER.trace(
121                     "connecting to blob storage at {} with authentication method {}", endpoint, authenticationMethod);
122         }
123 
124         //noinspection DataFlowIssue we want to force exhaustive switch here
125         clientBuilder = switch (authenticationMethod) {
126             case TOKEN -> clientBuilder.credential(getTokenCredential(parameters));
127             case SHARED_KEY -> {
128                 String accountName = parameters.value(PARAM_SHARED_ACCOUNT_NAME);
129                 if (StringUtils.isBlank(accountName)) {
130                     throw new NexusException(NexusError.builder()
131                             .message("Shared account name is required for authentication method SHARED_KEY")
132                             .build());
133                 }
134                 String key = parameters.value(PARAM_SHARED_KEY);
135                 if (StringUtils.isBlank(key)) {
136                     throw new NexusException(NexusError.builder()
137                             .message("Shared key is required for authentication method SHARED_KEY")
138                             .build());
139                 }
140                 yield clientBuilder.credential(new StorageSharedKeyCredential(accountName, key));
141             }
142             case SAS ->
143                 clientBuilder; // SAS authentication is done via the endpoint url, no additional credential needed
144         };
145 
146         BlobServiceClient blobServiceClient = clientBuilder.buildClient();
147 
148         BlobContainerClient blobContainerClient =
149                 blobServiceClient.getBlobContainerClient(parameters.value(PARAM_CONTAINER_NAME));
150 
151         final String filePattern = parameters.value(PARAM_TARGET_FILENAME);
152 
153         Message message = executionContext.getMessage();
154 
155         boolean overwrite = parameters.value(PARAM_OVERWRITE);
156 
157         for (ContextPayload payload : message.getPayloads()) {
158             String fileName = PatternReplaceUtils.replaceTokens(filePattern, executionContext, payload);
159             fileName = NEXUSe2eFileUtils.cleanFileName(fileName, "_");
160             LOGGER.trace("replaced filename pattern {} to {}", filePattern, fileName);
161 
162             // blobclient upload needs a stream that supports mark/reset, hence we wrap it in a BufferedInputStream
163             try (InputStream inputStream = payload.getContent().getInputStream();
164                     BufferedInputStream bufferedInputStream = new BufferedInputStream(inputStream)) {
165                 BlobClient blobClient = blobContainerClient.getBlobClient(fileName);
166                 StopWatch stopWatch = StopWatch.createStarted();
167                 blobClient.upload(bufferedInputStream, payload.getSize(), overwrite);
168                 stop();
169                 LOGGER.info(
170                         "uploaded file {} to container {} in {}.",
171                         fileName,
172                         parameters.value(PARAM_CONTAINER_NAME),
173                         stopWatch.getDuration());
174             } catch (IOException e) {
175                 LOGGER.error("error closing payload {} stream after writing file {}", payload, fileName, e);
176             }
177         }
178 
179         return executionContext.getMessageContext();
180     }

184         TOKEN,

186     }
