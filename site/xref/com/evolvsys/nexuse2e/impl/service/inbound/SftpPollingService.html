165     @Override
166     protected List<MessageContext> buildMessageContexts() throws NexusException {
167         Tika tika = new Tika();
168         String serverPath = parameters.value(PARAM_SERVER_PATH);
169 
170         boolean combinePayloads = parameters.value(PARAM_COMBINE_PAYLOADS);
171         PostProcessingTask postProcessing = PostProcessingTask.valueOf(parameters.value(PARAM_POST_PROCESSING_TASK));
172 
173         if (StringUtils.isBlank(serverPath) || !serverPath.endsWith("/")) {
174             serverPath = serverPath + "/";
175         }
176 
177         List<MessageContext> messageContexts = new ArrayList<>();
178         LOGGER.debug("fetching message from sftp server");
179 
180         try {
181 
182             LOGGER.info("fetching files");
183             List<SftpDirEntryWrapper> fileList = new ArrayList<>();
184             for (SftpClient.DirEntry entry :
185                     sftpTransactionContext.getSftpClient().readDir(serverPath)) {
186                 if (entry.getAttributes().isRegularFile()) {
187 
188                     SftpDirEntryWrapper sftpDirEntryWrapper = SftpDirEntryWrapper.of(entry, serverPath);
189                     fileList.add(sftpDirEntryWrapper);
190                 }
191             }
192 
193             List<SftpDirEntryWrapper> filteredFileList = filterFileList(fileList);
194 
195             List<SftpDirEntryWrapper> sortedFileList = sortFileList(filteredFileList);
196 
197             sftpTransactionContext.getSftpDirEntries().addAll(extendedProcessingFileList(sortedFileList));
198 
199             List<ContextPayload> contextPayloads = new ArrayList<>();
200 
201             for (SftpDirEntryWrapper sftpDirEntryWrapper : sftpTransactionContext.getSftpDirEntries()) {
202 
203                 applyProcessingConfig(sftpDirEntryWrapper, postProcessing, serverPath);
204 
205                 FileTime fileTime = sftpDirEntryWrapper.getAttributes().getModifyTime();
206                 long lastModifiedMillis = fileTime.toMillis();
207                 ZonedDateTime lastModifiedZoned = fileTime.toInstant().atZone(ZoneId.of("UTC"));
208 
209                 long fileLength = sftpDirEntryWrapper.getAttributes().getSize();
210 
211                 try (InputStream inputStream =
212                         sftpTransactionContext.getSftpClient().read(sftpDirEntryWrapper.getSource())) {
213 
214                     ContextPayload contextPayload = ContextPayload.builder()
215                             .content(PayloadContent.of(inputStream, fileLength))
216                             .size(fileLength)
217                             .encoding("utf-8")
218                             .contentType(tika.detect(sftpDirEntryWrapper.getFilename()))
219                             .metaData(new MetaData()
220                                     .set(FileConstants.FILE_NAME, sftpDirEntryWrapper.getFilename())
221                                     .set(FileConstants.FILE_SIZE, fileLength)
222                                     .set(FileConstants.FILE_PATH, sftpDirEntryWrapper.getSource())
223                                     .set("LAST_MODIFIED_READABLE", lastModifiedZoned)
224                                     .set(FileConstants.LAST_MODIFIED, lastModifiedMillis))
225                             .build();
226 
227                     LOGGER.debug("creating message context for file {}", sftpDirEntryWrapper.getFilename());
228 
229                     contextPayload.getMetaData().add(SOURCE_FILENAME, sftpDirEntryWrapper.getSource());
230                     contextPayload.getMetaData().add(TARGET_FILENAME, sftpDirEntryWrapper.getTarget());
231                     contextPayload
232                             .getMetaData()
233                             .add(
234                                     "ProcessingBehavior",
235                                     sftpDirEntryWrapper.getProcessingBehavior().name());
236 
237                     contextPayloads.add(contextPayload);
238                 }
239                 ProcessingBehavior processingBehavior = sftpDirEntryWrapper.getProcessingBehavior();
240                 if (processingBehavior.timing == ProcessingBehavior.Timing.PREPROCESSING) {
241                     processingBehavior.processEntry(sftpTransactionContext.getSftpClient(), sftpDirEntryWrapper);
242                 }
243             }
244 
245             TransformatorSupportService transformatorSupportService = parameters.value(PARAM_TRANSFORMATOR_SERVICE);
246 
247             if (transformatorSupportService != null) {
248                 contextPayloads = transformatorSupportService.transform(contextPayloads);
249             }
250 
251             prepareMessageContexts(contextPayloads, messageContexts, combinePayloads);
252 
253         } catch (IOException e) {
254             throw new NexusException(
255                     NexusError.builder().message("failed to process payload").build(), e);
256         }
257         return messageContexts;
258     }

260     @SuppressWarnings("PMD.ExhaustiveSwitchHasDefault") // false positive see https://github.com/pmd/pmd/issues/5514
261     private void applyProcessingConfig(
262             SftpDirEntryWrapper sftpDirEntryWrapper, PostProcessingTask postProcessing, String serverPath)
263             throws NexusException {
264         switch (postProcessing) {
265             case DELETE:
266                 sftpDirEntryWrapper.setProcessingBehavior(ProcessingBehavior.DELETE_AFTER_PROCESSING);
267                 break;
268             case RENAME:
269                 sftpDirEntryWrapper.setProcessingBehavior(ProcessingBehavior.RENAME_AFTER_PROCESSING);
270                 String prefix = parameters.value(PARAM_POST_PROCESSING_RENAMING_PREFIX);
271                 if (StringUtils.isBlank(prefix)) {
272                     prefix = "processed_";
273                 }
274                 sftpDirEntryWrapper.setTarget(serverPath + prefix + sftpDirEntryWrapper.getFilename());
275                 break;
276             case NOTHING:
277                 sftpDirEntryWrapper.setProcessingBehavior(ProcessingBehavior.DO_NOTHING_AFTER_PROCESSING);
278                 break;
279             default:
280                 throw new NexusException(NexusError.builder()
281                         .message("unknown post processing config")
282                         .extension("post_processing", postProcessing)
283                         .build());
284         }
285     }

287     private void prepareMessageContexts(
288             List<ContextPayload> contextPayloads, List<MessageContext> messageContexts, boolean combinePayloads) {
289         String sourcePartnerId = parameters.value(PARAM_SOURCE_PARTNER_ID);
290         String targetPartnerId = parameters.value(PARAM_TARGET_PARTNER_ID);
291         ActionInfo choreographyAndAction = parameters.value(PARAM_CHOREOGRAPHY_ACTION);
292 
293         for (int i = 0; i < contextPayloads.size(); i++) {
294             ContextPayload payload = contextPayloads.get(i);
295             if (combinePayloads && i > 0) {
296                 MessageContext messageContext = messageContexts.getFirst();
297                 messageContext.getSourceMessage().getPayloads().add(payload);
298             } else {
299                 MessageContext messageContext =
300                         generateMessageContext(payload, sourcePartnerId, targetPartnerId, choreographyAndAction);
301                 messageContexts.add(messageContext);
302             }
303         }
304     }

306     /**
307      * Executes the SFTP polling service.
308      * Deprecated algorithms can be enabled for interoperability reasons but require careful consideration of security implications.
309      *
310      * @throws NexusException
311      */
312     @Override
313     @SuppressWarnings("deprecation")
314     public void execute() throws NexusException {
315         sftpTransactionContext = new SFTPTransactionContext();
316         String password = parameters.value(PARAM_LOGIN_PASSWORD);
317         String privateKeyData = parameters.value(PARAM_SSH_KEY_DATA);
318         String privateKeyPassword = parameters.value(PARAM_SSH_KEY_PASSWORD);
319         String host = parameters.value(PARAM_HOST);
320         long port = parameters.value(PARAM_PORT);
321         String hostKeyData = parameters.value(PARAM_HOST_KEY_DATA);
322         String hostKeyAlgorithm = parameters.value(PARAM_HOST_KEY_ALGORITHM);
323         String user = parameters.value(PARAM_LOGIN_USERNAME);
324         boolean enableDeprecatedAlgorithms = parameters.value(PARAM_ENABLE_DEPRECATED_ALGORITHMS);
325         Duration serverConnectTimeout = parameters.value(PARAM_SERVER_CONNECT_TIMEOUT);
326 
327         if (StringUtils.isBlank(user)) {
328             throw new NexusException(NexusError.builder()
329                     .message("The username must not be empty")
330                     .build());
331         }
332 
333         try (SshClient client = SshClient.setUpDefaultClient()) {
334 
335             // TODO: replace with with multiselect and options from enum list as soon as its available
336             if (enableDeprecatedAlgorithms) {
337                 List<NamedFactory<Signature>> signatureFactories = client.getSignatureFactories();
338                 Set<BuiltinSignatures> signatures = EnumSet.of(BuiltinSignatures.dsa);
339                 signatureFactories.addAll(NamedFactory.setUpBuiltinFactories(false, signatures));
340                 client.setSignatureFactories(signatureFactories);
341             }
342             client.start();
343             SftpUtils.applyHostKeyValidation(client, hostKeyAlgorithm, hostKeyData);
344             try (ClientSession session = client.connect(user, host, (int) port)
345                     .verify(serverConnectTimeout)
346                     .getSession()) {
347                 sftpTransactionContext.setClient(client);
348                 sftpTransactionContext.setSession(session);
349 
350                 if (StringUtils.isNotBlank(privateKeyData)) {
351                     SftpUtils.authenticateSessionViaSshKey(session, privateKeyData, privateKeyPassword);
352                 } else if (StringUtils.isNotBlank(password)) {
353                     SftpUtils.authenticateSessionViaPassword(session, password);
354                 } else {
355                     throw new NexusException(NexusError.builder()
356                             .message("SSH Key or password must be provided for authentication")
357                             .build());
358                 }
359                 session.auth().verify(serverConnectTimeout);
360 
361                 sftpTransactionContext.setSftpClient(
362                         SftpClientFactory.instance().createSftpClient(session));
363 
364                 super.execute();
365                 cleanupSftpServer();
366             }
367         } catch (IOException | GeneralSecurityException e) {
368             throw new NexusException(
369                     NexusError.builder()
370                             .message("Error while connecting to SFTP server")
371                             .build(),
372                     e);
373         }
374     }

376     private void cleanupSftpServer() {
377         for (SftpDirEntryWrapper sftpDirEntryWrapper : sftpTransactionContext.getSftpDirEntries()) {
378             ProcessingBehavior processingBehavior = sftpDirEntryWrapper.getProcessingBehavior();
379             if (processingBehavior.timing == ProcessingBehavior.Timing.POSTPROCESSING) {
380                 try {
381                     processingBehavior.processEntry(sftpTransactionContext.getSftpClient(), sftpDirEntryWrapper);
382                 } catch (NexusException e) {
383                     LOGGER.error("can't cleanup file {}", sftpDirEntryWrapper.getSource(), e);
384                 }
385             }
386             LOGGER.info("finished cleanup of file {}", sftpDirEntryWrapper.getSource());
387         }
388     }

390     @Override
391     protected void afterResolving(MessageContext messageContext) {
392         super.afterResolving(messageContext);
393         String source = messageContext
394                 .getSourceMessage()
395                 .getPayloads()
396                 .getFirst()
397                 .getMetaData()
398                 .getFirstOrNull(SOURCE_FILENAME);
399         if (source != null) {
400             sftpTransactionContext.getSftpDirEntries().stream()
401                     .filter(sftpDirEntryWrapper ->
402                             sftpDirEntryWrapper.getSource().equals(source))
403                     .findFirst()
404                     .ifPresent(sftpDirEntryWrapper -> {
405                         sftpDirEntryWrapper.setProcessedCount(sftpDirEntryWrapper.getProcessedCount() + 1);
406                     });
407         }
408     }

410     protected List<SftpDirEntryWrapper> filterFileList(List<SftpDirEntryWrapper> fileList) {
411 
412         String pattern = parameters.value(PARAM_FILENAME_FILTER_PATTERN);
413         if (StringUtils.isBlank(pattern)) {
414             return fileList;
415         }
416         Pattern matcherPattern = Pattern.compile(pattern);
417         List<SftpDirEntryWrapper> filteredFileList = new ArrayList<>();
418         for (SftpDirEntryWrapper sftpDirEntryWrapper : fileList) {
419             Matcher matcher =
420                     matcherPattern.matcher(sftpDirEntryWrapper.getEntry().getFilename());
421             if (matcher.matches()) {
422                 filteredFileList.add(sftpDirEntryWrapper);
423             }
424         }
425         return filteredFileList;
426     }

428     protected List<SftpDirEntryWrapper> sortFileList(List<SftpDirEntryWrapper> fileList) {
429 
430         SftpSortingAttribute sftpSortingAttribute =
431                 SftpSortingAttribute.valueOf(parameters.value(PARAM_FILENAME_SORTING_FIELD));
432         SortingDirection direction = SortingDirection.valueOf(parameters.value(PARAM_FILENAME_SORTING_DIRECTION));
433         if (sftpSortingAttribute != SftpSortingAttribute.NONE) {
434             sftpSortingAttribute.sort(fileList);
435             if (direction == SortingDirection.DESCENDING) {
436                 Collections.reverse(fileList);
437             }
438         }
439         return fileList;
440     }

453     private MessageContext generateMessageContext(
454             ContextPayload payload, String sourcePartnerId, String targetPartnerId, ActionInfo choreographyAndAction) {
455 
456         MessageContext messageContext = contextFactory.createNew();
457         Message message = new Message();
458         message.setType(MessageType.NORMAL);
459         messageContext.setSourceMessage(message);
460         message.getPayloads().add(payload);
461 
462         if (StringUtils.isNotBlank(sourcePartnerId)) {
463             messageContext.setSourcePartnerId(sourcePartnerId);
464         }
465         if (StringUtils.isNotBlank(targetPartnerId)) {
466             messageContext.setTargetPartnerId(targetPartnerId);
467         }
468         if (choreographyAndAction != null) {
469             if (StringUtils.isNotBlank(choreographyAndAction.getChoreographyId())) {
470                 messageContext.setChoreographyId(choreographyAndAction.getChoreographyId());
471             }
472             if (StringUtils.isNotBlank(choreographyAndAction.getActionId())) {
473                 messageContext.setActionId(choreographyAndAction.getActionId());
474             }
475         }
476 
477         return messageContext;
478     }
