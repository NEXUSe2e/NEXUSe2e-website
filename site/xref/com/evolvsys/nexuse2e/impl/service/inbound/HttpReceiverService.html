104     protected ContextPayload errorToPayload(Error error) {
105         String message = error.getMessage();
106 
107         StringBuilder s =
108                 new StringBuilder(ObjectUtils.defaultIfNull(message, "").trim());
109         for (Map.Entry<String, Object> entry : error.getExtensions().entrySet()) {
110             s.append('\n').append(entry.getKey()).append(": ").append(entry.getValue());
111         }
112 
113         byte[] bytes = s.toString().getBytes(StandardCharsets.UTF_8);
114 
115         return ContextPayload.builder()
116                 .size(bytes.length)
117                 .encoding("utf-8")
118                 .contentType("text/plain")
119                 .content(PayloadContent.of(bytes))
120                 .build();
121     }

130     @Override
131     protected <T extends Exception & ErrorCodeException> MessageContext handleResolverError(
132             MessageContext messageContext, T errorCodeException) throws NexusException {
133         if (httpServletResponse != null && errorMessageResolver != null) {
134             messageContext = super.handleResolverError(messageContext, errorCodeException);
135 
136             List<ContextPayload> errorData = messageContext.getErrorData();
137             if (CollectionUtils.isEmpty(errorData)) {
138                 throw new NexusException(NexusError.builder()
139                         .message("Got no error data after error message resolver")
140                         .extension("service", getServiceId())
141                         .extension(
142                                 "sourceMessageId",
143                                 messageContext.getSourceMessage().getId())
144                         .build());
145             }
146 
147             // status must set before writing to the output stream, or it will stay at SC_OK
148             httpServletResponse.setStatus(HttpServletResponse.SC_BAD_REQUEST);
149             LOGGER.trace("set response status to {}", httpServletResponse.getStatus());
150 
151             long contentLength = 0;
152             LOGGER.trace("write {} error payloads to response", errorData.size());
153             for (ContextPayload payload : errorData) {
154                 try (InputStream inputStream = payload.getContent().getInputStream()) {
155                     LOGGER.trace("add payload {} to response", payload);
156                     inputStream.transferTo(httpServletResponse.getOutputStream());
157                     contentLength += payload.getSize();
158                 } catch (IOException e) {
159                     throw new NexusException(
160                             NexusError.builder()
161                                     .message("Could not read error message")
162                                     .extension("service", getServiceId())
163                                     .extension(
164                                             "sourceMessageId",
165                                             messageContext.getSourceMessage().getId())
166                                     .build(),
167                             e);
168                 }
169             }
170             httpServletResponse.setContentLengthLong(contentLength);
171         }
172         return messageContext;
173     }

175     @Override
176     @SuppressFBWarnings("EXS_EXCEPTION_SOFTENING_NO_CHECKED")
177     protected void afterResolving(MessageContext messageContext) {
178         super.afterResolving(messageContext);
179 
180         if (httpServletResponse != null
181                 && messageContext.getReturnMessage() == null
182                 && getReturnMessageResolver() != null) {
183             MessageContext returnContext;
184             try {
185                 // we need a new context to prevent concurrency issues with the original message context
186                 returnContext = contextFactory.createForResend(
187                         messageContext.getSourceMessage().getId());
188                 returnContext.setSourcePartnerId(messageContext.getTargetPartnerId());
189                 returnContext.setTargetPartnerId(messageContext.getSourcePartnerId());
190 
191                 Message returnMessage = new Message();
192                 returnMessage.setType(MessageType.RETURN);
193                 returnMessage.setConversationId(
194                         messageContext.getSourceMessage().getConversationId());
195                 returnMessage.setReferenceMessageId(
196                         messageContext.getSourceMessage().getId());
197 
198                 returnContext.setReturnMessage(returnMessage);
199                 returnContext = getReturnMessageResolver().resolve(returnContext);
200             } catch (NexusException e) {
201                 throw new NexusRuntimeException(
202                         NexusError.builder()
203                                 .message("Could not create response message")
204                                 .build(),
205                         e);
206             }
207 
208             long contentLength = 0;
209             ContextPayload payload = returnContext.getReturnData();
210             if (payload != null) {
211                 try (InputStream inputStream = payload.getContent().getInputStream()) {
212                     inputStream.transferTo(httpServletResponse.getOutputStream());
213                     contentLength += payload.getSize();
214                 } catch (IOException e) {
215                     throw new NexusRuntimeException(
216                             NexusError.builder()
217                                     .message("could not write return payload")
218                                     .build(),
219                             e);
220                 }
221             }
222             httpServletResponse.setContentLengthLong(contentLength);
223             httpServletResponse.setStatus(HttpServletResponse.SC_OK);
224         }
225     }

235                 if (payload != null) {
236                     try (InputStream inputStream = payload.getContent().getInputStream()) {
237                         inputStream.transferTo(httpServletResponse.getOutputStream());
238                         contentLength += payload.getSize();

240                         throw new NexusException(

244                                 e);
245                     }
246                 }
247                 httpServletResponse.setContentLengthLong(contentLength);
248                 httpServletResponse.setStatus(HttpServletResponse.SC_OK);
249             } else {
250                 httpServletResponse.setContentLengthLong(0);
251                 httpServletResponse.setStatus(HttpServletResponse.SC_OK);

256     protected void validateBasicAuth() throws NexusException {
257         String basicAuthUser = parameters.value(PARAM_BASIC_AUTH_USER);
258         String basicAuthPassword = parameters.value(PARAM_BASIC_AUTH_PASSWORD);
259 
260         if (StringUtils.isAnyBlank(basicAuthUser, basicAuthPassword)) {
261             LOGGER.trace("no basic auth requirement, skip header evaluation");
262         } else {
263             LOGGER.trace("basic auth requirement for user {}", basicAuthUser);
264 
265             Enumeration<String> authorizationHeaders = httpServletRequest.getHeaders(HttpHeaders.AUTHORIZATION);
266             boolean foundUser = false;
267             while (authorizationHeaders.hasMoreElements() && (foundUser == false)) {
268                 String header = authorizationHeaders.nextElement();
269                 if (StringUtils.startsWithIgnoreCase(header, "Basic ")) {
270                     String pair = new String(Base64.getDecoder().decode(header.substring(6)), StandardCharsets.UTF_8);
271                     String[] values = pair.split(":");
272                     if (values.length == 2) {
273                         LOGGER.trace("found basicAuth header with user {}", values[0]);
274                         foundUser = values[0].equals(basicAuthUser) && values[1].equals(basicAuthPassword);
275                     } else {
276                         LOGGER.warn("basic auth header was invalid: {}", header);
277                     }
278                 }
279             }
280 
281             if (foundUser) {
282                 LOGGER.trace("found valid basic auth header in request");
283             } else {
284                 throw new NexusException(NexusError.builder()
285                         .message("Request did not include a valid basic authorization header")
286                         .build());
287             }
288         }

291     @Override
292     @SuppressWarnings("PMD.NcssCount")
293     protected List<MessageContext> buildMessageContexts() throws NexusException {
294         LOGGER.trace("running http service");
295 
296         validateBasicAuth();
297 
298         List<ContextPayload> contextPayloads = new ArrayList<>();
299         List<MessageContext> messageContexts = new ArrayList<>();
300         try {
301             String ipAddress = httpServletRequest.getHeader("X-FORWARDED-FOR");
302             LOGGER.trace("X-FORWARDED-FOR-IP; {}", ipAddress);
303             LOGGER.trace("Client IP: {}", httpServletRequest.getRemoteAddr());
304 
305             MetaData metaData = new MetaData();
306 
307             metaData.set(HttpConstants.CLIENT_IP, httpServletRequest.getRemoteAddr());
308             metaData.set(HttpConstants.CLIENT_FORWARDED_IP, ipAddress);
309 
310             boolean isSecure = httpServletRequest.isSecure();
311             LOGGER.trace("Secure Request: {}", isSecure);
312             metaData.set(ConnectionConstants.SECURE, isSecure);
313 
314             List<String> requestAttributes = Collections.list(httpServletRequest.getAttributeNames());
315 
316             String contentType;
317             if (parameters.value(PARAM_FIX_CONTENT_TYPE)) {
318                 contentType = HttpUtils.fixContentType(httpServletRequest.getContentType());
319                 LOGGER.trace("modified content-type from {} to {}", httpServletRequest.getContentType(), contentType);
320             } else {
321                 contentType = httpServletRequest.getContentType();
322             }
323 
324             if (StringUtils.isNotBlank(contentType)) {
325                 metaData.set(HttpConstants.CONTENT_TYPE, MediaType.parseMediaType(contentType));
326             }
327 
328             String protocolVersion =
329                     (String) httpServletRequest.getAttribute(HttpConstants.REQUEST_PROTOCOL_VERSION_KEY);
330             LOGGER.debug("got protocol version {} from http request", protocolVersion);
331             metaData.set(HttpConstants.PROTOCOL_VERSION, protocolVersion);
332 
333             String cipherSuite = (String) httpServletRequest.getAttribute(HttpConstants.REQUEST_CIPHER_SUITE_KEY);
334             LOGGER.debug("got cipher suite {} from http request", cipherSuite);
335             metaData.set(HttpConstants.CIPHERS, cipherSuite);
336 
337             Enumeration<String> headerNames = httpServletRequest.getHeaderNames();
338 
339             while (headerNames.hasMoreElements()) {
340                 String headerName = headerNames.nextElement();
341                 Enumeration<String> values = httpServletRequest.getHeaders(headerName);
342                 while (values.hasMoreElements()) {
343                     metaData.add(HttpConstants.PREFIX_REQUEST_HEADER, headerName, values.nextElement());
344                 }
345             }
346 
347             Enumeration<String> parameterNames = httpServletRequest.getParameterNames();
348             while (parameterNames.hasMoreElements()) {
349                 String parameterName = parameterNames.nextElement();
350                 metaData.set(
351                         HttpConstants.PREFIX_RECEIVED_REQUEST_PARAMETER,
352                         parameterName,
353                         httpServletRequest.getParameterValues(parameterName));
354             }
355 
356             metaData.set(HttpConstants.CONTENT_SIZE, (long) httpServletRequest.getContentLength());
357             metaData.set(HttpConstants.ENCODING, httpServletRequest.getCharacterEncoding());
358 
359             if (MediaType.MULTIPART_RELATED.includes(metaData.getFirstOrNull(HttpConstants.CONTENT_TYPE))
360                     && !requestAttributes.contains(CustomServletMultipartResolver.MULTIPART_SINGLE_PAYLOAD)) {
361                 LOGGER.trace("getting payload from multipart request");
362                 int i = 0;
363                 for (Part multipart : httpServletRequest.getParts()) {
364                     ContextPayload contextPayload = buildContextPayload(
365                             multipart.getInputStream(),
366                             (int) multipart.getSize(),
367                             multipart.getContentType(),
368                             httpServletRequest.getCharacterEncoding());
369 
370                     contextPayload.setContentId(multipart.getHeader(HttpConstants.CONTENT_ID.getName()));
371                     contextPayload.setSequenceNumber(i++);
372                     contextPayload.getMetaData().set(HttpConstants.CONTENT_SIZE, multipart.getSize());
373                     contextPayload.getMetaData().set(HttpConstants.ENCODING, httpServletRequest.getCharacterEncoding());
374                     contextPayload.getMetaData().set("multipart-name", multipart.getName());
375 
376                     multipart
377                             .getHeaderNames()
378                             .forEach(header -> contextPayload.getMetaData().set(header, multipart.getHeader(header)));
379 
380                     contextPayloads.add(contextPayload);
381                 }
382             } else if (httpServletRequest.getContentLength() > 0) {
383                 LOGGER.trace(
384                         "getting payload from http request input stream with size {}",
385                         httpServletRequest.getContentLength());
386                 ContextPayload contextPayload = buildContextPayload(
387                         httpServletRequest.getInputStream(),
388                         httpServletRequest.getContentLength(),
389                         contentType,
390                         httpServletRequest.getCharacterEncoding());
391                 contextPayloads.add(contextPayload);
392                 setMultipartContentType(contextPayload, metaData);
393             } else {
394                 LOGGER.trace("request does not contain content length, trying to read from input stream");
395                 PayloadContent payloadContent;
396                 try (PayloadContentOutputStream outputStream = PayloadContent.createOutputStream();
397                         InputStream inputStream = httpServletRequest.getInputStream()) {
398                     inputStream.transferTo(outputStream);
399                     payloadContent = outputStream.finish();
400                 }
401 
402                 if (payloadContent.getSize() > 0) {
403                     LOGGER.trace("read payload from input stream with size {}", payloadContent.getSize());
404                     ContextPayload contextPayload = ContextPayload.builder()
405                             .content(payloadContent)
406                             .size(payloadContent.getSize())
407                             .contentType(contentType)
408                             .encoding(httpServletRequest.getCharacterEncoding())
409                             .contentId(httpServletRequest.getHeader(HttpConstants.CONTENT_ID.getName()))
410                             .build();
411                     contextPayloads.add(contextPayload);
412                     setMultipartContentType(contextPayload, metaData);
413                 } else {
414                     LOGGER.trace("no content found in request input stream");
415                 }
416             }
417 
418             TransformatorSupportService transformatorSupportService = parameters.value(PARAM_TRANSFORMATOR_SERVICE);
419 
420             if (transformatorSupportService == null) {
421                 MessageContext messageContext = contextFactory.createNew();
422                 Message message = new Message();
423                 message.setType(MessageType.NORMAL);
424                 messageContext.setSourceMessage(message);
425                 messageContext.getMetaData().add(metaData);
426                 messageContext.getSourceData().addAll(contextPayloads);
427                 setClientCert(messageContext);
428                 messageContexts.add(messageContext);
429             } else {
430                 contextPayloads = transformatorSupportService.transform(contextPayloads);
431                 for (ContextPayload contextPayload : contextPayloads) {
432                     MessageContext messageContext = contextFactory.createNew();
433                     Message message = new Message();
434                     message.setType(MessageType.NORMAL);
435                     messageContext.setSourceMessage(message);
436                     messageContext.getMetaData().add(metaData);
437                     messageContext.getSourceData().add(contextPayload);
438                     setClientCert(messageContext);
439                     messageContexts.add(messageContext);
440                 }
441             }
442 
443             if (LOGGER.isTraceEnabled()) {
444                 messageContexts.forEach(context -> {
445                     for (ContextPayload payload : context.getSourceData()) {
446                         LOGGER.trace(
447                                 "received payload {}: {}",
448                                 payload.getContentId(),
449                                 payload.getContent().getString(Charset.forName(payload.getEncoding())));
450                     }
451                 });
452             }
453 
454             LOGGER.trace("created http data with {}", metaData);
455         } catch (IOException e) {
456             throw new NexusException(
457                     NexusError.builder()
458                             .message("could not get content from http message")
459                             .build(),
460                     e);
461         } catch (ServletException e) {
462             throw new NexusException(
463                     NexusError.builder()
464                             .message("request ist not of type 'multipart'")
465                             .build(),
466                     e);
467         }
468 
469         return messageContexts;
470     }

472     private void setMultipartContentType(ContextPayload contextPayload, MetaData metaData) {
473         if (MediaType.MULTIPART_RELATED.includes(metaData.getFirstOrNull(HttpConstants.CONTENT_TYPE))) {
474             LOGGER.trace("content type is already multipart/related, skip multipart detection");
475         } else if (parameters.value(PARAM_MULTIPART_DETECTION)) {
476             byte[] mimeBoundaryBytes = MIME_BOUNDARY_START.getBytes(Charset.forName(contextPayload.getEncoding()));
477             byte[] firstBytes;
478             try (InputStream contentStream = contextPayload.getContent().getInputStream()) {
479                 firstBytes = contentStream.readNBytes(mimeBoundaryBytes.length);
480             } catch (IOException e) {
481                 LOGGER.warn("could not read first bytes from payload, skip multipart detection", e);
482                 firstBytes = new byte[0];
483             }
484 
485             if (Arrays.equals(mimeBoundaryBytes, firstBytes)) {
486                 LOGGER.info("detected mime boundary in payload - setting content type multipart/related");
487                 metaData.set(HttpConstants.CONTENT_TYPE, MediaType.MULTIPART_RELATED);
488             }
489         } else {
490             LOGGER.trace("multipart detection disabled, skip multipart detection");
491         }
492     }

494     private ContextPayload buildContextPayload(InputStream inputStream, int size, String contentType, String encoding) {
495         return ContextPayload.builder()
496                 .content(PayloadContent.of(inputStream, size))
497                 .size(size)
498                 .contentType(contentType)
499                 .encoding(encoding)
500                 .contentId(httpServletRequest.getHeader(HttpConstants.CONTENT_ID.getName()))
501                 .build();
502     }

504     private void setClientCert(MessageContext messageContext) {
505         X509Certificate[] certs =
506                 (X509Certificate[]) httpServletRequest.getAttribute(HttpConstants.REQUEST_CERTIFICATE_KEY);
507 
508         if (ArrayUtils.isNotEmpty(certs)) {
509             LOGGER.debug("found {} client certs", certs.length);
510             if (LOGGER.isTraceEnabled()) {
511                 for (X509Certificate cert : certs) {
512                     LOGGER.trace("cert: {}", cert);
513                 }
514             }
515             messageContext.setClientCertificate(certs[0]);
516         } else {
517             LOGGER.debug("no client certs found in request");
518         }
