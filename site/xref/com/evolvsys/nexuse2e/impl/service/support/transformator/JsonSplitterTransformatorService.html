34  public class JsonSplitterTransformatorService extends TransformatorSupportService implements BeanFactoryAware {
35      public static final String METADATA_CATEGORY = "json_splitter";
36  
37      private static final String PARAM_ADD_RAW = "add_raw";
38      private static final String PARAM_ARRAY_POINTER = "array_pointer";
39      private static final String PARAM_METADATA_POINTER = "metadata_pointer";
40      private static final String PARAM_CONTENT_POINTER = "content_pointer";

42      protected BeanFactory beanFactory;

44      @Override
45      public List<ParameterDefinition> parameterDefinitions() {
46          return List.of(
47                  ParameterDefinition.builder()
48                          .name(PARAM_ADD_RAW)
49                          .displayName("Add raw payload")
50                          .description(
51                                  "Adds the original raw payload as an additional message to the context with the meta-data 'raw-source' set to 'true'.")
52                          .type(ParameterType.BOOLEAN)
53                          .defaultValue(Boolean.FALSE)
54                          .build(),
55                  ParameterDefinition.builder()
56                          .name(PARAM_ARRAY_POINTER)
57                          .displayName("Json array pointer")
58                          .description(
59                                  "A json pointer to the node where the json splitting should start. Has to point to an array node.")
60                          .type(ParameterType.TRIMMED_STRING)
61                          .build(),
62                  ParameterDefinition.builder()
63                          .name(PARAM_METADATA_POINTER)
64                          .displayName("Json Metadata pointer")
65                          .description(
66                                  "A json pointer to the node whose content should be added as metadata to the payload. If the value is blank, no metadata is included.")
67                          .type(ParameterType.TRIMMED_STRING)
68                          .build(),
69                  ParameterDefinition.builder()
70                          .name(PARAM_CONTENT_POINTER)
71                          .displayName("Json content pointer")
72                          .description(
73                                  "A json pointer to the node whose content should be added as the payload. The pointer is applied for each payload after splitting the json.")
74                          .type(ParameterType.TRIMMED_STRING)
75                          .build());
76      }

78      @Override
79      public List<ContextPayload> transform(Collection<ContextPayload> payloads) throws NexusException {
80          List<ContextPayload> transformedPayloads;
81          if (CollectionUtils.isEmpty(payloads)) {
82              LOGGER.debug("No payloads to transform");
83              transformedPayloads = new ArrayList<>();
84          } else {
85              transformedPayloads = new ArrayList<>();
86              ObjectMapper objectMapper = beanFactory.getBean(ObjectMapper.class);
87  
88              for (ContextPayload payload : payloads) {
89                  try (InputStream inputStream = payload.getContent().getInputStream()) {
90                      if (LOGGER.isTraceEnabled()) {
91                          LOGGER.trace(
92                                  "splitting {} payload {}",
93                                  payload.getEncoding(),
94                                  payload.getContent().getString(Charset.forName(payload.getEncoding())));
95                      }
96  
97                      if (parameters.value(PARAM_ADD_RAW)) {
98                          transformedPayloads.add(payload);
99                          payload.getMetaData().add(META_DATA_RAW, Boolean.TRUE);
100                         LOGGER.trace("adding raw payload to result list");
101                     }
102 
103                     JsonNode jsonNode;
104                     // jackson requires inputStream content encoding to be utf-8. If it is not we need to convert the
105                     // content to in memory string
106                     if (Charset.forName(payload.getEncoding()).equals(StandardCharsets.UTF_8)) {
107                         jsonNode = objectMapper.readTree(inputStream);
108                     } else {
109                         jsonNode = objectMapper.readTree(
110                                 IOUtils.toString(inputStream, Charset.forName(payload.getEncoding())));
111                     }
112 
113                     Optional<JsonNode> optionalMetadata = Optional.empty();
114                     String metadataPointer = parameters.value(PARAM_METADATA_POINTER);
115                     if (StringUtils.isNotBlank(metadataPointer)) {
116                         JsonNode metadataContent = jsonNode.at(metadataPointer);
117                         optionalMetadata = Optional.of(metadataContent);
118                         LOGGER.trace("Extracted json metadata/headers {} from payload", metadataContent);
119                     }
120 
121                     String jsonPointer = parameters.value(PARAM_ARRAY_POINTER);
122                     if (StringUtils.isNotBlank(jsonPointer)) {
123                         LOGGER.trace("Splitting payload at json node via pointer {}", jsonPointer);
124                         jsonNode = jsonNode.at(jsonPointer);
125                     }
126 
127                     if (jsonNode.isArray()) {
128                         String contentPointer = parameters.value(PARAM_CONTENT_POINTER);
129                         List<String> extracted = StreamSupport.stream(jsonNode.spliterator(), false)
130                                 .map(node -> navigateWithContentPointer(node, contentPointer))
131                                 .map(JsonNode::toString)
132                                 .toList();
133                         LOGGER.debug(
134                                 "extracted {} payloads from base payload {}", extracted.size(), payload.getContentId());
135 
136                         for (int i = 0; i < extracted.size(); i++) {
137                             String string = extracted.get(i);
138                             ContextPayload contextPayload = stringToPayload(
139                                     string,
140                                     payload.getContentId() + "_" + payload.getSequenceNumber() + "_" + i,
141                                     payload.getEncoding());
142 
143                             optionalMetadata.ifPresent(
144                                     metadataNode -> addMetadataToPayload(contextPayload, metadataNode));
145                             transformedPayloads.add(contextPayload);
146                         }
147                     } else {
148                         LOGGER.error(
149                                 "Json node is not a json array: {}",
150                                 payload.getContent().getString(Charset.forName(payload.getEncoding())));
151                     }
152                 } catch (IOException e) {
153                     LOGGER.error("error while transforming payloads", e);
154                 }
155             }
156         }
157         return transformedPayloads;
158     }
159 
160     private ContextPayload stringToPayload(String content, String name, String charset) {
161         PayloadContent payloadContent = PayloadContent.of(content.getBytes(Charset.forName(charset)));
162         return ContextPayload.builder()
163                 .contentType(MediaType.APPLICATION_JSON_VALUE)
164                 .encoding(charset)
165                 .contentId(name)
166                 .size(payloadContent.getSize())
167                 .content(payloadContent)
168                 .build();
169     }
170 
171     private void addMetadataToPayload(ContextPayload payload, JsonNode metadataNode) {
172         Iterator<Map.Entry<String, JsonNode>> fields = metadataNode.fields();
173         while (fields.hasNext()) {
174             Map.Entry<String, JsonNode> entry = fields.next();
175             payload.getMetaData()
176                     .add(METADATA_CATEGORY, entry.getKey(), entry.getValue().toPrettyString());
177 
178             LOGGER.trace("Added metadata node {} with value {} to the payload.", entry.getKey(), entry.getValue());
179         }
180     }
181 
182     private JsonNode navigateWithContentPointer(JsonNode jsonNode, String contentPointer) {
183         if (StringUtils.isNotBlank(contentPointer)) {
184             LOGGER.debug("Extract payload at json via pointer '{}'", contentPointer);
185             jsonNode = jsonNode.at(contentPointer);
186         }
187         return jsonNode;
188     }
