20  /**
21   * Reader for the {@link JsonToCsvTransformerService}. Reads the context payload's json and procudes
22   * multiple {@link Map}, consisting of the key (json field name) and the optional value as a generic {@link Object}.
23   *
24   * @author Julian Drees
25   * @since 09.09.2025
26   */
27  @RequiredArgsConstructor
28  public class JsonToCsvReader {
29      private final ObjectMapper objectMapper;
30      private final List<KeyValue<String, String>> fieldColumnMappings;
31  
32      @NonNull
33      public Iterator<Map<String, Optional<Object>>> iterator(ContextPayload originalPayload) throws NexusException {
34          try {
35              JsonNode root = objectMapper.readTree(originalPayload.getContent().getInputStream());
36              if (!root.isArray()) {
37                  throw new NexusException(NexusError.builder()
38                          .message("Payload '%s' does not start with a root array."
39                                  .formatted(originalPayload.getContentId()))
40                          .build());
41              }
42  
43              return StreamEx.of(root.elements())
44                      .map(jsonNode -> StreamEx.of(fieldColumnMappings)
45                              .map(KeyValue::getKey)
46                              .toMap(Function.identity(), key -> getTypedValue(jsonNode.get(key))))
47                      .iterator();
48          } catch (IOException ex) {
49              throw new NexusException(
50                      NexusError.builder()
51                              .message("Failed to parse json content for payload '%s'."
52                                      .formatted(originalPayload.getContentId()))
53                              .build(),
54                      ex);
55          }
56      }
57  
58      @NonNull
59      private Optional<Object> getTypedValue(JsonNode jsonNode) {
60          if (jsonNode == null) {
61              return Optional.empty();
62          }
63          Object result =
64                  switch (jsonNode.getNodeType()) {
65                      case STRING -> jsonNode.asText();
66                      case NUMBER -> {
67                          if (jsonNode.isDouble()) {
68                              yield BigDecimal.valueOf(jsonNode.asDouble());
69                          }
70                          yield jsonNode.asLong();
71                      }
72                      case BOOLEAN -> jsonNode.asBoolean();
73                      case NULL -> null;
74                      default ->
75                          throw new UnsupportedOperationException("Unsupported node type: " + jsonNode.getNodeType());
76                  };
77          return Optional.ofNullable(result);
78      }
79  }
