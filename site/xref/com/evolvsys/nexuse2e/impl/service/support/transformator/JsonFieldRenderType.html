6   import java.text.ParseException;

8   import lombok.extern.slf4j.Slf4j;
9   import org.apache.commons.lang3.StringUtils;

11  @Slf4j
12  public enum JsonFieldRenderType {
13      NUMBER {
14          @Override
15          public void render(JsonGenerator currentJsonGenerator, String key, String value) {
16              try {
17                  if (StringUtils.isBlank(value)) {
18                      currentJsonGenerator.writeNullField(key);
19                  } else {
20                      parseAndWriteNumber(currentJsonGenerator, key, value, Locale.US);
21                  }
22  
23              } catch (IOException e) {
24                  LOGGER.error("failed to render number field", e);
25              }
26          }
27      },
28      NUMBER_DE {
29          @Override
30          public void render(JsonGenerator currentJsonGenerator, String key, String value) {
31              try {
32                  if (StringUtils.isBlank(value)) {
33                      currentJsonGenerator.writeNullField(key);
34                  } else {
35                      parseAndWriteNumber(currentJsonGenerator, key, value, Locale.GERMANY);
36                  }
37              } catch (IOException e) {
38                  LOGGER.error("failed to render number field", e);
39              }
40          }
41      },
42      STRING {
43          @Override
44          public void render(JsonGenerator currentJsonGenerator, String key, String value) {
45              try {
46                  currentJsonGenerator.writeStringField(key, value);
47              } catch (IOException e) {
48                  LOGGER.error("failed to render string field", e);
49              }
50          }
51      },
52      BOOLEAN {
53          @Override
54          public void render(JsonGenerator currentJsonGenerator, String key, String value) {
55              try {
56                  currentJsonGenerator.writeBooleanField(key, Boolean.parseBoolean(value));
57              } catch (IOException e) {
58                  LOGGER.error("failed to render boolean field", e);
59              }
60          }
61      };
62  
63      private static void parseAndWriteNumber(
64              JsonGenerator currentJsonGenerator, String key, String value, Locale locale) {
65          try {
66              Number number = NumberFormat.getInstance(locale).parse(value);
67              if (number instanceof Long) {
68                  long result = number.longValue();
69                  currentJsonGenerator.writeNumberField(key, result);
70              } else {
71                  double result = number.doubleValue();
72                  currentJsonGenerator.writeNumberField(key, result);
73              }
74          } catch (ParseException | IOException e) {
75              LOGGER.error("failed to render number field, not a valid number: >{}<", value, e);
76          }
77      }
78  
79      public abstract void render(JsonGenerator currentJsonGenerator, String key, String value);
80  }
