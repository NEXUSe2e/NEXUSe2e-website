3   import com.evolvsys.nexuse2e.core.exception.NexusError;
4   import com.evolvsys.nexuse2e.core.exception.NexusException;
5   import com.evolvsys.nexuse2e.core.parameter.ParameterDefinition;
6   import com.evolvsys.nexuse2e.core.parameter.ParameterType;
7   import java.util.List;
8   import java.util.function.Predicate;
9   import lombok.extern.slf4j.Slf4j;
10  import org.apache.commons.collections4.CollectionUtils;
11  import org.apache.commons.collections4.KeyValue;
12  
13  /**
14   * Maps string values based on a provided mapping table. Optionally fails if no mapping entry is found for the input value.
15   *
16   * @author SvenBarden
17   * @since 08.08.2025
18   */
19  @Slf4j
20  public class StringMapperSupportService extends StandardSupportService {
21      public static final String PARAM_MAPPING = "mapping";
22      public static final String PARAM_FAIL_ON_NOT_FOUND = "fail_on_not_found";
23      public static final String PARAM_IGNORE_CASE = "ignore_case";
24  
25      @Override
26      public List<ParameterDefinition> parameterDefinitions() {
27          return List.of(
28                  ParameterDefinition.builder()
29                          .name(PARAM_MAPPING)
30                          .displayName("Mapping")
31                          .description("The mapping table to use. ")
32                          .required(true)
33                          .type(ParameterType.KEY_VALUE_PAIR_LIST_TRIMMED)
34                          .build(),
35                  ParameterDefinition.builder()
36                          .name(PARAM_FAIL_ON_NOT_FOUND)
37                          .displayName("Fail on not found")
38                          .description(
39                                  "If true, the service throws an error if no mapping was provided for the given input.")
40                          .type(ParameterType.BOOLEAN)
41                          .defaultValue(Boolean.FALSE)
42                          .build(),
43                  ParameterDefinition.builder()
44                          .name(PARAM_IGNORE_CASE)
45                          .displayName("Ignore case")
46                          .description(
47                                  "If true, ignores the case for the mapping. Note that if the same key in multiple cases is provided there is no guarantee which mapping will be used.")
48                          .type(ParameterType.BOOLEAN)
49                          .defaultValue(Boolean.FALSE)
50                          .build());
51      }
52  
53      private Predicate<KeyValue<String, String>> match(String input) {
54          boolean ignoreCase = parameters.value(PARAM_IGNORE_CASE);
55  
56          return ignoreCase
57                  ? keyValue -> keyValue.getKey().equalsIgnoreCase(input)
58                  : keyValue -> keyValue.getKey().equals(input);
59      }
60  
61      public String map(String input) throws NexusException {
62          List<KeyValue<String, String>> mapping = parameters.value(PARAM_MAPPING);
63          boolean failOnNotFound = parameters.value(PARAM_FAIL_ON_NOT_FOUND);
64  
65          String mappedValue = CollectionUtils.emptyIfNull(mapping).stream()
66                  .filter(match(input))
67                  .map(KeyValue::getValue)
68                  .findFirst()
69                  .orElse(null);
70  
71          String result;
72          if (mappedValue == null) {
73              if (failOnNotFound) {
74                  LOGGER.trace("no mapping entries found, throwing exception");
75                  throw new NexusException(NexusError.builder()
76                          .message("No mapping entry found")
77                          .extension("key", input)
78                          .extension(
79                                  "entries",
80                                  mapping.stream().map(KeyValue::getKey).toList())
81                          .build());
82              } else {
83                  LOGGER.trace("no mapping entries found for {}, returning input as is", input);
84                  result = input;
85              }
86          } else {
87              result = mappedValue;
88              LOGGER.trace("mapping entry found, returning mapped value: {}", result);
89          }
90  
91          return result;
92      }
93  }
