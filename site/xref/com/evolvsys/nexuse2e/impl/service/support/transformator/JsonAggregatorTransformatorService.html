109     @Override
110     public List<ContextPayload> transform(Collection<ContextPayload> payloads) throws NexusException {
111         List<ContextPayload> transformedPayloads;
112         if (CollectionUtils.isEmpty(payloads)) {
113             LOGGER.debug("No payloads to transform");
114             transformedPayloads = new ArrayList<>();
115         } else {
116             transformedPayloads = transformInternal(payloads);
117         }
118         return transformedPayloads;
119     }
120 
121     private List<ContextPayload> transformInternal(Collection<ContextPayload> payloads) throws NexusException {
122         long objectsPerMessage = parameters.value(PARAM_PAYLOADS_PER_MESSAGE);
123 
124         List<ContextPayload> payloadsToProcess = new ArrayList<>(payloads);
125 
126         List<List<ContextPayload>> payloadPartition = ListUtils.partition(payloadsToProcess, (int) objectsPerMessage);
127         LOGGER.trace(
128                 "split {} payloads into {} partitions of size {}",
129                 payloadsToProcess.size(),
130                 payloadPartition.size(),
131                 objectsPerMessage);
132 
133         List<ContextPayload> processedPayloads = new ArrayList<>(payloadPartition.size());
134 
135         JsonAggregatorProcessor processor = JsonAggregatorProcessor.create(this);
136 
137         for (List<ContextPayload> contextPayloads : payloadPartition) {
138             processedPayloads.add(processor.process(contextPayloads));
139         }
140         return processedPayloads;
141     }
