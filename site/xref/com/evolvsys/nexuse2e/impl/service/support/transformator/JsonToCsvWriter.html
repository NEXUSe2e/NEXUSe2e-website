23  /**
24   * Csv writer for the {@link JsonToCsvTransformerService}.
25   *
26   * @author Julian Drees
27   * @since 09.09.2025
28   */
29  @Slf4j
30  public class JsonToCsvWriter {
31      private final List<String> headers;
32      private final Map<String, String> fieldColumnMapping;
33      private final boolean includeHeaders;
34      private final Charset charset;
35      private final char columnSeparator;
36  
37      public JsonToCsvWriter(
38              Collection<KeyValue<String, String>> fieldColumnMappings,
39              boolean includeHeaders,
40              Charset charset,
41              char columnSeparator) {
42          super();
43          this.headers = fieldColumnMappings.stream().map(KeyValue::getValue).toList();
44          this.fieldColumnMapping =
45                  fieldColumnMappings.stream().collect(Collectors.toMap(KeyValue::getKey, KeyValue::getValue));
46          this.includeHeaders = includeHeaders;
47          this.charset = charset;
48          this.columnSeparator = columnSeparator;
49      }
50  
51      @NonNull
52      public PayloadContent write(Iterator<Map<String, Optional<Object>>> iterator) throws IOException {
53          CsvMapper csvMapper = new CsvMapper();
54          CsvSchema csvSchema = createSchema();
55  
56          try (PayloadContentOutputStream outputStream = PayloadContent.createOutputStream();
57                  OutputStreamWriter outputStreamWriter = new OutputStreamWriter(outputStream, charset);
58                  SequenceWriter sequenceWriter = csvMapper.writer(csvSchema).writeValues(outputStreamWriter)) {
59              if (includeHeaders) {
60                  sequenceWriter.write(headers);
61              }
62  
63              while (iterator.hasNext()) {
64                  Map<String, Object> row = EntryStream.of(iterator.next().entrySet().stream())
65                          .collect(StreamUtils.toMapOfNullables(
66                                  entry -> fieldColumnMapping.get(entry.getKey()),
67                                  entry -> entry.getValue().orElse(null)));
68  
69                  if (row.isEmpty()) {
70                      LOGGER.warn("Tried to write empty csv row.");
71                      continue;
72                  }
73  
74                  LOGGER.trace("Writing csv row: {}", row);
75                  sequenceWriter.write(row);
76              }
77              return outputStream.finish();
78          }
79      }
80  
81      private CsvSchema createSchema() {
82          CsvSchema.Builder csvSchemaBuilder = CsvSchema.builder().setColumnSeparator(columnSeparator);
83          for (String column : headers) {
84              csvSchemaBuilder.addColumn(column);
85          }
86          return csvSchemaBuilder.build();
87      }
88  }
