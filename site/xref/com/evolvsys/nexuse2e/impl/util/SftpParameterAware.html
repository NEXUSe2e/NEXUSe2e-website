8   public interface SftpParameterAware {
9   
10      String PARAM_LOGIN_USERNAME = "login_username";
11      String PARAM_SSH_KEY_DATA = "ssh_key_data";
12      String PARAM_SSH_KEY_PASSWORD = "ssh_key_password";
13      String PARAM_LOGIN_PASSWORD = "login_password";
14      String PARAM_HOST = "host";
15      String PARAM_PORT = "port";
16      String PARAM_HOST_KEY_ALGORITHM = "host_key_algorithm";
17      String PARAM_HOST_KEY_DATA = "host_key_data";
18      String PARAM_SERVER_PATH = "server_path";
19      String PARAM_SERVER_CONNECT_TIMEOUT = "server_connect_timeout";
20      String PARAM_ENABLE_DEPRECATED_ALGORITHMS = "enable_deprecated_algorithms";
21  
22      default void addSftpParameters(List<ParameterDefinition> parameterDefinitions) {
23  
24          parameterDefinitions.addAll(List.of(
25                  ParameterDefinition.builder()
26                          .name(PARAM_LOGIN_USERNAME)
27                          .displayName("Login User Name")
28                          .description("The Server Login Username.")
29                          .type(ParameterType.TRIMMED_STRING)
30                          .required(true)
31                          .build(),
32                  ParameterDefinition.builder()
33                          .name(PARAM_LOGIN_PASSWORD)
34                          .displayName("Login Password")
35                          .description("The Login Password. Its optional, if the ssh key is provided.")
36                          .type(ParameterType.PASSWORD)
37                          .required(false)
38                          .build(),
39                  ParameterDefinition.builder()
40                          .name(PARAM_SSH_KEY_DATA)
41                          .displayName("SSH Key PEM Data")
42                          .description(
43                                  "Optional ssh key, used for authentication in combination with the proper username.")
44                          .type(ParameterType.LARGE_STRING)
45                          .required(false)
46                          .build(),
47                  ParameterDefinition.builder()
48                          .name(PARAM_SSH_KEY_PASSWORD)
49                          .displayName("SSH Key password")
50                          .description("Optional password for the also optional ssh key.")
51                          .type(ParameterType.PASSWORD)
52                          .required(false)
53                          .build(),
54                  ParameterDefinition.builder()
55                          .name(PARAM_HOST)
56                          .displayName("Hostname")
57                          .description("The DNS Hostname or IP for the server.")
58                          .type(ParameterType.TRIMMED_STRING)
59                          .required(true)
60                          .build(),
61                  ParameterDefinition.builder()
62                          .name(PARAM_PORT)
63                          .displayName("Port Number")
64                          .description("The tcp port number used to connect to the server.")
65                          .type(ParameterType.LONG)
66                          .defaultValue(22L)
67                          .required(true)
68                          .build(),
69                  ParameterDefinition.builder()
70                          .name(PARAM_HOST_KEY_ALGORITHM)
71                          .displayName("Host Key Algorithm")
72                          .description("The algorithm for the optional host key validation.")
73                          .type(ParameterType.TRIMMED_STRING)
74                          .required(false)
75                          .build(),
76                  ParameterDefinition.builder()
77                          .name(PARAM_HOST_KEY_DATA)
78                          .displayName("Host Key Data")
79                          .description("The base64 encoded key data for the optional host key validation.")
80                          .type(ParameterType.TRIMMED_LARGE_STRING)
81                          .required(false)
82                          .build(),
83                  ParameterDefinition.builder()
84                          .name(PARAM_SERVER_PATH)
85                          .displayName("Server Path")
86                          .description(
87                                  "The root path. Must exist as its not created on runtime. Patterns are not allowed.")
88                          .type(ParameterType.TRIMMED_STRING)
89                          .defaultValue("/")
90                          .required(true)
91                          .build(),
92                  ParameterDefinition.builder()
93                          .name(PARAM_SERVER_CONNECT_TIMEOUT)
94                          .displayName("Connect Timeout")
95                          .description("The initial connect and auth validation timeout.")
96                          .type(ParameterType.DURATION)
97                          .defaultValue(Duration.ofSeconds(10))
98                          .required(true)
99                          .build(),
100                 ParameterDefinition.builder()
101                         .name(PARAM_ENABLE_DEPRECATED_ALGORITHMS)
102                         .displayName("Enable deprecated algorithms")
103                         .description("Optionally enable deprecated/insecure algorithms for interoperability reasons.")
104                         .type(ParameterType.BOOLEAN)
105                         .defaultValue(Boolean.FALSE)
106                         .build()));
107     }
108 }
