9   public interface FileSaveAware {
10      String PARAM_TARGET_DIRECTORY = "target_directory";
11      String PARAM_TARGET_FILENAME = "target_filename";
12      String PARAM_CREATE_DIRECTORY = "create_dir";
13  
14      default ParameterDefinition getCreateMissingDirectories() {
15          return ParameterDefinition.builder()
16                  .name(PARAM_CREATE_DIRECTORY)
17                  .displayName("Create missing directories")
18                  .description("Automatically create missing directories before saving payloads")
19                  .type(ParameterType.BOOLEAN)
20                  .required(true)
21                  .defaultValue(Boolean.FALSE)
22                  .build();
23      }
24  
25      default ParameterDefinition getTargetFilename() {
26          return ParameterDefinition.builder()
27                  .name(PARAM_TARGET_FILENAME)
28                  .displayName("Target Filename")
29                  .description("The filename or pattern used when storing the message payload.")
30                  .defaultValue("{messageId}_{sequenceNumber}{extension}")
31                  .required(true)
32                  .type(ParameterType.TRIMMED_STRING)
33                  .build();
34      }
35  
36      default ParameterDefinition getTargetDirectory() {
37          return ParameterDefinition.builder()
38                  .name(PARAM_TARGET_DIRECTORY)
39                  .displayName("Target Directory")
40                  .description("The target directory where the file is stored.")
41                  .required(true)
42                  .type(ParameterType.TRIMMED_STRING)
43                  .build();
44      }
45  }
