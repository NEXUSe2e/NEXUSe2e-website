9   /**
10   *
11   * @author SvenBarden

13   */
14  public interface HttpConnectionParameters {
15      String PARAM_URL = "url";
16      String PARAM_METHOD = "method";
17      String PARAM_CONNECTION_TIMEOUT = "connection_timeout";
18      String PARAM_RESPONSE_TIMEOUT = "response_timeout";
19      String PARAM_SSL_PROTOCOLS = "ssl_protocols";
20      String PARAM_SSL_CIPHERS = "ssl_ciphers";
21      String PARAM_CONTENT_AS_STREAM = "content_as_stream";
22      String PARAM_ENABLE_HOSTNAME_VERIFICATION = "disable_hostname_verification";
23      String PARAM_VALIDATE_CERT_CHAIN = "validate_cert_chain";
24      String PARAM_AUTH_SERVICE = "auth_service";
25      String MESSAGE_ID = "messageId";
26  
27      Duration DEFAULT_RESPONSE_TIMEOUT = Duration.ofSeconds(30);
28      Duration DEFAULT_CONNECTION_TIMEOUT = Duration.ofSeconds(30);
29  
30      Supplier<ParameterDefinition.ParameterDefinitionBuilder> URL = () -> ParameterDefinition.builder()
31              .name(PARAM_URL)
32              .description("Url where the message should be send to")
33              .type(ParameterType.STRING)
34              .required(true);
35      Supplier<ParameterDefinition.ParameterDefinitionBuilder> METHOD = () -> ParameterDefinition.builder()
36              .name(PARAM_METHOD)
37              .description("Method to indicate which action is to be performed")
38              .type(ParameterType.DROPDOWN_LIST)
39              .required(true)
40              .defaultValue(HttpMethod.POST.toString())
41              .option(HttpMethod.POST.toString())
42              .option(HttpMethod.GET.toString())
43              .option(HttpMethod.PUT.toString());
44      Supplier<ParameterDefinition.ParameterDefinitionBuilder> CONTENT_AS_STREAM = () -> ParameterDefinition.builder()
45              .name(PARAM_CONTENT_AS_STREAM)
46              .displayName("Send content as stream")
47              .description("Sends content as stream. Saves memory for large payloads.")
48              .type(ParameterType.BOOLEAN)
49              .defaultValue(Boolean.FALSE);
50      Supplier<ParameterDefinition.ParameterDefinitionBuilder> CONNECTION_TIMEOUT = () -> ParameterDefinition.builder()
51              .name(PARAM_CONNECTION_TIMEOUT)
52              .displayName("Connection timeout")
53              .description("Amount of time allowed for the connection to be established")
54              .type(ParameterType.DURATION)
55              .defaultValue(DEFAULT_CONNECTION_TIMEOUT);
56      Supplier<ParameterDefinition.ParameterDefinitionBuilder> RESPONSE_TIMEOUT = () -> ParameterDefinition.builder()
57              .name(PARAM_RESPONSE_TIMEOUT)
58              .displayName("Response timeout")
59              .description("Amount of time allowed for the http response to be received")
60              .type(ParameterType.DURATION)
61              .defaultValue(DEFAULT_RESPONSE_TIMEOUT);
62      Supplier<ParameterDefinition.ParameterDefinitionBuilder> SSL_PROTOCOLS = () -> ParameterDefinition.builder()
63              .name(PARAM_SSL_PROTOCOLS)
64              .displayName("SSL protocols")
65              .description(
66                      "SSL/TLS protocol versions to use for the outbound https connection (e.g. \"TLSv1.2, TLSv1.3\") (optional)")
67              .type(ParameterType.STRING_LIST);
68      Supplier<ParameterDefinition.ParameterDefinitionBuilder> SSL_CIPHERS = () -> ParameterDefinition.builder()
69              .name(PARAM_SSL_CIPHERS)
70              .displayName("SSL ciphers")
71              .description(
72                      "SSL ciphers to use for the outbound https connection (\"ECDHE-ECDSA-AES128-GCM-SHA256, ECDHE-ECDSA-CHACHA20-POLY1305\") (optional) ")
73              .type(ParameterType.STRING_LIST);
74      Supplier<ParameterDefinition.ParameterDefinitionBuilder> ENABLE_HOSTNAME_VERIFICATION =
75              () -> ParameterDefinition.builder()
76                      .name(PARAM_ENABLE_HOSTNAME_VERIFICATION)
77                      .displayName("Enable hostname verification")
78                      .description("Enables the hostname verification for https connections")
79                      .type(ParameterType.BOOLEAN)
80                      .defaultValue(Boolean.TRUE);
81      Supplier<ParameterDefinition.ParameterDefinitionBuilder> VALIDATE_CERT_CHAIN = () -> ParameterDefinition.builder()
82              .name(PARAM_VALIDATE_CERT_CHAIN)
83              .displayName("Validate trusted cert chain")
84              .description("Validates the complete cert chain for each trusted partner certificate.")
85              .type(ParameterType.BOOLEAN)
86              .defaultValue(Boolean.TRUE);
87      Supplier<ParameterDefinition.ParameterDefinitionBuilder> AUTH_SERVICE = () -> ParameterDefinition.builder()
88              .name(PARAM_AUTH_SERVICE)
89              .displayName("Authentication service")
90              .description("The Http Support Service to use for authentication")
91              .type(ParameterType.SUPPORT_SERVICE);
92  }
