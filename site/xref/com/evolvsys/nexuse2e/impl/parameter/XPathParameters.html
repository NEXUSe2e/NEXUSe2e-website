1   package com.evolvsys.nexuse2e.impl.parameter;
2   
3   import com.evolvsys.nexuse2e.core.component.Configurable;
4   import com.evolvsys.nexuse2e.core.parameter.ParameterDefinition;
5   import com.evolvsys.nexuse2e.core.parameter.ParameterType;
6   import java.util.function.Supplier;
7   
8   /**
9    * @author SvenBarden
10   * @since 16.09.2024
11   */
12  public interface XPathParameters {
13      String PARAM_TRIM_RESULT = "trim_result";
14      String PARAM_STOP_ON_FAIL = "stop_on_error";
15      String PARAM_STOP_ON_NOT_FOUND = "stop_on_not_found";
16  
17      Supplier<ParameterDefinition.ParameterDefinitionBuilder> TRIM_RESULT = () -> ParameterDefinition.builder()
18              .name(PARAM_TRIM_RESULT)
19              .displayName("Strip whitespace from result")
20              .description("Whether to remove all leading or trailing whitespace characters from the result.")
21              .required(true)
22              .type(ParameterType.BOOLEAN)
23              .defaultValue(Boolean.TRUE);
24      Supplier<ParameterDefinition.ParameterDefinitionBuilder> STOP_ON_FAIL = () -> ParameterDefinition.builder()
25              .name(PARAM_STOP_ON_FAIL)
26              .displayName("Stop on Fail")
27              .description("Whether to cancel processing on failure.")
28              .required(true)
29              .type(ParameterType.BOOLEAN)
30              .defaultValue(Boolean.TRUE);
31      Supplier<ParameterDefinition.ParameterDefinitionBuilder> STOP_ON_NOT_FOUND = () -> ParameterDefinition.builder()
32              .name(PARAM_STOP_ON_NOT_FOUND)
33              .displayName("Stop on empty or not found value")
34              .description("Whether to cancel further processing when the result value for the xpath statement is empty.")
35              .required(true)
36              .type(ParameterType.BOOLEAN)
37              .defaultValue(Boolean.TRUE);
38  
39      default String trimResult(String s, Configurable configurable) {
40          Boolean trimResult = configurable.getParameters().value(PARAM_TRIM_RESULT);
41          if (trimResult) {
42              s = s.trim();
43          }
44          return s;
45      }
46  }
