13  import org.springframework.http.HttpHeaders;
14  import org.springframework.http.MediaType;
15  
16  @Slf4j
17  public class EbxmlMessagePackagerPipelet extends MessagePackagerPipelet {
18  
19      @Override
20      public void addContextPayloads(ExecutionContext executionContext) {
21          if (Objects.requireNonNull(executionContext.getMessage().getType()) == MessageType.ACKNOWLEDGMENT) {
22              addEbxmlHeaderContextPayload(executionContext);
23          } else {
24              addEbxmlHeaderContextPayload(executionContext);
25              super.addContextPayloads(executionContext);
26          }
27  
28          Message message = executionContext.getMessage();
29  
30          message.getMetaData()
31                  .set(HttpConstants.CONTENT_TYPE, EbxmlUtils.getMultipartRelatedContentType(executionContext));
32  
33          HttpHeaders additionalHeaders =
34                  message.getMetaData().getFirstOrDefault(HttpConstants.ADDITIONAL_HEADERS, new HttpHeaders());
35          additionalHeaders.set("SOAPAction", "\"ebXML\"");
36          message.getMetaData().set(HttpConstants.ADDITIONAL_HEADERS, additionalHeaders);
37      }
38  
39      public void addEbxmlHeaderContextPayload(ExecutionContext executionContext) {
40          ContextPayload contextPayload = new ContextPayload();
41          ContextPayload payload =
42                  executionContext.getMessage().getMetaData().getFirstOrThrows(EbxmlConstants.HEADER_DATA);
43          contextPayload.setContent(payload.getContent());
44          contextPayload.setSize(contextPayload.getContent().getSize());
45          contextPayload.setContentType(MediaType.TEXT_XML.toString());
46          contextPayload.setEncoding(payload.getEncoding());
47  
48          HttpHeaders httpHeaders = new HttpHeaders();
49          httpHeaders.set(HttpConstants.CONTENT_ID.getName(), getSoapId(executionContext));
50          contextPayload.getMetaData().set(HttpConstants.ADDITIONAL_HEADERS, httpHeaders);
51  
52          executionContext.setData(contextPayload);
53      }
54  
55      private String getSoapId(ExecutionContext executionContext) {
56          return getSoapId(executionContext.getMessage().getId());
57      }
58  
59      private String getSoapId(String messageId) {
60          return String.format("<%s%s-Header>", messageId, EbxmlConstants.PROTOCOL);
61      }
62  }
