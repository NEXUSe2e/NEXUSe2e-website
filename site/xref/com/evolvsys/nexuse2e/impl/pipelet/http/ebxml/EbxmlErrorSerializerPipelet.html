19   * @author SvenBarden

22  @Slf4j
23  public class EbxmlErrorSerializerPipelet extends Pipelet {
24      @Override
25      public List<ParameterDefinition> parameterDefinitions() {
26          return List.of();
27      }
28  
29      @Override
30      public ExecutionContext execute(ExecutionContext executionContext) throws NexusException {
31          try {
32              LOGGER.debug("serialize ebxml error message");
33              SOAPFactory soapFactory = SOAPFactory.newInstance();
34              MessageFactory messageFactory = MessageFactory.newInstance();
35              SOAPMessage soapMessage = messageFactory.createMessage();
36              soapMessage.setProperty(SOAPMessage.WRITE_XML_DECLARATION, "true");
37              SOAPPart soapPart = soapMessage.getSOAPPart();
38              SOAPEnvelope soapEnvelope = soapPart.getEnvelope();
39              soapEnvelope.addNamespaceDeclaration("xsi", "http://www.w3.org/2001/XMLSchema-instance");
40              soapEnvelope.addAttribute(
41                      soapFactory.createName("xsi:schemaLocation"),
42                      "http://schemas.xmlsoap.org/soap/envelope/ http://www.oasis-open.org/committees/ebxml-msg/schema/envelope.xsd");
43  
44              SOAPBody soapBody = soapEnvelope.getBody();
45              QName faultName = new QName(SOAPConstants.URI_NS_SOAP_ENVELOPE, "Server");
46              SOAPFault soapFault = soapBody.addFault();
47              soapFault.setFaultCode(faultName);
48  
49              soapFault.setFaultString(executionContext
50                      .getMessage()
51                      .getPayloads()
52                      .getFirst()
53                      .getContent()
54                      .getString(StandardCharsets.UTF_8));
55              soapMessage.saveChanges();
56  
57              try (PayloadContentOutputStream outputStream = PayloadContent.createOutputStream()) {
58                  soapMessage.writeTo(outputStream);
59  
60                  PayloadContent payloadContent = outputStream.finish();
61  
62                  executionContext
63                          .getErrorData()
64                          .add(ContextPayload.builder()
65                                  .contentType("text/xml")
66                                  .encoding("utf-8")
67                                  .content(payloadContent)
68                                  .size(payloadContent.getSize())
69                                  .build());
70              }
71          } catch (SOAPException | IOException e) {
72              throw new NexusException(
73                      NexusError.builder()
74                              .message("Could not serialize error message")
75                              .extension(
76                                      "messageId",
77                                      executionContext.getSourceMessage().getId())
78                              .build(),
79                      e);
80          }
81  
82          return executionContext;
83      }
84  }
