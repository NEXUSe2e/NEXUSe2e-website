34      private static final String MESSAGE_ERROR = "MessageError";

38      @Override
39      @SuppressFBWarnings("CFS_CONFUSING_FUNCTION_SEMANTICS")
40      public ExecutionContext execute(ExecutionContext executionContext) throws NexusException {
41          resolveParameters(parameters);
42  
43          try {
44              soapFactory = SOAPFactory.newInstance();
45              Message message = executionContext.getMessage();
46  
47              ContextPayload payload = message.getMetaData().getFirstOrThrows(EbxmlConstants.HEADER_DATA);
48  
49              Optional<String> optionalEncoding =
50                      Optional.ofNullable(payload.getEncoding()).or(() -> EbxmlUtils.getEncoding(executionContext));
51              if (optionalEncoding.isEmpty()) {
52                  LOGGER.error("Encoding of content (of message with id {}) was not provided", message.getId());
53                  throw new NexusException(NexusError.builder()
54                          .message("Encoding of content was not provided")
55                          .build());
56              }
57  
58              Charset charset = Charset.forName(optionalEncoding.orElseThrow());
59  
60              SOAPMessage soapMessage = getSoapMessage(payload.getContent(), charset);
61              SOAPHeader soapHeader = soapMessage.getSOAPHeader();
62  
63              Iterator<Node> soapHeaderElementIterator = soapHeader.getChildElements();
64              while (soapHeaderElementIterator.hasNext()) {
65                  Node node = soapHeaderElementIterator.next();
66                  if (node instanceof SOAPElement soapElement) {
67                      switch (soapElement.getElementName().getLocalName()) {
68                          case "MessageHeader" -> extractMessageHeader(soapElement, executionContext, message);
69                          case "Acknowledgment" -> extractAcknowledgment(soapElement, message);
70                          case "ErrorList" -> extractErrorList(soapElement, executionContext, message);
71                          case "AckRequested" -> extractAckRequested(message);
72                          case "SyncReply" -> extractSyncReply(message);
73                          default -> LOGGER.debug("skip element {}", soapElement.getElementName());
74                      }
75                  }
76              }
77  
78              if (MESSAGE_ERROR.equalsIgnoreCase(executionContext.getActionId())) {
79                  message.setType(MessageType.ERROR);
80              }
81  
82              executionContext.setMessage(message);
83          } catch (SOAPException | IOException e) {
84              throw new NexusException(
85                      NexusError.builder()
86                              .message("could not deserialize ebxml message")
87                              .build(),
88                      e);
89          }
90          return executionContext;
91      }

93      private void extractSyncReply(Message message) {
94          LOGGER.debug("inbound message requires sync reply");
95          message.getMetaData().set(ConnectionConstants.REQUIRES_REPLY, Boolean.TRUE);
96      }

99          LOGGER.debug("inbound message requires async acknowledgment");
100         message.getMetaData().set(ConnectionConstants.REQUIRES_ACK, Boolean.TRUE);
101     }

103     private void extractMessageHeader(SOAPElement soapMessageHeader, ExecutionContext executionContext, Message message)
104             throws SOAPException {
105         message.setType(MessageType.NORMAL);
106         Iterator<Node> headerElementIterator = soapMessageHeader.getChildElements();
107 
108         while (headerElementIterator.hasNext()) {
109             Node node = getNextNonTextNode(headerElementIterator);
110 
111             if (node instanceof SOAPElement soapElement) {
112                 switch (soapElement.getElementName().getLocalName()) {
113                     case "From" -> extractPartyElement(soapElement, executionContext, PartyType.FROM);
114                     case "To" -> extractPartyElement(soapElement, executionContext, PartyType.TO);
115                     case "CPAId" -> extractCpaId(soapElement, executionContext);
116                     case "ConversationId" -> extractConversationId(soapElement, message);
117                     case "Action" -> extractActionId(soapElement, executionContext);
118                     case "MessageData" -> extractMessageData(soapElement, message);
119                     case "DuplicateElimination" -> LOGGER.trace("duplicateElimination flag found!");
120                     case "RefToMessageId" -> extractReferencedMessage(soapElement, message);
121                     default -> LOGGER.debug("skip element {}", soapElement.getElementName());
122                 }
123             }
124         }
125     }

127     private void extractPartyElement(SOAPElement soapElement, ExecutionContext executionContext, PartyType partyType)
128             throws SOAPException {
129         Iterator<Node> innerElementIterator = soapElement.getChildElements();
130         if (innerElementIterator.hasNext()) {
131             Node node = getNextNonTextNode(innerElementIterator);
132             if (node instanceof SOAPElement innerSoapElement) {
133                 String partnerType = innerSoapElement.getAttributeValue(
134                         EbxmlUtils.createNameWithEbxmlNamespace(soapFactory, "type"));
135                 executionContext.getMessage().getMetaData().set(partyType.getPartnerTypeConstant(), partnerType);
136 
137                 String partnerId = innerSoapElement.getValue();
138                 if (StringUtils.isBlank(partnerType) && partnerId.startsWith("uri:")) {
139                     // Cut off 'uri:' as there was no type of party provided
140                     partnerId = partnerId.substring(partnerId.indexOf(':'));
141                 }
142 
143                 if (partyType == PartyType.FROM) {
144                     executionContext.setSourcePartnerId(partnerId);
145                 } else if (partyType == PartyType.TO) {
146                     executionContext.setTargetPartnerId(partnerId);
147                 }
148                 executionContext.getMessage().getMetaData().set("partnerId" + partyType.getDisplayName(), partnerId);
149             }
150         } else {
151             LOGGER.error("No '{}' party found in ebXML 2.0 message!", partyType.getDisplayName());
152             throw new SOAPException("No '" + partyType.getDisplayName() + "' party found in ebXML 2.0 message!");
153         }
154     }

156     private void extractCpaId(SOAPElement soapElement, ExecutionContext executionContext) {

158         executionContext.setChoreographyId(choreographyId);

174     private void extractMessageData(SOAPElement soapElement, Message message) {
175         Iterator<Node> innerElementIterator = soapElement.getChildElements();
176         while (innerElementIterator.hasNext()) {
177             Node node = getNextNonTextNode(innerElementIterator);
178             if (node instanceof SOAPElement) {
179                 soapElement = (SOAPElement) node;
180 
181                 switch (soapElement.getElementName().getLocalName()) {
182                     case "MessageId" -> message.setId(soapElement.getValue());
183                     case "Timestamp" -> extractTimestamp(soapElement, message);
184                     case "RefToMessageId" -> extractReferencedMessage(soapElement, message);
185                     default -> LOGGER.debug("skip element {}", soapElement.getElementName());
186                 }
187             }
188         }
189     }
190 
191     private void extractTimestamp(SOAPElement soapElement, Message message) {
192         ZonedDateTime createdDate;
193         try {
194             createdDate = EbxmlTimestampFormatter.getInstance().parseTimestamp(soapElement.getValue());
195         } catch (DateTimeParseException e) {
196             if (parameters.value(PARAM_IGNORE_MISSING_TIMEZONE)) {
197                 LOGGER.debug("Timestamp in ebxml header could not be parsed. Trying to parse as UTC instead.");
198                 createdDate = EbxmlTimestampFormatter.getInstance().parseTimestampWithoutZone(soapElement.getValue());
199             } else {
200                 throw e;
201             }
202         }
203         message.setOriginalCreated(createdDate);
204     }

207         message.setReferenceMessageId(soapElement.getValue());
208     }

210     private void extractAcknowledgment(SOAPElement soapAcknowledgmentHeader, Message message) {
211         message.setType(MessageType.ACKNOWLEDGMENT);
212 
213         Iterator<Node> acknowledgmentElementIterator = soapAcknowledgmentHeader.getChildElements();
214         if (acknowledgmentElementIterator.hasNext()) {
215             Node node = getNextNonTextNode(acknowledgmentElementIterator);
216             if (node instanceof SOAPElement soapElement) {
217                 String localName = soapElement.getElementName().getLocalName();
218                 if (EbxmlConstants.REFERENCE_TO_MESSAGE_ID.equalsIgnoreCase(localName)) {
219                     extractReferencedMessage(soapElement, message);
220                 }
221             }
222         }
223     }

225     private void extractErrorList(
226             SOAPElement soapErrorListElement, ExecutionContext executionContext, Message message) {
227         message.setType(MessageType.ERROR);
228 
229         soapErrorListElement.getChildElements().forEachRemaining(node -> {
230             if (node instanceof SOAPElement) {
231                 NodeList descriptionNodes = node.getChildNodes();
232                 for (int i = 0; i < descriptionNodes.getLength(); i++) {
233                     LOGGER.error(
234                             "RefId: {} - {}",
235                             executionContext.getSourceMessage().getId(),
236                             descriptionNodes.item(i).getFirstChild());
237                 }
238             }
239         });
240     }
241 
242     private Node getNextNonTextNode(Iterator<Node> iterator) {
243         Node node = iterator.next();
244         while (node instanceof Text && iterator.hasNext()) {
245             node = iterator.next();
246         }
247         return node;
248     }

250     private SOAPMessage getSoapMessage(PayloadContent payloadContent, Charset encoding)
251             throws SOAPException, IOException {
252 
253         SOAPMessage soapMessage;
254         try {
255             soapMessage = MessageFactory.newInstance().createMessage(null, payloadContent.getInputStream());
256         } catch (SOAPException | IOException e) {
257             LOGGER.debug("Got SoapException ({}), trying to fix soap header", e.getMessage());
258             String headerDataAsString = new String(payloadContent.getByteArray(), encoding);
259             String indexToken = ":Envelope";
260 
261             StringBuilder headerDataBuilder = new StringBuilder();
262             headerDataBuilder.append(headerDataAsString);
263             headerDataBuilder.insert(
264                     headerDataAsString.indexOf(indexToken) + indexToken.length(),
265                     String.format(" xmlns:xlink=\"%s\"", EbxmlConstants.XLINK_NAMESPACE));
266 
267             soapMessage = MessageFactory.newInstance()
268                     .createMessage(
269                             null,
270                             new ByteArrayInputStream(
271                                     headerDataBuilder.toString().getBytes(encoding)));
272         }
273         return soapMessage;
274     }

278         return List.of(
279                 ParameterDefinition.builder()
280                         .name(EbxmlConstants.CPA_ID_SCHEME.getName())
281                         .displayName("CPA id scheme")
282                         .description(
283                                 "scheme of cpaId used in inbound ebXML messages (URI scheme has the pattern uri://SOURCE_PARTNER_ID/TARGET_PARTNER_ID/CHOREOGRAPHY_ID)")
284                         .required(false)
285                         .defaultValue(CpaIdScheme.DEFAULT_SCHEME.name())
286                         .type(ParameterType.DROPDOWN_LIST)
287                         .option(CpaIdScheme.CHOREOGRAPHY_ID.name())
288                         .option(CpaIdScheme.URI.name())
289                         .build(),
290                 ParameterDefinition.builder()
291                         .name(PARAM_IGNORE_MISSING_TIMEZONE)
292                         .displayName("Ignore missing Timezone")
293                         .description(
294                                 "Assume 'Timestamp' in ebxml header is UTC, if no timezone information is provided")
295                         .type(ParameterType.BOOLEAN)
296                         .defaultValue(Boolean.FALSE)
297                         .build());
298     }

300     private void resolveParameters(Parameters parameters) {
301         if (parameters != null) {
302             if (parameters.containsKey(EbxmlConstants.CPA_ID_SCHEME.getName())) {
303                 Object cpaIdSchemeValue = parameters.value(EbxmlConstants.CPA_ID_SCHEME.getName());
304                 if (cpaIdSchemeValue instanceof String) {
305                     cpaIdScheme = CpaIdScheme.valueOf(cpaIdSchemeValue.toString());
306                 } else {
307                     cpaIdScheme = (CpaIdScheme) cpaIdSchemeValue;
308                 }
309                 LOGGER.trace("using {} as cpaId scheme", cpaIdScheme.name());
310             }
311         }
