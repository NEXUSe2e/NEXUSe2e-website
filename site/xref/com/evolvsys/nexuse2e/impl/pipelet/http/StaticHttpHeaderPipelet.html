16  /**
17   * Adds manual http headers to the processing metadata to be used by Http-Outflow-Services
18   *
19   * @author SvenBarden
20   * @since 10.04.2025
21   */
22  @Slf4j
23  public class StaticHttpHeaderPipelet extends Pipelet {
24      private static final String PARAM_HEADERS = "headers";
25      private static final String PARAM_OVERWRITE = "overwrite";
26  
27      @Override
28      public List<ParameterDefinition> parameterDefinitions() {
29          return List.of(
30                  ParameterDefinition.builder()
31                          .name(PARAM_HEADERS)
32                          .displayName("Header values")
33                          .description(
34                                  "Header key value pair to add as http headers. The value can also contain placeholders for key-value services.")
35                          .type(ParameterType.KEY_VALUE_PAIR_LIST_TRIMMED)
36                          .required(true)
37                          .build(),
38                  ParameterDefinition.builder()
39                          .name(PARAM_OVERWRITE)
40                          .displayName("Overwrite")
41                          .description(
42                                  "If true, any existing header value for the same key will be overwritten, otherwise the new value will just be added to the list.")
43                          .type(ParameterType.BOOLEAN)
44                          .defaultValue(Boolean.TRUE)
45                          .build());
46      }
47  
48      @Override
49      public ExecutionContext execute(ExecutionContext executionContext) throws NexusException {
50  
51          List<KeyValue<String, String>> headers = parameters.value(PARAM_HEADERS);
52  
53          if (headers != null) {
54              MetaData metaData = executionContext.getMessage().getMetaData();
55              HttpHeaders httpHeaders = metaData.getFirstOrNull(HttpConstants.ADDITIONAL_HEADERS);
56              if (httpHeaders == null) {
57                  metaData.set(HttpConstants.ADDITIONAL_HEADERS, httpHeaders = new HttpHeaders());
58              }
59  
60              boolean overwrite = parameters.value(PARAM_OVERWRITE);
61  
62              for (KeyValue<String, String> header : headers) {
63                  String key = header.getKey();
64                  String value = KeyValueLookup.lookup(header.getValue());
65  
66                  if (overwrite) {
67                      LOGGER.trace("Setting header {} with value {}", key, value);
68                      httpHeaders.set(key, value);
69                  } else {
70                      LOGGER.trace("Adding header {} with value {}", key, value);
71                      httpHeaders.add(key, value);
72                  }
73              }
74          }
75  
76          return executionContext;
77      }
78  }
