24  @Slf4j
25  public class MessageUnpackagerPipelet extends Pipelet {
26      private static final Pattern CONTENT_ID_PATTERN = Pattern.compile("<(.+)>");
27  
28      @Override
29      public ExecutionContext execute(ExecutionContext executionContext) throws NexusException {
30          if (CollectionUtils.isEmpty(executionContext.getSourceData())) {
31              throw new NexusException(
32                      NexusError.builder().message("No ebxml content provided").build());
33          }
34  
35          String encoding = EbxmlUtils.getEncoding(executionContext).orElse(null);
36          if (encoding == null) {
37              throw new NexusException(NexusError.builder()
38                      .message("Encoding of content was not provided")
39                      .build());
40          }
41          Charset charset = Charset.forName(encoding);
42  
43          MediaType contentType = executionContext.getMetaData().getFirstOrThrows(HttpConstants.CONTENT_TYPE);
44          switch (contentType.getType()) {
45              case "multipart" -> {
46                  if (executionContext.getSourceData().isEmpty()) {
47                      throw new NexusException(NexusError.builder()
48                              .message("MIME message contains no body parts!")
49                              .build());
50                  }
51              }
52              case "text" -> {}
53              default ->
54                  throw new NexusException(NexusError.builder()
55                          .message("MIME message doesn't seem to be of type multipart: " + contentType)
56                          .build());
57          }
58  
59          executionContext.getMessage().getMetaData().set(HttpConstants.ENCODING, encoding);
60  
61          decodeMessageHeader(executionContext.getSourceData().getFirst(), executionContext, charset);
62  
63          List<ContextPayload> messagePayloads = executionContext
64                  .getSourceData()
65                  .subList(1, executionContext.getSourceData().size());
66  
67          decodeMessagePayloads(messagePayloads, executionContext, charset);
68          return executionContext;
69      }
70  
71      private void decodeMessageHeader(
72              ContextPayload contextPayload, ExecutionContext executionContext, Charset encoding) {
73          if (LOGGER.isTraceEnabled()) {
74              LOGGER.trace(
75                      "Message-Header: {}", new String(contextPayload.getContent().getByteArray(), encoding));
76              LOGGER.trace("Message-Header Content Type: {}", contextPayload.getContentType());
77          }
78  
79          executionContext.getMessage().getMetaData().set(EbxmlConstants.HEADER_DATA, contextPayload);
80      }
81  
82      private void decodeMessagePayloads(
83              List<ContextPayload> messagePayloads, ExecutionContext executionContext, Charset encoding) {
84          List<ContextPayload> payloads = new ArrayList<>(messagePayloads.size());
85  
86          LOGGER.debug("decode message payloads");
87  
88          for (ContextPayload contextPayload : messagePayloads) {
89              String bodyPartMimeType =
90                      EbxmlUtils.extractMimeType(contextPayload.getContentType()).orElseThrow();
91              LOGGER.trace("BodyPart Mimetype: {}", bodyPartMimeType);
92  
93              PayloadContent payloadContent;
94              if (EbxmlUtils.isBinaryType(bodyPartMimeType)) {
95                  payloadContent = PayloadContent.of(
96                          Base64.getDecoder().decode(contextPayload.getContent().getByteArray()));
97              } else {
98                  payloadContent = contextPayload.getContent();
99              }
100 
101             ContextPayload payload = ContextPayload.builder()
102                     .contentType(bodyPartMimeType)
103                     .size(payloadContent.getSize())
104                     .encoding(encoding.name())
105                     .contentId(getContentId(ObjectUtils.defaultIfNull(contextPayload.getContentId(), "")))
106                     .sequenceNumber(messagePayloads.indexOf(contextPayload) + 1)
107                     .content(payloadContent)
108                     .build();
109             payloads.add(payload);
110         }
111         executionContext.getMessage().setPayloads(payloads);
112         LOGGER.debug("added {} payloads to message", payloads.size());
113     }
114 
115     private String getContentId(String rawContentId) {
116         Matcher matcher = CONTENT_ID_PATTERN.matcher(rawContentId);
117         if (matcher.matches()) {
118             return matcher.group(1);
119         }
120         return rawContentId;
121     }
122 
123     @Override
124     public List<ParameterDefinition> parameterDefinitions() {
125         return List.of();
126     }
