26  @Slf4j
27  public class MessagePackagerPipelet extends Pipelet {
28      private boolean trimContent;
29      private boolean contentDisposition;
30      private BinaryEncoding contentEncoding = BinaryEncoding.BASE64;
31  
32      @Override
33      @SuppressFBWarnings("CFS_CONFUSING_FUNCTION_SEMANTICS")
34      public ExecutionContext execute(ExecutionContext executionContext) {
35          resolveParameters(parameters);
36  
37          executionContext.clearData();
38          addContextPayloads(executionContext);
39  
40          if (LOGGER.isTraceEnabled()) {
41              LOGGER.trace("packed message parts: ");
42              executionContext
43                      .getDataAsList()
44                      .forEach(contextPayload -> LOGGER.trace(
45                              contextPayload.getContent().getString(Charset.forName(contextPayload.getEncoding()))));
46          }
47  
48          return executionContext;
49      }
50  
51      public void addContextPayloads(ExecutionContext executionContext) {
52          List<ContextPayload> payloads = executionContext.getMessage().getPayloads();
53          if (CollectionUtils.isNotEmpty(payloads)) {
54              for (ContextPayload payload : payloads) {
55                  ContextPayload contextPayload = new ContextPayload();
56                  HttpHeaders additionalHeaders = new HttpHeaders();
57  
58                  String contentId = ObjectUtils.defaultIfNull(payload.getContentId(), "");
59                  if (trimContent && StringUtils.isNotBlank(contentId)) {
60                      payload.setContentId(contentId = contentId.replaceAll("__body_\\d+", ""));
61                  }
62                  additionalHeaders.set(HttpConstants.CONTENT_ID.getName(), String.format("<%s>", contentId));
63  
64                  if (EbxmlUtils.isBinaryType(payload.getContentType())) {
65                      switch (contentEncoding) {
66                          case BINARY -> contextPayload.setContent(payload.getContent());
67                          case UNSUPPORTED -> {
68                              LOGGER.warn("BinaryEncoding.UNSUPPORTED found as target encoding. Will use base64 instead");
69                              addBase64EncodedContent(contextPayload, payload.getContent(), contentId, additionalHeaders);
70                          }
71                          default ->
72                              addBase64EncodedContent(contextPayload, payload.getContent(), contentId, additionalHeaders);
73                      }
74                  } else {
75                      contextPayload.setContent(payload.getContent());
76                  }
77  
78                  contextPayload.setSize(contextPayload.getContent().getSize());
79                  contextPayload.setContentType(payload.getContentType());
80                  contextPayload.setEncoding(payload.getEncoding());
81                  contextPayload.getMetaData().set(HttpConstants.ADDITIONAL_HEADERS, additionalHeaders);
82  
83                  executionContext.addData(contextPayload);
84              }
85          }
86      }
87  
88      private void addBase64EncodedContent(
89              ContextPayload payload, PayloadContent content, String contentId, HttpHeaders additionalHeaders) {
90          payload.setContent(PayloadContent.of(Base64.getEncoder().encode(content.getByteArray())));
91  
92          if (contentDisposition && StringUtils.isNotBlank(contentId)) {
93              additionalHeaders.setContentDisposition(
94                      ContentDisposition.attachment().filename(contentId).build());
95              additionalHeaders.set(
96                      "Content-Transfer-Encoding", contentEncoding.toString().toLowerCase());
97          }
98      }

100     @Override
101     public List<ParameterDefinition> parameterDefinitions() {
102         List<ParameterDefinition> parameterDefinitions = new ArrayList<>();
103         parameterDefinitions.add(ParameterDefinition.builder()
104                 .name(HttpConstants.ENCODING.getName())
105                 .displayName("Binary Encoding")
106                 .description("Which encoding to use for binary content")
107                 .required(false)
108                 .defaultValue(BinaryEncoding.BASE64.name())
109                 .type(ParameterType.DROPDOWN_LIST)
110                 .option(BinaryEncoding.BASE64.name())
111                 .option(BinaryEncoding.BINARY.name())
112                 .build());
113         parameterDefinitions.add(ParameterDefinition.builder()
114                 .name(HttpMessageConstants.USE_CONTENT_DISPOSITION.getName())
115                 .displayName("Use content disposition")
116                 .description(
117                         "Adds the content disposition mime header to the message (only for Base64 encoded binary type content)")
118                 .required(false)
119                 .defaultValue(Boolean.FALSE)
120                 .type(ParameterType.BOOLEAN)
121                 .build());
122         parameterDefinitions.add(ParameterDefinition.builder()
123                 .name(HttpMessageConstants.TRIM_CONTENT.getName())
124                 .displayName("Trim content")
125                 .description("Trims the content id in the mime header")
126                 .required(false)
127                 .defaultValue(Boolean.FALSE)
128                 .type(ParameterType.BOOLEAN)
129                 .build());
130         return parameterDefinitions;
131     }
132 
133     private void resolveParameters(Parameters parameters) {
134         if (parameters != null) {
135             if (parameters.containsKey(HttpConstants.ENCODING.getName())) {
136                 Object temporaryEncodingValue = parameters.value(HttpConstants.ENCODING.getName());
137                 BinaryEncoding temporaryEncoding;
138                 if (temporaryEncodingValue instanceof String) {
139                     temporaryEncoding = BinaryEncoding.of(temporaryEncodingValue.toString());
140                 } else {
141                     temporaryEncoding = (BinaryEncoding) temporaryEncodingValue;
142                 }
143 
144                 if (temporaryEncoding != BinaryEncoding.UNSUPPORTED) {
145                     contentEncoding = temporaryEncoding;
146                 }
147             }
148 
149             trimContent = parameters.value(HttpMessageConstants.TRIM_CONTENT.getName());
150             contentDisposition = parameters.value(HttpMessageConstants.USE_CONTENT_DISPOSITION.getName());
151         }
152     }
