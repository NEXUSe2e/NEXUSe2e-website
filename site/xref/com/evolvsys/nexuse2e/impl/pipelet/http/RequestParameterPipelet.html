15      static final String PARAM_CHOREOGRAPHY_ID_ENABLED = "choreography_id_enabled";
16      static final String PARAM_ACTION_ID_ENABLED = "action_id_enabled";
17      static final String PARAM_SOURCE_PARTNER_ID_ENABLED = "source_partner_id_enabled";

25      static final String PARAM_TARGET_PARTNER_ID = "target_partner_id";

36      @Override
37      public ExecutionContext execute(ExecutionContext executionContext) throws NexusException {
38  
39          // Choreography & Action
40          if (parameters.value(PARAM_CHOREOGRAPHY_ID_ENABLED)) {
41              setRequestParameter(executionContext, PARAM_CHOREOGRAPHY_ID, executionContext.getChoreographyId());
42          }
43          if (parameters.value(PARAM_ACTION_ID_ENABLED)) {
44              setRequestParameter(executionContext, PARAM_ACTION_ID, executionContext.getActionId());
45          }
46  
47          // Source & Target Partner
48          if (parameters.value(PARAM_SOURCE_PARTNER_ID_ENABLED)) {
49              setRequestParameter(executionContext, PARAM_SOURCE_PARTNER_ID, executionContext.getSourcePartnerId());
50          }
51          if (parameters.value(PARAM_TARGET_PARTNER_ID_ENABLED)) {
52              setRequestParameter(executionContext, PARAM_TARGET_PARTNER_ID, executionContext.getTargetPartnerId());
53          }
54  
55          // Conversation and Message
56          if (parameters.value(PARAM_CONVERSATION_ID_ENABLED)) {
57              setRequestParameter(
58                      executionContext,
59                      PARAM_CONVERSATION_ID,
60                      executionContext.getMessage().getConversationId());
61          }
62          if (parameters.value(PARAM_MESSAGE_ID_ENABLED)) {
63              setRequestParameter(
64                      executionContext,
65                      PARAM_MESSAGE_ID,
66                      executionContext.getMessage().getId());
67          }
68  
69          return executionContext;
70      }
71  
72      private void setRequestParameter(ExecutionContext executionContext, String param, String value) {
73          // parameter prefix is required as the HttpSenderService looks for request parameters in the metadata based on
74          // the prefix
75          executionContext
76                  .getMetaData()
77                  .set(HttpConstants.PREFIX_OUTBOUND_REQUEST_PARAMETER, parameters.value(param, String.class), value);
78      }
