21  @Slf4j
22  public class FileSavePipelet extends Pipelet implements FileSaveAware {
23  
24      public static final String PARAM_STOP_ON_FAIL = "stop_on_error";
25  
26      @Override
27      public List<ParameterDefinition> parameterDefinitions() {
28          return List.of(
29                  getTargetDirectory(),
30                  getTargetFilename(),
31                  getCreateMissingDirectories(),
32                  ParameterDefinition.builder()
33                          .name(PARAM_STOP_ON_FAIL)
34                          .displayName("Stop on Fail")
35                          .description("Whether to cancel processing on failure.")
36                          .required(true)
37                          .type(ParameterType.BOOLEAN)
38                          .defaultValue(Boolean.TRUE)
39                          .build());
40      }
41  
42      @Override
43      public ExecutionContext execute(ExecutionContext executionContext) throws NexusException {
44  
45          LOGGER.trace("running file save sender.");
46          Message message = executionContext.getMessage();
47  
48          if (StringUtils.isBlank(parameters.value(PARAM_TARGET_DIRECTORY))) {
49              throw new IllegalArgumentException("parameter 'Target Directory' must be provided to the sender service!");
50          }
51  
52          String directoryString = parameters.value(PARAM_TARGET_DIRECTORY);
53          LOGGER.trace("replacing directory pattern {}", directoryString);
54          directoryString = PatternReplaceUtils.replaceContextTokens(directoryString, executionContext);
55          directoryString = directoryString.replace('\\', '/');
56  
57          try {
58              LOGGER.trace("saving message payloads to directory {}", directoryString);
59              String filePattern = parameters.value(PARAM_TARGET_FILENAME);
60              persistPayloads(
61                      executionContext, message, directoryString, filePattern, parameters.value(PARAM_CREATE_DIRECTORY));
62  
63          } catch (Exception e) {
64  
65              Boolean stopOnFail = parameters.value(PARAM_STOP_ON_FAIL);
66              if (stopOnFail) {
67  
68                  throw new NexusException(
69                          NexusError.builder()
70                                  .message("Error while writing file to disk")
71                                  .extension("reason", e.getMessage())
72                                  .build(),
73                          e);
74              } else {
75                  LOGGER.warn("Error while writing file to disk, the error is ignored and processing is continued.", e);
76              }
77          }
78          return executionContext;
79      }
80  }
