20  /**
21   * Pipelet to check for duplicated metadata keys and values.
22   *
23   * @author Julian Drees
24   * @since 29.08.2025
25   */
26  @Slf4j
27  public class MetaDataDuplicationCheckPipelet extends Pipelet {
28      static final String PARAM_CHOREOGRAPHY_ID = "choreography_id";
29      static final String PARAM_METADATA_CATEGORY = "metadata_category";
30      static final String PARAM_METADATA_KEYS = "metadata_keys";
31      static final String PARAM_LINK_TYPE = "link_type";
32  
33      @Override
34      public List<ParameterDefinition> parameterDefinitions() {
35          return List.of(
36                  ParameterDefinition.builder()
37                          .name(PARAM_METADATA_CATEGORY)
38                          .displayName("Metadata category")
39                          .description("The category which holds the metadata.")
40                          .type(ParameterType.TRIMMED_STRING)
41                          .build(),
42                  ParameterDefinition.builder()
43                          .name(PARAM_METADATA_KEYS)
44                          .displayName("Metadata keys")
45                          .description(
46                                  "The metadata keys to check for duplicates. If any of the values for those keys already exist, the message will be rejected.")
47                          .type(ParameterType.STRING_LIST)
48                          .required(true)
49                          .build(),
50                  ParameterDefinition.builder()
51                          .name(PARAM_CHOREOGRAPHY_ID)
52                          .displayName("Choreography Id")
53                          .type(ParameterType.TRIMMED_STRING)
54                          .description(
55                                  "Optional id of the choreography to which messages's metadata the metadata from this message should be compared to.")
56                          .build(),
57                  ParameterDefinition.builder()
58                          .name(PARAM_LINK_TYPE)
59                          .displayName("Link type")
60                          .description(
61                                  "Whether this duplicate check should consider single key values as duplicates (or) or as a key combinations (and). Only matters when multiple keys are configured.")
62                          .type(ParameterType.DROPDOWN_LIST)
63                          .options(List.of(MetaDataLinkType.AND.name(), MetaDataLinkType.OR.name()))
64                          .required(true)
65                          .defaultValue(MetaDataLinkType.OR.name())
66                          .build());
67      }
68  
69      @Override
70      public ExecutionContext execute(ExecutionContext executionContext) throws NexusException {
71          String metadataCategory = getParameters().value(PARAM_METADATA_CATEGORY);
72          Collection<String> metadataKeys =
73                  CollectionUtils.emptyIfNull(getParameters().value(PARAM_METADATA_KEYS));
74          if (CollectionUtils.isEmpty(metadataKeys)) {
75              throw new NexusException(
76                      NexusError.builder().message("No metadata keys configured.").build());
77          }
78  
79          String choreographyId = getParameters().value(PARAM_CHOREOGRAPHY_ID);
80          MetaDataLinkType linkType = MetaDataLinkType.valueOf(getParameters().value(PARAM_LINK_TYPE));
81  
82          LOGGER.info(
83                  "Checking for duplicates for metadata keys {} in category {} using link type {}.",
84                  metadataKeys,
85                  metadataCategory,
86                  linkType);
87          if (StringUtils.isNotBlank(choreographyId)) {
88              LOGGER.debug("Duplicate checks only considers message metadata in choreography {}.", choreographyId);
89          }
90  
91          MetaDataService metaDataService = executionContext.getBeanFactory().getBean(MetaDataService.class);
92          Map<String, List<String>> duplicateKeys = metaDataService.getDuplicatesByKey(
93                  executionContext.getMessage().getMetaData(), metadataCategory, metadataKeys, choreographyId, linkType);
94          if (MapUtils.isNotEmpty(duplicateKeys)) {
95              NexusError nexusError = NexusError.builder()
96                      .code("DUPLICATE")
97                      .message("Duplication check failed, the following duplicate keys and values were found.")
98                      .extensions(EntryStream.of(duplicateKeys)
99                              .mapValues(values -> "[" + String.join(", ", values) + "]")
100                             .toMap())
101                     .build();
102             throw new NexusException(nexusError);
103         }
104         return executionContext;
105     }
106 }
