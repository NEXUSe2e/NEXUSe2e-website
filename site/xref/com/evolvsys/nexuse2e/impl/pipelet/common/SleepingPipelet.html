9   import java.time.Duration;
10  import java.util.List;
11  import lombok.extern.slf4j.Slf4j;
12  
13  /**
14   * @author SvenBarden
15   * @since 19.10.2023
16   */
17  @Slf4j
18  public class SleepingPipelet extends Pipelet {
19      private static final String PARAM_SLEEP_DURATION = "sleep_duration";
20  
21      @Override
22      public List<ParameterDefinition> parameterDefinitions() {
23          return List.of(ParameterDefinition.builder()
24                  .name(PARAM_SLEEP_DURATION)
25                  .displayName("Sleep duration")
26                  .description("Duration for which the current thread is paused.")
27                  .type(ParameterType.DURATION)
28                  .defaultValue(Duration.ofSeconds(5))
29                  .build());
30      }
31  
32      @Override
33      @SuppressWarnings("PMD.DoNotUseThreads")
34      public ExecutionContext execute(ExecutionContext executionContext) throws NexusException {
35          try {
36              Duration duration = parameters.value(PARAM_SLEEP_DURATION);
37              LOGGER.debug("waiting in sleep pipelet for {}", duration);
38              Thread.sleep(duration);
39          } catch (InterruptedException e) {
40              throw new NexusException(
41                      NexusError.builder()
42                              .message("Sleep pipelet was interrupted")
43                              .build(),
44                      e);
45          }
46          return executionContext;
47      }
48  }
