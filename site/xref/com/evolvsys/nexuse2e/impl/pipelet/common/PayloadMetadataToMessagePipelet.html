18  /**
19   * Pipelet to copy metadata from the payload into the message.
20   *
21   * @author Julian Drees
22   * @since 06.12.2024
23   */
24  @Slf4j
25  public class PayloadMetadataToMessagePipelet extends Pipelet {
26      public static final String PARAM_COPY_ALL = "copy_all";
27      public static final String PARAM_METADATA_FIELDS = "metadata_fields";
28      public static final String PARAM_CATEGORY = "category";
29  
30      @Override
31      public List<ParameterDefinition> parameterDefinitions() {
32          return List.of(
33                  ParameterDefinition.builder()
34                          .name(PARAM_METADATA_FIELDS)
35                          .displayName("Copy metadata fields")
36                          .description("The name of the metadata fields to copy")
37                          .type(ParameterType.STRING_LIST)
38                          .build(),
39                  ParameterDefinition.builder()
40                          .name(PARAM_CATEGORY)
41                          .displayName("Metadata category")
42                          .description(
43                                  "Metadata category from which the data is to be copied. If blank, all the categories are used.")
44                          .type(ParameterType.TRIMMED_STRING)
45                          .build(),
46                  ParameterDefinition.builder()
47                          .name(PARAM_COPY_ALL)
48                          .displayName("Copy all metadata fields")
49                          .description(
50                                  "Flag to copy all metadata fields for the category. This value has priority over \"Copy metadata fields\".")
51                          .type(ParameterType.BOOLEAN)
52                          .defaultValue(Boolean.FALSE)
53                          .build());
54      }
55  
56      @Override
57      public ExecutionContext execute(ExecutionContext executionContext) throws NexusException {
58          LOGGER.trace("Running PayloadMetadataToMessagePipelet.");
59  
60          boolean copyAll = parameters.value(PARAM_COPY_ALL);
61          String category = parameters.value(PARAM_CATEGORY);
62          List<String> metadataFields = parameters.value(PARAM_METADATA_FIELDS);
63  
64          if (!copyAll && CollectionUtils.isEmpty(metadataFields)) {
65              LOGGER.warn("Pipelet should not copy all metadata fields, but a field isn't configured either.");
66          }
67  
68          Message message = executionContext.getMessage();
69          MetaData targetMetaData = message.getMetaData();
70          for (ContextPayload payload : message.getPayloads()) {
71              LOGGER.trace("Copying metadata from payload {} to message.", payload);
72              MetaData sourceMetaData = payload.getMetaData();
73              Collection<String> categories =
74                      StringUtils.isBlank(category) ? sourceMetaData.categories() : Collections.singleton(category);
75  
76              for (String c : categories) {
77                  Collection<String> keys = copyAll ? sourceMetaData.keys(c) : metadataFields;
78                  keys.forEach(key -> targetMetaData.addAll(c, key, sourceMetaData.get(c, key)));
79              }
80          }
81          return executionContext;
82      }
83  }
