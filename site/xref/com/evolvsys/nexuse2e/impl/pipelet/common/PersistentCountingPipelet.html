11  import org.apache.commons.lang3.ObjectUtils;
12  
13  /**
14   * @author SvenBarden
15   * @since 18.06.2024
16   */
17  @Slf4j
18  public class PersistentCountingPipelet extends Pipelet {
19      private static final String PARAM_CATEGORY = "category";
20      private static final String PARAM_KEY = "key";
21      private static final String PARAM_INITIAL_VALUE = "initial_value";
22      private static final String PARAM_INCREMENT = "increment";
23  
24      @Override
25      public ExecutionContext execute(ExecutionContext executionContext) throws NexusException {
26          PersistentKeyValue persistentKeyValue = executionContext.getPersistentKeyValue();
27          String category = parameters.value(PARAM_CATEGORY);
28          String key = parameters.value(PARAM_KEY);
29          Long increment = ObjectUtils.defaultIfNull(parameters.value(PARAM_INCREMENT), 1L);
30          Long value = persistentKeyValue
31                  .getAsLong(category, key)
32                  .orElseGet(() -> ObjectUtils.defaultIfNull(parameters.value(PARAM_INITIAL_VALUE), 0L));
33          LOGGER.debug("incrementing {} - {} from {} to {}", category, key, value, value + increment);
34          persistentKeyValue.setLong(category, key, value + increment);
35          return executionContext;
36      }
37  
38      @Override
39      public List<ParameterDefinition> parameterDefinitions() {
40          return List.of(
41                  ParameterDefinition.builder()
42                          .name(PARAM_CATEGORY)
43                          .displayName("Category")
44                          .description("The category name to update")
45                          .type(ParameterType.STRING)
46                          .required(true)
47                          .build(),
48                  ParameterDefinition.builder()
49                          .name(PARAM_KEY)
50                          .displayName("Key")
51                          .description("The key to update")
52                          .type(ParameterType.STRING)
53                          .required(true)
54                          .build(),
55                  ParameterDefinition.builder()
56                          .name(PARAM_INITIAL_VALUE)
57                          .displayName("Initial value")
58                          .description(
59                                  "The initial value to set. The increment will be applied to this value on the first execution.")
60                          .type(ParameterType.LONG)
61                          .defaultValue(0L)
62                          .build(),
63                  ParameterDefinition.builder()
64                          .name(PARAM_INCREMENT)
65                          .displayName("Increment")
66                          .description("The increment to increase the value on each execution")
67                          .type(ParameterType.LONG)
68                          .defaultValue(1L)
69                          .build());
70      }
71  }
