29  @Slf4j
30  public class XSDSchemaValidationPipelet extends Pipelet {
31  
32      public static final String PARAM_SCHEMA_FILE_PATH = "schema_file_path";
33      public static final String PARAM_SCHEMA_FILE_CONTENT = "schema_file_content";
34      public static final String PARAM_DISABLE_VALIDATION = "disable_validation";
35  
36      @Override
37      public List<ParameterDefinition> parameterDefinitions() {
38          return List.of(
39                  ParameterDefinition.builder()
40                          .name(PARAM_SCHEMA_FILE_PATH)
41                          .displayName("File Path to XSD Schema")
42                          .description("The full path to the XSD schema file for payload validation.")
43                          .required(false)
44                          .type(ParameterType.TRIMMED_STRING)
45                          .build(),
46                  ParameterDefinition.builder()
47                          .name(PARAM_SCHEMA_FILE_CONTENT)
48                          .displayName("XSD Schema Content")
49                          .description(
50                                  "Holds the text content of the XSD Schema. If configured, its used instead of the file path.")
51                          .required(false)
52                          .type(ParameterType.LARGE_STRING)
53                          .build(),
54                  ParameterDefinition.builder()
55                          .name(PARAM_DISABLE_VALIDATION)
56                          .displayName("Temporary disable validation")
57                          .description("Used to disable validation without loosing the whole configuration.")
58                          .required(true)
59                          .defaultValue(Boolean.FALSE)
60                          .type(ParameterType.BOOLEAN)
61                          .build());
62      }
63  
64      @Override
65      public ExecutionContext execute(ExecutionContext executionContext) throws NexusException {
66  
67          Boolean disableValidation = getParameters().value(PARAM_DISABLE_VALIDATION);
68          if (disableValidation != null && disableValidation) {
69              LOGGER.info("Validation disabled");
70              return executionContext;
71          }
72  
73          SchemaFactory factory = SchemaFactory.newInstance(XMLConstants.W3C_XML_SCHEMA_NS_URI);
74  
75          try (InputStream schemaFile = getSchemaContent()) {
76              factory.setProperty(XMLConstants.ACCESS_EXTERNAL_DTD, "");
77              factory.setProperty(XMLConstants.ACCESS_EXTERNAL_SCHEMA, "");
78              StreamSource sourceSource = new StreamSource(schemaFile);
79              Schema schema = factory.newSchema(sourceSource);
80              Validator validator = schema.newValidator();
81  
82              validator.validate(new StreamSource(executionContext
83                      .getMessage()
84                      .getPayloads()
85                      .getFirst()
86                      .getContent()
87                      .getInputStream()));
88              LOGGER.info("Payload validation successful");
89          } catch (SAXException | IOException e) {
90              throw new NexusException(
91                      NexusError.builder().message("Payload validation failed").build(), e);
92          }
93  
94          return executionContext;
95      }
96  
97      @SuppressFBWarnings("PATH_TRAVERSAL_IN")
98      private InputStream getSchemaContent() throws NexusException, IOException {
99          String schemaFileString = getParameters().value(PARAM_SCHEMA_FILE_CONTENT);
100         if (StringUtils.isNotBlank(schemaFileString)) {
101             return new ByteArrayInputStream(schemaFileString.getBytes(StandardCharsets.UTF_8));
102         } else {
103             String schemaFilePath = getParameters().value(PARAM_SCHEMA_FILE_PATH);
104             if (StringUtils.isNotBlank(schemaFilePath)) {
105                 return FileUtils.openInputStream(new File(schemaFilePath));
106             }
107         }
108         throw new NexusException(NexusError.builder()
109                 .message("no valid schema configuration found")
110                 .build());
111     }
112 }
