3   import com.evolvsys.nexuse2e.core.component.Pipelet;
4   import com.evolvsys.nexuse2e.core.exception.NexusException;
5   import com.evolvsys.nexuse2e.core.message.context.ExecutionContext;
6   import com.evolvsys.nexuse2e.core.parameter.ParameterDefinition;
7   import com.evolvsys.nexuse2e.core.parameter.ParameterType;
8   import java.util.List;
9   import org.apache.commons.collections4.CollectionUtils;
10  import org.apache.commons.collections4.KeyValue;
11  import org.apache.commons.lang3.StringUtils;
12  
13  /**
14   * @author SvenBarden
15   * @since 05.07.2024
16   */
17  public class MetaDataPipelet extends Pipelet {
18      private static final String PARAM_CATEGORY = "category";
19      private static final String PARAM_META_DATAS = "meta_datas";
20  
21      @Override
22      public List<ParameterDefinition> parameterDefinitions() {
23          return List.of(
24                  ParameterDefinition.builder()
25                          .name(PARAM_CATEGORY)
26                          .displayName("Category")
27                          .description("The optional category to put the meta data in")
28                          .type(ParameterType.TRIMMED_STRING)
29                          .build(),
30                  ParameterDefinition.builder()
31                          .name(PARAM_META_DATAS)
32                          .displayName("Meta datas")
33                          .description("List of meta data entries to set in the message")
34                          .type(ParameterType.KEY_VALUE_PAIR_LIST_TRIMMED)
35                          .build());
36      }
37  
38      @Override
39      public ExecutionContext execute(ExecutionContext executionContext) throws NexusException {
40          List<KeyValue<String, String>> metaDatas = parameters.value(PARAM_META_DATAS);
41  
42          if (CollectionUtils.isNotEmpty(metaDatas)) {
43              String category = parameters.value(PARAM_CATEGORY);
44              for (KeyValue<String, String> metaData : metaDatas) {
45                  if (StringUtils.isBlank(category)) {
46                      executionContext.getMessage().getMetaData().add(metaData.getKey(), metaData.getValue());
47                  } else {
48                      executionContext.getMessage().getMetaData().add(category, metaData.getKey(), metaData.getValue());
49                  }
50              }
51          }
52  
53          return executionContext;
54      }
55  }
