1   package com.evolvsys.nexuse2e.impl.pipelet.routing;
2   
3   import com.evolvsys.nexuse2e.core.component.Pipelet;
4   import com.evolvsys.nexuse2e.core.message.context.ExecutionContext;
5   import com.evolvsys.nexuse2e.core.parameter.ParameterDefinition;
6   import com.evolvsys.nexuse2e.core.parameter.ParameterType;
7   import java.util.List;
8   import lombok.extern.slf4j.Slf4j;
9   
10  /**
11   * @author SvenBarden
12   * @since 06.10.2023
13   */
14  @Slf4j
15  public class RoutingPipelet extends Pipelet {
16      public static final String ACTION_ID = "action";
17      public static final String CHOREOGRAPHY_ID = "choreography";
18  
19      @Override
20      public List<ParameterDefinition> parameterDefinitions() {
21          return List.of(
22                  ParameterDefinition.builder()
23                          .name(CHOREOGRAPHY_ID)
24                          .displayName("Choreography ID")
25                          .description("Static choreography id used for every message")
26                          .type(ParameterType.STRING)
27                          .required(true)
28                          .build(),
29                  ParameterDefinition.builder()
30                          .name(ACTION_ID)
31                          .displayName("Action ID")
32                          .description("Static action id used for every message")
33                          .type(ParameterType.STRING)
34                          .required(true)
35                          .build());
36      }
37  
38      @Override
39      public ExecutionContext execute(ExecutionContext executionContext) {
40  
41          executionContext.setChoreographyId(parameters.value(CHOREOGRAPHY_ID));
42  
43          LOGGER.trace("setting choreographyId to {}", executionContext.getChoreographyId());
44  
45          executionContext.setActionId(parameters.value(ACTION_ID));
46  
47          LOGGER.trace("setting actionId to {}", executionContext.getActionId());
48  
49          return executionContext;
50      }
51  }
