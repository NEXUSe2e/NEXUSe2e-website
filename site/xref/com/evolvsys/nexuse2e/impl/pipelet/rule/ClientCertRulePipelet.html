12  import org.apache.commons.collections4.CollectionUtils;

14  /**
15   * @author SvenBarden

17   */
18  @Slf4j
19  public class ClientCertRulePipelet extends Pipelet {
20  
21      @Override
22      public List<ParameterDefinition> parameterDefinitions() {
23          return List.of();
24      }
25  
26      @Override
27      public ExecutionContext execute(ExecutionContext executionContext) throws NexusException {
28          X509Certificate clientCertificate = executionContext.getClientCertificate();
29  
30          if (clientCertificate == null) {
31              throw new NexusException(NexusError.builder()
32                      .message("No client certificate found")
33                      .extension("choreographyId", executionContext.getChoreographyId())
34                      .extension("actionId", executionContext.getActionId())
35                      .extension("sourcePartnerId", executionContext.getSourcePartnerId())
36                      .extension("targetPartnerId", executionContext.getTargetPartnerId())
37                      .extension("messageId", executionContext.getMessage().getId())
38                      .build());
39          } else {
40              List<X509Certificate> certificates = executionContext
41                      .getRoute()
42                      .getSourcePartnerTrustCertificates()
43                      .getNullable();
44              if (CollectionUtils.isEmpty(certificates)) {
45                  throw new NexusException(NexusError.builder()
46                          .message("No certificates found to validate the client certificate")
47                          .extension("choreographyId", executionContext.getChoreographyId())
48                          .extension("actionId", executionContext.getActionId())
49                          .extension("sourcePartnerId", executionContext.getSourcePartnerId())
50                          .extension("targetPartnerId", executionContext.getTargetPartnerId())
51                          .extension("messageId", executionContext.getMessage().getId())
52                          .build());
53              }
54              boolean valid = executionContext
55                      .getBeanFactory()
56                      .getBean(CertificateValidator.class)
57                      .validate(clientCertificate, certificates);
58              if (valid) {
59                  LOGGER.trace("client certificate was successfully validates against {} certs", certificates.size());
60              } else {
61                  throw new NexusException(NexusError.builder()
62                          .message("Client certificate is not valid")
63                          .extension("choreographyId", executionContext.getChoreographyId())
64                          .extension("actionId", executionContext.getActionId())
65                          .extension("sourcePartnerId", executionContext.getSourcePartnerId())
66                          .extension("targetPartnerId", executionContext.getTargetPartnerId())
67                          .extension("messageId", executionContext.getMessage().getId())
68                          .build());
69              }
70          }
71  
72          return executionContext;
73      }
74  }
