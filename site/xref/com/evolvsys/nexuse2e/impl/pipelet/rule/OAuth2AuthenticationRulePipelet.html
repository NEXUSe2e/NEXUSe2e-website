95      @Override
96      public ExecutionContext execute(ExecutionContext executionContext) throws NexusException {
97          String authorization =
98                  executionContext.getMetaData().getFirstOrThrows(HttpConstants.PREFIX_REQUEST_HEADER, "authorization");
99  
100         if (StringUtils.isBlank(authorization)) {
101             throw new NexusException(NexusError.builder()
102                     .message("Authorization header is missing")
103                     .build());
104         }
105 
106         if (authorization.startsWith("Bearer ") == false) {
107             throw new NexusException(NexusError.builder()
108                     .message("Authorization header is not a Bearer token")
109                     .build());
110         }
111 
112         String token = authorization.substring("Bearer ".length());
113 
114         LOGGER.trace("got bearer token: {}", token);
115 
116         NimbusJwtDecoder jwtDecoder = extendJwtDecoderBuilder(
117                         executionContext, NimbusJwtDecoder.withIssuerLocation(parameters.value(PARAM_ISSUER_URL)))
118                 .build();
119         JwtAuthenticationProvider jwtAuthenticationProvider = new JwtAuthenticationProvider(jwtDecoder);
120         JwtAuthenticationConverter jwtAuthenticationConverter = new JwtAuthenticationConverter();
121         jwtAuthenticationConverter.setJwtGrantedAuthoritiesConverter(
122                 new JwtAuthoritiesConverter(parameters.value(PARAM_AUTHORITY_CLAIMS)));
123         jwtAuthenticationProvider.setJwtAuthenticationConverter(jwtAuthenticationConverter);
124         BearerTokenAuthenticationToken bearerTokenAuthenticationToken = new BearerTokenAuthenticationToken(token);
125 
126         try {
127             Authentication authenticate = jwtAuthenticationProvider.authenticate(bearerTokenAuthenticationToken);
128             if (authenticate == null) {
129                 throw new NexusException(NexusError.builder()
130                         .message("Bearer token is invalid, no validation object could be created")
131                         .build());
132             }
133 
134             JwtAuthenticationToken jwtAuthenticationToken = (JwtAuthenticationToken) authenticate;
135 
136             List<String> validAudiences = parameters.value(PARAM_VALID_AUDIENCES);
137             if (CollectionUtils.isEmpty(validAudiences)) {
138                 LOGGER.warn(
139                         "No valid audiences are configured, for security reasons audience validation should be enabled");
140             } else {
141                 if (CollectionUtils.isNotEmpty(jwtAuthenticationToken.getToken().getAudience())
142                         && CollectionUtils.containsAny(
143                                 validAudiences,
144                                 jwtAuthenticationToken.getToken().getAudience())) {
145                     LOGGER.trace("audience validation successful");
146                 } else {
147                     throw new NexusException(NexusError.builder()
148                             .message("Bearer token is invalid, audience validation failed")
149                             .extension("validAudiences", validAudiences)
150                             .extension(
151                                     "audiences",
152                                     jwtAuthenticationToken.getToken().getAudience())
153                             .build());
154                 }
155             }
156 
157             List<KeyValue<String, String>> claimVerifications = parameters.value(PARAM_CLAIM_VERIFICATION);
158             if (CollectionUtils.isNotEmpty(claimVerifications)) {
159                 for (KeyValue<String, String> claimVerification : claimVerifications) {
160                     LOGGER.trace(
161                             "verifying claim: {} for value: {}",
162                             claimVerification.getKey(),
163                             claimVerification.getValue());
164                     Object tokenValue =
165                             jwtAuthenticationToken.getTokenAttributes().get(claimVerification.getKey());
166                     if (Objects.equals(tokenValue, claimVerification.getValue())) {
167                         LOGGER.trace("claim verification successful");
168                     } else {
169                         throw new NexusException(NexusError.builder()
170                                 .message("Bearer token is invalid, claim verification failed")
171                                 .extension("claim", claimVerification.getKey())
172                                 .extension("value", tokenValue)
173                                 .extension("expectedValue", claimVerification.getValue())
174                                 .build());
175                     }
176                 }
177             }
178 
179             List<String> validRoles = parameters.value(PARAM_VALID_ROLES);
180             if (CollectionUtils.isNotEmpty(validRoles)) {
181                 List<String> authorities = authenticate.getAuthorities().stream()
182                         .map(GrantedAuthority::getAuthority)
183                         .toList();
184                 LOGGER.trace("validating roles: {} against: {}", validRoles, authorities);
185                 if (CollectionUtils.containsAll(authorities, validRoles)) {
186                     LOGGER.trace("all roles are valid");
187                 } else {
188                     throw new NexusException(NexusError.builder()
189                             .message("Bearer token is invalid, role verification failed")
190                             .extension("requiredRoles", validRoles)
191                             .extension("roles", authorities)
192                             .build());
193                 }
194             }
195         } catch (AuthenticationException e) {
196             throw new NexusException(
197                     NexusError.builder().message("Bearer token is invalid").build(), e);
198         }
199 
200         return executionContext;
201     }

203     private NimbusJwtDecoder.JwkSetUriJwtDecoderBuilder extendJwtDecoderBuilder(
204             ExecutionContext executionContext, NimbusJwtDecoder.JwkSetUriJwtDecoderBuilder builder)
205             throws NexusException {
206         NexusRestClientFactory restClientFactory =
207                 executionContext.getBeanFactory().getBean(NexusRestClientFactory.class);
208         WebClientContext context = restClientFactory
209                 .createContextBuilder()
210                 .ssl(true)
211                 .disableHostNameVerification(parameters.value(PARAM_DISABLE_HOSTNAME_VERIFICATION))
212                 .build();
213         RestTemplate restTemplate = new RestTemplate();
214         try {
215             restTemplate.setRequestFactory(restClientFactory.createRequestFactory(context));
216             builder.restOperations(restTemplate);
217         } catch (UnrecoverableKeyException
218                 | CertificateException
219                 | NoSuchAlgorithmException
220                 | KeyStoreException
221                 | IOException
222                 | KeyManagementException e) {
223             throw new NexusException(
224                     NexusError.builder()
225                             .message("Could not create request factory for oauth2 validation")
226                             .build(),
227                     e);
228         }
229         return builder;
230     }
