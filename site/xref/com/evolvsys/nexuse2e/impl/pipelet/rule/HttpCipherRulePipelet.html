12  import org.apache.commons.collections4.CollectionUtils;
13  
14  /**
15   * @author SvenBarden
16   * @since 02.07.2024
17   */
18  @Slf4j
19  public class HttpCipherRulePipelet extends Pipelet {
20      private static final String PARAM_CIPHERS = "ciphers";
21  
22      @Override
23      public List<ParameterDefinition> parameterDefinitions() {
24          return List.of(ParameterDefinition.builder()
25                  .name(PARAM_CIPHERS)
26                  .displayName("Valid ciphers")
27                  .description("Valid http ciphers. An empty list counts as if all ciphers are valid.")
28                  .type(ParameterType.STRING_LIST)
29                  .build());
30      }
31  
32      @Override
33      public ExecutionContext execute(ExecutionContext executionContext) throws NexusException {
34          String cipher = executionContext.getMetaData().getFirstOrNull(HttpConstants.CIPHERS);
35  
36          List<String> validCiphers = parameters.value(PARAM_CIPHERS);
37  
38          LOGGER.trace("validating http cipher {}", cipher);
39  
40          if (cipher == null) {
41              throw new NexusException(NexusError.builder()
42                      .message("No http cipher found in message context")
43                      .extension("choreographyId", executionContext.getChoreographyId())
44                      .extension("actionId", executionContext.getActionId())
45                      .extension("sourcePartnerId", executionContext.getSourcePartnerId())
46                      .extension("targetPartnerId", executionContext.getTargetPartnerId())
47                      .extension("messageId", executionContext.getMessage().getId())
48                      .build());
49          } else {
50              if (CollectionUtils.isEmpty(validCiphers)) {
51                  LOGGER.trace("valid cipher list is empty, no validation occurred");
52              } else {
53                  if (validCiphers.stream().anyMatch(string -> string.equalsIgnoreCase(cipher))) {
54                      LOGGER.trace("cipher {} is in valid protocols {}", cipher, validCiphers);
55                  } else {
56                      throw new NexusException(NexusError.builder()
57                              .message("Invalid inbound http cipher")
58                              .extension("cipher", cipher)
59                              .extension("validCiphers", validCiphers)
60                              .extension("choreographyId", executionContext.getChoreographyId())
61                              .extension("actionId", executionContext.getActionId())
62                              .extension("sourcePartnerId", executionContext.getSourcePartnerId())
63                              .extension("targetPartnerId", executionContext.getTargetPartnerId())
64                              .extension(
65                                      "messageId", executionContext.getMessage().getId())
66                              .build());
67                  }
68              }
69          }
70  
71          return executionContext;
72      }
73  }
