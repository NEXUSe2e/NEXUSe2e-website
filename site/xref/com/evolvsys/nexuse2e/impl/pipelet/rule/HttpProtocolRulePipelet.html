12  import org.apache.commons.collections4.CollectionUtils;
13  
14  /**
15   * @author SvenBarden
16   * @since 02.07.2024
17   */
18  @Slf4j
19  public class HttpProtocolRulePipelet extends Pipelet {
20      private static final String PARAM_PROTOCOLS = "protocols";
21  
22      private boolean isValid(String validProtocol, String protocolToCheck) {
23          return protocolToCheck.equalsIgnoreCase(validProtocol)
24                  || protocolToCheck.equalsIgnoreCase(validProtocol + ":TLS"); // external tomcat adds :TLS suffix
25      }
26  
27      @Override
28      public List<ParameterDefinition> parameterDefinitions() {
29          return List.of(ParameterDefinition.builder()
30                  .name(PARAM_PROTOCOLS)
31                  .displayName("Valid protocols")
32                  .description("Valid http protocols. An empty list counts as if all protocols are valid.")
33                  .type(ParameterType.STRING_LIST)
34                  .build());
35      }
36  
37      @Override
38      public ExecutionContext execute(ExecutionContext executionContext) throws NexusException {
39          String protocol = executionContext.getMetaData().getFirstOrNull(HttpConstants.PROTOCOL_VERSION);
40  
41          List<String> validProtocols = parameters.value(PARAM_PROTOCOLS);
42  
43          LOGGER.trace("validating http protocol {}", protocol);
44  
45          if (protocol == null) {
46              throw new NexusException(NexusError.builder()
47                      .message("No http protocol found in message context")
48                      .extension("choreographyId", executionContext.getChoreographyId())
49                      .extension("actionId", executionContext.getActionId())
50                      .extension("sourcePartnerId", executionContext.getSourcePartnerId())
51                      .extension("targetPartnerId", executionContext.getTargetPartnerId())
52                      .extension("messageId", executionContext.getMessage().getId())
53                      .build());
54          } else {
55              if (CollectionUtils.isEmpty(validProtocols)) {
56                  LOGGER.trace("valid protocol list is empty, no validation occurred");
57              } else {
58                  if (validProtocols.stream().anyMatch(string -> isValid(string, protocol))) {
59                      LOGGER.trace("protocol {} is in valid protocols {}", protocol, validProtocols);
60                  } else {
61                      throw new NexusException(NexusError.builder()
62                              .message("Invalid inbound http protocol")
63                              .extension("protocol", protocol)
64                              .extension("validProtocols", validProtocols)
65                              .extension("choreographyId", executionContext.getChoreographyId())
66                              .extension("actionId", executionContext.getActionId())
67                              .extension("sourcePartnerId", executionContext.getSourcePartnerId())
68                              .extension("targetPartnerId", executionContext.getTargetPartnerId())
69                              .extension(
70                                      "messageId", executionContext.getMessage().getId())
71                              .build());
72                  }
73              }
74          }
75  
76          return executionContext;
77      }
78  }
