20  @Slf4j
21  public class HttpIpAddressRulePipelet extends Pipelet {
22      private static final String PARAM_VALID_IPV4_ADDRESSES = "valid_ipv4_addresses";
23      private static final String PARAM_VALID_IPV6_ADDRESSES = "valid_ipv6_addresses";
24  
25      @Override
26      public List<ParameterDefinition> parameterDefinitions() {
27          return List.of(
28                  ParameterDefinition.builder()
29                          .name(PARAM_VALID_IPV4_ADDRESSES)
30                          .displayName("Valid IPv4 Addresses")
31                          .description("List of valid IPv4 Addresses for the inbound message.")
32                          .type(ParameterType.STRING_LIST)
33                          .build(),
34                  ParameterDefinition.builder()
35                          .name(PARAM_VALID_IPV6_ADDRESSES)
36                          .displayName("Valid IPv6 Addresses")
37                          .description("List of valid IPv6 Addresses for the inbound message.")
38                          .type(ParameterType.STRING_LIST)
39                          .build());
40      }
41  
42      @Override
43      public ExecutionContext execute(ExecutionContext executionContext) throws NexusException {
44          String clientIpAddress = executionContext.getMetaData().getFirstOrNull(HttpConstants.CLIENT_IP);
45          String clientForwardedIpAddress =
46                  executionContext.getMetaData().getFirstOrNull(HttpConstants.CLIENT_FORWARDED_IP);
47  
48          List<String> ipAddresses = new ArrayList<>();
49          if (StringUtils.isNotEmpty(clientIpAddress)) {
50              ipAddresses.add(clientIpAddress);
51          }
52          if (StringUtils.isNotEmpty(clientForwardedIpAddress)) {
53              ipAddresses.add(clientForwardedIpAddress);
54          }
55  
56          List<String> validIpv4AddressRanges = parameters.value(PARAM_VALID_IPV4_ADDRESSES);
57          List<String> validIpv6AddressRanges = parameters.value(PARAM_VALID_IPV6_ADDRESSES);
58  
59          for (String ipAddress : ipAddresses) {
60              if (IpUtils.isValid(ipAddress, validIpv4AddressRanges, validIpv6AddressRanges)) {
61                  LOGGER.trace(
62                          "IP Address {} is valid for IPv4 {} and IPv6 {}",
63                          ipAddress,
64                          validIpv4AddressRanges,
65                          validIpv6AddressRanges);
66              } else {
67                  throw new NexusException(NexusError.builder()
68                          .message("Inbound IP Address is not valid")
69                          .extension("ipAddress", ipAddress)
70                          .extension("validIpV4Addresses", validIpv4AddressRanges)
71                          .extension("validIpV6Addresses", validIpv6AddressRanges)
72                          .extension("choreographyId", executionContext.getChoreographyId())
73                          .extension("actionId", executionContext.getActionId())
74                          .extension("sourcePartnerId", executionContext.getSourcePartnerId())
75                          .extension("targetPartnerId", executionContext.getTargetPartnerId())
76                          .extension("messageId", executionContext.getMessage().getId())
77                          .build());
78              }
79          }
80  
81          return executionContext;
82      }
83  }
