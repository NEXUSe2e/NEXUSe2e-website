package com.evolvsys.nexuse2e.core.pipeline.service;

import com.evolvsys.app.core.graphql.annotation.GraphQLService;
import com.evolvsys.app.core.graphql.dataloader.DataLoaderService;
import com.evolvsys.nexuse2e.core.config.service.ConfigGraphQLService;
import com.evolvsys.nexuse2e.core.pipelet.model.PipeletDto;
import com.evolvsys.nexuse2e.core.pipelet.service.PipeletBatchLoader;
import com.evolvsys.nexuse2e.core.pipeline.PipelineFilter;
import com.evolvsys.nexuse2e.core.pipeline.model.PipelineDto;
import io.leangen.graphql.annotations.*;
import io.leangen.graphql.execution.ResolutionEnvironment;
import jakarta.validation.Valid;
import java.util.List;
import java.util.Optional;
import java.util.Set;
import java.util.concurrent.CompletableFuture;
import lombok.RequiredArgsConstructor;
import lombok.extern.slf4j.Slf4j;

/**
 * @author Florian Drees
 * @since 20.11.2023
 */
@GraphQLService
@RequiredArgsConstructor
@Slf4j
public class PipelineGraphQLService {
    private final PipelineService pipelineService;
    private final DataLoaderService dataLoaderService;

    @GraphQLQuery(name = "pipeline_list", description = "List all pipelines")
    @GraphQLNonNull
    public List<@GraphQLNonNull PipelineDto> list(PipelineFilter pipelineFilter) {
        return pipelineService.list(pipelineFilter);
    }

    @GraphQLQuery(name = "pipeline_getById", description = "Gets a pipeline")
    public Optional<PipelineDto> getById(long pipelineId) {
        return pipelineService.getById(pipelineId);
    }

    @GraphQLMutation(name = "pipeline_update", description = "Updates an existing or creates a new pipeline")
    @GraphQLNonNull
    public PipelineDto update(@GraphQLNonNull @Valid PipelineDto pipelineDto) {
        return ConfigGraphQLService.notifySubscribers(pipelineService.save(pipelineDto));
    }

    @GraphQLMutation(name = "pipeline_updateWithPipelets")
    @GraphQLNonNull
    public PipelineDto update(@GraphQLNonNull @Valid PipelineDto pipeline, @Valid List<PipeletDto> pipelets) {
        return ConfigGraphQLService.notifySubscribers(pipelineService.save(pipeline, pipelets));
    }

    @GraphQLMutation(name = "pipeline_delete", description = "Delete a pipeline")
    public boolean delete(long pipelineId) {
        pipelineService.delete(pipelineId);
        ConfigGraphQLService.notifySubscribers();
        return true;
    }

    @GraphQLQuery(description = "list of pipelets of the provided pipeline")
    public CompletableFuture<Set<@GraphQLNonNull PipeletDto>> pipelets(
            @GraphQLContext PipelineDto pipelineDto, @GraphQLEnvironment ResolutionEnvironment resolutionEnvironment) {
        LOGGER.trace("getting pipelets for pipeline {}", pipelineDto);

        return dataLoaderService.loadRelation(resolutionEnvironment, PipeletBatchLoader.class, pipelineDto);
    }

    @GraphQLMutation(name = "pipeline_addPipelet", description = "Adds a pipelet to an existing pipeline")
    @GraphQLNonNull
    public PipelineDto addPipelet(long pipelineId, @Valid @GraphQLNonNull PipeletDto pipeletDto) {
        return ConfigGraphQLService.notifySubscribers(pipelineService.addPipelet(pipelineId, pipeletDto));
    }

    @GraphQLMutation(name = "pipeline_removePipelet", description = "Removes a pipelet from an existing pipeline")
    @GraphQLNonNull
    public PipelineDto removePipelet(long pipelineId, long pipeletId) {
        return ConfigGraphQLService.notifySubscribers(pipelineService.removePipelet(pipelineId, pipeletId));
    }

    @GraphQLMutation(name = "pipeline_rearrange", description = "rearrange the order of pipelets")
    public PipelineDto rearrangePipeline(long pipelineId, List<@GraphQLNonNull Long> pipeletIdsInOrder) {
        return ConfigGraphQLService.notifySubscribers(pipelineService.rearrange(pipelineId, pipeletIdsInOrder));
    }
}
