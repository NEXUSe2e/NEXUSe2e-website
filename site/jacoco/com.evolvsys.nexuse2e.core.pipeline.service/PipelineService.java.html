package com.evolvsys.nexuse2e.core.pipeline.service;

import com.evolvsys.app.core.auth.service.CurrentUserService;
import com.evolvsys.app.core.process.ProcessExecutionContext;
import com.evolvsys.app.core.process.ProcessService;
import com.evolvsys.app.core.service.BaseService;
import com.evolvsys.app.core.service.ListService;
import com.evolvsys.nexuse2e.core.pipelet.model.PipeletDto;
import com.evolvsys.nexuse2e.core.pipeline.PipelineFilter;
import com.evolvsys.nexuse2e.core.pipeline.action.PipelineActionType;
import com.evolvsys.nexuse2e.core.pipeline.action.PipelineAddPipeletAction;
import com.evolvsys.nexuse2e.core.pipeline.action.PipelineRearrangePipeletsAction;
import com.evolvsys.nexuse2e.core.pipeline.action.PipelineRemovePipeletAction;
import com.evolvsys.nexuse2e.core.pipeline.mapper.PipelineMapper;
import com.evolvsys.nexuse2e.core.pipeline.model.PipelineDto;
import com.evolvsys.nexuse2e.core.pipeline.model.PipelineEntity;
import com.evolvsys.nexuse2e.core.pipeline.repository.PipelineRepository;
import java.util.Comparator;
import java.util.List;
import org.apache.commons.lang3.Validate;
import org.springframework.stereotype.Service;
import org.springframework.transaction.annotation.Transactional;

/**
 * @author Florian Drees
 * @since 20.11.2023
 */
@Service
public class PipelineService extends BaseService<PipelineDto, PipelineEntity, PipelineRepository> {
    private final PipelineMapper mapper;
    private final ListService listService;
    private final ProcessService processService;

    public PipelineService(
            PipelineMapper mapper,
            PipelineRepository repository,
            CurrentUserService currentUserService,
            ListService listService,
            ProcessService processService) {
        super(mapper, repository, currentUserService);
        this.mapper = mapper;
        this.listService = listService;
        this.processService = processService;
    }

    @Override
    protected PipelineEntity createEntity() {
        return new PipelineEntity();
    }

    public List<PipelineDto> list(PipelineFilter pipelineFilter) {
        return listService.list(pipelineFilter, PipelineDto.class, repository::findAll).stream()
                .sorted(Comparator.comparing(PipelineEntity::getPipelineId))
                .map(this::mapEntityToDto)
                .toList();
    }

    @Transactional
    public PipelineDto save(PipelineDto pipelineDto, List<PipeletDto> pipeletDtos) {
        PipelineEntity entity = getEntity(pipelineDto);

        mapper.updatePipelets(pipeletDtos, entity, currentUserService.getCurrentUserData());

        entity = save(entity);

        return mapEntityToDto(entity);
    }

    public PipelineDto addPipelet(long pipelineId, PipeletDto pipeletDto) {
        Validate.notNull(pipeletDto, "pipelet dto must not be null");
        Validate.notNull(pipeletDto.getComponentId(), "pipelet id must not be null");

        PipelineEntity pipelineEntity = findWithActionCheck(pipelineId, PipelineActionType.ADD_PIPELET.name());
        pipelineEntity = processService.execute(ProcessExecutionContext.<PipelineEntity, PipelineDto>builder()
                .target(pipelineEntity)
                .actionType(PipelineActionType.ADD_PIPELET)
                .parameterValue(PipelineAddPipeletAction.PIPELET.of(pipeletDto))
                .build());

        return mapEntityToDto(pipelineEntity);
    }

    public PipelineDto removePipelet(long pipelineId, long pipeletId) {
        PipelineEntity pipelineEntity = findWithActionCheck(pipelineId, PipelineActionType.REMOVE_PIPELET.name());
        pipelineEntity = processService.execute(ProcessExecutionContext.<PipelineEntity, PipelineDto>builder()
                .target(pipelineEntity)
                .actionType(PipelineActionType.REMOVE_PIPELET)
                .parameterValue(PipelineRemovePipeletAction.PIPELET_ID.of(pipeletId))
                .build());

        return mapEntityToDto(pipelineEntity);
    }

    public PipelineDto rearrange(long pipelineId, List<Long> pipeletIdsInOrder) {
        PipelineEntity pipelineEntity = findWithActionCheck(pipelineId, PipelineActionType.REARRANGE_PIPELETS.name());
        pipelineEntity = processService.execute(ProcessExecutionContext.<PipelineEntity, PipelineDto>builder()
                .target(pipelineEntity)
                .actionType(PipelineActionType.REARRANGE_PIPELETS)
                .parameterValue(PipelineRearrangePipeletsAction.PIPELET_IDS.of(pipeletIdsInOrder))
                .build());

        return mapEntityToDto(pipelineEntity);
    }
}
