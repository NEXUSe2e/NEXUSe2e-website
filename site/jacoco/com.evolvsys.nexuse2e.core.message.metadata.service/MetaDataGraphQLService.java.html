package com.evolvsys.nexuse2e.core.message.metadata.service;

import com.evolvsys.app.core.graphql.annotation.GraphQLService;
import com.evolvsys.app.core.graphql.dataloader.DataLoaderService;
import com.evolvsys.nexuse2e.core.message.model.MetaDataDto;
import com.evolvsys.nexuse2e.core.message.payload.model.PayloadDto;
import com.evolvsys.nexuse2e.core.message.payload.service.PayloadBatchLoader;
import io.leangen.graphql.annotations.GraphQLContext;
import io.leangen.graphql.annotations.GraphQLEnvironment;
import io.leangen.graphql.annotations.GraphQLNonNull;
import io.leangen.graphql.annotations.GraphQLQuery;
import io.leangen.graphql.execution.ResolutionEnvironment;
import java.util.concurrent.CompletableFuture;
import lombok.RequiredArgsConstructor;
import lombok.extern.slf4j.Slf4j;
import org.apache.tika.config.TikaConfig;
import org.apache.tika.mime.MimeTypeException;

/**
 * @author Florian Drees
 * @since 29.02.2024
 */
@GraphQLService
@RequiredArgsConstructor
@Slf4j
public class MetaDataGraphQLService {
    private final DataLoaderService dataLoaderService;
    private final MetaDataService metaDataService;

    @GraphQLQuery
    public CompletableFuture<PayloadDto> binaryValue(
            @GraphQLContext MetaDataDto metaDataDto, @GraphQLEnvironment ResolutionEnvironment resolutionEnvironment) {
        String messageId = metaDataDto.getParentMessageId();

        return dataLoaderService
                .load(resolutionEnvironment, PayloadBatchLoader.class, metaDataDto, MetaDataDto::getPayloadId)
                .thenApply(payloadDto -> updateDownloadName(messageId, metaDataDto, payloadDto));
    }

    @GraphQLQuery(name = "metadata_keyList", description = "List all metadata key values")
    @GraphQLNonNull
    public MetaDataKeyListDto metaDataKeyList() {
        MetaDataKeyListDto metaDataKeyListDto = new MetaDataKeyListDto();
        metaDataKeyListDto.setMetaDataKeys(metaDataService.getMetaDataKeyList());
        return metaDataKeyListDto;
    }

    private static PayloadDto updateDownloadName(String messageId, MetaDataDto metaDataDto, PayloadDto payloadDto) {
        if (payloadDto != null) {
            payloadDto.setDownloadName(getDownloadName(messageId, metaDataDto, payloadDto));
        }
        return payloadDto;
    }

    private static String getDownloadName(String messageId, MetaDataDto metaDataDto, PayloadDto payloadDto) {
        String result;
        if (messageId == null) {
            result = metaDataDto.getName() + "_" + metaDataDto.getIndex();
        } else {
            result = messageId + "_" + metaDataDto.getName() + "_" + metaDataDto.getIndex();
        }
        try {
            result += TikaConfig.getDefaultConfig()
                    .getMimeRepository()
                    .forName(payloadDto.getContentType())
                    .getExtension();
        } catch (MimeTypeException e) {
            LOGGER.error("could not get extension for payload {} in message {}", payloadDto, messageId, e);
        }
        return result;
    }
}
