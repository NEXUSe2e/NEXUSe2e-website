package com.evolvsys.nexuse2e.core.message.metadata.service;

import com.evolvsys.nexuse2e.core.message.MetaData;
import com.evolvsys.nexuse2e.core.message.MetaDataLinkType;
import com.evolvsys.nexuse2e.core.message.repository.MetaDataRepository;
import com.evolvsys.nexuse2e.core.model.entity.MetaDataEntity;
import java.time.format.DateTimeFormatter;
import java.util.Collection;
import java.util.HashMap;
import java.util.List;
import java.util.Map;
import java.util.stream.Collectors;
import lombok.RequiredArgsConstructor;
import org.apache.commons.lang3.StringUtils;
import org.springframework.lang.NonNull;
import org.springframework.lang.Nullable;
import org.springframework.stereotype.Service;

/**
 *
 */
@RequiredArgsConstructor
@Service
public class MetaDataService {
    private final MetaDataRepository metaDataRepository;

    public List<MetaDataKeyDto> getMetaDataKeyList() {
        return metaDataRepository.findDistinctName();
    }

    /**
     * Returns the duplicate values (serialized as string) for the given keys and MetaData.
     *
     * @param metaData     the message metadata which should be checked for duplicates
     * @param category     the metadata category to be checked
     * @param metadataKeys the metadata keys to be checked
     * @param choreography the choreography id to narrow down the results and prevent issues with same metadata keys across choreographies
     * @param linkType     whether a duplicate key is considered a duplicate as single value or as a combination of all other keys
     * @return a map holding a metadata key and the duplicate values, serialized as string
     */
    @NonNull
    public Map<String, List<String>> getDuplicatesByKey(
            @NonNull MetaData metaData,
            @Nullable String category,
            @NonNull Collection<String> metadataKeys,
            @Nullable String choreography,
            @NonNull MetaDataLinkType linkType) {
        category = StringUtils.defaultIfBlank(category, MetaData.DEFAULT_CATEGORY);
        Map<String, List<Object>> metaDataValues = HashMap.newHashMap(metadataKeys.size());
        for (String metadataKey : metadataKeys) {
            List<Object> values = metaData.get(category, metadataKey);
            metaDataValues.put(metadataKey, values);
        }
        return metaDataRepository
                .getDuplicatesByKey(metaDataValues, category, choreography, linkType)
                .entrySet()
                .stream()
                .collect(Collectors.toMap(Map.Entry::getKey, value -> value.getValue().stream()
                        .map(this::valueToString)
                        .toList()));
    }

    @NonNull
    private String valueToString(@NonNull MetaDataEntity metaData) {
        return switch (metaData.getType()) {
            case LONG, INTEGER -> String.valueOf(metaData.getLongValue());
            case DOUBLE -> String.valueOf(metaData.getBigDecimalValue());
            case STRING -> String.valueOf(metaData.getStringValue());
            case BOOLEAN -> String.valueOf(metaData.getBooleanValue());
            case DATETIME -> DateTimeFormatter.ISO_ZONED_DATE_TIME.format(metaData.getDateTimeValue());
            case null, default -> {
                throw new IllegalArgumentException("Unsupported type: " + metaData.getType());
            }
        };
    }
}
