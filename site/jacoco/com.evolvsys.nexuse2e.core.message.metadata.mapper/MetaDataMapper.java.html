package com.evolvsys.nexuse2e.core.message.metadata.mapper;

import com.evolvsys.app.core.mapper.BaseDtoEntityMapper;
import com.evolvsys.app.core.mapper.DefaultMapperConfiguration;
import com.evolvsys.app.core.mapper.interceptor.InterceptingMappingContext;
import com.evolvsys.nexuse2e.core.message.model.MetaDataDto;
import com.evolvsys.nexuse2e.core.message.repository.MetaDataRepository;
import com.evolvsys.nexuse2e.core.model.entity.MetaDataEntity;
import org.mapstruct.Context;
import org.mapstruct.InheritConfiguration;
import org.mapstruct.Mapper;
import org.mapstruct.Mapping;
import org.mapstruct.MappingTarget;
import org.springframework.beans.factory.annotation.Autowired;

/**
 * <p>{@link Mapper} f√ºr {@link MetaDataEntity} und {@link MetaDataDto}.</p>
 *
 * @author Florian Drees
 * @since 27.02.2024
 */
@Mapper(config = DefaultMapperConfiguration.class)
public abstract class MetaDataMapper extends BaseDtoEntityMapper<MetaDataDto, MetaDataEntity> {
    private MetaDataRepository metaDataRepository;

    @Autowired
    protected void config(MetaDataRepository metaDataRepository) {
        this.metaDataRepository = metaDataRepository;
    }

    @Override
    public MetaDataEntity idToEntity(Long id, @Context InterceptingMappingContext interceptingMappingContext) {
        return idToEntity(id, interceptingMappingContext, metaDataRepository::findById);
    }

    @Mapping(target = "parentMessageId", ignore = true)
    @Mapping(target = "payloadId", source = "binaryValue.id")
    @InheritConfiguration
    @Override
    protected abstract MetaDataDto entityToDto(
            MetaDataEntity entity, @Context InterceptingMappingContext interceptingMappingContext);

    @Mapping(target = "binaryValue", ignore = true)
    @Override
    protected abstract void updateEntity(
            MetaDataDto dto,
            @MappingTarget MetaDataEntity entity,
            @Context InterceptingMappingContext interceptingMappingContext);
}
