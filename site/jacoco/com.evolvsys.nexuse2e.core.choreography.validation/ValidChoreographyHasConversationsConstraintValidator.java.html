package com.evolvsys.nexuse2e.core.choreography.validation;

import com.evolvsys.nexuse2e.core.message.repository.ConversationRepository;
import jakarta.validation.ConstraintValidator;
import jakarta.validation.ConstraintValidatorContext;
import lombok.RequiredArgsConstructor;
import org.springframework.stereotype.Component;

/**
 * @author SvenBarden
 * @since 22.04.2024
 */
@Component
@RequiredArgsConstructor
class ValidChoreographyHasConversationsConstraintValidator
        implements ConstraintValidator<ValidChoreographyHasConversations, Long> {
    private final ConversationRepository conversationRepository;

    @Override
    public boolean isValid(Long value, ConstraintValidatorContext context) {
        boolean isValid;
        if (value == null) {
            isValid = true;
        } else {
            isValid = conversationRepository.existsByChoreography_Id(value) == false;
        }
        return isValid;
    }
}
