package com.evolvsys.nexuse2e.core.message.queue;

import com.evolvsys.nexuse2e.core.async.MessageExecutor;
import com.evolvsys.nexuse2e.core.config.NEXUSe2eConfigurationProperties;
import com.evolvsys.nexuse2e.core.message.MessageStatus;
import com.evolvsys.nexuse2e.core.message.repository.MessageRepository;
import com.evolvsys.nexuse2e.core.message.service.MessageSenderService;
import com.evolvsys.nexuse2e.core.model.entity.MessageEntity;
import com.evolvsys.nexuse2e.core.model.entity.QMessageEntity;
import com.evolvsys.nexuse2e.core.utils.ErrorUtils;
import com.evolvsys.nexuse2e.internal.MessageType;
import com.querydsl.core.types.Predicate;
import java.time.ZonedDateTime;
import java.util.List;
import java.util.concurrent.CompletableFuture;
import java.util.concurrent.Executor;
import lombok.RequiredArgsConstructor;
import lombok.extern.slf4j.Slf4j;
import org.springframework.data.querydsl.QSort;
import org.springframework.stereotype.Service;

/**
 * @author SvenBarden
 * @since 05.03.2024
 */
@Service
@RequiredArgsConstructor
@Slf4j
class MessageQueueService {
    private final NEXUSe2eConfigurationProperties configurationProperties;
    private final MessageRepository messageRepository;
    private final MessageSenderService messageSenderService;

    @MessageExecutor
    private final Executor messageExecutor;

    private static Predicate buildPredicate() {
        return QMessageEntity.messageEntity
                .type
                .eq(MessageType.NORMAL)
                .and(QMessageEntity.messageEntity.status.in(
                        MessageStatus.RESOLVED,
                        MessageStatus.ACTION_SUCCESSFUL,
                        MessageStatus.QUEUE_ACTION,
                        MessageStatus.QUEUE_SENDER,
                        MessageStatus.WAITING_FOR_ACKNOWLEDGMENT))
                .and(QMessageEntity.messageEntity.nextRequeue.loe(ZonedDateTime.now()));
    }

    private List<MessageEntity> getMessagesToSend() {
        return messageRepository.findBy(buildPredicate(), fetchableFluentQuery -> fetchableFluentQuery
                .limit(configurationProperties.getMessageQueue().getMessagesPerRun())
                .sortBy(QSort.by(QMessageEntity.messageEntity.nextRequeue.asc()))
                .all());
    }

    private void runMessage(MessageEntity messageEntity) {
        try {
            LOGGER.debug("resending message {}", messageEntity.getMessageId());
            messageSenderService.resend(messageEntity.getMessageId());
        } catch (Throwable e) {
            LOGGER.error(
                    "error while resending message {}: {}", messageEntity.getMessageId(), ErrorUtils.aggregate(e), e);
        }
    }

    private void runMessageAsync(MessageEntity messageEntity) {
        CompletableFuture.runAsync(() -> runMessage(messageEntity), messageExecutor)
                .exceptionally(throwable -> {
                    LOGGER.error(
                            "error while async resending message {}: {}",
                            messageEntity.getMessageId(),
                            ErrorUtils.aggregate(throwable),
                            throwable);
                    return null;
                });
    }

    public void run() {
        List<MessageEntity> messagesToSend = getMessagesToSend();
        LOGGER.trace("found {} messages for resending", messagesToSend.size());

        for (MessageEntity messageEntity : messagesToSend) {
            if (configurationProperties.getMessageQueue().isSingleThread()) {
                runMessage(messageEntity);
            } else {
                runMessageAsync(messageEntity);
            }
        }
    }
}
