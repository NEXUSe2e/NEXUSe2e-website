package com.evolvsys.nexuse2e.core.message.queue;

import com.evolvsys.nexuse2e.core.config.NEXUSe2eConfigurationProperties;
import java.time.ZonedDateTime;
import java.util.Date;
import lombok.RequiredArgsConstructor;
import lombok.extern.slf4j.Slf4j;
import org.quartz.*;
import org.springframework.context.annotation.Bean;
import org.springframework.context.annotation.Configuration;

/**
 * @author SvenBarden
 * @since 05.03.2024
 */
@Configuration
@RequiredArgsConstructor
@Slf4j
class MessageQueueJobConfiguration {
    static final JobKey MESSAGE_QUEUE_JOB_KEY = new JobKey("queue", "message");
    static final TriggerKey MESSAGE_QUEUE_TRIGGER_KEY = new TriggerKey("queue", "message");

    private final NEXUSe2eConfigurationProperties nexuSe2eConfigurationProperties;

    @Bean
    JobDetail messageQueueJob() {
        return JobBuilder.newJob(MessageQueueJob.class)
                .withIdentity(MESSAGE_QUEUE_JOB_KEY)
                .storeDurably()
                .build();
    }

    @Bean
    Trigger messageQueueTrigger() {
        ZonedDateTime startTime = ZonedDateTime.now()
                .plus(nexuSe2eConfigurationProperties.getMessageQueue().getQueueStartupDelay());
        LOGGER.info(
                "create message queue trigger to run every {} seconds, starting at {}",
                nexuSe2eConfigurationProperties
                        .getMessageQueue()
                        .getSchedulingInterval()
                        .toSeconds(),
                startTime);
        return TriggerBuilder.newTrigger()
                .withIdentity(MESSAGE_QUEUE_TRIGGER_KEY)
                .forJob(MESSAGE_QUEUE_JOB_KEY)
                .withSchedule(SimpleScheduleBuilder.repeatSecondlyForever((int) nexuSe2eConfigurationProperties
                                .getMessageQueue()
                                .getSchedulingInterval()
                                .toSeconds())
                        .withMisfireHandlingInstructionNextWithRemainingCount())
                .startAt(Date.from(startTime.toInstant()))
                .build();
    }
}
