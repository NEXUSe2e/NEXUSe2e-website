package com.evolvsys.nexuse2e.core.config.service;

import com.evolvsys.nexuse2e.core.config.handler.ConfigHandler;
import com.evolvsys.nexuse2e.core.config.model.*;
import java.util.Collection;
import java.util.List;
import java.util.Optional;
import java.util.function.BiConsumer;
import lombok.RequiredArgsConstructor;
import lombok.extern.slf4j.Slf4j;
import org.apache.commons.collections4.CollectionUtils;
import org.springframework.stereotype.Service;

/**
 * @author SvenBarden
 * @since 21.11.2024
 */
@Service
@RequiredArgsConstructor
@Slf4j
class ConfigHandlerService {
    private static final List<Class<?>> ROOT_TYPES = List.of(
            PartnerConfig.class,
            PipelineConfig.class,
            ChoreographyConfig.class,
            RouteConfig.class,
            ServiceConfig.class,
            CaCertConfig.class,
            ClientKeyContainerConfig.class,
            UserConfig.class,
            KeyValuePropertyConfig.class);
    private final List<ConfigHandler<?>> configHandlers;

    private Optional<ConfigHandler<?>> findHandler(Class<?> configType) {
        return configHandlers.stream()
                .filter(configHandler -> configHandler.type().equals(configType))
                .findFirst();
    }

    @SuppressWarnings("unchecked")
    private <T> void handle(ConfigContext configContext, T config) {
        findHandler(config.getClass())
                .ifPresent(configHandler -> ((ConfigHandler<T>) configHandler).importConfig(configContext, config));
    }

    @SuppressWarnings("unchecked")
    private <T> void postHandle(ConfigContext configContext, T config) {
        findHandler(config.getClass())
                .ifPresent(configHandler -> ((ConfigHandler<T>) configHandler).postImportConfig(configContext, config));
    }

    private void handle(
            ConfigContext configContext, NexusConfig nexusConfig, BiConsumer<ConfigContext, Object> handleFunction) {
        for (PipelineConfig pipeline : nexusConfig.getPipelines()) {
            handleFunction.accept(configContext, pipeline);
        }
        if (CollectionUtils.isNotEmpty(nexusConfig.getServices())) {
            for (ServiceConfig service : nexusConfig.getServices()) {
                handleFunction.accept(configContext, service);
            }
        }
        for (PartnerConfig partner : nexusConfig.getPartners()) {
            handleFunction.accept(configContext, partner);
        }
        for (CaCertConfig caCertificate : nexusConfig.getCaCertificates()) {
            handleFunction.accept(configContext, caCertificate);
        }
        for (ClientKeyContainerConfig clientKey : nexusConfig.getClientKeyContainers()) {
            handleFunction.accept(configContext, clientKey);
        }
        for (RouteConfig route : nexusConfig.getRoutes()) {
            handleFunction.accept(configContext, route);
        }
        for (ChoreographyConfig choreography : nexusConfig.getChoreographies()) {
            handleFunction.accept(configContext, choreography);
        }
        for (UserConfig user : nexusConfig.getUsers()) {
            handleFunction.accept(configContext, user);
        }
        for (KeyValuePropertyConfig keyValueProperty : nexusConfig.getKeyValueProperties()) {
            handleFunction.accept(configContext, keyValueProperty);
        }
    }

    public void importConfig(ConfigContext configContext, NexusConfig nexusConfig) {
        handle(configContext, nexusConfig, this::handle);
    }

    public void postImportConfig(ConfigContext configContext, NexusConfig nexusConfig) {
        handle(configContext, nexusConfig, this::postHandle);
    }

    @SuppressWarnings("unchecked")
    private <T> List<T> merge(Class<T> type, Collection<T> oldConfigs, Collection<T> newConfigs) {
        return findHandler(type)
                .map(configHandler -> ((ConfigHandler<T>) configHandler).merge(oldConfigs, newConfigs))
                .orElseThrow();
    }

    public NexusConfig export(NexusConfig nexusConfig) {
        ConfigExportContext exportContext =
                ConfigExportContext.builder().config(nexusConfig).build();

        for (Class<?> rootType : ROOT_TYPES) {
            findHandler(rootType).ifPresent(configHandler -> configHandler.exportConfig(exportContext));
        }

        sortConfig(exportContext.getConfig());

        return exportContext.getConfig();
    }

    private void sortConfig(NexusConfig nexusConfig) {
        for (Class<?> rootType : ROOT_TYPES) {
            findHandler(rootType).ifPresent(configHandler -> configHandler.sort(nexusConfig));
        }
    }

    public NexusConfig merge(NexusConfig oldConfig, NexusConfig newConfig) {
        NexusConfig mergedConfig = new NexusConfig();
        mergedConfig.getPartners().addAll(merge(PartnerConfig.class, oldConfig.getPartners(), newConfig.getPartners()));
        mergedConfig
                .getPipelines()
                .addAll(merge(PipelineConfig.class, oldConfig.getPipelines(), newConfig.getPipelines()));
        mergedConfig.getRoutes().addAll(merge(RouteConfig.class, oldConfig.getRoutes(), newConfig.getRoutes()));
        mergedConfig
                .getChoreographies()
                .addAll(merge(ChoreographyConfig.class, oldConfig.getChoreographies(), newConfig.getChoreographies()));
        mergedConfig.getServices().addAll(merge(ServiceConfig.class, oldConfig.getServices(), newConfig.getServices()));
        mergedConfig
                .getCaCertificates()
                .addAll(merge(CaCertConfig.class, oldConfig.getCaCertificates(), newConfig.getCaCertificates()));
        mergedConfig
                .getClientKeyContainers()
                .addAll(merge(
                        ClientKeyContainerConfig.class,
                        oldConfig.getClientKeyContainers(),
                        newConfig.getClientKeyContainers()));
        mergedConfig.getUsers().addAll(merge(UserConfig.class, oldConfig.getUsers(), newConfig.getUsers()));
        mergedConfig
                .getKeyValueProperties()
                .addAll(merge(
                        KeyValuePropertyConfig.class,
                        oldConfig.getKeyValueProperties(),
                        newConfig.getKeyValueProperties()));
        return mergedConfig;
    }

    public void clear() {
        configHandlers.forEach(ConfigHandler::clear);
    }
}
