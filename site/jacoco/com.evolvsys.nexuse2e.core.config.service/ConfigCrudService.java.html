package com.evolvsys.nexuse2e.core.config.service;

import com.evolvsys.common.lock.LockService;
import com.evolvsys.nexuse2e.core.config.model.NexusConfig;
import com.evolvsys.nexuse2e.core.exception.NexusError;
import com.evolvsys.nexuse2e.core.exception.NexusException;
import com.evolvsys.nexuse2e.core.model.entity.ConfigEntity;
import com.fasterxml.jackson.core.JsonProcessingException;
import java.util.concurrent.TimeUnit;
import java.util.concurrent.locks.Lock;
import lombok.RequiredArgsConstructor;
import lombok.extern.slf4j.Slf4j;
import org.springframework.stereotype.Service;

/**
 * @author SvenBarden
 * @since 21.11.2024
 */
@Service
@RequiredArgsConstructor
@Slf4j
class ConfigCrudService {
    private final ConfigRepository configRepository;
    private final ConfigParserService configParserService;
    private final LockService lockService;

    public long count() {
        return configRepository.count();
    }

    public NexusConfig get(String config) throws JsonProcessingException {
        return configParserService.parse(config);
    }

    public void save(NexusConfig nexusConfig) throws NexusException {
        Lock lock = lockService.obtain("nexuse2e-config-update");
        try {
            if (lock.tryLock(2, TimeUnit.MINUTES)) {
                try {
                    configParserService.update(nexusConfig);
                    ConfigEntity configEntity = new ConfigEntity();
                    configEntity.setContent(configParserService.configToString(nexusConfig));
                    configRepository.save(configEntity);
                } finally {
                    lock.unlock();
                }
            } else {
                throw new NexusException(NexusError.builder()
                        .message("Could not acquire lock for config import within 2 minutes")
                        .build());
            }
        } catch (InterruptedException e) {
            throw new NexusException(
                    NexusError.builder()
                            .message("Thread was interrupted while acquiring config import lock")
                            .build(),
                    e);
        } catch (JsonProcessingException e) {
            throw new NexusException(
                    NexusError.builder()
                            .message("Error while processing config")
                            .build(),
                    e);
        }
    }
}
