package com.evolvsys.nexuse2e.core.config.service;

import com.evolvsys.app.core.exception.ObjectNotFoundException;
import com.evolvsys.app.core.file.mapper.FileMapper;
import com.evolvsys.app.core.file.model.FileDto;
import com.evolvsys.app.core.file.model.FileEntity;
import com.evolvsys.app.core.file.repository.FileRepository;
import com.evolvsys.app.core.file.service.FileFactory;
import com.evolvsys.app.core.system.service.SystemConfigurationProperties;
import java.io.ByteArrayOutputStream;
import java.io.InputStream;
import java.sql.SQLException;
import java.time.ZonedDateTime;
import java.time.format.DateTimeFormatter;
import lombok.RequiredArgsConstructor;
import org.apache.commons.lang3.StringUtils;
import org.springframework.stereotype.Component;

/**
 * @author SvenBarden
 * @since 09.09.2024
 */
@Component
@RequiredArgsConstructor
class ConfigImportFileHelper {
    private final FileRepository fileRepository;
    private final FileFactory fileFactory;
    private final FileMapper fileMapper;
    private final SystemConfigurationProperties systemConfigurationProperties;

    public InputStream getContent(long fileId) throws SQLException {
        FileEntity fileEntity = fileRepository
                .getFileEntityById(fileId)
                .orElseThrow(() -> new ObjectNotFoundException(FileDto.class, fileId));
        return fileEntity.getFileContent().getContent().getBinaryStream();
    }

    public FileDto create(ByteArrayOutputStream outputStream) {
        StringBuilder fileName = new StringBuilder("NEXUSe2e_config");
        String environmentName = systemConfigurationProperties.getEnvironmentName();
        if (StringUtils.isNotBlank(environmentName)) {
            fileName.append('_').append(environmentName);
        }
        fileName.append('_')
                .append(DateTimeFormatter.ISO_OFFSET_DATE_TIME.format(
                        ZonedDateTime.now().withNano(0)))
                .append(".xml");

        FileEntity fileEntity = fileFactory.create(
                fileName.toString().replaceAll("[^A-Za-z0-9.\\-_+]", "_"), "text/xml", outputStream.toByteArray());

        return fileMapper.entityToDto(fileEntity);
    }
}
