package com.evolvsys.nexuse2e.core.config.service;

import com.evolvsys.nexuse2e.core.config.model.NexusConfig;
import com.evolvsys.nexuse2e.core.exception.NexusException;
import com.evolvsys.nexuse2e.core.keyvalue.KeyValueService;
import com.evolvsys.nexuse2e.core.manualaction.service.ConfigImportManualActionService;
import com.evolvsys.nexuse2e.core.purge.service.PurgeService;
import lombok.RequiredArgsConstructor;
import org.springframework.stereotype.Component;

/**
 * @author SvenBarden
 * @since 09.09.2024
 */
@Component
@RequiredArgsConstructor
class ConfigImportEventHandler {
    private final PurgeService cleanupService;
    private final ConfigImportManualActionService configImportManualActionService;
    private final KeyValueService keyValueService;

    void preImport(NexusConfig config, boolean overwrite) {
        if (overwrite) {
            cleanupService.purgeAllConversations();
        }
    }

    void postImport(NexusConfig oldConfig, NexusConfig newConfig, boolean overwrite) throws NexusException {
        if (overwrite) {
            configImportManualActionService.overwrite(oldConfig, newConfig);
        } else {
            configImportManualActionService.merge(oldConfig, newConfig);
        }
        keyValueService.update();
    }
}
