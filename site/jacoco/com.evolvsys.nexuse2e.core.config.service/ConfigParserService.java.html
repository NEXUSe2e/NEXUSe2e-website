package com.evolvsys.nexuse2e.core.config.service;

import com.evolvsys.app.core.system.service.SystemConfigurationProperties;
import com.evolvsys.nexuse2e.core.config.model.ConfigContext;
import com.evolvsys.nexuse2e.core.config.model.NexusConfig;
import com.fasterxml.jackson.annotation.JsonInclude;
import com.fasterxml.jackson.core.JsonGenerator;
import com.fasterxml.jackson.core.JsonParser;
import com.fasterxml.jackson.core.JsonProcessingException;
import com.fasterxml.jackson.databind.DeserializationFeature;
import com.fasterxml.jackson.databind.SerializationFeature;
import com.fasterxml.jackson.dataformat.xml.XmlMapper;
import com.fasterxml.jackson.dataformat.xml.ser.ToXmlGenerator;
import com.fasterxml.jackson.module.jakarta.xmlbind.JakartaXmlBindAnnotationModule;
import java.io.IOException;
import java.io.InputStream;
import java.io.OutputStream;
import lombok.RequiredArgsConstructor;
import lombok.extern.slf4j.Slf4j;
import org.springframework.beans.factory.BeanFactory;
import org.springframework.beans.factory.BeanFactoryAware;
import org.springframework.beans.factory.InitializingBean;
import org.springframework.beans.factory.config.AutowireCapableBeanFactory;
import org.springframework.context.annotation.DependsOn;
import org.springframework.http.converter.json.SpringHandlerInstantiator;
import org.springframework.stereotype.Service;
import org.springframework.transaction.annotation.Transactional;

/**
 * @author SvenBarden
 * @since 26.10.2023
 */
@Service
@RequiredArgsConstructor
@Slf4j
@DependsOn("parameterConverterRegistrar")
public class ConfigParserService implements InitializingBean, BeanFactoryAware {
    private final XmlMapper xmlMapper = new XmlMapper();
    private final ConfigHandlerService configHandlerService;
    private final SystemConfigurationProperties systemConfigurationProperties;
    private AutowireCapableBeanFactory beanFactory;

    // @Transactional
    public void clear() {
        configHandlerService.clear();
    }

    public NexusConfig merge(NexusConfig oldConfig, NexusConfig newConfig) {
        NexusConfig mergedConfig = configHandlerService.merge(oldConfig, newConfig);
        mergedConfig.setVersion(systemConfigurationProperties.getVersion());
        return mergedConfig;
    }

    @Transactional(readOnly = true)
    public NexusConfig exportConfig() {
        NexusConfig nexusConfig = new NexusConfig();
        nexusConfig.setVersion(systemConfigurationProperties.getVersion());
        return configHandlerService.export(nexusConfig);
    }

    public void print(OutputStream outputStream, NexusConfig nexusConfig) throws IOException {
        xmlMapper.writeValue(outputStream, nexusConfig);
    }

    public String configToString(NexusConfig nexusConfig) throws JsonProcessingException {
        return xmlMapper.writeValueAsString(nexusConfig);
    }

    public void update(NexusConfig nexusConfig) {
        ConfigContext configContext =
                ConfigContext.builder().config(nexusConfig).build();
        configHandlerService.importConfig(configContext, nexusConfig);
        configHandlerService.postImportConfig(configContext, nexusConfig);
    }

    public NexusConfig parse(String content) throws JsonProcessingException {
        LOGGER.trace("parsing config from string");
        return xmlMapper.readValue(content, NexusConfig.class);
    }

    public NexusConfig parse(InputStream inputStream) throws IOException {
        LOGGER.trace("parsing config from stream");
        return xmlMapper.readValue(inputStream, NexusConfig.class);
    }

    @Override
    public void afterPropertiesSet() throws Exception {
        xmlMapper.setHandlerInstantiator(new SpringHandlerInstantiator(beanFactory));
        xmlMapper.enable(JsonGenerator.Feature.IGNORE_UNKNOWN);
        xmlMapper.enable(JsonParser.Feature.IGNORE_UNDEFINED);
        xmlMapper.disable(DeserializationFeature.FAIL_ON_UNKNOWN_PROPERTIES);
        xmlMapper.getDeserializationContext().checkUnresolvedObjectId();
        xmlMapper.enable(SerializationFeature.INDENT_OUTPUT);
        xmlMapper.registerModule(new JakartaXmlBindAnnotationModule());
        xmlMapper.setSerializationInclusion(JsonInclude.Include.NON_EMPTY);
        xmlMapper.configure(ToXmlGenerator.Feature.WRITE_XML_DECLARATION, true);
    }

    @Override
    public void setBeanFactory(BeanFactory beanFactory) {
        synchronized (this) {
            this.beanFactory = (AutowireCapableBeanFactory) beanFactory;
        }
    }
}
