package com.evolvsys.nexuse2e.core.config.service;

import com.evolvsys.app.core.exception.GeneralErrorException;
import com.evolvsys.app.core.file.model.FileDto;
import com.evolvsys.nexuse2e.core.config.ConfigValidatorService;
import com.evolvsys.nexuse2e.core.config.model.ConfigValidationResult;
import com.evolvsys.nexuse2e.core.config.model.NexusConfig;
import com.evolvsys.nexuse2e.core.config.model.NexusConfigMerge;
import com.evolvsys.nexuse2e.core.exception.NexusError;
import com.evolvsys.nexuse2e.core.exception.NexusException;
import edu.umd.cs.findbugs.annotations.SuppressFBWarnings;
import java.io.ByteArrayOutputStream;
import java.io.IOException;
import java.io.InputStream;
import java.io.OutputStream;
import lombok.RequiredArgsConstructor;
import org.springframework.stereotype.Service;
import org.springframework.transaction.annotation.Transactional;

/**
 * @author SvenBarden
 * @since 27.11.2023
 */
@Service
@RequiredArgsConstructor
public class NEXUSe2eConfigService {
    private static final String EXS_EXCEPTION_SOFTENING_NO_CONSTRAINTS = "EXS_EXCEPTION_SOFTENING_NO_CONSTRAINTS";
    private static final String COULD_NOT_MERGE_CONFIG = "could not merge config";
    private final ConfigParserService configParserService;
    private final ConfigValidatorService configValidatorService;
    private final ConfigCrudService configCrudService;
    private final ConfigImportFileHelper configImportFileHelper;
    private final ConfigImportEventHandler configImportEventHandler;

    public NexusConfig merge(NexusConfig oldConfig, NexusConfig newConfig) {
        return configParserService.merge(oldConfig, newConfig);
    }

    public NexusConfig getCurrentConfig() {
        return configParserService.exportConfig();
    }

    public void print(NexusConfig nexusConfig, OutputStream outputStream) throws IOException {
        configParserService.print(outputStream, nexusConfig);
    }

    public ConfigValidationResult validate(NexusConfig config) {
        return configValidatorService.validate(config);
    }

    @SuppressFBWarnings(EXS_EXCEPTION_SOFTENING_NO_CONSTRAINTS)
    public ConfigValidationResult validate(long fileId) {
        ConfigValidationResult validationResult;
        try (InputStream binaryStream = configImportFileHelper.getContent(fileId)) {
            NexusConfig nexusConfig = configParserService.parse(binaryStream);
            validationResult = validate(nexusConfig);
        } catch (Exception e) {
            throw new GeneralErrorException(COULD_NOT_MERGE_CONFIG, e);
        }
        return validationResult;
    }

    @SuppressWarnings("PMD.PrematureDeclaration") // oldConfig needs to be stored before we update the config
    private NexusConfig importConfig(NexusConfig config, boolean overwrite) throws NexusException {
        ConfigValidationResult validationResult = configValidatorService.validate(config);
        if (validationResult.getFatalCount() > 0) {
            throw new NexusException(NexusError.builder()
                    .message("Config contains fatal errors and cannot be applied.")
                    .extension("messages", String.join(", ", validationResult.getMessages()))
                    .build());
        }
        NexusConfig oldConfig = getCurrentConfig();
        configImportEventHandler.preImport(config, overwrite);
        if (overwrite) {
            configParserService.clear();
        }
        try {
            configCrudService.save(config);
        } catch (NexusException e) {
            throw new NexusException(
                    NexusError.builder().message(COULD_NOT_MERGE_CONFIG).build(), e);
        }
        NexusConfig currentConfig = getCurrentConfig();
        configImportEventHandler.postImport(oldConfig, currentConfig, overwrite);
        return currentConfig;
    }

    public NexusConfig merge(NexusConfig config) throws NexusException {
        return importConfig(config, false);
    }

    public NexusConfig overwrite(NexusConfig config) throws NexusException {
        return importConfig(config, true);
    }

    public NexusConfig overwrite(long fileId) throws NexusException {
        return overwrite(toConfig(fileId));
    }

    public NexusConfigMerge merge(long fileId) {
        NexusConfig currentConfig = getCurrentConfig();
        NexusConfig nexusConfig = toConfig(fileId);
        NexusConfig merge = merge(currentConfig, nexusConfig);

        return NexusConfigMerge.builder()
                .merged(merge)
                .upload(nexusConfig)
                .current(getCurrentConfig())
                .mergedValidationResult(validate(merge))
                .build();
    }

    @SuppressFBWarnings(EXS_EXCEPTION_SOFTENING_NO_CONSTRAINTS)
    public NexusConfig toConfig(long fileId) {
        NexusConfig nexusConfig;
        try (InputStream binaryStream = configImportFileHelper.getContent(fileId)) {
            nexusConfig = configParserService.parse(binaryStream);
        } catch (Exception e) {
            throw new GeneralErrorException(COULD_NOT_MERGE_CONFIG, e);
        }
        return nexusConfig;
    }

    @Transactional(rollbackFor = IOException.class)
    public FileDto currentConfigToFile() throws IOException {
        ByteArrayOutputStream outputStream = new ByteArrayOutputStream();
        configParserService.print(outputStream, getCurrentConfig());

        return configImportFileHelper.create(outputStream);
    }
}
