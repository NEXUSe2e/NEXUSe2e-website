package com.evolvsys.nexuse2e.core.component.creation;

import com.evolvsys.nexuse2e.core.component.BeanFactoryAware;
import lombok.RequiredArgsConstructor;
import org.springframework.beans.factory.BeanFactory;
import org.springframework.stereotype.Component;

/**
 * @author SvenBarden
 * @since 17.06.2024
 */
@Component
@RequiredArgsConstructor
class BeanFactoryPostCreationProcessor
        implements ComponentPostCreationProcessor<BeanFactoryAware>,
                org.springframework.beans.factory.BeanFactoryAware {
    private BeanFactory beanFactory;

    @Override
    public void execute(BeanFactoryAware component) {
        component.setBeanFactory(beanFactory);
    }

    @Override
    public void setBeanFactory(BeanFactory beanFactory) {
        synchronized (this) {
            this.beanFactory = beanFactory;
        }
    }
}
