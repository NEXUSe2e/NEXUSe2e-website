package com.evolvsys.nexuse2e.core.component.creation;

import com.evolvsys.app.core.util.GenericUtils;

/**
 * <p>Process component after creation.</p>
 *
 * @author Florian Drees
 * @since 30.01.2024
 */
@SuppressWarnings("PMD.ImplicitFunctionalInterface")
public interface ComponentPostCreationProcessor<ProcessingInterfaceT> {
    void execute(ProcessingInterfaceT component);

    /**
     * <p>Returns the interface which has post processing to be executed.</p>
     *
     * @return interface to be post-creation processed.
     */
    default Class<ProcessingInterfaceT> getProcessingInterfaceType() {
        return GenericUtils.getGenericType(getClass(), ComponentPostCreationProcessor.class, 0);
    }
}
