package com.evolvsys.nexuse2e.core.component.creation;

import com.evolvsys.nexuse2e.core.schedule.SchedulingService;
import com.evolvsys.nexuse2e.core.schedule.SchedulingServiceAware;
import lombok.RequiredArgsConstructor;
import lombok.extern.slf4j.Slf4j;
import org.springframework.stereotype.Component;

/**
 * <p>Process component with implementation of {@link SchedulingServiceAware} after creation.</p>
 *
 * @author Florian Drees
 * @since 30.01.2024
 * @see SchedulingServiceAware
 */
@Component
@RequiredArgsConstructor
@Slf4j
public class SchedulingServicePostCreationProcessor implements ComponentPostCreationProcessor<SchedulingServiceAware> {
    private final SchedulingService schedulingService;

    @Override
    public void execute(SchedulingServiceAware component) {
        component.setSchedulingService(schedulingService);
    }
}
