package com.evolvsys.nexuse2e.core.changelog.service;

import com.evolvsys.nexuse2e.core.changelog.model.ChangelogDto;
import com.evolvsys.nexuse2e.core.changelog.model.SingleChangelogDto;
import com.evolvsys.nexuse2e.core.exception.NexusException;
import com.evolvsys.nexuse2e.core.utils.ExtendedResource;
import com.evolvsys.nexuse2e.core.utils.ResourceUtils;
import java.io.IOException;
import java.nio.charset.StandardCharsets;
import java.time.LocalDate;
import lombok.extern.slf4j.Slf4j;
import org.apache.commons.io.IOUtils;
import org.springframework.core.io.ClassPathResource;
import org.springframework.stereotype.Component;

/**
 * @author SvenBarden
 * @since 15.04.2024
 */
@Component
@Slf4j
public class ChangelogRepository {
    private static final String BASE_PATH = "/nexuse2e/changelogs/";

    public ChangelogDto get() throws IOException, NexusException {
        ChangelogDto.ChangelogDtoBuilder builder = ChangelogDto.builder()
                .content(IOUtils.resourceToString("/nexuse2e/CHANGELOG.md", StandardCharsets.UTF_8));

        for (ExtendedResource changeLogDirResource :
                ResourceUtils.list("classpath:/nexuse2e/changelogs/**/version-summary.md")) {
            addChangelogFromFiles(changeLogDirResource, builder);
        }
        return builder.build();
    }

    private static void addChangelogFromFiles(ExtendedResource versionSummary, ChangelogDto.ChangelogDtoBuilder builder)
            throws IOException {
        String version = ResourceUtils.getParent(versionSummary.getFullPath());
        LOGGER.trace("adding changelog for version {}", version);
        ClassPathResource releaseDate = new ClassPathResource(BASE_PATH + version + "/release-date.txt");
        if (versionSummary.exists() && releaseDate.exists()) {
            builder.changelog(
                    version,
                    SingleChangelogDto.builder()
                            .version(version)
                            .content(IOUtils.toString(versionSummary.getInputStream(), StandardCharsets.UTF_8))
                            .releaseDate(LocalDate.parse(
                                    IOUtils.toString(releaseDate.getInputStream(), StandardCharsets.UTF_8)))
                            .build());
        }
    }
}
