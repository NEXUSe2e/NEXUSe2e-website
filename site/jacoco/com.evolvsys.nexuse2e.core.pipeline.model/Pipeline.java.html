package com.evolvsys.nexuse2e.core.pipeline.model;

import com.evolvsys.nexuse2e.core.component.Pipelet;
import com.evolvsys.nexuse2e.core.exception.NexusException;
import com.evolvsys.nexuse2e.core.message.context.ExecutionContext;
import java.util.List;
import lombok.Builder;
import lombok.Singular;
import lombok.Value;
import lombok.extern.slf4j.Slf4j;

/**
 * @author SvenBarden
 * @since 06.10.2023
 */
@Value
@Builder
@Slf4j
public class Pipeline {
    String pipelineId;

    @Singular
    List<Pipelet> pipelets;

    public ExecutionContext execute(ExecutionContext executionContext) throws NexusException {
        for (Pipelet pipelet : pipelets) {
            LOGGER.trace("Running pipelet {}", pipelet);
            executionContext = pipelet.execute(executionContext);
        }
        return executionContext;
    }
}
