package com.evolvsys.nexuse2e.core.database;

import com.evolvsys.nexuse2e.core.exception.NexusException;
import com.evolvsys.nexuse2e.core.message.context.ExecutionContext;
import com.evolvsys.nexuse2e.impl.service.outbound.DatabaseSenderService;
import org.springframework.jdbc.core.namedparam.NamedParameterJdbcTemplate;
import org.springframework.lang.NonNull;

/**
 * <p>Abstract class for dynamic database values, such as incremented primary key id's or timestamp columns.</p>
 * <p>Value is determined on execution of {@link DatabaseSenderService}.</p>
 *
 * @author Florian Drees
 * @since 20.06.2024
 */
public abstract class DynamicDatabaseValue<DbValue> {
    public abstract DbValue getValue(
            @NonNull NamedParameterJdbcTemplate namedParameterJdbcTemplate, @NonNull ExecutionContext executionContext)
            throws NexusException;
}
