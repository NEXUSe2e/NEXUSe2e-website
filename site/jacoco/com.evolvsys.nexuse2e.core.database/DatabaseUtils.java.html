package com.evolvsys.nexuse2e.core.database;

import com.evolvsys.nexuse2e.core.exception.NexusError;
import com.evolvsys.nexuse2e.core.exception.NexusException;
import java.sql.Connection;
import java.sql.SQLException;
import java.sql.Statement;
import java.util.ArrayList;
import java.util.Collection;
import java.util.List;
import java.util.Optional;
import javax.sql.DataSource;
import liquibase.util.Validate;
import lombok.experimental.UtilityClass;
import org.springframework.jdbc.core.namedparam.NamedParameterJdbcTemplate;
import org.springframework.jdbc.support.JdbcAccessor;
import org.springframework.lang.NonNull;

/**
 * @author Florian Drees
 * @since 01.07.2024
 */
@UtilityClass
public class DatabaseUtils {
    public static List<String> enquoteIdentifier(Statement statement, Collection<String> identifiers)
            throws NexusException {
        List<String> quotedIdentifiers = new ArrayList<>(identifiers.size());
        for (String identifier : identifiers) {
            quotedIdentifiers.add(enquoteIdentifier(statement, identifier));
        }
        return quotedIdentifiers;
    }

    public static String enquoteIdentifier(@NonNull Statement statement, @NonNull String identifier)
            throws NexusException {
        Validate.notNull(identifier, "identifier must not be null");
        try {
            return statement.enquoteIdentifier(identifier.toUpperCase(), true);
        } catch (SQLException e) {
            throw new NexusException(
                    NexusError.builder()
                            .message("Could not enquote identifier")
                            .extension("identifier", identifier)
                            .build(),
                    e);
        }
    }

    private static List<String> toNamedParameters(Collection<String> columnNames) {
        return columnNames.stream().map(columnName -> ":" + columnName).toList();
    }

    public static String buildNamedSqlInsert(
            NamedParameterJdbcTemplate namedParameterJdbcTemplate, String tableName, Collection<String> columnNames)
            throws NexusException {
        DataSource dataSource = Optional.ofNullable(namedParameterJdbcTemplate)
                .map(NamedParameterJdbcTemplate::getJdbcTemplate)
                .map(JdbcAccessor::getDataSource)
                .orElseThrow(() -> new NexusException(NexusError.builder()
                        .message("Could not get data source from named parameter jdbc template")
                        .build()));
        return buildNamedSqlInsert(dataSource, tableName, columnNames);
    }

    public static String buildNamedSqlInsert(DataSource dataSource, String tableName, Collection<String> columnNames)
            throws NexusException {
        try (Connection connection = dataSource.getConnection();
                Statement statement = connection.createStatement()) {
            return String.format(
                    "INSERT INTO %s (%s) VALUES (%s)",
                    enquoteIdentifier(statement, tableName),
                    String.join(", ", enquoteIdentifier(statement, columnNames)),
                    String.join(", ", toNamedParameters(columnNames)));
        } catch (SQLException e) {
            throw new NexusException(
                    NexusError.builder()
                            .message("Could not open database connection")
                            .build(),
                    e);
        }
    }
}
