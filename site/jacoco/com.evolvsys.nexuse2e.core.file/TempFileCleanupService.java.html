package com.evolvsys.nexuse2e.core.file;

import com.evolvsys.nexuse2e.core.config.NEXUSe2eConfigurationProperties;
import java.io.IOException;
import java.nio.file.DirectoryStream;
import java.nio.file.FileVisitResult;
import java.nio.file.Files;
import java.nio.file.Path;
import java.nio.file.SimpleFileVisitor;
import java.nio.file.attribute.BasicFileAttributes;
import java.time.Instant;
import java.util.concurrent.atomic.AtomicLong;
import lombok.RequiredArgsConstructor;
import lombok.extern.slf4j.Slf4j;
import org.apache.commons.io.file.PathUtils;
import org.springframework.beans.factory.InitializingBean;
import org.springframework.scheduling.annotation.Scheduled;
import org.springframework.stereotype.Component;

@Component
@RequiredArgsConstructor
@Slf4j
public class TempFileCleanupService implements InitializingBean {
    public static final String TEMP_FILE_NAME_PREFIX = "NEXUSe2e";

    private final NEXUSe2eConfigurationProperties configurationProperties;

    @Override
    public void afterPropertiesSet() throws Exception {
        cleanTempFiles(PathUtils.getTempDirectory());
    }

    @Scheduled(cron = "#{NEXUSe2eConfigurationProperties.tempFileCleanup.cron}")
    void cleanTempFilesScheduled() {
        cleanTempFiles(PathUtils.getTempDirectory());
    }

    void cleanTempFiles(Path tempDir) {
        Instant cutoff =
                Instant.now().minus(configurationProperties.getTempFileCleanup().getKeepDuration());

        if (!Files.exists(tempDir) || !Files.isDirectory(tempDir)) {
            LOGGER.warn("Temp directory does not exist or is not a directory: {}", tempDir);
            return;
        }

        final AtomicLong deletedCount = new AtomicLong(0);

        try {
            Files.walkFileTree(tempDir, new TempFileCleanupVisitor(tempDir, cutoff, deletedCount));

            LOGGER.info(
                    "Temp file cleanup completed for directory: {}. Total deleted entries: {}",
                    tempDir,
                    deletedCount.get());
        } catch (IOException e) {
            LOGGER.error("Error walking temp directory {}", tempDir, e);
        }
    }

    @RequiredArgsConstructor
    private static class TempFileCleanupVisitor extends SimpleFileVisitor<Path> {
        private final Path tempDir;
        private final Instant cutoff;
        private final AtomicLong deletedCount;

        @Override
        public FileVisitResult visitFile(Path file, BasicFileAttributes attrs) {
            // Skip files that do not match the prefix
            if (shouldSkip(file)) {
                return FileVisitResult.CONTINUE;
            }

            Instant lastModified = attrs.lastModifiedTime().toInstant();
            if (lastModified.isBefore(cutoff)) {
                try {
                    Files.delete(file);
                    deletedCount.incrementAndGet();
                    LOGGER.debug("Deleted file: {}", file);
                } catch (IOException e) {
                    LOGGER.warn("Failed to delete file: {}", file, e);
                }
            }
            return FileVisitResult.CONTINUE;
        }

        @Override
        public FileVisitResult postVisitDirectory(Path dir, IOException exc) {
            // Skip the root temp directory itself and directories that don't match the prefix
            if (dir.equals(tempDir) || shouldSkip(dir)) {
                return FileVisitResult.CONTINUE;
            }

            try (DirectoryStream<Path> entries = Files.newDirectoryStream(dir)) {
                if (!entries.iterator().hasNext()) {
                    Files.delete(dir);
                    deletedCount.incrementAndGet();
                    LOGGER.debug("Deleted empty directory: {}", dir);
                }
            } catch (IOException e) {
                LOGGER.warn("Failed to delete directory: {}", dir, e);
            }
            return FileVisitResult.CONTINUE;
        }

        @Override
        public FileVisitResult visitFileFailed(Path file, IOException exc) {
            // Only log warning for files that match the prefix
            if (!shouldSkip(file)) {
                LOGGER.warn("Failed to visit file: {}", file, exc);
            }
            return FileVisitResult.CONTINUE;
        }

        @Override
        public FileVisitResult preVisitDirectory(Path dir, BasicFileAttributes attrs) {
            LOGGER.trace("Entering directory: {}", dir);
            return FileVisitResult.CONTINUE;
        }

        private boolean shouldSkip(Path path) {
            Path fileName = path != null ? path.getFileName() : null;
            return fileName == null || !fileName.toString().startsWith(TEMP_FILE_NAME_PREFIX);
        }
    }
}
