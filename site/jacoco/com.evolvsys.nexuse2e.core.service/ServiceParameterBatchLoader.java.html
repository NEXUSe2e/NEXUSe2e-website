package com.evolvsys.nexuse2e.core.service;

import com.evolvsys.app.core.graphql.repository.BatchLoaderRepository;
import com.evolvsys.nexuse2e.core.component.ComponentRegistry;
import com.evolvsys.nexuse2e.core.model.entity.ParameterEntity;
import com.evolvsys.nexuse2e.core.model.entity.QServiceEntity;

import com.evolvsys.nexuse2e.core.parameter.ParameterBatchLoaderHelper;
import com.evolvsys.nexuse2e.core.parameter.repository.ParameterBatchLoader;
import org.springframework.stereotype.Component;

/**
 * <p>Batch-Loader zum Laden bestehender {@link ParameterEntity} eines {@link ServiceEntity}.</p>
 *
 * @author Florian Drees
 * @since 10.01.2024
 */
@Component
public class ServiceParameterBatchLoader extends ParameterBatchLoader {

    public ServiceParameterBatchLoader(
            ComponentRegistry componentRegistry,
            BatchLoaderRepository batchLoaderRepository,
            ParameterBatchLoaderHelper parameterBatchLoaderHelper) {
        super(
                componentRegistry,
                batchLoaderRepository,
                parameterBatchLoaderHelper,
                QServiceEntity.serviceEntity.parameters);
    }
}
