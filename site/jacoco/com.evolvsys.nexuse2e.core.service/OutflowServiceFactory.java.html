package com.evolvsys.nexuse2e.core.service;

import com.evolvsys.nexuse2e.core.component.ComponentFactory;
import com.evolvsys.nexuse2e.core.component.OutflowService;
import com.evolvsys.nexuse2e.core.exception.NexusError;
import com.evolvsys.nexuse2e.core.exception.NexusRuntimeException;
import com.evolvsys.nexuse2e.core.model.entity.ParameterEntity;
import com.evolvsys.nexuse2e.core.model.entity.ServiceEntity;
import com.evolvsys.nexuse2e.core.parameter.Parameters;
import com.evolvsys.nexuse2e.core.partner.model.ContextConnection;
import com.evolvsys.nexuse2e.core.partner.repository.PartnerConnectionRepository;
import com.evolvsys.nexuse2e.core.pipeline.PipelineFactory;
import com.evolvsys.nexuse2e.core.service.repository.ServiceRepository;
import jakarta.validation.constraints.NotNull;
import java.util.Set;
import lombok.RequiredArgsConstructor;
import lombok.extern.slf4j.Slf4j;
import org.springframework.stereotype.Component;
import org.springframework.transaction.annotation.Transactional;

/**
 * @author Florian Drees
 * @since 12.02.2024
 */
@Component
@RequiredArgsConstructor
@Slf4j
public class OutflowServiceFactory {
    private final ComponentFactory componentFactory;
    private final ServiceRepository serviceRepository;
    private final PipelineFactory pipelineFactory;
    private final PartnerConnectionRepository partnerConnectionRepository;

    @Transactional(readOnly = true)
    public OutflowService get(@NotNull ContextConnection partnerConnection) {
        return get(partnerConnection.getOutflowService(), partnerConnection);
    }

    @Transactional(readOnly = true)
    public OutflowService get(String name, @NotNull ContextConnection partnerConnection) {
        LOGGER.debug("search for sender service with name {}", name);
        return get(
                serviceRepository
                        .findByServiceIdAndEnabledTrue(name)
                        .orElseThrow(() -> new NexusRuntimeException(NexusError.builder()
                                .message("no active sender service found")
                                .extension("name", name)
                                .build())),
                partnerConnection);
    }

    @Transactional(readOnly = true)
    public OutflowService get(ServiceEntity entity, @NotNull ContextConnection partnerConnection) {
        OutflowService outflowService = componentFactory.create(entity.getComponentId(), entity.getParameters());
        LOGGER.debug(
                "creating service {} with component {} and implementation {}",
                entity.getServiceId(),
                entity.getComponentId(),
                entity.getClass().getName());

        if (outflowService instanceof NexusOutflowService nexusOutflowService) {
            if (entity.getResolverPipeline() != null) {
                nexusOutflowService.setPipeline(pipelineFactory.get(entity.getResolverPipeline()));
            }
            if (entity.getReturnPipeline() != null) {
                nexusOutflowService.setReturnPipeline(pipelineFactory.get(entity.getReturnPipeline()));
            }
            nexusOutflowService.setDefaultParameters(nexusOutflowService.getParameters());
        }

        Parameters connectionParameters = componentFactory.buildParameters(
                outflowService,
                getPartnerConnectionParameters(partnerConnection.getConnectionId(), partnerConnection.getPartnerId()),
                false);
        Parameters serviceParameters = outflowService.getParameters();
        outflowService.setParameters(serviceParameters.merge(connectionParameters));

        return outflowService;
    }

    protected Set<ParameterEntity> getPartnerConnectionParameters(String name, String partnerId) {
        return partnerConnectionRepository
                .findByConnectionIdAndPartner_PartnerId(name, partnerId)
                .orElseThrow(() -> new IllegalArgumentException(
                        "Connection with name '" + name + "' and partner id '" + partnerId + "' could not be found"))
                .getParameters();
    }
}
