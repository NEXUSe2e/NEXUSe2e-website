package com.evolvsys.nexuse2e.core.service;

import com.evolvsys.nexuse2e.core.annotation.NexusComponent;
import com.evolvsys.nexuse2e.core.component.OutflowService;
import com.evolvsys.nexuse2e.core.exception.NexusException;
import com.evolvsys.nexuse2e.core.message.context.ExecutionContext;
import com.evolvsys.nexuse2e.core.message.context.MessageContext;
import com.evolvsys.nexuse2e.core.parameter.Parameters;
import com.evolvsys.nexuse2e.core.pipeline.model.Pipeline;
import lombok.Getter;
import lombok.Setter;

/**
 * @author SvenBarden
 * @since 11.10.2023
 */
@NexusComponent
@Getter
@Setter
public abstract class NexusOutflowService implements OutflowService {
    protected Pipeline returnPipeline;
    protected Pipeline pipeline;
    protected Parameters parameters;
    protected Parameters defaultParameters;
    protected String serviceId;
    protected String componentId;

    @Override
    public MessageContext execute(ExecutionContext executionContext) throws NexusException {
        ExecutionContext localExecutionContext = executionContext;
        if (pipeline != null) {
            localExecutionContext = pipeline.execute(executionContext);
        }
        return executeInternal(localExecutionContext);
    }

    public abstract MessageContext executeInternal(ExecutionContext executionContext) throws NexusException;
}
