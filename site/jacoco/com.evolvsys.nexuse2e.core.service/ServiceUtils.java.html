package com.evolvsys.nexuse2e.core.service;

import com.evolvsys.nexuse2e.core.component.ComponentRegistry;
import com.evolvsys.nexuse2e.core.component.RegisteredNexusComponent;
import lombok.RequiredArgsConstructor;
import lombok.extern.slf4j.Slf4j;
import org.apache.commons.lang3.Validate;
import org.springframework.lang.NonNull;
import org.springframework.stereotype.Component;

/**
 * @author Florian Drees
 * @since 14.02.2024
 */
@Component
@RequiredArgsConstructor
@Slf4j
public class ServiceUtils {
    private final ComponentRegistry componentRegistry;

    public ServiceType getServiceType(@NonNull String componentId) {
        Validate.notNull(componentId, "component id must not be null");
        return componentRegistry
                .getComponent(componentId)
                .map(RegisteredNexusComponent::getComponentType)
                .map((componentType) -> switch (componentType) {
                    case INFLOW_SERVICE -> ServiceType.INFLOW_SERVICE;
                    case OUTFLOW_SERVICE -> ServiceType.OUTFLOW_SERVICE;
                    case SUPPORT_SERVICE -> ServiceType.SUPPORT_SERVICE;
                    default -> throw new IllegalArgumentException("Component type not supported");
                })
                .orElseThrow(() ->
                        new IllegalArgumentException("Component with id " + componentId + " could not be found!"));
    }
}
