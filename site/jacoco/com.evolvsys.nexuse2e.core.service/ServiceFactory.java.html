package com.evolvsys.nexuse2e.core.service;

import com.evolvsys.app.core.exception.ObjectNotFoundException;
import com.evolvsys.nexuse2e.core.component.ComponentFactory;
import com.evolvsys.nexuse2e.core.component.Service;
import com.evolvsys.nexuse2e.core.message.context.ContextFactory;
import com.evolvsys.nexuse2e.core.message.dispatcher.InboundMessageDispatcher;
import com.evolvsys.nexuse2e.core.model.entity.ServiceEntity;
import com.evolvsys.nexuse2e.core.resolver.MessageResolverFactory;
import com.evolvsys.nexuse2e.core.service.model.ServiceDto;
import com.evolvsys.nexuse2e.core.service.repository.ServiceRepository;
import com.evolvsys.nexuse2e.internal.MessageType;
import lombok.RequiredArgsConstructor;
import lombok.extern.slf4j.Slf4j;
import org.springframework.stereotype.Component;
import org.springframework.transaction.annotation.Transactional;

/**
 * @author SvenBarden
 * @since 11.10.2023
 */
@Component
@RequiredArgsConstructor
@Slf4j
public class ServiceFactory {
    private final ComponentFactory componentFactory;
    private final ServiceRepository serviceRepository;
    private final InboundMessageDispatcher inboundMessageDispatcher;
    private final ContextFactory contextFactory;
    private final MessageResolverFactory messageResolverFactory;

    @Transactional(readOnly = true)
    public <T extends Service> T get(Long id) {
        return get(serviceRepository.findById(id).orElseThrow(() -> new ObjectNotFoundException(ServiceDto.class, id)));
    }

    @Transactional(readOnly = true)
    public <T extends Service> T get(String serviceId) {
        return get(serviceRepository
                .findByServiceId(serviceId)
                .orElseThrow(() -> new IllegalArgumentException("Service with id " + serviceId + " not found")));
    }

    public <T extends Service> T get(ServiceEntity entity) {
        T service = componentFactory.create(entity.getComponentId(), entity.getParameters());
        LOGGER.debug(
                "creating service {} with component {} and implementation {}",
                entity.getServiceId(),
                entity.getComponentId(),
                service.getClass().getName());

        service.setServiceId(entity.getServiceId());
        service.setComponentId(entity.getComponentId());
        if (service instanceof BaseInflowNexusService nexusService) {
            nexusService.setMessageResolver(
                    messageResolverFactory.create(entity.getResolverPipeline(), MessageType.NORMAL));
            nexusService.setReturnMessageResolver(
                    messageResolverFactory.create(entity.getReturnPipeline(), MessageType.RETURN));
            if (entity.getErrorPipeline() != null) {
                nexusService.setErrorMessageResolver(
                        messageResolverFactory.create(entity.getErrorPipeline(), MessageType.ERROR));
            }

            nexusService.setMessageDispatcher(inboundMessageDispatcher);
            nexusService.setContextFactory(contextFactory);
        }

        return service;
    }
}
