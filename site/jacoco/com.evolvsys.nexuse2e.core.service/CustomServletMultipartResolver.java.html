package com.evolvsys.nexuse2e.core.service;

import com.evolvsys.nexuse2e.core.service.http.HttpServiceHandler;
import jakarta.servlet.http.HttpServletRequest;
import org.springframework.stereotype.Component;
import org.springframework.web.multipart.support.StandardServletMultipartResolver;

/**
 * @author Florian Drees
 * @since 11.12.2023
 */
@Component
public class CustomServletMultipartResolver extends StandardServletMultipartResolver {
    public static final String MULTIPART_SINGLE_PAYLOAD = "multipartAsSinglePayload";

    @Override
    public boolean isMultipart(HttpServletRequest request) {
        if (request.getServletPath().startsWith(HttpServiceHandler.SERVICE_PREFIX)) {
            request.setAttribute(MULTIPART_SINGLE_PAYLOAD, Boolean.TRUE);
            return false;
        }
        return super.isMultipart(request);
    }
}
