package com.evolvsys.nexuse2e.core.service;

import com.evolvsys.nexuse2e.core.exception.NexusException;
import com.evolvsys.nexuse2e.core.message.context.MessageContext;
import java.util.Collections;
import java.util.List;
import lombok.extern.slf4j.Slf4j;
import org.springframework.lang.Nullable;

/**
 * @author SvenBarden
 * @since 13.10.2023
 */
@Slf4j
public abstract class SingleContextNexusService extends StandardNexusInflowService {

    @Override
    protected final List<MessageContext> buildMessageContexts() throws NexusException {
        MessageContext messageContext = buildMessageContext();
        List<MessageContext> contexts;
        if (messageContext == null) {
            contexts = Collections.emptyList();
        } else {
            contexts = List.of(messageContext);
        }
        return contexts;
    }

    @Nullable
    protected abstract MessageContext buildMessageContext() throws NexusException;
}
