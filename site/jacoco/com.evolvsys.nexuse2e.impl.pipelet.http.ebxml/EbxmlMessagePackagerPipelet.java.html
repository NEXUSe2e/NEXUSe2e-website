package com.evolvsys.nexuse2e.impl.pipelet.http.ebxml;

import com.evolvsys.nexuse2e.core.constants.HttpConstants;
import com.evolvsys.nexuse2e.core.ebxml.EbxmlConstants;
import com.evolvsys.nexuse2e.core.ebxml.EbxmlUtils;
import com.evolvsys.nexuse2e.core.message.context.ExecutionContext;
import com.evolvsys.nexuse2e.impl.pipelet.http.MessagePackagerPipelet;
import com.evolvsys.nexuse2e.internal.ContextPayload;
import com.evolvsys.nexuse2e.internal.Message;
import com.evolvsys.nexuse2e.internal.MessageType;
import java.util.Objects;
import lombok.extern.slf4j.Slf4j;
import org.springframework.http.HttpHeaders;
import org.springframework.http.MediaType;

@Slf4j
public class EbxmlMessagePackagerPipelet extends MessagePackagerPipelet {

    @Override
    public void addContextPayloads(ExecutionContext executionContext) {
        if (Objects.requireNonNull(executionContext.getMessage().getType()) == MessageType.ACKNOWLEDGMENT) {
            addEbxmlHeaderContextPayload(executionContext);
        } else {
            addEbxmlHeaderContextPayload(executionContext);
            super.addContextPayloads(executionContext);
        }

        Message message = executionContext.getMessage();

        message.getMetaData()
                .set(HttpConstants.CONTENT_TYPE, EbxmlUtils.getMultipartRelatedContentType(executionContext));

        HttpHeaders additionalHeaders =
                message.getMetaData().getFirstOrDefault(HttpConstants.ADDITIONAL_HEADERS, new HttpHeaders());
        additionalHeaders.set("SOAPAction", "\"ebXML\"");
        message.getMetaData().set(HttpConstants.ADDITIONAL_HEADERS, additionalHeaders);
    }

    public void addEbxmlHeaderContextPayload(ExecutionContext executionContext) {
        ContextPayload contextPayload = new ContextPayload();
        ContextPayload payload =
                executionContext.getMessage().getMetaData().getFirstOrThrows(EbxmlConstants.HEADER_DATA);
        contextPayload.setContent(payload.getContent());
        contextPayload.setSize(contextPayload.getContent().getSize());
        contextPayload.setContentType(MediaType.TEXT_XML.toString());
        contextPayload.setEncoding(payload.getEncoding());

        HttpHeaders httpHeaders = new HttpHeaders();
        httpHeaders.set(HttpConstants.CONTENT_ID.getName(), getSoapId(executionContext));
        contextPayload.getMetaData().set(HttpConstants.ADDITIONAL_HEADERS, httpHeaders);

        executionContext.setData(contextPayload);
    }

    private String getSoapId(ExecutionContext executionContext) {
        return getSoapId(executionContext.getMessage().getId());
    }

    private String getSoapId(String messageId) {
        return String.format("<%s%s-Header>", messageId, EbxmlConstants.PROTOCOL);
    }
}
