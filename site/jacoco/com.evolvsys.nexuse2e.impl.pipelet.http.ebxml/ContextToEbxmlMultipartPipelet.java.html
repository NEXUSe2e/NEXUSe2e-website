package com.evolvsys.nexuse2e.impl.pipelet.http.ebxml;

import com.evolvsys.nexuse2e.core.component.Pipelet;
import com.evolvsys.nexuse2e.core.constants.HttpConstants;
import com.evolvsys.nexuse2e.core.ebxml.EbxmlUtils;
import com.evolvsys.nexuse2e.core.exception.NexusException;
import com.evolvsys.nexuse2e.core.message.context.ExecutionContext;
import com.evolvsys.nexuse2e.core.message.payload.content.PayloadContent;
import com.evolvsys.nexuse2e.core.parameter.ParameterDefinition;
import com.evolvsys.nexuse2e.internal.ContextPayload;
import edu.umd.cs.findbugs.annotations.SuppressFBWarnings;
import java.nio.charset.Charset;
import java.util.List;
import lombok.extern.slf4j.Slf4j;
import org.springframework.http.HttpHeaders;
import org.springframework.http.MediaType;

/**
 * <p>Pipelet for creating a multipart message that is to be sent as a single payload.</p>
 *
 * @author Florian Drees
 * @see EbxmlMessagePackagerPipelet
 * @since 11.12.2023
 */
@Slf4j
public class ContextToEbxmlMultipartPipelet extends Pipelet {
    private static final String CRLF = "\r\n";

    @Override
    @SuppressFBWarnings("CFS_CONFUSING_FUNCTION_SEMANTICS")
    public ExecutionContext execute(ExecutionContext executionContext) throws NexusException {
        StringBuilder builder = new StringBuilder(34);

        ContextPayload finalContextPayload = new ContextPayload();
        finalContextPayload.setEncoding(executionContext.getData().getEncoding());
        finalContextPayload.setContentType(
                EbxmlUtils.getMultipartRelatedContentType(executionContext).toString());

        for (ContextPayload contextPayload : executionContext.getDataAsList()) {
            builder.append("--MIME_boundary").append(CRLF);

            HttpHeaders httpHeaders =
                    contextPayload.getMetaData().getFirstOrDefault(HttpConstants.ADDITIONAL_HEADERS, new HttpHeaders());
            httpHeaders.setContentType(MediaType.parseMediaType(contextPayload.getContentType()));
            httpHeaders.forEach((header, values) -> {
                builder.append(header)
                        .append(": ")
                        .append(String.join(", ", values))
                        .append(CRLF);
            });
            builder.append(CRLF)
                    .append(contextPayload.getContent().getString(Charset.forName(contextPayload.getEncoding())))
                    .append(CRLF);
        }
        builder.append("--MIME_boundary--");

        finalContextPayload.setContent(
                PayloadContent.of(builder.toString().getBytes(Charset.forName(finalContextPayload.getEncoding()))));
        finalContextPayload.setSize(finalContextPayload.getContent().getSize());

        executionContext.setData(finalContextPayload);
        return executionContext;
    }

    @Override
    public List<ParameterDefinition> parameterDefinitions() {
        return List.of();
    }
}
