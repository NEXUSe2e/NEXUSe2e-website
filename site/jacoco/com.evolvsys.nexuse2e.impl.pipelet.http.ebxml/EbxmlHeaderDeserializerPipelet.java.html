package com.evolvsys.nexuse2e.impl.pipelet.http.ebxml;

import com.evolvsys.nexuse2e.core.component.Pipelet;
import com.evolvsys.nexuse2e.core.constants.ConnectionConstants;
import com.evolvsys.nexuse2e.core.ebxml.*;
import com.evolvsys.nexuse2e.core.exception.NexusError;
import com.evolvsys.nexuse2e.core.exception.NexusException;
import com.evolvsys.nexuse2e.core.message.context.ExecutionContext;
import com.evolvsys.nexuse2e.core.message.payload.content.PayloadContent;
import com.evolvsys.nexuse2e.core.parameter.ParameterDefinition;
import com.evolvsys.nexuse2e.core.parameter.ParameterType;
import com.evolvsys.nexuse2e.core.parameter.Parameters;
import com.evolvsys.nexuse2e.internal.ContextPayload;
import com.evolvsys.nexuse2e.internal.Message;
import com.evolvsys.nexuse2e.internal.MessageType;
import edu.umd.cs.findbugs.annotations.SuppressFBWarnings;
import jakarta.xml.soap.*;
import java.io.ByteArrayInputStream;
import java.io.IOException;
import java.nio.charset.Charset;
import java.time.ZonedDateTime;
import java.time.format.DateTimeParseException;
import java.util.Iterator;
import java.util.List;
import java.util.Optional;
import lombok.extern.slf4j.Slf4j;
import org.apache.tika.utils.StringUtils;
import org.w3c.dom.NodeList;

@Slf4j
@SuppressWarnings("PMD.GodClass")
public class EbxmlHeaderDeserializerPipelet extends Pipelet {
    static final String PARAM_IGNORE_MISSING_TIMEZONE = "ignore_missing_timezone";
    private static final String MESSAGE_ERROR = "MessageError";
    private SOAPFactory soapFactory;
    private CpaIdScheme cpaIdScheme = CpaIdScheme.DEFAULT_SCHEME;

    @Override
    @SuppressFBWarnings("CFS_CONFUSING_FUNCTION_SEMANTICS")
    public ExecutionContext execute(ExecutionContext executionContext) throws NexusException {
        resolveParameters(parameters);

        try {
            soapFactory = SOAPFactory.newInstance();
            Message message = executionContext.getMessage();

            ContextPayload payload = message.getMetaData().getFirstOrThrows(EbxmlConstants.HEADER_DATA);

            Optional<String> optionalEncoding =
                    Optional.ofNullable(payload.getEncoding()).or(() -> EbxmlUtils.getEncoding(executionContext));
            if (optionalEncoding.isEmpty()) {
                LOGGER.error("Encoding of content (of message with id {}) was not provided", message.getId());
                throw new NexusException(NexusError.builder()
                        .message("Encoding of content was not provided")
                        .build());
            }

            Charset charset = Charset.forName(optionalEncoding.orElseThrow());

            SOAPMessage soapMessage = getSoapMessage(payload.getContent(), charset);
            SOAPHeader soapHeader = soapMessage.getSOAPHeader();

            Iterator<Node> soapHeaderElementIterator = soapHeader.getChildElements();
            while (soapHeaderElementIterator.hasNext()) {
                Node node = soapHeaderElementIterator.next();
                if (node instanceof SOAPElement soapElement) {
                    switch (soapElement.getElementName().getLocalName()) {
                        case "MessageHeader" -> extractMessageHeader(soapElement, executionContext, message);
                        case "Acknowledgment" -> extractAcknowledgment(soapElement, message);
                        case "ErrorList" -> extractErrorList(soapElement, executionContext, message);
                        case "AckRequested" -> extractAckRequested(message);
                        case "SyncReply" -> extractSyncReply(message);
                        default -> LOGGER.debug("skip element {}", soapElement.getElementName());
                    }
                }
            }

            if (MESSAGE_ERROR.equalsIgnoreCase(executionContext.getActionId())) {
                message.setType(MessageType.ERROR);
            }

            executionContext.setMessage(message);
        } catch (SOAPException | IOException e) {
            throw new NexusException(
                    NexusError.builder()
                            .message("could not deserialize ebxml message")
                            .build(),
                    e);
        }
        return executionContext;
    }

    private void extractSyncReply(Message message) {
        LOGGER.debug("inbound message requires sync reply");
        message.getMetaData().set(ConnectionConstants.REQUIRES_REPLY, Boolean.TRUE);
    }

    private void extractAckRequested(Message message) {
        LOGGER.debug("inbound message requires async acknowledgment");
        message.getMetaData().set(ConnectionConstants.REQUIRES_ACK, Boolean.TRUE);
    }

    private void extractMessageHeader(SOAPElement soapMessageHeader, ExecutionContext executionContext, Message message)
            throws SOAPException {
        message.setType(MessageType.NORMAL);
        Iterator<Node> headerElementIterator = soapMessageHeader.getChildElements();

        while (headerElementIterator.hasNext()) {
            Node node = getNextNonTextNode(headerElementIterator);

            if (node instanceof SOAPElement soapElement) {
                switch (soapElement.getElementName().getLocalName()) {
                    case "From" -> extractPartyElement(soapElement, executionContext, PartyType.FROM);
                    case "To" -> extractPartyElement(soapElement, executionContext, PartyType.TO);
                    case "CPAId" -> extractCpaId(soapElement, executionContext);
                    case "ConversationId" -> extractConversationId(soapElement, message);
                    case "Action" -> extractActionId(soapElement, executionContext);
                    case "MessageData" -> extractMessageData(soapElement, message);
                    case "DuplicateElimination" -> LOGGER.trace("duplicateElimination flag found!");
                    case "RefToMessageId" -> extractReferencedMessage(soapElement, message);
                    default -> LOGGER.debug("skip element {}", soapElement.getElementName());
                }
            }
        }
    }

    private void extractPartyElement(SOAPElement soapElement, ExecutionContext executionContext, PartyType partyType)
            throws SOAPException {
        Iterator<Node> innerElementIterator = soapElement.getChildElements();
        if (innerElementIterator.hasNext()) {
            Node node = getNextNonTextNode(innerElementIterator);
            if (node instanceof SOAPElement innerSoapElement) {
                String partnerType = innerSoapElement.getAttributeValue(
                        EbxmlUtils.createNameWithEbxmlNamespace(soapFactory, "type"));
                executionContext.getMessage().getMetaData().set(partyType.getPartnerTypeConstant(), partnerType);

                String partnerId = innerSoapElement.getValue();
                if (StringUtils.isBlank(partnerType) && partnerId.startsWith("uri:")) {
                    // Cut off 'uri:' as there was no type of party provided
                    partnerId = partnerId.substring(partnerId.indexOf(':'));
                }

                if (partyType == PartyType.FROM) {
                    executionContext.setSourcePartnerId(partnerId);
                } else if (partyType == PartyType.TO) {
                    executionContext.setTargetPartnerId(partnerId);
                }
                executionContext.getMessage().getMetaData().set("partnerId" + partyType.getDisplayName(), partnerId);
            }
        } else {
            LOGGER.error("No '{}' party found in ebXML 2.0 message!", partyType.getDisplayName());
            throw new SOAPException("No '" + partyType.getDisplayName() + "' party found in ebXML 2.0 message!");
        }
    }

    private void extractCpaId(SOAPElement soapElement, ExecutionContext executionContext) {
        String choreographyId = cpaIdScheme.getChoreographyIdFromCpaId(soapElement.getValue());
        executionContext.setChoreographyId(choreographyId);
        LOGGER.trace("ChoreographyId: {}", choreographyId);
    }

    private void extractConversationId(SOAPElement soapElement, Message message) {
        String conversationId = soapElement.getValue();
        message.setConversationId(conversationId);
        LOGGER.trace("ConversationId: {}", conversationId);
    }

    private void extractActionId(SOAPElement soapElement, ExecutionContext executionContext) {
        String actionId = soapElement.getValue();
        executionContext.setActionId(actionId);
        LOGGER.trace("ActionId: {}", actionId);
    }

    private void extractMessageData(SOAPElement soapElement, Message message) {
        Iterator<Node> innerElementIterator = soapElement.getChildElements();
        while (innerElementIterator.hasNext()) {
            Node node = getNextNonTextNode(innerElementIterator);
            if (node instanceof SOAPElement) {
                soapElement = (SOAPElement) node;

                switch (soapElement.getElementName().getLocalName()) {
                    case "MessageId" -> message.setId(soapElement.getValue());
                    case "Timestamp" -> extractTimestamp(soapElement, message);
                    case "RefToMessageId" -> extractReferencedMessage(soapElement, message);
                    default -> LOGGER.debug("skip element {}", soapElement.getElementName());
                }
            }
        }
    }

    private void extractTimestamp(SOAPElement soapElement, Message message) {
        ZonedDateTime createdDate;
        try {
            createdDate = EbxmlTimestampFormatter.getInstance().parseTimestamp(soapElement.getValue());
        } catch (DateTimeParseException e) {
            if (parameters.value(PARAM_IGNORE_MISSING_TIMEZONE)) {
                LOGGER.debug("Timestamp in ebxml header could not be parsed. Trying to parse as UTC instead.");
                createdDate = EbxmlTimestampFormatter.getInstance().parseTimestampWithoutZone(soapElement.getValue());
            } else {
                throw e;
            }
        }
        message.setOriginalCreated(createdDate);
    }

    private void extractReferencedMessage(SOAPElement soapElement, Message message) {
        message.setReferenceMessageId(soapElement.getValue());
    }

    private void extractAcknowledgment(SOAPElement soapAcknowledgmentHeader, Message message) {
        message.setType(MessageType.ACKNOWLEDGMENT);

        Iterator<Node> acknowledgmentElementIterator = soapAcknowledgmentHeader.getChildElements();
        if (acknowledgmentElementIterator.hasNext()) {
            Node node = getNextNonTextNode(acknowledgmentElementIterator);
            if (node instanceof SOAPElement soapElement) {
                String localName = soapElement.getElementName().getLocalName();
                if (EbxmlConstants.REFERENCE_TO_MESSAGE_ID.equalsIgnoreCase(localName)) {
                    extractReferencedMessage(soapElement, message);
                }
            }
        }
    }

    private void extractErrorList(
            SOAPElement soapErrorListElement, ExecutionContext executionContext, Message message) {
        message.setType(MessageType.ERROR);

        soapErrorListElement.getChildElements().forEachRemaining(node -> {
            if (node instanceof SOAPElement) {
                NodeList descriptionNodes = node.getChildNodes();
                for (int i = 0; i < descriptionNodes.getLength(); i++) {
                    LOGGER.error(
                            "RefId: {} - {}",
                            executionContext.getSourceMessage().getId(),
                            descriptionNodes.item(i).getFirstChild());
                }
            }
        });
    }

    private Node getNextNonTextNode(Iterator<Node> iterator) {
        Node node = iterator.next();
        while (node instanceof Text && iterator.hasNext()) {
            node = iterator.next();
        }
        return node;
    }

    private SOAPMessage getSoapMessage(PayloadContent payloadContent, Charset encoding)
            throws SOAPException, IOException {

        SOAPMessage soapMessage;
        try {
            soapMessage = MessageFactory.newInstance().createMessage(null, payloadContent.getInputStream());
        } catch (SOAPException | IOException e) {
            LOGGER.debug("Got SoapException ({}), trying to fix soap header", e.getMessage());
            String headerDataAsString = new String(payloadContent.getByteArray(), encoding);
            String indexToken = ":Envelope";

            StringBuilder headerDataBuilder = new StringBuilder();
            headerDataBuilder.append(headerDataAsString);
            headerDataBuilder.insert(
                    headerDataAsString.indexOf(indexToken) + indexToken.length(),
                    String.format(" xmlns:xlink=\"%s\"", EbxmlConstants.XLINK_NAMESPACE));

            soapMessage = MessageFactory.newInstance()
                    .createMessage(
                            null,
                            new ByteArrayInputStream(
                                    headerDataBuilder.toString().getBytes(encoding)));
        }
        return soapMessage;
    }

    @Override
    public List<ParameterDefinition> parameterDefinitions() {
        return List.of(
                ParameterDefinition.builder()
                        .name(EbxmlConstants.CPA_ID_SCHEME.getName())
                        .displayName("CPA id scheme")
                        .description(
                                "scheme of cpaId used in inbound ebXML messages (URI scheme has the pattern uri://SOURCE_PARTNER_ID/TARGET_PARTNER_ID/CHOREOGRAPHY_ID)")
                        .required(false)
                        .defaultValue(CpaIdScheme.DEFAULT_SCHEME.name())
                        .type(ParameterType.DROPDOWN_LIST)
                        .option(CpaIdScheme.CHOREOGRAPHY_ID.name())
                        .option(CpaIdScheme.URI.name())
                        .build(),
                ParameterDefinition.builder()
                        .name(PARAM_IGNORE_MISSING_TIMEZONE)
                        .displayName("Ignore missing Timezone")
                        .description(
                                "Assume 'Timestamp' in ebxml header is UTC, if no timezone information is provided")
                        .type(ParameterType.BOOLEAN)
                        .defaultValue(Boolean.FALSE)
                        .build());
    }

    private void resolveParameters(Parameters parameters) {
        if (parameters != null) {
            if (parameters.containsKey(EbxmlConstants.CPA_ID_SCHEME.getName())) {
                Object cpaIdSchemeValue = parameters.value(EbxmlConstants.CPA_ID_SCHEME.getName());
                if (cpaIdSchemeValue instanceof String) {
                    cpaIdScheme = CpaIdScheme.valueOf(cpaIdSchemeValue.toString());
                } else {
                    cpaIdScheme = (CpaIdScheme) cpaIdSchemeValue;
                }
                LOGGER.trace("using {} as cpaId scheme", cpaIdScheme.name());
            }
        }
    }
}
