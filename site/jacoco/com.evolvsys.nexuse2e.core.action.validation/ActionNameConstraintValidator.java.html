package com.evolvsys.nexuse2e.core.action.validation;

import com.evolvsys.nexuse2e.core.action.model.ActionDto;
import com.evolvsys.nexuse2e.core.action.repository.ActionRepository;
import jakarta.validation.ConstraintValidator;
import jakarta.validation.ConstraintValidatorContext;
import java.util.Optional;
import lombok.RequiredArgsConstructor;
import org.springframework.stereotype.Component;

/**
 * <p>Validation of the name of an action to ensure that the name is unique in the choreography.</p>
 *
 * @author Florian Drees
 * @since 05.01.2024
 */
@Component
@RequiredArgsConstructor
public class ActionNameConstraintValidator implements ConstraintValidator<ValidActionNameInChoreography, ActionDto> {
    private final ActionRepository actionRepository;

    @Override
    public boolean isValid(ActionDto value, ConstraintValidatorContext context) {
        return Optional.ofNullable(value.getId())
                .map(id -> !actionRepository.existsByChoreography_IdAndActionIdAndIdNot(
                        value.getChoreographyId(), value.getActionId(), id))
                .orElseGet(() -> !actionRepository.existsByChoreography_IdAndActionId(
                        value.getChoreographyId(), value.getActionId()));
    }
}
