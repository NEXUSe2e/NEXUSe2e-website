package com.evolvsys.nexuse2e.core.logging.service;

import ch.qos.logback.classic.spi.ILoggingEvent;
import ch.qos.logback.classic.spi.ThrowableProxyUtil;
import com.evolvsys.app.core.auth.service.CurrentUserService;
import com.evolvsys.app.core.service.BaseService;
import com.evolvsys.app.core.service.ListService;
import com.evolvsys.common.mdc.MDCService;
import com.evolvsys.nexuse2e.core.logging.mapper.LogMapper;
import com.evolvsys.nexuse2e.core.logging.model.ForMessageLoggingContext;
import com.evolvsys.nexuse2e.core.logging.model.LogDto;
import com.evolvsys.nexuse2e.core.logging.model.LogEntity;
import com.evolvsys.nexuse2e.core.logging.model.LogFilter;
import com.evolvsys.nexuse2e.core.logging.repository.LogRepository;
import com.evolvsys.nexuse2e.core.message.context.ExecutionContext;
import com.evolvsys.nexuse2e.core.message.context.MessageContext;
import com.evolvsys.nexuse2e.core.settings.service.ApplicationSettingsService;
import com.evolvsys.nexuse2e.internal.Message;
import java.time.ZoneId;
import java.util.ArrayList;
import java.util.List;
import java.util.Optional;
import java.util.concurrent.BlockingQueue;
import java.util.concurrent.LinkedBlockingQueue;
import lombok.extern.slf4j.Slf4j;
import org.apache.commons.collections4.CollectionUtils;
import org.springframework.data.domain.Page;
import org.springframework.data.domain.Pageable;
import org.springframework.lang.Nullable;
import org.springframework.scheduling.annotation.Async;
import org.springframework.stereotype.Service;

/**
 * @author SvenBarden
 * @since 04.01.2024
 */
@Service
@Slf4j
public class LoggingService extends BaseService<LogDto, LogEntity, LogRepository> {
    private final BlockingQueue<LogEntity> logEntityQueue = new LinkedBlockingQueue<>();
    private final ListService listService;
    private final ApplicationSettingsService applicationSettingsService;
    private final MDCService mdcService;

    public LoggingService(
            LogMapper mapper,
            LogRepository repository,
            CurrentUserService currentUserService,
            ListService listService,
            ApplicationSettingsService applicationSettingsService,
            MDCService mdcService) {
        super(mapper, repository, currentUserService);
        this.listService = listService;
        this.applicationSettingsService = applicationSettingsService;
        this.mdcService = mdcService;
    }

    @Async
    public void persistQueue() {
        List<LogEntity> logEntities = new ArrayList<>(logEntityQueue.size());
        logEntityQueue.drainTo(logEntities);
        if (CollectionUtils.isNotEmpty(logEntities)) {
            LOGGER.debug("persisting {} log entities", logEntities.size());
            if (LOGGER.isTraceEnabled()) {
                for (LogEntity logEntity : logEntities) {
                    LOGGER.trace(
                            "persisting log entity with mdcId {} and messageId {}",
                            logEntity.getMdcId(),
                            logEntity.getMessageId());
                }
            }
            try {
                getRepository().saveAll(logEntities);
            } catch (RuntimeException e) {
                logEntityQueue.addAll(logEntities);
                LOGGER.warn("Could not persist logEntities, adding to queue for next run: {}", e.getMessage());
            }
        }
    }

    @Async
    public void updateMessageReference(String messageId, String mdcId) {
        try {
            LOGGER.trace("update message references for message {} and mdc {}", messageId, mdcId);
            getRepository().updateMessageIdByMdcId(messageId, mdcId);
        } catch (RuntimeException e) {
            LOGGER.error("could not update message references for messageId {}", messageId, e);
        }
    }

    @Async
    public void persistAndUpdate(String messageId, String mdcId) {
        // needs to be grouped in a single thread, to catch all logs with the following update query
        persistQueue();
        updateMessageReference(messageId, mdcId);
    }

    public void log(ForMessageLoggingContext forMessageLoggingContext, ILoggingEvent event) {
        LogEntity logEntity = new LogEntity();

        logEntity.setMdcId(mdcService.getId());
        logEntity.setCreated(event.getInstant().atZone(ZoneId.systemDefault()));
        logEntity.setMessage(event.getFormattedMessage());
        logEntity.setLevel(event.getLevel().toString());
        logEntity.setLogger(event.getLoggerName());
        logEntity.setInMessageContext(true);
        logEntity.setMessageId(forMessageLoggingContext.getMessageId());
        logEntity.setConversationId(forMessageLoggingContext.getConversationId());
        logEntity.setActionId(forMessageLoggingContext.getActionId());
        logEntity.setChoreographyId(forMessageLoggingContext.getChoreographyId());

        if (event.getThrowableProxy() != null) {
            logEntity.setStacktrace(ThrowableProxyUtil.asString(event.getThrowableProxy()));
        }

        logEntityQueue.add(logEntity);
    }

    public void log(
            @Nullable MessageContext messageContext, @Nullable ExecutionContext executionContext, ILoggingEvent event) {
        LogEntity logEntity = new LogEntity();
        if (messageContext != null) {
            logEntity.setActionId(messageContext.getActionId());
            logEntity.setChoreographyId(messageContext.getChoreographyId());

            Optional<Message> message = Optional.ofNullable(executionContext)
                    .map(ExecutionContext::getMessage)
                    .or(() -> Optional.of(messageContext).map(MessageContext::getSourceMessage));
            logEntity.setMessageId(message.map(Message::getId).orElse(null));
            logEntity.setConversationId(message.map(Message::getConversationId).orElse(null));
        }
        logEntity.setMdcId(mdcService.getId());
        logEntity.setCreated(event.getInstant().atZone(ZoneId.systemDefault()));
        logEntity.setMessage(event.getFormattedMessage());
        logEntity.setLevel(event.getLevel().toString());
        logEntity.setLogger(event.getLoggerName());
        logEntity.setInMessageContext(messageContext != null);

        if (event.getThrowableProxy() != null) {
            logEntity.setStacktrace(ThrowableProxyUtil.asString(event.getThrowableProxy()));
        }

        logEntityQueue.add(logEntity);
    }

    public Page<LogDto> getLogs(Pageable pageable, LogFilter logFilter) {
        return listService
                .list(pageable, logFilter, getDtoType(), (pageable1, filter, predicate) -> getRepository()
                        .findAllToRead(
                                pageable1,
                                filter,
                                applicationSettingsService
                                        .getSystemSettingsOrDefault()
                                        .getTimeZone(),
                                predicate))
                .map(this::mapEntityToDto);
    }

    @Override
    protected LogEntity createEntity() {
        return new LogEntity();
    }

    public long countAll() {
        return getRepository().count();
    }
}
