package com.evolvsys.nexuse2e.core.logging.service;

import com.evolvsys.nexuse2e.core.logging.repository.LogRepository;
import com.evolvsys.nexuse2e.core.message.context.MessageContext;
import com.evolvsys.nexuse2e.core.message.repository.MessageRepository;
import com.evolvsys.nexuse2e.core.utils.LoggingUtils;
import com.evolvsys.nexuse2e.internal.Message;
import java.util.Optional;
import liquibase.util.Validate;
import lombok.RequiredArgsConstructor;
import lombok.extern.slf4j.Slf4j;
import org.apache.commons.lang3.StringUtils;
import org.springframework.lang.NonNull;
import org.springframework.lang.Nullable;
import org.springframework.stereotype.Service;

/**
 * @author Florian Drees
 * @since 03.06.2024
 */
@Service
@RequiredArgsConstructor
@Slf4j
public class LoggingCleanupService {
    private final LoggingService loggingService;
    private final LogRepository logRepository;
    private final MessageRepository messageRepository;

    private void removeMessageReference(String mdcId, String messageId) {
        try {
            int updateLogEntries = logRepository.removeMessageIdByMdcId(mdcId, messageId);
            LoggingUtils.globalWrapper(LOGGER)
                    .trace(
                            "Removed message reference '{}' from {} log entries (mdcId: {})",
                            messageId,
                            updateLogEntries,
                            mdcId);
        } catch (RuntimeException e) {
            LOGGER.error(
                    "Error removing log messages for mdcId {} and messageId {}: {}", mdcId, messageId, e.getMessage());
        }
    }

    private void updateLogs(@Nullable Message message, @NonNull String mdcId) {
        String messageId = Optional.ofNullable(message).map(Message::getId).orElse(null);
        if (StringUtils.isNotEmpty(messageId)) {
            if (!messageRepository.existsByMessageId(messageId)) {
                removeMessageReference(mdcId, messageId);
            }
        }
    }

    public void updateLogs(@NonNull MessageContext messageContext, @NonNull String mdcId) {
        Validate.notNull(messageContext, "messageContext must not be null");
        Validate.notNull(mdcId, "mdcId must not be null");
        LoggingUtils.globalWrapper(LOGGER).trace("cleaning logs for mdcId {}", mdcId);
        loggingService.persistQueue();
        updateLogs(messageContext.getSourceMessage(), mdcId);
        updateLogs(messageContext.getErrorMessage(), mdcId);
        updateLogs(messageContext.getReturnMessage(), mdcId);
        updateLogs(messageContext.getAcknowledgmentMessage(), mdcId);
    }
}
