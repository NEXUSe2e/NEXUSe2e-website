package com.evolvsys.nexuse2e.core.logging.service;

import java.time.ZonedDateTime;
import java.util.Date;
import org.quartz.*;
import org.springframework.context.annotation.Bean;
import org.springframework.context.annotation.Configuration;

/**
 * @author SvenBarden
 * @since 21.03.2024
 */
@Configuration
class LogSchedulerJobConfiguration {
    static final JobKey PERSIST_LOGS_JOB_KEY = new JobKey("persist", "log");
    static final TriggerKey PERSIST_LOGS_TRIGGER_KEY = new TriggerKey("persist", "log");

    @Bean
    JobDetail persistLogsJob() {
        return JobBuilder.newJob(PersistLogsJob.class)
                .withIdentity(PERSIST_LOGS_JOB_KEY)
                .storeDurably()
                .build();
    }

    @Bean
    Trigger persistLogsTrigger() {
        return TriggerBuilder.newTrigger()
                .withIdentity(PERSIST_LOGS_TRIGGER_KEY)
                .forJob(PERSIST_LOGS_JOB_KEY)
                .withSchedule(SimpleScheduleBuilder.repeatSecondlyForever(10)
                        .withMisfireHandlingInstructionNextWithRemainingCount())
                .startAt(Date.from(ZonedDateTime.now().plusMinutes(2).toInstant()))
                .build();
    }
}
