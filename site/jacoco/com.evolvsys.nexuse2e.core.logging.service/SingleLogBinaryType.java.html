package com.evolvsys.nexuse2e.core.logging.service;

import com.evolvsys.app.core.binary.model.BinaryResult;
import com.evolvsys.app.core.binary.service.BinaryType;
import com.evolvsys.nexuse2e.core.config.NEXUSe2eConfigurationProperties;
import com.evolvsys.nexuse2e.core.utils.ErrorUtils;
import edu.umd.cs.findbugs.annotations.SuppressFBWarnings;
import java.io.File;
import java.io.IOException;
import java.util.Optional;
import lombok.RequiredArgsConstructor;
import lombok.extern.slf4j.Slf4j;
import org.apache.tika.Tika;
import org.springframework.core.io.FileSystemResource;
import org.springframework.core.io.Resource;
import org.springframework.stereotype.Component;

/**
 * @author SvenBarden
 * @since 18.03.2024
 */
@Component
@RequiredArgsConstructor
@Slf4j
public class SingleLogBinaryType implements BinaryType {
    public static final String NAME = "SINGLE_LOG_FILE";
    private final NEXUSe2eConfigurationProperties configurationProperties;

    @Override
    public String name() {
        return NAME;
    }

    @SuppressFBWarnings("PATH_TRAVERSAL_IN")
    private Optional<BinaryResult> singleFile(Resource directoryResource, String id) {
        Optional<BinaryResult> result;
        try {
            File directory = directoryResource.getFile();
            File file = new File(directory, id);

            if (file.exists() && file.isFile() && file.getParentFile().equals(directory)) {
                String type = new Tika().detect(file);
                result = Optional.of(BinaryResult.builder()
                        .fileName(file.getName())
                        .size(file.length())
                        .mimeType(type)
                        .resource(new FileSystemResource(file))
                        .build());
            } else {
                result = Optional.empty();
            }
        } catch (IOException e) {
            LOGGER.error("could not get log file {}: {}", id, ErrorUtils.aggregate(e), e);
            result = Optional.empty();
        }
        return result;
    }

    @Override
    public Optional<BinaryResult> get(String id) {
        return Optional.ofNullable(configurationProperties.getLogFileDirectory())
                .flatMap(resource -> singleFile(resource, id));
    }
}
