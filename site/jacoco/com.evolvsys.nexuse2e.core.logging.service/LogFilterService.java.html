package com.evolvsys.nexuse2e.core.logging.service;

import com.evolvsys.app.core.auth.service.CurrentUserService;
import com.evolvsys.app.core.service.BaseService;
import com.evolvsys.nexuse2e.core.logging.mapper.LogFilterMapper;
import com.evolvsys.nexuse2e.core.logging.model.LogFilterDto;
import com.evolvsys.nexuse2e.core.logging.model.LogFilterEntity;
import com.evolvsys.nexuse2e.core.logging.repository.LogFilterRepository;
import java.util.List;
import org.springframework.stereotype.Service;

/**
 * @author SvenBarden
 * @since 14.03.2024
 */
@Service
public class LogFilterService extends BaseService<LogFilterDto, LogFilterEntity, LogFilterRepository> {
    public LogFilterService(
            LogFilterMapper mapper, LogFilterRepository repository, CurrentUserService currentUserService) {
        super(mapper, repository, currentUserService);
    }

    @Override
    public LogFilterEntity getEntity(LogFilterDto dto) {
        return super.getEntity(dto, getRepository()::getEagerById);
    }

    @Override
    protected LogFilterEntity createEntity() {
        return new LogFilterEntity();
    }

    public List<LogFilterDto> getAll() {
        return getRepository().findAll().stream().map(this::mapEntityToDto).toList();
    }
}
