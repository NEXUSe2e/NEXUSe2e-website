package com.evolvsys.nexuse2e.core.logging.service;

import com.evolvsys.app.core.binary.model.BinaryObjectDto;
import com.evolvsys.app.core.binary.service.BinaryType;
import com.evolvsys.nexuse2e.core.config.NEXUSe2eConfigurationProperties;
import com.evolvsys.nexuse2e.core.exception.NexusError;
import com.evolvsys.nexuse2e.core.exception.NexusException;
import java.io.File;
import java.io.IOException;
import java.util.ArrayList;
import java.util.Collection;
import java.util.List;
import lombok.RequiredArgsConstructor;
import org.apache.tika.Tika;
import org.springframework.core.io.Resource;
import org.springframework.stereotype.Service;

/**
 * @author SvenBarden
 * @since 18.03.2024
 */
@Service
@RequiredArgsConstructor
public class LogFileService {
    private final NEXUSe2eConfigurationProperties configurationProperties;

    public List<BinaryObjectDto> list() throws NexusException {
        List<BinaryObjectDto> files = new ArrayList<>();

        Resource directory = configurationProperties.getLogFileDirectory();
        if (directory != null) {
            try {
                File directoryFile = directory.getFile();
                if (directoryFile.exists() && directoryFile.isDirectory()) {
                    addFiles(directoryFile, files);
                }
            } catch (IOException e) {
                throw new NexusException(
                        NexusError.builder()
                                .message("Could not load log file list")
                                .build(),
                        e);
            }
        }

        return files;
    }

    private static void addFiles(File directoryFile, Collection<BinaryObjectDto> files) throws IOException {
        File[] fileArray = directoryFile.listFiles();
        if (fileArray != null) {
            Tika tika = new Tika();
            for (File file : fileArray) {
                if (file.isFile()) {
                    files.add(BinaryObjectDto.builder()
                            .fileName(file.getName())
                            .size(file.length())
                            .mimeType(tika.detect(file))
                            .path(BinaryType.getPath(SingleLogBinaryType.NAME, file.getName()))
                            .build());
                }
            }
        }
    }
}
