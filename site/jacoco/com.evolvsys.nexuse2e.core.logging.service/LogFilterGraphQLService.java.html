package com.evolvsys.nexuse2e.core.logging.service;

import com.evolvsys.app.core.graphql.annotation.GraphQLService;
import com.evolvsys.app.core.graphql.dataloader.DataLoaderService;
import com.evolvsys.nexuse2e.core.logging.model.LogFilterDto;
import com.evolvsys.nexuse2e.core.message.MessageStatus;
import io.leangen.graphql.annotations.*;
import io.leangen.graphql.execution.ResolutionEnvironment;
import jakarta.validation.Valid;
import java.util.List;
import java.util.Optional;
import java.util.Set;
import java.util.concurrent.CompletableFuture;
import lombok.RequiredArgsConstructor;

/**
 * @author SvenBarden
 * @since 14.03.2024
 */
@GraphQLService
@RequiredArgsConstructor
public class LogFilterGraphQLService {
    private final LogFilterService logFilterService;
    private final DataLoaderService dataLoaderService;

    @GraphQLQuery(name = "logFilter_getAll")
    @GraphQLNonNull
    public List<@GraphQLNonNull LogFilterDto> getAll() {
        return logFilterService.getAll();
    }

    @GraphQLQuery(name = "logFilter_getById")
    public Optional<LogFilterDto> getById(long id) {
        return logFilterService.getById(id);
    }

    @GraphQLMutation(name = "logFilter_update")
    @GraphQLNonNull
    public LogFilterDto update(@GraphQLNonNull @Valid LogFilterDto logFilter) {
        return logFilterService.save(logFilter);
    }

    @GraphQLMutation(name = "logFilter_delete")
    public boolean delete(long id) {
        logFilterService.delete(id);
        return true;
    }

    @GraphQLQuery
    public CompletableFuture<Set<@GraphQLNonNull MessageStatus>> messageStatuses(
            @GraphQLContext LogFilterDto logFilterDto,
            @GraphQLEnvironment ResolutionEnvironment resolutionEnvironment) {
        return dataLoaderService.load(
                resolutionEnvironment,
                LogFilterMessageStatusesBatchLoader.class,
                logFilterDto,
                "messageStatuses",
                LogFilterDto::getId);
    }
}
