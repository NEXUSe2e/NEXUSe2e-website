package com.evolvsys.nexuse2e.core.logging.service;

import static com.evolvsys.nexuse2e.core.file.TempFileCleanupService.TEMP_FILE_NAME_PREFIX;

import com.evolvsys.app.core.binary.model.BinaryResult;
import com.evolvsys.app.core.binary.service.BinaryType;
import com.evolvsys.nexuse2e.core.config.NEXUSe2eConfigurationProperties;
import com.evolvsys.nexuse2e.core.utils.ErrorUtils;
import java.io.File;
import java.io.IOException;
import java.io.InputStream;
import java.io.OutputStream;
import java.nio.file.Files;
import java.time.LocalDateTime;
import java.util.Optional;
import java.util.zip.ZipEntry;
import java.util.zip.ZipOutputStream;
import lombok.RequiredArgsConstructor;
import lombok.extern.slf4j.Slf4j;
import org.springframework.core.io.FileSystemResource;
import org.springframework.core.io.Resource;
import org.springframework.stereotype.Component;

/**
 * @author SvenBarden
 * @since 18.03.2024
 */
@Component
@RequiredArgsConstructor
@Slf4j
public class AllLogsBinaryType implements BinaryType {
    public static final String NAME = "ALL_LOGS_FILE";
    private final NEXUSe2eConfigurationProperties configurationProperties;

    @Override
    public String name() {
        return NAME;
    }

    private Optional<BinaryResult> zip(Resource directoryResource) {
        Optional<BinaryResult> result;
        try {
            File directory = directoryResource.getFile();

            if (directory.exists() && directory.isDirectory()) {
                File tempFile =
                        Files.createTempFile(TEMP_FILE_NAME_PREFIX, "logs").toFile();

                try (OutputStream tempOutput = Files.newOutputStream(tempFile.toPath());
                        ZipOutputStream zipOutputStream = new ZipOutputStream(tempOutput)) {
                    File[] files = directory.listFiles();
                    if (files != null) {
                        for (File file : files) {
                            if (file.exists() && file.isFile()) {
                                ZipEntry zipEntry = new ZipEntry(file.getName());
                                zipEntry.setSize(file.length());
                                zipOutputStream.putNextEntry(zipEntry);
                                try (InputStream fileInputStream = Files.newInputStream(file.toPath())) {
                                    fileInputStream.transferTo(zipOutputStream);
                                }
                                zipOutputStream.closeEntry();
                            }
                        }
                        zipOutputStream.finish();
                    }
                }

                result = Optional.of(BinaryResult.builder()
                        .fileName("NEXUSe2e_logs_" + LocalDateTime.now().withNano(0) + ".zip")
                        .size(tempFile.length())
                        .mimeType("application/zip")
                        .resource(new FileSystemResource(tempFile))
                        .build());
            } else {
                result = Optional.empty();
            }
        } catch (IOException e) {
            LOGGER.error("could not zip log files: {}", ErrorUtils.aggregate(e), e);
            result = Optional.empty();
        }
        return result;
    }

    @Override
    public Optional<BinaryResult> get(String id) {
        return Optional.ofNullable(configurationProperties.getLogFileDirectory())
                .flatMap(this::zip);
    }
}
