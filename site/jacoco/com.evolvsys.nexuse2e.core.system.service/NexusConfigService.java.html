package com.evolvsys.nexuse2e.core.system.service;

import com.evolvsys.nexuse2e.core.config.NEXUSe2eConfigurationProperties;
import com.evolvsys.nexuse2e.core.system.model.NexusConfigDto;
import lombok.RequiredArgsConstructor;
import org.springframework.stereotype.Service;

/**
 * @author Florian Drees
 * @since 15.04.2024
 */
@Service
@RequiredArgsConstructor
public class NexusConfigService {
    private final NEXUSe2eConfigurationProperties nexusConfigurationProperties;

    public NexusConfigDto getConfig() {
        return NexusConfigDto.builder()
                .caCertsPath(nexusConfigurationProperties.getCaCertsPath())
                .defaultIdleConversationsDuration(nexusConfigurationProperties.getDefaultIdleConversationsDuration())
                .build();
    }
}
