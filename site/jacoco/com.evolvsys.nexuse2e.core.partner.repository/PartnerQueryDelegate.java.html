package com.evolvsys.nexuse2e.core.partner.repository;

import com.evolvsys.nexuse2e.core.model.entity.PartnerEntity;
import com.evolvsys.nexuse2e.core.model.entity.QPartnerEntity;
import com.evolvsys.nexuse2e.core.partner.PartnerFilter;
import com.querydsl.core.BooleanBuilder;
import com.querydsl.core.annotations.QueryDelegate;
import com.querydsl.core.types.Predicate;

/**
 * @author Florian Drees
 * @since 07.02.2024
 */
public final class PartnerQueryDelegate {
    private PartnerQueryDelegate() {}

    @QueryDelegate(PartnerEntity.class)
    public static Predicate filter(QPartnerEntity qPartnerEntity, PartnerFilter partnerFilter) {
        BooleanBuilder builder = new BooleanBuilder();
        if (partnerFilter != null) {
            if (partnerFilter.getPartnerType() != null) {
                builder.and(qPartnerEntity.partnerType.eq(partnerFilter.getPartnerType()));
            }
        }
        return builder.getValue();
    }
}
