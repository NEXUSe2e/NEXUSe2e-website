package com.evolvsys.nexuse2e.core.partner.repository;

import com.evolvsys.nexuse2e.core.config.model.PartnerType;
import com.evolvsys.nexuse2e.core.model.entity.PartnerEntity;
import com.evolvsys.nexuse2e.core.model.entity.QPartnerEntity;
import com.evolvsys.nexuse2e.core.partner.PartnerFilter;
import com.querydsl.core.types.Predicate;
import com.querydsl.core.types.dsl.BooleanExpression;
import java.util.List;
import java.util.Optional;
import org.apache.commons.collections4.IterableUtils;
import org.springframework.data.jpa.repository.JpaRepository;
import org.springframework.data.jpa.repository.Query;
import org.springframework.data.querydsl.QuerydslPredicateExecutor;

/**
 * @author SvenBarden
 * @since 25.10.2023
 */
public interface PartnerRepository
        extends JpaRepository<PartnerEntity, Long>, QuerydslPredicateExecutor<PartnerEntity> {
    Optional<PartnerEntity> findByPartnerId(String partnerId);

    boolean existsByPartnerId(String partnerId);

    boolean existsByPartnerIdAndIdNot(String partnerId, Long id);

    default List<PartnerEntity> findAll(PartnerFilter partnerFilter, BooleanExpression securityPredicate) {
        Predicate predicate = QPartnerEntity.partnerEntity.filter(partnerFilter);
        if (predicate != null) {
            securityPredicate = securityPredicate.and(predicate);
        }
        return IterableUtils.toList(findAll(securityPredicate));
    }

    @Query(
            "select p.partnerId from PartnerEntity p where p.partnerType = :partnerType and p.defaultLocalIdentity = true")
    Optional<String> getDefaultLocalIdentityName(PartnerType partnerType);

    boolean existsByDefaultLocalIdentityAndIdNot(boolean defaultLocalIdentity, Long id);

    boolean existsByDefaultLocalIdentity(boolean defaultLocalIdentity);
}
