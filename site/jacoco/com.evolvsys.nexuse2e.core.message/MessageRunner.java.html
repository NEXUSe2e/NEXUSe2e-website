package com.evolvsys.nexuse2e.core.message;

import com.evolvsys.nexuse2e.core.async.MessageExecutor;
import com.evolvsys.nexuse2e.core.exception.NexusException;
import com.evolvsys.nexuse2e.core.logging.MessageLogContextHolder;
import com.evolvsys.nexuse2e.core.message.context.MessageContext;
import java.util.concurrent.CompletableFuture;
import java.util.concurrent.Executor;
import java.util.function.Supplier;
import lombok.RequiredArgsConstructor;
import lombok.SneakyThrows;
import lombok.extern.slf4j.Slf4j;
import org.springframework.stereotype.Component;

/**
 * @author SvenBarden
 * @since 18.10.2023
 */
@Component
@Slf4j
@RequiredArgsConstructor
public class MessageRunner {
    @MessageExecutor
    private final Executor messageExecutor;

    private Supplier<MessageContext> createSupplier(MessageContext messageContext, Supplier<MessageContext> delegate) {
        return () -> {
            // assign the message context from the parent thread to the new created thread
            MessageLogContextHolder.getContext().setMessageContext(messageContext);
            MessageContext newContext = delegate.get();
            MessageLogContextHolder.clearContext();
            return newContext;
        };
    }

    public CompletableFuture<MessageContext> createFuture(
            MessageContext messageContext, Supplier<MessageContext> messageContextSupplier) {
        LOGGER.debug("creating future for {}", messageContext);

        return CompletableFuture.supplyAsync(createSupplier(messageContext, messageContextSupplier), messageExecutor);
    }

    public CompletableFuture<MessageContext> createWrappedFuture(
            MessageContext messageContext, SupplierWrapper messageContextSupplier) {
        LOGGER.debug("creating future for {}", messageContext);

        Supplier<MessageContext> supplier = new Supplier<>() {
            @SneakyThrows
            @Override
            public MessageContext get() {
                return messageContextSupplier.get();
            }
        };

        return CompletableFuture.supplyAsync(createSupplier(messageContext, supplier), messageExecutor);
    }

    @FunctionalInterface
    public interface SupplierWrapper {
        MessageContext get() throws NexusException;
    }
}
