package com.evolvsys.nexuse2e.core.message;

import com.evolvsys.common.lock.LockService;
import com.evolvsys.nexuse2e.core.action.Action;
import com.evolvsys.nexuse2e.core.action.model.ActionEntity;
import com.evolvsys.nexuse2e.core.action.repository.ActionRepository;
import com.evolvsys.nexuse2e.core.choreography.repository.ChoreographyRepository;
import com.evolvsys.nexuse2e.core.exception.NexusError;
import com.evolvsys.nexuse2e.core.exception.NexusRuntimeException;
import com.evolvsys.nexuse2e.core.message.mapper.ContextMessageMapper;
import com.evolvsys.nexuse2e.core.message.payload.model.BinaryEntity;
import com.evolvsys.nexuse2e.core.message.payload.model.PayloadEntity;
import com.evolvsys.nexuse2e.core.message.repository.ConversationRepository;
import com.evolvsys.nexuse2e.core.message.repository.MessageRepository;
import com.evolvsys.nexuse2e.core.model.entity.ConversationEntity;
import com.evolvsys.nexuse2e.core.model.entity.MessageEntity;
import com.evolvsys.nexuse2e.internal.ContextPayload;
import com.evolvsys.nexuse2e.internal.Message;
import java.time.ZonedDateTime;
import java.util.Optional;
import lombok.RequiredArgsConstructor;
import lombok.extern.slf4j.Slf4j;
import org.hibernate.engine.jdbc.BlobProxy;
import org.springframework.stereotype.Service;
import org.springframework.transaction.annotation.Transactional;

/**
 * @author SvenBarden
 * @since 10.10.2023
 */
@Slf4j
@Service
@RequiredArgsConstructor
public class ContextMessageService {
    private final MessageRepository messageRepository;
    private final ChoreographyRepository choreographyRepository;
    private final ConversationRepository conversationRepository;
    private final ActionRepository actionRepository;
    private final LockService lockService;
    private final ContextMessageMapper contextMessageMapper;

    private String getMessageLockKey(String messageId) {
        return "message_" + messageId;
    }

    private String getConversationLockKey(String conversationId) {
        return "conversation_" + conversationId;
    }

    private ActionEntity getActionEntity(Action action) {
        return actionRepository
                .findByActionIdAndChoreography_ChoreographyId(action.getActionId(), action.getChoreographyId())
                .stream()
                .filter(actionEntity ->
                        actionEntity.getChoreography().getChoreographyId().equals(action.getChoreographyId())
                                && actionEntity.getActionId().equals(action.getActionId()))
                .findFirst()
                .orElseThrow(() -> new NexusRuntimeException(NexusError.builder()
                        .message("action not found")
                        .extension("action", action.getActionId())
                        .extension("choreography", action.getChoreographyId())
                        .build()));
    }

    private ConversationEntity createConversation(Message message, Action action) {
        ConversationEntity conversationEntity = new ConversationEntity();
        conversationEntity.setConversationId(message.getConversationId());
        conversationEntity.setChoreography(
                choreographyRepository.findByChoreographyId(action.getChoreographyId()).stream()
                        .filter(choreographyEntity ->
                                choreographyEntity.getChoreographyId().equals(action.getChoreographyId()))
                        .findFirst()
                        .orElseThrow(() -> new NexusRuntimeException(NexusError.builder()
                                .message("choreography not found")
                                .extension("choreographyId", action.getChoreographyId())
                                .build())));
        conversationEntity.setCurrentAction(getActionEntity(action));
        return conversationEntity;
    }

    private MessageEntity create(Message message) {
        LOGGER.trace("create new messageEntity with id {}", message.getId());
        MessageEntity messageEntity = new MessageEntity();
        messageEntity.setMessageId(message.getId());
        messageEntity.setType(message.getType());
        return messageEntity;
    }

    private Message entityToMessage(MessageEntity messageEntity) {
        return contextMessageMapper.entityToMessage(messageEntity);
    }

    public Optional<Message> getMessage(String messageId) {
        return messageRepository.findByMessageId(messageId).map(this::entityToMessage);
    }

    @Transactional
    public void sent(Message message) {
        lockService.execute(getMessageLockKey(message.getId()), () -> {
            MessageEntity messageEntity =
                    messageRepository.findByMessageId(message.getId()).orElseGet(MessageEntity::new);
            messageEntity.setMessageId(message.getId());
            messageEntity.setStatus(MessageStatus.TRANSFERRED);
            messageEntity.setSent(ZonedDateTime.now());
            messageRepository.save(messageEntity);
        });
    }

    @Transactional
    public void save(Message message, Action action) {
        LOGGER.debug("saving message {}", message);

        String conversationId = message.getConversationId();
        ConversationEntity conversationEntity = lockService.evaluate(getConversationLockKey(conversationId), () -> {
            ConversationEntity internalConversationEntity = conversationRepository
                    .findByConversationId(conversationId)
                    .orElseGet(() -> createConversation(message, action));
            LOGGER.trace("setting currentAction to {} in conversation {}", action.getActionId(), conversationId);
            internalConversationEntity.setCurrentAction(getActionEntity(action));
            return conversationRepository.save(internalConversationEntity);
        });

        lockService.execute(getMessageLockKey(message.getId()), () -> {
            MessageEntity messageEntity =
                    messageRepository.findByMessageId(message.getId()).orElseGet(() -> create(message));
            boolean newMessage = messageEntity.getId() == null;

            contextMessageMapper.updateEntity(message, messageEntity);

            messageEntity.setConversation(conversationEntity);

            messageEntity.setStatus(message.getMessageStatus());
            if (messageEntity.getStatus() == MessageStatus.TRANSFERRED) {
                messageEntity.setSent(ZonedDateTime.now());
            }

            if (newMessage) {
                LOGGER.trace("saving message payloads");
                messageEntity.getPayloads().clear();
                for (ContextPayload payload : message.getPayloads()) {
                    BinaryEntity binaryEntity = new BinaryEntity();
                    binaryEntity.setContent(
                            BlobProxy.generateProxy(payload.getContent().getInputStream(), payload.getSize()));

                    PayloadEntity payloadEntity = new PayloadEntity();
                    payloadEntity.setSize(payload.getSize());
                    payloadEntity.setContentType(payload.getContentType());
                    payloadEntity.setEncoding(payload.getEncoding());
                    payloadEntity.setContent(binaryEntity);

                    messageEntity.getPayloads().add(payloadEntity);
                }
            }

            messageRepository.save(messageEntity);
        });
    }
}
