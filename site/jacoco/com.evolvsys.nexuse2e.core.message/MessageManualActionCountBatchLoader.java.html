package com.evolvsys.nexuse2e.core.message;

import com.evolvsys.app.core.graphql.dataloader.GraphQLBatchLoader;
import com.evolvsys.nexuse2e.core.manualaction.model.QMessageActionManualActionEntity;
import com.evolvsys.nexuse2e.core.model.entity.QMessageEntity;
import com.evolvsys.nexuse2e.core.utils.BatchLoaderUtils;
import com.querydsl.core.group.GroupBy;
import com.querydsl.jpa.JPQLQueryFactory;
import java.util.Map;
import java.util.Set;
import java.util.concurrent.CompletionStage;
import lombok.RequiredArgsConstructor;
import org.dataloader.BatchLoaderEnvironment;
import org.springframework.stereotype.Component;

/**
 * @author SvenBarden
 * @since 22.08.2024
 */
@Component
@RequiredArgsConstructor
public class MessageManualActionCountBatchLoader implements GraphQLBatchLoader<Long, Long> {
    private final JPQLQueryFactory queryFactory;

    @Override
    public CompletionStage<Map<Long, Long>> load(Set<Long> keys, BatchLoaderEnvironment batchLoaderEnvironment) {
        return BatchLoaderUtils.load(keys, this::loadInternal);
    }

    private Map<Long, Long> loadInternal(Set<Long> keys) {
        return queryFactory
                .select(
                        QMessageEntity.messageEntity.id,
                        QMessageActionManualActionEntity.messageActionManualActionEntity.count())
                .from(QMessageActionManualActionEntity.messageActionManualActionEntity)
                .join(
                        QMessageActionManualActionEntity.messageActionManualActionEntity.messages,
                        QMessageEntity.messageEntity)
                .on(QMessageEntity.messageEntity.id.in(keys))
                .groupBy(QMessageEntity.messageEntity.id)
                .transform(GroupBy.groupBy(QMessageEntity.messageEntity.id)
                        .as(QMessageActionManualActionEntity.messageActionManualActionEntity.count()));
    }
}
