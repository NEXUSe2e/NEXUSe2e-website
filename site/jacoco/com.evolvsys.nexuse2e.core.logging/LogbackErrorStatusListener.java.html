package com.evolvsys.nexuse2e.core.logging;

import ch.qos.logback.core.status.OnErrorConsoleStatusListener;
import ch.qos.logback.core.status.Status;

/**
 * @author SvenBarden
 * @since 27.05.2024
 */
public class LogbackErrorStatusListener extends OnErrorConsoleStatusListener {

    @Override
    public void addStatusEvent(Status status) {
        if (status.getLevel() >= Status.WARN) {
            super.addStatusEvent(status);
        }
    }
}
