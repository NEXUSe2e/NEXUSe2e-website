package com.evolvsys.nexuse2e.core.logging;

import ch.qos.logback.classic.Level;
import ch.qos.logback.classic.LoggerContext;
import ch.qos.logback.classic.filter.ThresholdFilter;
import ch.qos.logback.classic.spi.ILoggingEvent;
import ch.qos.logback.core.UnsynchronizedAppenderBase;
import ch.qos.logback.core.spi.FilterReply;
import com.evolvsys.nexuse2e.core.logging.model.ForMessageLoggingContext;
import com.evolvsys.nexuse2e.core.logging.model.LogFilterType;
import com.evolvsys.nexuse2e.core.logging.service.LogFilterCacheService;
import com.evolvsys.nexuse2e.core.logging.service.LoggingService;
import com.evolvsys.nexuse2e.core.utils.ErrorUtils;
import lombok.RequiredArgsConstructor;
import lombok.extern.slf4j.Slf4j;
import org.apache.commons.collections4.CollectionUtils;
import org.slf4j.Logger;
import org.slf4j.LoggerFactory;
import org.springframework.beans.factory.InitializingBean;
import org.springframework.context.SmartLifecycle;
import org.springframework.stereotype.Component;

/**
 * @author SvenBarden
 * @since 04.01.2024
 */
@Component
@RequiredArgsConstructor
@Slf4j
public class MessageLogbackAppender extends UnsynchronizedAppenderBase<ILoggingEvent>
        implements SmartLifecycle, InitializingBean {
    private final LoggingService loggingService;
    private final LogFilterCacheService logFilterCacheService;
    private DatabaseLogFilterHelper databaseLogFilterHelper;

    private static ThresholdFilter createThresholdFilter() {
        ThresholdFilter thresholdFilter = new ThresholdFilter();
        thresholdFilter.setLevel(Level.INFO.levelStr);
        thresholdFilter.start();
        return thresholdFilter;
    }

    @Override
    public boolean isAutoStartup() {
        return false;
    }

    private void updateFilters() {
        databaseLogFilterHelper.updateFilters();
    }

    @Override
    protected void append(ILoggingEvent event) {
        if (event.getLoggerName().equals(MessageLogbackAppender.class.getName()) == false) {
            try {
                LOGGER.debug("write log message {} from {}", event, event.getLoggerName());
                if (MessageLogContextHolder.exists()
                        && (MessageLogContextHolder.getContext().isForceGlobal() == false)) {
                    loggingService.log(
                            MessageLogContextHolder.getContext().getMessageContext(),
                            MessageLogContextHolder.getContext().getExecutionContext(),
                            event);
                } else {
                    ForMessageLoggingContext context = null;
                    if (CollectionUtils.isNotEmpty(event.getKeyValuePairs())) {
                        context = event.getKeyValuePairs().stream()
                                .filter(keyValuePair -> ForMessageLoggingContext.KEY_NAME.equals(keyValuePair.key))
                                .findFirst()
                                .map(keyValuePair -> (ForMessageLoggingContext) keyValuePair.value)
                                .orElse(null);
                    }

                    if (context == null) {
                        loggingService.log(null, null, event);
                    } else {
                        loggingService.log(context, event);
                    }
                }
            } catch (Throwable e) {
                LOGGER.error("could not write log message {}", ErrorUtils.aggregate(e), e);
            }
        }
    }

    @Override
    public FilterReply getFilterChainDecision(ILoggingEvent event) {
        updateFilters();
        return super.getFilterChainDecision(event);
    }

    @Override
    public boolean isRunning() {
        return isStarted();
    }

    @Override
    public void afterPropertiesSet() throws Exception {
        if (LoggerFactory.getILoggerFactory() instanceof LoggerContext loggerContext) {
            LOGGER.info("registering message logback appender");
            setContext(loggerContext);
            setName(getClass().getSimpleName());

            ch.qos.logback.classic.Logger rootLogger = loggerContext.getLogger(Logger.ROOT_LOGGER_NAME);
            synchronized (this) {
                databaseLogFilterHelper = new DatabaseLogFilterHelper(
                        logFilterCacheService, this, LogFilterType.DATABASE, createThresholdFilter());
                updateFilters();
            }
            start();
            rootLogger.addAppender(this);
        } else {
            throw new IllegalStateException("logger context ist not of type logback");
        }
    }
}
