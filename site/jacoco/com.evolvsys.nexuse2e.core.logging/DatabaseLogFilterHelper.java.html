package com.evolvsys.nexuse2e.core.logging;

import ch.qos.logback.classic.Level;
import ch.qos.logback.classic.spi.ILoggingEvent;
import ch.qos.logback.core.Appender;
import ch.qos.logback.core.filter.Filter;
import com.evolvsys.nexuse2e.core.logging.model.LogFilterCache;
import com.evolvsys.nexuse2e.core.logging.model.LogFilterType;
import com.evolvsys.nexuse2e.core.logging.service.LogFilterCacheService;
import edu.umd.cs.findbugs.annotations.SuppressFBWarnings;
import java.util.ArrayList;
import java.util.concurrent.locks.Lock;
import java.util.concurrent.locks.ReentrantLock;
import lombok.RequiredArgsConstructor;
import lombok.extern.slf4j.Slf4j;
import org.apache.commons.collections4.ListUtils;

/**
 * @author SvenBarden
 * @since 19.03.2024
 */
@RequiredArgsConstructor
@Slf4j
public class DatabaseLogFilterHelper {
    private final LogFilterCacheService logFilterCacheService;
    private final Appender<ILoggingEvent> appender;
    private final LogFilterType logFilterType;
    private final Filter<ILoggingEvent> defaultFilter;

    private int filterHash = -1;
    private final Lock filterLock = new ReentrantLock();

    private DatabaseLogFilter createFilter(LogFilterCache logFilter) {
        DatabaseLogFilter databaseLogFilter = new DatabaseLogFilter();
        databaseLogFilter.setLogger(logFilter.getLogger());
        databaseLogFilter.setLevel(Level.toLevel(logFilter.getLevel().name()));
        databaseLogFilter.setMessageStatuses(new ArrayList<>(ListUtils.emptyIfNull(logFilter.getMessageStatuses())));
        databaseLogFilter.setErrorPattern(logFilter.getErrorPattern());
        databaseLogFilter.start();
        return databaseLogFilter;
    }

    @SuppressFBWarnings(value = "USFW_UNSYNCHRONIZED_SINGLETON_FIELD_WRITES", justification = "we use a lock")
    public void updateFilters() {
        LogFilterCacheContainer logFilters = logFilterCacheService.getLogFilters(logFilterType);
        int newHash = logFilters.hashCode();
        if (newHash != filterHash) {
            if (filterLock.tryLock()) {
                try {
                    filterHash = logFilters.hashCode();
                    appender.clearAllFilters();
                    for (LogFilterCache logFilter : logFilters.getLogFilters()) {
                        try {
                            appender.addFilter(createFilter(logFilter));
                        } catch (Throwable e) {
                            LOGGER.error("could not add custom filter {}", logFilter, e);
                        }
                    }
                    appender.addFilter(defaultFilter);
                } finally {
                    filterLock.unlock();
                }
            } else {
                // filter update is already running in another thread
            }
        }
    }
}
