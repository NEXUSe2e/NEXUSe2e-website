package com.evolvsys.nexuse2e.core.logging;

import ch.qos.logback.classic.spi.ILoggingEvent;
import ch.qos.logback.classic.spi.ThrowableProxyUtil;
import ch.qos.logback.core.LayoutBase;
import lombok.RequiredArgsConstructor;
import org.thymeleaf.context.Context;
import org.thymeleaf.spring6.SpringTemplateEngine;

/**
 * @author SvenBarden
 * @since 19.03.2024
 */
@RequiredArgsConstructor
public class TemplateLayout extends LayoutBase<ILoggingEvent> {
    private final SpringTemplateEngine springTemplateEngine;

    @Override
    public String getContentType() {
        return "text/html";
    }

    @Override
    public String doLayout(ILoggingEvent event) {
        Context context = new Context();

        String layout;
        try {
            if (event instanceof LoggingEventWrapper loggingEventWrapper) {
                context.setVariable("messageContext", loggingEventWrapper.getMessageContext());
            }
            if (event.getThrowableProxy() != null) {
                context.setVariable("stacktrace", ThrowableProxyUtil.asString(event.getThrowableProxy()));
            }
            context.setVariable("logMessage", event.getFormattedMessage());
            layout = springTemplateEngine.process("error", context);
        } catch (Throwable e) {
            layout = "Could not generate email template: " + e.getMessage();
        }
        return layout;
    }
}
