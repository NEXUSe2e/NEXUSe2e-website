package com.evolvsys.nexuse2e.core.logging;

import ch.qos.logback.classic.Level;
import ch.qos.logback.classic.spi.ILoggingEvent;
import ch.qos.logback.classic.spi.ThrowableProxyUtil;
import ch.qos.logback.core.filter.Filter;
import ch.qos.logback.core.spi.FilterReply;
import com.evolvsys.nexuse2e.core.message.MessageStatus;
import com.evolvsys.nexuse2e.core.message.context.ExecutionContext;
import com.evolvsys.nexuse2e.internal.Message;
import java.util.List;
import java.util.Optional;
import java.util.regex.Pattern;
import lombok.AccessLevel;
import lombok.Getter;
import lombok.Setter;
import lombok.extern.slf4j.Slf4j;
import org.apache.commons.collections4.CollectionUtils;
import org.apache.commons.lang3.StringUtils;

/**
 * @author SvenBarden
 * @since 14.03.2024
 */
@Getter
@Setter
@Slf4j
public class DatabaseLogFilter extends Filter<ILoggingEvent> {
    private String logger;
    private Level level;
    private List<MessageStatus> messageStatuses;
    private String errorPattern;

    @Getter(AccessLevel.NONE)
    @Setter(AccessLevel.NONE)
    private Pattern compiledErrorPattern;

    public void setErrorPattern(String errorPattern) {
        this.errorPattern = errorPattern;
        compiledErrorPattern = errorPattern == null ? null : Pattern.compile(errorPattern);
    }

    private boolean checkLevel(ILoggingEvent event) {
        return level == null || event.getLevel().isGreaterOrEqual(level);
    }

    private boolean checkLogger(ILoggingEvent event) {
        return StringUtils.isBlank(logger)
                || logger.equals(event.getLoggerName())
                || event.getLoggerName().startsWith(logger + ".");
    }

    private boolean checkMessageStatus() {
        return CollectionUtils.isEmpty(messageStatuses)
                || Optional.ofNullable(MessageLogContextHolder.getContext())
                        .map(MessageLogContext::getExecutionContext)
                        .map(ExecutionContext::getMessage)
                        .map(Message::getMessageStatus)
                        .map(messageStatuses::contains)
                        .orElse(Boolean.FALSE);
    }

    private boolean matchErrorPattern(ILoggingEvent event) {
        return compiledErrorPattern.matcher(event.getFormattedMessage()).find()
                || (event.getThrowableProxy() != null
                        && compiledErrorPattern
                                .matcher(ThrowableProxyUtil.asString(event.getThrowableProxy()))
                                .find());
    }

    private boolean checkErrorPattern(ILoggingEvent event) {
        return compiledErrorPattern == null || matchErrorPattern(event);
    }

    @Override
    public FilterReply decide(ILoggingEvent event) {
        FilterReply reply;
        if (isStarted()) {
            if (checkLevel(event) && checkLogger(event) && checkMessageStatus() && checkErrorPattern(event)) {
                reply = FilterReply.ACCEPT;
            } else {
                reply = FilterReply.NEUTRAL;
            }
        } else {
            reply = FilterReply.NEUTRAL;
        }
        return reply;
    }
}
