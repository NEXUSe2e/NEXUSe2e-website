package com.evolvsys.nexuse2e.core.logging;

import java.util.function.Supplier;
import lombok.experimental.UtilityClass;

/**
 * @author SvenBarden
 * @since 04.01.2024
 */
@UtilityClass
public class MessageLogContextHolder {
    private static final ThreadLocal<Supplier<MessageLogContext>> CONTEXT_HOLDER = new ThreadLocal<>();

    public static void clearContext() {
        CONTEXT_HOLDER.remove();
    }

    public static boolean exists() {
        return CONTEXT_HOLDER.get() != null;
    }

    public MessageLogContext createEmptyContext() {
        return new MessageLogContext();
    }

    public static MessageLogContext getContext() {
        return getDeferredContext().get();
    }

    public static Supplier<MessageLogContext> getDeferredContext() {
        Supplier<MessageLogContext> result = CONTEXT_HOLDER.get();
        if (result == null) {
            MessageLogContext context = createEmptyContext();
            result = () -> context;
            CONTEXT_HOLDER.set(result);
        }

        return result;
    }

    public static void setContext(MessageLogContext messageLogContext) {
        CONTEXT_HOLDER.set(() -> messageLogContext);
    }

    public static void setDeferredContext(Supplier<MessageLogContext> deferredContext) {
        CONTEXT_HOLDER.set(deferredContext);
    }
}
