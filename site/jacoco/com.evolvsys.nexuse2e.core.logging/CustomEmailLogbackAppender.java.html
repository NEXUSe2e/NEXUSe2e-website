package com.evolvsys.nexuse2e.core.logging;

import ch.qos.logback.classic.Level;
import ch.qos.logback.classic.LoggerContext;
import ch.qos.logback.classic.filter.ThresholdFilter;
import ch.qos.logback.classic.html.HTMLLayout;
import ch.qos.logback.classic.net.SMTPAppender;
import ch.qos.logback.classic.spi.ILoggingEvent;
import ch.qos.logback.core.boolex.EventEvaluatorBase;
import ch.qos.logback.core.spi.FilterReply;
import com.evolvsys.app.core.system.service.SystemConfigurationProperties;
import com.evolvsys.nexuse2e.core.config.NEXUSe2eConfigurationProperties;
import com.evolvsys.nexuse2e.core.logging.model.LogFilterType;
import com.evolvsys.nexuse2e.core.logging.service.LogFilterCacheService;
import lombok.RequiredArgsConstructor;
import lombok.extern.slf4j.Slf4j;
import org.apache.commons.collections4.CollectionUtils;
import org.apache.commons.lang3.StringUtils;
import org.slf4j.Logger;
import org.slf4j.LoggerFactory;
import org.springframework.beans.factory.InitializingBean;
import org.springframework.boot.autoconfigure.mail.MailProperties;
import org.springframework.context.SmartLifecycle;
import org.springframework.stereotype.Component;
import org.thymeleaf.spring6.SpringTemplateEngine;

/**
 * @author SvenBarden
 * @since 19.03.2024
 */
@Component
@RequiredArgsConstructor
@Slf4j
public class CustomEmailLogbackAppender extends SMTPAppender implements SmartLifecycle, InitializingBean {
    private final NEXUSe2eConfigurationProperties configurationProperties;
    private final MailProperties mailProperties;
    private final SpringTemplateEngine springTemplateEngine;
    private final LogFilterCacheService logFilterCacheService;
    private final SystemConfigurationProperties systemConfigurationProperties;
    private DatabaseLogFilterHelper databaseLogFilterHelper;

    @Override
    public boolean isRunning() {
        return isStarted();
    }

    @Override
    public boolean isAutoStartup() {
        return false;
    }

    @Override
    public void doAppend(ILoggingEvent eventObject) {
        if (MessageLogContextHolder.exists()) {
            eventObject = new LoggingEventWrapper(
                    eventObject, MessageLogContextHolder.getContext().getMessageContext());
        }
        super.doAppend(eventObject);
    }

    @Override
    public FilterReply getFilterChainDecision(ILoggingEvent event) {
        databaseLogFilterHelper.updateFilters();
        return super.getFilterChainDecision(event);
    }

    protected boolean shouldEnable() {
        return systemConfigurationProperties.isSmtpEnabled()
                && configurationProperties.getLogging() != null
                && StringUtils.isNotBlank(configurationProperties.getLogging().getFrom())
                && CollectionUtils.isNotEmpty(
                        configurationProperties.getLogging().getTo())
                && StringUtils.isNotBlank(configurationProperties.getLogging().getSubject());
    }

    @Override
    public void afterPropertiesSet() throws Exception {
        if (LoggerFactory.getILoggerFactory() instanceof LoggerContext loggerContext) {
            if (shouldEnable()) {
                LOGGER.info("registering email logback appender");
                setContext(loggerContext);
                setName(getClass().getSimpleName());
                setEvaluator(new EventEvaluatorBase<>() {
                    @Override
                    public boolean evaluate(ILoggingEvent event) {
                        return true;
                    }
                });
                eventEvaluator.start();

                setSmtpHost(mailProperties.getHost());
                if (mailProperties.getPort() != null) {
                    setSmtpPort(mailProperties.getPort());
                }

                setUsername(mailProperties.getUsername());
                setPassword(mailProperties.getPassword());
                setFrom(configurationProperties.getLogging().getFrom());
                setSubject(configurationProperties.getLogging().getSubject());
                configurationProperties.getLogging().getTo().forEach(this::addTo);

                HTMLLayout htmlLayout = new HTMLLayout();
                htmlLayout.setContext(loggerContext);
                htmlLayout.start();

                TemplateLayout templateLayout = new TemplateLayout(springTemplateEngine);
                templateLayout.setContext(loggerContext);
                templateLayout.start();

                setLayout(templateLayout);

                ThresholdFilter thresholdFilter = new ThresholdFilter();
                thresholdFilter.setLevel(Level.OFF.levelStr);
                thresholdFilter.start();

                ch.qos.logback.classic.Logger rootLogger = loggerContext.getLogger(Logger.ROOT_LOGGER_NAME);
                start();
                rootLogger.addAppender(this);

                synchronized (this) {
                    databaseLogFilterHelper = new DatabaseLogFilterHelper(
                            logFilterCacheService, this, LogFilterType.EMAIL, thresholdFilter);
                    databaseLogFilterHelper.updateFilters();
                }
            } else {
                LOGGER.info("email logback appender will not be enabled, check required configuration properties");
            }
        } else {
            throw new IllegalStateException("logger context ist not of type logback");
        }
    }
}
