package com.evolvsys.nexuse2e.core.keyvalue.persistence.repository;

import com.evolvsys.app.core.repository.ListProviderRepository;
import com.evolvsys.nexuse2e.core.keyvalue.persistence.model.KeyValuePropertyEntity;
import com.evolvsys.nexuse2e.core.keyvalue.persistence.model.KeyValuePropertyFilter;
import com.evolvsys.nexuse2e.core.keyvalue.persistence.model.QKeyValuePropertyEntity;
import com.querydsl.core.types.dsl.BooleanExpression;
import java.util.Optional;
import org.springframework.data.domain.Page;
import org.springframework.data.domain.Pageable;
import org.springframework.data.querydsl.QuerydslPredicateExecutor;

/**
 * @author SvenBarden
 * @since 17.06.2024
 */
public interface KeyValuePropertyRepository
        extends ListProviderRepository<KeyValuePropertyEntity, Long, KeyValuePropertyFilter>,
                QuerydslPredicateExecutor<KeyValuePropertyEntity> {
    Optional<KeyValuePropertyEntity> findByCategoryAndKey(String category, String key);

    @Override
    default Page<KeyValuePropertyEntity> findAllToRead(
            Pageable pageable, KeyValuePropertyFilter filter, BooleanExpression securityPredicate) {
        return findAll(securityPredicate.and(QKeyValuePropertyEntity.keyValuePropertyEntity.filter(filter)), pageable);
    }
}
