package com.evolvsys.nexuse2e.core.keyvalue.persistence.repository;

import com.evolvsys.nexuse2e.core.keyvalue.persistence.model.KeyValuePropertyEntity;
import com.evolvsys.nexuse2e.core.keyvalue.persistence.model.KeyValuePropertyFilter;
import com.evolvsys.nexuse2e.core.keyvalue.persistence.model.QKeyValuePropertyEntity;
import com.querydsl.core.BooleanBuilder;
import com.querydsl.core.annotations.QueryDelegate;
import com.querydsl.core.types.Predicate;
import org.apache.commons.lang3.StringUtils;
import org.springframework.lang.Nullable;

/**
 * @author SvenBarden
 * @since 17.06.2024
 */
public final class KeyValuePropertyQueryDelegate {
    private KeyValuePropertyQueryDelegate() {}

    @QueryDelegate(KeyValuePropertyEntity.class)
    public static Predicate filter(
            QKeyValuePropertyEntity keyValuePropertyEntity, @Nullable KeyValuePropertyFilter keyValuePropertyFilter) {
        BooleanBuilder builder = new BooleanBuilder();
        if (keyValuePropertyFilter != null) {
            if (StringUtils.isNotBlank(keyValuePropertyFilter.getCategory())) {
                builder.and(keyValuePropertyEntity.category.contains(keyValuePropertyFilter.getCategory()));
            }
            if (StringUtils.isNotBlank(keyValuePropertyFilter.getKey())) {
                builder.and(keyValuePropertyEntity.key.contains(keyValuePropertyFilter.getKey()));
            }
            if (StringUtils.isNotBlank(keyValuePropertyFilter.getValue())) {
                builder.and(keyValuePropertyEntity.value.contains(keyValuePropertyFilter.getValue()));
            }
        }
        return builder.getValue();
    }
}
