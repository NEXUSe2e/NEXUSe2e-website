package com.evolvsys.nexuse2e.core.service.http.client.rest;

import com.evolvsys.nexuse2e.core.exception.NexusException;
import com.evolvsys.nexuse2e.core.service.http.WebClientContext;
import java.io.IOException;
import java.security.KeyManagementException;
import java.security.KeyStoreException;
import java.security.NoSuchAlgorithmException;
import java.security.UnrecoverableKeyException;
import java.security.cert.CertificateException;
import java.time.Duration;
import lombok.experimental.UtilityClass;
import org.apache.commons.lang3.ObjectUtils;
import org.apache.hc.client5.http.config.RequestConfig;
import org.apache.hc.client5.http.impl.classic.CloseableHttpClient;
import org.apache.hc.client5.http.impl.classic.HttpClientBuilder;
import org.apache.hc.client5.http.impl.classic.HttpClients;
import org.apache.hc.core5.util.Timeout;

/**
 * @author SvenBarden
 * @since 23.07.2024
 */
@UtilityClass
class RestHttpClientFactory {

    private static final Duration DEFAULT_RESPONSE_TIMEOUT = Duration.ofSeconds(60);

    public static CloseableHttpClient createHttpClient(WebClientContext context)
            throws NoSuchAlgorithmException, KeyStoreException, KeyManagementException, UnrecoverableKeyException,
                    CertificateException, IOException, NexusException {
        HttpClientBuilder httpClientBuilder = HttpClients.custom()
                .setDefaultRequestConfig(RequestConfig.custom()
                        .setResponseTimeout(Timeout.of(
                                ObjectUtils.defaultIfNull(context.getResponseTimeout(), DEFAULT_RESPONSE_TIMEOUT)))
                        .build());

        if (context.isSsl()) {
            httpClientBuilder.setConnectionManager(RestSslConnectionFactory.createSslConnection(context));
        }

        return httpClientBuilder.build();
    }
}
