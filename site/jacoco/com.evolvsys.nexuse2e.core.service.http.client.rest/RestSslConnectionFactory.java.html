package com.evolvsys.nexuse2e.core.service.http.client.rest;

import com.evolvsys.nexuse2e.core.exception.NexusException;
import com.evolvsys.nexuse2e.core.service.http.WebClientContext;
import com.evolvsys.nexuse2e.core.service.http.client.HttpContextSslUtils;
import java.io.IOException;
import java.security.*;
import java.security.cert.CertificateException;
import java.security.cert.X509Certificate;
import java.time.Duration;
import java.util.Collection;
import java.util.List;
import javax.net.ssl.SSLContext;
import lombok.experimental.UtilityClass;
import org.apache.commons.collections4.CollectionUtils;
import org.apache.commons.lang3.ObjectUtils;
import org.apache.hc.client5.http.config.ConnectionConfig;
import org.apache.hc.client5.http.impl.io.PoolingHttpClientConnectionManager;
import org.apache.hc.client5.http.impl.io.PoolingHttpClientConnectionManagerBuilder;
import org.apache.hc.client5.http.ssl.DefaultClientTlsStrategy;
import org.apache.hc.client5.http.ssl.HttpsSupport;
import org.apache.hc.client5.http.ssl.NoopHostnameVerifier;
import org.apache.hc.core5.reactor.ssl.SSLBufferMode;
import org.apache.hc.core5.ssl.SSLContextBuilder;
import org.apache.hc.core5.ssl.SSLContexts;
import org.apache.hc.core5.util.Timeout;

/**
 * @author SvenBarden
 * @since 23.07.2024
 */
@UtilityClass
class RestSslConnectionFactory {

    private static final Duration DEFAULT_CONNECTION_TIMEOUT = Duration.ofSeconds(30);

    private KeyStore buildKeystore(List<X509Certificate> certificates)
            throws KeyStoreException, CertificateException, IOException, NoSuchAlgorithmException {
        KeyStore ks = KeyStore.getInstance(KeyStore.getDefaultType());
        ks.load(null, null);

        for (int i = 0; i < certificates.size(); i++) {
            X509Certificate certificate = certificates.get(i);
            String alias = Integer.toString(i + 1);
            ks.setCertificateEntry(alias, certificate);
        }

        return ks;
    }

    public static PoolingHttpClientConnectionManager createSslConnection(WebClientContext context)
            throws KeyStoreException, NoSuchAlgorithmException, KeyManagementException, NexusException,
                    CertificateException, IOException, UnrecoverableKeyException {
        KeyStore trustStore;
        if (context.getTrustStore() == null) {
            trustStore = buildKeystore(HttpContextSslUtils.getTrustedCertificates(context));
        } else {
            trustStore = context.getTrustStore();
        }

        SSLContextBuilder sslContextBuilder = SSLContexts.custom().loadTrustMaterial(trustStore, null);

        if (context.getKeyStore() != null) {
            sslContextBuilder.loadKeyMaterial(
                    context.getKeyStore(), context.getKeyStorePassword().toCharArray());
        }

        SSLContext sslContext = sslContextBuilder.build();

        Collection<String> protocols = HttpContextSslUtils.getProtocols(context);
        String[] finalProtocols;
        if (CollectionUtils.isEmpty(protocols)) {
            finalProtocols = null;
        } else {
            finalProtocols = protocols.toArray(new String[0]);
        }

        Collection<String> ciphers = HttpContextSslUtils.getCiphers(context);
        String[] finalCiphers;
        if (CollectionUtils.isEmpty(ciphers)) {
            finalCiphers = null;
        } else {
            finalCiphers = ciphers.toArray(new String[0]);
        }

        DefaultClientTlsStrategy tlsStrategy = new DefaultClientTlsStrategy(
                sslContext,
                finalProtocols,
                finalCiphers,
                SSLBufferMode.STATIC,
                context.isDisableHostNameVerification()
                        ? NoopHostnameVerifier.INSTANCE
                        : HttpsSupport.getDefaultHostnameVerifier());

        return PoolingHttpClientConnectionManagerBuilder.create()
                .setTlsSocketStrategy(tlsStrategy)
                .setDefaultConnectionConfig(ConnectionConfig.custom()
                        .setConnectTimeout(Timeout.of(
                                ObjectUtils.defaultIfNull(context.getConnectionTimeout(), DEFAULT_CONNECTION_TIMEOUT)))
                        .setSocketTimeout(Timeout.of(
                                ObjectUtils.defaultIfNull(context.getConnectionTimeout(), DEFAULT_CONNECTION_TIMEOUT)))
                        .build())
                .build();
    }
}
