package com.evolvsys.nexuse2e.core.parameter.service;

import com.evolvsys.nexuse2e.core.exception.NexusError;
import com.evolvsys.nexuse2e.core.exception.NexusRuntimeException;
import com.evolvsys.nexuse2e.core.parameter.ParameterType;
import com.evolvsys.nexuse2e.core.parameter.converter.TypeConverter;
import java.util.ArrayList;
import java.util.List;
import lombok.extern.slf4j.Slf4j;
import org.springframework.stereotype.Component;
import org.springframework.util.ClassUtils;

/**
 * @author Florian Drees
 * @since 11.01.2024
 */
@Component
@Slf4j
public class ParameterConverterRegistry {
    private final List<TypeConverter<?, ?>> typeConverters = new ArrayList<>();

    @SuppressWarnings("unchecked")
    public <EntityTypeT, DtoTypeT> TypeConverter<EntityTypeT, DtoTypeT> getTypeConverter(ParameterType type) {
        return (TypeConverter<EntityTypeT, DtoTypeT>) typeConverters.stream()
                .filter(aTypeConverter ->
                        ClassUtils.getUserClass(aTypeConverter).equals(type.getConverterClass()))
                .findAny()
                .orElseThrow(() -> new NexusRuntimeException(NexusError.builder()
                        .message("Could not find converter for parameter type " + type)
                        .extension("type", type)
                        .build()));
    }

    void register(TypeConverter<?, ?> typeConverter) {
        typeConverters.add(typeConverter);
    }
}
