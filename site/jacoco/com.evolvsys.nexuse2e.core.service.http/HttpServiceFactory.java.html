package com.evolvsys.nexuse2e.core.service.http;

import com.evolvsys.nexuse2e.core.component.ComponentRegistry;
import com.evolvsys.nexuse2e.core.exception.NexusError;
import com.evolvsys.nexuse2e.core.exception.NexusException;
import com.evolvsys.nexuse2e.core.model.entity.ServiceEntity;
import com.evolvsys.nexuse2e.core.service.BaseInflowNexusService;
import com.evolvsys.nexuse2e.core.service.ServiceFactory;
import com.evolvsys.nexuse2e.core.service.ServiceType;
import com.evolvsys.nexuse2e.core.service.repository.ServiceRepository;
import java.util.List;
import lombok.RequiredArgsConstructor;
import org.apache.commons.collections4.CollectionUtils;
import org.springframework.stereotype.Component;

/**
 * @author SvenBarden
 * @since 21.02.2024
 */
@Component
@RequiredArgsConstructor
class HttpServiceFactory {
    private final ServiceRepository serviceRepository;
    private final ComponentRegistry componentRegistry;
    private final ServiceFactory serviceFactory;

    private boolean filterHttpService(ServiceEntity serviceEntity) {
        return componentRegistry
                .getComponent(serviceEntity.getComponentId())
                .map(component -> HttpReceiver.class.isAssignableFrom(component.getLoadedClass())
                        && BaseInflowNexusService.class.isAssignableFrom(component.getLoadedClass()))
                .orElse(Boolean.FALSE);
    }

    public HttpReceiver get(String path) throws NexusException {
        List<ServiceEntity> services = serviceRepository.getByTypeAndParameterValue(
                ServiceType.INFLOW_SERVICE, HttpReceiver.URL_PATH_PARAM_NAME, path);

        if (CollectionUtils.isEmpty(services)) {
            throw new NexusException(NexusError.builder()
                    .message("No service found for url path")
                    .extension("path", path)
                    .build());
        }

        List<ServiceEntity> filteredServices =
                services.stream().filter(this::filterHttpService).toList();
        if (CollectionUtils.isEmpty(filteredServices)) {
            throw new NexusException(NexusError.builder()
                    .message("Services found for url path missing HttpReceiver or NexusService dependency")
                    .extension("path", path)
                    .build());
        }

        if (filteredServices.size() > 1) {
            throw new NexusException(NexusError.builder()
                    .message("Found more than one HttpReceiver for the path")
                    .extension("path", path)
                    .build());
        }

        ServiceEntity serviceEntity = filteredServices.getFirst();

        return serviceFactory.get(serviceEntity.getId());
    }
}
