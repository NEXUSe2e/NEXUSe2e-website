package com.evolvsys.nexuse2e.core.service.http;

import com.evolvsys.app.core.exception.Error;
import com.evolvsys.nexuse2e.core.exception.NexusException;
import com.evolvsys.nexuse2e.core.utils.ErrorUtils;
import edu.umd.cs.findbugs.annotations.SuppressFBWarnings;
import jakarta.servlet.http.HttpServletRequest;
import jakarta.servlet.http.HttpServletResponse;
import java.io.IOException;
import java.nio.charset.StandardCharsets;
import java.util.Enumeration;
import java.util.Map;
import java.util.regex.Pattern;
import lombok.RequiredArgsConstructor;
import lombok.extern.slf4j.Slf4j;
import one.util.streamex.EntryStream;
import org.apache.commons.collections4.MapUtils;
import org.apache.commons.lang3.ArrayUtils;
import org.apache.commons.lang3.StringUtils;
import org.springframework.stereotype.Service;

/**
 * @author SvenBarden
 * @since 23.10.2023
 */
@Service
@RequiredArgsConstructor
@Slf4j
public class HttpServiceHandler {
    public static final String SERVICE_PREFIX = "/service/";
    private static final Pattern BASE_PATTERN = Pattern.compile("/service/([^/]+).*");
    private final HttpServiceFactory httpServiceFactory;

    private String serializeError(Error error) {
        StringBuilder s = new StringBuilder();
        s.append(error.getMessage());
        for (Map.Entry<String, Object> entry : error.getExtensions().entrySet()) {
            s.append('\n').append(entry.getKey()).append(": ").append(entry.getValue());
        }
        return s.toString();
    }

    private void error(String message, HttpServletResponse response) {
        try {
            response.setStatus(HttpServletResponse.SC_BAD_REQUEST);
            response.getOutputStream().write(message.getBytes(StandardCharsets.UTF_8));
        } catch (IOException e) {
            response.setStatus(HttpServletResponse.SC_INTERNAL_SERVER_ERROR);
            LOGGER.error(
                    "could not write message to response output stream: {}, {}", message, ErrorUtils.aggregate(e), e);
        }
    }

    private String getRequestInfo(HttpServletRequest httpServletRequest) {
        StringBuilder requestInfo = new StringBuilder(24);
        requestInfo
                .append("ip: ")
                .append(httpServletRequest.getRemoteAddr())
                .append(", method: ")
                .append(httpServletRequest.getMethod());
        Map<String, String[]> parameterMap = httpServletRequest.getParameterMap();
        if (MapUtils.isNotEmpty(parameterMap)) {
            requestInfo
                    .append(", params: ")
                    .append(EntryStream.of(parameterMap)
                            .map(stringEntry ->
                                    stringEntry.getKey() + ": [" + StringUtils.join(stringEntry.getValue(), ", ") + "]")
                            .joining(", "));
        }

        return requestInfo.toString();
    }

    private void logAdditionalInfos(HttpServletRequest httpServletRequest) {
        if (LOGGER.isTraceEnabled()) {
            LOGGER.trace("client ip {}", httpServletRequest.getRemoteAddr());
            LOGGER.trace("request method {}", httpServletRequest.getMethod());

            LOGGER.trace("request headers start");
            Enumeration<String> headerNames = httpServletRequest.getHeaderNames();
            if (headerNames != null) {
                while (headerNames.hasMoreElements()) {
                    String headerName = headerNames.nextElement();
                    Enumeration<String> values = httpServletRequest.getHeaders(headerName);
                    if (values == null) {
                        LOGGER.trace("header {}: ", headerNames);
                    } else {
                        while (values.hasMoreElements()) {
                            LOGGER.trace("header {}: {}", headerName, values.nextElement());
                        }
                    }
                }
            }
            LOGGER.trace("request headers finished");

            LOGGER.trace("request parameters start");
            Map<String, String[]> parameterMap = httpServletRequest.getParameterMap();
            if (MapUtils.isNotEmpty(parameterMap)) {
                for (Map.Entry<String, String[]> entry : parameterMap.entrySet()) {
                    String[] values = entry.getValue();
                    if (ArrayUtils.isEmpty(values)) {
                        LOGGER.trace("param {}: []", entry.getKey());
                    } else {
                        for (String value : values) {
                            LOGGER.trace("param {}: {}", entry.getKey(), value);
                        }
                    }
                }
            }
            LOGGER.trace("request parameters finished");
        }
    }

    @SuppressFBWarnings("EXS_EXCEPTION_SOFTENING_NO_CONSTRAINTS")
    public void handle(HttpServletRequest httpServletRequest, HttpServletResponse httpServletResponse) {
        String servletPath = httpServletRequest.getServletPath();

        LOGGER.trace("got request at {}", servletPath);

        logAdditionalInfos(httpServletRequest);

        String servicePath = BASE_PATTERN.matcher(servletPath).replaceAll("$1");

        LOGGER.debug("looking for http service with path {}", servicePath);

        try {
            HttpReceiver service = httpServiceFactory.get(servicePath);
            service.setHttpServletResponse(httpServletResponse);
            service.setHttpServletRequest(httpServletRequest);

            try {
                service.execute();
                int status = httpServletResponse.getStatus();
                if (status >= 200 && status < 300) {
                    LOGGER.debug("response status {} in 2xx range, request was successful", status);
                } else {
                    LOGGER.warn(
                            "response status {} not in 2xx range, request {} was unsuccessful",
                            status,
                            getRequestInfo(httpServletRequest));
                }
            } catch (NexusException e) {
                error(serializeError(e.getError()), httpServletResponse);
                LOGGER.error(
                        "error calling http receiver for path {} with request {}: {}",
                        servicePath,
                        getRequestInfo(httpServletRequest),
                        ErrorUtils.aggregate(e),
                        e);
            } catch (Throwable e) {
                error("Error calling http receiver: " + e.getMessage(), httpServletResponse);
                LOGGER.error(
                        "error calling http receiver for path {} with request {}: {}",
                        servicePath,
                        getRequestInfo(httpServletRequest),
                        ErrorUtils.aggregate(e),
                        e);
            }
        } catch (NexusException e) {
            error(serializeError(e.getError()), httpServletResponse);
            LOGGER.error(
                    "error creating http receiver for path {} with request {}: {}",
                    servicePath,
                    getRequestInfo(httpServletRequest),
                    ErrorUtils.aggregate(e),
                    e);
        } catch (Throwable e) {
            error("Error creating http receiver: " + e.getMessage(), httpServletResponse);
            LOGGER.error(
                    "error creating http receiver for path {} with request {}: {}",
                    servicePath,
                    getRequestInfo(httpServletRequest),
                    ErrorUtils.aggregate(e),
                    e);
        }
    }
}
