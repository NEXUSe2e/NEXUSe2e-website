package com.evolvsys.nexuse2e.core.service.http;

import com.evolvsys.nexuse2e.core.component.InflowService;
import com.evolvsys.nexuse2e.core.parameter.ParameterDefinition;
import com.evolvsys.nexuse2e.core.parameter.ParameterType;
import jakarta.servlet.http.HttpServletRequest;
import jakarta.servlet.http.HttpServletResponse;

/**
 * @author SvenBarden
 * @since 21.02.2024
 */
public interface HttpReceiver extends InflowService {
    String URL_PATH_PARAM_NAME = "http_url_path";
    ParameterDefinition URL_PATH_PARAM_DEFINITION = ParameterDefinition.builder()
            .name(URL_PATH_PARAM_NAME)
            .displayName("Url path")
            .description(
                    "Context path for this http service. The final url will be [NEXUSe2e Base Url]/service/[this value]. Only Alphanumerical characters are allowed.")
            .type(ParameterType.STRING)
            .required(true)
            .build();

    void setHttpServletRequest(HttpServletRequest httpServletRequest);

    void setHttpServletResponse(HttpServletResponse httpServletResponse);
}
