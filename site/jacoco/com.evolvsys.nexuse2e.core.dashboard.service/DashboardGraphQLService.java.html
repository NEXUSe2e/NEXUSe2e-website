package com.evolvsys.nexuse2e.core.dashboard.service;

import com.evolvsys.app.core.graphql.annotation.GraphQLService;
import com.evolvsys.nexuse2e.core.dashboard.DashboardFilter;
import com.evolvsys.nexuse2e.core.dashboard.model.ConversationStatusCountResultDto;
import com.evolvsys.nexuse2e.core.dashboard.model.LatestMessagesOfChoreographyDto;
import com.evolvsys.nexuse2e.core.dashboard.model.LatestMessagesOfPartnerDto;
import com.evolvsys.nexuse2e.core.dashboard.model.MessageStatusCountResultDto;
import io.leangen.graphql.annotations.GraphQLNonNull;
import io.leangen.graphql.annotations.GraphQLQuery;
import java.util.List;
import lombok.RequiredArgsConstructor;
import org.springframework.lang.Nullable;

/**
 * @author Florian Drees
 * @since 21.03.2024
 */
@GraphQLService
@RequiredArgsConstructor
public class DashboardGraphQLService {
    private final DashboardService dashboardService;

    @GraphQLQuery(name = "dashboard_getMessageCountByStatus", description = "Get count of messages by status")
    public List<@GraphQLNonNull MessageStatusCountResultDto> getMessageCountByStatus(
            @Nullable DashboardFilter dashboardFilter) {
        return dashboardService.getMessageCountByStatus(dashboardFilter);
    }

    @GraphQLQuery(name = "dashboard_getConversationCountByStatus", description = "Get count of conversations by status")
    public List<@GraphQLNonNull ConversationStatusCountResultDto> getConversationCountByStatus(
            @Nullable DashboardFilter dashboardFilter) {
        return dashboardService.getConversationCountByStatus(dashboardFilter);
    }

    @GraphQLQuery(
            name = "dashboard_getLatestMessagesOfChoreographies",
            description = "Get the time of the latest inbound and outbound message per choreography")
    public List<@GraphQLNonNull LatestMessagesOfChoreographyDto> getLatestMessagesOfChoreographies(
            @Nullable DashboardFilter dashboardFilter) {
        return dashboardService.getLatestMessagesOfChoreographies(dashboardFilter);
    }

    @GraphQLQuery(
            name = "dashboard_getLatestMessagesOfPartners",
            description = "Get the time of the latest inbound and outbound message per partner")
    public List<@GraphQLNonNull LatestMessagesOfPartnerDto> getLatestMessagesOfPartners(
            @Nullable DashboardFilter dashboardFilter) {
        return dashboardService.getLatestMessagesOfPartners(dashboardFilter);
    }
}
