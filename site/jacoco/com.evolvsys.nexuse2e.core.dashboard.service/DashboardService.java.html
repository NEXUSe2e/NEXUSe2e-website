package com.evolvsys.nexuse2e.core.dashboard.service;

import com.evolvsys.nexuse2e.core.dashboard.DashboardFilter;
import com.evolvsys.nexuse2e.core.dashboard.model.ConversationStatusCountResultDto;
import com.evolvsys.nexuse2e.core.dashboard.model.LatestMessagesOfChoreographyDto;
import com.evolvsys.nexuse2e.core.dashboard.model.LatestMessagesOfPartnerDto;
import com.evolvsys.nexuse2e.core.dashboard.model.MessageStatusCountResultDto;
import com.evolvsys.nexuse2e.core.dashboard.repository.DashboardRepository;
import java.util.List;
import lombok.RequiredArgsConstructor;
import org.springframework.lang.Nullable;
import org.springframework.stereotype.Service;

/**
 * @author Florian Drees
 * @since 21.03.2024
 */
@Service
@RequiredArgsConstructor
public class DashboardService {
    private final DashboardRepository dashboardRepository;

    public List<MessageStatusCountResultDto> getMessageCountByStatus(@Nullable DashboardFilter dashboardFilter) {
        return dashboardRepository.getMessageCountByStatus(dashboardFilter);
    }

    public List<ConversationStatusCountResultDto> getConversationCountByStatus(
            @Nullable DashboardFilter dashboardFilter) {
        return dashboardRepository.getConversationCountByStatus(dashboardFilter);
    }

    public List<LatestMessagesOfChoreographyDto> getLatestMessagesOfChoreographies(
            @Nullable DashboardFilter dashboardFilter) {
        return dashboardRepository.getLatestMessagesOfChoreographies(dashboardFilter);
    }

    public List<LatestMessagesOfPartnerDto> getLatestMessagesOfPartners(@Nullable DashboardFilter dashboardFilter) {
        return dashboardRepository.getLatestMessagesOfPartners(dashboardFilter);
    }
}
