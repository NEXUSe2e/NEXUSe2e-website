package com.evolvsys.nexuse2e.core.partner.validation;

import com.evolvsys.nexuse2e.core.partner.model.PartnerDto;
import com.evolvsys.nexuse2e.core.partner.repository.PartnerRepository;
import jakarta.validation.ConstraintValidator;
import jakarta.validation.ConstraintValidatorContext;
import java.util.Optional;
import lombok.RequiredArgsConstructor;
import org.springframework.stereotype.Component;

/**
 * <p>Validation of the (alphanumeric) id of a partner to ensure its unique across all partners.</p>
 *
 * @author Florian Drees
 * @since 17.01.2024
 */
@Component
@RequiredArgsConstructor
public class PartnerIdConstraintValidator implements ConstraintValidator<ValidPartnerId, PartnerDto> {
    private final PartnerRepository partnerRepository;

    @Override
    public boolean isValid(PartnerDto value, ConstraintValidatorContext context) {
        return Optional.ofNullable(value.getId())
                .map(id -> !partnerRepository.existsByPartnerIdAndIdNot(value.getPartnerId(), id))
                .orElseGet(() -> !partnerRepository.existsByPartnerId(value.getPartnerId()));
    }
}
