package com.evolvsys.nexuse2e.core.partner.validation;

import com.evolvsys.nexuse2e.core.partner.model.PartnerConnectionDto;
import com.evolvsys.nexuse2e.core.partner.repository.PartnerConnectionRepository;
import jakarta.validation.ConstraintValidator;
import jakarta.validation.ConstraintValidatorContext;
import java.util.Optional;
import lombok.RequiredArgsConstructor;
import org.springframework.stereotype.Component;

/**
 * <p>Validation of the name of the partner connection to be unique across all connections of a partner.</p>
 *
 * @author Florian Drees
 * @since 19.01.2024
 */
@Component
@RequiredArgsConstructor
public class ConnectionNameConstraintValidator
        implements ConstraintValidator<ValidPartnerConnectionName, PartnerConnectionDto> {
    private final PartnerConnectionRepository partnerConnectionRepository;

    @Override
    public boolean isValid(PartnerConnectionDto value, ConstraintValidatorContext context) {
        return Optional.ofNullable(value.getId())
                .map(id -> !partnerConnectionRepository.existsByConnectionIdAndPartner_IdAndIdNot(
                        value.getConnectionId(), value.getPartnerId(), id))
                .orElseGet(() -> !partnerConnectionRepository.existsByConnectionIdAndPartner_Id(
                        value.getConnectionId(), value.getPartnerId()));
    }
}
