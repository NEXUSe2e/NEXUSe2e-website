package com.evolvsys.nexuse2e.core.system;

import com.evolvsys.nexuse2e.core.config.NEXUSe2eConfigurationProperties;
import java.security.Security;
import java.util.Arrays;
import java.util.Set;
import java.util.stream.Collectors;
import lombok.RequiredArgsConstructor;
import lombok.extern.slf4j.Slf4j;
import org.springframework.beans.factory.InitializingBean;
import org.springframework.stereotype.Component;

/**
 * @author SvenBarden
 * @since 29.02.2024
 */
@Component
@RequiredArgsConstructor
@Slf4j
class SystemInitializer implements InitializingBean {
    private final NEXUSe2eConfigurationProperties configurationProperties;

    @Override
    public void afterPropertiesSet() throws Exception {
        if (configurationProperties.isEnableTLSv1() || configurationProperties.isEnableTLSv1_1()) {
            String defaultProperties = Security.getProperty("jdk.tls.disabledAlgorithms");

            Set<String> values =
                    Arrays.stream(defaultProperties.split("\\s*,\\s*")).collect(Collectors.toSet());

            if (configurationProperties.isEnableTLSv1()) {
                values.remove("TLSv1");
            }
            if (configurationProperties.isEnableTLSv1_1()) {
                values.remove("TLSv1.1");
            }

            String newValue = String.join(", ", values);

            LOGGER.info("changed disabledAlgorithms for non standalone tomcat {} to {}", defaultProperties, newValue);

            Security.setProperty("jdk.tls.disabledAlgorithms", newValue);
        }
    }
}
