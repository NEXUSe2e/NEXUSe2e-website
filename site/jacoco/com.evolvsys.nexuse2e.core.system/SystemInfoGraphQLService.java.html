package com.evolvsys.nexuse2e.core.system;

import com.evolvsys.app.core.graphql.annotation.GraphQLService;
import com.evolvsys.nexuse2e.core.system.model.SystemInfoDto;
import io.leangen.graphql.annotations.GraphQLNonNull;
import io.leangen.graphql.annotations.GraphQLQuery;
import lombok.RequiredArgsConstructor;

/**
 * @author SvenBarden
 * @since 15.12.2023
 */
@GraphQLService
@RequiredArgsConstructor
public class SystemInfoGraphQLService {
    private final SystemInfoService systemInfoService;

    @GraphQLQuery(name = "systemInfo_get")
    @GraphQLNonNull
    public SystemInfoDto getSystemInfo() {
        return systemInfoService.getSystemInfo();
    }
}
