package com.evolvsys.nexuse2e.core.config;

import jakarta.validation.Valid;
import jakarta.validation.constraints.NotBlank;
import jakarta.validation.constraints.NotNull;
import jakarta.validation.constraints.Positive;
import java.time.Duration;
import java.util.HashSet;
import java.util.List;
import java.util.Set;
import lombok.Data;
import lombok.ToString;
import lombok.extern.slf4j.Slf4j;
import org.hibernate.validator.constraints.time.DurationMin;
import org.springframework.beans.factory.InitializingBean;
import org.springframework.boot.context.properties.ConfigurationProperties;
import org.springframework.core.io.ClassPathResource;
import org.springframework.core.io.Resource;
import org.springframework.stereotype.Component;
import org.springframework.util.unit.DataSize;
import org.springframework.validation.annotation.Validated;

/**
 * @author SvenBarden
 * @since 23.11.2023
 */
@Component
@Validated
@ConfigurationProperties(prefix = "nexuse2e.core")
@Data
@Slf4j
public class NEXUSe2eConfigurationProperties implements InitializingBean {
    private Set<String> additionalComponentPackages;

    private String caCertsPath;

    @NotNull
    private Duration defaultIdleConversationsDuration = Duration.ofMinutes(10);

    @NotBlank
    @ToString.Exclude
    private String caCertsPassword = "changeit";

    @NotNull
    private Resource defaultConfig = new ClassPathResource("config.xml");

    @NotNull
    private String defaultManualMessageFiles = "classpath:manual-message";

    private Set<String> infoIgnoreProperties;

    @NotNull
    private DataSize inMemoryPayloadThreshold = DataSize.ofKilobytes(10);

    /**
     * Enables TSLv1 in embedded tomcat. Does NOT work if deployed as war file.
     */
    private boolean enableTLSv1;
    /**
     * Enables TSLv1.1 in embedded tomcat. Does NOT work if deployed as war file.
     */
    @SuppressWarnings("PMD.FieldNamingConventions")
    private boolean enableTLSv1_1;

    private boolean requireHttpsForUi;

    @Valid
    @NotNull
    private MessageQueueConfiguration messageQueue = new MessageQueueConfiguration();

    private Resource logFileDirectory;

    @Valid
    private Logging logging = new Logging();

    private Set<String> timeZones = new HashSet<>();

    @Valid
    private IpConfiguration ipConfiguration = new IpConfiguration();

    @Valid
    private Ebxml ebxml = new Ebxml();

    @Valid
    private TempFileCleanup tempFileCleanup = new TempFileCleanup();

    @Override
    public void afterPropertiesSet() throws Exception {
        LOGGER.info("loaded NEXUSe2e config {}", this);
    }

    @Data
    public static class MessageQueueConfiguration {
        /**
         * Interval between scheduler runs for requeueing messages
         */
        @NotNull
        @DurationMin(seconds = 1)
        private Duration schedulingInterval = Duration.ofMinutes(2);

        /**
         * Default retry interval for messages if the connection has no retry interval
         */
        @NotNull
        @DurationMin(seconds = 30)
        private Duration defaultRetryInterval = Duration.ofMinutes(5);

        /**
         * Delay until the trigger starts the message queue scheduler
         */
        @NotNull
        @DurationMin(seconds = 1)
        private Duration queueStartupDelay = Duration.ofMinutes(2);

        /**
         * Number of messages to requeue with each scheduler run
         */
        @Positive
        private int messagesPerRun = 10;

        private boolean singleThread = true;
    }

    @Data
    public static class Logging {
        private String from;

        private List<String> to;

        private String subject = "NEXUSe2e - error";
    }

    @Data
    public static class IpConfiguration {
        private List<String> validIpv4AddressRanges;
        private List<String> validIpv6AddressRanges;
    }

    @Data
    public static class Ebxml {
        private List<String> nonBinaryMimeTypes = List.of();
    }

    @Data
    public static class TempFileCleanup {
        private String cron = "0 0 3 * * *"; // Default: Cleanup every day at 3 AM
        private Duration keepDuration = Duration.ofDays(1); // Default: Keep temp files for 1 day
    }
}
