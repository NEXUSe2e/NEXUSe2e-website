package com.evolvsys.nexuse2e.core.config;

import com.evolvsys.nexuse2e.core.config.model.ConfigValidationResult;
import com.evolvsys.nexuse2e.core.config.model.NexusConfig;
import com.evolvsys.nexuse2e.core.config.validation.ValidationContext;
import com.evolvsys.nexuse2e.core.config.validation.ValidationContextHolder;
import com.evolvsys.nexuse2e.core.config.validation.ValidationLevel;
import jakarta.validation.ConstraintViolation;
import jakarta.validation.Payload;
import jakarta.validation.ValidatorFactory;
import java.util.Set;
import lombok.RequiredArgsConstructor;
import lombok.extern.slf4j.Slf4j;
import org.springframework.stereotype.Service;

/**
 * @author SvenBarden
 * @since 25.04.2024
 */
@Service
@RequiredArgsConstructor
@Slf4j
public class ConfigValidatorService {
    private final ValidatorFactory validatorFactory;

    public ConfigValidationResult validate(NexusConfig nexusConfig) {
        ValidationContextHolder.setContext(
                ValidationContext.builder().config(nexusConfig).build());

        LOGGER.info("validating nexus config");

        Set<ConstraintViolation<NexusConfig>> constraintViolations =
                validatorFactory.getValidator().validate(nexusConfig);

        ValidationContextHolder.clearContext();

        int errorCount = 0;
        int warningCount = 0;
        int fatalCount = 0;

        ConfigValidationResult.ConfigValidationResultBuilder resultBuilder = ConfigValidationResult.builder();

        for (ConstraintViolation<NexusConfig> constraintViolation : constraintViolations) {
            Set<Class<? extends Payload>> payload =
                    constraintViolation.getConstraintDescriptor().getPayload();

            if (payload.contains(ValidationLevel.Fatal.class)) {
                LOGGER.error(
                        "fatal validation error while parsing config at {}: {}",
                        constraintViolation.getPropertyPath(),
                        constraintViolation.getMessage());
                fatalCount++;
            } else if (payload.contains(ValidationLevel.Error.class)) {
                LOGGER.error(
                        "validation error while parsing config at {}: {}",
                        constraintViolation.getPropertyPath(),
                        constraintViolation.getMessage());
                errorCount++;
            } else {
                LOGGER.warn(
                        "validation warning while parsing config at {}: {}",
                        constraintViolation.getPropertyPath(),
                        constraintViolation.getMessage());
                warningCount++;
            }
            resultBuilder.message(constraintViolation.getPropertyPath() + ": " + constraintViolation.getMessage());
        }

        return resultBuilder
                .warningCount(warningCount)
                .errorCount(errorCount)
                .fatalCount(fatalCount)
                .build();
    }
}
