package com.evolvsys.nexuse2e.impl.service.support;

import com.evolvsys.nexuse2e.core.exception.NexusException;
import com.evolvsys.nexuse2e.core.parameter.ParameterDefinition;
import com.evolvsys.nexuse2e.core.parameter.ParameterType;
import java.util.List;
import org.apache.commons.lang3.StringUtils;
import org.springframework.http.client.support.BasicAuthenticationInterceptor;
import org.springframework.web.client.RestClient;
import org.springframework.web.reactive.function.client.ExchangeFilterFunctions;
import org.springframework.web.reactive.function.client.WebClient;

/**
 * @author SvenBarden
 * @since 11.06.2024
 */
public class HttpBasicAuthAuthenticationSupportService extends HttpAuthenticationSupportService {
    private static final String PARAM_BASIC_AUTH_USER = "basic_auth_user";
    private static final String PARAM_BASIC_AUTH_PASSWORD = "basic_auth_password";

    @Override
    public void addAuthentication(WebClient.Builder webClientBuilder) throws NexusException {
        String userName = parameters.value(PARAM_BASIC_AUTH_USER);
        String password = parameters.value(PARAM_BASIC_AUTH_PASSWORD);
        if (StringUtils.isNoneBlank(userName, password)) {
            webClientBuilder.filter(ExchangeFilterFunctions.basicAuthentication(userName, password));
        }
    }

    @Override
    public void addAuthentication(RestClient.Builder restClientBuilder) throws NexusException {
        String userName = parameters.value(PARAM_BASIC_AUTH_USER);
        String password = parameters.value(PARAM_BASIC_AUTH_PASSWORD);
        if (StringUtils.isNoneBlank(userName, password)) {
            restClientBuilder.requestInterceptor(new BasicAuthenticationInterceptor(userName, password));
        }
    }

    @Override
    public List<ParameterDefinition> parameterDefinitions() {
        return List.of(
                ParameterDefinition.builder()
                        .name(PARAM_BASIC_AUTH_USER)
                        .displayName("Username - basic auth")
                        .description("Username for basic auth")
                        .type(ParameterType.STRING)
                        .build(),
                ParameterDefinition.builder()
                        .name(PARAM_BASIC_AUTH_PASSWORD)
                        .displayName("Password - basic auth")
                        .description("Password for basic auth")
                        .type(ParameterType.PASSWORD)
                        .build());
    }
}
