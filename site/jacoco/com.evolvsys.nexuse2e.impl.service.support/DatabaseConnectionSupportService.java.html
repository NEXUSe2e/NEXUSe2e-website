package com.evolvsys.nexuse2e.impl.service.support;

import com.evolvsys.nexuse2e.core.exception.NexusError;
import com.evolvsys.nexuse2e.core.exception.NexusException;
import com.evolvsys.nexuse2e.core.parameter.ParameterDefinition;
import com.evolvsys.nexuse2e.core.parameter.ParameterType;
import java.sql.DriverManager;
import java.sql.SQLException;
import java.util.List;
import javax.sql.DataSource;
import lombok.extern.slf4j.Slf4j;
import org.springframework.boot.jdbc.DataSourceBuilder;
import org.springframework.jdbc.core.namedparam.NamedParameterJdbcTemplate;

/**
 * <p>Support-Service to get a configured database connection.</p>
 *
 * @author Florian Drees
 * @since 19.06.2024
 */
@Slf4j
public class DatabaseConnectionSupportService extends StandardSupportService {
    private static final String PARAMETER_URL = "url";
    private static final String PARAMETER_USERNAME = "username";
    private static final String PARAMETER_PASSWORD = "password";

    @Override
    public String getCategory() {
        return getClass().getSimpleName();
    }

    public DataSource getDataSource() throws NexusException {
        String url = parameters.value(PARAMETER_URL);
        DataSourceBuilder<?> dataSourceBuilder = DataSourceBuilder.create()
                .url(url)
                .username(parameters.value(PARAMETER_USERNAME))
                .password(parameters.value(PARAMETER_PASSWORD));

        try {
            dataSourceBuilder.driverClassName(
                    DriverManager.getDriver(url).getClass().getName());
        } catch (SQLException e) {
            throw new NexusException(
                    NexusError.builder()
                            .message("No suitable driver found")
                            .extension("url", url)
                            .build(),
                    e);
        }
        return dataSourceBuilder.build();
    }

    public NamedParameterJdbcTemplate getNamedParameterJdbcTemplate() throws NexusException {
        return getNamedParameterJdbcTemplate(getDataSource());
    }

    public NamedParameterJdbcTemplate getNamedParameterJdbcTemplate(DataSource dataSource) {
        return new NamedParameterJdbcTemplate(dataSource);
    }

    @Override
    public List<ParameterDefinition> parameterDefinitions() {
        return List.of(
                ParameterDefinition.builder()
                        .name(PARAMETER_URL)
                        .displayName("Url")
                        .description("Url of the database connection")
                        .type(ParameterType.STRING)
                        .required(true)
                        .build(),
                ParameterDefinition.builder()
                        .name(PARAMETER_USERNAME)
                        .displayName("Username")
                        .description("The user name used to login into the database")
                        .type(ParameterType.STRING)
                        .required(true)
                        .build(),
                ParameterDefinition.builder()
                        .name(PARAMETER_PASSWORD)
                        .displayName("Password")
                        .description("The password used to login into the database")
                        .type(ParameterType.PASSWORD)
                        .required(true)
                        .build());
    }
}
