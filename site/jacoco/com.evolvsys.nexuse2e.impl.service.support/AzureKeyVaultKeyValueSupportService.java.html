package com.evolvsys.nexuse2e.impl.service.support;

import com.azure.security.keyvault.secrets.SecretClient;
import com.azure.security.keyvault.secrets.SecretClientBuilder;
import com.evolvsys.nexuse2e.core.AzureAuthenticationProvider;
import com.evolvsys.nexuse2e.core.keyvalue.KeyValueProvider;
import com.evolvsys.nexuse2e.core.parameter.ParameterDefinition;
import com.evolvsys.nexuse2e.core.parameter.ParameterType;
import java.util.List;
import lombok.Value;
import org.apache.commons.text.lookup.StringLookup;

/**
 * @author SvenBarden
 * @since 17.06.2024
 */
public class AzureKeyVaultKeyValueSupportService extends KeyValueSupportService implements AzureAuthenticationProvider {
    private static final String PARAM_URL = "url";

    @Override
    public KeyValueProvider getKeyValueProvider() {
        SecretClient secretClient = new SecretClientBuilder()
                .vaultUrl(parameters.value(PARAM_URL))
                .credential(getTokenCredential(parameters))
                .buildClient();
        return new AzureKeyValueProvider(getServiceId(), secretClient);
    }

    @Override
    public List<ParameterDefinition> parameterDefinitions() {
        return List.of(
                ParameterDefinition.builder()
                        .name(PARAM_URL)
                        .displayName("Keyvault url")
                        .description("The url to the azure keyvault e.g. https://nexuse2e.vault.azure.net/")
                        .type(ParameterType.TRIMMED_STRING)
                        .required(true)
                        .build(),
                createClientIdParameter(),
                createClientSecretParameter(),
                createTenantIdParameter());
    }

    @Value
    static class AzureKeyValueProvider implements KeyValueProvider {
        String name;
        SecretClient secretClient;

        @Override
        public String getName() {
            return name;
        }

        @Override
        public StringLookup getLookup() {
            return key -> secretClient.getSecret(key).getValue();
        }
    }
}
