package com.evolvsys.nexuse2e.impl.service.support;

/**
 * @author SvenBarden
 * @since 11.06.2024

@Slf4j
public class HttpOauth2AuthenticationSupportService extends HttpAuthenticationSupportService
        implements BeanFactoryAware {
    private static final String PARAM_OAUTH2_TOKEN_URL = "oauth2_token_url";
    private static final String PARAM_OAUTH2_CLIENT_ID = "oauth2_client_id";
    private static final String PARAM_OAUTH2_CLIENT_SECRET = "oauth2_client_secret";
    private static final String PARAM_OAUTH2_SCOPES = "oauth2_scopes";
    private static final String PARAM_OAUTH2_ESCAPE_BASIC_SECRET = "oauth2_escape_basic_secret";
    private static final String PARAM_OAUTH2_AUTHORIZATION_GRANT_TYPE = "oauth2_authorization_grant_type";
    private static final String PARAM_OAUTH2_CLIENT_AUTHENTICATION_METHOD = "oauth2_client_authentication_method";
    private static final String PARAM_OAUTH2_USER_INFO_AUTHENTICATION_METHOD = "oauth2_user_info_authentication_method";
    private static final String PARAM_OAUTH2_DISABLE_HOST_NAME_VERIFICATION = "oauth2_disable_host_name_verification";
    private BeanFactory beanFactory;

    private ClientRegistration createClientRegistration() {
        return ClientRegistration.withRegistrationId(getServiceId())
                .tokenUri(parameters.value(PARAM_OAUTH2_TOKEN_URL))
                .clientId(parameters.value(PARAM_OAUTH2_CLIENT_ID))
                .clientSecret(parameters.value(PARAM_OAUTH2_CLIENT_SECRET))
                .authorizationGrantType(
                        new AuthorizationGrantType(parameters.value(PARAM_OAUTH2_AUTHORIZATION_GRANT_TYPE)))
                .scope((List<String>) parameters.value(PARAM_OAUTH2_SCOPES))
                .clientAuthenticationMethod(
                        new ClientAuthenticationMethod(parameters.value(PARAM_OAUTH2_CLIENT_AUTHENTICATION_METHOD)))
                .userInfoAuthenticationMethod(
                        new AuthenticationMethod(parameters.value(PARAM_OAUTH2_USER_INFO_AUTHENTICATION_METHOD)))
                .build();
    }

    private DefaultOAuth2TokenRequestHeadersConverter<OAuth2ClientCredentialsGrantRequest> createHeadersConverter() {
        DefaultOAuth2TokenRequestHeadersConverter<OAuth2ClientCredentialsGrantRequest> headersConverter =
                new DefaultOAuth2TokenRequestHeadersConverter<>();
        headersConverter.setEncodeClientCredentials(parameters.value(PARAM_OAUTH2_ESCAPE_BASIC_SECRET));
        return headersConverter;
    }

    @Override
    public void addAuthentication(WebClient.Builder webClientBuilder) throws NexusException {
        ClientRegistration clientRegistration = createClientRegistration();
        InMemoryReactiveClientRegistrationRepository registrationRepository =
                new InMemoryReactiveClientRegistrationRepository(clientRegistration);
        InMemoryReactiveOAuth2AuthorizedClientService clientService =
                new InMemoryReactiveOAuth2AuthorizedClientService(registrationRepository);
        WebClientReactiveClientCredentialsTokenResponseClient tokenResponseClient =
                new WebClientReactiveClientCredentialsTokenResponseClient();

        NexusWebClientFactory webClientFactory = beanFactory.getBean(NexusWebClientFactory.class);
        WebClientContext oauth2Context = webClientFactory
                .createContextBuilder()
                .ssl(true)
                .disableHostNameVerification(parameters.value(PARAM_OAUTH2_DISABLE_HOST_NAME_VERIFICATION))
                .build();
        WebClient.Builder webClient = webClientFactory.createWebClient(oauth2Context);
        tokenResponseClient.setWebClient(webClient.build());

        if ("client_secret_basic".equals(parameters.value(PARAM_OAUTH2_CLIENT_AUTHENTICATION_METHOD))) {
            tokenResponseClient.setHeadersConverter(createHeadersConverter());
        }

        ReactiveOAuth2AuthorizedClientProvider clientProvider = ReactiveOAuth2AuthorizedClientProviderBuilder.builder()
                .clientCredentials(clientCredentialsGrantBuilder -> clientCredentialsGrantBuilder
                        .accessTokenResponseClient(tokenResponseClient)
                        .build())
                .build();

        AuthorizedClientServiceReactiveOAuth2AuthorizedClientManager clientManager =
                new AuthorizedClientServiceReactiveOAuth2AuthorizedClientManager(registrationRepository, clientService);
        clientManager.setAuthorizedClientProvider(clientProvider);
        ServerOAuth2AuthorizedClientExchangeFilterFunction clientFilter =
                new ServerOAuth2AuthorizedClientExchangeFilterFunction(clientManager);

        RemoveAuthorizedClientReactiveOAuth2AuthorizationFailureHandler failureHandler =
                new RemoveAuthorizedClientReactiveOAuth2AuthorizationFailureHandler(
                        (clientRegistrationId, principal, attributes) -> Mono.empty()) {
                    @Override
                    public Mono<Void> onAuthorizationFailure(
                            OAuth2AuthorizationException authorizationException,
                            Authentication principal,
                            Map<String, Object> attributes) {
                        LOGGER.error("error using oauth2 authentication {}", attributes, authorizationException);
                        return super.onAuthorizationFailure(authorizationException, principal, attributes);
                    }
                };

        clientFilter.setAuthorizationFailureHandler(failureHandler);
        clientManager.setAuthorizationFailureHandler(failureHandler);
        clientFilter.setDefaultClientRegistrationId(getServiceId());

        webClientBuilder.filter(clientFilter);
    }

    @Override
    public void addAuthentication(RestClient.Builder restClientBuilder) throws NexusException {
        NexusRestClientFactory restClientFactory = beanFactory.getBean(NexusRestClientFactory.class);

        ClientRegistration clientRegistration = createClientRegistration();
        InMemoryClientRegistrationRepository registrationRepository =
                new InMemoryClientRegistrationRepository(clientRegistration);
        InMemoryOAuth2AuthorizedClientService clientService =
                new InMemoryOAuth2AuthorizedClientService(registrationRepository);

        ClientCredentialsOAuth2AuthorizedClientProvider clientProvider =
                new ClientCredentialsOAuth2AuthorizedClientProvider();
        RestClientClientCredentialsTokenResponseClient tokenResponseClient =
                new RestClientClientCredentialsTokenResponseClient();
        clientProvider.setAccessTokenResponseClient(tokenResponseClient);

        WebClientContext oauth2Context = restClientFactory
                .createContextBuilder()
                .ssl(true)
                .disableHostNameVerification(parameters.value(PARAM_OAUTH2_DISABLE_HOST_NAME_VERIFICATION))
                .build();

        tokenResponseClient.setRestClient(restClientFactory
                .createClientBuilder(oauth2Context)
                .messageConverters(
                        List.of(new FormHttpMessageConverter(), new OAuth2AccessTokenResponseHttpMessageConverter()))
                .defaultStatusHandler(new OAuth2ErrorResponseErrorHandler())
                .build());

        tokenResponseClient.setHeadersConverter(createHeadersConverter());

        AuthorizedClientServiceOAuth2AuthorizedClientManager clientManager =
                new AuthorizedClientServiceOAuth2AuthorizedClientManager(registrationRepository, clientService);

        clientManager.setAuthorizedClientProvider(clientProvider);

        OAuth2ClientHttpRequestInterceptor requestInterceptor =
                createRequestInterceptor(clientManager, clientRegistration);

        restClientBuilder.requestInterceptor(requestInterceptor);
    }

    private static OAuth2ClientHttpRequestInterceptor createRequestInterceptor(
            OAuth2AuthorizedClientManager clientManager, ClientRegistration clientRegistration) {
        OAuth2ClientHttpRequestInterceptor requestInterceptor = new OAuth2ClientHttpRequestInterceptor(clientManager);
        requestInterceptor.setClientRegistrationIdResolver(request -> clientRegistration.getRegistrationId());
        requestInterceptor.setPrincipalResolver(request -> new AbstractAuthenticationToken(null) {

            @Override
            public Object getCredentials() {
                return "";
            }

            @Override
            public Object getPrincipal() {
                return clientRegistration.getClientId();
            }
        });
        return requestInterceptor;
    }

    @Override
    public List<ParameterDefinition> parameterDefinitions() {
        return List.of(
                ParameterDefinition.builder()
                        .name(PARAM_OAUTH2_TOKEN_URL)
                        .displayName("Oauth2 Access token url")
                        .description("URL to get the oauth2 token from")
                        .required(true)
                        .type(ParameterType.TRIMMED_STRING)
                        .build(),
                ParameterDefinition.builder()
                        .name(PARAM_OAUTH2_CLIENT_ID)
                        .displayName("Oauth2 client id")
                        .description("The client id for retrieving the access token")
                        .required(true)
                        .type(ParameterType.TRIMMED_STRING)
                        .build(),
                ParameterDefinition.builder()
                        .name(PARAM_OAUTH2_CLIENT_SECRET)
                        .displayName("Oauth2 client secret")
                        .description("The client secret for retrieving the access token")
                        .required(true)
                        .type(ParameterType.PASSWORD)
                        .build(),
                ParameterDefinition.builder()
                        .name(PARAM_OAUTH2_SCOPES)
                        .displayName("Oauth2 scopes")
                        .description("Scopes to request for the oauth2 access token")
                        .type(ParameterType.STRING_LIST)
                        .build(),
                ParameterDefinition.builder()
                        .name(PARAM_OAUTH2_ESCAPE_BASIC_SECRET)
                        .displayName("Escape oauth2 basic auth")
                        .description("Escape the client id and client secret for the oauth2 token request.")
                        .required(true)
                        .type(ParameterType.BOOLEAN)
                        .defaultValue(Boolean.FALSE)
                        .build(),
                ParameterDefinition.builder()
                        .name(PARAM_OAUTH2_AUTHORIZATION_GRANT_TYPE)
                        .displayName("Authorization grant type")
                        .description("Grant type used for the oauth2 token request")
                        .required(true)
                        .type(ParameterType.DROPDOWN_LIST)
                        .option("authorization_code")
                        .option("refresh_token")
                        .option("client_credentials")
                        .defaultValue("client_credentials")
                        .build(),
                ParameterDefinition.builder()
                        .name(PARAM_OAUTH2_CLIENT_AUTHENTICATION_METHOD)
                        .displayName("Client authentication method")
                        .description("Authentication method used to get the oauth2 token")
                        .required(true)
                        .type(ParameterType.DROPDOWN_LIST)
                        .option("client_secret_basic")
                        .option("client_secret_post")
                        .option("client_secret_jwt")
                        .option("private_key_jwt")
                        .option("none")
                        .defaultValue("client_secret_basic")
                        .build(),
                ParameterDefinition.builder()
                        .name(PARAM_OAUTH2_USER_INFO_AUTHENTICATION_METHOD)
                        .displayName("User info authentication method")
                        .description("he authentication method used when sending bearer access tokens")
                        .required(true)
                        .type(ParameterType.DROPDOWN_LIST)
                        .option("header")
                        .option("form")
                        .option("query")
                        .defaultValue("header")
                        .build(),
                ParameterDefinition.builder()
                        .name(PARAM_OAUTH2_DISABLE_HOST_NAME_VERIFICATION)
                        .displayName("Disable ssl host name verification")
                        .description("Disables the ssl host name verification for the oauth2 token request")
                        .type(ParameterType.BOOLEAN)
                        .build());
    }

    @Override
    public void setBeanFactory(BeanFactory beanFactory) {
        this.beanFactory = beanFactory;
    }
}
