package com.evolvsys.nexuse2e.impl.service.support;

import com.evolvsys.nexuse2e.core.exception.NexusError;
import com.evolvsys.nexuse2e.core.exception.NexusException;
import com.evolvsys.nexuse2e.core.parameter.ParameterDefinition;
import com.evolvsys.nexuse2e.core.parameter.ParameterType;
import java.util.List;
import java.util.function.Predicate;
import lombok.extern.slf4j.Slf4j;
import org.apache.commons.collections4.CollectionUtils;
import org.apache.commons.collections4.KeyValue;

/**
 * Maps string values based on a provided mapping table. Optionally fails if no mapping entry is found for the input value.
 *
 * @author SvenBarden
 * @since 08.08.2025
 */
@Slf4j
public class StringMapperSupportService extends StandardSupportService {
    public static final String PARAM_MAPPING = "mapping";
    public static final String PARAM_FAIL_ON_NOT_FOUND = "fail_on_not_found";
    public static final String PARAM_IGNORE_CASE = "ignore_case";

    @Override
    public List<ParameterDefinition> parameterDefinitions() {
        return List.of(
                ParameterDefinition.builder()
                        .name(PARAM_MAPPING)
                        .displayName("Mapping")
                        .description("The mapping table to use. ")
                        .required(true)
                        .type(ParameterType.KEY_VALUE_PAIR_LIST_TRIMMED)
                        .build(),
                ParameterDefinition.builder()
                        .name(PARAM_FAIL_ON_NOT_FOUND)
                        .displayName("Fail on not found")
                        .description(
                                "If true, the service throws an error if no mapping was provided for the given input.")
                        .type(ParameterType.BOOLEAN)
                        .defaultValue(Boolean.FALSE)
                        .build(),
                ParameterDefinition.builder()
                        .name(PARAM_IGNORE_CASE)
                        .displayName("Ignore case")
                        .description(
                                "If true, ignores the case for the mapping. Note that if the same key in multiple cases is provided there is no guarantee which mapping will be used.")
                        .type(ParameterType.BOOLEAN)
                        .defaultValue(Boolean.FALSE)
                        .build());
    }

    private Predicate<KeyValue<String, String>> match(String input) {
        boolean ignoreCase = parameters.value(PARAM_IGNORE_CASE);

        return ignoreCase
                ? keyValue -> keyValue.getKey().equalsIgnoreCase(input)
                : keyValue -> keyValue.getKey().equals(input);
    }

    public String map(String input) throws NexusException {
        List<KeyValue<String, String>> mapping = parameters.value(PARAM_MAPPING);
        boolean failOnNotFound = parameters.value(PARAM_FAIL_ON_NOT_FOUND);

        String mappedValue = CollectionUtils.emptyIfNull(mapping).stream()
                .filter(match(input))
                .map(KeyValue::getValue)
                .findFirst()
                .orElse(null);

        String result;
        if (mappedValue == null) {
            if (failOnNotFound) {
                LOGGER.trace("no mapping entries found, throwing exception");
                throw new NexusException(NexusError.builder()
                        .message("No mapping entry found")
                        .extension("key", input)
                        .extension(
                                "entries",
                                mapping.stream().map(KeyValue::getKey).toList())
                        .build());
            } else {
                LOGGER.trace("no mapping entries found for {}, returning input as is", input);
                result = input;
            }
        } else {
            result = mappedValue;
            LOGGER.trace("mapping entry found, returning mapped value: {}", result);
        }

        return result;
    }
}
