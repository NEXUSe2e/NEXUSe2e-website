package com.evolvsys.nexuse2e.core.pipeline.repository;

import com.evolvsys.nexuse2e.core.pipeline.PipelineFilter;
import com.evolvsys.nexuse2e.core.pipeline.model.PipelineEntity;
import com.evolvsys.nexuse2e.core.pipeline.model.QPipelineEntity;
import com.querydsl.core.BooleanBuilder;
import com.querydsl.core.annotations.QueryDelegate;
import com.querydsl.core.types.Predicate;
import org.apache.commons.lang3.StringUtils;
import org.springframework.lang.Nullable;

/**
 * @author Florian Drees
 * @since 20.11.2023
 */
public final class PipelineQueryDelegate {
    private PipelineQueryDelegate() {}

    @QueryDelegate(PipelineEntity.class)
    public static Predicate filter(QPipelineEntity pipelineEntity, @Nullable PipelineFilter pipelineFilter) {
        BooleanBuilder builder = new BooleanBuilder();
        if (pipelineFilter != null) {
            if (StringUtils.isNotBlank(pipelineFilter.getPipelineId())) {
                builder.and(pipelineEntity.pipelineId.contains(pipelineFilter.getPipelineId()));
            }
        }
        return builder.getValue();
    }
}
