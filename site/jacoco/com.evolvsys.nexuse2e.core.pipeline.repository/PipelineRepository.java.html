package com.evolvsys.nexuse2e.core.pipeline.repository;

import com.evolvsys.nexuse2e.core.pipeline.PipelineFilter;
import com.evolvsys.nexuse2e.core.pipeline.model.PipelineEntity;
import com.evolvsys.nexuse2e.core.pipeline.model.QPipelineEntity;
import com.querydsl.core.types.Predicate;
import com.querydsl.core.types.dsl.BooleanExpression;
import java.util.List;
import java.util.Optional;
import org.apache.commons.collections4.IterableUtils;
import org.springframework.data.jpa.repository.JpaRepository;
import org.springframework.data.querydsl.QuerydslPredicateExecutor;

/**
 * @author SvenBarden
 * @since 11.10.2023
 */
public interface PipelineRepository
        extends JpaRepository<PipelineEntity, Long>, QuerydslPredicateExecutor<PipelineEntity> {
    Optional<PipelineEntity> findByPipelineId(String pipelineId);

    default List<PipelineEntity> findAll(PipelineFilter pipelineFilter, BooleanExpression securityPredicate) {
        Predicate predicate = QPipelineEntity.pipelineEntity.filter(pipelineFilter);
        if (predicate != null) {
            securityPredicate = securityPredicate.and(predicate);
        }
        return IterableUtils.toList(findAll(securityPredicate));
    }
}
