package com.evolvsys.nexuse2e.core.message.state;

import com.evolvsys.common.lock.LockService;
import com.evolvsys.nexuse2e.core.exception.NexusError;
import com.evolvsys.nexuse2e.core.exception.NexusException;
import com.evolvsys.nexuse2e.core.message.context.ExecutionContext;
import com.evolvsys.nexuse2e.core.message.state.event.MessageEventHandler;
import com.evolvsys.nexuse2e.internal.Message;
import java.util.List;
import java.util.concurrent.locks.Lock;
import lombok.RequiredArgsConstructor;
import lombok.extern.slf4j.Slf4j;
import org.apache.commons.collections4.CollectionUtils;
import org.apache.commons.lang3.Validate;
import org.springframework.stereotype.Service;

/**
 * @author SvenBarden
 * @since 18.01.2024
 */
@Service
@RequiredArgsConstructor
@Slf4j
public class MessageEventService {
    private final LockService lockService;
    private final List<MessageEventHandler> messageEventHandlers;

    private String getConversationLockKey(String conversationId) {
        return "conversation_" + conversationId;
    }

    private List<MessageEventHandler> getHandlers(MessageEventType messageEventType) {
        return messageEventHandlers.stream()
                .filter(messageEventHandler -> messageEventHandler.type() == messageEventType)
                .toList();
    }

    public void trigger(ExecutionContext executionContext, MessageEventType messageEventType) throws NexusException {
        Message message = executionContext.getMessage();
        Validate.notNull(message, "message must not be null");
        Validate.notBlank(message.getId(), "message.getId() must not be blank");
        Validate.notBlank(message.getConversationId(), "message.getConversationId() must not be blank");

        Lock messageLock = lockService.obtain(getConversationLockKey(message.getConversationId()));
        try {
            messageLock.lock();

            LOGGER.debug("triggering event {} for message {}", messageEventType, message);

            List<MessageEventHandler> handlers = getHandlers(messageEventType);
            if (CollectionUtils.isEmpty(handlers)) {
                throw new NexusException(NexusError.builder()
                        .message("no handler for event found")
                        .extension("event", messageEventType.name())
                        .build());
            }

            LOGGER.debug("found {} handlers for event {}", handlers.size(), messageEventType);
            for (MessageEventHandler handler : handlers) {
                LOGGER.trace("executing event handler {}", handler);
                handler.handle(executionContext);
            }

        } finally {
            messageLock.unlock();
        }
    }
}
