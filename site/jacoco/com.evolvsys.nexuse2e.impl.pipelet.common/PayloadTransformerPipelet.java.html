package com.evolvsys.nexuse2e.impl.pipelet.common;

import com.evolvsys.nexuse2e.core.component.Pipelet;
import com.evolvsys.nexuse2e.core.exception.NexusException;
import com.evolvsys.nexuse2e.core.message.context.ExecutionContext;
import com.evolvsys.nexuse2e.core.message.payload.content.PayloadContent;
import com.evolvsys.nexuse2e.core.parameter.ParameterDefinition;
import com.evolvsys.nexuse2e.core.parameter.ParameterType;
import com.evolvsys.nexuse2e.internal.ContextPayload;
import com.evolvsys.nexuse2e.internal.Message;
import edu.umd.cs.findbugs.annotations.SuppressFBWarnings;
import java.nio.charset.StandardCharsets;
import java.util.*;
import lombok.extern.slf4j.Slf4j;
import org.apache.commons.collections4.CollectionUtils;
import org.apache.commons.text.StringSubstitutor;

/**
 * Overwrites the payload of the message context with a new payload based on a template.
 *
 * @author SvenBarden
 * @since 28.07.2025
 */
@Slf4j
public class PayloadTransformerPipelet extends Pipelet {
    private static final String PAYLOAD_TEMPLATE = "payload_template";
    private static final String PAYLOAD_CONTENT_TYPE = "payload_content_type";
    private static final String PAYLOAD_ENCODING = "payload_encoding";

    @Override
    public List<ParameterDefinition> parameterDefinitions() {
        return List.of(
                ParameterDefinition.builder()
                        .name(PAYLOAD_TEMPLATE)
                        .displayName("Template")
                        .description(
                                "The template for the new payload to be transformed to. Has the following variables: "
                                        + "$[choreographyId], $[actionId], $[sourcePartnerId], $[targetPartnerId], "
                                        + "$[conversationId], $[messageId], $[referenceMessageId], $[payload].")
                        .type(ParameterType.TRIMMED_LARGE_STRING)
                        .required(true)
                        .build(),
                ParameterDefinition.builder()
                        .name(PAYLOAD_CONTENT_TYPE)
                        .displayName("Content Type")
                        .description("The content type of the payload after transformation.")
                        .defaultValue("text/xml")
                        .type(ParameterType.TRIMMED_STRING)
                        .required(true)
                        .build(),
                ParameterDefinition.builder()
                        .name(PAYLOAD_ENCODING)
                        .displayName("Encoding")
                        .description(
                                "The encoding of the payload after transformation. For text payloads, use CDATA encoding, for binary payloads use BASE64.")
                        .defaultValue(Encoding.CDATA.name())
                        .type(ParameterType.DROPDOWN_LIST)
                        .option(Encoding.CDATA.name())
                        .option(Encoding.BASE64.name())
                        .required(true)
                        .build());
    }

    private void addToMap(Map<String, Object> map, String key, Object value) {
        if (key != null && value != null) {
            map.put(key, value);
        }
    }

    @Override
    @SuppressFBWarnings("CFS_CONFUSING_FUNCTION_SEMANTICS")
    public ExecutionContext execute(ExecutionContext executionContext) throws NexusException {
        Message message = executionContext.getMessage();
        ContextPayload payload = Optional.of(message)
                .map(Message::getPayloads)
                .filter(CollectionUtils::isNotEmpty)
                .map(List::getFirst)
                .orElse(null);

        String template = parameters.value(PAYLOAD_TEMPLATE);

        LOGGER.trace("Replacing template {} ", template);

        Map<String, Object> values = new HashMap<>();

        String contentId;
        if (payload == null) {
            contentId = "transformed";
        } else {
            contentId = "transformed-" + payload.getContentId();
            Encoding encoding = Encoding.valueOf(parameters.value(PAYLOAD_ENCODING));
            String content =
                    switch (encoding) {
                        case BASE64 ->
                            Base64.getEncoder()
                                    .encodeToString(payload.getContent().getByteArray());
                        case CDATA ->
                            "<![CDATA[%s]]>".formatted(payload.getContent().getString(StandardCharsets.UTF_8));
                    };
            values.put("payload", content);
        }

        addToMap(values, "choreographyId", executionContext.getChoreographyId());
        addToMap(values, "actionId", executionContext.getActionId());
        addToMap(values, "sourcePartnerId", executionContext.getSourcePartnerId());
        addToMap(values, "targetPartnerId", executionContext.getTargetPartnerId());
        addToMap(values, "conversationId", message.getConversationId());
        addToMap(values, "messageId", message.getId());
        addToMap(values, "referenceMessageId", message.getReferenceMessageId());

        // we do not use the default ${} syntax to avoid conflicts with the key value services
        String replaced = StringSubstitutor.replace(template, values, "$[", "]");

        PayloadContent payloadContent = PayloadContent.of(replaced.getBytes(StandardCharsets.UTF_8));
        ContextPayload transformedPayload = ContextPayload.builder()
                .encoding("UTF-8")
                .contentType(parameters.value(PAYLOAD_CONTENT_TYPE))
                .content(payloadContent)
                .size(payloadContent.getSize())
                .contentId(contentId)
                .build();

        executionContext.setData(transformedPayload);

        return executionContext;
    }

    public enum Encoding {
        BASE64,
        CDATA
    }
}
