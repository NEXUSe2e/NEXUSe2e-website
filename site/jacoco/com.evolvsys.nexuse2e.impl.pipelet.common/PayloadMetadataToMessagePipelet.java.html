package com.evolvsys.nexuse2e.impl.pipelet.common;

import com.evolvsys.nexuse2e.core.component.Pipelet;
import com.evolvsys.nexuse2e.core.exception.NexusException;
import com.evolvsys.nexuse2e.core.message.MetaData;
import com.evolvsys.nexuse2e.core.message.context.ExecutionContext;
import com.evolvsys.nexuse2e.core.parameter.ParameterDefinition;
import com.evolvsys.nexuse2e.core.parameter.ParameterType;
import com.evolvsys.nexuse2e.internal.ContextPayload;
import com.evolvsys.nexuse2e.internal.Message;
import java.util.Collection;
import java.util.Collections;
import java.util.List;
import lombok.extern.slf4j.Slf4j;
import org.apache.commons.collections4.CollectionUtils;
import org.apache.commons.lang3.StringUtils;

/**
 * Pipelet to copy metadata from the payload into the message.
 *
 * @author Julian Drees
 * @since 06.12.2024
 */
@Slf4j
public class PayloadMetadataToMessagePipelet extends Pipelet {
    public static final String PARAM_COPY_ALL = "copy_all";
    public static final String PARAM_METADATA_FIELDS = "metadata_fields";
    public static final String PARAM_CATEGORY = "category";

    @Override
    public List<ParameterDefinition> parameterDefinitions() {
        return List.of(
                ParameterDefinition.builder()
                        .name(PARAM_METADATA_FIELDS)
                        .displayName("Copy metadata fields")
                        .description("The name of the metadata fields to copy")
                        .type(ParameterType.STRING_LIST)
                        .build(),
                ParameterDefinition.builder()
                        .name(PARAM_CATEGORY)
                        .displayName("Metadata category")
                        .description(
                                "Metadata category from which the data is to be copied. If blank, all the categories are used.")
                        .type(ParameterType.TRIMMED_STRING)
                        .build(),
                ParameterDefinition.builder()
                        .name(PARAM_COPY_ALL)
                        .displayName("Copy all metadata fields")
                        .description(
                                "Flag to copy all metadata fields for the category. This value has priority over \"Copy metadata fields\".")
                        .type(ParameterType.BOOLEAN)
                        .defaultValue(Boolean.FALSE)
                        .build());
    }

    @Override
    public ExecutionContext execute(ExecutionContext executionContext) throws NexusException {
        LOGGER.trace("Running PayloadMetadataToMessagePipelet.");

        boolean copyAll = parameters.value(PARAM_COPY_ALL);
        String category = parameters.value(PARAM_CATEGORY);
        List<String> metadataFields = parameters.value(PARAM_METADATA_FIELDS);

        if (!copyAll && CollectionUtils.isEmpty(metadataFields)) {
            LOGGER.warn("Pipelet should not copy all metadata fields, but a field isn't configured either.");
        }

        Message message = executionContext.getMessage();
        MetaData targetMetaData = message.getMetaData();
        for (ContextPayload payload : message.getPayloads()) {
            LOGGER.trace("Copying metadata from payload {} to message.", payload);
            MetaData sourceMetaData = payload.getMetaData();
            Collection<String> categories =
                    StringUtils.isBlank(category) ? sourceMetaData.categories() : Collections.singleton(category);

            for (String c : categories) {
                Collection<String> keys = copyAll ? sourceMetaData.keys(c) : metadataFields;
                keys.forEach(key -> targetMetaData.addAll(c, key, sourceMetaData.get(c, key)));
            }
        }
        return executionContext;
    }
}
