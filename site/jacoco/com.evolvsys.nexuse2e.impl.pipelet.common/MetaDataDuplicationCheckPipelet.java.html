package com.evolvsys.nexuse2e.impl.pipelet.common;

import com.evolvsys.nexuse2e.core.component.Pipelet;
import com.evolvsys.nexuse2e.core.exception.NexusError;
import com.evolvsys.nexuse2e.core.exception.NexusException;
import com.evolvsys.nexuse2e.core.message.MetaDataLinkType;
import com.evolvsys.nexuse2e.core.message.context.ExecutionContext;
import com.evolvsys.nexuse2e.core.message.metadata.service.MetaDataService;
import com.evolvsys.nexuse2e.core.parameter.ParameterDefinition;
import com.evolvsys.nexuse2e.core.parameter.ParameterType;
import java.util.Collection;
import java.util.List;
import java.util.Map;
import lombok.extern.slf4j.Slf4j;
import one.util.streamex.EntryStream;
import org.apache.commons.collections4.CollectionUtils;
import org.apache.commons.collections4.MapUtils;
import org.apache.commons.lang3.StringUtils;

/**
 * Pipelet to check for duplicated metadata keys and values.
 *
 * @author Julian Drees
 * @since 29.08.2025
 */
@Slf4j
public class MetaDataDuplicationCheckPipelet extends Pipelet {
    static final String PARAM_CHOREOGRAPHY_ID = "choreography_id";
    static final String PARAM_METADATA_CATEGORY = "metadata_category";
    static final String PARAM_METADATA_KEYS = "metadata_keys";
    static final String PARAM_LINK_TYPE = "link_type";

    @Override
    public List<ParameterDefinition> parameterDefinitions() {
        return List.of(
                ParameterDefinition.builder()
                        .name(PARAM_METADATA_CATEGORY)
                        .displayName("Metadata category")
                        .description("The category which holds the metadata.")
                        .type(ParameterType.TRIMMED_STRING)
                        .build(),
                ParameterDefinition.builder()
                        .name(PARAM_METADATA_KEYS)
                        .displayName("Metadata keys")
                        .description(
                                "The metadata keys to check for duplicates. If any of the values for those keys already exist, the message will be rejected.")
                        .type(ParameterType.STRING_LIST)
                        .required(true)
                        .build(),
                ParameterDefinition.builder()
                        .name(PARAM_CHOREOGRAPHY_ID)
                        .displayName("Choreography Id")
                        .type(ParameterType.TRIMMED_STRING)
                        .description(
                                "Optional id of the choreography to which messages's metadata the metadata from this message should be compared to.")
                        .build(),
                ParameterDefinition.builder()
                        .name(PARAM_LINK_TYPE)
                        .displayName("Link type")
                        .description(
                                "Whether this duplicate check should consider single key values as duplicates (or) or as a key combinations (and). Only matters when multiple keys are configured.")
                        .type(ParameterType.DROPDOWN_LIST)
                        .options(List.of(MetaDataLinkType.AND.name(), MetaDataLinkType.OR.name()))
                        .required(true)
                        .defaultValue(MetaDataLinkType.OR.name())
                        .build());
    }

    @Override
    public ExecutionContext execute(ExecutionContext executionContext) throws NexusException {
        String metadataCategory = getParameters().value(PARAM_METADATA_CATEGORY);
        Collection<String> metadataKeys =
                CollectionUtils.emptyIfNull(getParameters().value(PARAM_METADATA_KEYS));
        if (CollectionUtils.isEmpty(metadataKeys)) {
            throw new NexusException(
                    NexusError.builder().message("No metadata keys configured.").build());
        }

        String choreographyId = getParameters().value(PARAM_CHOREOGRAPHY_ID);
        MetaDataLinkType linkType = MetaDataLinkType.valueOf(getParameters().value(PARAM_LINK_TYPE));

        LOGGER.info(
                "Checking for duplicates for metadata keys {} in category {} using link type {}.",
                metadataKeys,
                metadataCategory,
                linkType);
        if (StringUtils.isNotBlank(choreographyId)) {
            LOGGER.debug("Duplicate checks only considers message metadata in choreography {}.", choreographyId);
        }

        MetaDataService metaDataService = executionContext.getBeanFactory().getBean(MetaDataService.class);
        Map<String, List<String>> duplicateKeys = metaDataService.getDuplicatesByKey(
                executionContext.getMessage().getMetaData(), metadataCategory, metadataKeys, choreographyId, linkType);
        if (MapUtils.isNotEmpty(duplicateKeys)) {
            NexusError nexusError = NexusError.builder()
                    .code("DUPLICATE")
                    .message("Duplication check failed, the following duplicate keys and values were found.")
                    .extensions(EntryStream.of(duplicateKeys)
                            .mapValues(values -> "[" + String.join(", ", values) + "]")
                            .toMap())
                    .build();
            throw new NexusException(nexusError);
        }
        return executionContext;
    }
}
