package com.evolvsys.nexuse2e.impl.pipelet.common;

import com.evolvsys.nexuse2e.core.component.Pipelet;
import com.evolvsys.nexuse2e.core.exception.NexusException;
import com.evolvsys.nexuse2e.core.message.MetaData;
import com.evolvsys.nexuse2e.core.message.context.ExecutionContext;
import com.evolvsys.nexuse2e.core.parameter.ParameterDefinition;
import com.evolvsys.nexuse2e.internal.Message;
import java.util.List;
import java.util.Set;
import java.util.stream.Collectors;
import lombok.extern.slf4j.Slf4j;
import org.apache.commons.lang3.StringUtils;

/**
 * Check Context Metadata for valuable fields added in the ingress process.
 * Currently supported by default are the keyword METADATA_KEYWORD and METADATA_KEYWORD_<key>. METADATA_KEYWORD
 * expects key value pairs (key=value) as the value while METADATA_KEYWORD_<key> only receives a value.
 */
@Slf4j
public class ParamsToMetaDataPipelet extends Pipelet {

    private static final String METADATA_KEYWORD = "meta_data";

    @Override
    public List<ParameterDefinition> parameterDefinitions() {
        return List.of();
    }

    @Override
    public ExecutionContext execute(ExecutionContext executionContext) throws NexusException {
        Message message = executionContext.getMessage();
        MetaData metaData = executionContext.getMetaData();
        Set<String> keys = metaData.keys().stream()
                .filter(key -> key.startsWith(METADATA_KEYWORD))
                .collect(Collectors.toSet());

        for (String key : keys) {
            List<String> values = metaData.getAll(key);
            if (METADATA_KEYWORD.equals(key)) {
                extractKeyValue(values, message);
            } else {
                extractSuffixKeyValue(key, values, message);
            }
        }

        return executionContext;
    }

    private static void extractSuffixKeyValue(String key, Iterable<String> values, Message message) {
        String newKey = key.substring(METADATA_KEYWORD.length() + 1);
        values.forEach(value -> {
            if (StringUtils.isNoneEmpty(value)) {
                message.getMetaData().add(newKey.trim(), value.trim());
            }
        });
    }

    private static void extractKeyValue(Iterable<String> values, Message message) {
        for (String value : values) {
            if (StringUtils.isNoneEmpty(value)) {
                if (value.contains("=")) {
                    String[] keyValue = value.split("=", 2);
                    message.getMetaData().add(keyValue[0].trim(), keyValue[1].trim());
                } else {
                    message.getMetaData().add(value, Boolean.TRUE);
                }
            }
        }
    }
}
