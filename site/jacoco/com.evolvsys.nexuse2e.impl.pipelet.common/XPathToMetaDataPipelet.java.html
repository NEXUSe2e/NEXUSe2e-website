package com.evolvsys.nexuse2e.impl.pipelet.common;

import com.evolvsys.nexuse2e.core.component.Pipelet;
import com.evolvsys.nexuse2e.core.exception.NexusError;
import com.evolvsys.nexuse2e.core.exception.NexusException;
import com.evolvsys.nexuse2e.core.message.context.ExecutionContext;
import com.evolvsys.nexuse2e.core.parameter.ParameterDefinition;
import com.evolvsys.nexuse2e.core.parameter.ParameterType;
import com.evolvsys.nexuse2e.impl.parameter.XPathParameters;
import com.evolvsys.nexuse2e.internal.ContextPayload;
import com.evolvsys.nexuse2e.internal.Message;
import edu.umd.cs.findbugs.annotations.SuppressFBWarnings;
import java.io.IOException;
import java.util.List;
import java.util.Optional;
import javax.xml.XMLConstants;
import javax.xml.parsers.DocumentBuilder;
import javax.xml.parsers.DocumentBuilderFactory;
import javax.xml.parsers.ParserConfigurationException;
import javax.xml.xpath.XPath;
import javax.xml.xpath.XPathConstants;
import javax.xml.xpath.XPathExpressionException;
import javax.xml.xpath.XPathFactory;
import lombok.extern.slf4j.Slf4j;
import org.apache.commons.collections4.KeyValue;
import org.w3c.dom.Document;
import org.xml.sax.SAXException;

/**
 *
 */
@Slf4j
public class XPathToMetaDataPipelet extends Pipelet implements XPathParameters {

    public static final String PARAM_XPATH_KEY_VALUE_PAIRS = "xpath_key_value_pairs";

    @Override
    public List<ParameterDefinition> parameterDefinitions() {
        return List.of(
                ParameterDefinition.builder()
                        .name(PARAM_XPATH_KEY_VALUE_PAIRS)
                        .displayName("MetaData/Xpath Pairs")
                        .description("Define metadata keys and xpath statements as key value pairs.")
                        .required(true)
                        .type(ParameterType.KEY_VALUE_PAIR_LIST_TRIMMED)
                        .build(),
                STOP_ON_FAIL.get().build(),
                STOP_ON_NOT_FOUND.get().build(),
                TRIM_RESULT.get().build());
    }

    @Override
    @SuppressFBWarnings({"XPATH_INJECTION", "XXE_DOCUMENT", "SPP_PASSING_THIS_AS_PARM"})
    public ExecutionContext execute(ExecutionContext executionContext) throws NexusException {
        DocumentBuilderFactory builderFactory = DocumentBuilderFactory.newInstance();
        builderFactory.setAttribute(XMLConstants.ACCESS_EXTERNAL_DTD, "");
        builderFactory.setAttribute(XMLConstants.ACCESS_EXTERNAL_SCHEMA, "");
        DocumentBuilder builder;
        Document xmlDocument;
        try {
            builder = builderFactory.newDocumentBuilder();

            ContextPayload payload = Optional.ofNullable(executionContext)
                    .map(ExecutionContext::getMessage)
                    .map(Message::getPayloads)
                    .map(List::getFirst)
                    .orElseThrow(() -> new IOException("No payload found in message."));

            xmlDocument = builder.parse(payload.getContent().getInputStream());

            XPathFactory xPathFactory = XPathFactory.newInstance();
            XPath xPath = xPathFactory.newXPath();
            List<KeyValue<String, String>> pairs = getParameters().value(PARAM_XPATH_KEY_VALUE_PAIRS);
            for (KeyValue<String, String> keyValuePair : pairs) {

                String expression = keyValuePair.getValue();
                String result = (String) xPath.compile(expression).evaluate(xmlDocument, XPathConstants.STRING);
                Boolean stopOnNotFound = getParameters().value(PARAM_STOP_ON_NOT_FOUND);
                if (result.isEmpty()) {
                    if (stopOnNotFound) {
                        throw new NexusException(NexusError.builder()
                                .message("No value found for xpath statement.")
                                .extension("key", keyValuePair.getKey())
                                .extension("xpath", keyValuePair.getValue())
                                .build());
                    } else {
                        return executionContext;
                    }
                }
                String key = keyValuePair.getKey();
                result = trimResult(result, this);
                executionContext.getMessage().getMetaData().add(key, result);
            }

        } catch (ParserConfigurationException | SAXException | IOException | XPathExpressionException e) {
            if (getParameters().value(PARAM_STOP_ON_FAIL)) {
                throw new NexusException(
                        NexusError.builder()
                                .message("Error while executing xpath statement.")
                                .build(),
                        e);
            }
        }
        return executionContext;
    }
}
