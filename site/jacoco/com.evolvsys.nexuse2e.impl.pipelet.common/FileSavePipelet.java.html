package com.evolvsys.nexuse2e.impl.pipelet.common;

import static com.evolvsys.nexuse2e.impl.util.FileSaveUtils.*;

import com.evolvsys.nexuse2e.core.component.Pipelet;
import com.evolvsys.nexuse2e.core.exception.NexusError;
import com.evolvsys.nexuse2e.core.exception.NexusException;
import com.evolvsys.nexuse2e.core.message.context.ExecutionContext;
import com.evolvsys.nexuse2e.core.parameter.ParameterDefinition;
import com.evolvsys.nexuse2e.core.parameter.ParameterType;
import com.evolvsys.nexuse2e.core.utils.PatternReplaceUtils;
import com.evolvsys.nexuse2e.impl.util.FileSaveAware;
import com.evolvsys.nexuse2e.internal.Message;
import java.util.List;
import lombok.extern.slf4j.Slf4j;
import org.apache.tika.utils.StringUtils;

/**
 *
 */
@Slf4j
public class FileSavePipelet extends Pipelet implements FileSaveAware {

    public static final String PARAM_STOP_ON_FAIL = "stop_on_error";

    @Override
    public List<ParameterDefinition> parameterDefinitions() {
        return List.of(
                getTargetDirectory(),
                getTargetFilename(),
                getCreateMissingDirectories(),
                ParameterDefinition.builder()
                        .name(PARAM_STOP_ON_FAIL)
                        .displayName("Stop on Fail")
                        .description("Whether to cancel processing on failure.")
                        .required(true)
                        .type(ParameterType.BOOLEAN)
                        .defaultValue(Boolean.TRUE)
                        .build());
    }

    @Override
    public ExecutionContext execute(ExecutionContext executionContext) throws NexusException {

        LOGGER.trace("running file save sender.");
        Message message = executionContext.getMessage();

        if (StringUtils.isBlank(parameters.value(PARAM_TARGET_DIRECTORY))) {
            throw new IllegalArgumentException("parameter 'Target Directory' must be provided to the sender service!");
        }

        String directoryString = parameters.value(PARAM_TARGET_DIRECTORY);
        LOGGER.trace("replacing directory pattern {}", directoryString);
        directoryString = PatternReplaceUtils.replaceContextTokens(directoryString, executionContext);
        directoryString = directoryString.replace('\\', '/');

        try {
            LOGGER.trace("saving message payloads to directory {}", directoryString);
            String filePattern = parameters.value(PARAM_TARGET_FILENAME);
            persistPayloads(
                    executionContext, message, directoryString, filePattern, parameters.value(PARAM_CREATE_DIRECTORY));

        } catch (Exception e) {

            Boolean stopOnFail = parameters.value(PARAM_STOP_ON_FAIL);
            if (stopOnFail) {

                throw new NexusException(
                        NexusError.builder()
                                .message("Error while writing file to disk")
                                .extension("reason", e.getMessage())
                                .build(),
                        e);
            } else {
                LOGGER.warn("Error while writing file to disk, the error is ignored and processing is continued.", e);
            }
        }
        return executionContext;
    }
}
