package com.evolvsys.nexuse2e.impl.pipelet.common;

import com.evolvsys.nexuse2e.core.component.Pipelet;
import com.evolvsys.nexuse2e.core.exception.NexusError;
import com.evolvsys.nexuse2e.core.exception.NexusException;
import com.evolvsys.nexuse2e.core.message.context.ExecutionContext;
import com.evolvsys.nexuse2e.core.message.payload.content.PayloadContent;
import com.evolvsys.nexuse2e.core.message.payload.content.PayloadContentOutputStream;
import com.evolvsys.nexuse2e.core.parameter.ParameterDefinition;
import com.evolvsys.nexuse2e.core.parameter.ParameterType;
import com.evolvsys.nexuse2e.core.utils.ErrorUtils;
import com.evolvsys.nexuse2e.core.utils.NEXUSe2eFileUtils;
import com.evolvsys.nexuse2e.internal.ContextPayload;
import com.evolvsys.nexuse2e.internal.Message;
import java.io.IOException;
import java.io.InputStream;
import java.nio.charset.Charset;
import java.time.LocalDate;
import java.time.ZonedDateTime;
import java.time.chrono.ChronoZonedDateTime;
import java.time.format.DateTimeFormatter;
import java.util.List;
import java.util.zip.ZipEntry;
import java.util.zip.ZipOutputStream;
import lombok.extern.slf4j.Slf4j;
import org.apache.commons.collections4.CollectionUtils;
import org.apache.commons.lang3.StringUtils;

/**
 * @author Florian Drees
 * @since 09.11.2023
 */
@Slf4j
public class ZipPipelet extends Pipelet {
    public static final String PARAMETER_COMBINE_PAYLOADS = "combine_payloads";
    public static final String PARAMETER_USE_CONTENT_ID = "use_content_id";
    public static final String PARAMETER_PATTERN_FILE_NAME = "file_name_pattern";
    public static final String PARAMETER_PATTERN_TIMESTAMP = "timestamp_pattern";
    public static final String PARAMETER_SEQUENCE_DIGITS = "sequence_digits";
    public static final String PARAMETER_FILE_EXTENSION = "file_extension";

    private boolean combinePayloads;
    private boolean useContentId;
    private int sequenceDigits = 2;
    private String fileNamePattern;
    private String timestampPattern;
    private String fileExtension;

    @Override
    public List<ParameterDefinition> parameterDefinitions() {
        return List.of(
                ParameterDefinition.builder()
                        .name(PARAMETER_COMBINE_PAYLOADS)
                        .displayName("Combine payloads")
                        .description("Create a single zip file to include all payloads")
                        .required(false)
                        .type(ParameterType.BOOLEAN)
                        .defaultValue(Boolean.FALSE)
                        .build(),
                ParameterDefinition.builder()
                        .name(PARAMETER_USE_CONTENT_ID)
                        .displayName("Use contentId")
                        .description(
                                "Whether to use the content id as the file name. This overrides the other file name related settings.")
                        .required(false)
                        .type(ParameterType.BOOLEAN)
                        .defaultValue(Boolean.FALSE)
                        .build(),
                ParameterDefinition.builder()
                        .name(PARAMETER_SEQUENCE_DIGITS)
                        .displayName("Amount of prepended zeroes")
                        .description("Number of digits in sequence counter (prepended zeroes)")
                        .required(false)
                        .type(ParameterType.LONG)
                        .defaultValue(2L)
                        .build(),
                ParameterDefinition.builder()
                        .name(PARAMETER_PATTERN_FILE_NAME)
                        .displayName("File name pattern")
                        .description("The file name pattern to use for payloads")
                        .required(false)
                        .type(ParameterType.STRING)
                        .defaultValue("Payload_${sequence}")
                        .build(),
                ParameterDefinition.builder()
                        .name(PARAMETER_PATTERN_TIMESTAMP)
                        .displayName("Timestamp pattern")
                        .description("The timestamp pattern to use")
                        .required(true)
                        .type(ParameterType.STRING)
                        .defaultValue("yyyy-MM-dd HH.mm.ssSSS")
                        .build(),
                ParameterDefinition.builder()
                        .name(PARAMETER_FILE_EXTENSION)
                        .displayName("File extension")
                        .description("The file extension to append")
                        .required(false)
                        .type(ParameterType.STRING)
                        .defaultValue(".dat")
                        .build());
    }

    private void resolveParameters() {
        if (parameters.containsKey(PARAMETER_PATTERN_FILE_NAME)) {
            fileNamePattern = parameters.value(PARAMETER_PATTERN_FILE_NAME);
        } else {
            fileNamePattern = "Payload_${sequence}";
        }
        if (parameters.containsKey(PARAMETER_COMBINE_PAYLOADS)) {
            combinePayloads = parameters.value(PARAMETER_COMBINE_PAYLOADS);
        }
        if (parameters.containsKey(PARAMETER_USE_CONTENT_ID)) {
            useContentId = parameters.value(PARAMETER_USE_CONTENT_ID);
        }
        if (parameters.containsKey(PARAMETER_PATTERN_TIMESTAMP)) {
            timestampPattern = parameters.value(PARAMETER_PATTERN_TIMESTAMP);
        }
        if (parameters.containsKey(PARAMETER_SEQUENCE_DIGITS)) {
            try {
                sequenceDigits = parameters.value(PARAMETER_SEQUENCE_DIGITS);
            } catch (NumberFormatException e) {
                LOGGER.error(
                        "Value of parameter '{}' is not a number!: {}",
                        PARAMETER_SEQUENCE_DIGITS,
                        ErrorUtils.aggregate(e));
                sequenceDigits = 2;
            }
        }
        String fileExtension = ".dat";
        if (parameters.containsKey(PARAMETER_FILE_EXTENSION)) {
            fileExtension = parameters.value(PARAMETER_FILE_EXTENSION);
        }
        if (!fileExtension.startsWith(".")) {
            fileExtension = "." + fileExtension;
        }
        this.fileExtension = fileExtension;
    }

    @Override
    @SuppressWarnings(
            "PMD.CloseResource") // TODO refactor to try-with-resources, the current code with conditional variables
    // makes this quite hard
    public ExecutionContext execute(ExecutionContext executionContext) throws NexusException {
        resolveParameters();

        Message message = executionContext.getMessage();
        if (CollectionUtils.isNotEmpty(message.getPayloads())) {
            PayloadContentOutputStream outputStream = combinePayloads ? PayloadContent.createOutputStream() : null;
            ZipOutputStream zipOutputStream = outputStream == null ? null : new ZipOutputStream(outputStream);
            if (zipOutputStream != null) {
                zipOutputStream.setMethod(ZipOutputStream.DEFLATED);
            }

            int currentPayloadSequence = 1;

            for (ContextPayload payload : message.getPayloads()) {
                if (!combinePayloads) {
                    outputStream = PayloadContent.createOutputStream();
                    zipOutputStream = new ZipOutputStream(outputStream);
                    zipOutputStream.setMethod(ZipOutputStream.DEFLATED);
                }

                String fileName = getFileName(
                        payload.getContentId(),
                        dateTimeToTimestamp(ZonedDateTime.now(), timestampPattern),
                        currentPayloadSequence++);

                ZipEntry zipEntry = new ZipEntry(fileName);
                zipEntry.setSize(payload.getSize());

                if (StringUtils.isNotBlank(payload.getContentType())) {
                    zipEntry.setComment(payload.getContentType());
                    zipEntry.setExtra(payload.getContentType().getBytes(Charset.forName(payload.getEncoding())));
                }
                try (InputStream inputStream = payload.getContent().getInputStream()) {
                    zipOutputStream.putNextEntry(zipEntry);
                    inputStream.transferTo(zipOutputStream);
                    zipOutputStream.closeEntry();

                    if (!combinePayloads) {
                        zipOutputStream.finish();
                        PayloadContent zipContent = outputStream.finish();

                        payload.setContent(zipContent);
                        payload.setContentType("application/zip");
                    }
                } catch (IOException e) {
                    throw new NexusException(
                            NexusError.builder()
                                    .message("Error compressing payload")
                                    .build(),
                            e);
                }
            }

            if (combinePayloads) {
                try {
                    zipOutputStream.finish();
                } catch (IOException e) {
                    throw new NexusException(
                            NexusError.builder()
                                    .message("Error compressing payload")
                                    .extension("messageId", message.getId())
                                    .build(),
                            e);
                }

                if (CollectionUtils.isNotEmpty(message.getPayloads())) {
                    PayloadContent payloadContent = outputStream.finish();
                    ContextPayload combinedPayload = ContextPayload.builder()
                            .contentType("application/zip")
                            .content(payloadContent)
                            .size(payloadContent.getSize())
                            .contentId("COMBINED_PAYLOADS")
                            .build();
                    message.setPayloads(List.of(combinedPayload));
                } else {
                    LOGGER.warn("No payloads to compress!");
                }
            }
        }
        return executionContext;
    }

    private String dateTimeToTimestamp(ChronoZonedDateTime<LocalDate> zonedDateTime, String timestampPattern) {
        if (timestampPattern == null) {
            return null;
        }
        return zonedDateTime.format(DateTimeFormatter.ofPattern(timestampPattern));
    }

    private String getFileName(String contentId, String timestamp, int sequenceId) {
        String fileName;
        if (useContentId) {
            fileName = contentId;
        } else {
            fileName = fileNamePattern;
            if (fileName.contains("${timestamp}") && StringUtils.isNotBlank(timestamp)) {
                fileName = fileName.replace("${timestamp}", timestamp);
            }
            if (fileName.contains("${sequence}")) {
                fileName = fileName.replace("${sequence}", String.format("%0" + sequenceDigits + "d", sequenceId));
            }
            fileName = fileName.concat(fileExtension);
        }
        return NEXUSe2eFileUtils.cleanFileName(fileName);
    }
}
