package com.evolvsys.nexuse2e.impl.pipelet.common;

import com.evolvsys.nexuse2e.core.component.Pipelet;
import com.evolvsys.nexuse2e.core.message.context.ExecutionContext;
import com.evolvsys.nexuse2e.core.parameter.ParameterDefinition;
import com.evolvsys.nexuse2e.core.parameter.ParameterType;
import com.evolvsys.nexuse2e.core.rhino.RhinoScript;
import java.util.List;

/**
 * @author SvenBarden
 * @since 16.11.2023
 */
public class RhinoPipelet extends Pipelet {
    @Override
    public List<ParameterDefinition> parameterDefinitions() {
        return List.of(ParameterDefinition.builder()
                .required(true)
                .name("script")
                .displayName("Rhino Script")
                .type(ParameterType.LARGE_STRING)
                .description("Rhino script to execute. See help text for details.")
                .build());
    }

    @Override
    public ExecutionContext execute(ExecutionContext executionContext) {
        new RhinoScript()
                .evaluate(
                        parameters.value("script"),
                        String.class,
                        new RhinoScript.Parameter("context", executionContext));
        return executionContext;
    }
}
