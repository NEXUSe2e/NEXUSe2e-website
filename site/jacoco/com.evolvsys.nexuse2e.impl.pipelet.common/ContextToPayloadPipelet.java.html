package com.evolvsys.nexuse2e.impl.pipelet.common;

import com.evolvsys.nexuse2e.core.component.Pipelet;
import com.evolvsys.nexuse2e.core.message.context.ExecutionContext;
import com.evolvsys.nexuse2e.core.parameter.ParameterDefinition;
import com.evolvsys.nexuse2e.internal.ContextPayload;
import com.evolvsys.nexuse2e.internal.Message;
import java.util.List;
import org.apache.commons.collections4.CollectionUtils;
import org.apache.commons.lang3.StringUtils;

/**
 * <p>Pipelet zur Umwandlung einer {@link ContextPayload} zur weiteren Verwendung in eine {@link ContextPayload}.</p>
 *
 * @author Florian Drees
 * @since 23.11.2023
 */
public class ContextToPayloadPipelet extends Pipelet {
    @Override
    public ExecutionContext execute(ExecutionContext executionContext) {
        List<ContextPayload> dataAsList = executionContext.getDataAsList();
        if (CollectionUtils.isNotEmpty(dataAsList)) {
            Message message = executionContext.getMessage();
            for (ContextPayload contextPayload : dataAsList) {
                if (StringUtils.isBlank(contextPayload.getContentId())) {
                    contextPayload.setContentId(
                            String.format("%s_%d", message.getId(), contextPayload.getSequenceNumber()));
                }
                message.getPayloads().add(contextPayload);
            }
        }
        return executionContext;
    }

    @Override
    public List<ParameterDefinition> parameterDefinitions() {
        return List.of();
    }
}
