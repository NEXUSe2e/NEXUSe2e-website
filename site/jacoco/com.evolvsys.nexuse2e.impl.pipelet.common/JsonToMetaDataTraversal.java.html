package com.evolvsys.nexuse2e.impl.pipelet.common;

import com.evolvsys.nexuse2e.core.message.MetaData;
import com.evolvsys.nexuse2e.internal.ContextPayload;
import com.fasterxml.jackson.core.JsonFactory;
import com.fasterxml.jackson.core.JsonParser;
import com.fasterxml.jackson.core.JsonToken;
import edu.umd.cs.findbugs.annotations.SuppressFBWarnings;
import java.io.IOException;
import java.io.InputStream;
import java.util.List;
import java.util.Map;
import java.util.Optional;
import java.util.stream.Collectors;
import javax.annotation.concurrent.NotThreadSafe;
import lombok.AccessLevel;
import lombok.NoArgsConstructor;
import lombok.extern.slf4j.Slf4j;
import org.apache.commons.collections4.KeyValue;
import org.apache.commons.lang3.StringUtils;

/**
 * Utility class to aggregate field values from a given json and return them grouped by the mapped metadata key (and category)
 * in a new {@link MetaData} object.
 *
 * @author Julian Drees
 * @since 27.08.2025
 */
@Slf4j
@NotThreadSafe
@NoArgsConstructor(access = AccessLevel.PRIVATE)
@SuppressFBWarnings("EI_EXPOSE_REP")
public class JsonToMetaDataTraversal {
    private static final String ROOT_TRAVERSAL = "";
    private static final JsonFactory JSON_FACTORY = new JsonFactory();

    private String category;
    private Map<String, String> fieldToKeys;
    private MetaData metaData;

    /**
     * Creates a new instance from the {@link JsonToMetaDataPipelet}.
     *
     * @param jsonToMetaDataPipelet the pipelet used for creating a new instance
     * @return a new instance based of the {@link JsonToMetaDataPipelet}
     */
    public static JsonToMetaDataTraversal forInstance(JsonToMetaDataPipelet jsonToMetaDataPipelet) {
        JsonToMetaDataTraversal traversal = new JsonToMetaDataTraversal();
        traversal.initialize(jsonToMetaDataPipelet);
        return traversal;
    }

    /**
     * Aggregates the json fields based on the mapping (paramater {@link JsonToMetaDataPipelet#PARAM_META_DATA_KEYS})
     * and returns them group by the metadata key in a new {@link MetaData} object.
     * This new {@link MetaData} object can then be used to add the json field metadata to an existing Metadata object using
     * {@link MetaData#add(MetaData)}.
     *
     * @param contextPayload the context payload which holds the json content
     * @return the new MetaData object holding all the json field values
     * @throws IOException thrown when an error occurs while reading the json/json field values
     */
    public MetaData resolve(ContextPayload contextPayload) throws IOException {
        try (InputStream contentStream = contextPayload.getContent().getInputStream();
                JsonParser jsonParser = JSON_FACTORY.createParser(contentStream)) {
            String latestFieldName = null;
            while (jsonParser.nextToken() != null) {
                JsonToken currentToken = jsonParser.getCurrentToken();
                if (currentToken == JsonToken.FIELD_NAME) {
                    latestFieldName = jsonParser.getText();
                    LOGGER.trace("Found json field '{}'.", latestFieldName);
                } else {
                    handleValue(jsonParser, currentToken, latestFieldName);
                }
            }
            return metaData;
        } finally {
            reset();
        }
    }

    /**
     * Reads the value from the latest field.
     * If the field name is null, and the metadata key mapping contains the key {@link #ROOT_TRAVERSAL}, the values will be read from the root array.
     * If the latestFieldName is not null, the value will be mapped via the mapping for the fieldName.
     *
     * @param jsonParser      the {@link JsonParser} for the payload
     * @param currentToken    the current json token
     * @param latestFieldName the latest field name
     * @throws IOException thrown when an error occurs while reading the field value
     */
    private void handleValue(JsonParser jsonParser, JsonToken currentToken, String latestFieldName) throws IOException {
        Object value =
                switch (currentToken) {
                    case VALUE_STRING -> jsonParser.getValueAsString();
                    case VALUE_NUMBER_INT -> jsonParser.getValueAsInt();
                    case VALUE_NUMBER_FLOAT -> jsonParser.getDecimalValue();
                    case VALUE_TRUE -> Boolean.TRUE;
                    case VALUE_FALSE -> Boolean.FALSE;
                    default -> {
                        LOGGER.trace("JsonToken {} is currently not supported, yielding null.", currentToken);
                        yield null;
                    }
                };
        if (value == null) {
            return;
        }

        String metaDataKey = fieldToKeys.get(StringUtils.defaultIfBlank(latestFieldName, ROOT_TRAVERSAL));
        if (metaDataKey != null) {
            LOGGER.debug("Added value '{}' to metadata key list '{}'.", value, metaDataKey);
            metaData.add(category, metaDataKey, value);
        }
    }

    private void initialize(JsonToMetaDataPipelet jsonToMetaDataPipelet) {
        List<KeyValue<String, String>> fieldToKeyMappings =
                jsonToMetaDataPipelet.getParameters().value(JsonToMetaDataPipelet.PARAM_META_DATA_KEYS);

        fieldToKeys = fieldToKeyMappings.stream().collect(Collectors.toMap(KeyValue::getKey, KeyValue::getValue));
        metaData = new MetaData();
        category = Optional.ofNullable(jsonToMetaDataPipelet
                        .getParameters()
                        .value(JsonToMetaDataPipelet.PARAM_META_DATA_CATEGORY, String.class))
                .orElse(MetaData.DEFAULT_CATEGORY);

        LOGGER.info(
                "Successfully configured json traversal: category = {}, fields = {}.", category, fieldToKeys.keySet());
    }

    private void reset() {
        metaData = new MetaData();
    }
}
