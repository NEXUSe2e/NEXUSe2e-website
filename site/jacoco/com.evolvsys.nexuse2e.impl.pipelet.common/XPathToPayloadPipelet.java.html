package com.evolvsys.nexuse2e.impl.pipelet.common;

import com.evolvsys.nexuse2e.core.component.Pipelet;
import com.evolvsys.nexuse2e.core.exception.NexusError;
import com.evolvsys.nexuse2e.core.exception.NexusException;
import com.evolvsys.nexuse2e.core.message.context.ExecutionContext;
import com.evolvsys.nexuse2e.core.message.payload.content.PayloadContent;
import com.evolvsys.nexuse2e.core.parameter.ParameterDefinition;
import com.evolvsys.nexuse2e.core.parameter.ParameterType;
import com.evolvsys.nexuse2e.impl.util.XmlUtils;
import com.evolvsys.nexuse2e.internal.ContextPayload;
import edu.umd.cs.findbugs.annotations.SuppressFBWarnings;
import java.io.IOException;
import java.nio.charset.StandardCharsets;
import java.util.ArrayList;
import java.util.List;
import javax.xml.parsers.ParserConfigurationException;
import javax.xml.xpath.*;
import lombok.extern.slf4j.Slf4j;
import org.w3c.dom.Document;
import org.xml.sax.SAXException;

/**
 * Uses a XPath statement to extract a new payload from the existing payload. The original payload is overwritten and lost after the pipelet.
 *
 * @author SvenBarden
 * @since 24.07.2025
 */
@Slf4j
public class XPathToPayloadPipelet extends Pipelet {
    private static final String PARAM_XPATH_PATTERN = "xpath_pattern";
    private static final String PARAM_PAYLOAD_INDEX = "payload_index";
    private static final String PARAM_TRIM_PAYLOAD_CONTENT = "trim_payload_content";
    private static final String PARAM_PAYLOAD_CONTENT_TYPE = "payload_content_type";
    private static final String PARAM_SAVE_ORIGINAL_PAYLOAD = "save_original_payload";

    private final XPath xpath = XPathFactory.newInstance().newXPath();

    @Override
    public List<ParameterDefinition> parameterDefinitions() {
        return List.of(
                ParameterDefinition.builder()
                        .name(PARAM_XPATH_PATTERN)
                        .type(ParameterType.TRIMMED_STRING)
                        .displayName("XPath Pattern")
                        .description(
                                "The XPath pattern to extract content from the payloads. Only text nodes will be extracted, no full XML nodes.")
                        .required(true)
                        .build(),
                ParameterDefinition.builder()
                        .name(PARAM_PAYLOAD_CONTENT_TYPE)
                        .type(ParameterType.TRIMMED_STRING)
                        .displayName("Payload Content Type")
                        .description("The content type of the extracted payloads.")
                        .defaultValue("text/plain")
                        .required(true)
                        .build(),
                ParameterDefinition.builder()
                        .name(PARAM_TRIM_PAYLOAD_CONTENT)
                        .type(ParameterType.BOOLEAN)
                        .displayName("Trim the extracted content")
                        .description(
                                "If set to true, the extracted content will be trimmed before creating the new payload.")
                        .defaultValue(Boolean.TRUE)
                        .build(),
                ParameterDefinition.builder()
                        .name(PARAM_PAYLOAD_INDEX)
                        .type(ParameterType.LONG)
                        .displayName("Payload Index")
                        .description("The index of the payload to process. If not set, all payloads will be processed.")
                        .build(),
                ParameterDefinition.builder()
                        .name(PARAM_SAVE_ORIGINAL_PAYLOAD)
                        .type(ParameterType.BOOLEAN)
                        .displayName("Save original payload")
                        .description("Saves the original payload in the message meta data.")
                        .defaultValue(Boolean.FALSE)
                        .build());
    }

    @Override
    @SuppressFBWarnings("XPATH_INJECTION")
    public ExecutionContext execute(ExecutionContext executionContext) throws NexusException {
        List<ContextPayload> payloads = executionContext.getMessage().getPayloads();
        Long payloadIndex = parameters.value(PARAM_PAYLOAD_INDEX);

        if (payloadIndex != null && payloadIndex >= 0) {
            LOGGER.trace("Processing payload at index: {}", payloadIndex);
            payloads = List.of(payloads.get(payloadIndex.intValue()));
        } else {
            LOGGER.trace("Processing all payloads");
        }

        XPathExpression xPathExpression;
        try {
            xPathExpression = xpath.compile(parameters.value(PARAM_XPATH_PATTERN));
        } catch (XPathExpressionException e) {
            throw new NexusException(
                    NexusError.builder()
                            .message("XPath expression could not be compiled")
                            .extension("expression", parameters.value(PARAM_XPATH_PATTERN))
                            .build(),
                    e);
        }

        boolean saveOriginalPayload = parameters.value(PARAM_SAVE_ORIGINAL_PAYLOAD);

        List<ContextPayload> newPayloads = new ArrayList<>();
        for (ContextPayload payload : payloads) {
            Document document;
            try {
                document = XmlUtils.getDocument(payload);
            } catch (ParserConfigurationException | SAXException | IOException e) {
                throw new NexusException(
                        NexusError.builder()
                                .message("Could not create XML document from payload")
                                .extension("contentId", payload.getContentId())
                                .extension("type", payload.getContentType())
                                .build(),
                        e);
            }

            if (saveOriginalPayload) {
                LOGGER.trace("Saving original payload with contentId: {}", payload.getContentId());
                executionContext
                        .getMessage()
                        .getMetaData()
                        .add("xpath-to-payload", "original-" + payload.getContentId(), payload.copy());
            } else {
                LOGGER.trace("Not saving original payload with contentId: {}", payload.getContentId());
            }

            try {
                // Serializes all text nodes for that xpath expression and all child nodes
                // note that all xml text will be striped from serialization.
                // if we want to extract the xml nodes itself we need to change the extraction from STRING to NODESET
                String content = (String) xPathExpression.evaluate(document, XPathConstants.STRING);

                if (parameters.value(PARAM_TRIM_PAYLOAD_CONTENT)) {
                    LOGGER.trace("Trimming content for payload with contentId: {}", payload.getContentId());
                    content = content.trim();
                }

                PayloadContent payloadContent = PayloadContent.of(content.getBytes(StandardCharsets.UTF_8));
                ContextPayload newPayload = ContextPayload.builder()
                        .content(payloadContent)
                        .size(payloadContent.getSize())
                        .contentType(parameters.value(PARAM_PAYLOAD_CONTENT_TYPE))
                        .encoding(payload.getEncoding())
                        .contentId(payload.getContentId())
                        .build();

                newPayloads.add(newPayload);

            } catch (XPathExpressionException e) {
                throw new NexusException(
                        NexusError.builder()
                                .message("Could not evaluate xpath expression on payload")
                                .extension("xpath", parameters.value(PARAM_XPATH_PATTERN))
                                .extension("contentId", payload.getContentId())
                                .build(),
                        e);
            }
        }

        executionContext.getMessage().setPayloads(newPayloads);

        return executionContext;
    }
}
