package com.evolvsys.nexuse2e.impl.pipelet.common;

import com.evolvsys.nexuse2e.core.component.Pipelet;
import com.evolvsys.nexuse2e.core.exception.NexusError;
import com.evolvsys.nexuse2e.core.exception.NexusException;
import com.evolvsys.nexuse2e.core.message.MetaData;
import com.evolvsys.nexuse2e.core.message.context.ExecutionContext;
import com.evolvsys.nexuse2e.core.parameter.ParameterDefinition;
import com.evolvsys.nexuse2e.core.parameter.ParameterType;
import com.evolvsys.nexuse2e.internal.ContextPayload;
import com.evolvsys.nexuse2e.internal.Message;
import java.io.IOException;
import java.util.List;
import lombok.extern.slf4j.Slf4j;

/**
 * Pipelet to copy json field values to the message metadata.
 * Supports both root arrays and specific field values.
 *
 * @author Julian Drees
 * @since 27.08.2025
 */
@Slf4j
public class JsonToMetaDataPipelet extends Pipelet {
    public static final String PARAM_META_DATA_CATEGORY = "meta_data_category";
    public static final String PARAM_META_DATA_KEYS = "meta_data_keys";
    public static final String PARAM_FAIL_ON_ERROR = "fail_on_error";

    @Override
    public List<ParameterDefinition> parameterDefinitions() {
        return List.of(
                ParameterDefinition.builder()
                        .name(PARAM_META_DATA_CATEGORY)
                        .displayName("Metadata target category")
                        .description(
                                "The metadata category where the extracted values should be stored in. Use '%s' if you want them to be stored in the default category."
                                        .formatted(MetaData.DEFAULT_CATEGORY))
                        .type(ParameterType.TRIMMED_STRING)
                        .defaultValue(MetaData.DEFAULT_CATEGORY)
                        .build(),
                ParameterDefinition.builder()
                        .name(PARAM_META_DATA_KEYS)
                        .displayName("Metadata target keys")
                        .description(
                                "The json property and it's metadata key where the value should be stored in. The key is the json field, the value the target metadata key. Use \"\" for a root string array.")
                        .type(ParameterType.KEY_VALUE_PAIR_LIST_TRIMMED)
                        .build(),
                ParameterDefinition.builder()
                        .name(PARAM_FAIL_ON_ERROR)
                        .displayName("Fail on error")
                        .description("If the processing should stop when one of the payloads could not be read.")
                        .type(ParameterType.BOOLEAN)
                        .defaultValue(Boolean.TRUE)
                        .build());
    }

    @Override
    public ExecutionContext execute(ExecutionContext executionContext) throws NexusException {
        Message message = executionContext.getMessage();
        MetaData metaData = message.getMetaData();
        boolean failOnError = getParameters().value(PARAM_FAIL_ON_ERROR);

        JsonToMetaDataTraversal traversal = JsonToMetaDataTraversal.forInstance(this);
        for (ContextPayload payload : message.getPayloads()) {
            try {
                MetaData additionalMetaData = traversal.resolve(payload);
                metaData.add(additionalMetaData);
                LOGGER.info("Successfully extracted metadata values from payload '{}'.", payload.getContentId());
            } catch (IOException ex) {
                LOGGER.error("Could not read or extract metadata from payload '{}'.", payload.getContentId());
                if (failOnError) {
                    throw new NexusException(
                            NexusError.builder()
                                    .message("Could not read json field values from payload " + payload.getContentId())
                                    .build(),
                            ex);
                }
            }
        }
        return executionContext;
    }
}
