package com.evolvsys.nexuse2e.impl.pipelet.common;

import com.evolvsys.nexuse2e.core.component.Pipelet;
import com.evolvsys.nexuse2e.core.exception.NexusException;
import com.evolvsys.nexuse2e.core.message.context.ExecutionContext;
import com.evolvsys.nexuse2e.core.parameter.ParameterDefinition;
import com.evolvsys.nexuse2e.core.parameter.ParameterType;
import java.util.List;
import org.apache.commons.collections4.CollectionUtils;
import org.apache.commons.collections4.KeyValue;
import org.apache.commons.lang3.StringUtils;

/**
 * @author SvenBarden
 * @since 05.07.2024
 */
public class MetaDataPipelet extends Pipelet {
    private static final String PARAM_CATEGORY = "category";
    private static final String PARAM_META_DATAS = "meta_datas";

    @Override
    public List<ParameterDefinition> parameterDefinitions() {
        return List.of(
                ParameterDefinition.builder()
                        .name(PARAM_CATEGORY)
                        .displayName("Category")
                        .description("The optional category to put the meta data in")
                        .type(ParameterType.TRIMMED_STRING)
                        .build(),
                ParameterDefinition.builder()
                        .name(PARAM_META_DATAS)
                        .displayName("Meta datas")
                        .description("List of meta data entries to set in the message")
                        .type(ParameterType.KEY_VALUE_PAIR_LIST_TRIMMED)
                        .build());
    }

    @Override
    public ExecutionContext execute(ExecutionContext executionContext) throws NexusException {
        List<KeyValue<String, String>> metaDatas = parameters.value(PARAM_META_DATAS);

        if (CollectionUtils.isNotEmpty(metaDatas)) {
            String category = parameters.value(PARAM_CATEGORY);
            for (KeyValue<String, String> metaData : metaDatas) {
                if (StringUtils.isBlank(category)) {
                    executionContext.getMessage().getMetaData().add(metaData.getKey(), metaData.getValue());
                } else {
                    executionContext.getMessage().getMetaData().add(category, metaData.getKey(), metaData.getValue());
                }
            }
        }

        return executionContext;
    }
}
