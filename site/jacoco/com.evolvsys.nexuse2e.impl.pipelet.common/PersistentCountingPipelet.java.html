package com.evolvsys.nexuse2e.impl.pipelet.common;

import com.evolvsys.nexuse2e.core.component.Pipelet;
import com.evolvsys.nexuse2e.core.exception.NexusException;
import com.evolvsys.nexuse2e.core.keyvalue.persistence.PersistentKeyValue;
import com.evolvsys.nexuse2e.core.message.context.ExecutionContext;
import com.evolvsys.nexuse2e.core.parameter.ParameterDefinition;
import com.evolvsys.nexuse2e.core.parameter.ParameterType;
import java.util.List;
import lombok.extern.slf4j.Slf4j;
import org.apache.commons.lang3.ObjectUtils;

/**
 * @author SvenBarden
 * @since 18.06.2024
 */
@Slf4j
public class PersistentCountingPipelet extends Pipelet {
    private static final String PARAM_CATEGORY = "category";
    private static final String PARAM_KEY = "key";
    private static final String PARAM_INITIAL_VALUE = "initial_value";
    private static final String PARAM_INCREMENT = "increment";

    @Override
    public ExecutionContext execute(ExecutionContext executionContext) throws NexusException {
        PersistentKeyValue persistentKeyValue = executionContext.getPersistentKeyValue();
        String category = parameters.value(PARAM_CATEGORY);
        String key = parameters.value(PARAM_KEY);
        Long increment = ObjectUtils.defaultIfNull(parameters.value(PARAM_INCREMENT), 1L);
        Long value = persistentKeyValue
                .getAsLong(category, key)
                .orElseGet(() -> ObjectUtils.defaultIfNull(parameters.value(PARAM_INITIAL_VALUE), 0L));
        LOGGER.debug("incrementing {} - {} from {} to {}", category, key, value, value + increment);
        persistentKeyValue.setLong(category, key, value + increment);
        return executionContext;
    }

    @Override
    public List<ParameterDefinition> parameterDefinitions() {
        return List.of(
                ParameterDefinition.builder()
                        .name(PARAM_CATEGORY)
                        .displayName("Category")
                        .description("The category name to update")
                        .type(ParameterType.STRING)
                        .required(true)
                        .build(),
                ParameterDefinition.builder()
                        .name(PARAM_KEY)
                        .displayName("Key")
                        .description("The key to update")
                        .type(ParameterType.STRING)
                        .required(true)
                        .build(),
                ParameterDefinition.builder()
                        .name(PARAM_INITIAL_VALUE)
                        .displayName("Initial value")
                        .description(
                                "The initial value to set. The increment will be applied to this value on the first execution.")
                        .type(ParameterType.LONG)
                        .defaultValue(0L)
                        .build(),
                ParameterDefinition.builder()
                        .name(PARAM_INCREMENT)
                        .displayName("Increment")
                        .description("The increment to increase the value on each execution")
                        .type(ParameterType.LONG)
                        .defaultValue(1L)
                        .build());
    }
}
