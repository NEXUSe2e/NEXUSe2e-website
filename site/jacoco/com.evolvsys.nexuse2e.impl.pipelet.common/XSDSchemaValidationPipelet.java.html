package com.evolvsys.nexuse2e.impl.pipelet.common;

import com.evolvsys.nexuse2e.core.component.Pipelet;
import com.evolvsys.nexuse2e.core.exception.NexusError;
import com.evolvsys.nexuse2e.core.exception.NexusException;
import com.evolvsys.nexuse2e.core.message.context.ExecutionContext;
import com.evolvsys.nexuse2e.core.parameter.ParameterDefinition;
import com.evolvsys.nexuse2e.core.parameter.ParameterType;
import edu.umd.cs.findbugs.annotations.SuppressFBWarnings;
import java.io.ByteArrayInputStream;
import java.io.File;
import java.io.IOException;
import java.io.InputStream;
import java.nio.charset.StandardCharsets;
import java.util.List;
import javax.xml.XMLConstants;
import javax.xml.transform.stream.StreamSource;
import javax.xml.validation.Schema;
import javax.xml.validation.SchemaFactory;
import javax.xml.validation.Validator;
import lombok.extern.slf4j.Slf4j;
import org.apache.commons.io.FileUtils;
import org.apache.commons.lang3.StringUtils;
import org.xml.sax.SAXException;

/**
 *
 */
@Slf4j
public class XSDSchemaValidationPipelet extends Pipelet {

    public static final String PARAM_SCHEMA_FILE_PATH = "schema_file_path";
    public static final String PARAM_SCHEMA_FILE_CONTENT = "schema_file_content";
    public static final String PARAM_DISABLE_VALIDATION = "disable_validation";

    @Override
    public List<ParameterDefinition> parameterDefinitions() {
        return List.of(
                ParameterDefinition.builder()
                        .name(PARAM_SCHEMA_FILE_PATH)
                        .displayName("File Path to XSD Schema")
                        .description("The full path to the XSD schema file for payload validation.")
                        .required(false)
                        .type(ParameterType.TRIMMED_STRING)
                        .build(),
                ParameterDefinition.builder()
                        .name(PARAM_SCHEMA_FILE_CONTENT)
                        .displayName("XSD Schema Content")
                        .description(
                                "Holds the text content of the XSD Schema. If configured, its used instead of the file path.")
                        .required(false)
                        .type(ParameterType.LARGE_STRING)
                        .build(),
                ParameterDefinition.builder()
                        .name(PARAM_DISABLE_VALIDATION)
                        .displayName("Temporary disable validation")
                        .description("Used to disable validation without loosing the whole configuration.")
                        .required(true)
                        .defaultValue(Boolean.FALSE)
                        .type(ParameterType.BOOLEAN)
                        .build());
    }

    @Override
    public ExecutionContext execute(ExecutionContext executionContext) throws NexusException {

        Boolean disableValidation = getParameters().value(PARAM_DISABLE_VALIDATION);
        if (disableValidation != null && disableValidation) {
            LOGGER.info("Validation disabled");
            return executionContext;
        }

        SchemaFactory factory = SchemaFactory.newInstance(XMLConstants.W3C_XML_SCHEMA_NS_URI);

        try (InputStream schemaFile = getSchemaContent()) {
            factory.setProperty(XMLConstants.ACCESS_EXTERNAL_DTD, "");
            factory.setProperty(XMLConstants.ACCESS_EXTERNAL_SCHEMA, "");
            StreamSource sourceSource = new StreamSource(schemaFile);
            Schema schema = factory.newSchema(sourceSource);
            Validator validator = schema.newValidator();

            validator.validate(new StreamSource(executionContext
                    .getMessage()
                    .getPayloads()
                    .getFirst()
                    .getContent()
                    .getInputStream()));
            LOGGER.info("Payload validation successful");
        } catch (SAXException | IOException e) {
            throw new NexusException(
                    NexusError.builder().message("Payload validation failed").build(), e);
        }

        return executionContext;
    }

    @SuppressFBWarnings("PATH_TRAVERSAL_IN")
    private InputStream getSchemaContent() throws NexusException, IOException {
        String schemaFileString = getParameters().value(PARAM_SCHEMA_FILE_CONTENT);
        if (StringUtils.isNotBlank(schemaFileString)) {
            return new ByteArrayInputStream(schemaFileString.getBytes(StandardCharsets.UTF_8));
        } else {
            String schemaFilePath = getParameters().value(PARAM_SCHEMA_FILE_PATH);
            if (StringUtils.isNotBlank(schemaFilePath)) {
                return FileUtils.openInputStream(new File(schemaFilePath));
            }
        }
        throw new NexusException(NexusError.builder()
                .message("no valid schema configuration found")
                .build());
    }
}
