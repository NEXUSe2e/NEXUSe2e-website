package com.evolvsys.nexuse2e.impl.pipelet.common;

import com.evolvsys.nexuse2e.core.component.Pipelet;
import com.evolvsys.nexuse2e.core.exception.NexusException;
import com.evolvsys.nexuse2e.core.message.MetaData;
import com.evolvsys.nexuse2e.core.message.context.ExecutionContext;
import com.evolvsys.nexuse2e.core.parameter.ParameterDefinition;
import com.evolvsys.nexuse2e.core.parameter.ParameterType;
import com.evolvsys.nexuse2e.internal.ContextPayload;
import com.evolvsys.nexuse2e.internal.Message;
import java.util.List;
import java.util.Set;
import lombok.extern.slf4j.Slf4j;
import org.apache.commons.collections4.CollectionUtils;
import org.apache.commons.lang3.ObjectUtils;
import org.apache.commons.lang3.StringUtils;

/**
 * Copies the metadata from the payloads to the message.
 *
 * @author SvenBarden
 * @since 11.04.2025
 */
@Slf4j
public class CopyPayloadMetaDataPipelet extends Pipelet {
    private static final String PARAM_CATEGORY = "category";
    private static final String PARAM_META_DATA_KEYS = "meta_data_keys";

    @Override
    public List<ParameterDefinition> parameterDefinitions() {
        return List.of(
                ParameterDefinition.builder()
                        .name(PARAM_CATEGORY)
                        .displayName("Category")
                        .description(
                                "The metadata category to copy from. If left blank all categories will be copied. Use '%s' if you only want to copy the default metadata category."
                                        .formatted(MetaData.DEFAULT_CATEGORY))
                        .type(ParameterType.TRIMMED_STRING)
                        .defaultValue(MetaData.DEFAULT_CATEGORY)
                        .build(),
                ParameterDefinition.builder()
                        .name(PARAM_META_DATA_KEYS)
                        .displayName("Meta data keys")
                        .description("Metadata keys to copy. If the parameter is empty all keys will be copied.")
                        .type(ParameterType.STRING_LIST)
                        .build());
    }

    protected void copyCategory(MetaData source, MetaData target, String category) {
        Set<String> keys = source.keys(category);
        List<String> metaDataKeys = ObjectUtils.defaultIfNull(parameters.value(PARAM_META_DATA_KEYS), List.of());
        boolean allKeys = CollectionUtils.isEmpty(metaDataKeys);
        if (allKeys) {
            LOGGER.trace("copying all keys {} from category {}", keys, category);
        } else {
            LOGGER.trace("copying keys {} from {} from category {}", metaDataKeys, keys, category);
        }

        for (String key : keys) {
            if (allKeys || metaDataKeys.contains(key)) {
                target.addAll(category, key, source.get(category, key));
            }
        }
    }

    @Override
    public ExecutionContext execute(ExecutionContext executionContext) throws NexusException {
        String category = parameters.value(PARAM_CATEGORY);
        boolean allCategories = StringUtils.isBlank(category);

        LOGGER.debug("copying meta data from payloads to message with parameters: {}", parameters);

        Message message = executionContext.getMessage();
        MetaData messageMetaData = message.getMetaData();
        for (ContextPayload payload : message.getPayloads()) {
            MetaData payloadMetaData = payload.getMetaData();

            if (allCategories) {
                LOGGER.trace("copying all categories from payload {}", payload);
                for (String payloadCategory : payloadMetaData.categories()) {
                    copyCategory(payloadMetaData, messageMetaData, payloadCategory);
                }
            } else {
                LOGGER.trace("copying category {} from payload {}", category, payload);
                copyCategory(payloadMetaData, messageMetaData, category);
            }
        }

        return executionContext;
    }
}
