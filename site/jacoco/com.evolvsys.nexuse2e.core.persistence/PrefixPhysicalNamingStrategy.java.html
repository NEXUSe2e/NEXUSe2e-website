package com.evolvsys.nexuse2e.core.persistence;

import org.apache.commons.lang3.StringUtils;
import org.hibernate.boot.model.naming.CamelCaseToUnderscoresNamingStrategy;
import org.hibernate.boot.model.naming.Identifier;
import org.hibernate.engine.jdbc.env.spi.JdbcEnvironment;

/**
 * @author SvenBarden
 * @since 28.02.2024
 */
public class PrefixPhysicalNamingStrategy extends CamelCaseToUnderscoresNamingStrategy {
    private static final String PREFIX = "NXS_";

    @Override
    public Identifier toPhysicalTableName(Identifier logicalName, JdbcEnvironment jdbcEnvironment) {
        Identifier physicalTableName = super.toPhysicalTableName(logicalName, jdbcEnvironment);
        if (StringUtils.startsWithIgnoreCase(physicalTableName.getText(), PREFIX)) { // required for audit tables
            return physicalTableName;
        }
        return new Identifier(PREFIX + physicalTableName.getText(), physicalTableName.isQuoted());
    }
}
