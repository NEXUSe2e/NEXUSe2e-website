package com.evolvsys.nexuse2e.impl.pipelet.http.soap;

import com.evolvsys.nexuse2e.core.component.Pipelet;
import com.evolvsys.nexuse2e.core.exception.NexusError;
import com.evolvsys.nexuse2e.core.exception.NexusException;
import com.evolvsys.nexuse2e.core.message.context.ExecutionContext;
import com.evolvsys.nexuse2e.core.message.payload.content.PayloadContent;
import com.evolvsys.nexuse2e.core.message.payload.content.PayloadContentOutputStream;
import com.evolvsys.nexuse2e.core.parameter.ParameterDefinition;
import com.evolvsys.nexuse2e.core.parameter.ParameterType;
import com.evolvsys.nexuse2e.internal.ContextPayload;
import edu.umd.cs.findbugs.annotations.SuppressFBWarnings;
import jakarta.xml.soap.MessageFactory;
import jakarta.xml.soap.SOAPBody;
import jakarta.xml.soap.SOAPException;
import jakarta.xml.soap.SOAPMessage;
import java.io.IOException;
import java.util.List;
import javax.xml.transform.*;
import javax.xml.transform.dom.DOMSource;
import javax.xml.transform.stream.StreamResult;
import lombok.extern.slf4j.Slf4j;
import org.apache.commons.lang3.StringUtils;
import org.w3c.dom.Document;

/**
 * Parses a SOAP message and extracts the SOAP body as an XML document and adds it to the message payloads.
 *
 * @author SvenBarden
 * @since 24.07.2025
 */
@Slf4j
public class SoapUnpackagerPipelet extends Pipelet {
    private static final String ENCODING = "encoding";

    @Override
    public List<ParameterDefinition> parameterDefinitions() {
        return List.of(ParameterDefinition.builder()
                .name(ENCODING)
                .type(ParameterType.TRIMMED_STRING)
                .displayName("Encoding")
                .description(
                        "The encoding of the extracted SOAP body payload. Defaults to the encoding of the SOAP message if not set.")
                .build());
    }

    @Override
    @SuppressFBWarnings({"XXE_XSLT_TRANSFORM_FACTORY", "XXE_DTD_TRANSFORM_FACTORY"})
    public ExecutionContext execute(ExecutionContext executionContext) throws NexusException {
        ContextPayload payload = executionContext.getData();

        String encoding = StringUtils.firstNonBlank(parameters.value(ENCODING), payload.getEncoding(), "UTF-8");
        LOGGER.trace("Using encoding '{}' for SOAP body extraction", encoding);

        SOAPMessage soapMessage;
        try {
            soapMessage = MessageFactory.newInstance()
                    .createMessage(null, payload.getContent().getInputStream());
        } catch (IOException | SOAPException e) {
            throw new NexusException(
                    NexusError.builder()
                            .message("Could not create message factory to parse soap message")
                            .build(),
                    e);
        }

        SOAPBody soapBody;
        try {
            soapBody = soapMessage.getSOAPBody();
        } catch (SOAPException e) {
            throw new NexusException(
                    NexusError.builder()
                            .message("Could not read soap body from message")
                            .build(),
                    e);
        }
        if (soapBody.hasFault()) {
            throw new NexusException(NexusError.builder()
                    .message("Faults cannot be processed as business messages")
                    .build());
        }

        Document document;
        try {
            document = soapBody.extractContentAsDocument();
        } catch (SOAPException e) {
            throw new NexusException(
                    NexusError.builder()
                            .message("Could not extract soap body as xml document")
                            .build(),
                    e);
        }
        TransformerFactory transformerFactory = TransformerFactory.newInstance();
        Transformer transformer;
        try {
            transformer = transformerFactory.newTransformer();
        } catch (TransformerConfigurationException e) {
            throw new NexusException(
                    NexusError.builder()
                            .message("Could not create new xml transformer")
                            .build(),
                    e);
        }

        PayloadContent payloadContent;
        try (PayloadContentOutputStream outputStream = PayloadContent.createOutputStream()) {
            try {
                transformer.setOutputProperty(OutputKeys.ENCODING, encoding);
                transformer.transform(new DOMSource(document), new StreamResult(outputStream));
            } catch (TransformerException e) {
                throw new NexusException(
                        NexusError.builder()
                                .message("Could not transform soap body document to output stream")
                                .build(),
                        e);
            }
            payloadContent = outputStream.finish();
        } catch (IOException e) {
            throw new NexusException(
                    NexusError.builder()
                            .message("Error while reading payload from SOAP body")
                            .build(),
                    e);
        }
        ContextPayload contextPayload = ContextPayload.builder()
                .content(payloadContent)
                .size(payloadContent.getSize())
                .contentType("text/xml")
                .encoding(encoding)
                .contentId("soap-body")
                .build();
        executionContext.getMessage().getPayloads().add(contextPayload);

        return executionContext;
    }
}
