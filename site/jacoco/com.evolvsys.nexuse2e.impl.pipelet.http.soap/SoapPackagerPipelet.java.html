package com.evolvsys.nexuse2e.impl.pipelet.http.soap;

import com.evolvsys.nexuse2e.core.component.Pipelet;
import com.evolvsys.nexuse2e.core.constants.HttpConstants;
import com.evolvsys.nexuse2e.core.exception.NexusError;
import com.evolvsys.nexuse2e.core.exception.NexusException;
import com.evolvsys.nexuse2e.core.message.context.ExecutionContext;
import com.evolvsys.nexuse2e.core.message.payload.content.PayloadContent;
import com.evolvsys.nexuse2e.core.message.payload.content.PayloadContentOutputStream;
import com.evolvsys.nexuse2e.core.parameter.ParameterDefinition;
import com.evolvsys.nexuse2e.core.parameter.ParameterType;
import com.evolvsys.nexuse2e.internal.ContextPayload;
import edu.umd.cs.findbugs.annotations.SuppressFBWarnings;
import jakarta.xml.soap.MessageFactory;
import jakarta.xml.soap.SOAPBody;
import jakarta.xml.soap.SOAPException;
import jakarta.xml.soap.SOAPMessage;
import java.io.IOException;
import java.util.List;
import javax.xml.parsers.DocumentBuilderFactory;
import javax.xml.parsers.ParserConfigurationException;
import org.springframework.http.HttpHeaders;
import org.w3c.dom.Document;
import org.xml.sax.SAXException;

/**
 * Creates a SOAP message with a SOAP body containing the XML payload from the message. The SOAP message is written to the ExecutionContext's data.
 *
 * @author SvenBarden
 * @since 25.07.2025
 */
public class SoapPackagerPipelet extends Pipelet {
    private static final String SOAP_ACTION = "soap_action";

    @Override
    public List<ParameterDefinition> parameterDefinitions() {
        return List.of(ParameterDefinition.builder()
                .name(SOAP_ACTION)
                .type(ParameterType.TRIMMED_STRING)
                .displayName("SOAP Action")
                .description("The SOAP action to set in the HTTP headers.")
                .build());
    }

    @Override
    @SuppressFBWarnings({"XXE_DOCUMENT", "CFS_CONFUSING_FUNCTION_SEMANTICS"})
    public ExecutionContext execute(ExecutionContext executionContext) throws NexusException {

        SOAPMessage soapMessage;
        try {
            soapMessage = MessageFactory.newInstance().createMessage();
        } catch (SOAPException e) {
            throw new NexusException(
                    NexusError.builder()
                            .message("Could not create soap message factory")
                            .build(),
                    e);
        }

        SOAPBody soapBody;
        try {
            soapBody = soapMessage.getSOAPBody();
        } catch (SOAPException e) {
            throw new NexusException(
                    NexusError.builder().message("Could not create soap body").build(), e);
        }

        ContextPayload payload = executionContext.getData();

        DocumentBuilderFactory builderFactory = DocumentBuilderFactory.newInstance();
        builderFactory.setNamespaceAware(true);
        Document document;
        try {
            document = builderFactory
                    .newDocumentBuilder()
                    .parse(payload.getContent().getInputStream());
        } catch (SAXException | IOException | ParserConfigurationException e) {
            throw new NexusException(
                    NexusError.builder()
                            .message("Could not create document from message payload")
                            .build(),
                    e);
        }

        try {
            soapBody.addDocument(document);
        } catch (SOAPException e) {
            throw new NexusException(
                    NexusError.builder()
                            .message("Could not add payload to soap body")
                            .build(),
                    e);
        }

        PayloadContent payloadContent;
        try (PayloadContentOutputStream outputStream = PayloadContent.createOutputStream()) {
            try {
                soapMessage.writeTo(outputStream);
            } catch (SOAPException | IOException e) {
                throw new NexusException(
                        NexusError.builder()
                                .message("Could not write soap message to output stream")
                                .build(),
                        e);
            }

            payloadContent = outputStream.finish();
        } catch (IOException e) {
            throw new NexusException(
                    NexusError.builder()
                            .message("Error while writing soap message")
                            .build(),
                    e);
        }

        ContextPayload contextPayload = ContextPayload.builder()
                .size(payloadContent.getSize())
                .content(payloadContent)
                .contentType("text/xml")
                .encoding("utf-8")
                .build();
        executionContext.setData(contextPayload);

        HttpHeaders httpHeaders = new HttpHeaders();
        httpHeaders.set("SOAPAction", parameters.value(SOAP_ACTION));
        executionContext.getMessage().getMetaData().set(HttpConstants.ADDITIONAL_HEADERS, httpHeaders);

        return executionContext;
    }
}
