package com.evolvsys.nexuse2e.impl.pipelet.http.soap;

import com.evolvsys.nexuse2e.core.component.Pipelet;
import com.evolvsys.nexuse2e.core.exception.NexusError;
import com.evolvsys.nexuse2e.core.exception.NexusException;
import com.evolvsys.nexuse2e.core.message.context.ExecutionContext;
import com.evolvsys.nexuse2e.core.parameter.ParameterDefinition;
import com.evolvsys.nexuse2e.core.parameter.ParameterType;
import com.evolvsys.nexuse2e.internal.ContextPayload;
import edu.umd.cs.findbugs.annotations.SuppressFBWarnings;
import java.io.IOException;
import java.util.List;
import javax.security.auth.callback.Callback;
import javax.security.auth.callback.CallbackHandler;
import javax.xml.parsers.DocumentBuilder;
import javax.xml.parsers.DocumentBuilderFactory;
import javax.xml.parsers.ParserConfigurationException;
import lombok.extern.slf4j.Slf4j;
import org.apache.wss4j.common.ext.WSPasswordCallback;
import org.apache.wss4j.common.ext.WSSecurityException;
import org.apache.wss4j.dom.engine.WSSecurityEngine;
import org.apache.wss4j.dom.engine.WSSecurityEngineResult;
import org.apache.wss4j.dom.handler.RequestData;
import org.apache.wss4j.dom.handler.WSHandlerResult;
import org.w3c.dom.Document;
import org.xml.sax.SAXException;

/**
 * Valides the WS-Security header in a SOAP message using a UsernameToken.
 *
 * @author SvenBarden
 * @since 24.07.2025
 */
@Slf4j
public class WsSecurityRulePipelet extends Pipelet {
    private static final String PARAM_USER_NAME = "user_name";
    private static final String PARAM_PASSWORD = "password";

    @Override
    public List<ParameterDefinition> parameterDefinitions() {
        return List.of(
                ParameterDefinition.builder()
                        .name(PARAM_USER_NAME)
                        .type(ParameterType.TRIMMED_STRING)
                        .displayName("User Name")
                        .description("The user name for the WS-Security UsernameToken.")
                        .required(true)
                        .build(),
                ParameterDefinition.builder()
                        .name(PARAM_PASSWORD)
                        .type(ParameterType.PASSWORD)
                        .displayName("Password")
                        .description("The password for the WS-Security UsernameToken.")
                        .required(true)
                        .build());
    }

    @Override
    @SuppressFBWarnings("XXE_DOCUMENT")
    public ExecutionContext execute(ExecutionContext executionContext) throws NexusException {
        LOGGER.debug("Validating WS-Security");
        ContextPayload payload = executionContext.getData();

        DocumentBuilderFactory builderFactory = DocumentBuilderFactory.newInstance();
        builderFactory.setNamespaceAware(true);

        DocumentBuilder documentBuilder;
        try {
            documentBuilder = builderFactory.newDocumentBuilder();
        } catch (ParserConfigurationException e) {
            throw new NexusException(
                    NexusError.builder()
                            .message("Could not create document builder for WSSecurity validation")
                            .build(),
                    e);
        }
        Document document;
        try {
            document = documentBuilder.parse(payload.getContent().getInputStream());
        } catch (SAXException | IOException e) {
            throw new NexusException(
                    NexusError.builder()
                            .message("Could not parse content to create SOAP Document")
                            .build(),
                    e);
        }

        CallbackHandler callbackHandler = new CallbackHandler() {
            @Override
            public void handle(Callback[] callbacks) {
                for (Callback callback : callbacks) {
                    if (callback instanceof WSPasswordCallback wsPasswordCallback
                            && wsPasswordCallback.getUsage() == WSPasswordCallback.USERNAME_TOKEN) {
                        String identifier = wsPasswordCallback.getIdentifier();
                        LOGGER.trace("verifying username token with identifier: {}", identifier);
                        if (identifier.equals(parameters.value(PARAM_USER_NAME))) {
                            wsPasswordCallback.setPassword(parameters.value(PARAM_PASSWORD));
                        }
                    }
                }
            }
        };

        WSSecurityEngine securityEngine = new WSSecurityEngine();
        RequestData requestData = new RequestData();
        requestData.setAllowUsernameTokenNoPassword(false);
        requestData.setCallbackHandler(callbackHandler);
        try {
            WSHandlerResult wsHandlerResult = securityEngine.processSecurityHeader(document, requestData);
            if (wsHandlerResult == null) {
                throw new NexusException(NexusError.builder()
                        .message("No WSSecurity header found in payload")
                        .build());
            }
            if (LOGGER.isTraceEnabled()) {
                for (@SuppressWarnings("PMD.LooseCoupling")
                WSSecurityEngineResult result : wsHandlerResult.getResults()) {
                    LOGGER.trace("WSSecurity result: {}", result);
                }
            }
        } catch (WSSecurityException e) {
            throw new NexusException(
                    NexusError.builder().message("WSSecurity validation failed").build(), e);
        }

        return executionContext;
    }
}
