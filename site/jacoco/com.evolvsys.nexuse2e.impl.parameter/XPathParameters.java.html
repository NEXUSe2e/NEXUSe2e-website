package com.evolvsys.nexuse2e.impl.parameter;

import com.evolvsys.nexuse2e.core.component.Configurable;
import com.evolvsys.nexuse2e.core.parameter.ParameterDefinition;
import com.evolvsys.nexuse2e.core.parameter.ParameterType;
import java.util.function.Supplier;

/**
 * @author SvenBarden
 * @since 16.09.2024
 */
public interface XPathParameters {
    String PARAM_TRIM_RESULT = "trim_result";
    String PARAM_STOP_ON_FAIL = "stop_on_error";
    String PARAM_STOP_ON_NOT_FOUND = "stop_on_not_found";

    Supplier<ParameterDefinition.ParameterDefinitionBuilder> TRIM_RESULT = () -> ParameterDefinition.builder()
            .name(PARAM_TRIM_RESULT)
            .displayName("Strip whitespace from result")
            .description("Whether to remove all leading or trailing whitespace characters from the result.")
            .required(true)
            .type(ParameterType.BOOLEAN)
            .defaultValue(Boolean.TRUE);
    Supplier<ParameterDefinition.ParameterDefinitionBuilder> STOP_ON_FAIL = () -> ParameterDefinition.builder()
            .name(PARAM_STOP_ON_FAIL)
            .displayName("Stop on Fail")
            .description("Whether to cancel processing on failure.")
            .required(true)
            .type(ParameterType.BOOLEAN)
            .defaultValue(Boolean.TRUE);
    Supplier<ParameterDefinition.ParameterDefinitionBuilder> STOP_ON_NOT_FOUND = () -> ParameterDefinition.builder()
            .name(PARAM_STOP_ON_NOT_FOUND)
            .displayName("Stop on empty or not found value")
            .description("Whether to cancel further processing when the result value for the xpath statement is empty.")
            .required(true)
            .type(ParameterType.BOOLEAN)
            .defaultValue(Boolean.TRUE);

    default String trimResult(String s, Configurable configurable) {
        Boolean trimResult = configurable.getParameters().value(PARAM_TRIM_RESULT);
        if (trimResult) {
            s = s.trim();
        }
        return s;
    }
}
