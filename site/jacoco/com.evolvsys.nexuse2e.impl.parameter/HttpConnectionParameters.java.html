package com.evolvsys.nexuse2e.impl.parameter;

import com.evolvsys.nexuse2e.core.parameter.ParameterDefinition;
import com.evolvsys.nexuse2e.core.parameter.ParameterType;
import java.time.Duration;
import java.util.function.Supplier;
import org.springframework.http.HttpMethod;

/**
 *
 * @author SvenBarden
 * @since 16.09.2024
 */
public interface HttpConnectionParameters {
    String PARAM_URL = "url";
    String PARAM_METHOD = "method";
    String PARAM_CONNECTION_TIMEOUT = "connection_timeout";
    String PARAM_RESPONSE_TIMEOUT = "response_timeout";
    String PARAM_SSL_PROTOCOLS = "ssl_protocols";
    String PARAM_SSL_CIPHERS = "ssl_ciphers";
    String PARAM_CONTENT_AS_STREAM = "content_as_stream";
    String PARAM_ENABLE_HOSTNAME_VERIFICATION = "disable_hostname_verification";
    String PARAM_VALIDATE_CERT_CHAIN = "validate_cert_chain";
    String PARAM_AUTH_SERVICE = "auth_service";
    String MESSAGE_ID = "messageId";

    Duration DEFAULT_RESPONSE_TIMEOUT = Duration.ofSeconds(30);
    Duration DEFAULT_CONNECTION_TIMEOUT = Duration.ofSeconds(30);

    Supplier<ParameterDefinition.ParameterDefinitionBuilder> URL = () -> ParameterDefinition.builder()
            .name(PARAM_URL)
            .description("Url where the message should be send to")
            .type(ParameterType.STRING)
            .required(true);
    Supplier<ParameterDefinition.ParameterDefinitionBuilder> METHOD = () -> ParameterDefinition.builder()
            .name(PARAM_METHOD)
            .description("Method to indicate which action is to be performed")
            .type(ParameterType.DROPDOWN_LIST)
            .required(true)
            .defaultValue(HttpMethod.POST.toString())
            .option(HttpMethod.POST.toString())
            .option(HttpMethod.GET.toString())
            .option(HttpMethod.PUT.toString());
    Supplier<ParameterDefinition.ParameterDefinitionBuilder> CONTENT_AS_STREAM = () -> ParameterDefinition.builder()
            .name(PARAM_CONTENT_AS_STREAM)
            .displayName("Send content as stream")
            .description("Sends content as stream. Saves memory for large payloads.")
            .type(ParameterType.BOOLEAN)
            .defaultValue(Boolean.FALSE);
    Supplier<ParameterDefinition.ParameterDefinitionBuilder> CONNECTION_TIMEOUT = () -> ParameterDefinition.builder()
            .name(PARAM_CONNECTION_TIMEOUT)
            .displayName("Connection timeout")
            .description("Amount of time allowed for the connection to be established")
            .type(ParameterType.DURATION)
            .defaultValue(DEFAULT_CONNECTION_TIMEOUT);
    Supplier<ParameterDefinition.ParameterDefinitionBuilder> RESPONSE_TIMEOUT = () -> ParameterDefinition.builder()
            .name(PARAM_RESPONSE_TIMEOUT)
            .displayName("Response timeout")
            .description("Amount of time allowed for the http response to be received")
            .type(ParameterType.DURATION)
            .defaultValue(DEFAULT_RESPONSE_TIMEOUT);
    Supplier<ParameterDefinition.ParameterDefinitionBuilder> SSL_PROTOCOLS = () -> ParameterDefinition.builder()
            .name(PARAM_SSL_PROTOCOLS)
            .displayName("SSL protocols")
            .description(
                    "SSL/TLS protocol versions to use for the outbound https connection (e.g. \"TLSv1.2, TLSv1.3\") (optional)")
            .type(ParameterType.STRING_LIST);
    Supplier<ParameterDefinition.ParameterDefinitionBuilder> SSL_CIPHERS = () -> ParameterDefinition.builder()
            .name(PARAM_SSL_CIPHERS)
            .displayName("SSL ciphers")
            .description(
                    "SSL ciphers to use for the outbound https connection (\"ECDHE-ECDSA-AES128-GCM-SHA256, ECDHE-ECDSA-CHACHA20-POLY1305\") (optional) ")
            .type(ParameterType.STRING_LIST);
    Supplier<ParameterDefinition.ParameterDefinitionBuilder> ENABLE_HOSTNAME_VERIFICATION =
            () -> ParameterDefinition.builder()
                    .name(PARAM_ENABLE_HOSTNAME_VERIFICATION)
                    .displayName("Enable hostname verification")
                    .description("Enables the hostname verification for https connections")
                    .type(ParameterType.BOOLEAN)
                    .defaultValue(Boolean.TRUE);
    Supplier<ParameterDefinition.ParameterDefinitionBuilder> VALIDATE_CERT_CHAIN = () -> ParameterDefinition.builder()
            .name(PARAM_VALIDATE_CERT_CHAIN)
            .displayName("Validate trusted cert chain")
            .description("Validates the complete cert chain for each trusted partner certificate.")
            .type(ParameterType.BOOLEAN)
            .defaultValue(Boolean.TRUE);
    Supplier<ParameterDefinition.ParameterDefinitionBuilder> AUTH_SERVICE = () -> ParameterDefinition.builder()
            .name(PARAM_AUTH_SERVICE)
            .displayName("Authentication service")
            .description("The Http Support Service to use for authentication")
            .type(ParameterType.SUPPORT_SERVICE);
}
