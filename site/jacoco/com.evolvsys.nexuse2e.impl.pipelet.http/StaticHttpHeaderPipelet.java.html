package com.evolvsys.nexuse2e.impl.pipelet.http;

import com.evolvsys.nexuse2e.core.component.Pipelet;
import com.evolvsys.nexuse2e.core.constants.HttpConstants;
import com.evolvsys.nexuse2e.core.exception.NexusException;
import com.evolvsys.nexuse2e.core.keyvalue.KeyValueLookup;
import com.evolvsys.nexuse2e.core.message.MetaData;
import com.evolvsys.nexuse2e.core.message.context.ExecutionContext;
import com.evolvsys.nexuse2e.core.parameter.ParameterDefinition;
import com.evolvsys.nexuse2e.core.parameter.ParameterType;
import java.util.List;
import lombok.extern.slf4j.Slf4j;
import org.apache.commons.collections4.KeyValue;
import org.springframework.http.HttpHeaders;

/**
 * Adds manual http headers to the processing metadata to be used by Http-Outflow-Services
 *
 * @author SvenBarden
 * @since 10.04.2025
 */
@Slf4j
public class StaticHttpHeaderPipelet extends Pipelet {
    private static final String PARAM_HEADERS = "headers";
    private static final String PARAM_OVERWRITE = "overwrite";

    @Override
    public List<ParameterDefinition> parameterDefinitions() {
        return List.of(
                ParameterDefinition.builder()
                        .name(PARAM_HEADERS)
                        .displayName("Header values")
                        .description(
                                "Header key value pair to add as http headers. The value can also contain placeholders for key-value services.")
                        .type(ParameterType.KEY_VALUE_PAIR_LIST_TRIMMED)
                        .required(true)
                        .build(),
                ParameterDefinition.builder()
                        .name(PARAM_OVERWRITE)
                        .displayName("Overwrite")
                        .description(
                                "If true, any existing header value for the same key will be overwritten, otherwise the new value will just be added to the list.")
                        .type(ParameterType.BOOLEAN)
                        .defaultValue(Boolean.TRUE)
                        .build());
    }

    @Override
    public ExecutionContext execute(ExecutionContext executionContext) throws NexusException {

        List<KeyValue<String, String>> headers = parameters.value(PARAM_HEADERS);

        if (headers != null) {
            MetaData metaData = executionContext.getMessage().getMetaData();
            HttpHeaders httpHeaders = metaData.getFirstOrNull(HttpConstants.ADDITIONAL_HEADERS);
            if (httpHeaders == null) {
                metaData.set(HttpConstants.ADDITIONAL_HEADERS, httpHeaders = new HttpHeaders());
            }

            boolean overwrite = parameters.value(PARAM_OVERWRITE);

            for (KeyValue<String, String> header : headers) {
                String key = header.getKey();
                String value = KeyValueLookup.lookup(header.getValue());

                if (overwrite) {
                    LOGGER.trace("Setting header {} with value {}", key, value);
                    httpHeaders.set(key, value);
                } else {
                    LOGGER.trace("Adding header {} with value {}", key, value);
                    httpHeaders.add(key, value);
                }
            }
        }

        return executionContext;
    }
}
