package com.evolvsys.nexuse2e.impl.pipelet.http;

import com.evolvsys.nexuse2e.core.component.Pipelet;
import com.evolvsys.nexuse2e.core.exception.NexusException;
import com.evolvsys.nexuse2e.core.message.context.ExecutionContext;
import com.evolvsys.nexuse2e.core.parameter.ParameterDefinition;
import com.evolvsys.nexuse2e.internal.ContextPayload;
import java.util.ArrayList;
import java.util.List;
import lombok.extern.slf4j.Slf4j;
import org.apache.commons.collections4.CollectionUtils;
import org.apache.commons.lang3.StringUtils;
import org.springframework.http.MediaType;

/**
 * @author Florian Drees
 * @since 23.01.2024
 */
@Slf4j
public class PlainHttpMessageUnpackagerPipelet extends Pipelet {
    @Override
    public ExecutionContext execute(ExecutionContext executionContext) throws NexusException {
        if (CollectionUtils.isNotEmpty(executionContext.getDataAsList())) {
            ContextPayload contextPayload = executionContext.getData();
            ContextPayload payload = new ContextPayload();

            if (StringUtils.isNotEmpty(contextPayload.getContentType())) {
                payload.setContentType(contextPayload.getContentType());
            } else {
                LOGGER.warn("Content-Type not set! Using default type {}", MediaType.TEXT_XML);
                payload.setContentType(MediaType.TEXT_XML_VALUE);
            }

            payload.setContent(contextPayload.getContent());
            payload.setEncoding(contextPayload.getEncoding());
            payload.setSize(contextPayload.getSize());

            List<ContextPayload> payloads = new ArrayList<>();
            payloads.add(payload);
            executionContext.getMessage().setPayloads(payloads);
        }
        return executionContext;
    }

    @Override
    public List<ParameterDefinition> parameterDefinitions() {
        return List.of();
    }
}
