package com.evolvsys.nexuse2e.impl.pipelet.http;

import com.evolvsys.nexuse2e.core.component.Pipelet;
import com.evolvsys.nexuse2e.core.constants.HttpConstants;
import com.evolvsys.nexuse2e.core.exception.NexusException;
import com.evolvsys.nexuse2e.core.message.context.ExecutionContext;
import com.evolvsys.nexuse2e.core.parameter.ParameterDefinition;
import com.evolvsys.nexuse2e.core.parameter.ParameterType;
import java.util.List;

/**
 * <p>{@link Pipelet} to add request parameters to metadata based on routing ids.</p>
 */
public class RequestParameterPipelet extends Pipelet {
    static final String PARAM_CHOREOGRAPHY_ID_ENABLED = "choreography_id_enabled";
    static final String PARAM_ACTION_ID_ENABLED = "action_id_enabled";
    static final String PARAM_SOURCE_PARTNER_ID_ENABLED = "source_partner_id_enabled";
    static final String PARAM_TARGET_PARTNER_ID_ENABLED = "target_partner_id_enabled";
    static final String PARAM_CONVERSATION_ID_ENABLED = "conversation_id_enabled";
    static final String PARAM_MESSAGE_ID_ENABLED = "message_id_enabled";

    static final String PARAM_CHOREOGRAPHY_ID = "choreography_id";
    static final String PARAM_ACTION_ID = "action_id";
    static final String PARAM_SOURCE_PARTNER_ID = "source_partner_id";
    static final String PARAM_TARGET_PARTNER_ID = "target_partner_id";
    static final String PARAM_CONVERSATION_ID = "conversation_id";
    static final String PARAM_MESSAGE_ID = "message_id";

    static final String PARAM_DEFAULT_VALUE_CHOREOGRAPHY_ID = "choreography";
    static final String PARAM_DEFAULT_VALUE_ACTION_ID = "action";
    static final String PARAM_DEFAULT_VALUE_SOURCE_PARTNER_ID = "partner";
    static final String PARAM_DEFAULT_VALUE_TARGET_PARTNER_ID = "local_identity";
    static final String PARAM_DEFAULT_VALUE_CONVERSATION_ID = "conversation";
    static final String PARAM_DEFAULT_VALUE_MESSAGE_ID = "message";

    @Override
    public ExecutionContext execute(ExecutionContext executionContext) throws NexusException {

        // Choreography & Action
        if (parameters.value(PARAM_CHOREOGRAPHY_ID_ENABLED)) {
            setRequestParameter(executionContext, PARAM_CHOREOGRAPHY_ID, executionContext.getChoreographyId());
        }
        if (parameters.value(PARAM_ACTION_ID_ENABLED)) {
            setRequestParameter(executionContext, PARAM_ACTION_ID, executionContext.getActionId());
        }

        // Source & Target Partner
        if (parameters.value(PARAM_SOURCE_PARTNER_ID_ENABLED)) {
            setRequestParameter(executionContext, PARAM_SOURCE_PARTNER_ID, executionContext.getSourcePartnerId());
        }
        if (parameters.value(PARAM_TARGET_PARTNER_ID_ENABLED)) {
            setRequestParameter(executionContext, PARAM_TARGET_PARTNER_ID, executionContext.getTargetPartnerId());
        }

        // Conversation and Message
        if (parameters.value(PARAM_CONVERSATION_ID_ENABLED)) {
            setRequestParameter(
                    executionContext,
                    PARAM_CONVERSATION_ID,
                    executionContext.getMessage().getConversationId());
        }
        if (parameters.value(PARAM_MESSAGE_ID_ENABLED)) {
            setRequestParameter(
                    executionContext,
                    PARAM_MESSAGE_ID,
                    executionContext.getMessage().getId());
        }

        return executionContext;
    }

    private void setRequestParameter(ExecutionContext executionContext, String param, String value) {
        // parameter prefix is required as the HttpSenderService looks for request parameters in the metadata based on
        // the prefix
        executionContext
                .getMetaData()
                .set(HttpConstants.PREFIX_OUTBOUND_REQUEST_PARAMETER, parameters.value(param, String.class), value);
    }

    @Override
    public List<ParameterDefinition> parameterDefinitions() {
        return List.of(
                ParameterDefinition.builder()
                        .name(PARAM_CHOREOGRAPHY_ID_ENABLED)
                        .displayName("Add Choreography Id request parameter")
                        .description("Whether or not to add a request parameter containing the choreography id")
                        .type(ParameterType.BOOLEAN)
                        .defaultValue(Boolean.TRUE)
                        .build(),
                ParameterDefinition.builder()
                        .name(PARAM_CHOREOGRAPHY_ID)
                        .displayName("Choreography Id request parameter name")
                        .description("Name of the request parameter containing the choreography id")
                        .type(ParameterType.TRIMMED_STRING)
                        .defaultValue(PARAM_DEFAULT_VALUE_CHOREOGRAPHY_ID)
                        .build(),
                ParameterDefinition.builder()
                        .name(PARAM_ACTION_ID_ENABLED)
                        .displayName("Add Action Id request parameter")
                        .description("Whether or not to add a request parameter containing the action id")
                        .type(ParameterType.BOOLEAN)
                        .defaultValue(Boolean.TRUE)
                        .build(),
                ParameterDefinition.builder()
                        .name(PARAM_ACTION_ID)
                        .displayName("Action Id request parameter name")
                        .description("Name of the request parameter containing the action id ")
                        .type(ParameterType.TRIMMED_STRING)
                        .defaultValue(PARAM_DEFAULT_VALUE_ACTION_ID)
                        .build(),
                ParameterDefinition.builder()
                        .name(PARAM_SOURCE_PARTNER_ID_ENABLED)
                        .displayName("Add Source Partner Id request parameter")
                        .description("Whether or not to add a request parameter containing the source partner id")
                        .type(ParameterType.BOOLEAN)
                        .defaultValue(Boolean.TRUE)
                        .build(),
                ParameterDefinition.builder()
                        .name(PARAM_SOURCE_PARTNER_ID)
                        .displayName("Source Partner Id request parameter name")
                        .description("Name of the request parameter containing the source partner id")
                        .type(ParameterType.TRIMMED_STRING)
                        .defaultValue(PARAM_DEFAULT_VALUE_SOURCE_PARTNER_ID)
                        .build(),
                ParameterDefinition.builder()
                        .name(PARAM_TARGET_PARTNER_ID_ENABLED)
                        .displayName("Add Target Partner Id request parameter")
                        .description("Whether or not to add a request parameter containing the target partner id")
                        .type(ParameterType.BOOLEAN)
                        .defaultValue(Boolean.TRUE)
                        .build(),
                ParameterDefinition.builder()
                        .name(PARAM_TARGET_PARTNER_ID)
                        .displayName("Target Partner Id request parameter name")
                        .description("Name of the request parameter containing the target partner id")
                        .type(ParameterType.TRIMMED_STRING)
                        .defaultValue(PARAM_DEFAULT_VALUE_TARGET_PARTNER_ID)
                        .build(),
                ParameterDefinition.builder()
                        .name(PARAM_CONVERSATION_ID_ENABLED)
                        .displayName("Add Conversation Id request parameter")
                        .description("Whether or not to add a request parameter containing the conversation id")
                        .type(ParameterType.BOOLEAN)
                        .defaultValue(Boolean.TRUE)
                        .build(),
                ParameterDefinition.builder()
                        .name(PARAM_CONVERSATION_ID)
                        .displayName("Conversation Id request parameter name")
                        .description("Name of the request parameter containing the conversation id")
                        .type(ParameterType.TRIMMED_STRING)
                        .defaultValue(PARAM_DEFAULT_VALUE_CONVERSATION_ID)
                        .build(),
                ParameterDefinition.builder()
                        .name(PARAM_MESSAGE_ID_ENABLED)
                        .displayName("Add Message Id request parameter")
                        .description("Whether or not to add a the request parameter containing the message id")
                        .type(ParameterType.BOOLEAN)
                        .defaultValue(Boolean.TRUE)
                        .build(),
                ParameterDefinition.builder()
                        .name(PARAM_MESSAGE_ID)
                        .displayName("Message Id request parameter name")
                        .description("Name of the request parameter containing the message id")
                        .type(ParameterType.TRIMMED_STRING)
                        .defaultValue(PARAM_DEFAULT_VALUE_MESSAGE_ID)
                        .build());
    }
}
