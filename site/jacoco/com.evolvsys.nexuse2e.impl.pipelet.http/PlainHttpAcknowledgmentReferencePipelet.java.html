package com.evolvsys.nexuse2e.impl.pipelet.http;

import com.evolvsys.nexuse2e.core.component.Pipelet;
import com.evolvsys.nexuse2e.core.constants.HttpConstants;
import com.evolvsys.nexuse2e.core.exception.NexusError;
import com.evolvsys.nexuse2e.core.exception.NexusException;
import com.evolvsys.nexuse2e.core.message.context.ExecutionContext;
import com.evolvsys.nexuse2e.core.parameter.ParameterDefinition;
import com.evolvsys.nexuse2e.core.parameter.ParameterType;
import com.evolvsys.nexuse2e.internal.Message;
import com.evolvsys.nexuse2e.internal.MessageType;
import java.util.List;
import lombok.extern.slf4j.Slf4j;
import org.apache.commons.lang3.StringUtils;

/**
 * Pipelet to set the message type to acknowledgment and reference message id based on request parameters.
 *
 * @author SvenBarden
 * @since 14.11.2024
 */
@Slf4j
public class PlainHttpAcknowledgmentReferencePipelet extends Pipelet {
    private static final String PARAM_REFERENCE_MESSAGE_ID = "reference_message_id";

    @Override
    public List<ParameterDefinition> parameterDefinitions() {
        return List.of(ParameterDefinition.builder()
                .name(PARAM_REFERENCE_MESSAGE_ID)
                .displayName("Reference message id parameter name")
                .description("Name of the request parameter containing the reference message id.")
                .type(ParameterType.TRIMMED_STRING)
                .defaultValue("reference_message_id")
                .required(true)
                .build());
    }

    @Override
    public ExecutionContext execute(ExecutionContext executionContext) throws NexusException {
        Message message = executionContext.getMessage();

        String referenceMessageIdName = parameters.value(PARAM_REFERENCE_MESSAGE_ID);
        LOGGER.trace("looking for reference message id parameter: {}", referenceMessageIdName);

        if (StringUtils.isBlank(referenceMessageIdName)) {
            throw new NexusException(NexusError.builder()
                    .message("Reference message id parameter is required")
                    .build());
        }

        String referenceMessageIdValue = executionContext
                .getMetaData()
                .getFirstOrNull(HttpConstants.PREFIX_RECEIVED_REQUEST_PARAMETER, referenceMessageIdName);

        LOGGER.trace("found reference message id: {}", referenceMessageIdValue);
        if (StringUtils.isBlank(referenceMessageIdValue)) {
            throw new NexusException(NexusError.builder()
                    .message("Reference message id not found in request parameters")
                    .extension("expectedParameterName", referenceMessageIdName)
                    .build());
        }

        LOGGER.debug("setting reference message id: {}", referenceMessageIdValue);
        message.setReferenceMessageId(referenceMessageIdValue);

        LOGGER.debug("setting message type to acknowledgment for message {}", message.getId());
        message.setType(MessageType.ACKNOWLEDGMENT);
        executionContext.setActionId("Acknowledgment");

        return executionContext;
    }
}
