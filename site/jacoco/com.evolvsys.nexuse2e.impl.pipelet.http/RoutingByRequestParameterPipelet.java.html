package com.evolvsys.nexuse2e.impl.pipelet.http;

import com.evolvsys.nexuse2e.core.component.Pipelet;
import com.evolvsys.nexuse2e.core.constants.HttpConstants;
import com.evolvsys.nexuse2e.core.exception.NexusException;
import com.evolvsys.nexuse2e.core.message.context.ExecutionContext;
import com.evolvsys.nexuse2e.core.parameter.ParameterDefinition;
import com.evolvsys.nexuse2e.core.parameter.ParameterType;
import java.util.List;
import org.apache.commons.lang3.StringUtils;

/**
 * <p>{@link Pipelet} to use request parameters from metadata for routing ids.</p>
 */
public class RoutingByRequestParameterPipelet extends Pipelet {
    static final String PARAM_CHOREOGRAPHY_ID = "choreography_id";
    static final String PARAM_SOURCE_PARTNER_ID = "source_partner_id";
    static final String PARAM_TARGET_PARTNER_ID = "target_partner_id";
    static final String PARAM_ACTION_ID = "action_id";
    static final String PARAM_CONVERSATION_ID = "conversation_id";
    static final String PARAM_MESSAGE_ID = "message_id";

    static final String PARAM_DEFAULT_VALUE_CHOREOGRAPHY_ID = "choreography";
    static final String PARAM_DEFAULT_VALUE_SOURCE_PARTNER_ID = "local_identity";
    static final String PARAM_DEFAULT_VALUE_TARGET_PARTNER_ID = "partner";
    static final String PARAM_DEFAULT_VALUE_ACTION_ID = "action";
    static final String PARAM_DEFAULT_VALUE_CONVERSATION_ID = "conversation";
    static final String PARAM_DEFAULT_VALUE_MESSAGE_ID = "message";

    @Override
    public ExecutionContext execute(ExecutionContext executionContext) throws NexusException {
        // Choreography & Action
        String choreographyId = getRequestParameterValue(executionContext, parameters.value(PARAM_CHOREOGRAPHY_ID));
        if (StringUtils.isNotBlank(choreographyId)) {
            executionContext.setChoreographyId(choreographyId);
        }

        String actionId = getRequestParameterValue(executionContext, parameters.value(PARAM_ACTION_ID));
        if (StringUtils.isNotBlank(actionId)) {
            executionContext.setActionId(actionId);
        }

        // Source & Target Partner
        String sourcePartnerId = getRequestParameterValue(executionContext, parameters.value(PARAM_SOURCE_PARTNER_ID));
        String targetPartnerId = getRequestParameterValue(executionContext, parameters.value(PARAM_TARGET_PARTNER_ID));

        if (StringUtils.isNotEmpty(targetPartnerId)) {
            executionContext.setTargetPartnerId(targetPartnerId);
        }
        if (StringUtils.isNotEmpty(sourcePartnerId)) {
            executionContext.setSourcePartnerId(sourcePartnerId);
        }

        // Conversation and Message
        String conversationId = getRequestParameterValue(executionContext, parameters.value(PARAM_CONVERSATION_ID));
        if (StringUtils.isNotEmpty(conversationId)) {
            executionContext.getMessage().setConversationId(conversationId);
        }
        String messageId = getRequestParameterValue(executionContext, parameters.value(PARAM_MESSAGE_ID));
        if (StringUtils.isNotEmpty(messageId)) {
            executionContext.getMessage().setId(messageId);
        }

        return executionContext;
    }

    private String getRequestParameterValue(ExecutionContext executionContext, String key) {
        // parameter prefix is required as the HttpReceiverService adds it to ensure the key is unique in the metadata
        return executionContext.getMetaData().getFirstOrNull(HttpConstants.PREFIX_RECEIVED_REQUEST_PARAMETER, key);
    }

    @Override
    public List<ParameterDefinition> parameterDefinitions() {
        return List.of(
                ParameterDefinition.builder()
                        .name(PARAM_CHOREOGRAPHY_ID)
                        .displayName("Choreography Id request parameter name")
                        .description("Name of the required request parameter containing the choreography id")
                        .type(ParameterType.TRIMMED_STRING)
                        .required(true)
                        .defaultValue(PARAM_DEFAULT_VALUE_CHOREOGRAPHY_ID)
                        .build(),
                ParameterDefinition.builder()
                        .name(PARAM_ACTION_ID)
                        .displayName("Action Id request parameter name")
                        .description("Name of the required request parameter containing the action id")
                        .type(ParameterType.TRIMMED_STRING)
                        .required(true)
                        .defaultValue(PARAM_DEFAULT_VALUE_ACTION_ID)
                        .build(),
                ParameterDefinition.builder()
                        .name(PARAM_SOURCE_PARTNER_ID)
                        .displayName("Source Partner Id request parameter name")
                        .description(
                                "Name of the optional request parameter containing the source partner id "
                                        + "(Source partner id defaults to partner marked as default local identity if not provided in request)")
                        .type(ParameterType.TRIMMED_STRING)
                        .required(true)
                        .defaultValue(PARAM_DEFAULT_VALUE_SOURCE_PARTNER_ID)
                        .build(),
                ParameterDefinition.builder()
                        .name(PARAM_TARGET_PARTNER_ID)
                        .displayName("Target Partner Id request parameter name")
                        .description(
                                "Name of the optional request parameter containing the target partner id "
                                        + "(Target partner id defaults to partner marked as default local identity if not provided in request)")
                        .type(ParameterType.TRIMMED_STRING)
                        .required(true)
                        .defaultValue(PARAM_DEFAULT_VALUE_TARGET_PARTNER_ID)
                        .build(),
                ParameterDefinition.builder()
                        .name(PARAM_CONVERSATION_ID)
                        .displayName("Conversation Id request parameter name")
                        .description("Name of the optional request parameter containing the conversation id")
                        .type(ParameterType.TRIMMED_STRING)
                        .required(true)
                        .defaultValue(PARAM_DEFAULT_VALUE_CONVERSATION_ID)
                        .build(),
                ParameterDefinition.builder()
                        .name(PARAM_MESSAGE_ID)
                        .displayName("Message Id request parameter name")
                        .description("Name of the optional request parameter containing the message id")
                        .type(ParameterType.TRIMMED_STRING)
                        .required(true)
                        .defaultValue(PARAM_DEFAULT_VALUE_MESSAGE_ID)
                        .build());
    }
}
