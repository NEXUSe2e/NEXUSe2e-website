package com.evolvsys.nexuse2e.impl.pipelet.http;

import com.evolvsys.nexuse2e.core.component.Pipelet;
import com.evolvsys.nexuse2e.core.constants.HttpConstants;
import com.evolvsys.nexuse2e.core.constants.HttpMessageConstants;
import com.evolvsys.nexuse2e.core.ebxml.EbxmlUtils;
import com.evolvsys.nexuse2e.core.message.context.ExecutionContext;
import com.evolvsys.nexuse2e.core.message.payload.content.PayloadContent;
import com.evolvsys.nexuse2e.core.parameter.ParameterDefinition;
import com.evolvsys.nexuse2e.core.parameter.ParameterType;
import com.evolvsys.nexuse2e.core.parameter.Parameters;
import com.evolvsys.nexuse2e.internal.BinaryEncoding;
import com.evolvsys.nexuse2e.internal.ContextPayload;
import edu.umd.cs.findbugs.annotations.SuppressFBWarnings;
import java.nio.charset.Charset;
import java.util.ArrayList;
import java.util.Base64;
import java.util.List;
import lombok.extern.slf4j.Slf4j;
import org.apache.commons.collections4.CollectionUtils;
import org.apache.commons.lang3.ObjectUtils;
import org.apache.commons.lang3.StringUtils;
import org.springframework.http.ContentDisposition;
import org.springframework.http.HttpHeaders;

@Slf4j
public class MessagePackagerPipelet extends Pipelet {
    private boolean trimContent;
    private boolean contentDisposition;
    private BinaryEncoding contentEncoding = BinaryEncoding.BASE64;

    @Override
    @SuppressFBWarnings("CFS_CONFUSING_FUNCTION_SEMANTICS")
    public ExecutionContext execute(ExecutionContext executionContext) {
        resolveParameters(parameters);

        executionContext.clearData();
        addContextPayloads(executionContext);

        if (LOGGER.isTraceEnabled()) {
            LOGGER.trace("packed message parts: ");
            executionContext
                    .getDataAsList()
                    .forEach(contextPayload -> LOGGER.trace(
                            contextPayload.getContent().getString(Charset.forName(contextPayload.getEncoding()))));
        }

        return executionContext;
    }

    public void addContextPayloads(ExecutionContext executionContext) {
        List<ContextPayload> payloads = executionContext.getMessage().getPayloads();
        if (CollectionUtils.isNotEmpty(payloads)) {
            for (ContextPayload payload : payloads) {
                ContextPayload contextPayload = new ContextPayload();
                HttpHeaders additionalHeaders = new HttpHeaders();

                String contentId = ObjectUtils.defaultIfNull(payload.getContentId(), "");
                if (trimContent && StringUtils.isNotBlank(contentId)) {
                    payload.setContentId(contentId = contentId.replaceAll("__body_\\d+", ""));
                }
                additionalHeaders.set(HttpConstants.CONTENT_ID.getName(), String.format("<%s>", contentId));

                if (EbxmlUtils.isBinaryType(payload.getContentType())) {
                    switch (contentEncoding) {
                        case BINARY -> contextPayload.setContent(payload.getContent());
                        case UNSUPPORTED -> {
                            LOGGER.warn("BinaryEncoding.UNSUPPORTED found as target encoding. Will use base64 instead");
                            addBase64EncodedContent(contextPayload, payload.getContent(), contentId, additionalHeaders);
                        }
                        default ->
                            addBase64EncodedContent(contextPayload, payload.getContent(), contentId, additionalHeaders);
                    }
                } else {
                    contextPayload.setContent(payload.getContent());
                }

                contextPayload.setSize(contextPayload.getContent().getSize());
                contextPayload.setContentType(payload.getContentType());
                contextPayload.setEncoding(payload.getEncoding());
                contextPayload.getMetaData().set(HttpConstants.ADDITIONAL_HEADERS, additionalHeaders);

                executionContext.addData(contextPayload);
            }
        }
    }

    private void addBase64EncodedContent(
            ContextPayload payload, PayloadContent content, String contentId, HttpHeaders additionalHeaders) {
        payload.setContent(PayloadContent.of(Base64.getEncoder().encode(content.getByteArray())));

        if (contentDisposition && StringUtils.isNotBlank(contentId)) {
            additionalHeaders.setContentDisposition(
                    ContentDisposition.attachment().filename(contentId).build());
            additionalHeaders.set(
                    "Content-Transfer-Encoding", contentEncoding.toString().toLowerCase());
        }
    }

    @Override
    public List<ParameterDefinition> parameterDefinitions() {
        List<ParameterDefinition> parameterDefinitions = new ArrayList<>();
        parameterDefinitions.add(ParameterDefinition.builder()
                .name(HttpConstants.ENCODING.getName())
                .displayName("Binary Encoding")
                .description("Which encoding to use for binary content")
                .required(false)
                .defaultValue(BinaryEncoding.BASE64.name())
                .type(ParameterType.DROPDOWN_LIST)
                .option(BinaryEncoding.BASE64.name())
                .option(BinaryEncoding.BINARY.name())
                .build());
        parameterDefinitions.add(ParameterDefinition.builder()
                .name(HttpMessageConstants.USE_CONTENT_DISPOSITION.getName())
                .displayName("Use content disposition")
                .description(
                        "Adds the content disposition mime header to the message (only for Base64 encoded binary type content)")
                .required(false)
                .defaultValue(Boolean.FALSE)
                .type(ParameterType.BOOLEAN)
                .build());
        parameterDefinitions.add(ParameterDefinition.builder()
                .name(HttpMessageConstants.TRIM_CONTENT.getName())
                .displayName("Trim content")
                .description("Trims the content id in the mime header")
                .required(false)
                .defaultValue(Boolean.FALSE)
                .type(ParameterType.BOOLEAN)
                .build());
        return parameterDefinitions;
    }

    private void resolveParameters(Parameters parameters) {
        if (parameters != null) {
            if (parameters.containsKey(HttpConstants.ENCODING.getName())) {
                Object temporaryEncodingValue = parameters.value(HttpConstants.ENCODING.getName());
                BinaryEncoding temporaryEncoding;
                if (temporaryEncodingValue instanceof String) {
                    temporaryEncoding = BinaryEncoding.of(temporaryEncodingValue.toString());
                } else {
                    temporaryEncoding = (BinaryEncoding) temporaryEncodingValue;
                }

                if (temporaryEncoding != BinaryEncoding.UNSUPPORTED) {
                    contentEncoding = temporaryEncoding;
                }
            }

            trimContent = parameters.value(HttpMessageConstants.TRIM_CONTENT.getName());
            contentDisposition = parameters.value(HttpMessageConstants.USE_CONTENT_DISPOSITION.getName());
        }
    }
}
