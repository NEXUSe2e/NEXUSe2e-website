package com.evolvsys.nexuse2e.impl.pipelet.http;

import com.evolvsys.nexuse2e.core.component.Pipelet;
import com.evolvsys.nexuse2e.core.constants.HttpConstants;
import com.evolvsys.nexuse2e.core.ebxml.EbxmlConstants;
import com.evolvsys.nexuse2e.core.ebxml.EbxmlUtils;
import com.evolvsys.nexuse2e.core.exception.NexusError;
import com.evolvsys.nexuse2e.core.exception.NexusException;
import com.evolvsys.nexuse2e.core.message.context.ExecutionContext;
import com.evolvsys.nexuse2e.core.message.payload.content.PayloadContent;
import com.evolvsys.nexuse2e.core.parameter.ParameterDefinition;
import com.evolvsys.nexuse2e.internal.ContextPayload;
import java.nio.charset.Charset;
import java.util.ArrayList;
import java.util.Base64;
import java.util.List;
import java.util.regex.Matcher;
import java.util.regex.Pattern;
import lombok.extern.slf4j.Slf4j;
import org.apache.commons.collections4.CollectionUtils;
import org.apache.commons.lang3.ObjectUtils;
import org.springframework.http.MediaType;

@Slf4j
public class MessageUnpackagerPipelet extends Pipelet {
    private static final Pattern CONTENT_ID_PATTERN = Pattern.compile("<(.+)>");

    @Override
    public ExecutionContext execute(ExecutionContext executionContext) throws NexusException {
        if (CollectionUtils.isEmpty(executionContext.getSourceData())) {
            throw new NexusException(
                    NexusError.builder().message("No ebxml content provided").build());
        }

        String encoding = EbxmlUtils.getEncoding(executionContext).orElse(null);
        if (encoding == null) {
            throw new NexusException(NexusError.builder()
                    .message("Encoding of content was not provided")
                    .build());
        }
        Charset charset = Charset.forName(encoding);

        MediaType contentType = executionContext.getMetaData().getFirstOrThrows(HttpConstants.CONTENT_TYPE);
        switch (contentType.getType()) {
            case "multipart" -> {
                if (executionContext.getSourceData().isEmpty()) {
                    throw new NexusException(NexusError.builder()
                            .message("MIME message contains no body parts!")
                            .build());
                }
            }
            case "text" -> {}
            default ->
                throw new NexusException(NexusError.builder()
                        .message("MIME message doesn't seem to be of type multipart: " + contentType)
                        .build());
        }

        executionContext.getMessage().getMetaData().set(HttpConstants.ENCODING, encoding);

        decodeMessageHeader(executionContext.getSourceData().getFirst(), executionContext, charset);

        List<ContextPayload> messagePayloads = executionContext
                .getSourceData()
                .subList(1, executionContext.getSourceData().size());

        decodeMessagePayloads(messagePayloads, executionContext, charset);
        return executionContext;
    }

    private void decodeMessageHeader(
            ContextPayload contextPayload, ExecutionContext executionContext, Charset encoding) {
        if (LOGGER.isTraceEnabled()) {
            LOGGER.trace(
                    "Message-Header: {}", new String(contextPayload.getContent().getByteArray(), encoding));
            LOGGER.trace("Message-Header Content Type: {}", contextPayload.getContentType());
        }

        executionContext.getMessage().getMetaData().set(EbxmlConstants.HEADER_DATA, contextPayload);
    }

    private void decodeMessagePayloads(
            List<ContextPayload> messagePayloads, ExecutionContext executionContext, Charset encoding) {
        List<ContextPayload> payloads = new ArrayList<>(messagePayloads.size());

        LOGGER.debug("decode message payloads");

        for (ContextPayload contextPayload : messagePayloads) {
            String bodyPartMimeType =
                    EbxmlUtils.extractMimeType(contextPayload.getContentType()).orElseThrow();
            LOGGER.trace("BodyPart Mimetype: {}", bodyPartMimeType);

            PayloadContent payloadContent;
            if (EbxmlUtils.isBinaryType(bodyPartMimeType)) {
                payloadContent = PayloadContent.of(
                        Base64.getDecoder().decode(contextPayload.getContent().getByteArray()));
            } else {
                payloadContent = contextPayload.getContent();
            }

            ContextPayload payload = ContextPayload.builder()
                    .contentType(bodyPartMimeType)
                    .size(payloadContent.getSize())
                    .encoding(encoding.name())
                    .contentId(getContentId(ObjectUtils.defaultIfNull(contextPayload.getContentId(), "")))
                    .sequenceNumber(messagePayloads.indexOf(contextPayload) + 1)
                    .content(payloadContent)
                    .build();
            payloads.add(payload);
        }
        executionContext.getMessage().setPayloads(payloads);
        LOGGER.debug("added {} payloads to message", payloads.size());
    }

    private String getContentId(String rawContentId) {
        Matcher matcher = CONTENT_ID_PATTERN.matcher(rawContentId);
        if (matcher.matches()) {
            return matcher.group(1);
        }
        return rawContentId;
    }

    @Override
    public List<ParameterDefinition> parameterDefinitions() {
        return List.of();
    }
}
