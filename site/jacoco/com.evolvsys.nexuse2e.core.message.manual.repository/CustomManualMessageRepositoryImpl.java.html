package com.evolvsys.nexuse2e.core.message.manual.repository;

import com.evolvsys.nexuse2e.core.action.model.QActionEntity;
import com.evolvsys.nexuse2e.core.message.manual.model.ManualMessageActionInfoDto;
import com.evolvsys.nexuse2e.core.message.manual.model.ManualMessageRoutingInfoDto;
import com.evolvsys.nexuse2e.core.model.entity.QChoreographyEntity;
import com.evolvsys.nexuse2e.core.model.entity.QRouteEntity;
import com.querydsl.core.annotations.QueryProjection;
import com.querydsl.core.types.dsl.BooleanExpression;
import com.querydsl.core.types.dsl.Expressions;
import com.querydsl.jpa.JPQLQueryFactory;
import java.util.Collection;
import java.util.List;
import lombok.RequiredArgsConstructor;
import one.util.streamex.StreamEx;
import org.springframework.stereotype.Repository;

/**
 * @author SvenBarden
 * @since 16.07.2024
 */
@Repository
@RequiredArgsConstructor
class CustomManualMessageRepositoryImpl implements CustomManualMessageRepository {
    private final JPQLQueryFactory queryFactory;

    @Override
    public Collection<ManualMessageRoutingInfoDto> getRoutingInfos(Long sourcePartnerId, Long targetPartnerId) {
        QChoreographyEntity choreographyEntity = QChoreographyEntity.choreographyEntity;
        QActionEntity actionEntity = QActionEntity.actionEntity;
        QRouteEntity routeEntity = QRouteEntity.routeEntity;

        List<ManualMessageRoutingInfo> choreographyResults = queryFactory
                .select(new QCustomManualMessageRepositoryImpl_ManualMessageRoutingInfo(
                        choreographyEntity.id,
                        choreographyEntity.choreographyId,
                        actionEntity.id,
                        actionEntity.actionId))
                .from(choreographyEntity)
                .join(choreographyEntity.routes, routeEntity)
                .on(getPartnerPredicate(routeEntity, sourcePartnerId, targetPartnerId))
                .join(choreographyEntity.actions, actionEntity)
                .fetch();

        List<ManualMessageRoutingInfo> actionResults = queryFactory
                .select(new QCustomManualMessageRepositoryImpl_ManualMessageRoutingInfo(
                        choreographyEntity.id,
                        choreographyEntity.choreographyId,
                        actionEntity.id,
                        actionEntity.actionId))
                .from(choreographyEntity)
                .join(choreographyEntity.actions, actionEntity)
                .join(actionEntity.routes, routeEntity)
                .on(getPartnerPredicate(routeEntity, sourcePartnerId, targetPartnerId))
                .fetch();

        return StreamEx.of(choreographyResults)
                .append(actionResults)
                .toMap(
                        CustomManualMessageRepositoryImpl.ManualMessageRoutingInfo::choreographyId,
                        CustomManualMessageRepositoryImpl.ManualMessageRoutingInfo::toDto,
                        this::merge)
                .values();
    }

    private BooleanExpression getPartnerPredicate(
            QRouteEntity routeEntity, Long sourcePartnerId, Long targetPartnerId) {
        BooleanExpression firstPartnerPredicate = Expressions.allOf(
                routeEntity.firstPartner.id.eq(sourcePartnerId), routeEntity.secondPartner.id.eq(targetPartnerId));
        BooleanExpression secondPartnerPredicate = Expressions.allOf(
                routeEntity.firstPartner.id.eq(targetPartnerId), routeEntity.secondPartner.id.eq(sourcePartnerId));

        return Expressions.anyOf(firstPartnerPredicate, secondPartnerPredicate);
    }

    private ManualMessageRoutingInfoDto merge(ManualMessageRoutingInfoDto left, ManualMessageRoutingInfoDto right) {
        right.getActions().addAll(left.getActions());
        return right;
    }

    public record ManualMessageRoutingInfo(
            Long choreographyId, String choreographyName, Long actionId, String actionName) {
        @QueryProjection
        public ManualMessageRoutingInfo {}

        public ManualMessageRoutingInfoDto toDto() {
            ManualMessageRoutingInfoDto dto = new ManualMessageRoutingInfoDto();
            dto.setId(choreographyId);
            dto.setName(choreographyName);
            ManualMessageActionInfoDto actionDto = new ManualMessageActionInfoDto();
            actionDto.setId(actionId);
            actionDto.setName(actionName);
            dto.getActions().add(actionDto);
            return dto;
        }
    }
}
