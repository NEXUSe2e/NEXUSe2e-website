package com.evolvsys.nexuse2e.core.certificate.service;

import com.evolvsys.app.core.graphql.annotation.GraphQLService;
import com.evolvsys.nexuse2e.core.certificate.model.*;
import com.evolvsys.nexuse2e.core.exception.NexusException;
import io.leangen.graphql.annotations.*;
import io.leangen.graphql.execution.ResolutionEnvironment;
import jakarta.validation.Valid;
import java.util.List;
import java.util.Optional;
import lombok.RequiredArgsConstructor;
import org.springframework.data.domain.Page;
import org.springframework.data.domain.Pageable;

/**
 * @author SvenBarden
 * @since 06.02.2024
 */
@GraphQLService
@RequiredArgsConstructor
public class CertificateGraphQLService {
    private final CertificateCrudService certificateCrudService;
    private final CertificateService certificateService;
    private final CertificateChainLoader certificateChainLoader;

    @GraphQLQuery(name = "certificate_list")
    public Page<@GraphQLNonNull CertificateDto> list(Pageable pageable, CertificateFilter certificateFilter) {
        return certificateCrudService.list(pageable, certificateFilter);
    }

    @GraphQLQuery(name = "certificate_listAll")
    public List<@GraphQLNonNull CertificateDto> getAll() {
        return certificateCrudService.getAll();
    }

    @GraphQLQuery(name = "certificate_getById")
    public Optional<CertificateDto> getById(long id) {
        return certificateCrudService.getById(id);
    }

    @GraphQLQuery(name = "certificate_exportCaKeystore")
    public boolean exportCaKeystore() throws NexusException {
        certificateService.exportCaKeystore();
        return true;
    }

    @GraphQLMutation(name = "certificate_update")
    @GraphQLNonNull
    public CertificateDto update(@GraphQLNonNull @Valid CertificateDto certificate) {
        return certificateCrudService.save(certificate);
    }

    @GraphQLMutation(name = "certificate_uploadLocal")
    @GraphQLNonNull
    public CertificateDto uploadPrivateKey(@GraphQLNonNull @Valid KeystoreUpload keystoreUpload) throws NexusException {
        long certId = certificateService.uploadPrivateKey(keystoreUpload);
        return certificateCrudService.getOrThrow(certId);
    }

    @GraphQLMutation(name = "certificate_uploadCertificate")
    @GraphQLNonNull
    public CertificateDto uploadCertificate(@GraphQLNonNull @Valid CertificateUpload certificateUpload)
            throws NexusException {
        long certId = certificateService.uploadPartner(certificateUpload);
        return certificateCrudService.getOrThrow(certId);
    }

    @GraphQLMutation(name = "certificate_uploadCaCertificate")
    @GraphQLNonNull
    public CertificateDto uploadCaCertificate(@GraphQLNonNull @Valid CaCertificateUpload certificateUpload)
            throws NexusException {
        long certId = certificateService.uploadCa(certificateUpload);
        return certificateCrudService.getOrThrow(certId);
    }

    @GraphQLMutation(name = "certificate_uploadCaKeystore")
    public boolean uploadCaKeystore(@GraphQLNonNull long fileId, String password) throws NexusException {
        certificateService.uploadCa(fileId, password);
        return true;
    }

    @GraphQLMutation(name = "certificate_delete")
    public boolean delete(long certificateId) {
        certificateCrudService.delete(certificateId);
        return true;
    }

    @GraphQLMutation(name = "certificate_enable")
    public boolean enable(long certificateId) {
        certificateCrudService.enable(certificateId);
        return true;
    }

    @GraphQLMutation(name = "certificate_disable")
    public boolean disable(long certificateId) {
        certificateCrudService.disable(certificateId);
        return true;
    }

    @GraphQLQuery
    public List<@GraphQLNonNull CertificateInfoDto> certificateChain(
            @GraphQLContext CertificateDto certificateDto,
            @GraphQLEnvironment ResolutionEnvironment resolutionEnvironment)
            throws NexusException {
        return certificateChainLoader.getCertificateChain(certificateDto.getId());
    }
}
