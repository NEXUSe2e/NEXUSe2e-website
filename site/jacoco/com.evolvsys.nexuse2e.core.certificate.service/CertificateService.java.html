package com.evolvsys.nexuse2e.core.certificate.service;

import com.evolvsys.app.core.exception.ObjectNotFoundException;
import com.evolvsys.app.core.file.model.FileDto;
import com.evolvsys.app.core.file.model.FileEntity;
import com.evolvsys.app.core.file.model.FileStatus;
import com.evolvsys.app.core.file.repository.FileRepository;
import com.evolvsys.nexuse2e.core.certificate.mapper.CertificateInfoMapper;
import com.evolvsys.nexuse2e.core.certificate.model.*;
import com.evolvsys.nexuse2e.core.certificate.repository.CertificateContainerRepository;
import com.evolvsys.nexuse2e.core.certificate.repository.CertificateRepository;
import com.evolvsys.nexuse2e.core.config.NEXUSe2eConfigurationProperties;
import com.evolvsys.nexuse2e.core.exception.NexusError;
import com.evolvsys.nexuse2e.core.exception.NexusException;
import edu.umd.cs.findbugs.annotations.SuppressFBWarnings;
import java.io.File;
import java.io.FileOutputStream;
import java.io.IOException;
import java.security.*;
import java.security.cert.CertificateException;
import java.security.cert.X509Certificate;
import java.sql.Blob;
import java.sql.SQLException;
import java.time.ZonedDateTime;
import java.util.List;
import java.util.Set;
import java.util.stream.Collectors;
import lombok.RequiredArgsConstructor;
import lombok.extern.slf4j.Slf4j;
import org.apache.commons.io.FileUtils;
import org.springframework.stereotype.Service;
import org.springframework.transaction.annotation.Transactional;

/**
 * @author SvenBarden
 * @since 06.02.2024
 */
@Service
@RequiredArgsConstructor
@Slf4j
public class CertificateService {
    private final NEXUSe2eConfigurationProperties configurationProperties;
    private final FileRepository fileRepository;
    private final CertificateEntityFactory certificateEntityFactory;
    private final CertificateRepository certificateRepository;
    private final CertificateContainerRepository certificateContainerRepository;
    private final CertificateInfoMapper certificateInfoMapper;

    private Set<String> getCaFingerprints() {
        List<CertificateEntity> caCertificates = certificateRepository.findByType(CertificateType.CA);
        return caCertificates.stream()
                .map(certificateEntity -> certificateEntity.getInfo().getSha1Fingerprint())
                .collect(Collectors.toSet());
    }

    @Transactional(rollbackFor = NexusException.class)
    public long uploadPartner(CertificateUpload certificateUpload) throws NexusException {
        CertificateContainerEntity certificateContainerEntity = certificateContainerRepository
                .findById(certificateUpload.getCertificateContainerId())
                .orElseThrow(() -> new ObjectNotFoundException(
                        CertificateContainerDto.class, certificateUpload.getCertificateContainerId()));

        if (certificateContainerEntity.getType() != CertificateContainerType.PUBLIC_KEY) {
            throw new NexusException(NexusError.builder()
                    .message("Public certs can only be uploaded to public containers")
                    .build());
        }

        return uploadSingleCert(certificateUpload.getFileId(), certificateUpload.getName(), certificateContainerEntity)
                .getId();
    }

    @Transactional(rollbackFor = NexusException.class)
    public long uploadPrivateKey(KeystoreUpload keystoreUpload) throws NexusException {
        try {
            FileEntity fileEntity = fileRepository
                    .getFileEntityById(keystoreUpload.getFileId())
                    .orElseThrow(() -> new ObjectNotFoundException(FileDto.class, keystoreUpload.getFileId()));
            Blob content = fileEntity.getFileContent().getContent();

            KeyStore keyStore =
                    CertificateKeystoreUtils.toKeystore(content.getBinaryStream(), keystoreUpload.getPassword());

            CertificateContainerEntity certificateContainerEntity = certificateContainerRepository
                    .findById(keystoreUpload.getCertificateContainerId())
                    .orElseThrow(() -> new ObjectNotFoundException(
                            CertificateContainerDto.class, keystoreUpload.getCertificateContainerId()));

            if (certificateContainerEntity.getType() != CertificateContainerType.PRIVATE_KEY) {
                throw new NexusException(NexusError.builder()
                        .message("key stores can only be uploaded to local identity partner containers")
                        .build());
            }

            CertificateEntity certificateEntity = certificateEntityFactory.createFromPKCS12(keyStore, keystoreUpload);

            if (certificateRepository.existsByCertificateContainerAndInfo_sha1Fingerprint(
                    certificateContainerEntity, certificateEntity.getInfo().getSha1Fingerprint())) {
                throw new NexusException(NexusError.builder()
                        .message("certificate exists already in container")
                        .extension("fingerprint", certificateEntity.getInfo().getSha1Fingerprint())
                        .build());
            }

            Set<String> caFingerprints = getCaFingerprints();
            List<CertificateEntity> caCertificateEntities =
                    certificateEntityFactory.createMissingCaEntities(caFingerprints, keyStore);

            certificateEntity.setCertificateContainer(certificateContainerEntity);

            certificateEntity = certificateRepository.save(certificateEntity);
            certificateRepository.saveAll(caCertificateEntities);

            fileRepository.delete(fileEntity);

            return certificateEntity.getId();
        } catch (CertificateException | SQLException e) {
            throw new NexusException(
                    NexusError.builder()
                            .message("could not upload keystore")
                            .extension("reason", e.getMessage())
                            .extension("fileId", keystoreUpload.getFileId())
                            .build(),
                    e);
        }
    }

    @Transactional(rollbackFor = NexusException.class)
    public long uploadCa(CaCertificateUpload certificateUpload) throws NexusException {
        return uploadSingleCert(certificateUpload.getFileId(), certificateUpload.getName(), null)
                .getId();
    }

    @Transactional(rollbackFor = NexusException.class)
    public void uploadCa(Long fileId, String password) throws NexusException {
        try {
            FileEntity fileEntity = fileRepository
                    .getFileEntityById(fileId)
                    .orElseThrow(() -> new ObjectNotFoundException(FileDto.class, fileId));
            Blob content = fileEntity.getFileContent().getContent();

            KeyStore keyStore = CertificateKeystoreUtils.toKeystore(content.getBinaryStream(), password);

            Set<String> caFingerprints = getCaFingerprints();
            List<CertificateEntity> caEntities =
                    certificateEntityFactory.createMissingCaEntitiesFromCaKeyStore(caFingerprints, keyStore);

            LOGGER.trace("adding {} out of {} ca certificates from keystore", caEntities.size(), keyStore.size());

            certificateRepository.saveAll(caEntities);
        } catch (KeyStoreException | CertificateException | SQLException e) {
            throw new NexusException(
                    NexusError.builder()
                            .message("could not upload keystore")
                            .extension("reason", e.getMessage())
                            .extension("fileId", fileId)
                            .build(),
                    e);
        }
    }

    @Transactional(readOnly = true, rollbackFor = NexusException.class)
    @SuppressFBWarnings("PATH_TRAVERSAL_IN")
    public void exportCaKeystore() throws NexusException {
        List<CertificateEntity> caCertificateEntities = certificateRepository.findByType(CertificateType.CA);

        try {
            String path = configurationProperties.getCaCertsPath();

            File file = new File(path);
            try (FileOutputStream fileOutputStream = FileUtils.openOutputStream(file)) {

                KeyStore jks = KeyStore.getInstance("JKS");
                jks.load(null, null);
                for (CertificateEntity caCertificateEntity : caCertificateEntities) {
                    X509Certificate certificate = CertificateHelper.getCertificate(caCertificateEntity
                            .getFile()
                            .getFileContent()
                            .getContent()
                            .getBinaryStream());
                    jks.setCertificateEntry(caCertificateEntity.getName(), certificate);
                }

                jks.store(
                        fileOutputStream,
                        configurationProperties.getCaCertsPassword().toCharArray());
            }
        } catch (KeyStoreException
                | SQLException
                | NexusException
                | CertificateException
                | IOException
                | NoSuchAlgorithmException e) {
            throw new NexusException(
                    NexusError.builder()
                            .message("could not export ca keystore to server file path")
                            .build(),
                    e);
        }
    }

    private CertificateEntity uploadSingleCert(
            Long fileId, String name, CertificateContainerEntity certificateContainer) throws NexusException {
        FileEntity fileEntity = fileRepository
                .getFileEntityById(fileId)
                .orElseThrow(() -> new ObjectNotFoundException(FileDto.class, fileId));
        Blob content = fileEntity.getFileContent().getContent();

        X509Certificate x509Certificate;
        try {
            x509Certificate = CertificateHelper.getCertificate(content.getBinaryStream());
        } catch (SQLException e) {
            throw new NexusException(
                    NexusError.builder()
                            .message("could not load certificate from database")
                            .build(),
                    e);
        }

        CertificateInfo certificateInfo = certificateInfoMapper.certificateToInfo(x509Certificate);

        if ((certificateContainer == null
                        && certificateRepository.existsByCertificateContainerNullAndInfo_sha1Fingerprint(
                                certificateInfo.getSha1Fingerprint()))
                || (certificateContainer != null
                        && certificateRepository.existsByCertificateContainerAndInfo_sha1Fingerprint(
                                certificateContainer, certificateInfo.getSha1Fingerprint()))) {
            throw new NexusException(NexusError.builder()
                    .message("certificate exists already in container")
                    .extension("fingerprint", certificateInfo.getSha1Fingerprint())
                    .build());
        }

        CertificateEntity certificateEntity = new CertificateEntity();
        certificateEntity.setName(name);
        certificateEntity.setType(certificateContainer == null ? CertificateType.CA : CertificateType.PARTNER);
        certificateEntity.setCertificateContainer(certificateContainer);
        certificateEntity.setFile(fileEntity);
        certificateEntity.setInfo(certificateInfo);
        certificateEntity.setEnabled(ZonedDateTime.now());
        certificateEntity.setActive(true);

        fileEntity.setStatus(FileStatus.SAVED);

        certificateEntity = certificateRepository.save(certificateEntity);
        fileRepository.save(fileEntity);

        return certificateEntity;
    }
}
