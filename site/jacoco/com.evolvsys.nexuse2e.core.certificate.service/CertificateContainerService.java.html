package com.evolvsys.nexuse2e.core.certificate.service;

import com.evolvsys.app.core.auth.service.CurrentUserService;
import com.evolvsys.app.core.service.BaseService;
import com.evolvsys.app.core.service.ListService;
import com.evolvsys.nexuse2e.core.certificate.mapper.CertificateContainerMapper;
import com.evolvsys.nexuse2e.core.certificate.model.CertificateContainerDto;
import com.evolvsys.nexuse2e.core.certificate.model.CertificateContainerEntity;
import com.evolvsys.nexuse2e.core.certificate.model.CertificateContainerFilter;
import com.evolvsys.nexuse2e.core.certificate.repository.CertificateContainerRepository;
import org.springframework.data.domain.Page;
import org.springframework.data.domain.Pageable;
import org.springframework.stereotype.Service;

/**
 * @author SvenBarden
 * @since 07.02.2024
 */
@Service
public class CertificateContainerService
        extends BaseService<CertificateContainerDto, CertificateContainerEntity, CertificateContainerRepository> {
    private final ListService listService;

    public CertificateContainerService(
            CertificateContainerMapper mapper,
            CertificateContainerRepository repository,
            CurrentUserService currentUserService,
            ListService listService) {
        super(mapper, repository, currentUserService);
        this.listService = listService;
    }

    @Override
    protected CertificateContainerEntity createEntity() {
        return new CertificateContainerEntity();
    }

    protected Page<CertificateContainerDto> list(
            Pageable pageable, CertificateContainerFilter certificateContainerFilter) {
        return listService.list(pageable, certificateContainerFilter, this).map(this::mapEntityToDto);
    }
}
