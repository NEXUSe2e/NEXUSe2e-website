package com.evolvsys.nexuse2e.core.certificate.service;

import com.evolvsys.nexuse2e.core.certificate.CertificateConstants;
import com.evolvsys.nexuse2e.core.exception.NexusError;
import com.evolvsys.nexuse2e.core.exception.NexusException;
import java.io.InputStream;
import java.security.*;
import java.security.cert.Certificate;
import java.security.cert.CertificateException;
import java.security.cert.X509Certificate;
import java.util.ArrayList;
import java.util.Enumeration;
import java.util.List;
import java.util.SequencedCollection;
import javax.security.auth.x500.X500Principal;
import lombok.experimental.UtilityClass;
import org.apache.commons.lang3.ArrayUtils;

/**
 * @author SvenBarden
 * @since 06.02.2024
 */
@UtilityClass
public class CertificateHelper {

    private static void addCertsToChain(
            SequencedCollection<X509Certificate> chain, SequencedCollection<X509Certificate> caCerts) {
        X509Certificate x509Certificate = chain.getLast();

        X500Principal issuerX500Principal = x509Certificate.getIssuerX500Principal();

        if (issuerX500Principal.equals(x509Certificate.getSubjectX500Principal())) {
            // self sign, we are finished
        } else {
            for (X509Certificate caCert : caCerts) {
                if (issuerX500Principal.equals(caCert.getSubjectX500Principal())) {
                    try {
                        x509Certificate.verify(caCert.getPublicKey());

                        chain.add(caCert);

                        addCertsToChain(chain, caCerts);
                        break;
                    } catch (CertificateException
                            | NoSuchAlgorithmException
                            | InvalidKeyException
                            | NoSuchProviderException
                            | SignatureException e) {
                        // do nothing
                    }
                }
            }
        }
    }

    public static List<X509Certificate> buildChain(X509Certificate cert, SequencedCollection<X509Certificate> caCerts) {
        List<X509Certificate> chain = new ArrayList<>();
        chain.add(cert);

        addCertsToChain(chain, caCerts);

        return chain;
    }

    public static X509Certificate getCertificate(InputStream inputStream) throws NexusException {
        X509Certificate certificate;
        try {
            java.security.cert.CertificateFactory factory;
            try {
                factory = java.security.cert.CertificateFactory.getInstance(
                        CertificateConstants.DEFAULT_CERT_TYPE, CertificateConstants.DEFAULT_JCE_PROVIDER);
            } catch (NoSuchProviderException e) {
                factory =
                        java.security.cert.CertificateFactory.getInstance(CertificateConstants.DEFAULT_KEY_STORE_TYPE);
            }
            certificate = (X509Certificate) factory.generateCertificate(inputStream);
        } catch (CertificateException e) {
            throw new NexusException(
                    NexusError.builder()
                            .message("could not create certificate from inputStream")
                            .build(),
                    e);
        }

        return certificate;
    }

    public static X509Certificate getPublicCertFromKey(KeyStore keyStore) throws NexusException {
        X509Certificate certificate = null;
        try {
            Enumeration<String> aliases = keyStore.aliases();

            if (aliases.hasMoreElements() == false) {
                throw new NexusException(
                        NexusError.builder().message("keystore has no aliases").build());
            }

            while (aliases.hasMoreElements() && certificate == null) {
                String alias = aliases.nextElement();
                if (keyStore.isKeyEntry(alias)) {
                    Certificate[] certificateChain = keyStore.getCertificateChain(alias);
                    if (ArrayUtils.isNotEmpty(certificateChain)) {
                        certificate = (X509Certificate) certificateChain[0];
                    }
                }
            }

            if (certificate == null) {
                throw new NexusException(NexusError.builder()
                        .message("no certificate in keystore found")
                        .build());
            }

        } catch (KeyStoreException e) {
            throw new NexusException(
                    NexusError.builder()
                            .message("could not get certificate from keystore")
                            .build(),
                    e);
        }
        return certificate;
    }
}
