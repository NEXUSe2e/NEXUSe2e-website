package com.evolvsys.nexuse2e.core.certificate.service;

import static com.evolvsys.nexuse2e.core.file.TempFileCleanupService.TEMP_FILE_NAME_PREFIX;

import com.evolvsys.app.core.binary.model.BinaryResult;
import com.evolvsys.app.core.binary.service.BinaryType;
import com.evolvsys.nexuse2e.core.certificate.mapper.CertificateInfoMapper;
import com.evolvsys.nexuse2e.core.certificate.model.CertificateInfo;
import com.evolvsys.nexuse2e.core.exception.NexusException;
import com.evolvsys.nexuse2e.core.utils.ErrorUtils;
import com.evolvsys.nexuse2e.core.utils.NEXUSe2eFileUtils;
import java.io.File;
import java.io.FileOutputStream;
import java.io.IOException;
import java.nio.charset.StandardCharsets;
import java.nio.file.Files;
import java.security.cert.X509Certificate;
import java.time.LocalDateTime;
import java.util.Collection;
import java.util.List;
import java.util.Optional;
import lombok.RequiredArgsConstructor;
import lombok.extern.slf4j.Slf4j;
import org.apache.commons.io.FileUtils;
import org.apache.commons.io.IOUtils;
import org.springframework.core.io.FileSystemResource;
import org.springframework.stereotype.Component;

/**
 * <p>Provide all public certificates in a single pem file to ensure a complete certificate chain for download.</p>
 *
 * @author Florian Drees
 * @since 24.05.2024
 */
@Component
@RequiredArgsConstructor
@Slf4j
public class CertificateChainBinaryType implements BinaryType {
    public static final String NAME = "CERTIFICATE_CHAIN_FILE";

    private final CertificateChainLoader certificateChainLoader;
    private final CertificateInfoMapper certificateInfoMapper;

    @Override
    public String name() {
        return NAME;
    }

    @Override
    public Optional<BinaryResult> get(String id) {
        Optional<BinaryResult> binaryResult;
        try {
            List<X509Certificate> certificateChain = certificateChainLoader.getX509CertificateChain(Long.parseLong(id));

            if (certificateChain.isEmpty()) {
                binaryResult = Optional.empty();
            } else {
                try {
                    File tempFile = Files.createTempFile(TEMP_FILE_NAME_PREFIX, "certificateChain")
                            .toFile();

                    writeCertChain(tempFile, certificateChain);

                    CertificateInfo leafCertificateInfo =
                            certificateInfoMapper.certificateToInfo(certificateChain.getFirst());

                    binaryResult = Optional.of(BinaryResult.builder()
                            .fileName(getFileName(leafCertificateInfo.getCommonName()))
                            .size(tempFile.length())
                            .mimeType("application/x-pem-file")
                            .resource(new FileSystemResource(tempFile))
                            .build());
                } catch (IOException e) {
                    LOGGER.error("Could not build pem file for certificate chain: {}", ErrorUtils.aggregate(e), e);
                    binaryResult = Optional.empty();
                }
            }
        } catch (NexusException e) {
            LOGGER.error("Could not build certificate chain: {}", ErrorUtils.aggregate(e), e);
            binaryResult = Optional.empty();
        }
        return binaryResult;
    }

    private static void writeCertChain(File tempFile, Collection<X509Certificate> certificateChain)
            throws IOException, NexusException {
        try (FileOutputStream fileOutputStream = FileUtils.openOutputStream(tempFile)) {
            IOUtils.write(
                    CertificatePemWriter.toCertificateChain(certificateChain),
                    fileOutputStream,
                    StandardCharsets.UTF_8);
        }
    }

    public static String getFileName(String commonName) {
        return NEXUSe2eFileUtils.cleanFileName("NEXUSe2e_publicKeyChain_" + commonName + "_"
                + LocalDateTime.now().withNano(0) + ".pem");
    }
}
