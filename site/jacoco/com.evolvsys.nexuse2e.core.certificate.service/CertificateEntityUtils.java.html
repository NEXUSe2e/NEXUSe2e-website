package com.evolvsys.nexuse2e.core.certificate.service;

import com.evolvsys.app.core.exception.ObjectNotFoundException;
import com.evolvsys.nexuse2e.core.certificate.model.CertificateDto;
import com.evolvsys.nexuse2e.core.certificate.model.CertificateEntity;
import com.evolvsys.nexuse2e.core.certificate.model.CertificateType;
import com.evolvsys.nexuse2e.core.certificate.repository.CertificateRepository;
import com.evolvsys.nexuse2e.core.exception.NexusError;
import com.evolvsys.nexuse2e.core.exception.NexusException;
import java.security.PrivateKey;
import lombok.RequiredArgsConstructor;
import lombok.extern.slf4j.Slf4j;
import org.springframework.stereotype.Component;
import org.springframework.transaction.annotation.Transactional;

/**
 * @author Florian Drees
 * @since 04.06.2024
 */
@Component
@RequiredArgsConstructor
@Slf4j
public class CertificateEntityUtils {
    private final CertificateRepository certificateRepository;

    public boolean isPrivateCert(long certificateId) {
        return certificateRepository
                .findById(certificateId)
                .map(entity -> entity.getType() == CertificateType.LOCAL)
                .orElseThrow(() -> new ObjectNotFoundException(CertificateDto.class, certificateId));
    }

    @Transactional(readOnly = true)
    public PrivateKey getPrivateKey(long certificateId) throws NexusException {
        CertificateEntity certificateEntity = certificateRepository
                .findById(certificateId)
                .orElseThrow(() -> new ObjectNotFoundException(CertificateDto.class, certificateId));
        LOGGER.debug(
                "Getting private key for certificate with id '{}' and name '{}'",
                certificateEntity.getId(),
                certificateEntity.getName());
        if (!isPrivateCert(certificateId)) {
            throw new NexusException(NexusError.builder()
                    .message("Unable to load private key as the entity is not a keystore")
                    .extension("id", certificateEntity.getId())
                    .extension("name", certificateEntity.getName())
                    .build());
        }
        return CertificateKeystoreUtils.getPrivateKey(
                CertificateKeystoreUtils.toKeyStore(certificateEntity), certificateEntity.getPassword());
    }
}
