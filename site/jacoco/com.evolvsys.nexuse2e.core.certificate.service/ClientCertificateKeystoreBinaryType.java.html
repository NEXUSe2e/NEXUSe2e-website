package com.evolvsys.nexuse2e.core.certificate.service;

import static com.evolvsys.nexuse2e.core.file.TempFileCleanupService.TEMP_FILE_NAME_PREFIX;

import com.evolvsys.app.core.binary.model.BinaryResult;
import com.evolvsys.app.core.binary.service.BinaryType;
import com.evolvsys.app.core.exception.ObjectNotFoundException;
import com.evolvsys.nexuse2e.core.certificate.mapper.CertificateInfoMapper;
import com.evolvsys.nexuse2e.core.certificate.model.CertificateDto;
import com.evolvsys.nexuse2e.core.certificate.model.CertificateEntity;
import com.evolvsys.nexuse2e.core.certificate.model.CertificateInfo;
import com.evolvsys.nexuse2e.core.certificate.repository.CertificateRepository;
import com.evolvsys.nexuse2e.core.exception.NexusException;
import com.evolvsys.nexuse2e.core.utils.ErrorUtils;
import com.evolvsys.nexuse2e.core.utils.NEXUSe2eFileUtils;
import java.io.File;
import java.io.IOException;
import java.io.OutputStream;
import java.nio.charset.StandardCharsets;
import java.nio.file.Files;
import java.security.cert.X509Certificate;
import java.time.LocalDateTime;
import java.util.List;
import java.util.Optional;
import java.util.zip.ZipEntry;
import java.util.zip.ZipOutputStream;
import lombok.RequiredArgsConstructor;
import lombok.extern.slf4j.Slf4j;
import org.apache.commons.io.IOUtils;
import org.springframework.core.io.FileSystemResource;
import org.springframework.stereotype.Component;

/**
 * @author Florian Drees
 * @since 04.06.2024
 */
@Component
@RequiredArgsConstructor
@Slf4j
public class ClientCertificateKeystoreBinaryType implements BinaryType {
    public static final String NAME = "CLIENT_CERTIFICATE_KEYSTORE_CONTENTS_FILE";

    private final CertificateEntityUtils certificateEntityUtils;
    private final CertificateChainLoader certificateChainLoader;
    private final CertificateInfoMapper certificateInfoMapper;
    private final CertificateRepository certificateRepository;

    @Override
    public String name() {
        return NAME;
    }

    @Override
    public Optional<BinaryResult> get(String id) {
        return zip(id);
    }

    private Optional<BinaryResult> zip(String id) {
        long certificateId = Long.parseLong(id);
        if (!certificateEntityUtils.isPrivateCert(certificateId)) {
            LOGGER.error("Certificate is not a keystore, unable to load private key.");
            return Optional.empty();
        }
        Optional<BinaryResult> result;

        CertificateEntity certificateEntity = certificateRepository
                .findById(certificateId)
                .orElseThrow(() -> new ObjectNotFoundException(CertificateDto.class, certificateId));

        try {
            File tempFile =
                    Files.createTempFile(TEMP_FILE_NAME_PREFIX, "certificates").toFile();

            List<X509Certificate> certificateChain = certificateChainLoader.getX509CertificateChain(certificateId);

            CertificateInfo leafCertificateInfo = certificateInfoMapper.certificateToInfo(certificateChain.getFirst());

            try (OutputStream outputStream = Files.newOutputStream(tempFile.toPath());
                    ZipOutputStream zipOutputStream = new ZipOutputStream(outputStream)) {

                String cleanedCommonName = NEXUSe2eFileUtils.cleanFileName(leafCertificateInfo.getCommonName());
                // Private key into zip file
                String privateKey = CertificatePemWriter.toEncryptedPrivateKey(
                        certificateEntityUtils.getPrivateKey(certificateId), certificateEntity.getPassword());
                ZipEntry zipEntry = new ZipEntry(cleanedCommonName + "_private_key.pem");
                zipOutputStream.putNextEntry(zipEntry);
                zipEntry.setSize(privateKey.length());
                IOUtils.write(privateKey, zipOutputStream, StandardCharsets.UTF_8);
                zipOutputStream.closeEntry();

                // Certificate chain into zip file
                String certificateChainContent = CertificatePemWriter.toCertificateChain(certificateChain);
                zipEntry = new ZipEntry(cleanedCommonName + "_certificate_chain.pem");
                zipOutputStream.putNextEntry(zipEntry);
                zipEntry.setSize(certificateChainContent.length());
                IOUtils.write(certificateChainContent, zipOutputStream, StandardCharsets.UTF_8);
                zipOutputStream.closeEntry();

                zipOutputStream.finish();
            }

            // CPD-OFF CertificateBinaryType
            result = Optional.of(BinaryResult.builder()
                    .fileName(getFileName(certificateEntity.getName()))
                    .size(tempFile.length())
                    .mimeType("application/zip")
                    .resource(new FileSystemResource(tempFile))
                    .build());
            // CPD-ON
        } catch (IOException e) {
            LOGGER.error("Could not zip certificates: {}", ErrorUtils.aggregate(e), e);
            result = Optional.empty();
        } catch (NexusException e) {
            LOGGER.error("Could not build certificate chain: {}", ErrorUtils.aggregate(e), e);
            result = Optional.empty();
        }
        return result;
    }

    public static String getFileName(String certificateName) {
        return NEXUSe2eFileUtils.cleanFileName("NEXUSe2e_privateKey_withChain_" + certificateName + "_"
                + LocalDateTime.now().withNano(0) + ".zip");
    }
}
