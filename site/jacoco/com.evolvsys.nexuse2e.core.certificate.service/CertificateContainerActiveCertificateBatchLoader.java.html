package com.evolvsys.nexuse2e.core.certificate.service;

import com.evolvsys.app.core.auth.model.CurrentUserData;
import com.evolvsys.app.core.graphql.dataloader.BatchLoaderContext;
import com.evolvsys.app.core.graphql.dataloader.GraphQLBatchLoader;
import com.evolvsys.nexuse2e.core.certificate.mapper.CertificateMapper;
import com.evolvsys.nexuse2e.core.certificate.model.CertificateContainerEntity;
import com.evolvsys.nexuse2e.core.certificate.model.CertificateDto;
import com.evolvsys.nexuse2e.core.certificate.model.CertificateEntity;
import com.evolvsys.nexuse2e.core.certificate.repository.CertificateRepository;
import java.util.HashMap;
import java.util.Map;
import java.util.Objects;
import java.util.Set;
import java.util.concurrent.CompletableFuture;
import java.util.concurrent.CompletionStage;
import java.util.stream.Collectors;
import lombok.RequiredArgsConstructor;
import org.apache.commons.collections4.CollectionUtils;
import org.dataloader.BatchLoaderEnvironment;
import org.springframework.stereotype.Component;

/**
 * <p>data loader to get the first active {@link CertificateEntity} of a {@link CertificateContainerEntity}.</p>
 *
 * @author Florian Drees
 * @since 06.03.2024
 */
@Component
@RequiredArgsConstructor
public class CertificateContainerActiveCertificateBatchLoader implements GraphQLBatchLoader<Long, CertificateDto> {
    private final CertificateMapper certificateMapper;
    private final CertificateRepository certificateRepository;

    @Override
    public CompletionStage<Map<Long, CertificateDto>> load(Set<Long> keys, BatchLoaderEnvironment environment) {
        BatchLoaderContext context = environment.getContext();
        CurrentUserData currentUserData = context.getCurrentUserData();

        Set<Long> notNullKeys = keys.stream().filter(Objects::nonNull).collect(Collectors.toSet());

        CompletableFuture<Map<Long, CertificateDto>> result;
        if (CollectionUtils.isEmpty(notNullKeys)) {
            result = CompletableFuture.completedFuture(new HashMap<>());
        } else {
            result = CompletableFuture.supplyAsync(() -> loadInternal(notNullKeys, currentUserData));
        }
        return result;
    }

    private Map<Long, CertificateDto> loadInternal(Set<Long> notNullKeys, CurrentUserData currentUserData) {
        Map<Long, CertificateDto> map = HashMap.newHashMap(notNullKeys.size());
        for (CertificateEntity certificateEntity :
                certificateRepository.findFirstByActiveAndCertificateContainer_Id(true, notNullKeys)) {
            map.put(
                    certificateEntity.getCertificateContainer().getId(),
                    certificateMapper.entityToDto(certificateEntity, currentUserData));
        }
        return map;
    }
}
