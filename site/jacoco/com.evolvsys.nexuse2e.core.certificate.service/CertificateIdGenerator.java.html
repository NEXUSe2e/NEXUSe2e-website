package com.evolvsys.nexuse2e.core.certificate.service;

import java.security.cert.X509Certificate;
import java.util.Optional;
import java.util.regex.Pattern;
import java.util.stream.Stream;
import javax.security.auth.x500.X500Principal;
import lombok.RequiredArgsConstructor;
import lombok.extern.slf4j.Slf4j;
import org.bouncycastle.asn1.ASN1ObjectIdentifier;
import org.bouncycastle.asn1.DERBMPString;
import org.bouncycastle.asn1.pkcs.PKCSObjectIdentifiers;
import org.bouncycastle.asn1.x500.RDN;
import org.bouncycastle.asn1.x500.X500Name;
import org.bouncycastle.asn1.x500.style.BCStyle;
import org.bouncycastle.asn1.x500.style.IETFUtils;
import org.bouncycastle.jce.interfaces.PKCS12BagAttributeCarrier;
import org.springframework.stereotype.Component;

/**
 * @author SvenBarden
 * @since 06.02.2024
 */
@Component
@RequiredArgsConstructor
@Slf4j
class CertificateIdGenerator {
    private static final Pattern NON_PRINTABLE_PATTERN = Pattern.compile("\\P{Alpha}+");

    private Optional<String> createFromFriendlyNameId(X509Certificate x509Certificate, boolean replaceNonPrintable) {
        Optional<String> result = Optional.empty();

        LOGGER.trace("creating id from friendly name for cert {}", x509Certificate);

        if (x509Certificate instanceof PKCS12BagAttributeCarrier pkcs12BagAttributeCarrier) {
            DERBMPString friendlyName = (DERBMPString)
                    pkcs12BagAttributeCarrier.getBagAttribute(PKCSObjectIdentifiers.pkcs_9_at_friendlyName);
            if (friendlyName != null) {
                String certId = friendlyName.getString();
                if (replaceNonPrintable) {
                    certId = NON_PRINTABLE_PATTERN.matcher(certId).replaceAll("");
                    LOGGER.trace("removed non printable characters from certId {}", certId);
                }

                LOGGER.debug("created cert id {}", certId);
                result = Optional.of(certId);
            }
        } else {
            LOGGER.error("certificate is not an instance of {}", PKCS12BagAttributeCarrier.class);
        }

        return result;
    }

    private Optional<String> getInfo(X500Name x500Name, ASN1ObjectIdentifier objectIdentifier) {
        RDN[] rdNs = x500Name.getRDNs(objectIdentifier);

        LOGGER.debug("createCertificateId from {}", objectIdentifier);

        return Stream.of(rdNs)
                .map(rdn -> IETFUtils.valueToString(rdn.getFirst().getValue()))
                .findFirst();
    }

    private Optional<String> createFromPrincipal(X509Certificate x509Certificate) {
        X500Principal x500Principal = x509Certificate.getSubjectX500Principal();
        X500Name x500Name = new X500Name(x500Principal.getName());

        return getInfo(x500Name, BCStyle.CN)
                .or(() -> getInfo(x500Name, BCStyle.O))
                .or(() -> getInfo(x500Name, BCStyle.OU));
    }

    public Optional<String> createCertificateId(X509Certificate x509Certificate) {
        return createCertificateId(x509Certificate, true);
    }

    public Optional<String> createCertificateId(X509Certificate x509Certificate, boolean replaceNonPrintable) {
        return createFromFriendlyNameId(x509Certificate, replaceNonPrintable)
                .or(() -> createFromPrincipal(x509Certificate));
    }
}
