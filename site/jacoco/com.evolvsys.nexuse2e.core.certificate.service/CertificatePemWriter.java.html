package com.evolvsys.nexuse2e.core.certificate.service;

import com.evolvsys.nexuse2e.core.exception.NexusError;
import com.evolvsys.nexuse2e.core.exception.NexusException;
import java.io.IOException;
import java.io.StringWriter;
import java.security.PrivateKey;
import java.security.cert.CertificateEncodingException;
import java.security.cert.X509Certificate;
import java.util.Collection;
import lombok.experimental.UtilityClass;
import lombok.extern.slf4j.Slf4j;
import org.apache.commons.lang3.StringUtils;
import org.bouncycastle.openssl.PKCS8Generator;
import org.bouncycastle.openssl.jcajce.JcaPEMWriter;
import org.bouncycastle.openssl.jcajce.JcaPKCS8Generator;
import org.bouncycastle.openssl.jcajce.JceOpenSSLPKCS8EncryptorBuilder;
import org.bouncycastle.operator.OperatorCreationException;
import org.bouncycastle.util.io.pem.PemObject;
import org.bouncycastle.util.io.pem.PemWriter;
import org.springframework.lang.Nullable;

/**
 * @author Florian Drees
 * @since 04.06.2024
 */
@UtilityClass
@Slf4j
public class CertificatePemWriter {
    private static final String CARRIAGE_RETURN = "\r";

    public static String toEncryptedPrivateKey(PrivateKey privateKey, @Nullable String password) throws NexusException {
        StringWriter stringWriter;
        if (StringUtils.isNotEmpty(password)) {
            stringWriter = new StringWriter();
            try (JcaPEMWriter pemWriter = new JcaPEMWriter(stringWriter)) {
                pemWriter.writeObject(new JcaPKCS8Generator(
                        privateKey,
                        new JceOpenSSLPKCS8EncryptorBuilder(PKCS8Generator.DES3_CBC)
                                .setPassword(password.toCharArray())
                                .build()));
            } catch (IOException | OperatorCreationException e) {
                throw new NexusException(
                        NexusError.builder()
                                .message("Private key could not be encrypted or written")
                                .build(),
                        e);
            }
        } else {
            LOGGER.trace("Writing unencrypted private key of keystore as pem format");
            stringWriter = toPem(privateKey.getEncoded(), "PRIVATE KEY");
        }
        return stringWriter.toString().replace(CARRIAGE_RETURN, "");
    }

    public static String toCertificate(X509Certificate certificate) throws NexusException {
        try {
            return toPem(certificate.getEncoded(), "CERTIFICATE").toString().replace(CARRIAGE_RETURN, "");
        } catch (CertificateEncodingException e) {
            throw new NexusException(
                    NexusError.builder()
                            .message("Certificate could not be encoded")
                            .extension("certificate", certificate)
                            .build(),
                    e);
        }
    }

    public static String toCertificateChain(Collection<X509Certificate> certificateChain) throws NexusException {
        StringBuilder stringBuilder = new StringBuilder();
        for (X509Certificate certificate : certificateChain) {
            stringBuilder.append(toCertificate(certificate));
        }
        return stringBuilder.toString().replace(CARRIAGE_RETURN, "");
    }

    private static void toPem(StringWriter stringWriter, byte[] content, String type) throws NexusException {
        try (PemWriter pemWriter = new PemWriter(stringWriter)) {
            pemWriter.writeObject(new PemObject(type, content));
        } catch (IOException e) {
            throw new NexusException(
                    NexusError.builder()
                            .message("Content could not be transformed into a string")
                            .extension("type", type)
                            .build(),
                    e);
        }
    }

    private static StringWriter toPem(byte[] content, String type) throws NexusException {
        StringWriter stringWriter = new StringWriter();
        toPem(stringWriter, content, type);
        return stringWriter;
    }
}
