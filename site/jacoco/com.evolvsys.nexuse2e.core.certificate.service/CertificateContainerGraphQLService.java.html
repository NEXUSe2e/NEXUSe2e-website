package com.evolvsys.nexuse2e.core.certificate.service;

import com.evolvsys.app.core.graphql.annotation.GraphQLService;
import com.evolvsys.app.core.graphql.dataloader.DataLoaderService;
import com.evolvsys.nexuse2e.core.certificate.model.CertificateContainerDto;
import com.evolvsys.nexuse2e.core.certificate.model.CertificateContainerFilter;
import com.evolvsys.nexuse2e.core.certificate.model.CertificateDto;
import com.evolvsys.nexuse2e.core.certificate.validation.ValidCertificateContainerHasCertificates;
import com.evolvsys.nexuse2e.core.certificate.validation.ValidCertificateContainerHasRoutes;
import com.evolvsys.nexuse2e.core.partner.model.PartnerDto;
import com.evolvsys.nexuse2e.core.partner.service.PartnerBatchLoader;
import io.leangen.graphql.annotations.*;
import io.leangen.graphql.execution.ResolutionEnvironment;
import jakarta.validation.Valid;
import java.util.Optional;
import java.util.Set;
import java.util.concurrent.CompletableFuture;
import lombok.RequiredArgsConstructor;
import org.springframework.data.domain.Page;
import org.springframework.data.domain.Pageable;

/**
 * @author SvenBarden
 * @since 07.02.2024
 */
@GraphQLService
@RequiredArgsConstructor
public class CertificateContainerGraphQLService {
    private final DataLoaderService dataLoaderService;
    private final CertificateContainerService certificateContainerService;

    @GraphQLMutation(name = "certificateContainer_delete")
    public boolean delete(
            @ValidCertificateContainerHasCertificates @ValidCertificateContainerHasRoutes long certificateContainerId) {
        certificateContainerService.delete(certificateContainerId);
        return true;
    }

    @GraphQLMutation(name = "certificateContainer_update")
    @GraphQLNonNull
    public CertificateContainerDto update(@GraphQLNonNull @Valid CertificateContainerDto certificateContainer) {
        return certificateContainerService.save(certificateContainer);
    }

    @GraphQLQuery(name = "certificateContainer_getById")
    public Optional<CertificateContainerDto> getById(long id) {
        return certificateContainerService.getById(id);
    }

    @GraphQLQuery(name = "certificateContainer_list")
    @GraphQLNonNull
    public Page<@GraphQLNonNull CertificateContainerDto> list(Pageable pageable, CertificateContainerFilter filter) {
        return certificateContainerService.list(pageable, filter);
    }

    @GraphQLQuery
    public CompletableFuture<Set<@GraphQLNonNull CertificateDto>> certificates(
            @GraphQLContext CertificateContainerDto certificateContainerDto,
            @GraphQLEnvironment ResolutionEnvironment resolutionEnvironment) {
        return dataLoaderService.loadRelation(
                resolutionEnvironment, CertificateContainerCertificatesBatchLoader.class, certificateContainerDto);
    }

    @GraphQLQuery
    public CompletableFuture<PartnerDto> partner(
            @GraphQLContext CertificateContainerDto certificateContainerDto,
            @GraphQLEnvironment ResolutionEnvironment resolutionEnvironment) {
        return dataLoaderService.load(
                resolutionEnvironment,
                PartnerBatchLoader.class,
                certificateContainerDto,
                "partnerId",
                CertificateContainerDto::getPartnerId);
    }

    @GraphQLQuery
    public CompletableFuture<CertificateDto> activeCertificate(
            @GraphQLContext CertificateContainerDto certificateContainerDto,
            @GraphQLEnvironment ResolutionEnvironment resolutionEnvironment) {
        return dataLoaderService.load(
                resolutionEnvironment,
                CertificateContainerActiveCertificateBatchLoader.class,
                certificateContainerDto,
                CertificateContainerDto::getId);
    }
}
