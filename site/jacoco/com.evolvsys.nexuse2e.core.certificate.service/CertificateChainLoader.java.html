package com.evolvsys.nexuse2e.core.certificate.service;

import com.evolvsys.app.core.exception.ObjectNotFoundException;
import com.evolvsys.nexuse2e.core.certificate.mapper.CertificateInfoMapper;
import com.evolvsys.nexuse2e.core.certificate.model.*;
import com.evolvsys.nexuse2e.core.certificate.repository.CertificateRepository;
import com.evolvsys.nexuse2e.core.exception.NexusException;
import java.security.cert.X509Certificate;
import java.util.ArrayList;
import java.util.List;
import javax.security.auth.x500.X500Principal;
import lombok.RequiredArgsConstructor;
import lombok.extern.slf4j.Slf4j;
import org.springframework.stereotype.Component;
import org.springframework.transaction.annotation.Transactional;

/**
 * @author SvenBarden
 * @since 22.02.2024
 */
@Component
@RequiredArgsConstructor
@Slf4j
public class CertificateChainLoader {
    private final CertificateRepository certificateRepository;
    private final CertificateInfoMapper certificateInfoMapper;

    @Transactional(readOnly = true)
    public List<CertificateInfoDto> getCertificateChain(long certificateId) throws NexusException {
        List<X509Certificate> certChain = getX509CertificateChain(certificateId);

        List<CertificateInfoDto> result;

        X509Certificate lastInChain = certChain.getLast();
        X500Principal issuerX500Principal = lastInChain.getIssuerX500Principal();
        if (lastInChain.getSubjectX500Principal().getName().equals(issuerX500Principal.getName()) == false) {
            LOGGER.trace(
                    "last cert in chain is not self signed, issuer: {}, subject: {}",
                    issuerX500Principal.getName(),
                    lastInChain.getSubjectX500Principal().getName());
            result = new ArrayList<>(certChain.size() + 1);
            result.addAll(certChain.stream()
                    .map(certificateInfoMapper::certificateToDto)
                    .toList());
            CertificateInfoDto.CertificateInfoDtoBuilder missingBuilder =
                    CertificateInfoDto.builder().validity(CertificateValidity.MISSING);
            certificateInfoMapper.updateInfo(issuerX500Principal, missingBuilder);
            result.add(missingBuilder.build());
        } else {
            LOGGER.trace("last cert in chain is self signed");
            result = certChain.stream()
                    .map(certificateInfoMapper::certificateToDto)
                    .toList();
        }
        return result;
    }

    @Transactional(readOnly = true)
    public List<X509Certificate> getX509CertificateChain(long certificateId) throws NexusException {
        List<CertificateEntity> activeCas = certificateRepository.findByTypeAndActive(CertificateType.CA, true);

        CertificateEntity certificateEntity = certificateRepository
                .findById(certificateId)
                .orElseThrow(() -> new ObjectNotFoundException(CertificateDto.class, certificateId));

        X509Certificate x509Certificate;
        if (certificateEntity.getType() == CertificateType.LOCAL) {
            x509Certificate = certificateInfoMapper.getPrivateCert(certificateEntity);
        } else {
            x509Certificate = certificateInfoMapper.getPublicCert(certificateEntity);
        }

        List<X509Certificate> cas = new ArrayList<>(activeCas.size());
        for (CertificateEntity activeCa : activeCas) {
            cas.add(certificateInfoMapper.getPublicCert(activeCa));
        }
        return CertificateHelper.buildChain(x509Certificate, cas);
    }
}
