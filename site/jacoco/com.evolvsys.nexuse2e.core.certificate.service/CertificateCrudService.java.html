package com.evolvsys.nexuse2e.core.certificate.service;

import com.evolvsys.app.core.auth.service.CurrentUserService;
import com.evolvsys.app.core.service.BaseService;
import com.evolvsys.app.core.service.ListService;
import com.evolvsys.nexuse2e.core.certificate.mapper.CertificateMapper;
import com.evolvsys.nexuse2e.core.certificate.model.CertificateDto;
import com.evolvsys.nexuse2e.core.certificate.model.CertificateEntity;
import com.evolvsys.nexuse2e.core.certificate.model.CertificateFilter;
import com.evolvsys.nexuse2e.core.certificate.model.CertificateType;
import com.evolvsys.nexuse2e.core.certificate.repository.CertificateRepository;
import edu.umd.cs.findbugs.annotations.SuppressFBWarnings;
import java.time.ZonedDateTime;
import java.util.ArrayList;
import java.util.List;
import lombok.extern.slf4j.Slf4j;
import org.springframework.data.domain.Page;
import org.springframework.data.domain.Pageable;
import org.springframework.stereotype.Service;
import org.springframework.transaction.annotation.Transactional;

/**
 * @author SvenBarden
 * @since 07.02.2024
 */
@Service
@Slf4j
public class CertificateCrudService extends BaseService<CertificateDto, CertificateEntity, CertificateRepository> {
    private final ListService listService;

    public CertificateCrudService(
            CertificateMapper mapper,
            CertificateRepository repository,
            CurrentUserService currentUserService,
            ListService listService) {
        super(mapper, repository, currentUserService);
        this.listService = listService;
    }

    @Override
    @Transactional
    public CertificateDto save(CertificateDto dto) {
        CertificateEntity entity = getEntity(dto);

        if (dto.getWriteMetaData().getChangedProperties().contains("active")) {
            LOGGER.debug("changing active state for certificate {}", dto);
            if (dto.isActive()) {
                enable(entity);
            } else {
                disable(entity);
            }
        }
        entity = save(entity);

        return mapEntityToDto(entity);
    }

    @Override
    protected CertificateEntity createEntity() {
        return new CertificateEntity();
    }

    public List<CertificateDto> getAll() {
        return getRepository().findAll().stream().map(this::mapEntityToDto).toList();
    }

    public Page<CertificateDto> list(Pageable pageable, CertificateFilter certificateFilter) {
        return listService.list(pageable, certificateFilter, this).map(this::mapEntityToDto);
    }

    @SuppressFBWarnings("UAA_USE_ADD_ALL")
    private void enable(CertificateEntity entity) {
        List<CertificateEntity> entitiesToPersist = new ArrayList<>();
        entitiesToPersist.add(entity);

        if (entity.getType() == CertificateType.LOCAL) {
            List<CertificateEntity> otherEntities =
                    getRepository().findByCertificateContainerAndActive(entity.getCertificateContainer(), true);

            for (CertificateEntity otherEntity : otherEntities) {
                otherEntity.setActive(false);
                otherEntity.setDisabled(ZonedDateTime.now());
                entitiesToPersist.add(otherEntity);
            }
        }

        entity.setEnabled(ZonedDateTime.now());
        entity.setActive(true);

        getRepository().saveAll(entitiesToPersist);
    }

    private void disable(CertificateEntity entity) {
        entity.setDisabled(ZonedDateTime.now());
        entity.setActive(false);

        getRepository().save(entity);
    }

    @Transactional
    public void enable(long certificateId) {
        CertificateEntity entity = findOrThrow(certificateId);

        enable(entity);
    }

    public void disable(long certificateId) {
        CertificateEntity entity = findOrThrow(certificateId);

        disable(entity);
    }
}
