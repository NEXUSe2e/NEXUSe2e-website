package com.evolvsys.nexuse2e.core.certificate.service;

import com.evolvsys.nexuse2e.core.certificate.CertificateConstants;
import com.evolvsys.nexuse2e.core.certificate.model.CertificateEntity;
import com.evolvsys.nexuse2e.core.exception.NexusError;
import com.evolvsys.nexuse2e.core.exception.NexusException;
import java.io.IOException;
import java.io.InputStream;
import java.security.*;
import java.security.cert.CertificateException;
import java.sql.SQLException;
import java.util.Enumeration;
import java.util.Optional;
import java.util.stream.Collectors;
import java.util.stream.Stream;
import lombok.NonNull;
import lombok.experimental.UtilityClass;
import org.springframework.lang.Nullable;

/**
 * @author Florian Drees
 * @since 06.06.2024
 */
@UtilityClass
public class CertificateKeystoreUtils {
    public static KeyStore toKeyStore(CertificateEntity certificateEntity) throws NexusException {
        try {
            return toKeystore(
                    certificateEntity.getFile().getFileContent().getContent().getBinaryStream(),
                    certificateEntity.getPassword());
        } catch (NexusException e) {
            throw new NexusException(
                    NexusError.builder()
                            .message(e.getMessage())
                            .extension("id", certificateEntity.getId())
                            .extension("name", certificateEntity.getName())
                            .build(),
                    e);
        } catch (SQLException e) {
            throw new NexusException(
                    NexusError.builder()
                            .message("Unable to read keystore from certificate entity")
                            .extension("id", certificateEntity.getId())
                            .extension("name", certificateEntity.getName())
                            .build(),
                    e);
        }
    }

    public static KeyStore toKeystore(@NonNull InputStream inputStream, @Nullable String password)
            throws NexusException {
        try {
            KeyStore keyStore = KeyStore.getInstance(
                    CertificateConstants.DEFAULT_KEY_STORE_TYPE, CertificateConstants.DEFAULT_JCE_PROVIDER);
            keyStore.load(
                    inputStream,
                    Optional.ofNullable(password).map(String::toCharArray).orElse(null));
            return keyStore;
        } catch (CertificateException | KeyStoreException | IOException | NoSuchAlgorithmException e) {
            throw new NexusException(
                    NexusError.builder()
                            .message("Could not create keystore from entity")
                            .build(),
                    e);
        } catch (NoSuchProviderException e) {
            String allProviders =
                    Stream.of(Security.getProviders()).map(Provider::getName).collect(Collectors.joining(", "));
            throw new NexusException(
                    NexusError.builder()
                            .message("jce provider not found")
                            .extension("provider", CertificateConstants.DEFAULT_JCE_PROVIDER)
                            .extension("availableProviders", allProviders)
                            .build(),
                    e);
        }
    }

    public static PrivateKey getPrivateKey(@NonNull KeyStore keyStore, @Nullable String password)
            throws NexusException {
        String keyAlias = getKeyAlias(keyStore);
        if (keyAlias == null) {
            throw new NexusException(NexusError.builder()
                    .message("No key alias found in keystore")
                    .build());
        }

        try {
            return (PrivateKey) keyStore.getKey(
                    keyAlias,
                    Optional.ofNullable(password).map(String::toCharArray).orElse(null));
        } catch (KeyStoreException e) {
            throw new NexusException(
                    NexusError.builder()
                            .message("Error loading aliases from keystore")
                            .build(),
                    e);
        } catch (NoSuchAlgorithmException | UnrecoverableKeyException e) {
            throw new NexusException(
                    NexusError.builder()
                            .message("Error loading private key from keystore")
                            .build(),
                    e);
        }
    }

    public static String getKeyAlias(@NonNull KeyStore keyStore) throws NexusException {
        String keyAlias = null;
        try {
            Enumeration<String> aliases = keyStore.aliases();
            while (aliases.hasMoreElements() && keyAlias == null) {
                String alias = aliases.nextElement();
                if (keyStore.isKeyEntry(alias)) {
                    keyAlias = alias;
                }
            }
        } catch (KeyStoreException e) {
            throw new NexusException(
                    NexusError.builder()
                            .message("Could not read aliases from keystore")
                            .build(),
                    e);
        }
        return keyAlias;
    }
}
