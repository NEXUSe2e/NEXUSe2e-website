package com.evolvsys.nexuse2e.core.settings.service;

import com.evolvsys.app.core.auth.service.CurrentUserService;
import com.evolvsys.app.core.service.BaseService;
import com.evolvsys.nexuse2e.core.config.NEXUSe2eConfigurationProperties;
import com.evolvsys.nexuse2e.core.settings.mapper.ApplicationsSettingsMapper;
import com.evolvsys.nexuse2e.core.settings.model.ApplicationSettingsDto;
import com.evolvsys.nexuse2e.core.settings.model.ApplicationSettingsEntity;
import com.evolvsys.nexuse2e.core.settings.repository.ApplicationSettingsRepository;
import java.time.ZoneId;
import java.util.Optional;
import java.util.Set;
import java.util.stream.Collectors;
import org.apache.commons.collections4.CollectionUtils;
import org.springframework.stereotype.Service;

/**
 * @author SvenBarden
 * @since 27.03.2024
 */
@Service
public class ApplicationSettingsCrudService
        extends BaseService<ApplicationSettingsDto, ApplicationSettingsEntity, ApplicationSettingsRepository> {
    private final NEXUSe2eConfigurationProperties configurationProperties;
    private final ApplicationSettingsService applicationSettingsService;

    public ApplicationSettingsCrudService(
            ApplicationsSettingsMapper mapper,
            ApplicationSettingsRepository repository,
            CurrentUserService currentUserService,
            NEXUSe2eConfigurationProperties configurationProperties,
            ApplicationSettingsService applicationSettingsService) {
        super(mapper, repository, currentUserService);
        this.configurationProperties = configurationProperties;
        this.applicationSettingsService = applicationSettingsService;
    }

    public Set<String> listTimeZones() {
        Set<String> allZones = ZoneId.getAvailableZoneIds();
        Set<String> result;
        if (CollectionUtils.isNotEmpty(configurationProperties.getTimeZones())) {
            result = allZones.stream()
                    .filter(s -> configurationProperties.getTimeZones().contains(s))
                    .collect(Collectors.toSet());
        } else {
            result = allZones;
        }
        return result;
    }

    public Optional<ApplicationSettingsDto> getSystemSettings() {
        return applicationSettingsService.getSystemSettings().map(this::mapEntityToDto);
    }

    @Override
    protected ApplicationSettingsEntity createEntity() {
        ApplicationSettingsEntity applicationSettingsEntity = applicationSettingsService.getDefault();
        applicationSettingsEntity.setTenant(currentUserService.getActiveTenant());
        return applicationSettingsEntity;
    }
}
