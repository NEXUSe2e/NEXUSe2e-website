package com.evolvsys.nexuse2e.core.settings.service;

import com.evolvsys.nexuse2e.core.settings.model.ApplicationSettingsEntity;
import com.evolvsys.nexuse2e.core.settings.repository.ApplicationSettingsRepository;
import java.util.Optional;
import java.util.TimeZone;
import lombok.RequiredArgsConstructor;
import org.springframework.stereotype.Service;

/**
 * @author SvenBarden
 * @since 02.04.2024
 */
@Service
@RequiredArgsConstructor
public class ApplicationSettingsService {
    private final ApplicationSettingsRepository applicationSettingsRepository;

    public Optional<ApplicationSettingsEntity> getSystemSettings() {
        return applicationSettingsRepository.findFirstByTenant_Id(1L);
    }

    public ApplicationSettingsEntity getSystemSettingsOrDefault() {
        return getSystemSettings().orElseGet(this::getDefault);
    }

    public ApplicationSettingsEntity getDefault() {
        ApplicationSettingsEntity applicationSettingsEntity = new ApplicationSettingsEntity();
        applicationSettingsEntity.setThemeColor("#FF6F00");
        applicationSettingsEntity.setItemsPerPage(25);
        applicationSettingsEntity.setTimeZone(TimeZone.getDefault().getID());
        return applicationSettingsEntity;
    }
}
