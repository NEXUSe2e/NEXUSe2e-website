package com.evolvsys.nexuse2e.core.settings.service;

import com.evolvsys.app.core.graphql.annotation.GraphQLService;
import com.evolvsys.nexuse2e.core.settings.model.ApplicationSettingsDto;
import io.leangen.graphql.annotations.GraphQLMutation;
import io.leangen.graphql.annotations.GraphQLNonNull;
import io.leangen.graphql.annotations.GraphQLQuery;
import jakarta.validation.Valid;
import java.util.Optional;
import java.util.Set;
import lombok.RequiredArgsConstructor;

/**
 * @author SvenBarden
 * @since 27.03.2024
 */
@GraphQLService
@RequiredArgsConstructor
public class ApplicationSettingsGraphQLService {
    private final ApplicationSettingsCrudService applicationSettingsService;

    @GraphQLMutation(name = "applicationSettings_update")
    public ApplicationSettingsDto update(@Valid ApplicationSettingsDto applicationSettings) {
        return applicationSettingsService.save(applicationSettings);
    }

    @GraphQLQuery(name = "applicationSettings_create")
    public ApplicationSettingsDto create() {
        return applicationSettingsService.create();
    }

    @GraphQLQuery(name = "applicationSettings_get")
    public Optional<ApplicationSettingsDto> get() {
        return applicationSettingsService.getSystemSettings();
    }

    @GraphQLQuery(name = "applicationSettings_listTimeZones")
    public Set<@GraphQLNonNull String> listTimeZones() {
        return applicationSettingsService.listTimeZones();
    }
}
