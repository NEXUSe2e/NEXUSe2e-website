package com.evolvsys.nexuse2e.core.message.state.event;

import static com.evolvsys.nexuse2e.core.message.MessageStatus.RESOLVED;

import com.evolvsys.common.mdc.MDCService;
import com.evolvsys.nexuse2e.core.exception.NexusError;
import com.evolvsys.nexuse2e.core.exception.NexusException;
import com.evolvsys.nexuse2e.core.logging.service.LoggingService;
import com.evolvsys.nexuse2e.core.message.context.ExecutionContext;
import com.evolvsys.nexuse2e.core.message.mapper.ContextConversationMapper;
import com.evolvsys.nexuse2e.core.message.mapper.ContextMessageMapper;
import com.evolvsys.nexuse2e.core.message.model.ContextConversation;
import com.evolvsys.nexuse2e.core.message.repository.ConversationRepository;
import com.evolvsys.nexuse2e.core.message.repository.MessageRepository;
import com.evolvsys.nexuse2e.core.message.state.MessageEventType;
import com.evolvsys.nexuse2e.core.model.entity.ConversationEntity;
import com.evolvsys.nexuse2e.core.model.entity.ConversationStatus;
import com.evolvsys.nexuse2e.core.model.entity.MessageEntity;
import com.evolvsys.nexuse2e.internal.Message;
import com.evolvsys.nexuse2e.internal.MessageType;
import lombok.RequiredArgsConstructor;
import lombok.extern.slf4j.Slf4j;
import org.springframework.stereotype.Component;
import org.springframework.transaction.annotation.Transactional;

/**
 * @author SvenBarden
 * @since 23.01.2024
 */
@Component
@RequiredArgsConstructor
@Slf4j
class MessageResolveEventHandler implements MessageEventHandler {
    private final MessageRepository messageRepository;
    private final ConversationRepository conversationRepository;
    private final ContextMessageMapper contextMessageMapper;
    private final ContextConversationMapper contextConversationMapper;
    private final LoggingService loggingService;

    protected void handleInternal(ExecutionContext executionContext, Message message, MessageEntity messageEntity)
            throws NexusException {
        contextMessageMapper.updateEntityWithPayloads(message, executionContext.getMessageContext(), messageEntity);

        messageEntity.setStatus(RESOLVED);
        message.setMessageStatus(messageEntity.getStatus());

        ConversationEntity conversationEntity = messageEntity.getConversation();
        if (message.getType() == MessageType.NORMAL) {
            handleNormalMessage(executionContext, message, messageEntity, conversationEntity);
        }

        messageEntity = messageRepository.save(messageEntity);

        contextMessageMapper.updateMessage(messageEntity, message);
    }

    private void handleNormalMessage(
            ExecutionContext executionContext,
            Message message,
            MessageEntity messageEntity,
            ConversationEntity conversationEntity)
            throws NexusException {
        if (conversationEntity == null) {
            if (executionContext.getAction().isValidStartAction() == false) {
                throw new NexusException(NexusError.builder()
                        .message("action is not a valid start action")
                        .extension("action", executionContext.getActionId())
                        .build());
            }
            conversationEntity = new ConversationEntity();
        }

        conversationEntity.setStatus(ConversationStatus.QUEUED);
        conversationEntity.setLastSourcePartner(messageEntity.getSourcePartner());
        conversationEntity.setLastTargetPartner(messageEntity.getTargetPartner());
        conversationEntity.setLastDirection(messageEntity.getDirection());

        ContextConversation contextConversation = ContextConversation.builder()
                .conversationId(message.getConversationId())
                .choreographyId(executionContext.getChoreographyId())
                .currentAction(executionContext.getActionId())
                .build();

        loggingService.persistAndUpdate(
                message.getId(),
                executionContext.getBeanFactory().getBean(MDCService.class).getId());

        contextConversationMapper.updateEntity(contextConversation, conversationEntity);

        conversationEntity = conversationRepository.save(conversationEntity);

        messageEntity.setConversation(conversationEntity);
    }

    @Override
    @Transactional(rollbackFor = NexusException.class)
    public void handle(ExecutionContext executionContext) throws NexusException {
        Message message = executionContext.getMessage();

        MessageEntity messageEntity =
                messageRepository.findByMessageId(message.getId()).orElseGet(MessageEntity::new);

        if (messageEntity.getId() == null) {
            LOGGER.trace("update new resolved message {}", message.getId());
            if (message.getType() == MessageType.ACKNOWLEDGMENT) {
                LOGGER.info(
                        "Receiving acknowledgment {} for message {}", message.getId(), message.getReferenceMessageId());
            } else {
                LOGGER.info(
                        "Receiving {} message {} from {}",
                        message.getType(),
                        message.getId(),
                        executionContext.getSourcePartnerId());
            }
            handleInternal(executionContext, message, messageEntity);
        } else {
            LOGGER.info("Received message {} already exists", message.getId());
        }
    }

    @Override
    public MessageEventType type() {
        return MessageEventType.RESOLVED;
    }

    @Override
    public int getOrder() {
        return 100;
    }
}
