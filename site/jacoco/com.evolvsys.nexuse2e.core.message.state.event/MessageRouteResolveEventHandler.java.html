package com.evolvsys.nexuse2e.core.message.state.event;

import com.evolvsys.nexuse2e.core.exception.NexusError;
import com.evolvsys.nexuse2e.core.exception.NexusException;
import com.evolvsys.nexuse2e.core.message.context.ExecutionContext;
import com.evolvsys.nexuse2e.core.message.context.RouteKey;
import com.evolvsys.nexuse2e.core.message.repository.MessageRepository;
import com.evolvsys.nexuse2e.core.message.state.MessageEventType;
import com.evolvsys.nexuse2e.core.model.entity.MessageEntity;
import com.evolvsys.nexuse2e.core.route.service.ContextRouteFactory;
import com.evolvsys.nexuse2e.internal.MessageType;
import lombok.RequiredArgsConstructor;
import lombok.extern.slf4j.Slf4j;
import org.apache.commons.lang3.StringUtils;
import org.springframework.stereotype.Component;
import org.springframework.transaction.annotation.Transactional;

/**
 * @author SvenBarden
 * @since 16.02.2024
 */
@Component
@RequiredArgsConstructor
@Slf4j
class MessageRouteResolveEventHandler implements MessageEventHandler {
    private final ContextRouteFactory contextRouteFactory;
    private final MessageRepository messageRepository;
    private static final String ACK_ACTION_ID = "Acknowledgment";

    @Override
    @Transactional(readOnly = true)
    public void handle(ExecutionContext executionContext) throws NexusException {
        RouteKey routeKey;
        if (executionContext.getMessage().getType() == MessageType.ACKNOWLEDGMENT) {
            if (ACK_ACTION_ID.equalsIgnoreCase(executionContext.getActionId())) {
                String referenceMessageId = executionContext.getMessage().getReferenceMessageId();
                if (StringUtils.isBlank(referenceMessageId)) {
                    throw new NexusException(NexusError.builder()
                            .message("Acknowledgment message requires referenceMessageId")
                            .extension("sourcePartnerId", executionContext.getSourcePartnerId())
                            .build());
                }
                MessageEntity messageEntity =
                        messageRepository.findByMessageId(referenceMessageId).orElse(null);
                if (messageEntity == null) {
                    routeKey = null;
                } else {
                    String actionId = messageEntity.getAction().getActionId();
                    LOGGER.debug("setting actionId from Acknowledgment to originalActionId {}", actionId);
                    executionContext.setActionId(actionId);

                    routeKey = new RouteKey(
                            executionContext.getChoreography().getId(),
                            messageEntity.getAction().getId(),
                            executionContext.getSourcePartnerId(),
                            executionContext.getTargetPartnerId());
                }
            } else {
                throw new NexusException(NexusError.builder()
                        .message("Acknowledgment message requires actionId 'Acknowledgment'")
                        .extension("action", executionContext.getActionId())
                        .extension("sourcePartnerId", executionContext.getSourcePartnerId())
                        .build());
            }
        } else if (StringUtils.isNoneBlank(
                executionContext.getActionId(),
                executionContext.getChoreographyId(),
                executionContext.getSourcePartnerId(),
                executionContext.getTargetPartnerId())) {
            routeKey = new RouteKey(
                    executionContext.getChoreography().getId(),
                    executionContext.getAction().getId(),
                    executionContext.getSourcePartnerId(),
                    executionContext.getTargetPartnerId());
        } else {
            routeKey = null;
        }

        if (routeKey != null) {
            EventUtils.setRoute(routeKey, contextRouteFactory, executionContext);
        }
    }

    @Override
    public MessageEventType type() {
        return MessageEventType.RESOLVING;
    }

    @Override
    public int getOrder() {
        return 20;
    }
}
