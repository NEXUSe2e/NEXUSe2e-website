package com.evolvsys.nexuse2e.core.message.state.event;

import com.evolvsys.nexuse2e.core.config.NEXUSe2eConfigurationProperties;
import com.evolvsys.nexuse2e.core.exception.NexusException;
import com.evolvsys.nexuse2e.core.message.MessageStatus;
import com.evolvsys.nexuse2e.core.message.context.ExecutionContext;
import com.evolvsys.nexuse2e.core.message.mapper.ContextMessageMapper;
import com.evolvsys.nexuse2e.core.message.repository.ConversationRepository;
import com.evolvsys.nexuse2e.core.message.repository.MessageRepository;
import com.evolvsys.nexuse2e.core.model.entity.ConversationEntity;
import com.evolvsys.nexuse2e.core.model.entity.ConversationStatus;
import com.evolvsys.nexuse2e.core.model.entity.MessageEntity;
import com.evolvsys.nexuse2e.core.partner.model.ContextConnection;
import com.evolvsys.nexuse2e.internal.Message;
import com.evolvsys.nexuse2e.internal.MessageType;
import java.time.ZonedDateTime;
import lombok.RequiredArgsConstructor;
import lombok.extern.slf4j.Slf4j;
import org.springframework.transaction.annotation.Transactional;

/**
 * @author SvenBarden
 * @since 05.03.2024
 */
@RequiredArgsConstructor
@Slf4j
abstract class BaseMessageErrorEventHandler implements MessageEventHandler {
    private final MessageRepository messageRepository;
    private final ConversationRepository conversationRepository;
    private final ContextMessageMapper contextMessageMapper;
    private final NEXUSe2eConfigurationProperties configurationProperties;

    protected abstract MessageStatus requeueStatusToSet();

    protected abstract MessageStatus errorStatusToSet();

    protected MessageEntity newMessage(ExecutionContext executionContext) {
        MessageEntity messageEntity = new MessageEntity();
        contextMessageMapper.updateEntity(
                executionContext.getMessage(), executionContext.getMessageContext(), messageEntity);
        return messageEntity;
    }

    protected void handleNormalMessage(ExecutionContext executionContext, Message message) {
        MessageEntity messageEntity =
                messageRepository.findByMessageId(message.getId()).orElseGet(() -> newMessage(executionContext));

        ContextConnection targetConnection = executionContext.getRoute().getTargetConnection();
        messageEntity.setMaxRetries(targetConnection.getMaxRetries());

        EventUtils.updateRetries(
                messageEntity,
                targetConnection,
                configurationProperties.getMessageQueue().getDefaultRetryInterval());

        messageEntity.setLastTry(ZonedDateTime.now());

        ConversationEntity conversation = messageEntity.getConversation();

        if (messageEntity.getRetries() < messageEntity.getMaxRetries()) {
            LOGGER.trace("setting message {} to queue status {}", message.getId(), requeueStatusToSet());
            messageEntity.setStatus(requeueStatusToSet());
            message.setMessageStatus(requeueStatusToSet());
            conversation.setStatus(ConversationStatus.QUEUED);
        } else {
            updateErrorStatus(messageEntity, message);
            conversation.setStatus(ConversationStatus.FAILED);
        }

        LOGGER.trace("set message {} nextRequeue time to {}", message.getId(), messageEntity.getNextRequeue());

        messageRepository.save(messageEntity);
        conversationRepository.save(conversation);
    }

    protected void handleAcknowledgment(ExecutionContext executionContext, Message message) {
        MessageEntity messageEntity =
                messageRepository.findByMessageId(message.getId()).orElseGet(() -> newMessage(executionContext));

        messageEntity.setMaxRetries(0);
        messageEntity.setLastTry(ZonedDateTime.now());

        updateErrorStatus(messageEntity, message);

        messageRepository.save(messageEntity);
    }

    protected void updateErrorStatus(MessageEntity messageEntity, Message message) {
        LOGGER.trace("setting message {} to error status {}", message.getId(), errorStatusToSet());
        messageEntity.setStatus(errorStatusToSet());
        messageEntity.setNextRequeue(null);
        message.setMessageStatus(errorStatusToSet());
    }

    @Override
    @Transactional(rollbackFor = NexusException.class)
    public void handle(ExecutionContext executionContext) throws NexusException {
        Message message = executionContext.getMessage();

        if (message.getType() == MessageType.ACKNOWLEDGMENT) {
            handleAcknowledgment(executionContext, message);
        } else {
            handleNormalMessage(executionContext, message);
        }
    }
}
