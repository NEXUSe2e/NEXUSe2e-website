package com.evolvsys.nexuse2e.core.message.state.event;

import com.evolvsys.nexuse2e.core.config.model.PartnerType;
import com.evolvsys.nexuse2e.core.exception.NexusError;
import com.evolvsys.nexuse2e.core.exception.NexusException;
import com.evolvsys.nexuse2e.core.message.context.ExecutionContext;
import com.evolvsys.nexuse2e.core.message.state.MessageEventType;
import com.evolvsys.nexuse2e.core.partner.repository.PartnerRepository;
import lombok.RequiredArgsConstructor;
import lombok.extern.slf4j.Slf4j;
import org.apache.commons.lang3.StringUtils;
import org.springframework.stereotype.Component;

/**
 * @author SvenBarden
 * @since 16.02.2024
 */
@Component
@RequiredArgsConstructor
@Slf4j
class MessageDefaultPartnerResolveEventHandler implements MessageEventHandler {
    private final PartnerRepository partnerRepository;

    @Override
    public void handle(ExecutionContext executionContext) throws NexusException {
        if (StringUtils.isBlank(executionContext.getSourcePartnerId())) {
            LOGGER.debug("no source partner was set, using default local identity");
            executionContext.setSourcePartnerId(partnerRepository
                    .getDefaultLocalIdentityName(PartnerType.INTERN)
                    .orElseThrow(() -> new NexusException(NexusError.builder()
                            .message("no default local identity found for source partner")
                            .build())));
        }
        if (StringUtils.isBlank(executionContext.getTargetPartnerId())) {
            LOGGER.debug("no target partner was set, using default local identity");
            executionContext.setTargetPartnerId(partnerRepository
                    .getDefaultLocalIdentityName(PartnerType.INTERN)
                    .orElseThrow(() -> new NexusException(NexusError.builder()
                            .message("no default local identity found for target partner")
                            .build())));
        }
    }

    @Override
    public MessageEventType type() {
        return MessageEventType.RESOLVING;
    }

    @Override
    public int getOrder() {
        return 10;
    }
}
