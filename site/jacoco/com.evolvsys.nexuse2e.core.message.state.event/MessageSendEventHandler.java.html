package com.evolvsys.nexuse2e.core.message.state.event;

import com.evolvsys.nexuse2e.core.config.NEXUSe2eConfigurationProperties;
import com.evolvsys.nexuse2e.core.exception.NexusException;
import com.evolvsys.nexuse2e.core.message.MessageStatus;
import com.evolvsys.nexuse2e.core.message.context.ExecutionContext;
import com.evolvsys.nexuse2e.core.message.mapper.ContextMessageMapper;
import com.evolvsys.nexuse2e.core.message.repository.ConversationRepository;
import com.evolvsys.nexuse2e.core.message.repository.MessageRepository;
import com.evolvsys.nexuse2e.core.message.state.MessageEventType;
import com.evolvsys.nexuse2e.core.model.entity.ConversationEntity;
import com.evolvsys.nexuse2e.core.model.entity.ConversationStatus;
import com.evolvsys.nexuse2e.core.model.entity.MessageEntity;
import com.evolvsys.nexuse2e.core.partner.model.ContextConnection;
import com.evolvsys.nexuse2e.internal.Message;
import com.evolvsys.nexuse2e.internal.MessageType;
import java.time.ZonedDateTime;
import lombok.RequiredArgsConstructor;
import lombok.extern.slf4j.Slf4j;
import org.springframework.stereotype.Component;
import org.springframework.transaction.annotation.Transactional;

/**
 * @author SvenBarden
 * @since 29.01.2024
 */
@Component
@RequiredArgsConstructor
@Slf4j
class MessageSendEventHandler implements MessageEventHandler {
    private final MessageRepository messageRepository;
    private final ContextMessageMapper contextMessageMapper;
    private final ConversationRepository conversationRepository;
    private final NEXUSe2eConfigurationProperties configurationProperties;

    @Override
    @Transactional(rollbackFor = NexusException.class)
    public void handle(ExecutionContext executionContext) throws NexusException {
        Message message = executionContext.getMessage();

        MessageEntity messageEntity =
                messageRepository.findByMessageId(message.getId()).orElseGet(MessageEntity::new);

        contextMessageMapper.updateEntity(message, executionContext.getMessageContext(), messageEntity);

        ConversationEntity conversationEntity = messageEntity.getConversation();

        MessageStatus lastMessageStatus = messageEntity.getStatus();

        messageEntity.setStatus(MessageStatus.TRANSFERRED);
        messageEntity.setSent(ZonedDateTime.now());

        ContextConnection targetConnection = executionContext.getRoute().getTargetConnection();
        EventUtils.updateRetries(
                messageEntity,
                targetConnection,
                configurationProperties.getMessageQueue().getDefaultRetryInterval());

        messageEntity.setLastTry(ZonedDateTime.now());
        if (messageEntity.getType() == MessageType.ACKNOWLEDGMENT) {
            messageEntity.setOutboundAcknowledgment(true);
            messageEntity.setFinished(ZonedDateTime.now());
        } else if (message.getType() == MessageType.NORMAL) {
            messageEntity.setMaxRetries(targetConnection.getMaxRetries());
            boolean reliable = targetConnection.isReliable();

            if (reliable) {
                messageEntity.setStatus(MessageStatus.WAITING_FOR_ACKNOWLEDGMENT);
                conversationEntity.setStatus(ConversationStatus.WAITING_FOR_ACKNOWLEDGMENT);
                LOGGER.trace("target connection is reliable, set status to: {}", messageEntity.getStatus());
                if (messageEntity.getRetries() > 0 && lastMessageStatus == MessageStatus.WAITING_FOR_ACKNOWLEDGMENT) {
                    LOGGER.warn(
                            "missing acknowledgment from {}, retrying message {}",
                            executionContext.getTargetPartnerId(),
                            message.getId());
                }
            } else {
                messageEntity.setNextRequeue(null);
                messageEntity.setFinished(ZonedDateTime.now());
                // we check the action in the entity, because the action might have been changed during the sending
                // pipeline to an "invalid" action just for the outbound payload
                if (messageEntity.getAction().isValidTerminationAction()) {
                    conversationEntity.setStatus(ConversationStatus.SUCCESS);
                    conversationEntity.setFinished(messageEntity.getFinished());
                } else {
                    conversationEntity.setStatus(ConversationStatus.IDLE);
                }
                LOGGER.trace("target connection is not reliable, set status to: {}", messageEntity.getStatus());
            }
        }

        message.setMessageStatus(messageEntity.getStatus());

        conversationRepository.save(conversationEntity);
        messageEntity = messageRepository.save(messageEntity);

        contextMessageMapper.updateMessage(messageEntity, message);
    }

    @Override
    public MessageEventType type() {
        return MessageEventType.SENT;
    }
}
