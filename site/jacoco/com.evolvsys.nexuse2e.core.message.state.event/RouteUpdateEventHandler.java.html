package com.evolvsys.nexuse2e.core.message.state.event;

import com.evolvsys.nexuse2e.core.exception.NexusException;
import com.evolvsys.nexuse2e.core.message.context.ExecutionContext;
import com.evolvsys.nexuse2e.core.message.context.RouteKey;
import com.evolvsys.nexuse2e.core.message.state.MessageEventType;
import com.evolvsys.nexuse2e.core.route.service.ContextRouteFactory;
import lombok.RequiredArgsConstructor;
import org.springframework.stereotype.Component;

/**
 * @author SvenBarden
 * @since 03.07.2024
 */
@RequiredArgsConstructor
@Component
class RouteUpdateEventHandler implements MessageEventHandler {
    private final ContextRouteFactory contextRouteFactory;

    @Override
    public void handle(ExecutionContext executionContext) throws NexusException {
        RouteKey routeKey = new RouteKey(
                executionContext.getChoreography().getId(),
                executionContext.getAction().getId(),
                executionContext.getSourcePartnerId(),
                executionContext.getTargetPartnerId());
        EventUtils.setRoute(routeKey, contextRouteFactory, executionContext);
    }

    @Override
    public MessageEventType type() {
        return MessageEventType.POST_ROUTE_RESOLVING;
    }
}
