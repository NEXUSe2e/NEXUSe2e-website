package com.evolvsys.nexuse2e.core.message.state.event;

import com.evolvsys.nexuse2e.core.exception.NexusError;
import com.evolvsys.nexuse2e.core.exception.NexusException;
import com.evolvsys.nexuse2e.core.message.context.ExecutionContext;
import com.evolvsys.nexuse2e.core.message.context.RouteKey;
import com.evolvsys.nexuse2e.core.model.entity.MessageEntity;
import com.evolvsys.nexuse2e.core.partner.model.ContextConnection;
import com.evolvsys.nexuse2e.core.route.model.ContextRoute;
import com.evolvsys.nexuse2e.core.route.service.ContextRouteFactory;
import java.time.Duration;
import java.time.ZonedDateTime;
import java.util.List;
import lombok.experimental.UtilityClass;
import lombok.extern.slf4j.Slf4j;
import org.apache.commons.collections4.CollectionUtils;

/**
 * @author SvenBarden
 * @since 02.04.2024
 */
@UtilityClass
@Slf4j
class EventUtils {

    public static void updateRetries(
            MessageEntity messageEntity, ContextConnection targetConnection, Duration defaultRetryInterval) {
        int nextRetryIndex;
        if (messageEntity.getLastTry() != null) { // increment retry count if not the first try
            messageEntity.setRetries(messageEntity.getRetries() + 1);
            nextRetryIndex = messageEntity.getRetries();
            LOGGER.trace("set retries for message to {}", messageEntity.getRetries());
        } else {
            nextRetryIndex = 0;
        }

        if (nextRetryIndex == 0 || messageEntity.getRetries() < messageEntity.getMaxRetries()) {
            List<Duration> requeueIntervals = targetConnection.getRequeueIntervals();

            if (CollectionUtils.isEmpty(requeueIntervals)) {
                messageEntity.setNextRequeue(ZonedDateTime.now().plus(defaultRetryInterval));
            } else if (requeueIntervals.size() <= nextRetryIndex) {
                messageEntity.setNextRequeue(ZonedDateTime.now().plus(requeueIntervals.getLast()));
            } else {
                messageEntity.setNextRequeue(ZonedDateTime.now().plus(requeueIntervals.get(nextRetryIndex)));
            }
        }
    }

    public static void setRoute(
            RouteKey routeKey, ContextRouteFactory contextRouteFactory, ExecutionContext executionContext)
            throws NexusException {
        ContextRoute contextRoute = contextRouteFactory
                .getRoute(routeKey)
                .orElseThrow(() -> new NexusException(NexusError.builder()
                        .message("could not find route for message")
                        .extension("message", executionContext.getMessage().getId())
                        .extension("sourcePartner", executionContext.getSourcePartnerId())
                        .extension("targetPartner", executionContext.getTargetPartnerId())
                        .extension("choreography", executionContext.getChoreographyId())
                        .extension("action", executionContext.getActionId())
                        .build()));
        executionContext.setRoute(contextRoute);
    }
}
