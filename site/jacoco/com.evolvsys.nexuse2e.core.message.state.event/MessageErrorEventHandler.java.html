package com.evolvsys.nexuse2e.core.message.state.event;

import com.evolvsys.nexuse2e.core.config.NEXUSe2eConfigurationProperties;
import com.evolvsys.nexuse2e.core.message.MessageStatus;
import com.evolvsys.nexuse2e.core.message.mapper.ContextMessageMapper;
import com.evolvsys.nexuse2e.core.message.repository.ConversationRepository;
import com.evolvsys.nexuse2e.core.message.repository.MessageRepository;
import com.evolvsys.nexuse2e.core.message.state.MessageEventType;
import lombok.extern.slf4j.Slf4j;
import org.springframework.stereotype.Component;

/**
 * @author SvenBarden
 * @since 05.02.2024
 */
@Component
@Slf4j
class MessageErrorEventHandler extends BaseMessageErrorEventHandler {

    public MessageErrorEventHandler(
            MessageRepository messageRepository,
            ConversationRepository conversationRepository,
            ContextMessageMapper contextMessageMapper,
            NEXUSe2eConfigurationProperties configurationProperties) {
        super(messageRepository, conversationRepository, contextMessageMapper, configurationProperties);
    }

    @Override
    public MessageEventType type() {
        return MessageEventType.ERROR;
    }

    @Override
    protected MessageStatus errorStatusToSet() {
        return MessageStatus.FAILED;
    }

    @Override
    protected MessageStatus requeueStatusToSet() {
        return MessageStatus.QUEUE_ACTION;
    }
}
