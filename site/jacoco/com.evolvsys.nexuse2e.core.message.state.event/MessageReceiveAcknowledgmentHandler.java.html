package com.evolvsys.nexuse2e.core.message.state.event;

import com.evolvsys.nexuse2e.core.exception.NexusError;
import com.evolvsys.nexuse2e.core.exception.NexusException;
import com.evolvsys.nexuse2e.core.message.MessageStatus;
import com.evolvsys.nexuse2e.core.message.context.ExecutionContext;
import com.evolvsys.nexuse2e.core.message.repository.ConversationRepository;
import com.evolvsys.nexuse2e.core.message.repository.MessageRepository;
import com.evolvsys.nexuse2e.core.message.state.MessageEventType;
import com.evolvsys.nexuse2e.core.model.entity.ConversationEntity;
import com.evolvsys.nexuse2e.core.model.entity.ConversationStatus;
import com.evolvsys.nexuse2e.core.model.entity.MessageEntity;
import com.evolvsys.nexuse2e.internal.Message;
import java.time.ZonedDateTime;
import java.util.EnumSet;
import java.util.Set;
import lombok.RequiredArgsConstructor;
import lombok.extern.slf4j.Slf4j;
import org.apache.commons.lang3.StringUtils;
import org.springframework.stereotype.Component;
import org.springframework.transaction.annotation.Transactional;

/**
 * @author SvenBarden
 * @since 29.01.2024
 */
@Component
@RequiredArgsConstructor
@Slf4j
class MessageReceiveAcknowledgmentHandler implements MessageEventHandler {
    private static final Set<MessageStatus> VALID_MESSAGE_STATUS =
            EnumSet.of(MessageStatus.FAILED_MISSING_ACKNOWLEDGMENT, MessageStatus.WAITING_FOR_ACKNOWLEDGMENT);
    private final MessageRepository messageRepository;
    private final ConversationRepository conversationRepository;

    private MessageEntity createEntity(Message message) {
        LOGGER.trace("create new messageEntity with id {}", message.getId());
        MessageEntity messageEntity = new MessageEntity();
        messageEntity.setMessageId(message.getId());
        messageEntity.setType(message.getType());
        return messageEntity;
    }

    private boolean updateReference(MessageEntity messageEntity, MessageEntity acknowledgmentEntity) {
        boolean updateConversation = false;
        if (VALID_MESSAGE_STATUS.contains(messageEntity.getStatus())) {
            LOGGER.debug("update message to status transferred {}", messageEntity.getMessageId());
            messageEntity.setStatus(MessageStatus.TRANSFERRED);
            messageEntity.setNextRequeue(null);
            messageEntity.setFinished(ZonedDateTime.now());
            messageEntity = messageRepository.save(messageEntity);
            updateConversation = true;
        }
        acknowledgmentEntity.setReferenceMessage(messageEntity);
        return updateConversation;
    }

    @Override
    @Transactional(rollbackFor = NexusException.class)
    public void handle(ExecutionContext executionContext) throws NexusException {
        Message message = executionContext.getMessage();

        MessageEntity acknowledgmentEntity =
                messageRepository.findByMessageId(message.getId()).orElseGet(() -> createEntity(message));
        boolean updateConversation = false;
        if (StringUtils.isNotBlank(message.getReferenceMessageId())) {
            LOGGER.trace("setting referenceId to {}", message.getReferenceMessageId());
            MessageEntity referenceMessageEntity = messageRepository
                    .findByMessageId(message.getReferenceMessageId())
                    .orElseThrow(() -> new NexusException(NexusError.builder().build()));
            updateConversation = updateReference(referenceMessageEntity, acknowledgmentEntity);
        }

        acknowledgmentEntity.setStatus(MessageStatus.RECEIVED);
        acknowledgmentEntity.setFinished(ZonedDateTime.now());

        if (updateConversation) {
            ConversationEntity conversation = acknowledgmentEntity.getConversation();
            if (executionContext.getAction().isValidTerminationAction()) {
                conversation.setStatus(ConversationStatus.SUCCESS);
                conversation.setFinished(ZonedDateTime.now());
            } else {
                conversation.setStatus(ConversationStatus.IDLE);
            }

            conversationRepository.save(conversation);
        }

        messageRepository.save(acknowledgmentEntity);
    }

    @Override
    public MessageEventType type() {
        return MessageEventType.RECEIVE_ACKNOWLEDGMENT;
    }
}
