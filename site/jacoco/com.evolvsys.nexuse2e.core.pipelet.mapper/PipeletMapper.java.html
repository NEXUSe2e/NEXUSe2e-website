package com.evolvsys.nexuse2e.core.pipelet.mapper;

import com.evolvsys.app.core.mapper.BaseDtoEntityMapper;
import com.evolvsys.app.core.mapper.DefaultMapperConfiguration;
import com.evolvsys.app.core.mapper.interceptor.InterceptingMappingContext;
import com.evolvsys.nexuse2e.core.model.entity.PipeletEntity;
import com.evolvsys.nexuse2e.core.parameter.mapper.ParameterMapper;
import com.evolvsys.nexuse2e.core.pipelet.PipeletRepository;
import com.evolvsys.nexuse2e.core.pipelet.model.PipeletDto;
import org.mapstruct.*;
import org.springframework.beans.factory.annotation.Autowired;

/**
 * <p>Mapper f√ºr {@link PipeletEntity} und {@link PipeletDto}.</p>
 *
 * @author Florian Drees
 * @since 28.11.2023
 * @see BaseDtoEntityMapper
 */
@Mapper(
        config = DefaultMapperConfiguration.class,
        uses = {ParameterMapper.class})
public abstract class PipeletMapper extends BaseDtoEntityMapper<PipeletDto, PipeletEntity> {
    private PipeletRepository pipeletRepository;

    @Autowired
    protected void config(PipeletRepository pipeletRepository) {
        this.pipeletRepository = pipeletRepository;
    }

    @Override
    public PipeletEntity idToEntity(Long id, @Context InterceptingMappingContext interceptingMappingContext) {
        return idToEntity(id, interceptingMappingContext, pipeletRepository::findById);
    }

    @Mapping(target = "parameters", ignore = true)
    @InheritConfiguration
    @Override
    protected abstract PipeletDto entityToDto(
            PipeletEntity entity, @Context InterceptingMappingContext interceptingMappingContext);

    @InheritConfiguration
    @Override
    protected abstract void updateEntity(
            PipeletDto dto,
            @MappingTarget PipeletEntity entity,
            @Context InterceptingMappingContext interceptingMappingContext);
}
