package com.evolvsys.nexuse2e.core.message.metadata;

import com.evolvsys.nexuse2e.core.exception.NexusError;
import com.evolvsys.nexuse2e.core.exception.NexusRuntimeException;
import com.evolvsys.nexuse2e.core.model.MetaDataType;
import com.evolvsys.nexuse2e.core.model.entity.MetaDataEntity;
import com.fasterxml.jackson.core.JsonProcessingException;
import com.fasterxml.jackson.databind.JsonNode;
import com.fasterxml.jackson.databind.ObjectMapper;
import edu.umd.cs.findbugs.annotations.SuppressFBWarnings;
import lombok.Builder;
import lombok.RequiredArgsConstructor;
import lombok.Value;
import lombok.extern.jackson.Jacksonized;
import org.springframework.core.annotation.Order;
import org.springframework.stereotype.Component;

/**
 * @author SvenBarden
 * @since 31.01.2024
 */
@Component
@RequiredArgsConstructor
@Order
class JsonMetaDataConverter implements MetaDataConverter {
    private final ObjectMapper objectMapper;

    @Override
    @SuppressFBWarnings("EXS_EXCEPTION_SOFTENING_NO_CHECKED")
    public void updateEntity(Object value, MetaDataEntity metaDataEntity) {
        try {
            Container container = Container.builder()
                    .value(value)
                    .type(value.getClass().getName())
                    .build();
            metaDataEntity.setStringValue(objectMapper.writeValueAsString(container));
        } catch (JsonProcessingException e) {
            throw new NexusRuntimeException(
                    NexusError.builder()
                            .message("could not generate string value for meta data")
                            .extension("type", value.getClass().getName())
                            .build(),
                    e);
        }
    }

    @Override
    @SuppressFBWarnings("EXS_EXCEPTION_SOFTENING_NO_CHECKED")
    public Object getValue(MetaDataEntity metaDataEntity) {
        Object value;
        try {
            JsonNode jsonNode = objectMapper.readTree(metaDataEntity.getStringValue());

            JsonNode valueNode = jsonNode.get("value");
            String type = jsonNode.get("type").asText();

            Class<?> clazz = Class.forName(type);

            value = objectMapper.convertValue(valueNode, clazz);
        } catch (JsonProcessingException e) {
            throw new NexusRuntimeException(
                    NexusError.builder()
                            .message("could not parse json for metaData")
                            .extension("name", metaDataEntity.getName())
                            .extension("json", metaDataEntity.getStringValue())
                            .build(),
                    e);
        } catch (ClassNotFoundException e) {
            throw new NexusRuntimeException(
                    NexusError.builder()
                            .message("unknown class in meta data json")
                            .extension("name", metaDataEntity.getName())
                            .extension("json", metaDataEntity.getStringValue())
                            .build(),
                    e);
        }
        return value;
    }

    @Override
    public boolean supports(Class<?> type) {
        return true;
    }

    @Override
    public MetaDataType type() {
        return MetaDataType.JSON;
    }

    @Jacksonized
    @Value
    @Builder
    public static class Container {
        String type;
        Object value;
    }
}
