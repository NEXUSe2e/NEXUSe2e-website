package com.evolvsys.nexuse2e.core.message.metadata;

import com.evolvsys.nexuse2e.core.model.MetaDataType;
import com.evolvsys.nexuse2e.core.model.entity.MetaDataEntity;
import org.springframework.stereotype.Component;

/**
 * @author SvenBarden
 * @since 30.01.2024
 */
@Component
class IntegerMetaDataConverter extends TypeSafeMetaDataConverter<Integer> {
    @Override
    protected void updateEntityInternal(Integer value, MetaDataEntity metaDataEntity) {
        metaDataEntity.setLongValue(value == null ? null : value.longValue());
    }

    @Override
    protected Integer getValueInternal(MetaDataEntity metaDataEntity) {
        return metaDataEntity.getLongValue() == null
                ? null
                : metaDataEntity.getLongValue().intValue();
    }

    @Override
    public MetaDataType type() {
        return MetaDataType.INTEGER;
    }
}
