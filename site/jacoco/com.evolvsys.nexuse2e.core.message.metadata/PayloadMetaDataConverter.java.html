package com.evolvsys.nexuse2e.core.message.metadata;

import com.evolvsys.nexuse2e.core.message.payload.content.PayloadContentFactory;
import com.evolvsys.nexuse2e.core.message.payload.model.BinaryEntity;
import com.evolvsys.nexuse2e.core.message.payload.model.PayloadEntity;
import com.evolvsys.nexuse2e.core.model.MetaDataType;
import com.evolvsys.nexuse2e.core.model.entity.MetaDataEntity;
import com.evolvsys.nexuse2e.internal.ContextPayload;
import edu.umd.cs.findbugs.annotations.SuppressFBWarnings;
import lombok.RequiredArgsConstructor;
import org.hibernate.engine.jdbc.BlobProxy;
import org.springframework.stereotype.Component;

/**
 * @author SvenBarden
 * @since 31.01.2024
 */
@Component
@RequiredArgsConstructor
class PayloadMetaDataConverter extends TypeSafeMetaDataConverter<ContextPayload> {
    private final PayloadContentFactory payloadContentFactory;

    @Override
    protected void updateEntityInternal(ContextPayload value, MetaDataEntity metaDataEntity) {
        PayloadEntity payloadEntity = metaDataEntity.getBinaryValue();

        if (payloadEntity == null) {
            metaDataEntity.setBinaryValue(payloadEntity = new PayloadEntity());
        }

        BinaryEntity binaryEntity = payloadEntity.getContent();
        if (binaryEntity == null) {
            payloadEntity.setContent(binaryEntity = new BinaryEntity());
        }

        binaryEntity.setContent(BlobProxy.generateProxy(value.getContent().getInputStream(), value.getSize()));

        payloadEntity.setSize(value.getSize());
        payloadEntity.setContentType(value.getContentType());
        payloadEntity.setEncoding(value.getEncoding());
        payloadEntity.setContent(binaryEntity);
    }

    @Override
    @SuppressFBWarnings("EXS_EXCEPTION_SOFTENING_NO_CONSTRAINTS")
    protected ContextPayload getValueInternal(MetaDataEntity metaDataEntity) {
        ContextPayload payload = new ContextPayload();

        PayloadEntity payloadEntity = metaDataEntity.getBinaryValue();

        payload.setContentType(payloadEntity.getContentType());
        payload.setSize(payloadEntity.getSize());
        payload.setEncoding(payloadEntity.getEncoding());
        payload.setContent(payloadContentFactory.createFromEntity(payloadEntity));

        return payload;
    }

    @Override
    public MetaDataType type() {
        return MetaDataType.PAYLOAD;
    }
}
