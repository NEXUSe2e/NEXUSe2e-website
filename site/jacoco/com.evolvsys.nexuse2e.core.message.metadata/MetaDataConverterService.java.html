package com.evolvsys.nexuse2e.core.message.metadata;

import com.evolvsys.nexuse2e.core.exception.NexusError;
import com.evolvsys.nexuse2e.core.exception.NexusRuntimeException;
import com.evolvsys.nexuse2e.core.model.MetaDataType;
import com.evolvsys.nexuse2e.core.model.entity.MetaDataEntity;
import java.util.List;
import java.util.Optional;
import lombok.RequiredArgsConstructor;
import lombok.extern.slf4j.Slf4j;
import org.apache.commons.lang3.Validate;
import org.springframework.stereotype.Service;

/**
 * @author SvenBarden
 * @since 31.01.2024
 */
@Service
@RequiredArgsConstructor
@Slf4j
public class MetaDataConverterService {
    private final List<MetaDataConverter> metaDataConverters;

    private Optional<MetaDataConverter> getConverter(MetaDataType metaDataType) {
        return metaDataConverters.stream()
                .filter(metaDataConverter -> metaDataConverter.type() == metaDataType)
                .findFirst();
    }

    private Optional<MetaDataConverter> getConverter(Object value) {
        return metaDataConverters.stream()
                .filter(metaDataConverter -> metaDataConverter.supports(value.getClass()))
                .findFirst();
    }

    private void updateNullValue(MetaDataEntity metaDataEntity) {
        metaDataEntity.setLongValue(null);
        metaDataEntity.setStringValue(null);
        metaDataEntity.setDateTimeValue(null);
        metaDataEntity.setBinaryValue(null);
        metaDataEntity.setBooleanValue(null);
        metaDataEntity.setBigDecimalValue(null);
    }

    public void updateEntity(Object value, MetaDataEntity metaDataEntity) {
        Validate.notNull(value, "value must not be null");
        Validate.notNull(metaDataEntity, "metaDataEntity must not be null");

        updateNullValue(metaDataEntity);
        MetaDataConverter metaDataConverter = getConverter(value)
                .orElseThrow(() -> new NexusRuntimeException(NexusError.builder()
                        .message("no metaDataConverter for class found")
                        .extension("class", value.getClass().getName())
                        .build()));
        LOGGER.trace("converting {} with {}", value.getClass(), metaDataConverter);

        metaDataConverter.updateEntity(value, metaDataEntity);
        metaDataEntity.setType(metaDataConverter.type());
    }

    public Object getValue(MetaDataEntity metaDataEntity) {
        return getConverter(metaDataEntity.getType())
                .map(metaDataConverter -> metaDataConverter.getValue(metaDataEntity))
                .orElseThrow(() -> new NexusRuntimeException(NexusError.builder()
                        .message("no metaDataConverter found for type")
                        .extension("type", metaDataEntity.getType().name())
                        .build()));
    }
}
