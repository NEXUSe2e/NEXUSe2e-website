package com.evolvsys.nexuse2e.core.message.metadata;

import com.evolvsys.nexuse2e.core.model.MetaDataType;
import com.evolvsys.nexuse2e.core.model.entity.MetaDataEntity;
import java.math.BigDecimal;
import org.springframework.stereotype.Component;

/**
 * @author SvenBarden
 * @since 18.04.2024
 */
@Component
class DoubleMetaDataConverter implements MetaDataConverter {

    @Override
    public void updateEntity(Object value, MetaDataEntity metaDataEntity) {
        if (value instanceof Double doubleValue) {
            metaDataEntity.setBigDecimalValue(BigDecimal.valueOf(doubleValue));
        } else {
            metaDataEntity.setBigDecimalValue(null);
        }
    }

    @Override
    public Object getValue(MetaDataEntity metaDataEntity) {
        return metaDataEntity.getBigDecimalValue();
    }

    @Override
    public boolean supports(Class<?> type) {
        return Double.class.equals(type) || double.class.equals(type);
    }

    @Override
    public MetaDataType type() {
        return MetaDataType.DOUBLE;
    }
}
