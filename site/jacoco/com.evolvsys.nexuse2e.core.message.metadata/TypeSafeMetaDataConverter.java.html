package com.evolvsys.nexuse2e.core.message.metadata;

import com.evolvsys.app.core.util.GenericUtils;
import com.evolvsys.nexuse2e.core.model.entity.MetaDataEntity;

/**
 * @author SvenBarden
 * @since 30.01.2024
 */
abstract class TypeSafeMetaDataConverter<T> implements MetaDataConverter {

    @SuppressWarnings("unchecked")
    @Override
    public final void updateEntity(Object value, MetaDataEntity metaDataEntity) {
        updateEntityInternal((T) value, metaDataEntity);
    }

    @Override
    public final Object getValue(MetaDataEntity metaDataEntity) {
        return getValueInternal(metaDataEntity);
    }

    @Override
    public boolean supports(Class<?> type) {
        return getType().isAssignableFrom(type);
    }

    protected abstract void updateEntityInternal(T value, MetaDataEntity metaDataEntity);

    protected abstract T getValueInternal(MetaDataEntity metaDataEntity);

    protected Class<T> getType() {
        return GenericUtils.getGenericType(getClass(), TypeSafeMetaDataConverter.class, 0);
    }
}
