package com.evolvsys.nexuse2e.core.message.metadata;

import com.evolvsys.nexuse2e.core.model.MetaDataType;
import com.evolvsys.nexuse2e.core.model.entity.MetaDataEntity;
import java.time.ZonedDateTime;
import org.springframework.stereotype.Component;

/**
 * @author SvenBarden
 * @since 30.01.2024
 */
@Component
class ZonedDateTimeMetaDataConverter extends TypeSafeMetaDataConverter<ZonedDateTime> {
    @Override
    protected void updateEntityInternal(ZonedDateTime value, MetaDataEntity metaDataEntity) {
        metaDataEntity.setDateTimeValue(value);
    }

    @Override
    protected ZonedDateTime getValueInternal(MetaDataEntity metaDataEntity) {
        return metaDataEntity.getDateTimeValue();
    }

    @Override
    public MetaDataType type() {
        return MetaDataType.DATETIME;
    }
}
