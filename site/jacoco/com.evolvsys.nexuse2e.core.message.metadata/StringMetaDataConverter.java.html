package com.evolvsys.nexuse2e.core.message.metadata;

import com.evolvsys.nexuse2e.core.model.MetaDataType;
import com.evolvsys.nexuse2e.core.model.entity.MetaDataEntity;
import org.springframework.stereotype.Component;

/**
 * @author SvenBarden
 * @since 30.01.2024
 */
@Component
class StringMetaDataConverter extends TypeSafeMetaDataConverter<String> {
    @Override
    protected void updateEntityInternal(String value, MetaDataEntity metaDataEntity) {
        metaDataEntity.setStringValue(value);
    }

    @Override
    protected String getValueInternal(MetaDataEntity metaDataEntity) {
        return metaDataEntity.getStringValue();
    }

    @Override
    public MetaDataType type() {
        return MetaDataType.STRING;
    }
}
