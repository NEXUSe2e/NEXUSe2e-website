package com.evolvsys.nexuse2e.core.action.mapper;

import com.evolvsys.app.core.mapper.DefaultMapperConfiguration;
import com.evolvsys.nexuse2e.core.action.Action;
import com.evolvsys.nexuse2e.core.action.model.ActionEntity;
import com.evolvsys.nexuse2e.core.action.repository.ActionRepository;
import com.evolvsys.nexuse2e.core.pipeline.PipelineFactory;
import com.evolvsys.nexuse2e.core.pipeline.model.Pipeline;
import com.evolvsys.nexuse2e.core.pipeline.model.PipelineEntity;
import org.mapstruct.Mapper;
import org.mapstruct.Mapping;
import org.springframework.beans.factory.annotation.Autowired;

/**
 * @author SvenBarden
 * @since 24.01.2024
 */
@Mapper(config = DefaultMapperConfiguration.class)
public abstract class ContextActionMapper {
    private PipelineFactory pipelineFactory;
    private ActionRepository actionRepository;

    @Autowired
    protected void config(PipelineFactory pipelineFactory, ActionRepository actionRepository) {
        this.pipelineFactory = pipelineFactory;
        this.actionRepository = actionRepository;
    }

    protected Pipeline stringToPipeline(String pipelineId) {
        return pipelineId == null ? null : pipelineFactory.get(pipelineId);
    }

    protected Pipeline pipelineToContext(PipelineEntity pipelineEntity) {
        return pipelineEntity == null ? null : stringToPipeline(pipelineEntity.getPipelineId());
    }

    protected String actionToName(ActionEntity actionEntity) {
        return actionEntity == null ? null : actionEntity.getActionId();
    }

    public ActionEntity actionToEntity(Action action) {
        return namesToEntity(action.getChoreographyId(), action.getActionId());
    }

    public ActionEntity namesToEntity(String choreographyId, String actionId) {
        return actionRepository.findByActionIdAndChoreography_ChoreographyId(actionId, choreographyId).stream()
                .filter(actionEntity ->
                        actionEntity.getChoreography().getChoreographyId().equals(choreographyId)
                                && actionEntity.getActionId().equals(actionId))
                .findFirst()
                .orElse(null);
    }

    @Mapping(target = "validNextAction", ignore = true)
    @Mapping(target = "choreographyId", source = "choreography.choreographyId")
    @Mapping(target = "choreographyRefId", source = "choreography.id")
    public abstract Action entityToAction(ActionEntity actionEntity);
}
