package com.evolvsys.nexuse2e.core.action.mapper;

import com.evolvsys.app.core.mapper.BaseDtoEntityMapper;
import com.evolvsys.app.core.mapper.DefaultMapperConfiguration;
import com.evolvsys.app.core.mapper.interceptor.InterceptingMappingContext;
import com.evolvsys.nexuse2e.core.action.model.ActionDto;
import com.evolvsys.nexuse2e.core.action.model.ActionEntity;
import com.evolvsys.nexuse2e.core.action.repository.ActionRepository;
import com.evolvsys.nexuse2e.core.choreography.mapper.BaseChoreographyMapper;
import com.evolvsys.nexuse2e.core.pipeline.mapper.PipelineMapper;
import com.evolvsys.nexuse2e.core.route.mapper.RouteMapper;
import java.util.Collection;
import java.util.Map;
import java.util.Objects;
import java.util.function.Function;
import java.util.stream.Collectors;
import org.apache.commons.collections4.CollectionUtils;
import org.mapstruct.*;
import org.springframework.beans.factory.annotation.Autowired;

/**
 * <p>Mapper f√ºr {@link ActionEntity} und {@link ActionDto}.</p>
 *
 * @author Florian Drees
 * @since 19.12.2023
 * @see BaseDtoEntityMapper
 */
@Mapper(
        config = DefaultMapperConfiguration.class,
        uses = {BaseChoreographyMapper.class, PipelineMapper.class, RouteMapper.class})
public abstract class ActionMapper extends BaseDtoEntityMapper<ActionDto, ActionEntity> {
    private ActionRepository actionRepository;

    @Autowired
    protected void config(ActionRepository actionRepository) {
        this.actionRepository = actionRepository;
    }

    @Override
    public ActionEntity idToEntity(Long id, @Context InterceptingMappingContext interceptingMappingContext) {
        return idToEntity(id, interceptingMappingContext, actionRepository::findById);
    }

    @Mapping(target = "returnPipelineId", source = "returnPipeline.id")
    @Mapping(target = "forwardPipelineId", source = "forwardPipeline.id")
    @Mapping(target = "acknowledgmentPipelineId", source = "acknowledgmentPipeline.id")
    @Mapping(target = "choreographyId", source = "choreography.id")
    @InheritConfiguration
    @Override
    protected abstract ActionDto entityToDto(
            ActionEntity entity, @Context InterceptingMappingContext interceptingMappingContext);

    @Mapping(target = "routes", ignore = true)
    @Mapping(target = "returnPipeline", source = "returnPipelineId")
    @Mapping(target = "forwardPipeline", source = "forwardPipelineId")
    @Mapping(target = "acknowledgmentPipeline", source = "acknowledgmentPipelineId")
    @Mapping(target = "validNextActions", ignore = true)
    @Mapping(target = "choreography", source = "choreographyId")
    @InheritConfiguration
    @Override
    protected abstract void updateEntity(
            ActionDto dto,
            @MappingTarget ActionEntity entity,
            @Context InterceptingMappingContext interceptingMappingContext);

    public void updateValidNextActions(
            Collection<ActionDto> actionDtos, ActionEntity actionEntity, Collection<ActionEntity> choreographyActions) {

        actionEntity.getValidNextActions().clear();
        if (CollectionUtils.isNotEmpty(actionDtos)) {
            Map<String, ActionEntity> actionEntityMap = choreographyActions.stream()
                    .collect(Collectors.toMap(ActionEntity::getActionId, Function.identity()));

            actionDtos.stream()
                    .map(actionDto -> actionEntityMap.get(actionDto.getActionId()))
                    .filter(Objects::nonNull)
                    .forEach(nextAction -> actionEntity.getValidNextActions().add(nextAction));
        }
    }
}
