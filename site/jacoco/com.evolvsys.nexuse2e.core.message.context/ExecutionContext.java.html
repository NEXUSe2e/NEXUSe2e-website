package com.evolvsys.nexuse2e.core.message.context;

import com.evolvsys.nexuse2e.core.exception.NexusError;
import com.evolvsys.nexuse2e.core.exception.NexusRuntimeException;
import com.evolvsys.nexuse2e.internal.ContextPayload;
import com.evolvsys.nexuse2e.internal.Message;
import com.evolvsys.nexuse2e.internal.MessageType;
import jakarta.validation.Valid;
import java.util.Collection;
import java.util.List;
import java.util.Optional;
import lombok.*;
import lombok.experimental.Delegate;

/**
 * @author SvenBarden
 * @since 20.10.2023
 */
@AllArgsConstructor(access = AccessLevel.PACKAGE)
@Getter
@EqualsAndHashCode
@ToString
public class ExecutionContext {

    private static final String INVALID_MESSAGE_TYPE_IN_EXECUTION_CONTEXT = "Invalid message type in execution context";
    private static final String TYPE = "type";

    @Delegate
    @Valid
    private final MessageContext messageContext;

    private final MessageType messageType;

    public ContextPayload getData() {
        return switch (getMessageType()) {
            case NORMAL -> getSourceData().stream().findFirst().orElse(null);
            case RETURN -> getReturnData();
            case ERROR -> getErrorData().stream().findFirst().orElse(null);
            case ACKNOWLEDGMENT -> getAcknowledgmentData().stream().findFirst().orElse(null);
            default ->
                throw new NexusRuntimeException(NexusError.builder()
                        .message(INVALID_MESSAGE_TYPE_IN_EXECUTION_CONTEXT)
                        .extension(TYPE, getMessageType())
                        .build());
        };
    }

    public void setData(ContextPayload data) {
        switch (getMessageType()) {
            case NORMAL -> setSourceData(data);
            case RETURN -> setReturnData(data);
            case ERROR -> setErrorData(data);
            case ACKNOWLEDGMENT -> setAcknowledgmentData(data);
            default ->
                throw new NexusRuntimeException(NexusError.builder()
                        .message(INVALID_MESSAGE_TYPE_IN_EXECUTION_CONTEXT)
                        .extension(TYPE, getMessageType())
                        .build());
        }
    }

    public void setData(Collection<ContextPayload> contextPayloads) {
        switch (getMessageType()) {
            case NORMAL -> {
                getSourceData().clear();
                getSourceData().addAll(contextPayloads);
            }
            case ACKNOWLEDGMENT -> {
                getAcknowledgmentData().clear();
                getAcknowledgmentData().addAll(contextPayloads);
            }
            case ERROR -> {
                getErrorData().clear();
                getErrorData().addAll(contextPayloads);
            }
            case RETURN -> throw new IllegalStateException("collections not allowed for return type");
            default ->
                throw new NexusRuntimeException(NexusError.builder()
                        .message(INVALID_MESSAGE_TYPE_IN_EXECUTION_CONTEXT)
                        .extension(TYPE, getMessageType())
                        .build());
        }
    }

    public void addData(ContextPayload contextPayload) {
        switch (getMessageType()) {
            case NORMAL -> getSourceData().add(contextPayload);
            case ACKNOWLEDGMENT -> getAcknowledgmentData().add(contextPayload);
            case ERROR -> getErrorData().add(contextPayload);
            case RETURN -> setData(contextPayload);
            default ->
                throw new NexusRuntimeException(NexusError.builder()
                        .message(INVALID_MESSAGE_TYPE_IN_EXECUTION_CONTEXT)
                        .extension(TYPE, getMessageType())
                        .build());
        }
    }

    public Message getMessage() {
        return switch (getMessageType()) {
            case NORMAL -> getSourceMessage();
            case RETURN -> getReturnMessage();
            case ERROR -> getErrorMessage();
            case ACKNOWLEDGMENT -> getAcknowledgmentMessage();
            default ->
                throw new NexusRuntimeException(NexusError.builder()
                        .message(INVALID_MESSAGE_TYPE_IN_EXECUTION_CONTEXT)
                        .extension(TYPE, getMessageType())
                        .build());
        };
    }

    public void setMessage(Message message) {
        switch (getMessageType()) {
            case NORMAL -> setSourceMessage(message);
            case RETURN -> setReturnMessage(message);
            case ERROR -> setErrorMessage(message);
            case ACKNOWLEDGMENT -> setAcknowledgmentMessage(message);
            default ->
                throw new NexusRuntimeException(NexusError.builder()
                        .message(INVALID_MESSAGE_TYPE_IN_EXECUTION_CONTEXT)
                        .extension(TYPE, getMessageType())
                        .build());
        }
    }

    public List<ContextPayload> getDataAsList() {
        return switch (getMessageType()) {
            case NORMAL -> getSourceData();
            case RETURN -> Optional.of(getData()).stream().toList();
            case ACKNOWLEDGMENT -> getAcknowledgmentData();
            case ERROR -> getErrorData();
            default ->
                throw new NexusRuntimeException(NexusError.builder()
                        .message(INVALID_MESSAGE_TYPE_IN_EXECUTION_CONTEXT)
                        .extension(TYPE, getMessageType())
                        .build());
        };
    }

    public void clearData() {
        switch (getMessageType()) {
            case NORMAL -> getSourceData().clear();
            case RETURN -> setData((ContextPayload) null);
            case ACKNOWLEDGMENT -> getAcknowledgmentData().clear();
            case ERROR -> getErrorData().clear();
            default ->
                throw new NexusRuntimeException(NexusError.builder()
                        .message(INVALID_MESSAGE_TYPE_IN_EXECUTION_CONTEXT)
                        .extension(TYPE, getMessageType())
                        .build());
        }
    }
}
