package com.evolvsys.nexuse2e.core.message.context;

import java.util.Objects;
import java.util.function.Function;
import lombok.RequiredArgsConstructor;

/**
 * @author SvenBarden
 * @since 24.01.2024
 */
@RequiredArgsConstructor
final class LazyObject<K, T> {
    private final Function<K, T> get;
    private K key;
    private T value;

    public T getValue(K key) {
        if (value == null || (Objects.equals(this.key, key) == false)) {
            if (key == null) {
                value = null;
            } else {
                value = get.apply(key);
            }
            this.key = key;
        }
        return value;
    }
}
