package com.evolvsys.nexuse2e.core.message.context;

import com.evolvsys.nexuse2e.core.exception.NexusError;
import com.evolvsys.nexuse2e.core.exception.NexusRuntimeException;
import com.evolvsys.nexuse2e.core.message.mapper.ContextMessageMapper;
import com.evolvsys.nexuse2e.core.message.repository.MessageRepository;
import com.evolvsys.nexuse2e.core.model.entity.MessageEntity;
import com.evolvsys.nexuse2e.core.route.model.ContextRoute;
import com.evolvsys.nexuse2e.core.route.service.ContextRouteFactory;
import com.evolvsys.nexuse2e.internal.Message;
import com.evolvsys.nexuse2e.internal.MessageType;
import lombok.RequiredArgsConstructor;
import lombok.extern.slf4j.Slf4j;
import org.springframework.stereotype.Service;
import org.springframework.transaction.annotation.Transactional;

/**
 * @author SvenBarden
 * @since 02.02.2024
 */
@Service
@Slf4j
@RequiredArgsConstructor
class MessageContextLoaderService {
    private final MessageRepository messageRepository;
    private final ContextMessageMapper contextMessageMapper;
    private final ContextRouteFactory contextRouteFactory;

    private void update(MessageContext messageContext, MessageEntity messageEntity) {
        Message message = contextMessageMapper.entityToMessage(messageEntity);

        contextMessageMapper.updateContext(messageEntity, messageContext);

        if (message.getType() == MessageType.NORMAL) {
            LOGGER.trace("set source message {}", message);
            messageRepository
                    .findByReferenceMessageAndTypeAndOutboundAcknowledgment(
                            messageEntity, MessageType.ACKNOWLEDGMENT, true)
                    .ifPresent(ackEntity -> {
                        messageContext.setAcknowledgmentMessage(contextMessageMapper.entityToMessage(ackEntity));
                        LOGGER.trace("set acknowledgment {}", messageContext.getAcknowledgmentMessage());
                    });
            messageContext.setSourceMessage(message);
        } else if (message.getType() == MessageType.ACKNOWLEDGMENT) {
            LOGGER.trace("set acknowledgment {}", message);
            messageContext.setAcknowledgmentMessage(message);
            MessageEntity sourceMessageEntity = messageRepository
                    .findByMessageId(message.getReferenceMessageId())
                    .orElseThrow();

            Message sourceMessage = contextMessageMapper.entityToMessage(sourceMessageEntity);

            LOGGER.trace("set source message {}", sourceMessage);

            messageContext.setSourceMessage(sourceMessage);
        }

        LOGGER.debug("create context for {} and {}", messageContext.getChoreographyId(), messageContext.getActionId());

        ContextRoute contextRoute = contextRouteFactory
                .getRoute(new RouteKey(
                        messageContext.getChoreography().getId(),
                        messageContext.getAction().getId(),
                        messageContext.getSourcePartnerId(),
                        messageContext.getTargetPartnerId()))
                .orElseThrow(() -> new NexusRuntimeException(NexusError.builder()
                        .message("no route found for rebuilding message")
                        .extension(
                                "messageId", messageContext.getSourceMessage().getId())
                        .extension("sourcePartner", messageContext.getSourcePartnerId())
                        .extension("targetPartner", messageContext.getTargetPartnerId())
                        .extension("choreography", messageContext.getChoreographyId())
                        .extension("action", messageContext.getActionId())
                        .build()));

        messageContext.setRoute(contextRoute);
    }

    @Transactional(readOnly = true)
    public void update(MessageContext messageContext, String messageId) {
        messageRepository.findByMessageId(messageId).ifPresent(messageEntity -> update(messageContext, messageEntity));
    }

    public boolean exists(String messageId) {
        return messageRepository.existsByMessageId(messageId);
    }
}
