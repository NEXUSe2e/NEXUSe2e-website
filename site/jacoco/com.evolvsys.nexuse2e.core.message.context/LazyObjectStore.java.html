package com.evolvsys.nexuse2e.core.message.context;

import com.evolvsys.nexuse2e.core.action.Action;
import com.evolvsys.nexuse2e.core.choreography.model.ContextChoreography;
import com.evolvsys.nexuse2e.core.message.model.ContextConversation;
import com.evolvsys.nexuse2e.core.partner.model.ContextPartner;
import lombok.AccessLevel;
import lombok.Getter;
import lombok.RequiredArgsConstructor;

/**
 * @author SvenBarden
 * @since 24.01.2024
 */
@RequiredArgsConstructor
class LazyObjectStore {
    @Getter(AccessLevel.PRIVATE)
    private final ContextLazyObjectProvider contextLazyObjectProvider;

    private final LazyObject<ActionKey, Action> action =
            new LazyObject<>(actionKey -> getContextLazyObjectProvider().getAction(actionKey));
    private final LazyObject<String, ContextPartner> sourcePartner =
            new LazyObject<>(s -> getContextLazyObjectProvider().getPartner(s));
    private final LazyObject<String, ContextPartner> targetPartner =
            new LazyObject<>(s -> getContextLazyObjectProvider().getPartner(s));
    private final LazyObject<String, ContextConversation> conversation =
            new LazyObject<>(s -> getContextLazyObjectProvider().getConversation(s));
    private final LazyObject<String, ContextChoreography> choreography =
            new LazyObject<>(s -> getContextLazyObjectProvider().getChoreography(s));

    public Action getAction(MessageContext messageContext) {
        return action.getValue(new ActionKey(messageContext.getChoreographyId(), messageContext.getActionId()));
    }

    public ContextPartner getSourcePartner(MessageContext messageContext) {
        return sourcePartner.getValue(messageContext.getSourcePartnerId());
    }

    public ContextPartner getTargetPartner(MessageContext messageContext) {
        return targetPartner.getValue(messageContext.getTargetPartnerId());
    }

    public ContextChoreography getChoreography(MessageContext messageContext) {
        return choreography.getValue(messageContext.getChoreographyId());
    }
}
