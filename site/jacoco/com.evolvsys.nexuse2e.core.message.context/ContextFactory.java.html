package com.evolvsys.nexuse2e.core.message.context;

import com.evolvsys.nexuse2e.core.keyvalue.persistence.PersistentKeyValue;
import com.evolvsys.nexuse2e.core.keyvalue.persistence.PersistentKeyValueRepository;
import com.evolvsys.nexuse2e.core.logging.MessageLogContextHolder;
import com.evolvsys.nexuse2e.core.message.generator.ConversationIdGenerator;
import com.evolvsys.nexuse2e.core.message.generator.MessageIdGenerator;
import com.evolvsys.nexuse2e.internal.MessageType;
import lombok.RequiredArgsConstructor;
import lombok.extern.slf4j.Slf4j;
import org.springframework.beans.factory.BeanFactory;
import org.springframework.beans.factory.BeanFactoryAware;
import org.springframework.context.ApplicationContext;
import org.springframework.context.ApplicationContextAware;
import org.springframework.stereotype.Component;

/**
 * @author SvenBarden
 * @since 21.01.2024
 */
@Component
@RequiredArgsConstructor
@Slf4j
public class ContextFactory implements BeanFactoryAware, ApplicationContextAware {
    private final MessageIdGenerator messageIdGenerator;
    private final ConversationIdGenerator conversationIdGenerator;
    private final ContextLazyObjectProvider contextLazyObjectProvider;
    private final MessageContextLoaderService messageContextLoaderService;
    private final PersistentKeyValueRepository persistentKeyValueRepository;
    private BeanFactory beanFactory;
    private ApplicationContext applicationContext;

    public static ExecutionContext createExecutionContext(MessageContext messageContext, MessageType messageType) {
        ExecutionContext executionContext = new ExecutionContext(messageContext, messageType);
        MessageLogContextHolder.getContext().setExecutionContext(executionContext);
        return executionContext;
    }

    public MessageContext createNew() {
        LOGGER.trace("creating new message context");
        MessageContext messageContext = new MessageContext(
                beanFactory,
                applicationContext,
                messageIdGenerator,
                conversationIdGenerator,
                new PersistentKeyValue(persistentKeyValueRepository),
                new LazyObjectStore(contextLazyObjectProvider));
        messageContext.setNewContext(true);
        MessageLogContextHolder.getContext().setMessageContext(messageContext);
        return messageContext;
    }

    public MessageContext createForResend(String messageId) {
        MessageContext messageContext = createNew();
        messageContextLoaderService.update(messageContext, messageId);
        return messageContext;
    }

    public MessageContext createForInboundMessage(String messageId) {
        MessageContext messageContext = createNew();
        messageContext.setNewContext(false);
        messageContextLoaderService.update(messageContext, messageId);
        if (messageContext.getSourceMessage().getId().equals(messageId) == false
                && messageContext.getAcknowledgmentMessage() != null
                && messageContext.getAcknowledgmentMessage().getId().equals(messageId)) {
            // TODO workaround for inbound dispatcher, required refactoring
            messageContext.setSourceMessage(messageContext.getAcknowledgmentMessage());
        }

        return messageContext;
    }

    public boolean exists(String messageId) {
        return messageContextLoaderService.exists(messageId);
    }

    @Override
    public void setBeanFactory(BeanFactory beanFactory) {
        synchronized (this) {
            this.beanFactory = beanFactory;
        }
    }

    @Override
    public void setApplicationContext(ApplicationContext applicationContext) {
        synchronized (this) {
            this.applicationContext = applicationContext;
        }
    }
}
