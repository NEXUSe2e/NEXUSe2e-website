package com.evolvsys.nexuse2e.core.message.context;

import com.evolvsys.nexuse2e.core.action.Action;
import com.evolvsys.nexuse2e.core.choreography.model.ContextChoreography;
import com.evolvsys.nexuse2e.core.message.model.ContextConversation;
import com.evolvsys.nexuse2e.core.partner.model.ContextPartner;
import java.util.List;
import lombok.RequiredArgsConstructor;
import org.springframework.stereotype.Component;

/**
 * @author SvenBarden
 * @since 24.01.2024
 */
@Component
@RequiredArgsConstructor
public class ContextLazyObjectProvider {
    private final List<ContextObjectFactory<?, ?>> contextObjectFactories;

    @SuppressWarnings("unchecked")
    private <K, T> T create(Class<T> valueType, K key) {
        T result;
        if (key == null) {
            result = null;
        } else {
            ContextObjectFactory<K, T> factory = (ContextObjectFactory<K, T>) contextObjectFactories.stream()
                    .filter(contextObjectFactory ->
                            contextObjectFactory.getValueType().equals(valueType)
                                    && contextObjectFactory.getKeyType().equals(key.getClass()))
                    .findFirst()
                    .orElseThrow(() -> new IllegalArgumentException(
                            "could not find factory or value for " + valueType + " (" + key + ")"));
            result = factory.create(key);
        }
        return result;
    }

    public Action getAction(ActionKey actionKey) {
        return create(Action.class, actionKey);
    }

    public ContextPartner getPartner(String partnerId) {
        return create(ContextPartner.class, partnerId);
    }

    public ContextChoreography getChoreography(String choreographyId) {
        return create(ContextChoreography.class, choreographyId);
    }

    public ContextConversation getConversation(String conversationId) {
        return create(ContextConversation.class, conversationId);
    }
}
