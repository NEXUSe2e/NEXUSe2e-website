package com.evolvsys.nexuse2e.core.message.action;

import com.evolvsys.app.core.process.*;
import com.evolvsys.nexuse2e.core.exception.NexusError;
import com.evolvsys.nexuse2e.core.exception.NexusException;
import com.evolvsys.nexuse2e.core.exception.NexusRuntimeException;
import com.evolvsys.nexuse2e.core.manualaction.service.MessageActionManualActionService;
import com.evolvsys.nexuse2e.core.message.repository.MessageRepository;
import com.evolvsys.nexuse2e.core.message.service.MessageSenderService;
import com.evolvsys.nexuse2e.core.model.entity.MessageEntity;
import edu.umd.cs.findbugs.annotations.SuppressFBWarnings;
import lombok.RequiredArgsConstructor;
import lombok.extern.slf4j.Slf4j;
import org.springframework.stereotype.Component;

/**
 * @author SvenBarden
 * @since 22.05.2024
 */
@Component
@RequiredArgsConstructor
@Slf4j
public class MessageRequeueAction implements ProcessAction<MessageEntity> {
    public static final ProcessParameter<Long> MANUAL_ACTION_ID = new ProcessParameter<>() {};
    private final MessageRepository messageRepository;
    private final MessageSenderService messageSenderService;
    private final MessageActionManualActionService messageActionManualActionService;

    @Override
    @SuppressFBWarnings("EXS_EXCEPTION_SOFTENING_NO_CONSTRAINTS")
    public MessageEntity execute(MessageEntity entity, ProcessActionChain chain, ProcessParameters processParameters) {
        entity.setRetries(0);
        entity.setLastTry(null);
        entity = messageRepository.save(entity);

        Long manualActionId = processParameters.get(MANUAL_ACTION_ID);
        manualActionId = messageActionManualActionService.requeue(manualActionId, entity);
        chain.addOutput(MANUAL_ACTION_ID.of(manualActionId));
        try {
            messageSenderService.resend(entity.getMessageId());
        } catch (NexusException e) {
            throw new NexusRuntimeException(
                    NexusError.builder()
                            .message("Could not requeue message")
                            .extension("messageId", entity.getMessageId())
                            .build(),
                    e);
        }
        return entity;
    }

    @Override
    public ProcessActionConditions conditions(MessageEntity entity, ProcessParameters processParameters) {
        return ProcessActionConditions.EMPTY;
    }
}
