package com.evolvsys.nexuse2e.core.message.action;

import com.evolvsys.app.core.process.ProcessActionType;
import com.evolvsys.app.core.process.ProcessDefinition;
import com.evolvsys.nexuse2e.core.message.model.MessageDto;
import com.evolvsys.nexuse2e.core.model.entity.MessageEntity;
import lombok.Getter;

/**
 * @author SvenBarden
 * @since 22.05.2024
 */
@Getter
public enum MessageActionType implements ProcessActionType<MessageEntity, MessageDto> {
    REQUEUE(ProcessDefinition.<MessageEntity, MessageDto>builder()
            .action(MessageRequeueAction.class)
            .requiredPermission("NEXUS.SEND_MESSAGE")
            .build()),
    STOP(ProcessDefinition.<MessageEntity, MessageDto>builder()
            .action(MessageStopAction.class)
            .requiredPermission("NEXUS.SEND_MESSAGE")
            .isValidDto(messageDto -> MessageStopAction.VALID_STATUSES.contains(messageDto.getStatus()))
            .isValidEntity(messageEntity -> MessageStopAction.VALID_STATUSES.contains(messageEntity.getStatus()))
            .build());

    private final ProcessDefinition<MessageEntity, MessageDto> process;

    MessageActionType(ProcessDefinition<MessageEntity, MessageDto> process) {
        this.process = process;
    }
}
