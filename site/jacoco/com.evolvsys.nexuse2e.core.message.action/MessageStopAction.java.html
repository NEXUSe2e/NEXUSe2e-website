package com.evolvsys.nexuse2e.core.message.action;

import com.evolvsys.app.core.process.*;
import com.evolvsys.nexuse2e.core.manualaction.service.MessageActionManualActionService;
import com.evolvsys.nexuse2e.core.message.MessageStatus;
import com.evolvsys.nexuse2e.core.message.repository.ConversationRepository;
import com.evolvsys.nexuse2e.core.message.repository.MessageRepository;
import com.evolvsys.nexuse2e.core.model.entity.ConversationStatus;
import com.evolvsys.nexuse2e.core.model.entity.MessageEntity;
import java.util.List;
import lombok.RequiredArgsConstructor;
import lombok.extern.slf4j.Slf4j;
import org.springframework.stereotype.Component;

/**
 * @author SvenBarden
 * @since 22.05.2024
 */
@Component
@RequiredArgsConstructor
@Slf4j
public class MessageStopAction implements ProcessAction<MessageEntity> {
    public static final ProcessParameter<Long> MANUAL_ACTION_ID = new ProcessParameter<>() {};
    public static final List<MessageStatus> VALID_STATUSES = List.of(
            MessageStatus.RESOLVED,
            MessageStatus.ACTION_SUCCESSFUL,
            MessageStatus.QUEUE_ACTION,
            MessageStatus.QUEUE_SENDER,
            MessageStatus.WAITING_FOR_ACKNOWLEDGMENT);
    private final MessageRepository messageRepository;
    private final ConversationRepository conversationRepository;
    private final MessageActionManualActionService messageActionManualActionService;

    @Override
    public MessageEntity execute(MessageEntity entity, ProcessActionChain chain, ProcessParameters processParameters) {

        entity.setStatus(MessageStatus.STOPPED);
        entity.setNextRequeue(null);
        entity = messageRepository.save(entity);

        entity.getConversation().setStatus(ConversationStatus.STOPPED);
        conversationRepository.save(entity.getConversation());

        Long manualActionId = processParameters.get(MANUAL_ACTION_ID);
        manualActionId = messageActionManualActionService.stop(manualActionId, entity);
        chain.addOutput(MANUAL_ACTION_ID.of(manualActionId));

        return entity;
    }

    @Override
    public ProcessActionConditions conditions(MessageEntity entity, ProcessParameters processParameters) {
        return ProcessActionConditions.builder()
                .condition(VALID_STATUSES.contains(entity.getStatus()), "Status is not valid for stopping")
                .build();
    }
}
