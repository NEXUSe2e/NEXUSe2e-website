package com.evolvsys.nexuse2e.core.choreography.service;

import com.evolvsys.nexuse2e.core.choreography.mapper.ContextChoreographyMapper;
import com.evolvsys.nexuse2e.core.choreography.model.ContextChoreography;
import com.evolvsys.nexuse2e.core.choreography.repository.ChoreographyRepository;
import com.evolvsys.nexuse2e.core.message.context.ContextObjectFactory;
import lombok.RequiredArgsConstructor;
import org.springframework.stereotype.Component;

/**
 * @author SvenBarden
 * @since 24.01.2024
 */
@Component
@RequiredArgsConstructor
public class ChoreographyFactory implements ContextObjectFactory<String, ContextChoreography> {
    private final ChoreographyRepository choreographyRepository;
    private final ContextChoreographyMapper contextChoreographyMapper;

    @Override
    public ContextChoreography create(String choreographyId) {
        return choreographyRepository.findByChoreographyId(choreographyId).stream()
                .filter(choreographyEntity ->
                        choreographyEntity.getChoreographyId().equals(choreographyId))
                .findFirst()
                .map(contextChoreographyMapper::entityToChoreography)
                .orElseThrow(
                        () -> new IllegalArgumentException("no choreography with name " + choreographyId + " found"));
    }
}
