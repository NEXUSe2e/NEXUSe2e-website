package com.evolvsys.nexuse2e.core.choreography.service;

import com.evolvsys.app.core.graphql.annotation.GraphQLService;
import com.evolvsys.app.core.graphql.dataloader.DataLoaderService;
import com.evolvsys.nexuse2e.core.action.model.ActionDto;
import com.evolvsys.nexuse2e.core.choreography.model.ChoreographyDto;
import com.evolvsys.nexuse2e.core.choreography.repository.ChoreographyActionBatchLoader;
import com.evolvsys.nexuse2e.core.choreography.repository.ChoreographyRouteBatchLoader;
import com.evolvsys.nexuse2e.core.choreography.validation.ValidChoreographyHasActions;
import com.evolvsys.nexuse2e.core.choreography.validation.ValidChoreographyHasConversations;
import com.evolvsys.nexuse2e.core.config.service.ConfigGraphQLService;
import com.evolvsys.nexuse2e.core.route.model.RouteDto;
import io.leangen.graphql.annotations.*;
import io.leangen.graphql.execution.ResolutionEnvironment;
import jakarta.validation.Valid;
import java.util.List;
import java.util.Optional;
import java.util.Set;
import java.util.concurrent.CompletableFuture;
import lombok.RequiredArgsConstructor;
import lombok.extern.slf4j.Slf4j;

/**
 * @author Florian Drees
 * @since 18.12.2023
 */
@GraphQLService
@RequiredArgsConstructor
@Slf4j
public class ChoreographyGraphQLService {
    private final ChoreographyService choreographyService;
    private final DataLoaderService dataLoaderService;

    @GraphQLMutation(name = "choreography_update", description = "Updates an existing or creates a new choreography")
    @GraphQLNonNull
    public ChoreographyDto update(@GraphQLNonNull @Valid ChoreographyDto choreographyDto) {
        return ConfigGraphQLService.notifySubscribers(choreographyService.save(choreographyDto));
    }

    @GraphQLMutation(name = "choreography_delete", description = "Deletes a choreography")
    public boolean delete(@ValidChoreographyHasConversations @ValidChoreographyHasActions long choreographyId) {
        choreographyService.delete(choreographyId);
        ConfigGraphQLService.notifySubscribers();
        return true;
    }

    @GraphQLMutation(name = "choreography_addRoute")
    @GraphQLNonNull
    public ChoreographyDto addRoute(long choreographyId, @GraphQLNonNull @Valid RouteDto route) {
        return ConfigGraphQLService.notifySubscribers(choreographyService.addRoute(choreographyId, route));
    }

    @GraphQLMutation(name = "choreography_linkRoutes", description = "Links existing routes to the choreography")
    @GraphQLNonNull
    public ChoreographyDto linkRoutes(long choreographyId, List<@GraphQLNonNull Long> routeIds) {
        return ConfigGraphQLService.notifySubscribers(choreographyService.linkRoutes(choreographyId, routeIds));
    }

    @GraphQLMutation(name = "choreography_removeRoute", description = "Removes the route from the choreography")
    @GraphQLNonNull
    public ChoreographyDto removeRoute(long choreographyId, long routeId) {
        return ConfigGraphQLService.notifySubscribers(choreographyService.removeRoute(choreographyId, routeId));
    }

    @GraphQLQuery(name = "choreography_list", description = "List all choreographies")
    @GraphQLNonNull
    public List<@GraphQLNonNull ChoreographyDto> list() {
        return choreographyService.list();
    }

    @GraphQLQuery(name = "choreography_getById", description = "Gets a choreography")
    public Optional<ChoreographyDto> getById(long choreographyId) {
        return choreographyService.getById(choreographyId);
    }

    @GraphQLQuery(description = "list of actions of the provided choreography")
    public CompletableFuture<Set<@GraphQLNonNull ActionDto>> actions(
            @GraphQLContext ChoreographyDto choreographyDto,
            @GraphQLEnvironment ResolutionEnvironment resolutionEnvironment) {
        LOGGER.trace("getting actions for choreography {}", choreographyDto);
        return dataLoaderService.loadRelation(
                resolutionEnvironment, ChoreographyActionBatchLoader.class, choreographyDto);
    }

    @GraphQLQuery(description = "list of routes of the provided choreography")
    public CompletableFuture<Set<@GraphQLNonNull RouteDto>> routes(
            @GraphQLContext ChoreographyDto choreographyDto,
            @GraphQLEnvironment ResolutionEnvironment resolutionEnvironment) {
        return dataLoaderService.loadRelation(
                resolutionEnvironment, ChoreographyRouteBatchLoader.class, choreographyDto);
    }
}
