package com.evolvsys.nexuse2e.core.choreography.service;

import com.evolvsys.app.core.auth.service.CurrentUserService;
import com.evolvsys.app.core.process.ProcessExecutionContext;
import com.evolvsys.app.core.process.ProcessService;
import com.evolvsys.app.core.service.BaseService;
import com.evolvsys.app.core.service.ListService;
import com.evolvsys.nexuse2e.core.choreography.action.ChoreographyActionType;
import com.evolvsys.nexuse2e.core.choreography.action.ChoreographyAddRouteAction;
import com.evolvsys.nexuse2e.core.choreography.action.ChoreographyLinkRouteAction;
import com.evolvsys.nexuse2e.core.choreography.action.ChoreographyRemoveRouteAction;
import com.evolvsys.nexuse2e.core.choreography.mapper.ChoreographyMapper;
import com.evolvsys.nexuse2e.core.choreography.model.ChoreographyDto;
import com.evolvsys.nexuse2e.core.choreography.repository.ChoreographyRepository;
import com.evolvsys.nexuse2e.core.model.entity.ChoreographyEntity;
import com.evolvsys.nexuse2e.core.route.model.RouteDto;
import java.util.Comparator;
import java.util.List;
import org.springframework.stereotype.Service;

/**
 * @author Florian Drees
 * @since 18.12.2023
 */
@Service
public class ChoreographyService extends BaseService<ChoreographyDto, ChoreographyEntity, ChoreographyRepository> {
    private final ProcessService processService;
    private final ListService listService;

    public ChoreographyService(
            ChoreographyMapper mapper,
            ChoreographyRepository repository,
            CurrentUserService currentUserService,
            ProcessService processService,
            ListService listService) {
        super(mapper, repository, currentUserService);
        this.processService = processService;
        this.listService = listService;
    }

    @Override
    protected ChoreographyEntity createEntity() {
        return new ChoreographyEntity();
    }

    public List<ChoreographyDto> list() {
        return listService.list(this).stream()
                .sorted(Comparator.comparing(ChoreographyEntity::getChoreographyId))
                .map(this::mapEntityToDto)
                .toList();
    }

    public ChoreographyDto addRoute(long choreographyId, RouteDto routeDto) {
        ChoreographyEntity choreographyEntity =
                findWithActionCheck(choreographyId, ChoreographyActionType.ADD_ROUTE.name());

        choreographyEntity =
                processService.execute(ProcessExecutionContext.<ChoreographyEntity, ChoreographyDto>builder()
                        .target(choreographyEntity)
                        .actionType(ChoreographyActionType.ADD_ROUTE)
                        .parameterValue(ChoreographyAddRouteAction.ROUTE_PARAMETER.of(routeDto))
                        .build());

        return mapEntityToDto(choreographyEntity);
    }

    public ChoreographyDto removeRoute(long choreographyId, long routeId) {
        ChoreographyEntity choreographyEntity =
                findWithActionCheck(choreographyId, ChoreographyActionType.REMOVE_ROUTE.name());

        choreographyEntity =
                processService.execute(ProcessExecutionContext.<ChoreographyEntity, ChoreographyDto>builder()
                        .target(choreographyEntity)
                        .actionType(ChoreographyActionType.REMOVE_ROUTE)
                        .parameterValue(ChoreographyRemoveRouteAction.ROUTE_ID_PARAMETER.of(routeId))
                        .build());

        return mapEntityToDto(choreographyEntity);
    }

    public ChoreographyDto linkRoutes(long choreographyId, List<Long> routeIds) {
        ChoreographyEntity choreographyEntity =
                findWithActionCheck(choreographyId, ChoreographyActionType.LINK_ROUTE.name());

        choreographyEntity =
                processService.execute(ProcessExecutionContext.<ChoreographyEntity, ChoreographyDto>builder()
                        .target(choreographyEntity)
                        .actionType(ChoreographyActionType.LINK_ROUTE)
                        .parameterValue(ChoreographyLinkRouteAction.ROUTE_ID_PARAMETER.of(routeIds))
                        .build());

        return mapEntityToDto(choreographyEntity);
    }
}
