package com.evolvsys.nexuse2e.core.ebxml;

import com.evolvsys.nexuse2e.core.message.context.ExecutionContext;
import java.util.regex.Matcher;
import java.util.regex.Pattern;

public enum CpaIdScheme {
    CHOREOGRAPHY_ID {
        @Override
        public String generateCpaId(ExecutionContext executionContext) {
            return executionContext.getChoreographyId();
        }

        @Override
        public String getChoreographyIdFromCpaId(String cpaId) {
            return cpaId;
        }
    },

    URI {
        @Override
        public String generateCpaId(ExecutionContext executionContext) {
            return String.format(
                    "uri://%s/%s/%s",
                    executionContext.getSourcePartnerId(),
                    executionContext.getTargetPartnerId(),
                    executionContext.getChoreographyId());
        }

        @Override
        public String getChoreographyIdFromCpaId(String cpaId) {
            Matcher matcher = URI_PATTERN.matcher(cpaId);
            return matcher.matches() ? matcher.group(1) : cpaId;
        }
    };

    public static final CpaIdScheme DEFAULT_SCHEME = CHOREOGRAPHY_ID;
    private static final Pattern URI_PATTERN = Pattern.compile("uri://[^/]+/[^/]+/(.+)");

    public abstract String generateCpaId(ExecutionContext executionContext);

    public abstract String getChoreographyIdFromCpaId(String cpaId);
}
