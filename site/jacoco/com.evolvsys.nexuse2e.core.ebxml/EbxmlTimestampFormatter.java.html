package com.evolvsys.nexuse2e.core.ebxml;

import java.time.LocalDateTime;
import java.time.OffsetDateTime;
import java.time.ZoneId;
import java.time.ZonedDateTime;
import java.time.format.DateTimeFormatter;

public class EbxmlTimestampFormatter {

    private static final String FORMAT = "yyyy-MM-dd'T'HH:mm:ss'Z'";
    private static final DateTimeFormatter DATE_TIME_FORMATTER =
            DateTimeFormatter.ofPattern(FORMAT).withZone(ZoneId.of("UTC"));

    public static final EbxmlTimestampFormatter INSTANCE = new EbxmlTimestampFormatter();

    public static EbxmlTimestampFormatter getInstance() {
        return INSTANCE;
    }

    public String getTimestamp(ZonedDateTime dateTime) {
        return dateTime.format(DATE_TIME_FORMATTER);
    }

    public ZonedDateTime parseTimestamp(String timestamp) {
        return OffsetDateTime.parse(timestamp).atZoneSameInstant(ZoneId.of("UTC"));
    }

    public ZonedDateTime parseTimestampWithoutZone(String timestamp) {
        return LocalDateTime.parse(timestamp).atZone(ZoneId.of("UTC"));
    }
}
