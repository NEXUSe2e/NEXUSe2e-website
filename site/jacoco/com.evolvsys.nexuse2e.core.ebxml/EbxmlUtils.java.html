package com.evolvsys.nexuse2e.core.ebxml;

import com.evolvsys.nexuse2e.core.constants.HttpConstants;
import com.evolvsys.nexuse2e.core.message.context.ExecutionContext;
import com.evolvsys.nexuse2e.internal.ContextPayload;
import jakarta.xml.soap.Name;
import jakarta.xml.soap.SOAPException;
import jakarta.xml.soap.SOAPFactory;
import java.util.HashMap;
import java.util.Map;
import java.util.Optional;
import java.util.regex.Matcher;
import java.util.regex.Pattern;
import lombok.experimental.UtilityClass;
import lombok.extern.slf4j.Slf4j;
import org.springframework.http.MediaType;
import org.springframework.util.InvalidMimeTypeException;

/**
 * @author Florian Drees
 * @since 14.11.2023
 */
@UtilityClass
@Slf4j
public class EbxmlUtils {
    private static final Pattern CONTENT_TYPE_PATTERN = Pattern.compile("(?<contentType>[^;]+);?|([^;=]+=[^;=]+);?");

    public static boolean isBinaryType(String contentType) {
        Optional<MediaType> optionalMediaType = tryToParseMediaType(contentType);
        if (optionalMediaType.isEmpty()) {
            LOGGER.debug("Unable to find media type for type {}", contentType);
            return true;
        } else {
            return EbxmlConfig.getNonBinaryMimeTypes().contains(contentType) == false;
        }
    }

    public static Optional<MediaType> tryToParseMediaType(String contentType) {
        try {
            return Optional.of(MediaType.parseMediaType(contentType));
        } catch (InvalidMimeTypeException e) {
            LOGGER.trace("unable to parse media type for content type {}", contentType);
        }
        return Optional.empty();
    }

    public static Optional<String> extractMimeType(String mimeBodyPartString) {
        Matcher matcher = CONTENT_TYPE_PATTERN.matcher(mimeBodyPartString);
        if (matcher.matches()) {
            return Optional.ofNullable(matcher.group("contentType"));
        } else {
            return Optional.empty();
        }
    }

    public static Optional<String> getEncoding(ExecutionContext executionContext) {
        Optional<String> optionalEncoding;
        ContextPayload data = executionContext.getData();
        if (data != null) {
            optionalEncoding = Optional.ofNullable(data.getEncoding());
        } else {
            LOGGER.debug(
                    "ContextPayload of message with id {} is null, trying to access encoding from meta data",
                    executionContext.getMessage().getId());
            optionalEncoding = executionContext.getMetaData().getFirst(HttpConstants.ENCODING);
        }
        return optionalEncoding;
    }

    public static Name createNameWithEbxmlNamespace(SOAPFactory soapFactory, String localName) throws SOAPException {
        return soapFactory.createName(localName, EbxmlConstants.EBXML_NAMESPACE_PREFIX, EbxmlConstants.EBXML_NAMESPACE);
    }

    public static MediaType getMultipartRelatedContentType(ExecutionContext executionContext) {
        return getMultipartRelatedContentType(executionContext.getMessage().getId(), EbxmlConstants.PROTOCOL);
    }

    public static MediaType getMultipartRelatedContentType(String messageId, String protocol) {
        Map<String, String> parameterMap = new HashMap<>();
        parameterMap.put("type", "\"text/xml\"");
        parameterMap.put("boundary", "\"MIME_boundary\"");
        parameterMap.put("start", String.format("\"%s%s-Header\"", messageId, protocol));
        return new MediaType("multipart", "related", parameterMap);
    }
}
