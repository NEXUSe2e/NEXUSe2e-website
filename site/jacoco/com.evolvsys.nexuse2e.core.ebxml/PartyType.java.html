package com.evolvsys.nexuse2e.core.ebxml;

import com.evolvsys.nexuse2e.core.constants.Constant;
import lombok.Getter;

@Getter
public enum PartyType {
    FROM("From", EbxmlConstants.PARTNER_TYPE_FROM, EbxmlConstants.PARTNER_ROLE_FROM),
    TO("To", EbxmlConstants.PARTNER_TYPE_TO, EbxmlConstants.PARTNER_ROLE_TO);

    private final String displayName;
    private final Constant<String> partnerTypeConstant;
    private final Constant<String> partnerRoleConstant;

    PartyType(String displayName, Constant<String> partnerTypeConstant, Constant<String> partnerRoleConstant) {
        this.displayName = displayName;
        this.partnerTypeConstant = partnerTypeConstant;
        this.partnerRoleConstant = partnerRoleConstant;
    }
}
