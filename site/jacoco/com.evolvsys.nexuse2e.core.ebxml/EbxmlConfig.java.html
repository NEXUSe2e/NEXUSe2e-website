package com.evolvsys.nexuse2e.core.ebxml;

import com.evolvsys.nexuse2e.core.config.NEXUSe2eConfigurationProperties;
import java.util.ArrayList;
import java.util.Collections;
import java.util.List;
import lombok.RequiredArgsConstructor;
import org.apache.commons.collections4.CollectionUtils;
import org.springframework.beans.factory.InitializingBean;
import org.springframework.stereotype.Component;

/**
 * @author SvenBarden
 * @since 12.04.2024
 */
@Component
@RequiredArgsConstructor
class EbxmlConfig implements InitializingBean {
    private static final List<String> NON_BINARY_MIME_TYPES = new ArrayList<>();
    private final NEXUSe2eConfigurationProperties configurationProperties;

    static {
        CollectionUtils.addAll(
                NON_BINARY_MIME_TYPES,
                "text/plain",
                "text/html",
                "application/xhtml+xml",
                "text/xml",
                "application/xml",
                "text/csv");
    }

    static List<String> getNonBinaryMimeTypes() {
        return Collections.unmodifiableList(NON_BINARY_MIME_TYPES);
    }

    @Override
    public void afterPropertiesSet() throws Exception {
        NEXUSe2eConfigurationProperties.Ebxml ebxml = configurationProperties.getEbxml();
        if (ebxml != null && CollectionUtils.isNotEmpty(ebxml.getNonBinaryMimeTypes())) {
            NON_BINARY_MIME_TYPES.addAll(ebxml.getNonBinaryMimeTypes());
        }
    }
}
