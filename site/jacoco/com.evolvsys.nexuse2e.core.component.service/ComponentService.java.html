package com.evolvsys.nexuse2e.core.component.service;

import com.evolvsys.app.core.auth.service.CurrentUserService;
import com.evolvsys.nexuse2e.core.component.ComponentFilter;
import com.evolvsys.nexuse2e.core.component.ComponentRegistry;
import com.evolvsys.nexuse2e.core.component.RegisteredNexusComponent;
import com.evolvsys.nexuse2e.core.component.mapper.ComponentMapper;
import com.evolvsys.nexuse2e.core.component.model.dto.ComponentDto;
import com.evolvsys.nexuse2e.core.parameter.mapper.ParameterMapper;
import com.evolvsys.nexuse2e.core.parameter.model.ParameterDto;
import com.evolvsys.nexuse2e.core.system.security.BaseConfigAccessInterceptor;
import java.util.Collections;
import java.util.Comparator;
import java.util.List;
import lombok.RequiredArgsConstructor;
import lombok.extern.slf4j.Slf4j;
import org.apache.commons.collections4.CollectionUtils;
import org.springframework.lang.NonNull;
import org.springframework.lang.Nullable;
import org.springframework.stereotype.Service;

/**
 * @author Florian Drees
 * @since 16.11.2023
 */
@Service
@RequiredArgsConstructor
@Slf4j
public class ComponentService {
    private final ComponentRegistry componentRegistry;
    private final ComponentMapper componentMapper;
    private final ParameterMapper parameterMapper;
    private final CurrentUserService currentUserService;

    public List<ComponentDto> list(@Nullable ComponentFilter componentFilter) {
        return currentUserService.hasAnySystemPermission(BaseConfigAccessInterceptor.READ_CONFIG)
                ? componentRegistry.getRegisteredComponents().stream()
                        .filter(c -> ComponentFilterUtils.filter(c, componentFilter))
                        .sorted(Comparator.comparing(RegisteredNexusComponent::getComponentId))
                        .map(componentMapper::componentToDto)
                        .toList()
                : Collections.emptyList();
    }

    public List<ParameterDto> getParameterDefinitions(@NonNull String component) {
        return componentRegistry
                .getComponent(component)
                .map(registeredComponent -> CollectionUtils.emptyIfNull(registeredComponent.getParameterDefinitions()))
                .orElse(Collections.emptyList())
                .stream()
                .map(definition -> parameterMapper.definitionToDto(definition, currentUserService.getCurrentUserData()))
                .toList();
    }
}
