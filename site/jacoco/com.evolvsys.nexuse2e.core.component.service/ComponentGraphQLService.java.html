package com.evolvsys.nexuse2e.core.component.service;

import com.evolvsys.app.core.graphql.annotation.GraphQLService;
import com.evolvsys.nexuse2e.core.component.ComponentFilter;
import com.evolvsys.nexuse2e.core.component.model.dto.ComponentDto;
import com.evolvsys.nexuse2e.core.parameter.model.ParameterDto;
import io.leangen.graphql.annotations.GraphQLContext;
import io.leangen.graphql.annotations.GraphQLNonNull;
import io.leangen.graphql.annotations.GraphQLQuery;
import java.util.List;
import lombok.RequiredArgsConstructor;

/**
 * @author Florian Drees
 * @since 16.11.2023
 */
@GraphQLService
@RequiredArgsConstructor
public class ComponentGraphQLService {
    private final ComponentService componentService;

    @GraphQLQuery(name = "component_list")
    @GraphQLNonNull
    public List<@GraphQLNonNull ComponentDto> list(ComponentFilter componentFilter) {
        return componentService.list(componentFilter);
    }

    @GraphQLQuery(description = "Gets all parameter definitions of the provided component")
    public List<@GraphQLNonNull ParameterDto> parameterDefinitions(@GraphQLContext ComponentDto componentDto) {
        return componentService.getParameterDefinitions(componentDto.getComponentId());
    }
}
