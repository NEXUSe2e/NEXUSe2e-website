package com.evolvsys.nexuse2e.core.component.service;

import com.evolvsys.nexuse2e.core.component.ComponentFilter;
import com.evolvsys.nexuse2e.core.component.RegisteredNexusComponent;
import java.util.List;
import java.util.function.Function;
import lombok.Value;
import lombok.experimental.UtilityClass;
import org.apache.commons.lang3.StringUtils;

/**
 * @author SvenBarden
 * @since 11.12.2023
 */
@UtilityClass
public class ComponentFilterUtils {
    private static final List<Filter> FILTERS = List.of(
            (componentFilter, component) -> componentFilter.getComponentType() == null
                    || componentFilter.getComponentType() == component.getComponentType(),
            new StringFilter(ComponentFilter::getComponentId, RegisteredNexusComponent::getComponentId),
            new StringFilter(ComponentFilter::getDescription, RegisteredNexusComponent::getDescription),
            new StringFilter(ComponentFilter::getHelpText, RegisteredNexusComponent::getHelpText));

    public static boolean filter(RegisteredNexusComponent nexusComponent, ComponentFilter componentFilter) {
        boolean result;
        if (componentFilter == null) {
            result = true;
        } else {
            result = FILTERS.stream().allMatch(filter -> filter.filter(componentFilter, nexusComponent));
        }
        return result;
    }

    @FunctionalInterface
    private interface Filter {

        boolean filter(ComponentFilter componentFilter, RegisteredNexusComponent component);
    }

    @Value
    private static class StringFilter implements Filter {
        Function<ComponentFilter, String> filterGetter;
        Function<RegisteredNexusComponent, String> componentGetter;

        @Override
        public boolean filter(ComponentFilter componentFilter, RegisteredNexusComponent component) {
            boolean result;

            String filterValue = filterGetter.apply(componentFilter);
            if (StringUtils.isNotBlank(filterValue)) {
                String componentValue = componentGetter.apply(component);
                if (StringUtils.isBlank(componentValue)) {
                    result = false;
                } else {
                    result = componentValue.toLowerCase().contains(filterValue.toLowerCase());
                }
            } else {
                result = true;
            }
            return result;
        }
    }
}
