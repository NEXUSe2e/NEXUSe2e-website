package com.evolvsys.nexuse2e.core.component.service;

import com.evolvsys.nexuse2e.core.component.creation.ComponentPostCreationProcessor;
import java.util.List;
import lombok.RequiredArgsConstructor;
import lombok.extern.slf4j.Slf4j;
import org.apache.commons.collections4.CollectionUtils;
import org.springframework.stereotype.Service;

/**
 * <p>Service to handle the processing after a component has been created.</p>
 *
 * @author Florian Drees
 * @since 01.02.2024
 */
@Service
@RequiredArgsConstructor
@Slf4j
public class ComponentPostCreationService {
    private final List<ComponentPostCreationProcessor<?>> postCreationProcessors;

    @SuppressWarnings("unchecked")
    public <T> void processComponentPostCreation(T component) {
        CollectionUtils.emptyIfNull(postCreationProcessors).stream()
                .filter(processor -> processor.getProcessingInterfaceType().isInstance(component))
                .forEach(processor -> {
                    LOGGER.trace(
                            "executing post-creation processing {} for component {}",
                            processor.getProcessingInterfaceType().getSimpleName(),
                            component.getClass().getSimpleName());
                    ((ComponentPostCreationProcessor<T>) processor).execute(component);
                });
    }
}
