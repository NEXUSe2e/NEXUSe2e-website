package com.evolvsys.nexuse2e.core.message.service;

import com.evolvsys.app.core.graphql.dataloader.GraphQLBatchLoader;
import com.evolvsys.app.core.persistence.model.IdDto;
import com.evolvsys.nexuse2e.core.message.model.ConversationDto;
import com.evolvsys.nexuse2e.core.message.model.ConversationPartnerDto;
import com.evolvsys.nexuse2e.core.model.entity.QMessageEntity;
import com.querydsl.core.CloseableIterator;
import com.querydsl.core.FetchableQuery;
import com.querydsl.core.ResultTransformer;
import com.querydsl.core.Tuple;
import com.querydsl.jpa.JPQLQueryFactory;
import java.util.*;
import java.util.concurrent.CompletableFuture;
import java.util.concurrent.CompletionStage;
import java.util.function.Function;
import java.util.stream.Collectors;
import lombok.RequiredArgsConstructor;
import org.apache.commons.collections4.CollectionUtils;
import org.dataloader.BatchLoaderEnvironment;
import org.springframework.stereotype.Component;

/**
 * @author SvenBarden
 * @since 20.03.2024
 */
@Component
@RequiredArgsConstructor
public class ConversationPartnersBatchLoader
        implements GraphQLBatchLoader<ConversationDto, Set<ConversationPartnerDto>> {
    private final JPQLQueryFactory queryFactory;

    @Override
    public CompletionStage<Map<ConversationDto, Set<ConversationPartnerDto>>> load(
            Set<ConversationDto> keys, BatchLoaderEnvironment environment) {
        Set<ConversationDto> nonNullKeys =
                keys.stream().filter(Objects::nonNull).collect(Collectors.toSet());

        CompletionStage<Map<ConversationDto, Set<ConversationPartnerDto>>> result;
        if (CollectionUtils.isEmpty(nonNullKeys)) {
            result = CompletableFuture.completedStage(new HashMap<>());
        } else {
            result = CompletableFuture.supplyAsync(() -> loadInternal(nonNullKeys));
        }
        return result;
    }

    private Map<ConversationDto, Set<ConversationPartnerDto>> loadInternal(Set<ConversationDto> keys) {
        List<Long> idKeys = keys.stream().map(IdDto::getId).distinct().toList();

        return queryFactory
                .from(QMessageEntity.messageEntity)
                .where(QMessageEntity.messageEntity.conversation.id.in(idKeys))
                .groupBy(
                        QMessageEntity.messageEntity.conversation.id,
                        QMessageEntity.messageEntity.sourcePartner.id,
                        QMessageEntity.messageEntity.targetPartner.id)
                .transform(new CustomTransformer(keys));
    }

    @RequiredArgsConstructor
    private static class CustomTransformer
            implements ResultTransformer<Map<ConversationDto, Set<ConversationPartnerDto>>> {
        private final Set<ConversationDto> conversationDtos;

        @Override
        public Map<ConversationDto, Set<ConversationPartnerDto>> transform(FetchableQuery<?, ?> query) {
            Map<ConversationDto, Set<ConversationPartnerDto>> result = new HashMap<>();

            Map<Long, ConversationDto> conversationMap =
                    conversationDtos.stream().collect(Collectors.toMap(IdDto::getId, Function.identity()));

            try (CloseableIterator<Tuple> iterate = query.select(
                            QMessageEntity.messageEntity.conversation.id,
                            QMessageEntity.messageEntity.sourcePartner.id,
                            QMessageEntity.messageEntity.targetPartner.id)
                    .iterate()) {
                while (iterate.hasNext()) {
                    Tuple tuple = iterate.next();
                    Long conversationId = tuple.get(QMessageEntity.messageEntity.conversation.id);
                    Long sourcePartnerId = tuple.get(QMessageEntity.messageEntity.sourcePartner.id);
                    Long targetPartnerId = tuple.get(QMessageEntity.messageEntity.targetPartner.id);

                    ConversationDto conversationDto = conversationMap.get(conversationId);

                    Set<ConversationPartnerDto> conversationPartnerDtos =
                            result.computeIfAbsent(conversationDto, aLong -> new HashSet<>());
                    conversationPartnerDtos.add(new ConversationPartnerDto(sourcePartnerId));
                    conversationPartnerDtos.add(new ConversationPartnerDto(targetPartnerId));
                }
            }
            return result;
        }
    }
}
