package com.evolvsys.nexuse2e.core.message.service;

import com.evolvsys.nexuse2e.core.message.MessageFilter;
import com.evolvsys.nexuse2e.core.message.repository.MessageRepository;
import com.evolvsys.nexuse2e.core.settings.service.ApplicationSettingsService;
import lombok.RequiredArgsConstructor;
import org.springframework.stereotype.Service;

/**
 * @author Florian Drees
 * @since 24.09.2024
 */
@Service
@RequiredArgsConstructor
public class MessageDataService {
    private final MessageRepository messageRepository;
    private final ApplicationSettingsService applicationSettingsService;

    public boolean exists(MessageFilter messageFilter) {
        return messageRepository.exists(
                messageFilter,
                applicationSettingsService.getSystemSettingsOrDefault().getTimeZone());
    }
}
