package com.evolvsys.nexuse2e.core.message.service;

import com.evolvsys.app.core.auth.service.CurrentUserService;
import com.evolvsys.app.core.exception.NoActionAccessException;
import com.evolvsys.app.core.process.ProcessExecutionContext;
import com.evolvsys.app.core.process.ProcessResult;
import com.evolvsys.app.core.process.ProcessService;
import com.evolvsys.app.core.service.BaseService;
import com.evolvsys.app.core.service.ListService;
import com.evolvsys.nexuse2e.core.exception.NexusException;
import com.evolvsys.nexuse2e.core.message.MessageFilter;
import com.evolvsys.nexuse2e.core.message.action.MessageActionType;
import com.evolvsys.nexuse2e.core.message.action.MessageRequeueAction;
import com.evolvsys.nexuse2e.core.message.action.MessageStopAction;
import com.evolvsys.nexuse2e.core.message.mapper.MessageMapper;
import com.evolvsys.nexuse2e.core.message.model.MessageDto;
import com.evolvsys.nexuse2e.core.message.repository.MessageRepository;
import com.evolvsys.nexuse2e.core.model.entity.MessageEntity;
import com.evolvsys.nexuse2e.core.settings.service.ApplicationSettingsService;
import java.util.List;
import lombok.extern.slf4j.Slf4j;
import org.springframework.data.domain.Page;
import org.springframework.data.domain.Pageable;
import org.springframework.lang.Nullable;
import org.springframework.stereotype.Service;

/**
 * @author Florian Drees
 * @since 27.12.2023
 */
@Service("EntityMessageService")
@Slf4j
public class MessageService extends BaseService<MessageDto, MessageEntity, MessageRepository> {
    private final ListService listService;
    private final ApplicationSettingsService applicationSettingsService;
    private final ProcessService processService;

    public MessageService(
            MessageMapper mapper,
            MessageRepository repository,
            CurrentUserService currentUserService,
            ListService listService,
            ApplicationSettingsService applicationSettingsService,
            ProcessService processService) {
        super(mapper, repository, currentUserService);
        this.listService = listService;
        this.applicationSettingsService = applicationSettingsService;
        this.processService = processService;
    }

    @Override
    protected MessageEntity createEntity() {
        return new MessageEntity();
    }

    public Page<MessageDto> list(Pageable pageable, @Nullable MessageFilter messageFilter) {
        return listService
                .list(pageable, messageFilter, getDtoType(), (pageable1, filter, predicate) -> getRepository()
                        .findAllToRead(
                                pageable1,
                                filter,
                                applicationSettingsService
                                        .getSystemSettingsOrDefault()
                                        .getTimeZone(),
                                predicate))
                .map(this::mapEntityToDto);
    }

    public void resend(List<Long> messageIds) throws NexusException {
        Long manualActionId = null;
        for (long messageId : messageIds) {
            try {
                MessageEntity messageEntity = findWithActionCheck(messageId, MessageActionType.REQUEUE.name());
                ProcessResult<MessageEntity> result =
                        processService.executeAsResult(ProcessExecutionContext.<MessageEntity, MessageDto>builder()
                                .target(messageEntity)
                                .actionType(MessageActionType.REQUEUE)
                                .parameterValue(MessageRequeueAction.MANUAL_ACTION_ID.of(manualActionId))
                                .build());
                manualActionId = result.getOutput(MessageRequeueAction.MANUAL_ACTION_ID);
            } catch (NoActionAccessException e) {
                LOGGER.warn("message with internalId {} is not in a valid state for requeue", messageId, e);
                // ignore
            }
        }
    }

    public void stop(List<Long> messageIds) {
        Long manualActionId = null;
        for (Long messageId : messageIds) {
            try {
                MessageEntity messageEntity = findWithActionCheck(messageId, MessageActionType.STOP.name());
                ProcessResult<MessageEntity> result =
                        processService.executeAsResult(ProcessExecutionContext.<MessageEntity, MessageDto>builder()
                                .target(messageEntity)
                                .actionType(MessageActionType.STOP)
                                .parameterValue(MessageStopAction.MANUAL_ACTION_ID.of(manualActionId))
                                .build());
                manualActionId = result.getOutput(MessageStopAction.MANUAL_ACTION_ID);
            } catch (NoActionAccessException e) {
                LOGGER.warn("message with internalId {} is not in a valid state for stopping", messageId, e);
                // ignore
            }
        }
    }
}
