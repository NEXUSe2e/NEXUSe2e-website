package com.evolvsys.nexuse2e.core.message.service;

import com.evolvsys.nexuse2e.core.exception.NexusException;
import com.evolvsys.nexuse2e.core.logging.MessageLogContextHolder;
import com.evolvsys.nexuse2e.core.message.context.ContextFactory;
import com.evolvsys.nexuse2e.core.message.context.MessageContext;
import com.evolvsys.nexuse2e.core.message.dispatcher.OutboundMessageDispatcher;
import com.evolvsys.nexuse2e.internal.MessageType;
import lombok.RequiredArgsConstructor;
import lombok.extern.slf4j.Slf4j;
import org.apache.commons.lang3.Validate;
import org.springframework.stereotype.Service;

/**
 * @author SvenBarden
 * @since 02.02.2024
 */
@Service
@RequiredArgsConstructor
@Slf4j
public class MessageSenderService {
    private final ContextFactory contextFactory;
    private final OutboundMessageDispatcher outboundMessageDispatcher;

    public void resend(String messageId) throws NexusException {
        try {
            MessageContext messageContext = contextFactory.createForResend(messageId);

            Validate.notNull(messageContext.getSourceMessage(), "messageContext.getSourceMessage() must not be null");

            MessageType messageType = MessageType.NORMAL;
            if (messageContext.getAcknowledgmentMessage() != null
                    && messageId.equals(
                            messageContext.getAcknowledgmentMessage().getId())) {
                messageType = MessageType.ACKNOWLEDGMENT;
            }

            LOGGER.debug("sending message {} as {}", messageId, messageType);

            outboundMessageDispatcher.dispatch(messageContext.createExecutionContext(messageType));
        } finally {
            MessageLogContextHolder.clearContext();
        }
    }
}
