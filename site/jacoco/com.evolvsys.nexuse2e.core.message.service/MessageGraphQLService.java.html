package com.evolvsys.nexuse2e.core.message.service;

import com.evolvsys.app.core.graphql.annotation.GraphQLService;
import com.evolvsys.app.core.graphql.dataloader.DataLoaderService;
import com.evolvsys.nexuse2e.core.action.model.ActionDto;
import com.evolvsys.nexuse2e.core.action.service.ActionBatchLoader;
import com.evolvsys.nexuse2e.core.exception.NexusException;
import com.evolvsys.nexuse2e.core.message.MessageFilter;
import com.evolvsys.nexuse2e.core.message.MessageManualActionCountBatchLoader;
import com.evolvsys.nexuse2e.core.message.model.ConversationDto;
import com.evolvsys.nexuse2e.core.message.model.MessageDto;
import com.evolvsys.nexuse2e.core.message.model.MetaDataDto;
import com.evolvsys.nexuse2e.core.message.payload.MetaDataDtoComparator;
import com.evolvsys.nexuse2e.core.message.payload.model.PayloadDto;
import com.evolvsys.nexuse2e.core.message.repository.ConversationBatchLoader;
import com.evolvsys.nexuse2e.core.partner.model.PartnerDto;
import com.evolvsys.nexuse2e.core.partner.service.PartnerBatchLoader;
import io.leangen.graphql.annotations.*;
import io.leangen.graphql.execution.ResolutionEnvironment;
import java.util.*;
import java.util.concurrent.CompletableFuture;
import lombok.RequiredArgsConstructor;
import lombok.extern.slf4j.Slf4j;
import org.apache.commons.collections4.CollectionUtils;
import org.apache.tika.config.TikaConfig;
import org.apache.tika.mime.MimeTypeException;
import org.springframework.data.domain.Page;
import org.springframework.data.domain.Pageable;
import org.springframework.lang.Nullable;

/**
 * @author Florian Drees
 * @since 27.12.2023
 */
@GraphQLService
@RequiredArgsConstructor
@Slf4j
public class MessageGraphQLService {
    private final MessageService messageService;
    private final DataLoaderService dataLoaderService;

    @GraphQLQuery(name = "message_list", description = "List all messages")
    @GraphQLNonNull
    public Page<@GraphQLNonNull MessageDto> list(Pageable pageable, @Nullable MessageFilter messageFilter) {
        return messageService.list(pageable, messageFilter);
    }

    @GraphQLQuery(name = "message_getById", description = "Gets a message")
    public Optional<MessageDto> getById(long messageId) {
        return messageService.getById(messageId);
    }

    @GraphQLMutation(name = "message_resend")
    public boolean resend(List<@GraphQLNonNull Long> messageIds) throws NexusException {
        messageService.resend(messageIds);
        return true;
    }

    @GraphQLMutation(name = "message_stop", description = "Stop messages in queue")
    public boolean stop(List<@GraphQLNonNull Long> messageIds) {
        messageService.stop(messageIds);
        return true;
    }

    @GraphQLQuery(description = "conversation of the provided message")
    public CompletableFuture<ConversationDto> conversation(
            @GraphQLContext MessageDto messageDto, @GraphQLEnvironment ResolutionEnvironment resolutionEnvironment) {
        return dataLoaderService.load(
                resolutionEnvironment,
                ConversationBatchLoader.class,
                messageDto,
                "conversationId",
                MessageDto::getConversationId);
    }

    @GraphQLQuery
    public CompletableFuture<MessageDto> referenceMessage(
            @GraphQLContext MessageDto messageDto, @GraphQLEnvironment ResolutionEnvironment resolutionEnvironment) {
        return dataLoaderService.load(
                resolutionEnvironment,
                MessageBatchLoader.class,
                messageDto,
                "referenceMessageId",
                MessageDto::getReferenceMessageId);
    }

    @GraphQLQuery
    public CompletableFuture<ActionDto> action(
            @GraphQLContext MessageDto messageDto, @GraphQLEnvironment ResolutionEnvironment resolutionEnvironment) {
        return dataLoaderService.load(
                resolutionEnvironment, ActionBatchLoader.class, messageDto, "actionId", MessageDto::getActionId);
    }

    @GraphQLQuery
    public CompletableFuture<PartnerDto> sourcePartner(
            @GraphQLContext MessageDto messageDto, @GraphQLEnvironment ResolutionEnvironment resolutionEnvironment) {
        return dataLoaderService.load(
                resolutionEnvironment,
                PartnerBatchLoader.class,
                messageDto,
                "sourcePartnerId",
                MessageDto::getSourcePartnerId);
    }

    @GraphQLQuery
    public CompletableFuture<PartnerDto> targetPartner(
            @GraphQLContext MessageDto messageDto, @GraphQLEnvironment ResolutionEnvironment resolutionEnvironment) {
        return dataLoaderService.load(
                resolutionEnvironment,
                PartnerBatchLoader.class,
                messageDto,
                "targetPartnerId",
                MessageDto::getTargetPartnerId);
    }

    @GraphQLQuery(description = "set of meta data's of the provided message")
    public CompletableFuture<Set<@GraphQLNonNull MetaDataDto>> metaDatas(
            @GraphQLContext MessageDto messageDto, @GraphQLEnvironment ResolutionEnvironment resolutionEnvironment) {
        return dataLoaderService
                .loadRelation(resolutionEnvironment, MessageMetaDataBatchLoader.class, messageDto)
                .thenApply(metaDataDtos -> updateMetaDatas(messageDto, metaDataDtos));
    }

    @GraphQLQuery(description = "payloads of the message")
    public CompletableFuture<Set<@GraphQLNonNull PayloadDto>> payloads(
            @GraphQLContext MessageDto messageDto, @GraphQLEnvironment ResolutionEnvironment resolutionEnvironment) {
        return dataLoaderService
                .loadRelation(resolutionEnvironment, MessagePayloadBatchLoader.class, messageDto)
                .thenApply(payloadDtos -> updatePayloadDtos(messageDto, payloadDtos));
    }

    @GraphQLQuery
    public CompletableFuture<Long> manualActionCount(
            @GraphQLContext MessageDto messageDto, @GraphQLEnvironment ResolutionEnvironment resolutionEnvironment) {
        return dataLoaderService.load(
                resolutionEnvironment, MessageManualActionCountBatchLoader.class, messageDto, MessageDto::getId);
    }

    private static String getDownloadName(MessageDto messageDto, PayloadDto payloadDto) {
        try {
            return messageDto.getMessageId() + "_" + payloadDto.getPosition()
                    + TikaConfig.getDefaultConfig()
                            .getMimeRepository()
                            .forName(payloadDto.getContentType())
                            .getExtension();
        } catch (MimeTypeException e) {
            LOGGER.error("could not get extension for payload {} in message {}", payloadDto, messageDto, e);
            return payloadDto.getName();
        }
    }

    private static Set<PayloadDto> updatePayloadDtos(MessageDto messageDto, Set<PayloadDto> payloadDtos) {
        if (CollectionUtils.isNotEmpty(payloadDtos)) {
            for (PayloadDto payloadDto : payloadDtos) {
                payloadDto.setDownloadName(getDownloadName(messageDto, payloadDto));
            }
        }
        return payloadDtos;
    }

    private static Set<MetaDataDto> updateMetaDatas(MessageDto messageDto, Set<MetaDataDto> metaDataDtos) {
        if (CollectionUtils.isNotEmpty(metaDataDtos)) {
            for (MetaDataDto metaDataDto : metaDataDtos) {
                metaDataDto.setParentMessageId(messageDto.getMessageId());
            }
            SortedSet<MetaDataDto> sorted = new TreeSet<>(MetaDataDtoComparator.INSTANCE);
            sorted.addAll(metaDataDtos);
            metaDataDtos = sorted;
        }
        return metaDataDtos;
    }
}
