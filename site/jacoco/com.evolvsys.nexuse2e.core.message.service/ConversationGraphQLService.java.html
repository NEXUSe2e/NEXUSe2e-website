package com.evolvsys.nexuse2e.core.message.service;

import com.evolvsys.app.core.graphql.annotation.GraphQLService;
import com.evolvsys.app.core.graphql.dataloader.DataLoaderService;
import com.evolvsys.nexuse2e.core.action.model.ActionDto;
import com.evolvsys.nexuse2e.core.action.service.ActionBatchLoader;
import com.evolvsys.nexuse2e.core.choreography.model.ChoreographyDto;
import com.evolvsys.nexuse2e.core.choreography.service.ChoreographyBatchLoader;
import com.evolvsys.nexuse2e.core.message.ConversationFilter;
import com.evolvsys.nexuse2e.core.message.model.ConversationDto;
import com.evolvsys.nexuse2e.core.message.model.ConversationPartnerDto;
import com.evolvsys.nexuse2e.core.message.model.MessageDto;
import com.evolvsys.nexuse2e.core.partner.model.PartnerDto;
import com.evolvsys.nexuse2e.core.partner.service.PartnerBatchLoader;
import io.leangen.graphql.annotations.GraphQLContext;
import io.leangen.graphql.annotations.GraphQLEnvironment;
import io.leangen.graphql.annotations.GraphQLNonNull;
import io.leangen.graphql.annotations.GraphQLQuery;
import io.leangen.graphql.execution.ResolutionEnvironment;
import java.util.Optional;
import java.util.Set;
import java.util.concurrent.CompletableFuture;
import java.util.function.Function;
import lombok.RequiredArgsConstructor;
import org.springframework.data.domain.Page;
import org.springframework.data.domain.Pageable;
import org.springframework.lang.Nullable;

/**
 * @author SvenBarden
 * @since 02.02.2024
 */
@GraphQLService
@RequiredArgsConstructor
public class ConversationGraphQLService {
    private final ConversationService conversationService;
    private final DataLoaderService dataLoaderService;

    @GraphQLQuery(name = "conversation_list", description = "List all conversations")
    @GraphQLNonNull
    public Page<@GraphQLNonNull ConversationDto> list(
            Pageable pageable, @Nullable ConversationFilter conversationFilter) {
        return conversationService.list(pageable, conversationFilter);
    }

    @GraphQLQuery(name = "conversation_getById", description = "Gets a conversation")
    public Optional<ConversationDto> getById(long conversationId) {
        return conversationService.getById(conversationId);
    }

    @GraphQLQuery(name = "conversation_countAll")
    public long countAll() {
        return conversationService.countAll();
    }

    @GraphQLQuery
    public CompletableFuture<ChoreographyDto> choreography(
            @GraphQLContext ConversationDto conversationDto,
            @GraphQLEnvironment ResolutionEnvironment resolutionEnvironment) {
        return dataLoaderService.load(
                resolutionEnvironment,
                ChoreographyBatchLoader.class,
                conversationDto,
                "choreographyId",
                ConversationDto::getChoreographyId);
    }

    @GraphQLQuery
    public CompletableFuture<ActionDto> currentAction(
            @GraphQLContext ConversationDto conversationDto,
            @GraphQLEnvironment ResolutionEnvironment resolutionEnvironment) {
        return dataLoaderService.load(
                resolutionEnvironment,
                ActionBatchLoader.class,
                conversationDto,
                "currentActionId",
                ConversationDto::getCurrentActionId);
    }

    @GraphQLQuery
    public CompletableFuture<Set<@GraphQLNonNull MessageDto>> messages(
            @GraphQLContext ConversationDto conversationDto,
            @GraphQLEnvironment ResolutionEnvironment resolutionEnvironment) {
        return dataLoaderService.load(
                resolutionEnvironment, ConversationMessageBatchLoader.class, conversationDto, ConversationDto::getId);
    }

    @GraphQLQuery
    public CompletableFuture<PartnerDto> lastSourcePartner(
            @GraphQLContext ConversationDto conversationDto,
            @GraphQLEnvironment ResolutionEnvironment resolutionEnvironment) {
        return dataLoaderService.load(
                resolutionEnvironment,
                PartnerBatchLoader.class,
                conversationDto,
                "lastSourcePartnerId",
                ConversationDto::getLastSourcePartnerId);
    }

    @GraphQLQuery
    public CompletableFuture<PartnerDto> lastTargetPartner(
            @GraphQLContext ConversationDto conversationDto,
            @GraphQLEnvironment ResolutionEnvironment resolutionEnvironment) {
        return dataLoaderService.load(
                resolutionEnvironment,
                PartnerBatchLoader.class,
                conversationDto,
                "lastTargetPartnerId",
                ConversationDto::getLastTargetPartnerId);
    }

    @GraphQLQuery
    public CompletableFuture<Set<@GraphQLNonNull ConversationPartnerDto>> partners(
            @GraphQLContext ConversationDto conversationDto,
            @GraphQLEnvironment ResolutionEnvironment resolutionEnvironment) {
        return dataLoaderService.load(
                resolutionEnvironment, ConversationPartnersBatchLoader.class, conversationDto, Function.identity());
    }

    @GraphQLQuery
    public CompletableFuture<PartnerDto> partner(
            @GraphQLContext ConversationPartnerDto conversationPartnerDto,
            @GraphQLEnvironment ResolutionEnvironment resolutionEnvironment) {
        return dataLoaderService.load(
                resolutionEnvironment,
                PartnerBatchLoader.class,
                conversationPartnerDto,
                ConversationPartnerDto::getPartnerId);
    }
}
