package com.evolvsys.nexuse2e.core.message.service;

import com.evolvsys.app.core.auth.service.CurrentUserService;
import com.evolvsys.app.core.service.BaseService;
import com.evolvsys.app.core.service.ListService;
import com.evolvsys.nexuse2e.core.message.ConversationFilter;
import com.evolvsys.nexuse2e.core.message.mapper.ConversationMapper;
import com.evolvsys.nexuse2e.core.message.model.ConversationDto;
import com.evolvsys.nexuse2e.core.message.repository.ConversationRepository;
import com.evolvsys.nexuse2e.core.model.entity.ConversationEntity;
import com.evolvsys.nexuse2e.core.settings.service.ApplicationSettingsService;
import org.springframework.data.domain.Page;
import org.springframework.data.domain.Pageable;
import org.springframework.lang.Nullable;
import org.springframework.stereotype.Service;

/**
 * @author Florian Drees
 * @since 05.03.2024
 */
@Service
public class ConversationService extends BaseService<ConversationDto, ConversationEntity, ConversationRepository> {
    private final ListService listService;
    private final ApplicationSettingsService applicationSettingsService;

    public ConversationService(
            ConversationMapper mapper,
            ConversationRepository repository,
            CurrentUserService currentUserService,
            ListService listService,
            ApplicationSettingsService applicationSettingsService) {
        super(mapper, repository, currentUserService);
        this.listService = listService;
        this.applicationSettingsService = applicationSettingsService;
    }

    @Override
    protected ConversationEntity createEntity() {
        return new ConversationEntity();
    }

    public Page<ConversationDto> list(Pageable pageable, @Nullable ConversationFilter conversationFilter) {
        return listService
                .list(pageable, conversationFilter, getDtoType(), (pageable1, filter, predicate) -> getRepository()
                        .findAllToRead(
                                pageable1,
                                filter,
                                applicationSettingsService
                                        .getSystemSettingsOrDefault()
                                        .getTimeZone(),
                                predicate))
                .map(this::mapEntityToDto);
    }

    public long countAll() {
        return getRepository().count();
    }
}
