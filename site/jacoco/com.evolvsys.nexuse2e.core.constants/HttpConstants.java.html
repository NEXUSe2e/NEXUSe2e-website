package com.evolvsys.nexuse2e.core.constants;

import java.util.List;
import java.util.Map;
import org.apache.tomcat.util.net.SSLSupport;
import org.springframework.http.HttpHeaders;
import org.springframework.http.MediaType;

/**
 * @author SvenBarden
 * @since 23.10.2023
 */
public interface HttpConstants {
    Constant<String> ENCODING = Constant.ofString("encoding");
    Constant<Long> CONTENT_SIZE = Constant.ofLong("contentSize");
    Constant<Map<String, List<String>>> ORIGINAL_HEADER = Constant.ofType("header", Map.class);
    Constant<HttpHeaders> ADDITIONAL_HEADERS = Constant.ofType("additionalHeaders", List.class);
    Constant<MediaType> CONTENT_TYPE = Constant.ofType("Content-Type", MediaType.class);
    Constant<String> CONTENT_ID = Constant.ofString("Content-ID");
    Constant<String> PROTOCOL_VERSION = Constant.ofString("protocol-version");
    Constant<String> CIPHERS = Constant.ofString("ciphers");
    Constant<String> CLIENT_IP = Constant.ofString("client-ip");
    Constant<String> CLIENT_FORWARDED_IP = Constant.ofString("client-forwarded-ip");

    String PREFIX_RECEIVED_REQUEST_PARAMETER = "received_parameter";
    String PREFIX_OUTBOUND_REQUEST_PARAMETER = "outbound_parameter";
    String PREFIX_REQUEST_HEADER = "header";
    String PREFIX_QUERY_PARAMETERS = "query_parameters";

    String REQUEST_PROTOCOL_VERSION_KEY = SSLSupport.PROTOCOL_VERSION_KEY;
    String REQUEST_CIPHER_SUITE_KEY = SSLSupport.CIPHER_SUITE_KEY;
    String REQUEST_CERTIFICATE_KEY = SSLSupport.CERTIFICATE_KEY;
}
