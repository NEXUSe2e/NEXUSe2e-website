package com.evolvsys.nexuse2e.core.constants;

import lombok.Value;

/**
 * @author SvenBarden
 * @since 16.10.2023
 */
@Value(staticConstructor = "of")
public class Constant<T> {
    String name;
    Class<T> type;

    @SuppressWarnings("unchecked")
    public static <TP> Constant<TP> ofType(String name, Class<?> type) {
        return (Constant<TP>) of(name, type);
    }

    public static Constant<String> ofString(String name) {
        return of(name, String.class);
    }

    public static Constant<Integer> ofInt(String name) {
        return of(name, Integer.class);
    }

    public static Constant<Long> ofLong(String name) {
        return of(name, Long.class);
    }

    public static Constant<Boolean> ofBoolean(String name) {
        return of(name, Boolean.class);
    }
}
