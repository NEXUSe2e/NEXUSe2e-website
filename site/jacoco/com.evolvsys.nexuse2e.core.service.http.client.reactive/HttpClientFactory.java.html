package com.evolvsys.nexuse2e.core.service.http.client.reactive;

import com.evolvsys.nexuse2e.core.exception.NexusException;
import com.evolvsys.nexuse2e.core.service.http.WebClientContext;
import io.netty.channel.ChannelOption;
import io.netty.handler.ssl.SslContext;
import java.security.KeyStoreException;
import java.security.NoSuchAlgorithmException;
import java.security.UnrecoverableKeyException;
import javax.net.ssl.SSLEngine;
import javax.net.ssl.SSLException;
import javax.net.ssl.SSLParameters;
import lombok.experimental.UtilityClass;
import reactor.netty.http.client.HttpClient;
import reactor.netty.tcp.SslProvider;

/**
 * @author SvenBarden
 * @since 16.02.2024
 */
@UtilityClass
public class HttpClientFactory {

    private static void sslProviderBuilder(
            SslProvider.SslContextSpec sslContextSpec, SslContext sslContext, WebClientContext context) {
        SslProvider.Builder builder = sslContextSpec.sslContext(sslContext);
        if (context.isDisableHostNameVerification()) {
            builder.handlerConfigurator(handler -> {
                SSLEngine engine = handler.engine();
                SSLParameters sslParameters = engine.getSSLParameters();
                SSLParameters params = new SSLParameters(sslParameters.getCipherSuites(), sslParameters.getProtocols());
                // With java update null value is no longer sufficient to skip host name verification
                params.setEndpointIdentificationAlgorithm("");
                engine.setSSLParameters(params);
            });
        }
    }

    public static HttpClient createHttpClient(WebClientContext context)
            throws SSLException, NexusException, UnrecoverableKeyException, NoSuchAlgorithmException,
                    KeyStoreException {
        HttpClient httpClient = HttpClient.create();

        if (context.isSsl()) {
            SslContext sslContext = SslContextFactory.createSSLContext(context);
            httpClient = httpClient.secure(sslContextSpec -> sslProviderBuilder(sslContextSpec, sslContext, context));
        }

        if (context.getResponseTimeout() != null) {
            httpClient.responseTimeout(context.getResponseTimeout());
        }
        if (context.getConnectionTimeout() != null) {
            httpClient.option(ChannelOption.CONNECT_TIMEOUT_MILLIS, (int)
                    context.getConnectionTimeout().toMillis());
        }

        return httpClient;
    }
}
