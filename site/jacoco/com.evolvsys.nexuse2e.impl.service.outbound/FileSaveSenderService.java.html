package com.evolvsys.nexuse2e.impl.service.outbound;

import static com.evolvsys.nexuse2e.impl.util.FileSaveUtils.persistPayloads;

import com.evolvsys.nexuse2e.core.exception.NexusError;
import com.evolvsys.nexuse2e.core.exception.NexusException;
import com.evolvsys.nexuse2e.core.message.context.ExecutionContext;
import com.evolvsys.nexuse2e.core.message.context.MessageContext;
import com.evolvsys.nexuse2e.core.parameter.ParameterDefinition;
import com.evolvsys.nexuse2e.core.service.NexusOutflowService;
import com.evolvsys.nexuse2e.core.utils.PatternReplaceUtils;
import com.evolvsys.nexuse2e.impl.util.FileSaveAware;
import com.evolvsys.nexuse2e.internal.Message;
import java.io.IOException;
import java.util.List;
import lombok.extern.slf4j.Slf4j;
import org.apache.tika.utils.StringUtils;

/**
 * @author GuidoEsch
 * @since 02.02.2024
 */
@Slf4j
public class FileSaveSenderService extends NexusOutflowService implements FileSaveAware {

    @Override
    public MessageContext executeInternal(ExecutionContext executionContext) throws NexusException {
        LOGGER.trace("running file save sender.");
        MessageContext messageContext = executionContext.getMessageContext();
        Message message = executionContext.getMessage();

        if (StringUtils.isBlank(parameters.value(PARAM_TARGET_DIRECTORY))) {
            throw new IllegalArgumentException("parameter 'Directory' must be provided to the sender service!");
        }

        String directoryString = parameters.value(PARAM_TARGET_DIRECTORY);
        LOGGER.trace("replacing directory pattern {}", directoryString);
        directoryString = PatternReplaceUtils.replaceContextTokens(directoryString, executionContext);
        directoryString = directoryString.replace('\\', '/');

        try {
            String filePattern = parameters.value(PARAM_TARGET_FILENAME);
            persistPayloads(
                    executionContext, message, directoryString, filePattern, parameters.value(PARAM_CREATE_DIRECTORY));

        } catch (IOException e) {
            throw new NexusException(
                    NexusError.builder()
                            .message("Error while writing file to disk")
                            .extension("reason", e.getMessage())
                            .build(),
                    e);
        }
        return messageContext;
    }

    @Override
    public List<ParameterDefinition> parameterDefinitions() {
        return List.of(getTargetDirectory(), getTargetFilename(), getCreateMissingDirectories());
    }
}
