package com.evolvsys.nexuse2e.impl.service.outbound;

import com.evolvsys.nexuse2e.core.component.SupportService;
import com.evolvsys.nexuse2e.core.database.DatabaseSenderContext;
import com.evolvsys.nexuse2e.core.database.DatabaseUtils;
import com.evolvsys.nexuse2e.core.database.DynamicDatabaseValue;
import com.evolvsys.nexuse2e.core.exception.NexusError;
import com.evolvsys.nexuse2e.core.exception.NexusException;
import com.evolvsys.nexuse2e.core.message.context.ExecutionContext;
import com.evolvsys.nexuse2e.core.message.context.MessageContext;
import com.evolvsys.nexuse2e.core.parameter.ParameterDefinition;
import com.evolvsys.nexuse2e.core.parameter.ParameterType;
import com.evolvsys.nexuse2e.core.service.NexusOutflowService;
import com.evolvsys.nexuse2e.impl.service.support.DatabaseConnectionSupportService;
import java.util.Arrays;
import java.util.List;
import java.util.Map;
import lombok.extern.slf4j.Slf4j;
import org.springframework.jdbc.core.namedparam.MapSqlParameterSource;
import org.springframework.jdbc.core.namedparam.NamedParameterJdbcTemplate;

/**
 * @author Florian Drees
 * @since 19.06.2024
 */
@Slf4j
public class DatabaseSenderService extends NexusOutflowService {
    private static final String PARAMETER_DATABASE_CONNECTION_SUPPORT_SERVICE = "database_connection_support_service";

    public static final String CONTEXT_KEY = "database_sender_context_key";
    public static final String CONTEXT_CATEGORY = "database_sender_context_category";

    @Override
    public MessageContext executeInternal(ExecutionContext executionContext) throws NexusException {
        LOGGER.trace("Running database sender");

        MessageContext messageContext = executionContext.getMessageContext();

        DatabaseConnectionSupportService databaseConnectionSupportService = getDatabaseConnectionSupportService();

        NamedParameterJdbcTemplate namedParameterJdbcTemplate =
                databaseConnectionSupportService.getNamedParameterJdbcTemplate();

        DatabaseSenderContext databaseSenderContext =
                executionContext.getMetaData().getFirstOrThrows(CONTEXT_CATEGORY, CONTEXT_KEY);

        MapSqlParameterSource mapSqlParameterSource = new MapSqlParameterSource();

        for (Map.Entry<String, Object> entry :
                databaseSenderContext.getKeyValues().entrySet()) {
            if (entry.getValue() instanceof DynamicDatabaseValue<?> dynamicDatabaseValue) {
                mapSqlParameterSource.addValue(
                        entry.getKey(), dynamicDatabaseValue.getValue(namedParameterJdbcTemplate, executionContext));
            } else {
                mapSqlParameterSource.addValue(entry.getKey(), entry.getValue());
            }
        }

        List<String> parameterNames = Arrays.asList(mapSqlParameterSource.getParameterNames());

        final String sql = DatabaseUtils.buildNamedSqlInsert(
                namedParameterJdbcTemplate, databaseSenderContext.getTableName(), parameterNames);
        LOGGER.debug("Built sql for insert operation: {}", sql);

        try {
            int affectedRows = namedParameterJdbcTemplate.update(sql, mapSqlParameterSource);
            LOGGER.debug("Inserted '{}' rows into table '{}'", affectedRows, databaseSenderContext.getTableName());
        } catch (Exception e) {
            throw new NexusException(
                    NexusError.builder()
                            .message("An unexpected error occurred while trying to save payloads")
                            .extension("sql", sql)
                            .build(),
                    e);
        }
        return messageContext;
    }

    private DatabaseConnectionSupportService getDatabaseConnectionSupportService() throws NexusException {
        SupportService supportService = parameters.value(PARAMETER_DATABASE_CONNECTION_SUPPORT_SERVICE);
        if (supportService == null) {
            throw new NexusException(NexusError.builder()
                    .message("A configured support service is required for further processing.")
                    .extension("type", DatabaseConnectionSupportService.class.getSimpleName())
                    .build());
        }
        if (supportService instanceof DatabaseConnectionSupportService databaseConnectionSupportService) {
            LOGGER.debug("Using database connection service '{}'", databaseConnectionSupportService.getServiceId());
            return databaseConnectionSupportService;
        } else {
            throw new NexusException(NexusError.builder()
                    .message("The configured support service can not be used for this service")
                    .extension("type", DatabaseConnectionSupportService.class.getSimpleName())
                    .build());
        }
    }

    @Override
    public List<ParameterDefinition> parameterDefinitions() {
        return List.of(ParameterDefinition.builder()
                .name(PARAMETER_DATABASE_CONNECTION_SUPPORT_SERVICE)
                .displayName("Database connection support service")
                .description("The database connection support service to provide the connection")
                .type(ParameterType.SUPPORT_SERVICE)
                .required(true)
                .build());
    }
}
