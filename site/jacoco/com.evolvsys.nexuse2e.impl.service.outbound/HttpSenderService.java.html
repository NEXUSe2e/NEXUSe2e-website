package com.evolvsys.nexuse2e.impl.service.outbound;

import com.evolvsys.nexuse2e.core.exception.NexusException;
import com.evolvsys.nexuse2e.core.message.context.ExecutionContext;
import com.evolvsys.nexuse2e.core.message.context.MessageContext;
import com.evolvsys.nexuse2e.core.service.http.WebClientContext;
import com.evolvsys.nexuse2e.core.service.http.client.rest.NexusRestClientFactory;
import lombok.extern.slf4j.Slf4j;
import org.springframework.http.HttpMethod;
import org.springframework.util.MultiValueMap;
import org.springframework.util.StreamUtils;
import org.springframework.web.client.RestClient;

@Slf4j
public class HttpSenderService extends BaseHttpSenderService {

    @Override
    public MessageContext executeInternal(ExecutionContext executionContext) throws NexusException {
        LOGGER.trace("running http sender");

        validateParameters();

        NexusRestClientFactory restClientFactory =
                executionContext.getBeanFactory().getBean(NexusRestClientFactory.class);

        WebClientContext webClientContext =
                buildWebClientContext(executionContext, restClientFactory::createContextBuilder);
        RestClient.Builder restClientBuilder = restClientFactory.createClientBuilder(webClientContext);

        MultiValueMap<String, String> requestParameters = getRequestParametersFromMetadata(executionContext);

        addAuthenticationSupport(httpAuthenticationSupportService ->
                httpAuthenticationSupportService.addAuthentication(restClientBuilder));

        RestClient.RequestBodySpec requestBodySpec = restClientBuilder
                .baseUrl((String) parameters.value(PARAM_URL))
                .build()
                .method(HttpMethod.valueOf(parameters.value(PARAM_METHOD)))
                .uri(uriBuilder -> uriBuilder.queryParams(requestParameters).build());
        setContentType(requestBodySpec::contentType, executionContext);
        setAdditionalHeaders(requestBodySpec::headers, executionContext);
        addBody(
                requestBodySpec::body,
                requestBodySpec::header,
                inputStream -> requestBodySpec.body(outputStream -> StreamUtils.copy(inputStream, outputStream)),
                requestBodySpec::body,
                executionContext);

        return handleResponse(() -> requestBodySpec.retrieve().toEntity(String.class), executionContext);
    }
}
