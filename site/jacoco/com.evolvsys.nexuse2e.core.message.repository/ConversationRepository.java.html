package com.evolvsys.nexuse2e.core.message.repository;

import com.evolvsys.nexuse2e.core.message.ConversationFilter;
import com.evolvsys.nexuse2e.core.model.entity.ConversationEntity;
import com.evolvsys.nexuse2e.core.model.entity.QConversationEntity;
import com.querydsl.core.types.dsl.BooleanExpression;
import java.util.Optional;
import org.springframework.data.domain.Page;
import org.springframework.data.domain.Pageable;
import org.springframework.data.jpa.repository.JpaRepository;
import org.springframework.data.querydsl.QuerydslPredicateExecutor;
import org.springframework.lang.Nullable;

/**
 * @author SvenBarden
 * @since 12.10.2023
 */
public interface ConversationRepository
        extends JpaRepository<ConversationEntity, Long>, QuerydslPredicateExecutor<ConversationEntity> {
    Optional<ConversationEntity> findByConversationId(String conversationId);

    default Page<ConversationEntity> findAllToRead(
            Pageable pageable,
            @Nullable ConversationFilter conversationFilter,
            String timeZone,
            BooleanExpression securityPredicate) {
        return findAll(
                securityPredicate.and(QConversationEntity.conversationEntity.filter(conversationFilter, timeZone)),
                pageable);
    }

    boolean existsByChoreography_Id(Long id);
}
