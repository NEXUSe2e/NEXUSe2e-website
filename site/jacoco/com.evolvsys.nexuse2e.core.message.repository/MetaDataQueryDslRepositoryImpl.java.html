package com.evolvsys.nexuse2e.core.message.repository;

import com.evolvsys.nexuse2e.core.message.MetaDataLinkType;
import com.evolvsys.nexuse2e.core.model.entity.MetaDataEntity;
import com.evolvsys.nexuse2e.core.model.entity.QChoreographyEntity;
import com.evolvsys.nexuse2e.core.model.entity.QConversationEntity;
import com.evolvsys.nexuse2e.core.model.entity.QMessageEntity;
import com.evolvsys.nexuse2e.core.model.entity.QMetaDataEntity;
import com.querydsl.core.BooleanBuilder;
import com.querydsl.core.group.GroupBy;
import com.querydsl.core.types.Predicate;
import com.querydsl.jpa.JPAExpressions;
import com.querydsl.jpa.JPQLQuery;
import com.querydsl.jpa.JPQLQueryFactory;
import com.querydsl.jpa.JPQLSubQuery;
import java.util.List;
import java.util.Map;
import java.util.Optional;
import lombok.RequiredArgsConstructor;
import lombok.extern.slf4j.Slf4j;
import org.apache.commons.lang3.StringUtils;
import org.springframework.lang.NonNull;
import org.springframework.lang.Nullable;
import org.springframework.stereotype.Repository;

/**
 * QueryDsl-Repository for the message metadata.
 *
 * @author Julian Drees
 * @since 29.08.2025
 */
@Slf4j
@Repository
@RequiredArgsConstructor
class MetaDataQueryDslRepositoryImpl implements MetaDataQueryDslRepository {
    private final JPQLQueryFactory jpqlQueryFactory;

    @NonNull
    @Override
    public Map<String, List<MetaDataEntity>> getDuplicatesByKey(
            @NonNull Map<String, List<Object>> metaData,
            @NonNull String category,
            @Nullable String choreographyId,
            @NonNull MetaDataLinkType linkType) {
        QMessageEntity qMessageEntity = QMessageEntity.messageEntity;
        QMetaDataEntity qMetaDataEntity = QMetaDataEntity.metaDataEntity;

        JPQLSubQuery<Long> subQuery = createSubQuery(metaData, linkType, category, metaData.size());
        JPQLQuery<MetaDataEntity> query = jpqlQueryFactory
                .select(qMetaDataEntity)
                .from(qMessageEntity)
                .innerJoin(qMessageEntity.metaDatas, qMetaDataEntity)
                .where(
                        qMessageEntity.id.in(subQuery),
                        qMetaDataEntity.category.eq(category),
                        createPredicate(metaData, qMetaDataEntity));

        if (StringUtils.isNotBlank(choreographyId)) {
            LOGGER.debug("Searching for duplicates with choreography '{}'.", choreographyId);
            QConversationEntity qConversationEntity = QConversationEntity.conversationEntity;
            QChoreographyEntity qChoreographyEntity = QChoreographyEntity.choreographyEntity;
            query = query.innerJoin(qMessageEntity.conversation, qConversationEntity)
                    .innerJoin(qConversationEntity.choreography, qChoreographyEntity)
                    .on(qChoreographyEntity.choreographyId.eq(choreographyId));
        }
        return query.transform(GroupBy.groupBy(qMetaDataEntity.name).as(GroupBy.list(qMetaDataEntity)));
    }

    @NonNull
    private JPQLSubQuery<Long> createSubQuery(
            @NonNull Map<String, List<Object>> metaData,
            @NonNull MetaDataLinkType linkType,
            @Nullable String category,
            long keys) {
        QMessageEntity qMessageEntity = new QMessageEntity("innerMessageEntity");
        QMetaDataEntity qMetaDataEntity = new QMetaDataEntity("innerMetaDataEntity");
        JPQLSubQuery<Long> subQuery = JPAExpressions.select(qMessageEntity.id)
                .from(qMessageEntity)
                .innerJoin(qMessageEntity.metaDatas, qMetaDataEntity)
                .where(createPredicate(metaData, qMetaDataEntity), qMetaDataEntity.category.eq(category))
                .groupBy(qMessageEntity.id);
        if (linkType == MetaDataLinkType.AND) {
            subQuery = subQuery.having(qMetaDataEntity.name.countDistinct().eq(keys));
        }
        return subQuery;
    }

    @NonNull
    private Predicate createPredicate(
            @NonNull Map<String, List<Object>> metaData, @NonNull QMetaDataEntity qMetaDataEntity) {
        BooleanBuilder builder = new BooleanBuilder();
        for (Map.Entry<String, List<Object>> entry : metaData.entrySet()) {
            builder.or(qMetaDataEntity.filterByValues(entry.getKey(), entry.getValue()));
        }
        return Optional.ofNullable(builder.getValue())
                .orElseThrow(() -> new IllegalArgumentException(
                        "Creating this predicate without metadata entries is not allowed."));
    }
}
