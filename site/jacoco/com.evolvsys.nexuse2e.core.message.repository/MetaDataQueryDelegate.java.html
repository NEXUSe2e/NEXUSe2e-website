package com.evolvsys.nexuse2e.core.message.repository;

import com.evolvsys.nexuse2e.core.message.MetaDataFilter;
import com.evolvsys.nexuse2e.core.model.entity.MetaDataEntity;
import com.evolvsys.nexuse2e.core.model.entity.QMetaDataEntity;
import com.querydsl.core.BooleanBuilder;
import com.querydsl.core.annotations.QueryDelegate;
import com.querydsl.core.types.Predicate;
import java.math.BigDecimal;
import java.time.ZonedDateTime;
import java.time.format.DateTimeParseException;
import java.util.List;
import java.util.stream.Stream;
import one.util.streamex.StreamEx;
import org.apache.commons.collections4.CollectionUtils;
import org.apache.commons.lang3.BooleanUtils;
import org.apache.commons.lang3.StringUtils;
import org.springframework.lang.NonNull;

/**
 * @author Florian Drees
 * @since 25.02.2024
 */
public final class MetaDataQueryDelegate {
    private MetaDataQueryDelegate() {}

    @QueryDelegate(MetaDataEntity.class)
    public static Predicate filter(QMetaDataEntity qMetaDataEntity, MetaDataFilter metaDataFilter) {
        BooleanBuilder builder = new BooleanBuilder();
        if (StringUtils.isNoneBlank(metaDataFilter.getKey(), metaDataFilter.getValue())) {
            builder.and(
                    qMetaDataEntity.name.eq(metaDataFilter.getKey()).and(filterValue(qMetaDataEntity, metaDataFilter)));
        } else if (StringUtils.isNotEmpty(metaDataFilter.getKey())) {
            builder.and(qMetaDataEntity.name.eq(metaDataFilter.getKey()));
        } else if (StringUtils.isNotEmpty(metaDataFilter.getValue())) {
            builder.and(filterValue(qMetaDataEntity, metaDataFilter));
        }
        return builder.getValue();
    }

    private static Predicate filterValue(QMetaDataEntity qMetaDataEntity, MetaDataFilter metaDataFilter) {
        BooleanBuilder builder = new BooleanBuilder();
        builder.or(qMetaDataEntity.stringValue.eq(metaDataFilter.getValue()));

        try {
            builder.or(qMetaDataEntity.longValue.eq(Long.valueOf(metaDataFilter.getValue())));
        } catch (NumberFormatException e) {
            // do nothing and ignore result. Filtering of field longValue is ignored
        }

        try {
            builder.or(qMetaDataEntity.dateTimeValue.eq(ZonedDateTime.parse(metaDataFilter.getValue())));
        } catch (DateTimeParseException e) {
            // do nothing and ignore result. Filtering of field dateTimeValue is ignored
        }

        Boolean booleanValue = BooleanUtils.toBooleanObject(metaDataFilter.getValue());
        if (booleanValue != null) {
            builder.or(qMetaDataEntity.booleanValue.eq(booleanValue));
        }
        return builder.getValue();
    }

    @QueryDelegate(MetaDataEntity.class)
    public static Predicate filterByValues(
            QMetaDataEntity qMetaDataEntity, @NonNull String metaDataKey, @NonNull List<Object> values) {
        BooleanBuilder builder = new BooleanBuilder();

        List<String> stringValues = getTypedValues(values, String.class);
        if (CollectionUtils.isNotEmpty(stringValues)) {
            builder.or(qMetaDataEntity.stringValue.in(stringValues));
        }

        List<Long> longValues = getLongValues(values);
        if (CollectionUtils.isNotEmpty(longValues)) {
            builder.or(qMetaDataEntity.longValue.in(longValues));
        }

        List<Boolean> booleanValues = getTypedValues(values, Boolean.class);
        if (CollectionUtils.isNotEmpty(booleanValues)) {
            builder.or(qMetaDataEntity.booleanValue.in(booleanValues));
        }

        List<BigDecimal> bigDecimalValues = getTypedValues(values, BigDecimal.class);
        if (CollectionUtils.isNotEmpty(bigDecimalValues)) {
            builder.or(qMetaDataEntity.bigDecimalValue.in(bigDecimalValues));
        }

        List<ZonedDateTime> dateTimeValues = getTypedValues(values, ZonedDateTime.class);
        if (CollectionUtils.isNotEmpty(dateTimeValues)) {
            builder.or(qMetaDataEntity.dateTimeValue.in(dateTimeValues));
        }

        BooleanBuilder wrapper = new BooleanBuilder(qMetaDataEntity.name.eq(metaDataKey));
        if (builder.hasValue()) {
            wrapper.and(builder.getValue());
        }
        return wrapper;
    }

    @NonNull
    private static <T> List<T> getTypedValues(@NonNull List<Object> values, @NonNull Class<T> target) {
        return CollectionUtils.emptyIfNull(values).stream()
                .filter(object -> target.isAssignableFrom(object.getClass()))
                .map(target::cast)
                .toList();
    }

    @NonNull
    private static List<Long> getLongValues(@NonNull List<Object> values) {
        Stream<Long> longValues = getTypedValues(values, Long.class).stream();
        return StreamEx.of(longValues)
                .append(getTypedValues(values, Integer.class).stream()
                        .mapToLong(Long::valueOf)
                        .boxed())
                .toList();
    }
}
