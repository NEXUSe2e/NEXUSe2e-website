package com.evolvsys.nexuse2e.core.message.repository;

import com.evolvsys.nexuse2e.core.message.MessageFilter;
import com.evolvsys.nexuse2e.core.message.MetaDataFilter;
import com.evolvsys.nexuse2e.core.model.entity.MessageEntity;
import com.evolvsys.nexuse2e.core.model.entity.QMessageEntity;
import com.evolvsys.nexuse2e.core.model.entity.QMetaDataEntity;
import com.evolvsys.nexuse2e.core.utils.DateRangeFilterUtils;
import com.querydsl.core.BooleanBuilder;
import com.querydsl.core.annotations.QueryDelegate;
import com.querydsl.core.types.Expression;
import com.querydsl.core.types.Predicate;
import com.querydsl.core.types.dsl.DateTimePath;
import com.querydsl.core.types.dsl.Expressions;
import com.querydsl.jpa.JPAExpressions;
import java.time.ZonedDateTime;
import java.util.List;
import org.apache.commons.collections4.CollectionUtils;
import org.apache.commons.lang3.StringUtils;
import org.springframework.lang.Nullable;

/**
 * @author Florian Drees
 * @since 28.12.2023
 */
public final class MessageQueryDelegate {
    private MessageQueryDelegate() {}

    @QueryDelegate(MessageEntity.class)
    public static Predicate filter(
            QMessageEntity qMessageEntity, @Nullable MessageFilter messageFilter, String timeZone) {
        BooleanBuilder builder = new BooleanBuilder();
        if (messageFilter != null) {
            if (StringUtils.isNotBlank(messageFilter.getMessageId())) {
                builder.and(qMessageEntity.messageId.contains(messageFilter.getMessageId()));
            }
            if (CollectionUtils.isNotEmpty(messageFilter.getTypes())) {
                builder.and(qMessageEntity.type.in(messageFilter.getTypes()));
            }
            if (CollectionUtils.isNotEmpty(messageFilter.getStatuses())) {
                builder.and(qMessageEntity.status.in(messageFilter.getStatuses()));
            }
            // filter fields of the conversation, not the message
            if (StringUtils.isNotBlank(messageFilter.getConversationId())) {
                builder.and(qMessageEntity.conversation.conversationId.contains(messageFilter.getConversationId()));
            }
            if (CollectionUtils.isNotEmpty(messageFilter.getChoreographyIds())) {
                builder.and(qMessageEntity.conversation.choreography.id.in(messageFilter.getChoreographyIds()));
            }
            if (StringUtils.isNotBlank(messageFilter.getSourcePartnerId())) {
                builder.and(qMessageEntity.sourcePartner.partnerId.eq(messageFilter.getSourcePartnerId()));
            }
            if (StringUtils.isNotBlank(messageFilter.getSourcePartnerDescription())) {
                builder.and(
                        qMessageEntity.sourcePartner.description.contains(messageFilter.getSourcePartnerDescription()));
            }
            if (StringUtils.isNotBlank(messageFilter.getTargetPartnerId())) {
                builder.and(qMessageEntity.targetPartner.partnerId.eq(messageFilter.getTargetPartnerId()));
            }
            if (StringUtils.isNotBlank(messageFilter.getTargetPartnerDescription())) {
                builder.and(
                        qMessageEntity.targetPartner.description.contains(messageFilter.getTargetPartnerDescription()));
            }

            // Date where the message has been sent
            filterFrom(builder, qMessageEntity.sent, messageFilter.getSentFrom());
            filterTill(builder, qMessageEntity.sent, messageFilter.getSentTill());
            // Date where the message has been processed
            filterFrom(builder, qMessageEntity.processed, messageFilter.getProcessedFrom());
            filterTill(builder, qMessageEntity.processed, messageFilter.getProcessedTill());
            // Date of the last try to send the message
            filterFrom(builder, qMessageEntity.lastTry, messageFilter.getLastTryFrom());
            filterTill(builder, qMessageEntity.lastTry, messageFilter.getLastTryTill());

            filterFrom(builder, qMessageEntity.finished, messageFilter.getFinishedFrom());
            filterTill(builder, qMessageEntity.finished, messageFilter.getFinishedTill());

            // MetaData filter
            if (CollectionUtils.isNotEmpty(messageFilter.getMetaDataFilters())) {
                List<MetaDataFilter> metaDataFilters = messageFilter.getMetaDataFilters();
                filterMetaData(qMessageEntity, metaDataFilters, builder);
            }

            if (CollectionUtils.isNotEmpty(messageFilter.getDirections())) {
                builder.and(qMessageEntity.direction.in(messageFilter.getDirections()));
            }

            if (messageFilter.getMinRetries() != null) {
                builder.and(qMessageEntity.retries.goe(messageFilter.getMinRetries()));
            }
            if (messageFilter.getMaxRetries() != null) {
                builder.and(qMessageEntity.retries.loe(messageFilter.getMaxRetries()));
            }
            Predicate predicate = DateRangeFilterUtils.filter(
                    qMessageEntity.created,
                    timeZone,
                    messageFilter.getCreatedRange(),
                    messageFilter.getCreatedFrom(),
                    messageFilter.getCreatedTill());
            if (predicate != null) {
                builder.and(predicate);
            }

            if (messageFilter.getPartnerId() != null) {
                builder.and(Expressions.anyOf(
                        qMessageEntity.sourcePartner.id.eq(messageFilter.getPartnerId()),
                        qMessageEntity.targetPartner.id.eq(messageFilter.getPartnerId())));
            }
        }
        return builder.getValue();
    }

    private static void filterMetaData(
            Expression<MessageEntity> qMessageEntity, List<MetaDataFilter> metaDataFilters, BooleanBuilder builder) {
        for (int i = 0; i < metaDataFilters.size(); i++) {
            MetaDataFilter metaDataFilter = metaDataFilters.get(i);
            if (StringUtils.isNotBlank(metaDataFilter.getKey()) || StringUtils.isNotBlank(metaDataFilter.getValue())) {
                QMessageEntity qMessageEntity1 = new QMessageEntity("message" + i);
                QMetaDataEntity qMetaDataEntity1 = new QMetaDataEntity("metaData" + i);
                builder.and(JPAExpressions.select(qMessageEntity1.count())
                        .from(qMessageEntity1)
                        .join(qMessageEntity1.metaDatas, qMetaDataEntity1)
                        .on(qMetaDataEntity1.filter(metaDataFilter))
                        .where(qMessageEntity1.eq(qMessageEntity))
                        .exists());
            }
        }
    }

    private static void filterFrom(BooleanBuilder builder, DateTimePath<ZonedDateTime> path, ZonedDateTime from) {
        if (from != null) {
            builder.and(path.goe(from));
        }
    }

    private static void filterTill(BooleanBuilder builder, DateTimePath<ZonedDateTime> path, ZonedDateTime till) {
        if (till != null) {
            builder.and(path.loe(till));
        }
    }
}
