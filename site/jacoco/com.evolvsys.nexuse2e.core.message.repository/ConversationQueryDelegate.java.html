package com.evolvsys.nexuse2e.core.message.repository;

import com.evolvsys.nexuse2e.core.message.ConversationFilter;
import com.evolvsys.nexuse2e.core.model.entity.ConversationEntity;
import com.evolvsys.nexuse2e.core.model.entity.QConversationEntity;
import com.evolvsys.nexuse2e.core.utils.DateRangeFilterUtils;
import com.querydsl.core.BooleanBuilder;
import com.querydsl.core.annotations.QueryDelegate;
import com.querydsl.core.types.Predicate;
import com.querydsl.core.types.dsl.Expressions;
import java.time.ZonedDateTime;
import org.apache.commons.collections4.CollectionUtils;
import org.apache.commons.lang3.StringUtils;
import org.springframework.lang.Nullable;

/**
 * @author Florian Drees
 * @since 05.03.2024
 */
public final class ConversationQueryDelegate {
    private ConversationQueryDelegate() {}

    @QueryDelegate(ConversationEntity.class)
    public static Predicate filter(
            QConversationEntity qConversationEntity, @Nullable ConversationFilter conversationFilter, String timeZone) {
        BooleanBuilder builder = new BooleanBuilder();
        if (conversationFilter != null) {
            if (StringUtils.isNotEmpty(conversationFilter.getConversationId())) {
                builder.and(qConversationEntity.conversationId.contains(conversationFilter.getConversationId()));
            }
            if (CollectionUtils.isNotEmpty(conversationFilter.getChoreographyIds())) {
                builder.and(qConversationEntity.choreography.id.in(conversationFilter.getChoreographyIds()));
            }
            if (conversationFilter.getFinishedFrom() != null) {
                builder.and(qConversationEntity.finished.goe(conversationFilter.getFinishedFrom()));
            }
            if (conversationFilter.getFinishedTill() != null) {
                builder.and(qConversationEntity.finished.loe(conversationFilter.getFinishedTill()));
            }
            if (CollectionUtils.isNotEmpty(conversationFilter.getLastDirections())) {
                builder.and(qConversationEntity.lastDirection.in(conversationFilter.getLastDirections()));
            }
            if (CollectionUtils.isNotEmpty(conversationFilter.getStatuses())) {
                builder.and(qConversationEntity.status.in(conversationFilter.getStatuses()));
            }
            if (StringUtils.isNotBlank(conversationFilter.getLastSourcePartnerId())) {
                builder.and(qConversationEntity.lastSourcePartner.partnerId.contains(
                        conversationFilter.getLastSourcePartnerId()));
            }
            if (StringUtils.isNotBlank(conversationFilter.getLastSourcePartnerDescription())) {
                builder.and(qConversationEntity.lastSourcePartner.description.contains(
                        conversationFilter.getLastSourcePartnerDescription()));
            }
            if (StringUtils.isNotBlank(conversationFilter.getLastTargetPartnerId())) {
                builder.and(qConversationEntity.lastTargetPartner.partnerId.contains(
                        conversationFilter.getLastTargetPartnerId()));
            }
            if (StringUtils.isNotBlank(conversationFilter.getLastTargetPartnerDescription())) {
                builder.and(qConversationEntity.lastTargetPartner.description.contains(
                        conversationFilter.getLastTargetPartnerDescription()));
            }
            if (conversationFilter.getLastUpdatedSince() != null) {
                builder.and(qConversationEntity.lastUpdated.loe(
                        ZonedDateTime.now().minus(conversationFilter.getLastUpdatedSince())));
            }
            Predicate predicate = DateRangeFilterUtils.filter(
                    qConversationEntity.created,
                    timeZone,
                    conversationFilter.getCreatedRange(),
                    conversationFilter.getCreatedFrom(),
                    conversationFilter.getCreatedTill());
            if (predicate != null) {
                builder.and(predicate);
            }
            if (conversationFilter.getLastPartnerId() != null) {
                builder.and(Expressions.anyOf(
                        qConversationEntity.lastSourcePartner.id.eq(conversationFilter.getLastPartnerId()),
                        qConversationEntity.lastTargetPartner.id.eq(conversationFilter.getLastPartnerId())));
            }
        }
        return builder.getValue();
    }
}
