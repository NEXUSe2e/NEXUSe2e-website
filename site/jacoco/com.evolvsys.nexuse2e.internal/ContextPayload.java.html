package com.evolvsys.nexuse2e.internal;

import com.evolvsys.nexuse2e.core.exception.NexusError;
import com.evolvsys.nexuse2e.core.exception.NexusException;
import com.evolvsys.nexuse2e.core.message.MetaData;
import com.evolvsys.nexuse2e.core.message.payload.content.PayloadContent;
import com.evolvsys.nexuse2e.core.message.payload.content.PayloadContentOutputStream;
import java.io.IOException;
import java.io.InputStream;
import lombok.*;

/**
 * @author SvenBarden
 * @since 27.10.2023
 */
@Data
@NoArgsConstructor
@AllArgsConstructor
@Builder(toBuilder = true)
public class ContextPayload {
    private Long id;

    private String contentId;

    @ToString.Exclude
    private PayloadContent content;

    private String contentType;
    private long size;
    private String encoding;
    private int sequenceNumber;

    @Builder.Default
    private MetaData metaData = new MetaData();

    /**
     * Creates a copy of the ContextPayload. The content is copied to ensure that the original content is not modified. The metaData is also copied to
     * maintain the original state.
     *
     * @return a new ContextPayload instance with copied content
     * @throws NexusException if there is an error during the content copy process
     * @see MetaData#copy()
     */
    public ContextPayload copy() throws NexusException {
        PayloadContent contentCopy;
        try (PayloadContentOutputStream outputStream = PayloadContent.createOutputStream();
                InputStream inputStream = content.getInputStream()) {
            inputStream.transferTo(outputStream);
            contentCopy = outputStream.finish();
        } catch (IOException e) {
            throw new NexusException(
                    NexusError.builder()
                            .message("Could not copy content for payload")
                            .extension("contentId", contentId)
                            .extension("id", id)
                            .build(),
                    e);
        }
        return toBuilder()
                .id(null)
                .content(contentCopy)
                .metaData(metaData.copy())
                .build();
    }
}
