package com.evolvsys.nexuse2e.internal;

import java.util.Arrays;
import lombok.Getter;

/**
 * @author Florian Drees
 * @since 26.10.2023
 */
@Getter
public enum BinaryEncoding {
    UNSUPPORTED("Unsupported encoding"),
    BINARY("Binary"),
    BASE64("Base 64");

    private final String encoding;

    BinaryEncoding(String encoding) {
        this.encoding = encoding;
    }

    public static BinaryEncoding of(String encoding) {
        return Arrays.stream(values())
                .filter(s -> s.getEncoding().equalsIgnoreCase(encoding))
                .findAny()
                .orElse(UNSUPPORTED);
    }
}
