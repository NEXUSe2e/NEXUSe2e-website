package com.evolvsys.nexuse2e.core.pipeline;

import com.evolvsys.nexuse2e.core.component.ComponentFactory;
import com.evolvsys.nexuse2e.core.component.Pipelet;
import com.evolvsys.nexuse2e.core.exception.NexusError;
import com.evolvsys.nexuse2e.core.exception.NexusRuntimeException;
import com.evolvsys.nexuse2e.core.model.entity.PipeletEntity;
import com.evolvsys.nexuse2e.core.pipeline.model.Pipeline;
import com.evolvsys.nexuse2e.core.pipeline.model.PipelineEntity;
import com.evolvsys.nexuse2e.core.pipeline.repository.PipelineRepository;
import com.evolvsys.nexuse2e.core.utils.LoggingUtils;
import lombok.RequiredArgsConstructor;
import lombok.extern.slf4j.Slf4j;
import org.springframework.stereotype.Component;
import org.springframework.transaction.annotation.Transactional;

/**
 * @author SvenBarden
 * @since 11.10.2023
 */
@Component
@RequiredArgsConstructor
@Slf4j
public class PipelineFactory {
    private final ComponentFactory componentFactory;
    private final PipelineRepository pipelineRepository;

    @Transactional(readOnly = true)
    public Pipeline get(PipelineEntity pipelineEntity) {
        Pipeline.PipelineBuilder pipelineBuilder = Pipeline.builder().pipelineId(pipelineEntity.getPipelineId());

        for (PipeletEntity pipeletEntity : pipelineEntity.getPipelets()) {
            try {
                Pipelet pipelet =
                        componentFactory.create(pipeletEntity.getComponentId(), pipeletEntity.getParameters());
                pipelineBuilder.pipelet(pipelet);
            } catch (IllegalArgumentException e) {
                LoggingUtils.globalWrapper(LOGGER)
                        .error(
                                "could not create pipelet {} for pipeline {}",
                                pipeletEntity.getComponentId(),
                                pipelineEntity.getPipelineId(),
                                e);
            }
        }

        return pipelineBuilder.build();
    }

    @Transactional(readOnly = true)
    public Pipeline get(String pipelineId) {
        return get(pipelineRepository
                .findByPipelineId(pipelineId)
                .orElseThrow(() -> new NexusRuntimeException(NexusError.builder()
                        .message(String.format("pipeline with id %s could not be found", pipelineId))
                        .build())));
    }

    @Transactional(readOnly = true)
    public Pipeline get(Long id) {
        PipelineEntity pipelineEntity = pipelineRepository.findById(id).orElseThrow();

        return get(pipelineEntity);
    }
}
