package com.evolvsys.nexuse2e.core.action.service;

import com.evolvsys.app.core.graphql.annotation.GraphQLService;
import com.evolvsys.app.core.graphql.dataloader.DataLoaderService;
import com.evolvsys.app.core.persistence.model.IdDto;
import com.evolvsys.nexuse2e.core.action.model.ActionDto;
import com.evolvsys.nexuse2e.core.action.validation.ValidActionHasMessages;
import com.evolvsys.nexuse2e.core.action.validation.ValidActionHasNextActions;
import com.evolvsys.nexuse2e.core.config.service.ConfigGraphQLService;
import com.evolvsys.nexuse2e.core.pipeline.PipelineBatchLoader;
import com.evolvsys.nexuse2e.core.pipeline.model.PipelineDto;
import com.evolvsys.nexuse2e.core.route.model.RouteDto;
import io.leangen.graphql.annotations.*;
import io.leangen.graphql.execution.ResolutionEnvironment;
import jakarta.validation.Valid;
import java.util.Collection;
import java.util.List;
import java.util.Optional;
import java.util.Set;
import java.util.concurrent.CompletableFuture;
import lombok.RequiredArgsConstructor;
import lombok.extern.slf4j.Slf4j;

/**
 * @author Florian Drees
 * @since 20.12.2023
 */
@GraphQLService
@RequiredArgsConstructor
@Slf4j
public class ActionGraphQLService {
    private final ActionService actionService;
    private final DataLoaderService dataLoaderService;

    @GraphQLMutation(name = "action_update", description = "Update an existing or create a new action")
    @GraphQLNonNull
    public ActionDto update(@GraphQLNonNull @Valid ActionDto actionDto) {
        return ConfigGraphQLService.notifySubscribers(actionService.save(actionDto));
    }

    @GraphQLMutation(name = "action_delete", description = "Delete an existing action")
    public boolean delete(@ValidActionHasMessages @ValidActionHasNextActions long actionId) {
        actionService.delete(actionId);
        ConfigGraphQLService.notifySubscribers();
        return true;
    }

    @GraphQLMutation(name = "action_updateWithValidNextActions")
    @GraphQLNonNull
    public ActionDto updateWithValidNextActions(
            @GraphQLNonNull @Valid ActionDto action,
            @GraphQLNonNull Collection<@GraphQLNonNull ActionDto> validNextActions) {
        return ConfigGraphQLService.notifySubscribers(actionService.save(action, validNextActions));
    }

    @GraphQLMutation(name = "action_addRoute")
    @GraphQLNonNull
    public ActionDto addRoute(long actionId, @GraphQLNonNull @Valid RouteDto route) {
        return ConfigGraphQLService.notifySubscribers(actionService.addRoute(actionId, route));
    }

    @GraphQLMutation(name = "action_linkRoutes", description = "Links existing routes to the action")
    @GraphQLNonNull
    public ActionDto linkRoutes(long actionId, List<@GraphQLNonNull Long> routeIds) {
        return ConfigGraphQLService.notifySubscribers(actionService.linkRoutes(actionId, routeIds));
    }

    @GraphQLMutation(name = "action_removeRoute", description = "Removes the route from the action")
    @GraphQLNonNull
    public ActionDto removeRoute(long actionId, long routeId) {
        return ConfigGraphQLService.notifySubscribers(actionService.removeRoute(actionId, routeId));
    }

    @GraphQLQuery(name = "action_getById", description = "Gets an action")
    public Optional<ActionDto> getById(long actionId) {
        return actionService.getById(actionId);
    }

    @GraphQLQuery
    public CompletableFuture<Set<@GraphQLNonNull ActionDto>> validNextActions(
            @GraphQLContext ActionDto actionDto, @GraphQLEnvironment ResolutionEnvironment environment) {
        return dataLoaderService.load(environment, ValidNextActionsBatchLoader.class, actionDto, IdDto::getId);
    }

    @GraphQLQuery
    public CompletableFuture<Set<@GraphQLNonNull RouteDto>> routes(
            @GraphQLContext ActionDto actionDto, @GraphQLEnvironment ResolutionEnvironment resolutionEnvironment) {
        return dataLoaderService.loadRelation(resolutionEnvironment, ActionRouteBatchLoader.class, actionDto);
    }

    @GraphQLQuery
    public CompletableFuture<PipelineDto> forwardPipeline(
            @GraphQLContext ActionDto actionDto, @GraphQLEnvironment ResolutionEnvironment environment) {
        return dataLoaderService.load(
                environment,
                PipelineBatchLoader.class,
                actionDto,
                "forwardPipelineId",
                ActionDto::getForwardPipelineId);
    }

    @GraphQLQuery
    public CompletableFuture<PipelineDto> returnPipeline(
            @GraphQLContext ActionDto actionDto, @GraphQLEnvironment ResolutionEnvironment environment) {
        return dataLoaderService.load(
                environment, PipelineBatchLoader.class, actionDto, "returnPipelineId", ActionDto::getReturnPipelineId);
    }

    @GraphQLQuery
    public CompletableFuture<PipelineDto> acknowledgmentPipeline(
            @GraphQLContext ActionDto actionDto, @GraphQLEnvironment ResolutionEnvironment environment) {
        return dataLoaderService.load(
                environment,
                PipelineBatchLoader.class,
                actionDto,
                "acknowledgmentPipelineId",
                ActionDto::getAcknowledgmentPipelineId);
    }
}
