package com.evolvsys.nexuse2e.core.action.service;

import com.evolvsys.app.core.auth.service.CurrentUserService;
import com.evolvsys.app.core.process.ProcessExecutionContext;
import com.evolvsys.app.core.process.ProcessService;
import com.evolvsys.app.core.service.BaseService;
import com.evolvsys.nexuse2e.core.action.action.ActionActionType;
import com.evolvsys.nexuse2e.core.action.action.ActionAddRouteAction;
import com.evolvsys.nexuse2e.core.action.action.ActionLinkRouteAction;
import com.evolvsys.nexuse2e.core.action.action.ActionRemoveRouteAction;
import com.evolvsys.nexuse2e.core.action.mapper.ActionMapper;
import com.evolvsys.nexuse2e.core.action.model.ActionDto;
import com.evolvsys.nexuse2e.core.action.model.ActionEntity;
import com.evolvsys.nexuse2e.core.action.repository.ActionRepository;
import com.evolvsys.nexuse2e.core.route.model.RouteDto;
import java.util.Collection;
import java.util.HashSet;
import java.util.List;
import java.util.Set;
import org.springframework.stereotype.Service;
import org.springframework.transaction.annotation.Transactional;

/**
 * @author Florian Drees
 * @since 20.12.2023
 */
@Service
public class ActionService extends BaseService<ActionDto, ActionEntity, ActionRepository> {
    private final ProcessService processService;
    private final ActionMapper actionMapper;

    public ActionService(
            ActionMapper mapper,
            ActionRepository repository,
            CurrentUserService currentUserService,
            ProcessService processService) {
        super(mapper, repository, currentUserService);
        this.processService = processService;
        this.actionMapper = mapper;
    }

    @Override
    protected ActionEntity createEntity() {
        return new ActionEntity();
    }

    public ActionDto addRoute(long actionId, RouteDto routeDto) {
        ActionEntity actionEntity = findWithActionCheck(actionId, ActionActionType.ADD_ROUTE.name());

        actionEntity = processService.execute(ProcessExecutionContext.<ActionEntity, ActionDto>builder()
                .target(actionEntity)
                .actionType(ActionActionType.ADD_ROUTE)
                .parameterValue(ActionAddRouteAction.ROUTE_PARAMETER.of(routeDto))
                .build());

        return mapEntityToDto(actionEntity);
    }

    @Transactional
    public ActionDto save(ActionDto actionDto, Collection<ActionDto> validNextActions) {
        ActionEntity entity = getEntity(actionDto);

        actionMapper.updateEntity(actionDto, entity, currentUserService.getCurrentUserData());

        Set<ActionEntity> choreographyActions =
                new HashSet<>(entity.getChoreography().getActions());
        if (actionDto.getId() == null) {
            choreographyActions.add(entity);
        }

        actionMapper.updateValidNextActions(validNextActions, entity, choreographyActions);

        entity = save(entity);

        return mapEntityToDto(entity);
    }

    public ActionDto removeRoute(long actionId, long routeId) {
        ActionEntity actionEntity = findWithActionCheck(actionId, ActionActionType.REMOVE_ROUTE.name());

        actionEntity = processService.execute(ProcessExecutionContext.<ActionEntity, ActionDto>builder()
                .target(actionEntity)
                .actionType(ActionActionType.REMOVE_ROUTE)
                .parameterValue(ActionRemoveRouteAction.ROUTE_ID_PARAMETER.of(routeId))
                .build());

        return mapEntityToDto(actionEntity);
    }

    public ActionDto linkRoutes(long actionId, List<Long> routeIds) {
        ActionEntity actionEntity = findWithActionCheck(actionId, ActionActionType.LINK_ROUTE.name());

        actionEntity = processService.execute(ProcessExecutionContext.<ActionEntity, ActionDto>builder()
                .target(actionEntity)
                .actionType(ActionActionType.LINK_ROUTE)
                .parameterValue(ActionLinkRouteAction.ROUTE_ID_PARAMETER.of(routeIds))
                .build());

        return mapEntityToDto(actionEntity);
    }
}
