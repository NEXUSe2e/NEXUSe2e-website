package com.evolvsys.nexuse2e.impl.pipelet.rule;

import com.evolvsys.nexuse2e.core.component.Pipelet;
import com.evolvsys.nexuse2e.core.constants.HttpConstants;
import com.evolvsys.nexuse2e.core.exception.NexusError;
import com.evolvsys.nexuse2e.core.exception.NexusException;
import com.evolvsys.nexuse2e.core.message.context.ExecutionContext;
import com.evolvsys.nexuse2e.core.parameter.ParameterDefinition;
import com.evolvsys.nexuse2e.core.parameter.ParameterType;
import java.util.List;
import lombok.extern.slf4j.Slf4j;
import org.apache.commons.collections4.CollectionUtils;

/**
 * @author SvenBarden
 * @since 02.07.2024
 */
@Slf4j
public class HttpProtocolRulePipelet extends Pipelet {
    private static final String PARAM_PROTOCOLS = "protocols";

    private boolean isValid(String validProtocol, String protocolToCheck) {
        return protocolToCheck.equalsIgnoreCase(validProtocol)
                || protocolToCheck.equalsIgnoreCase(validProtocol + ":TLS"); // external tomcat adds :TLS suffix
    }

    @Override
    public List<ParameterDefinition> parameterDefinitions() {
        return List.of(ParameterDefinition.builder()
                .name(PARAM_PROTOCOLS)
                .displayName("Valid protocols")
                .description("Valid http protocols. An empty list counts as if all protocols are valid.")
                .type(ParameterType.STRING_LIST)
                .build());
    }

    @Override
    public ExecutionContext execute(ExecutionContext executionContext) throws NexusException {
        String protocol = executionContext.getMetaData().getFirstOrNull(HttpConstants.PROTOCOL_VERSION);

        List<String> validProtocols = parameters.value(PARAM_PROTOCOLS);

        LOGGER.trace("validating http protocol {}", protocol);

        if (protocol == null) {
            throw new NexusException(NexusError.builder()
                    .message("No http protocol found in message context")
                    .extension("choreographyId", executionContext.getChoreographyId())
                    .extension("actionId", executionContext.getActionId())
                    .extension("sourcePartnerId", executionContext.getSourcePartnerId())
                    .extension("targetPartnerId", executionContext.getTargetPartnerId())
                    .extension("messageId", executionContext.getMessage().getId())
                    .build());
        } else {
            if (CollectionUtils.isEmpty(validProtocols)) {
                LOGGER.trace("valid protocol list is empty, no validation occurred");
            } else {
                if (validProtocols.stream().anyMatch(string -> isValid(string, protocol))) {
                    LOGGER.trace("protocol {} is in valid protocols {}", protocol, validProtocols);
                } else {
                    throw new NexusException(NexusError.builder()
                            .message("Invalid inbound http protocol")
                            .extension("protocol", protocol)
                            .extension("validProtocols", validProtocols)
                            .extension("choreographyId", executionContext.getChoreographyId())
                            .extension("actionId", executionContext.getActionId())
                            .extension("sourcePartnerId", executionContext.getSourcePartnerId())
                            .extension("targetPartnerId", executionContext.getTargetPartnerId())
                            .extension(
                                    "messageId", executionContext.getMessage().getId())
                            .build());
                }
            }
        }

        return executionContext;
    }
}
