package com.evolvsys.nexuse2e.impl.pipelet.rule;

import com.evolvsys.nexuse2e.core.component.Pipelet;
import com.evolvsys.nexuse2e.core.constants.HttpConstants;
import com.evolvsys.nexuse2e.core.exception.NexusError;
import com.evolvsys.nexuse2e.core.exception.NexusException;
import com.evolvsys.nexuse2e.core.message.context.ExecutionContext;
import com.evolvsys.nexuse2e.core.parameter.ParameterDefinition;
import com.evolvsys.nexuse2e.core.parameter.ParameterType;
import com.evolvsys.nexuse2e.core.service.http.WebClientContext;
import com.evolvsys.nexuse2e.core.service.http.client.rest.NexusRestClientFactory;
import com.evolvsys.nexuse2e.impl.util.JwtAuthoritiesConverter;
import java.io.IOException;
import java.security.KeyManagementException;
import java.security.KeyStoreException;
import java.security.NoSuchAlgorithmException;
import java.security.UnrecoverableKeyException;
import java.security.cert.CertificateException;
import java.util.List;
import java.util.Objects;
import lombok.extern.slf4j.Slf4j;
import org.apache.commons.collections4.CollectionUtils;
import org.apache.commons.collections4.KeyValue;
import org.apache.commons.lang3.StringUtils;
import org.springframework.security.core.Authentication;
import org.springframework.security.core.AuthenticationException;
import org.springframework.security.core.GrantedAuthority;
import org.springframework.security.oauth2.jwt.NimbusJwtDecoder;
import org.springframework.security.oauth2.server.resource.authentication.BearerTokenAuthenticationToken;
import org.springframework.security.oauth2.server.resource.authentication.JwtAuthenticationConverter;
import org.springframework.security.oauth2.server.resource.authentication.JwtAuthenticationProvider;
import org.springframework.security.oauth2.server.resource.authentication.JwtAuthenticationToken;
import org.springframework.web.client.RestTemplate;

/**
 * @author SvenBarden
 * @since 13.08.2024
 */
@Slf4j
public class OAuth2AuthenticationRulePipelet extends Pipelet {
    private static final String PARAM_ISSUER_URL = "issuer_url";
    private static final String PARAM_CLAIM_VERIFICATION = "claim_verification";
    private static final String PARAM_AUTHORITY_CLAIMS = "authority_claims";
    private static final String PARAM_VALID_ROLES = "valid_roles";
    private static final String PARAM_DISABLE_HOSTNAME_VERIFICATION = "disable_hostname_verification";
    private static final String PARAM_VALID_AUDIENCES = "valid_audiences";

    @Override
    public List<ParameterDefinition> parameterDefinitions() {
        return List.of(
                ParameterDefinition.builder()
                        .name(PARAM_ISSUER_URL)
                        .displayName("Issuer url")
                        .description("Name of the issuer url to validate the token against.")
                        .type(ParameterType.TRIMMED_STRING)
                        .required(true)
                        .build(),
                ParameterDefinition.builder()
                        .name(PARAM_CLAIM_VERIFICATION)
                        .displayName("Claim verification")
                        .description("Verifies the values of claims against a static value.")
                        .type(ParameterType.KEY_VALUE_PAIR_LIST_TRIMMED)
                        .build(),
                ParameterDefinition.builder()
                        .name(PARAM_AUTHORITY_CLAIMS)
                        .displayName("Authority claim names")
                        .description("Claim names that may contains authorities to be verified against the role list.")
                        .type(ParameterType.STRING_LIST)
                        .defaultValue(List.of("scp", "scope", "roles"))
                        .build(),
                ParameterDefinition.builder()
                        .name(PARAM_VALID_ROLES)
                        .displayName("Valid roles")
                        .description("Roles that all must be in the role claims")
                        .type(ParameterType.STRING_LIST)
                        .build(),
                ParameterDefinition.builder()
                        .name(PARAM_VALID_AUDIENCES)
                        .displayName("Valid audiences")
                        .description(
                                "A List of audiences that are validated against the token. At least one of this audiences must be part of the token.")
                        .type(ParameterType.STRING_LIST)
                        .required(true)
                        .build(),
                ParameterDefinition.builder()
                        .name(PARAM_DISABLE_HOSTNAME_VERIFICATION)
                        .displayName("Disable ssl host name verification")
                        .description("Disables the ssl host name verification for the oauth2 token request.")
                        .type(ParameterType.BOOLEAN)
                        .required(true)
                        .defaultValue(Boolean.FALSE)
                        .build());
    }

    @Override
    public ExecutionContext execute(ExecutionContext executionContext) throws NexusException {
        String authorization =
                executionContext.getMetaData().getFirstOrThrows(HttpConstants.PREFIX_REQUEST_HEADER, "authorization");

        if (StringUtils.isBlank(authorization)) {
            throw new NexusException(NexusError.builder()
                    .message("Authorization header is missing")
                    .build());
        }

        if (authorization.startsWith("Bearer ") == false) {
            throw new NexusException(NexusError.builder()
                    .message("Authorization header is not a Bearer token")
                    .build());
        }

        String token = authorization.substring("Bearer ".length());

        LOGGER.trace("got bearer token: {}", token);

        NimbusJwtDecoder jwtDecoder = extendJwtDecoderBuilder(
                        executionContext, NimbusJwtDecoder.withIssuerLocation(parameters.value(PARAM_ISSUER_URL)))
                .build();
        JwtAuthenticationProvider jwtAuthenticationProvider = new JwtAuthenticationProvider(jwtDecoder);
        JwtAuthenticationConverter jwtAuthenticationConverter = new JwtAuthenticationConverter();
        jwtAuthenticationConverter.setJwtGrantedAuthoritiesConverter(
                new JwtAuthoritiesConverter(parameters.value(PARAM_AUTHORITY_CLAIMS)));
        jwtAuthenticationProvider.setJwtAuthenticationConverter(jwtAuthenticationConverter);
        BearerTokenAuthenticationToken bearerTokenAuthenticationToken = new BearerTokenAuthenticationToken(token);

        try {
            Authentication authenticate = jwtAuthenticationProvider.authenticate(bearerTokenAuthenticationToken);
            if (authenticate == null) {
                throw new NexusException(NexusError.builder()
                        .message("Bearer token is invalid, no validation object could be created")
                        .build());
            }

            JwtAuthenticationToken jwtAuthenticationToken = (JwtAuthenticationToken) authenticate;

            List<String> validAudiences = parameters.value(PARAM_VALID_AUDIENCES);
            if (CollectionUtils.isEmpty(validAudiences)) {
                LOGGER.warn(
                        "No valid audiences are configured, for security reasons audience validation should be enabled");
            } else {
                if (CollectionUtils.isNotEmpty(jwtAuthenticationToken.getToken().getAudience())
                        && CollectionUtils.containsAny(
                                validAudiences,
                                jwtAuthenticationToken.getToken().getAudience())) {
                    LOGGER.trace("audience validation successful");
                } else {
                    throw new NexusException(NexusError.builder()
                            .message("Bearer token is invalid, audience validation failed")
                            .extension("validAudiences", validAudiences)
                            .extension(
                                    "audiences",
                                    jwtAuthenticationToken.getToken().getAudience())
                            .build());
                }
            }

            List<KeyValue<String, String>> claimVerifications = parameters.value(PARAM_CLAIM_VERIFICATION);
            if (CollectionUtils.isNotEmpty(claimVerifications)) {
                for (KeyValue<String, String> claimVerification : claimVerifications) {
                    LOGGER.trace(
                            "verifying claim: {} for value: {}",
                            claimVerification.getKey(),
                            claimVerification.getValue());
                    Object tokenValue =
                            jwtAuthenticationToken.getTokenAttributes().get(claimVerification.getKey());
                    if (Objects.equals(tokenValue, claimVerification.getValue())) {
                        LOGGER.trace("claim verification successful");
                    } else {
                        throw new NexusException(NexusError.builder()
                                .message("Bearer token is invalid, claim verification failed")
                                .extension("claim", claimVerification.getKey())
                                .extension("value", tokenValue)
                                .extension("expectedValue", claimVerification.getValue())
                                .build());
                    }
                }
            }

            List<String> validRoles = parameters.value(PARAM_VALID_ROLES);
            if (CollectionUtils.isNotEmpty(validRoles)) {
                List<String> authorities = authenticate.getAuthorities().stream()
                        .map(GrantedAuthority::getAuthority)
                        .toList();
                LOGGER.trace("validating roles: {} against: {}", validRoles, authorities);
                if (CollectionUtils.containsAll(authorities, validRoles)) {
                    LOGGER.trace("all roles are valid");
                } else {
                    throw new NexusException(NexusError.builder()
                            .message("Bearer token is invalid, role verification failed")
                            .extension("requiredRoles", validRoles)
                            .extension("roles", authorities)
                            .build());
                }
            }
        } catch (AuthenticationException e) {
            throw new NexusException(
                    NexusError.builder().message("Bearer token is invalid").build(), e);
        }

        return executionContext;
    }

    private NimbusJwtDecoder.JwkSetUriJwtDecoderBuilder extendJwtDecoderBuilder(
            ExecutionContext executionContext, NimbusJwtDecoder.JwkSetUriJwtDecoderBuilder builder)
            throws NexusException {
        NexusRestClientFactory restClientFactory =
                executionContext.getBeanFactory().getBean(NexusRestClientFactory.class);
        WebClientContext context = restClientFactory
                .createContextBuilder()
                .ssl(true)
                .disableHostNameVerification(parameters.value(PARAM_DISABLE_HOSTNAME_VERIFICATION))
                .build();
        RestTemplate restTemplate = new RestTemplate();
        try {
            restTemplate.setRequestFactory(restClientFactory.createRequestFactory(context));
            builder.restOperations(restTemplate);
        } catch (UnrecoverableKeyException
                | CertificateException
                | NoSuchAlgorithmException
                | KeyStoreException
                | IOException
                | KeyManagementException e) {
            throw new NexusException(
                    NexusError.builder()
                            .message("Could not create request factory for oauth2 validation")
                            .build(),
                    e);
        }
        return builder;
    }
}
