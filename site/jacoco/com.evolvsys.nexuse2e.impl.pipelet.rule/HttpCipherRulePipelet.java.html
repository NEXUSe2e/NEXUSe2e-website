package com.evolvsys.nexuse2e.impl.pipelet.rule;

import com.evolvsys.nexuse2e.core.component.Pipelet;
import com.evolvsys.nexuse2e.core.constants.HttpConstants;
import com.evolvsys.nexuse2e.core.exception.NexusError;
import com.evolvsys.nexuse2e.core.exception.NexusException;
import com.evolvsys.nexuse2e.core.message.context.ExecutionContext;
import com.evolvsys.nexuse2e.core.parameter.ParameterDefinition;
import com.evolvsys.nexuse2e.core.parameter.ParameterType;
import java.util.List;
import lombok.extern.slf4j.Slf4j;
import org.apache.commons.collections4.CollectionUtils;

/**
 * @author SvenBarden
 * @since 02.07.2024
 */
@Slf4j
public class HttpCipherRulePipelet extends Pipelet {
    private static final String PARAM_CIPHERS = "ciphers";

    @Override
    public List<ParameterDefinition> parameterDefinitions() {
        return List.of(ParameterDefinition.builder()
                .name(PARAM_CIPHERS)
                .displayName("Valid ciphers")
                .description("Valid http ciphers. An empty list counts as if all ciphers are valid.")
                .type(ParameterType.STRING_LIST)
                .build());
    }

    @Override
    public ExecutionContext execute(ExecutionContext executionContext) throws NexusException {
        String cipher = executionContext.getMetaData().getFirstOrNull(HttpConstants.CIPHERS);

        List<String> validCiphers = parameters.value(PARAM_CIPHERS);

        LOGGER.trace("validating http cipher {}", cipher);

        if (cipher == null) {
            throw new NexusException(NexusError.builder()
                    .message("No http cipher found in message context")
                    .extension("choreographyId", executionContext.getChoreographyId())
                    .extension("actionId", executionContext.getActionId())
                    .extension("sourcePartnerId", executionContext.getSourcePartnerId())
                    .extension("targetPartnerId", executionContext.getTargetPartnerId())
                    .extension("messageId", executionContext.getMessage().getId())
                    .build());
        } else {
            if (CollectionUtils.isEmpty(validCiphers)) {
                LOGGER.trace("valid cipher list is empty, no validation occurred");
            } else {
                if (validCiphers.stream().anyMatch(string -> string.equalsIgnoreCase(cipher))) {
                    LOGGER.trace("cipher {} is in valid protocols {}", cipher, validCiphers);
                } else {
                    throw new NexusException(NexusError.builder()
                            .message("Invalid inbound http cipher")
                            .extension("cipher", cipher)
                            .extension("validCiphers", validCiphers)
                            .extension("choreographyId", executionContext.getChoreographyId())
                            .extension("actionId", executionContext.getActionId())
                            .extension("sourcePartnerId", executionContext.getSourcePartnerId())
                            .extension("targetPartnerId", executionContext.getTargetPartnerId())
                            .extension(
                                    "messageId", executionContext.getMessage().getId())
                            .build());
                }
            }
        }

        return executionContext;
    }
}
