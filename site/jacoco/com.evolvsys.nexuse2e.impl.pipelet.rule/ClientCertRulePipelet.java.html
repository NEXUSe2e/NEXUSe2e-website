package com.evolvsys.nexuse2e.impl.pipelet.rule;

import com.evolvsys.nexuse2e.core.certificate.service.CertificateValidator;
import com.evolvsys.nexuse2e.core.component.Pipelet;
import com.evolvsys.nexuse2e.core.exception.NexusError;
import com.evolvsys.nexuse2e.core.exception.NexusException;
import com.evolvsys.nexuse2e.core.message.context.ExecutionContext;
import com.evolvsys.nexuse2e.core.parameter.ParameterDefinition;
import java.security.cert.X509Certificate;
import java.util.List;
import lombok.extern.slf4j.Slf4j;
import org.apache.commons.collections4.CollectionUtils;

/**
 * @author SvenBarden
 * @since 02.07.2024
 */
@Slf4j
public class ClientCertRulePipelet extends Pipelet {

    @Override
    public List<ParameterDefinition> parameterDefinitions() {
        return List.of();
    }

    @Override
    public ExecutionContext execute(ExecutionContext executionContext) throws NexusException {
        X509Certificate clientCertificate = executionContext.getClientCertificate();

        if (clientCertificate == null) {
            throw new NexusException(NexusError.builder()
                    .message("No client certificate found")
                    .extension("choreographyId", executionContext.getChoreographyId())
                    .extension("actionId", executionContext.getActionId())
                    .extension("sourcePartnerId", executionContext.getSourcePartnerId())
                    .extension("targetPartnerId", executionContext.getTargetPartnerId())
                    .extension("messageId", executionContext.getMessage().getId())
                    .build());
        } else {
            List<X509Certificate> certificates = executionContext
                    .getRoute()
                    .getSourcePartnerTrustCertificates()
                    .getNullable();
            if (CollectionUtils.isEmpty(certificates)) {
                throw new NexusException(NexusError.builder()
                        .message("No certificates found to validate the client certificate")
                        .extension("choreographyId", executionContext.getChoreographyId())
                        .extension("actionId", executionContext.getActionId())
                        .extension("sourcePartnerId", executionContext.getSourcePartnerId())
                        .extension("targetPartnerId", executionContext.getTargetPartnerId())
                        .extension("messageId", executionContext.getMessage().getId())
                        .build());
            }
            boolean valid = executionContext
                    .getBeanFactory()
                    .getBean(CertificateValidator.class)
                    .validate(clientCertificate, certificates);
            if (valid) {
                LOGGER.trace("client certificate was successfully validates against {} certs", certificates.size());
            } else {
                throw new NexusException(NexusError.builder()
                        .message("Client certificate is not valid")
                        .extension("choreographyId", executionContext.getChoreographyId())
                        .extension("actionId", executionContext.getActionId())
                        .extension("sourcePartnerId", executionContext.getSourcePartnerId())
                        .extension("targetPartnerId", executionContext.getTargetPartnerId())
                        .extension("messageId", executionContext.getMessage().getId())
                        .build());
            }
        }

        return executionContext;
    }
}
