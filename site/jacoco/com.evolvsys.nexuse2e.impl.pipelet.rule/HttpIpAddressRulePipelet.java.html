package com.evolvsys.nexuse2e.impl.pipelet.rule;

import com.evolvsys.nexuse2e.core.component.Pipelet;
import com.evolvsys.nexuse2e.core.constants.HttpConstants;
import com.evolvsys.nexuse2e.core.exception.NexusError;
import com.evolvsys.nexuse2e.core.exception.NexusException;
import com.evolvsys.nexuse2e.core.message.context.ExecutionContext;
import com.evolvsys.nexuse2e.core.parameter.ParameterDefinition;
import com.evolvsys.nexuse2e.core.parameter.ParameterType;
import com.evolvsys.nexuse2e.core.utils.validation.IpUtils;
import java.util.ArrayList;
import java.util.List;
import lombok.extern.slf4j.Slf4j;
import org.apache.commons.lang3.StringUtils;

/**
 * @author SvenBarden
 * @since 02.07.2024
 */
@Slf4j
public class HttpIpAddressRulePipelet extends Pipelet {
    private static final String PARAM_VALID_IPV4_ADDRESSES = "valid_ipv4_addresses";
    private static final String PARAM_VALID_IPV6_ADDRESSES = "valid_ipv6_addresses";

    @Override
    public List<ParameterDefinition> parameterDefinitions() {
        return List.of(
                ParameterDefinition.builder()
                        .name(PARAM_VALID_IPV4_ADDRESSES)
                        .displayName("Valid IPv4 Addresses")
                        .description("List of valid IPv4 Addresses for the inbound message.")
                        .type(ParameterType.STRING_LIST)
                        .build(),
                ParameterDefinition.builder()
                        .name(PARAM_VALID_IPV6_ADDRESSES)
                        .displayName("Valid IPv6 Addresses")
                        .description("List of valid IPv6 Addresses for the inbound message.")
                        .type(ParameterType.STRING_LIST)
                        .build());
    }

    @Override
    public ExecutionContext execute(ExecutionContext executionContext) throws NexusException {
        String clientIpAddress = executionContext.getMetaData().getFirstOrNull(HttpConstants.CLIENT_IP);
        String clientForwardedIpAddress =
                executionContext.getMetaData().getFirstOrNull(HttpConstants.CLIENT_FORWARDED_IP);

        List<String> ipAddresses = new ArrayList<>();
        if (StringUtils.isNotEmpty(clientIpAddress)) {
            ipAddresses.add(clientIpAddress);
        }
        if (StringUtils.isNotEmpty(clientForwardedIpAddress)) {
            ipAddresses.add(clientForwardedIpAddress);
        }

        List<String> validIpv4AddressRanges = parameters.value(PARAM_VALID_IPV4_ADDRESSES);
        List<String> validIpv6AddressRanges = parameters.value(PARAM_VALID_IPV6_ADDRESSES);

        for (String ipAddress : ipAddresses) {
            if (IpUtils.isValid(ipAddress, validIpv4AddressRanges, validIpv6AddressRanges)) {
                LOGGER.trace(
                        "IP Address {} is valid for IPv4 {} and IPv6 {}",
                        ipAddress,
                        validIpv4AddressRanges,
                        validIpv6AddressRanges);
            } else {
                throw new NexusException(NexusError.builder()
                        .message("Inbound IP Address is not valid")
                        .extension("ipAddress", ipAddress)
                        .extension("validIpV4Addresses", validIpv4AddressRanges)
                        .extension("validIpV6Addresses", validIpv6AddressRanges)
                        .extension("choreographyId", executionContext.getChoreographyId())
                        .extension("actionId", executionContext.getActionId())
                        .extension("sourcePartnerId", executionContext.getSourcePartnerId())
                        .extension("targetPartnerId", executionContext.getTargetPartnerId())
                        .extension("messageId", executionContext.getMessage().getId())
                        .build());
            }
        }

        return executionContext;
    }
}
