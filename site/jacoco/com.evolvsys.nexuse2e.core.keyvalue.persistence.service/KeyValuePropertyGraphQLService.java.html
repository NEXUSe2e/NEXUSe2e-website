package com.evolvsys.nexuse2e.core.keyvalue.persistence.service;

import com.evolvsys.app.core.graphql.annotation.GraphQLService;
import com.evolvsys.nexuse2e.core.keyvalue.persistence.model.KeyValuePropertyDto;
import com.evolvsys.nexuse2e.core.keyvalue.persistence.model.KeyValuePropertyFilter;
import io.leangen.graphql.annotations.GraphQLMutation;
import io.leangen.graphql.annotations.GraphQLNonNull;
import io.leangen.graphql.annotations.GraphQLQuery;
import jakarta.validation.Valid;
import java.util.Optional;
import lombok.RequiredArgsConstructor;
import org.springframework.data.domain.Page;
import org.springframework.data.domain.Pageable;

/**
 * @author SvenBarden
 * @since 17.06.2024
 */
@GraphQLService
@RequiredArgsConstructor
public class KeyValuePropertyGraphQLService {

    private final KeyValuePropertyService keyValuePropertyService;

    @GraphQLQuery(name = "keyValueProperty_getById")
    public Optional<KeyValuePropertyDto> getById(long id) {
        return keyValuePropertyService.getById(id);
    }

    @GraphQLQuery(name = "keyValueProperty_list")
    @GraphQLNonNull
    public Page<@GraphQLNonNull KeyValuePropertyDto> list(
            Pageable pageable, @Valid KeyValuePropertyFilter keyValuePropertyFilter) {
        return keyValuePropertyService.list(pageable, keyValuePropertyFilter);
    }

    @GraphQLNonNull
    @GraphQLMutation(name = "keyValueProperty_update")
    public KeyValuePropertyDto update(@GraphQLNonNull @Valid KeyValuePropertyDto keyValuePropertyDto) {
        return keyValuePropertyService.save(keyValuePropertyDto);
    }

    @GraphQLMutation(name = "keyValueProperty_delete")
    public boolean delete(long id) {
        keyValuePropertyService.delete(id);
        return true;
    }
}
