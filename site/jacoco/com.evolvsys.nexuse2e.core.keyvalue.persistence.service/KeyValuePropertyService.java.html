package com.evolvsys.nexuse2e.core.keyvalue.persistence.service;

import com.evolvsys.app.core.auth.service.CurrentUserService;
import com.evolvsys.app.core.service.BaseService;
import com.evolvsys.app.core.service.ListService;
import com.evolvsys.nexuse2e.core.keyvalue.persistence.mapper.KeyValuePropertyMapper;
import com.evolvsys.nexuse2e.core.keyvalue.persistence.model.KeyValuePropertyDto;
import com.evolvsys.nexuse2e.core.keyvalue.persistence.model.KeyValuePropertyEntity;
import com.evolvsys.nexuse2e.core.keyvalue.persistence.model.KeyValuePropertyFilter;
import com.evolvsys.nexuse2e.core.keyvalue.persistence.repository.KeyValuePropertyRepository;
import com.evolvsys.nexuse2e.core.utils.StreamUtils;
import java.util.Collection;
import java.util.List;
import java.util.Map;
import one.util.streamex.EntryStream;
import one.util.streamex.StreamEx;
import org.apache.commons.text.lookup.StringLookup;
import org.apache.commons.text.lookup.StringLookupFactory;
import org.springframework.data.domain.Page;
import org.springframework.data.domain.Pageable;
import org.springframework.stereotype.Service;

/**
 * @author SvenBarden
 * @since 17.06.2024
 */
@Service
public class KeyValuePropertyService
        extends BaseService<KeyValuePropertyDto, KeyValuePropertyEntity, KeyValuePropertyRepository> {
    private final ListService listService;

    public KeyValuePropertyService(
            KeyValuePropertyMapper mapper,
            KeyValuePropertyRepository repository,
            CurrentUserService currentUserService,
            ListService listService) {
        super(mapper, repository, currentUserService);
        this.listService = listService;
    }

    private StringLookup toLookup(Collection<KeyValuePropertyEntity> entities) {
        return StringLookupFactory.INSTANCE.mapStringLookup(entities.stream()
                .collect(StreamUtils.toMapOfNullables(
                        KeyValuePropertyEntity::getKey, KeyValuePropertyEntity::getValue)));
    }

    public Map<String, StringLookup> loadLookups() {
        List<KeyValuePropertyEntity> entities = getRepository().findAll();
        return EntryStream.of(StreamEx.of(entities).groupingBy(KeyValuePropertyEntity::getCategory))
                .mapValues(this::toLookup)
                .toMap();
    }

    @Override
    protected KeyValuePropertyEntity createEntity() {
        return new KeyValuePropertyEntity();
    }

    public Page<KeyValuePropertyDto> list(Pageable pageable, KeyValuePropertyFilter filter) {
        return listService.map(listService.list(pageable, filter, this), mapper);
    }
}
