package com.evolvsys.nexuse2e.core.component;

import com.evolvsys.nexuse2e.core.component.exception.ComponentCreationException;
import com.evolvsys.nexuse2e.core.component.service.ComponentPostCreationService;
import com.evolvsys.nexuse2e.core.model.entity.ParameterEntity;
import com.evolvsys.nexuse2e.core.parameter.ParameterDefinition;
import com.evolvsys.nexuse2e.core.parameter.ParameterType;
import com.evolvsys.nexuse2e.core.parameter.Parameters;
import com.evolvsys.nexuse2e.core.parameter.service.ParameterConverterRegistry;
import edu.umd.cs.findbugs.annotations.SuppressFBWarnings;
import java.lang.reflect.InvocationTargetException;
import java.util.Collection;
import java.util.Collections;
import java.util.Optional;
import lombok.RequiredArgsConstructor;
import lombok.extern.slf4j.Slf4j;
import org.apache.commons.collections4.CollectionUtils;
import org.apache.commons.lang3.ObjectUtils;
import org.springframework.beans.factory.BeanFactory;
import org.springframework.beans.factory.BeanFactoryAware;
import org.springframework.lang.Nullable;
import org.springframework.stereotype.Component;

/**
 * @author SvenBarden
 * @since 09.10.2023
 */
@RequiredArgsConstructor
@Component
@Slf4j
public class ComponentFactory implements BeanFactoryAware {
    private final ComponentRegistry componentRegistry;
    private final ParameterConverterRegistry parameterConverterUtils;
    private final ComponentPostCreationService componentPostCreationService;
    private BeanFactory beanFactory;

    @SuppressWarnings("unchecked")
    public <T> T create(String name, @Nullable Collection<ParameterEntity> parameterEntities) {
        Class<T> componentType = (Class<T>) getComponentType(name);

        return create(componentType, parameterEntities);
    }

    private Class<?> getComponentType(String name) {
        return componentRegistry
                .get(name)
                .orElseThrow(() -> new IllegalArgumentException("no component with name " + name + " found"));
    }

    @SuppressFBWarnings("EXS_EXCEPTION_SOFTENING_NO_CONSTRAINTS")
    private <T> T create(Class<T> type, @Nullable Collection<ParameterEntity> parameterEntities) {
        T component;
        try {
            component = type.getConstructor().newInstance();
            if (component instanceof Configurable configurable) {
                configurable.setParameters(buildParameters(
                        configurable, ObjectUtils.defaultIfNull(parameterEntities, Collections.emptyList())));
            }
            componentPostCreationService.processComponentPostCreation(component);
        } catch (InstantiationException
                | IllegalAccessException
                | InvocationTargetException
                | NoSuchMethodException e) {
            throw new ComponentCreationException(type, e);
        }
        return component;
    }

    private Object getParameterValue(ParameterDefinition parameterDefinition, ParameterEntity parameterEntity) {
        if (ParameterType.SPRING_BEAN == parameterDefinition.getType()) {
            return beanFactory.getBean(parameterEntity.getStringValue());
        } else {
            return parameterConverterUtils
                    .getTypeConverter(parameterDefinition.getType())
                    .toContextValue(parameterEntity);
        }
    }

    public Parameters buildParameters(Configurable configurable, Collection<ParameterEntity> parameterEntities) {
        return buildParameters(configurable, parameterEntities, true);
    }

    public Parameters buildParameters(
            Configurable configurable, Collection<ParameterEntity> parameterEntities, boolean addDefault) {
        Parameters parameters = new Parameters();

        if (CollectionUtils.isNotEmpty(configurable.parameterDefinitions())) {
            for (ParameterDefinition parameterDefinition : configurable.parameterDefinitions()) {
                Optional<ParameterEntity> optionalEntity = parameterEntities.stream()
                        .filter(parameterEntity -> parameterEntity.getName().equals(parameterDefinition.getName()))
                        .findFirst();

                if (optionalEntity.isPresent()) {
                    Object value = getParameterValue(parameterDefinition, optionalEntity.orElseThrow());

                    LOGGER.trace(
                            "add parameter {} with value {} to component {}",
                            parameterDefinition.getName(),
                            value,
                            configurable);
                    parameters.put(parameterDefinition.getName(), value);
                } else if (addDefault) {
                    Object value = parameterDefinition.getDefaultValue();

                    LOGGER.trace(
                            "add default parameter {} with value {} to component {}",
                            parameterDefinition.getName(),
                            value,
                            configurable);
                    parameters.put(parameterDefinition.getName(), value);
                }
            }
        }

        return parameters;
    }

    @Override
    public void setBeanFactory(BeanFactory beanFactory) {
        synchronized (this) {
            this.beanFactory = beanFactory;
        }
    }
}
