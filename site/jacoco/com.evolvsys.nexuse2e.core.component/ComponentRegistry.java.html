package com.evolvsys.nexuse2e.core.component;

import com.evolvsys.common.bundle.XMLResourceBundleUtils;
import com.evolvsys.nexuse2e.core.annotation.NexusComponent;
import com.evolvsys.nexuse2e.core.config.NEXUSe2eConfigurationProperties;
import io.github.classgraph.*;
import java.util.*;
import lombok.AccessLevel;
import lombok.Getter;
import lombok.RequiredArgsConstructor;
import lombok.Setter;
import lombok.extern.slf4j.Slf4j;
import org.apache.commons.lang3.ObjectUtils;
import org.apache.commons.lang3.StringUtils;
import org.apache.commons.lang3.Validate;
import org.springframework.beans.factory.InitializingBean;
import org.springframework.stereotype.Component;

/**
 * @author SvenBarden
 * @since 09.10.2023
 */
@RequiredArgsConstructor
@Component
@Slf4j
public class ComponentRegistry implements InitializingBean {
    private final Map<String, RegisteredNexusComponent> componentMap = new HashMap<>();
    private static final String DEFAULT_BUNDLE = "i18n/component";

    @Setter(AccessLevel.PACKAGE)
    @Getter
    private String defaultBundle = DEFAULT_BUNDLE;

    private final NEXUSe2eConfigurationProperties nexuSe2eConfigurationProperties;

    public static Locale getDefaultLocale() {
        return Locale.ENGLISH;
    }

    private static String getI18n(String bundle, String key, Locale locale) {
        try {
            return XMLResourceBundleUtils.getBundle(bundle, ObjectUtils.defaultIfNull(locale, getDefaultLocale()))
                    .getString(key);
        } catch (MissingResourceException e) {
            LOGGER.debug("could not find resource key for {} in bundle {}", key, bundle);
            return "";
        }
    }

    private String getI18nText(AnnotationParameterValueList parameterValues, String componentName, String baseName) {

        String text = Optional.ofNullable(parameterValues.getValue(baseName))
                .map(Object::toString)
                .orElse(null);

        if (StringUtils.isBlank(text)) {
            String key = parameterValues.getValue(baseName + "Key").toString();

            if (StringUtils.isBlank(key)) {
                key = String.format("component.%s.%s", componentName, baseName);
            }

            String bundle = StringUtils.defaultIfBlank(
                    parameterValues.getValue("bundle").toString(), getDefaultBundle());

            text = getI18n(bundle, key, getDefaultLocale());
        }

        return text;
    }

    public Optional<Class<?>> get(String componentId) {
        return Optional.ofNullable(componentMap.get(componentId)).map(RegisteredNexusComponent::getLoadedClass);
    }

    public Optional<RegisteredNexusComponent> getComponent(String componentId) {
        return Optional.ofNullable(componentMap.get(componentId));
    }

    public void register(RegisteredNexusComponent nexusComponent) {
        Validate.notNull(nexusComponent.getComponentId(), "id must not be null");
        if (componentMap.containsKey(nexusComponent.getComponentId())) {
            throw new IllegalArgumentException(
                    "component with id " + nexusComponent.getComponentId() + " is already registered");
        }
        LOGGER.debug(
                "registering component {} of type {}",
                nexusComponent.getComponentId(),
                nexusComponent.getComponentType());
        componentMap.put(nexusComponent.getComponentId(), nexusComponent);
    }

    public List<RegisteredNexusComponent> getRegisteredComponents() {
        return componentMap.values().stream().toList();
    }

    public List<RegisteredNexusComponent> getRegisteredComponents(ComponentType componentType) {
        return componentMap.values().stream()
                .filter(component -> component.getComponentType() == componentType)
                .toList();
    }

    public boolean exists(String componentName) {
        return componentMap.keySet().stream()
                .anyMatch(registeredComponentName -> registeredComponentName.equals(componentName));
    }

    @Override
    public void afterPropertiesSet() throws Exception {
        LOGGER.debug("scanning for components");
        Set<String> additionalComponentPackages = nexuSe2eConfigurationProperties.getAdditionalComponentPackages();
        Set<String> componentPackages = new HashSet<>();
        componentPackages.add("com.evolvsys.nexuse2e");
        if (additionalComponentPackages != null) {
            componentPackages.addAll(additionalComponentPackages);
        }
        try (ScanResult scanResult = new ClassGraph()
                .enableAllInfo()
                .acceptPackages(componentPackages.toArray(new String[0]))
                .scan()) {
            ClassInfoList classInfos = scanResult.getClassesWithAnnotation(NexusComponent.class);
            for (ClassInfo classInfo : classInfos) {

                String className = classInfo.getName();

                if (classInfo.isAbstract()) {
                    LOGGER.trace("skip abstract class {}", className);
                    continue;
                }

                AnnotationParameterValueList parameterValues =
                        classInfo.getAnnotationInfo(NexusComponent.class).getParameterValues();

                String componentId = parameterValues.getValue("value").toString();
                if ("".equals(componentId)) {
                    componentId = classInfo.getSimpleName();
                }

                Optional<ComponentType> componentType = getComponentType(classInfo);
                if (componentType.isEmpty()) {
                    LOGGER.error("Unable to resolve type for component of class '{}'", classInfo.getSimpleName());
                    continue;
                }

                LOGGER.trace("identified component {} as {}", componentId, componentType.orElseThrow());

                String description = getI18nText(parameterValues, componentId, "description");
                String helpText = getI18nText(parameterValues, componentId, "helpText");

                Class<?> loadedClass = classInfo.loadClass();

                RegisteredNexusComponent.RegisteredNexusComponentBuilder registeredNexusComponentBuilder =
                        RegisteredNexusComponent.builder()
                                .componentId(componentId)
                                .componentType(componentType.orElseThrow())
                                .loadedClass(loadedClass)
                                .description(description)
                                .helpText(helpText);

                if (Configurable.class.isAssignableFrom(loadedClass)) {
                    try {
                        Configurable configurable = (Configurable)
                                classInfo.loadClass().getConstructor().newInstance();

                        registeredNexusComponentBuilder.parameterDefinitions(ObjectUtils.defaultIfNull(
                                configurable.parameterDefinitions(), Collections.emptyList()));
                    } catch (NoSuchMethodException e) {
                        LOGGER.error(
                                "Class '{}' could not be instantiated as a Configurable", loadedClass.getSimpleName());
                    }
                }

                RegisteredNexusComponent registeredNexusComponent = registeredNexusComponentBuilder.build();

                LOGGER.trace("scanned component {}", registeredNexusComponent);

                register(registeredNexusComponent);
            }
        }
        LOGGER.debug("finished component scanning");
    }

    private Optional<ComponentType> getComponentType(ClassInfo classInfo) {
        if (classInfo.implementsInterface(InflowService.class)) {
            return Optional.of(ComponentType.INFLOW_SERVICE);
        }
        if (classInfo.implementsInterface(OutflowService.class)) {
            return Optional.of(ComponentType.OUTFLOW_SERVICE);
        }
        if (classInfo.extendsSuperclass(Pipelet.class)) {
            return Optional.of(ComponentType.PIPELET);
        }
        if (classInfo.implementsInterface(SupportService.class)) {
            return Optional.of(ComponentType.SUPPORT_SERVICE);
        }
        return Optional.empty();
    }
}
