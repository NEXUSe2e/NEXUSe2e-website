package com.evolvsys.nexuse2e.core.keyvalue.persistence.mapper;

import com.evolvsys.app.core.mapper.BaseDtoEntityMapper;
import com.evolvsys.app.core.mapper.DefaultMapperConfiguration;
import com.evolvsys.app.core.mapper.interceptor.InterceptingMappingContext;
import com.evolvsys.nexuse2e.core.keyvalue.persistence.model.KeyValuePropertyDto;
import com.evolvsys.nexuse2e.core.keyvalue.persistence.model.KeyValuePropertyEntity;
import com.evolvsys.nexuse2e.core.keyvalue.persistence.repository.KeyValuePropertyRepository;
import org.mapstruct.Context;
import org.mapstruct.InheritConfiguration;
import org.mapstruct.Mapper;
import org.mapstruct.MappingTarget;
import org.springframework.beans.factory.annotation.Autowired;

/**
 * @author SvenBarden
 * @since 17.06.2024
 */
@Mapper(config = DefaultMapperConfiguration.class)
public abstract class KeyValuePropertyMapper extends BaseDtoEntityMapper<KeyValuePropertyDto, KeyValuePropertyEntity> {
    protected KeyValuePropertyRepository keyValuePropertyRepository;

    @Autowired
    protected void config(KeyValuePropertyRepository keyValuePropertyRepository) {
        this.keyValuePropertyRepository = keyValuePropertyRepository;
    }

    @Override
    public KeyValuePropertyEntity idToEntity(Long id, @Context InterceptingMappingContext interceptingMappingContext) {
        return idToEntity(id, interceptingMappingContext, keyValuePropertyRepository::findById);
    }

    @InheritConfiguration
    @Override
    protected abstract KeyValuePropertyDto entityToDto(
            KeyValuePropertyEntity entity, @Context InterceptingMappingContext interceptingMappingContext);

    @Override
    @InheritConfiguration
    protected abstract void updateEntity(
            KeyValuePropertyDto dto,
            @MappingTarget KeyValuePropertyEntity entity,
            @Context InterceptingMappingContext interceptingMappingContext);
}
