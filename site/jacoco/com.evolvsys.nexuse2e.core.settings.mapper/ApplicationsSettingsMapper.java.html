package com.evolvsys.nexuse2e.core.settings.mapper;

import com.evolvsys.app.core.mapper.BaseDtoEntityMapper;
import com.evolvsys.app.core.mapper.DefaultMapperConfiguration;
import com.evolvsys.app.core.mapper.interceptor.InterceptingMappingContext;
import com.evolvsys.app.core.tenant.mapper.TenantMapper;
import com.evolvsys.nexuse2e.core.settings.model.ApplicationSettingsDto;
import com.evolvsys.nexuse2e.core.settings.model.ApplicationSettingsEntity;
import com.evolvsys.nexuse2e.core.settings.repository.ApplicationSettingsRepository;
import org.mapstruct.*;
import org.springframework.beans.factory.annotation.Autowired;

/**
 * @author SvenBarden
 * @since 27.03.2024
 */
@Mapper(config = DefaultMapperConfiguration.class, uses = TenantMapper.class)
public abstract class ApplicationsSettingsMapper
        extends BaseDtoEntityMapper<ApplicationSettingsDto, ApplicationSettingsEntity> {
    private ApplicationSettingsRepository applicationSettingsRepository;

    @Autowired
    protected void configure(ApplicationSettingsRepository applicationSettingsRepository) {
        this.applicationSettingsRepository = applicationSettingsRepository;
    }

    @Override
    public ApplicationSettingsEntity idToEntity(Long id, InterceptingMappingContext interceptingMappingContext) {
        return idToEntity(id, interceptingMappingContext, applicationSettingsRepository::findById);
    }

    @Mapping(target = "tenantId", source = "tenant.id")
    @InheritConfiguration
    @Override
    protected abstract ApplicationSettingsDto entityToDto(
            ApplicationSettingsEntity entity, @Context InterceptingMappingContext interceptingMappingContext);

    @Mapping(target = "tenant", source = "tenantId")
    @InheritConfiguration
    @Override
    protected abstract void updateEntity(
            ApplicationSettingsDto dto,
            @MappingTarget ApplicationSettingsEntity entity,
            @Context InterceptingMappingContext interceptingMappingContext);
}
