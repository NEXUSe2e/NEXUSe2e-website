package com.evolvsys.nexuse2e.core.pipeline.mapper;

import com.evolvsys.app.core.auth.model.CurrentUserData;
import com.evolvsys.app.core.mapper.BaseDtoEntityMapper;
import com.evolvsys.app.core.mapper.DefaultMapperConfiguration;
import com.evolvsys.app.core.mapper.interceptor.InterceptingMappingContext;
import com.evolvsys.app.core.persistence.model.IdDto;
import com.evolvsys.nexuse2e.core.model.entity.PipeletEntity;
import com.evolvsys.nexuse2e.core.pipelet.mapper.PipeletMapper;
import com.evolvsys.nexuse2e.core.pipelet.model.PipeletDto;
import com.evolvsys.nexuse2e.core.pipeline.model.PipelineDto;
import com.evolvsys.nexuse2e.core.pipeline.model.PipelineEntity;
import com.evolvsys.nexuse2e.core.pipeline.repository.PipelineRepository;
import java.util.List;
import java.util.Objects;
import java.util.Set;
import java.util.stream.Collectors;
import org.mapstruct.*;
import org.springframework.beans.factory.annotation.Autowired;

/**
 * @author Florian Drees
 * @since 20.11.2023
 */
@Mapper(config = DefaultMapperConfiguration.class)
public abstract class PipelineMapper extends BaseDtoEntityMapper<PipelineDto, PipelineEntity> {
    private PipelineRepository pipelineRepository;
    private PipeletMapper pipeletMapper;

    @Autowired
    protected void config(PipelineRepository pipelineRepository, PipeletMapper pipeletMapper) {
        this.pipelineRepository = pipelineRepository;
        this.pipeletMapper = pipeletMapper;
    }

    @Override
    public PipelineEntity idToEntity(Long id, @Context InterceptingMappingContext interceptingMappingContext) {
        return idToEntity(id, interceptingMappingContext, pipelineRepository::findById);
    }

    @InheritConfiguration
    @Override
    protected abstract PipelineDto entityToDto(
            PipelineEntity entity, @Context InterceptingMappingContext interceptingMappingContext);

    @Mapping(target = "pipelets", ignore = true)
    @InheritConfiguration
    @Override
    protected abstract void updateEntity(
            PipelineDto dto,
            @MappingTarget PipelineEntity entity,
            @Context InterceptingMappingContext interceptingMappingContext);

    public void updatePipelets(List<PipeletDto> pipeletDtos, PipelineEntity entity, CurrentUserData currentUserData) {
        Set<Long> pipeletIds =
                pipeletDtos.stream().map(IdDto::getId).filter(Objects::nonNull).collect(Collectors.toSet());

        entity.getPipelets().removeIf(pipeletEntity -> pipeletIds.contains(pipeletEntity.getId()) == false);

        for (int i = 0; i < pipeletDtos.size(); i++) {
            PipeletDto pipeletDto = pipeletDtos.get(i);

            PipeletEntity pipeletEntity = entity.getPipelets().stream()
                    .filter(p -> p.getId() != null && p.getId().equals(pipeletDto.getId()))
                    .findFirst()
                    .orElseGet(PipeletEntity::new);

            if (pipeletEntity.getId() == null) {
                entity.getPipelets().add(pipeletEntity);
            }
            pipeletDto.setOrder(i + 1);

            pipeletMapper.updateEntity(pipeletDto, pipeletEntity, currentUserData);
        }
    }
}
