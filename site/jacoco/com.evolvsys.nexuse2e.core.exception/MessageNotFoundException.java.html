package com.evolvsys.nexuse2e.core.exception;

/**
 * @author SvenBarden
 * @since 19.10.2023
 */
public class MessageNotFoundException extends NexusRuntimeException {
    public MessageNotFoundException(String messageId) {
        this(messageId, null);
    }

    public MessageNotFoundException(String messageId, Throwable cause) {
        super(
                NexusError.builder()
                        .message(String.format("Message with Id '%s' could not be found", messageId))
                        .code("MESSAGE_NOT_FOUND")
                        .extension("messageId", messageId)
                        .build(),
                cause);
    }
}
