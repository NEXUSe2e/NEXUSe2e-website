package com.evolvsys.nexuse2e.core.exception;

import com.evolvsys.app.core.exception.ErrorCodeException;
import lombok.Getter;
import lombok.ToString;

/**
 * @author SvenBarden
 * @since 17.10.2023
 */
@ToString
@Getter
public non-sealed class NexusException extends Exception implements NexusErrorContainer, ErrorCodeException {
    private final NexusError error;

    public NexusException(NexusError error) {
        this.error = error;
    }

    public NexusException(NexusError error, Throwable cause) {
        super(cause);
        this.error = error;
    }

    @Override
    public String getMessage() {
        return error.toString();
    }
}
