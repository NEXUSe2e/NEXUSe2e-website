package com.evolvsys.nexuse2e.core.parameter.model;

import com.evolvsys.nexuse2e.core.exception.NexusError;
import com.evolvsys.nexuse2e.core.exception.NexusRuntimeException;
import lombok.Builder;
import lombok.Value;

/**
 * @author SvenBarden
 * @since 02.04.2024
 */
@Value
@Builder
public class ActionInfo {
    public static final String SEPARATOR = "_~~_";
    String choreographyId;
    String actionId;

    public static ActionInfo of(String parse) {
        String[] split = parse.split(SEPARATOR);
        if (split.length != 2) {
            throw new NexusRuntimeException(NexusError.builder()
                    .message("ActionInfo string '" + parse + " 'was invalid")
                    .extension("string", parse)
                    .extension("separator", SEPARATOR)
                    .build());
        }
        return new ActionInfo(split[0], split[1]);
    }

    public String format() {
        return choreographyId + SEPARATOR + actionId;
    }
}
