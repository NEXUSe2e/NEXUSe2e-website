package com.evolvsys.nexuse2e.core.message.payload.mapper;

import com.evolvsys.app.core.mapper.BaseDtoEntityMapper;
import com.evolvsys.app.core.mapper.DefaultMapperConfiguration;
import com.evolvsys.app.core.mapper.interceptor.InterceptingMappingContext;
import com.evolvsys.nexuse2e.core.message.payload.model.PayloadDto;
import com.evolvsys.nexuse2e.core.message.payload.model.PayloadEntity;
import com.evolvsys.nexuse2e.core.message.payload.repository.PayloadRepository;
import org.mapstruct.*;
import org.springframework.beans.factory.annotation.Autowired;

/**
 * <p>{@link Mapper} f√ºr {@link PayloadEntity} und {@link PayloadDto}.</p>
 *
 * @author Florian Drees
 * @since 29.02.2024
 */
@Mapper(config = DefaultMapperConfiguration.class)
public abstract class PayloadMapper extends BaseDtoEntityMapper<PayloadDto, PayloadEntity> {
    private PayloadRepository payloadRepository;

    @Autowired
    protected void config(PayloadRepository payloadRepository) {
        this.payloadRepository = payloadRepository;
    }

    @Override
    public PayloadEntity idToEntity(Long id, @Context InterceptingMappingContext interceptingMappingContext) {
        return idToEntity(id, interceptingMappingContext, payloadRepository::findById);
    }

    @Mapping(target = "downloadName", ignore = true)
    @Mapping(target = "binaryId", source = "content.id")
    @InheritConfiguration
    @Override
    protected abstract PayloadDto entityToDto(
            PayloadEntity entity, @Context InterceptingMappingContext interceptingMappingContext);

    @Mapping(target = "metaDatas", ignore = true)
    @Mapping(target = "content", ignore = true)
    @Override
    protected abstract void updateEntity(
            PayloadDto dto,
            @MappingTarget PayloadEntity entity,
            @Context InterceptingMappingContext interceptingMappingContext);
}
