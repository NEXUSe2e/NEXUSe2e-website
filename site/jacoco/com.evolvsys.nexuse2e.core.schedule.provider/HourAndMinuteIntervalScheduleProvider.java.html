package com.evolvsys.nexuse2e.core.schedule.provider;

import com.evolvsys.nexuse2e.core.exception.NexusError;
import com.evolvsys.nexuse2e.core.exception.NexusRuntimeException;
import java.util.regex.Matcher;
import java.util.regex.Pattern;
import lombok.RequiredArgsConstructor;
import org.apache.commons.lang3.StringUtils;
import org.quartz.CronScheduleBuilder;

/**
 * @author Florian Drees
 * @since 30.01.2024
 */
@RequiredArgsConstructor
public class HourAndMinuteIntervalScheduleProvider implements ScheduleProvider<CronScheduleBuilder> {
    private static final String REGEX = "^([0-9]|0[0-9]|1[0-9]|2[0-3]):([0-5][0-9])$";
    private static final Pattern PATTERN = Pattern.compile(REGEX);

    private final String pattern;
    private final MisfireInstruction misfireInstruction;

    public static HourAndMinuteIntervalScheduleProvider create(String pattern) {
        return create(pattern, MisfireInstruction.NEXT_SCHEDULE);
    }

    public static HourAndMinuteIntervalScheduleProvider create(String pattern, MisfireInstruction misfireInstruction) {
        if (StringUtils.isNotEmpty(pattern) && pattern.matches(REGEX)) {
            return new HourAndMinuteIntervalScheduleProvider(pattern, misfireInstruction);
        } else {
            throw new IllegalArgumentException("The value for the hour and minute pattern is not valid: " + pattern);
        }
    }

    @Override
    public CronScheduleBuilder getScheduleBuilder() {
        Matcher matcher = PATTERN.matcher(pattern);
        if (matcher.find()) {
            CronScheduleBuilder cronScheduleBuilder = CronScheduleBuilder.dailyAtHourAndMinute(
                    Integer.parseInt(matcher.group(1)), Integer.parseInt(matcher.group(2)));
            return switch (misfireInstruction) {
                case NEXT_SCHEDULE -> cronScheduleBuilder.withMisfireHandlingInstructionDoNothing();
                case FIRE_ONCE -> cronScheduleBuilder.withMisfireHandlingInstructionFireAndProceed();
            };
        } else {
            throw new NexusRuntimeException(NexusError.builder()
                    .message("invalid hour and minute pattern")
                    .extension("pattern", pattern)
                    .extension("required pattern", REGEX)
                    .build());
        }
    }
}
