package com.evolvsys.nexuse2e.core.schedule.provider;

import lombok.RequiredArgsConstructor;
import org.quartz.CronExpression;
import org.quartz.CronScheduleBuilder;

/**
 * @author Florian Drees
 * @since 30.01.2024
 */
@RequiredArgsConstructor
public class CronScheduleProvider implements ScheduleProvider<CronScheduleBuilder> {
    private final String cron;
    private final MisfireInstruction misfireInstruction;

    public static CronScheduleProvider create(String cron) {
        return create(cron, MisfireInstruction.NEXT_SCHEDULE);
    }

    public static CronScheduleProvider create(String cron, MisfireInstruction misfireInstruction) {
        if (CronExpression.isValidExpression(cron)) {
            return new CronScheduleProvider(cron, misfireInstruction);
        } else {
            throw new IllegalArgumentException("The provided cron pattern is not valid: " + cron);
        }
    }

    @Override
    public CronScheduleBuilder getScheduleBuilder() {
        CronScheduleBuilder cronScheduleBuilder = CronScheduleBuilder.cronSchedule(cron);
        return switch (misfireInstruction) {
            case FIRE_ONCE -> cronScheduleBuilder.withMisfireHandlingInstructionFireAndProceed();
            case NEXT_SCHEDULE -> cronScheduleBuilder.withMisfireHandlingInstructionDoNothing();
        };
    }
}
