package com.evolvsys.nexuse2e.core.schedule.provider;

import lombok.RequiredArgsConstructor;
import lombok.extern.slf4j.Slf4j;
import org.quartz.SimpleScheduleBuilder;

/**
 * @author Florian Drees
 * @since 30.01.2024
 */
@RequiredArgsConstructor
@Slf4j
public class MillisecondScheduleProvider implements ScheduleProvider<SimpleScheduleBuilder> {
    private final Long milliseconds;
    private final MisfireInstruction misfireInstruction;

    public static MillisecondScheduleProvider create(Long milliseconds) {
        return create(milliseconds, MisfireInstruction.NEXT_SCHEDULE);
    }

    public static MillisecondScheduleProvider create(Long milliseconds, MisfireInstruction misfireInstruction) {
        if (milliseconds != null && milliseconds > 0) {
            return new MillisecondScheduleProvider(milliseconds, misfireInstruction);
        } else {
            throw new IllegalArgumentException("scheduling with " + milliseconds + " milliseconds is not valid");
        }
    }

    @Override
    public SimpleScheduleBuilder getScheduleBuilder() {
        SimpleScheduleBuilder simpleScheduleBuilder = SimpleScheduleBuilder.simpleSchedule()
                .withIntervalInMilliseconds(milliseconds)
                .repeatForever();
        return switch (misfireInstruction) {
            case FIRE_ONCE -> simpleScheduleBuilder.withMisfireHandlingInstructionFireNow();
            case NEXT_SCHEDULE -> simpleScheduleBuilder.withMisfireHandlingInstructionNextWithExistingCount();
        };
    }
}
