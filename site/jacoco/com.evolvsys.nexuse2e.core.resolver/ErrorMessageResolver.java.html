package com.evolvsys.nexuse2e.core.resolver;

import com.evolvsys.nexuse2e.core.exception.NexusException;
import com.evolvsys.nexuse2e.core.message.context.MessageContext;
import com.evolvsys.nexuse2e.core.pipeline.model.Pipeline;
import com.evolvsys.nexuse2e.internal.MessageType;
import lombok.RequiredArgsConstructor;
import lombok.extern.slf4j.Slf4j;

/**
 * @author SvenBarden
 * @since 01.03.2024
 */
@Slf4j
@RequiredArgsConstructor
public class ErrorMessageResolver extends MessageResolver {
    private final Pipeline pipeline;

    @Override
    public MessageContext resolve(MessageContext messageContext) throws NexusException {
        LOGGER.debug("running error message resolver for pipeline {}", pipeline);

        if (pipeline != null) {
            messageContext = pipeline.execute(messageContext.createExecutionContext(MessageType.ERROR))
                    .getMessageContext();
        }

        return messageContext;
    }
}
