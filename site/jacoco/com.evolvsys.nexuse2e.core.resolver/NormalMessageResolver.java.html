package com.evolvsys.nexuse2e.core.resolver;

import com.evolvsys.nexuse2e.core.exception.NexusError;
import com.evolvsys.nexuse2e.core.exception.NexusException;
import com.evolvsys.nexuse2e.core.message.MetaData;
import com.evolvsys.nexuse2e.core.message.context.ContextFactory;
import com.evolvsys.nexuse2e.core.message.context.ExecutionContext;
import com.evolvsys.nexuse2e.core.message.context.MessageContext;
import com.evolvsys.nexuse2e.core.message.state.MessageEventService;
import com.evolvsys.nexuse2e.core.message.state.MessageEventType;
import com.evolvsys.nexuse2e.core.pipeline.model.Pipeline;
import com.evolvsys.nexuse2e.core.route.model.ContextRoute;
import com.evolvsys.nexuse2e.internal.ContextPayload;
import com.evolvsys.nexuse2e.internal.Message;
import com.evolvsys.nexuse2e.internal.MessageType;
import java.nio.charset.Charset;
import java.time.ZonedDateTime;
import java.util.LinkedHashSet;
import java.util.Optional;
import java.util.Set;
import lombok.RequiredArgsConstructor;
import lombok.extern.slf4j.Slf4j;

/**
 * @author SvenBarden
 * @since 01.03.2024
 */
@Slf4j
@RequiredArgsConstructor
public class NormalMessageResolver extends MessageResolver {
    private final Pipeline pipeline;
    private final MessageType messageType;
    private final MessageResolverValidator messageResolverValidator;
    private final MessageEventService messageEventService;
    private final ContextFactory contextFactory;

    @Override
    public MessageContext resolve(MessageContext messageContext) throws NexusException {
        LOGGER.debug("running message resolver for for pipeline {}", pipeline);
        if (LOGGER.isTraceEnabled()) {
            for (ContextPayload sourceData :
                    messageContext.createExecutionContext(messageType).getDataAsList()) {
                LOGGER.trace(
                        "source dataReceived {} with content: {}",
                        sourceData,
                        sourceData.getContent().getString(Charset.forName(sourceData.getEncoding())));
            }
        }

        if (pipeline != null) {
            messageContext = pipeline.execute(messageContext.createExecutionContext(messageType))
                    .getMessageContext();
        }

        ExecutionContext executionContext = messageContext.createExecutionContext(messageType);

        Message message = executionContext.getMessage();

        if (message.getId() == null) {
            LOGGER.trace("generate new message id");
            message.setId(messageContext.getMessageIdGenerator().generateId());
        }
        if (message.getConversationId() == null) {
            LOGGER.trace("generate new conversation id");
            message.setConversationId(
                    messageContext.getConversationIdGenerator().generateId());
        }

        messageEventService.trigger(executionContext, MessageEventType.RESOLVING);

        executionContext = runRouteInflowPipelines(executionContext);

        messageResolverValidator.validate(messageContext);

        LOGGER.debug(
                "processed message with {} payloads with messageType {}",
                message.getPayloads().size(),
                messageType);

        if (contextFactory.exists(message.getId())) {
            LOGGER.debug("message already exists, building context from database");
            MetaData metaData = messageContext.getMetaData();
            messageContext = contextFactory.createForInboundMessage(message.getId());
            messageContext.getMetaData().set(metaData);
        } else {
            LOGGER.debug("message is new, triggering resolved");
            message.setProcessed(ZonedDateTime.now());

            messageEventService.trigger(executionContext, MessageEventType.RESOLVED);
        }

        return messageContext;
    }

    private ExecutionContext runRouteInflowPipelines(ExecutionContext executionContext) throws NexusException {
        LOGGER.trace("running route inflow pipelines");
        String lastRouteId;
        Set<String> usedRouteIds = new LinkedHashSet<>();
        do {
            ContextRoute newRoute = executionContext.getRoute();
            lastRouteId = newRoute == null ? null : newRoute.getRouteId();
            if (lastRouteId != null) {
                if (usedRouteIds.add(lastRouteId) == false) {
                    throw new NexusException(NexusError.builder()
                            .message("Route circle configuration found")
                            .extension("routes", usedRouteIds)
                            .extension("choreographyId", executionContext.getChoreographyId())
                            .extension("actionId", executionContext.getActionId())
                            .build());
                }
                LOGGER.info(
                        "using route '{}' from '{}' to '{}' for '{}/{}'",
                        lastRouteId,
                        executionContext.getSourcePartnerId(),
                        executionContext.getTargetPartnerId(),
                        executionContext.getChoreography().getChoreographyId(),
                        executionContext.getAction().getActionId());
                Pipeline inflowPipeline = newRoute.getInflowPipeline();
                if (inflowPipeline != null) {
                    LOGGER.trace("running route inflow pipeline {}", inflowPipeline.getPipelineId());
                    executionContext = inflowPipeline.execute(executionContext);
                    messageEventService.trigger(executionContext, MessageEventType.POST_ROUTE_RESOLVING);
                }
            }
        } while (lastRouteId != null
                && (lastRouteId.equals(Optional.ofNullable(executionContext.getRoute())
                                .map(ContextRoute::getRouteId)
                                .orElse(null))
                        == false));

        return executionContext;
    }
}
