package com.evolvsys.nexuse2e.core.resolver;

import com.evolvsys.nexuse2e.core.exception.NexusError;
import com.evolvsys.nexuse2e.core.exception.NexusException;
import com.evolvsys.nexuse2e.core.message.context.MessageContext;
import com.evolvsys.nexuse2e.internal.MessageContextValidationGroups;
import jakarta.validation.ConstraintViolation;
import jakarta.validation.ValidatorFactory;
import jakarta.validation.groups.Default;
import java.util.*;
import lombok.RequiredArgsConstructor;
import org.apache.commons.lang3.ObjectUtils;
import org.springframework.core.annotation.AnnotationUtils;
import org.springframework.stereotype.Component;

/**
 * @author SvenBarden
 * @since 23.01.2024
 */
@RequiredArgsConstructor
@Component
public class MessageResolverValidator {
    private final ValidatorFactory validatorFactory;

    public void validate(MessageContext messageContext, Class<?>... groups) throws NexusException {
        Set<ConstraintViolation<MessageContext>> violations =
                validatorFactory.getValidator().validate(messageContext, groups);

        List<ConstraintViolation<MessageContext>> violationList = violations.stream()
                .sorted((o1, o2) -> {
                    Integer order1 = (Integer) AnnotationUtils.getValue(
                            o1.getConstraintDescriptor().getAnnotation(), "order");
                    Integer order2 = (Integer) AnnotationUtils.getValue(
                            o2.getConstraintDescriptor().getAnnotation(), "order");

                    return ObjectUtils.defaultIfNull(order1, Integer.MAX_VALUE)
                            .compareTo(ObjectUtils.defaultIfNull(order2, Integer.MAX_VALUE));
                })
                .toList();

        // AnnotationUtils.getValue()
        List<Map<String, Object>> constraintViolations = new ArrayList<>(violationList.size());

        ConstraintViolation<MessageContext> first = null;
        for (ConstraintViolation<MessageContext> violation : violationList) {
            if (first == null) {
                first = violation;
            }
            Map<String, Object> map = LinkedHashMap.newLinkedHashMap(3);
            map.put("message", violation.getMessage());
            map.put("key", violation.getMessageTemplate());
            map.put("path", violation.getPropertyPath().toString());
            constraintViolations.add(map);
        }

        if (first != null) {
            NexusError nexusError = NexusError.builder()
                    .message(first.getMessage())
                    .code("RESOLVER_VALIDATION")
                    .extension("key", first.getMessageTemplate())
                    .extension("path", first.getPropertyPath().toString())
                    .extension("violations", constraintViolations.toString())
                    .build();

            throw new NexusException(nexusError);
        }
    }

    public void validate(MessageContext messageContext) throws NexusException {
        validate(messageContext, Default.class, MessageContextValidationGroups.Resolver.class);
    }
}
