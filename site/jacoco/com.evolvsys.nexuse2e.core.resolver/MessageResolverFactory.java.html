package com.evolvsys.nexuse2e.core.resolver;

import com.evolvsys.nexuse2e.core.message.context.ContextFactory;
import com.evolvsys.nexuse2e.core.message.state.MessageEventService;
import com.evolvsys.nexuse2e.core.pipeline.PipelineFactory;
import com.evolvsys.nexuse2e.core.pipeline.model.PipelineEntity;
import com.evolvsys.nexuse2e.internal.MessageType;
import lombok.RequiredArgsConstructor;
import lombok.extern.slf4j.Slf4j;
import org.springframework.lang.Nullable;
import org.springframework.stereotype.Component;

/**
 * @author SvenBarden
 * @since 23.01.2024
 */
@Component
@RequiredArgsConstructor
@Slf4j
public class MessageResolverFactory {
    private final MessageResolverValidator messageResolverValidator;
    private final PipelineFactory pipelineFactory;
    private final MessageEventService messageEventService;
    private final ContextFactory contextFactory;

    public MessageResolver create(@Nullable PipelineEntity pipeline, MessageType messageType) {
        return switch (messageType) {
            case ERROR -> new ErrorMessageResolver(pipeline == null ? null : pipelineFactory.get(pipeline));
            case RETURN -> new ReturnMessageResolver(pipeline == null ? null : pipelineFactory.get(pipeline));
            default ->
                new NormalMessageResolver(
                        pipeline == null ? null : pipelineFactory.get(pipeline),
                        messageType,
                        messageResolverValidator,
                        messageEventService,
                        contextFactory);
        };
    }
}
