package com.evolvsys.nexuse2e.core.logging.mapper;

import com.evolvsys.app.core.mapper.BaseDtoEntityMapper;
import com.evolvsys.app.core.mapper.DefaultMapperConfiguration;
import com.evolvsys.app.core.mapper.interceptor.InterceptingMappingContext;
import com.evolvsys.nexuse2e.core.logging.model.LogDto;
import com.evolvsys.nexuse2e.core.logging.model.LogEntity;
import com.evolvsys.nexuse2e.core.logging.repository.LogRepository;
import org.mapstruct.*;
import org.springframework.beans.factory.annotation.Autowired;

/**
 * @author SvenBarden
 * @since 09.01.2024
 */
@Mapper(config = DefaultMapperConfiguration.class)
public abstract class LogMapper extends BaseDtoEntityMapper<LogDto, LogEntity> {
    private LogRepository logRepository;

    @Autowired
    protected void config(LogRepository logRepository) {
        this.logRepository = logRepository;
    }

    @Override
    public LogEntity idToEntity(Long id, InterceptingMappingContext interceptingMappingContext) {
        return idToEntity(id, interceptingMappingContext, logRepository::findById);
    }

    @Override
    @InheritConfiguration
    protected abstract LogDto entityToDto(
            LogEntity entity, @Context InterceptingMappingContext interceptingMappingContext);

    @Mapping(target = "lastUpdated", ignore = true)
    @Mapping(target = "created", ignore = true)
    @Mapping(target = "inMessageContext", ignore = true)
    @Override
    @InheritConfiguration
    protected abstract void updateEntity(
            LogDto dto,
            @MappingTarget LogEntity entity,
            @Context InterceptingMappingContext interceptingMappingContext);
}
