package com.evolvsys.nexuse2e.core.logging.mapper;

import com.evolvsys.app.core.mapper.BaseDtoEntityMapper;
import com.evolvsys.app.core.mapper.DefaultMapperConfiguration;
import com.evolvsys.app.core.mapper.interceptor.InterceptingMappingContext;
import com.evolvsys.nexuse2e.core.logging.model.LogFilterDto;
import com.evolvsys.nexuse2e.core.logging.model.LogFilterEntity;
import com.evolvsys.nexuse2e.core.logging.repository.LogFilterRepository;
import org.mapstruct.*;
import org.springframework.beans.factory.annotation.Autowired;

/**
 * @author SvenBarden
 * @since 14.03.2024
 */
@Mapper(config = DefaultMapperConfiguration.class)
public abstract class LogFilterMapper extends BaseDtoEntityMapper<LogFilterDto, LogFilterEntity> {
    private LogFilterRepository logFilterRepository;

    @Autowired
    protected void config(LogFilterRepository logFilterRepository) {
        this.logFilterRepository = logFilterRepository;
    }

    @Override
    public LogFilterEntity idToEntity(Long id, InterceptingMappingContext interceptingMappingContext) {
        return idToEntity(id, interceptingMappingContext, logFilterRepository::findById);
    }

    @Override
    @InheritConfiguration
    @Mapping(target = "messageStatuses", ignore = true)
    protected abstract LogFilterDto entityToDto(
            LogFilterEntity entity, @Context InterceptingMappingContext interceptingMappingContext);

    @Override
    protected abstract void updateEntity(
            LogFilterDto dto,
            @MappingTarget LogFilterEntity entity,
            @Context InterceptingMappingContext interceptingMappingContext);
}
