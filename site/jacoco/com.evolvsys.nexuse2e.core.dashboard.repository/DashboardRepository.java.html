package com.evolvsys.nexuse2e.core.dashboard.repository;

import com.evolvsys.nexuse2e.core.dashboard.DashboardFilter;
import com.evolvsys.nexuse2e.core.dashboard.model.ConversationStatusCountResultDto;
import com.evolvsys.nexuse2e.core.dashboard.model.LatestMessagesOfChoreographyDto;
import com.evolvsys.nexuse2e.core.dashboard.model.LatestMessagesOfPartnerDto;
import com.evolvsys.nexuse2e.core.dashboard.model.MessageStatusCountResultDto;
import com.evolvsys.nexuse2e.core.message.MessageStatus;
import com.evolvsys.nexuse2e.core.model.MessageDirection;
import com.evolvsys.nexuse2e.core.model.entity.*;
import com.evolvsys.nexuse2e.core.settings.service.ApplicationSettingsService;
import com.evolvsys.nexuse2e.internal.MessageType;
import com.querydsl.core.annotations.QueryProjection;
import com.querydsl.jpa.JPAExpressions;
import com.querydsl.jpa.JPQLQueryFactory;
import java.time.Duration;
import java.time.ZoneId;
import java.time.ZonedDateTime;
import java.time.temporal.Temporal;
import java.util.*;
import java.util.function.BiConsumer;
import lombok.RequiredArgsConstructor;
import org.springframework.lang.Nullable;
import org.springframework.stereotype.Repository;

/**
 * @author Florian Drees
 * @since 21.03.2024
 */
@Repository
@RequiredArgsConstructor
public class DashboardRepository {
    private final JPQLQueryFactory queryFactory;
    private final ApplicationSettingsService applicationSettingsService;

    public List<MessageStatusCountResultDto> getMessageCountByStatus(@Nullable DashboardFilter dashboardFilter) {
        QMessageEntity message = QMessageEntity.messageEntity;
        return queryFactory
                .select(new QDashboardRepository_MessageStatusCountResult(message.status, message.count()))
                .from(message)
                .where(
                        message.type.eq(MessageType.NORMAL),
                        message.filter(
                                dashboardFilter,
                                applicationSettingsService
                                        .getSystemSettingsOrDefault()
                                        .getTimeZone()))
                .groupBy(message.status)
                .orderBy(message.count().desc())
                .fetch()
                .stream()
                .map(MessageStatusCountResult::toDto)
                .toList();
    }

    public List<ConversationStatusCountResultDto> getConversationCountByStatus(
            @Nullable DashboardFilter dashboardFilter) {
        QConversationEntity conversation = QConversationEntity.conversationEntity;
        return queryFactory
                .select(new QDashboardRepository_ConversationStatusCountResult(
                        conversation.status, conversation.count()))
                .from(conversation)
                .where(conversation.filter(
                        dashboardFilter,
                        applicationSettingsService.getSystemSettingsOrDefault().getTimeZone()))
                .groupBy(conversation.status)
                .orderBy(conversation.count().desc())
                .fetch()
                .stream()
                .map(ConversationStatusCountResult::toDto)
                .toList();
    }

    public List<LatestMessagesOfChoreographyDto> getLatestMessagesOfChoreographies(
            @Nullable DashboardFilter dashboardFilter) {
        List<LatestMessageOfChoreographyResult> latestOutboundMessages =
                fetchLatestMessagesByChoreography(MessageDirection.INTERN_TO_EXTERN, dashboardFilter);
        List<LatestMessageOfChoreographyResult> latestInboundMessages =
                fetchLatestMessagesByChoreography(MessageDirection.EXTERN_TO_INTERN, dashboardFilter);

        Map<Long, LatestMessagesOfChoreographyDto> dtos = HashMap.newHashMap(latestOutboundMessages.size());
        for (LatestMessageOfChoreographyResult outboundResult : latestOutboundMessages) {
            LatestMessagesOfChoreographyDto outboundDto =
                    outboundResult.toDto(LatestMessagesOfChoreographyDto::setLatestOutboundMessageId);
            outboundDto.setLatestOutbound(toDuration(outboundResult.created()));
            dtos.put(outboundResult.id(), outboundDto);
        }
        for (LatestMessageOfChoreographyResult inboundResult : latestInboundMessages) {
            LatestMessagesOfChoreographyDto dto = dtos.computeIfAbsent(
                    inboundResult.id(),
                    id -> inboundResult.toDto(LatestMessagesOfChoreographyDto::setLatestInboundMessageId));
            dto.setLatestInbound(toDuration(inboundResult.created()));
        }
        return new ArrayList<>(dtos.values());
    }

    public List<LatestMessagesOfPartnerDto> getLatestMessagesOfPartners(@Nullable DashboardFilter dashboardFilter) {
        List<LatestMessageOfPartnerResult> latestOutboundMessages =
                fetchLatestMessagesByPartner(MessageDirection.INTERN_TO_EXTERN, dashboardFilter);
        List<LatestMessageOfPartnerResult> latestInboundMessages =
                fetchLatestMessagesByPartner(MessageDirection.EXTERN_TO_INTERN, dashboardFilter);

        Map<Long, LatestMessagesOfPartnerDto> dtos = HashMap.newHashMap(latestOutboundMessages.size());
        for (LatestMessageOfPartnerResult outboundResult : latestOutboundMessages) {
            LatestMessagesOfPartnerDto outboundDto =
                    outboundResult.toDto(LatestMessagesOfPartnerDto::setLatestOutboundMessageId);
            outboundDto.setLatestOutbound(toDuration(outboundResult.created()));
            dtos.put(outboundResult.id(), outboundDto);
        }
        for (LatestMessageOfPartnerResult inboundResult : latestInboundMessages) {
            LatestMessagesOfPartnerDto dto = dtos.computeIfAbsent(
                    inboundResult.id(),
                    id -> inboundResult.toDto(LatestMessagesOfPartnerDto::setLatestInboundMessageId));
            dto.setLatestInbound(toDuration(inboundResult.created()));
        }
        return new ArrayList<>(dtos.values());
    }

    private List<LatestMessageOfChoreographyResult> fetchLatestMessagesByChoreography(
            MessageDirection direction, @Nullable DashboardFilter dashboardFilter) {

        QMessageEntity message = QMessageEntity.messageEntity;
        QChoreographyEntity choreography = message.conversation.choreography;

        return queryFactory
                .select(new QDashboardRepository_LatestMessageOfChoreographyResult(
                        choreography.id, choreography.choreographyId,
                        message.id, message.created))
                .from(message)
                .where(message.created.in(JPAExpressions.select(message.created.max())
                        .from(message)
                        .where(
                                message.type.eq(MessageType.NORMAL),
                                message.direction.eq(direction),
                                message.filter(
                                        dashboardFilter,
                                        applicationSettingsService
                                                .getSystemSettingsOrDefault()
                                                .getTimeZone()))
                        .groupBy(message.conversation.choreography.id)))
                .fetch();
    }

    private List<LatestMessageOfPartnerResult> fetchLatestMessagesByPartner(
            MessageDirection direction, @Nullable DashboardFilter dashboardFilter) {
        QMessageEntity message = QMessageEntity.messageEntity;
        QPartnerEntity partner;

        if (EnumSet.of(MessageDirection.INTERN_TO_EXTERN, MessageDirection.INTERN_TO_INTERN)
                .contains(direction)) {
            partner = message.targetPartner;
        } else {
            partner = message.sourcePartner;
        }

        return queryFactory
                .select(new QDashboardRepository_LatestMessageOfPartnerResult(
                        partner.id, partner.partnerId, message.id, message.created))
                .from(message)
                .join(partner)
                .where(message.created.in(JPAExpressions.select(message.created.max())
                        .from(message)
                        .where(
                                message.type.eq(MessageType.NORMAL),
                                message.direction.eq(direction),
                                message.filter(
                                        dashboardFilter,
                                        applicationSettingsService
                                                .getSystemSettingsOrDefault()
                                                .getTimeZone()))
                        .groupBy(partner.id)))
                .fetch();
    }

    public record MessageStatusCountResult(MessageStatus messageStatus, Long count) {
        @QueryProjection
        public MessageStatusCountResult {}

        public MessageStatusCountResultDto toDto() {
            return new MessageStatusCountResultDto().setStatus(messageStatus).setCount(count);
        }
    }

    public record ConversationStatusCountResult(ConversationStatus conversationStatus, Long count) {
        @QueryProjection
        public ConversationStatusCountResult {}

        public ConversationStatusCountResultDto toDto() {
            return new ConversationStatusCountResultDto()
                    .setConversationStatus(conversationStatus)
                    .setCount(count);
        }
    }

    public record LatestMessageOfChoreographyResult(
            Long id, String choreographyId, Long messageId, ZonedDateTime created) {
        @QueryProjection
        public LatestMessageOfChoreographyResult {}

        public LatestMessagesOfChoreographyDto toDto(
                BiConsumer<LatestMessagesOfChoreographyDto, Long> messageIdSetter) {
            LatestMessagesOfChoreographyDto dto =
                    new LatestMessagesOfChoreographyDto().setId(id).setChoreographyId(choreographyId);
            messageIdSetter.accept(dto, messageId);
            return dto;
        }
    }

    public record LatestMessageOfPartnerResult(Long id, String partnerId, Long messageId, ZonedDateTime created) {
        @QueryProjection
        public LatestMessageOfPartnerResult {}

        public LatestMessagesOfPartnerDto toDto(BiConsumer<LatestMessagesOfPartnerDto, Long> messageIdSetter) {
            LatestMessagesOfPartnerDto dto =
                    new LatestMessagesOfPartnerDto().setId(id).setPartnerId(partnerId);
            messageIdSetter.accept(dto, messageId);
            return dto;
        }
    }

    private Duration toDuration(Temporal temporal) {
        return Duration.between(
                temporal,
                ZonedDateTime.now()
                        .toInstant()
                        .atZone(ZoneId.of(applicationSettingsService
                                .getSystemSettingsOrDefault()
                                .getTimeZone())));
    }
}
