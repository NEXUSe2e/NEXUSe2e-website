package com.evolvsys.nexuse2e.core.dashboard.repository;

import com.evolvsys.nexuse2e.core.dashboard.DashboardFilter;
import com.evolvsys.nexuse2e.core.model.entity.ConversationEntity;
import com.evolvsys.nexuse2e.core.model.entity.MessageEntity;
import com.evolvsys.nexuse2e.core.model.entity.QConversationEntity;
import com.evolvsys.nexuse2e.core.model.entity.QMessageEntity;
import com.evolvsys.nexuse2e.core.utils.DateRangeFilterUtils;
import com.querydsl.core.BooleanBuilder;
import com.querydsl.core.annotations.QueryDelegate;
import com.querydsl.core.types.Predicate;
import com.querydsl.core.types.dsl.DateTimePath;
import java.time.ZonedDateTime;
import org.springframework.lang.Nullable;

/**
 * @author Florian Drees
 * @since 22.03.2024
 */
public final class DashboardQueryDelegate {
    private DashboardQueryDelegate() {}

    @QueryDelegate(MessageEntity.class)
    public static Predicate filter(QMessageEntity message, @Nullable DashboardFilter dashboardFilter, String timeZone) {
        return buildPredicate(dashboardFilter, timeZone, message.created);
    }

    @QueryDelegate(ConversationEntity.class)
    public static Predicate filter(
            QConversationEntity conversation, @Nullable DashboardFilter dashboardFilter, String timeZone) {
        return buildPredicate(dashboardFilter, timeZone, conversation.created);
    }

    private static Predicate buildPredicate(
            @Nullable DashboardFilter dashboardFilter, String timeZone, DateTimePath<ZonedDateTime> datePath) {
        BooleanBuilder builder = new BooleanBuilder();
        if (dashboardFilter != null) {
            Predicate predicate = DateRangeFilterUtils.filter(
                    datePath,
                    timeZone,
                    dashboardFilter.getRange(),
                    dashboardFilter.getFrom(),
                    dashboardFilter.getTill());
            if (predicate != null) {
                builder.and(predicate);
            }
        }
        return builder;
    }
}
