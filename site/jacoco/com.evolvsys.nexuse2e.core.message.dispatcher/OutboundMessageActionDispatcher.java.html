package com.evolvsys.nexuse2e.core.message.dispatcher;

import com.evolvsys.nexuse2e.core.action.Action;
import com.evolvsys.nexuse2e.core.exception.NexusError;
import com.evolvsys.nexuse2e.core.exception.NexusException;
import com.evolvsys.nexuse2e.core.message.MessageRunner;
import com.evolvsys.nexuse2e.core.message.MessageStatus;
import com.evolvsys.nexuse2e.core.message.context.ExecutionContext;
import com.evolvsys.nexuse2e.core.message.context.MessageContext;
import com.evolvsys.nexuse2e.core.message.state.MessageEventService;
import com.evolvsys.nexuse2e.core.message.state.MessageEventType;
import com.evolvsys.nexuse2e.internal.Message;
import com.evolvsys.nexuse2e.internal.MessageType;
import java.util.EnumSet;
import java.util.Set;
import lombok.RequiredArgsConstructor;
import lombok.extern.slf4j.Slf4j;
import org.springframework.stereotype.Component;

/**
 * @author SvenBarden
 * @since 05.03.2024
 */
@RequiredArgsConstructor
@Slf4j
@Component
class OutboundMessageActionDispatcher {
    private static final Set<MessageStatus> VALID_STATUS = EnumSet.of(
            MessageStatus.FAILED,
            MessageStatus.RESOLVED,
            MessageStatus.FAILED_IN_ACTION_PIPELINE,
            MessageStatus.QUEUE_ACTION);

    private final ReturnDispatcher returnDispatcher;
    private final MessageEventService messageEventService;
    private final MessageRunner messageRunner;

    private MessageContext dispatchInternal(MessageContext messageContext) throws NexusException {
        Action action = messageContext.getAction();
        if (action.getForwardPipeline() == null) {
            LOGGER.trace("action has no forward pipeline, continuing with outbound dispatcher");
        } else {
            try {
                LOGGER.info(
                        "run action pipeline for message {}",
                        messageContext.getSourceMessage().getId());
                messageContext = action.getForwardPipeline()
                        .execute(messageContext.createExecutionContext(MessageType.NORMAL))
                        .getMessageContext();

                messageEventService.trigger(
                        messageContext.createExecutionContext(MessageType.NORMAL), MessageEventType.ACTION);
            } catch (Throwable e) {
                messageEventService.trigger(
                        messageContext.createExecutionContext(MessageType.NORMAL), MessageEventType.ACTION_ERROR);
                throw new NexusException(
                        NexusError.builder()
                                .message("error while calling action pipeline")
                                .extension(
                                        "messageId",
                                        messageContext.getSourceMessage().getId())
                                .build(),
                        e);
            }
        }

        return messageContext;
    }

    public ExecutionContext dispatch(ExecutionContext executionContext) throws NexusException {
        MessageContext messageContext = executionContext.getMessageContext();
        Message message = executionContext.getMessage();

        if (VALID_STATUS.contains(message.getMessageStatus())) {
            messageContext = dispatchInternal(messageContext);

            LOGGER.debug("called sync forward pipeline");
            messageContext = returnDispatcher.dispatch(messageContext);

        } else {
            LOGGER.debug("message in status {} will not execute action pipeline", message.getMessageStatus());
        }

        return messageContext.createExecutionContext(executionContext.getMessageType());
    }
}
