package com.evolvsys.nexuse2e.core.message.dispatcher;

import com.evolvsys.nexuse2e.core.action.Action;
import com.evolvsys.nexuse2e.core.action.ActionFactory;
import com.evolvsys.nexuse2e.core.exception.NexusException;
import com.evolvsys.nexuse2e.core.message.ContextMessageService;
import com.evolvsys.nexuse2e.core.message.context.MessageContext;
import com.evolvsys.nexuse2e.core.utils.ErrorUtils;
import com.evolvsys.nexuse2e.internal.MessageType;
import java.util.UUID;
import lombok.RequiredArgsConstructor;
import lombok.extern.slf4j.Slf4j;
import org.springframework.stereotype.Component;

/**
 * @author SvenBarden
 * @since 21.01.2024
 */
@Component
@RequiredArgsConstructor
@Slf4j
class ReturnDispatcher implements Dispatcher {
    private final ActionFactory actionFactory;
    private final ContextMessageService messageService;

    @Override
    public MessageContext dispatch(MessageContext messageContext) throws NexusException {
        String actionId = messageContext.getActionId();
        String choreographyId = messageContext.getChoreographyId();

        Action action = actionFactory.getAction(choreographyId, actionId);

        LOGGER.debug(
                "running return pipeline for message {} in action {} from choreography {}",
                messageContext.getSourceMessage().getId(),
                action.getActionId(),
                action.getChoreographyId());
        if (messageContext.getReturnMessage() != null) {
            if (action.hasReturnPipeline()) {
                try {
                    messageContext = action.getReturnPipeline()
                            .execute(messageContext.createExecutionContext(MessageType.RETURN))
                            .getMessageContext();
                } catch (NexusException e) {
                    // TODO error handling
                    LOGGER.error("error running return pipeline: {}", ErrorUtils.aggregate(e), e);
                }
            }

            if (messageContext.getReturnMessage() != null) {
                if (messageContext.getReturnMessage().getId() == null) {
                    messageContext.getReturnMessage().setId(UUID.randomUUID().toString());
                    messageContext
                            .getReturnMessage()
                            .setConversationId(messageContext.getSourceMessage().getConversationId());
                }
                messageService.save(messageContext.getReturnMessage(), action);
            }
        } else {
            LOGGER.trace("context has no return message, do not run return pipeline");
        }
        return messageContext;
    }
}
