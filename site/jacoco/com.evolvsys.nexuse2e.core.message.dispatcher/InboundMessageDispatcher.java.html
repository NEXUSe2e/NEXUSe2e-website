package com.evolvsys.nexuse2e.core.message.dispatcher;

import com.evolvsys.nexuse2e.core.exception.NexusError;
import com.evolvsys.nexuse2e.core.exception.NexusException;
import com.evolvsys.nexuse2e.core.message.context.MessageContext;
import lombok.RequiredArgsConstructor;
import lombok.extern.slf4j.Slf4j;
import org.springframework.stereotype.Component;

/**
 * @author SvenBarden
 * @since 21.01.2024
 */
@Component
@RequiredArgsConstructor
@Slf4j
public class InboundMessageDispatcher {
    private final InboundNormalDispatcher inboundNormalDispatcher;
    private final InboundAcknowledgmentDispatcher inboundAcknowledgmentDispatcher;

    public MessageContext dispatch(MessageContext messageContext) throws NexusException {
        LOGGER.debug("dispatching inbound message {}", messageContext);

        return switch (messageContext.getSourceMessage().getType()) {
            case NORMAL -> inboundNormalDispatcher.dispatch(messageContext);
            case ACKNOWLEDGMENT -> inboundAcknowledgmentDispatcher.dispatch(messageContext);
            default ->
                throw new NexusException(NexusError.builder()
                        .message("Invalid inbound message type")
                        .code("INVALID_MESSAGE_TYPE")
                        .extension(
                                "type",
                                messageContext.getSourceMessage().getType().name())
                        .extension(
                                "messageId", messageContext.getSourceMessage().getId())
                        .extension(
                                "conversationId",
                                messageContext.getSourceMessage().getConversationId())
                        .build());
        };
    }
}
