package com.evolvsys.nexuse2e.core.message.dispatcher;

import com.evolvsys.nexuse2e.core.action.Action;
import com.evolvsys.nexuse2e.core.constants.ConnectionConstants;
import com.evolvsys.nexuse2e.core.exception.NexusException;
import com.evolvsys.nexuse2e.core.message.MessageRunner;
import com.evolvsys.nexuse2e.core.message.MetaData;
import com.evolvsys.nexuse2e.core.message.context.ContextFactory;
import com.evolvsys.nexuse2e.core.message.context.MessageContext;
import com.evolvsys.nexuse2e.core.message.state.MessageEventService;
import com.evolvsys.nexuse2e.core.message.state.MessageEventType;
import com.evolvsys.nexuse2e.core.route.service.ContextRouteFactory;
import com.evolvsys.nexuse2e.core.utils.ErrorUtils;
import com.evolvsys.nexuse2e.internal.Message;
import com.evolvsys.nexuse2e.internal.MessageType;
import lombok.RequiredArgsConstructor;
import lombok.extern.slf4j.Slf4j;
import org.springframework.stereotype.Component;

/**
 * @author SvenBarden
 * @since 21.01.2024
 */
@Component
@RequiredArgsConstructor
@Slf4j
class OutboundAcknowledgmentDispatcher implements Dispatcher {
    private final MessageRunner messageRunner;
    private final OutboundMessageDispatcher outboundMessageDispatcher;
    private final MessageEventService messageEventService;
    private final ContextFactory contextFactory;
    private final ContextRouteFactory contextRouteFactory;

    private MessageContext dispatchInternal(MessageContext originalMessageContext) {
        Action action = originalMessageContext.getAction();

        LOGGER.debug(
                "dispatching acknowledgment for message {} in action {} from choreography {}",
                originalMessageContext.getSourceMessage().getId(),
                action.getActionId(),
                action.getChoreographyId());

        MessageContext messageContext = contextFactory.createForResend(
                originalMessageContext.getSourceMessage().getId());
        messageContext.setSourcePartnerId(originalMessageContext.getTargetPartnerId());
        messageContext.setTargetPartnerId(originalMessageContext.getSourcePartnerId());

        messageContext.setRoute(contextRouteFactory.create(
                originalMessageContext.getRoute().getId(), messageContext.getSourcePartnerId()));

        try {
            Message sourceMessage = messageContext.getSourceMessage();
            Message ackMessage = messageContext.getAcknowledgmentMessage();

            if (ackMessage == null) {
                LOGGER.debug("creating new acknowledgment message");
                ackMessage = new Message();

                ackMessage.setReferenceMessageId(sourceMessage.getId());
                ackMessage.setId(messageContext.getMessageIdGenerator().generateId());
                ackMessage.setType(MessageType.ACKNOWLEDGMENT);
                ackMessage.setMetaData(new MetaData());
                ackMessage.setConversationId(sourceMessage.getConversationId());

                messageContext.setAcknowledgmentMessage(ackMessage);

            } else {
                LOGGER.debug("reusing existing acknowledgment message");
            }

            if (action.getAcknowledgmentPipeline() != null) {
                messageContext = action.getAcknowledgmentPipeline()
                        .execute(messageContext.createExecutionContext(MessageType.ACKNOWLEDGMENT))
                        .getMessageContext();
            }

            messageContext = outboundMessageDispatcher.dispatch(
                    messageContext.createExecutionContext(MessageType.ACKNOWLEDGMENT));
        } catch (NexusException e) {
            try {
                LOGGER.error("error running acknowledgment pipeline {}", ErrorUtils.aggregate(e), e);
                messageEventService.trigger(
                        messageContext.createExecutionContext(MessageType.ACKNOWLEDGMENT), MessageEventType.ERROR);
            } catch (NexusException ex) {
                LOGGER.error("could not execute error event {}", ErrorUtils.aggregate(ex), ex);
            }
        }

        return messageContext;
    }

    @Override
    public MessageContext dispatch(MessageContext messageContext) throws NexusException {
        if (messageContext
                .getSourceMessage()
                .getMetaData()
                .getFirstOrDefault(ConnectionConstants.REQUIRES_ACK, Boolean.FALSE)) {
            LOGGER.trace("connection is reliable, sending acknowledgment");
            LOGGER.trace("connection is async, creating new thread");

            messageRunner
                    .createWrappedFuture(messageContext, () -> dispatchInternal(messageContext))
                    .thenAccept(mc -> LOGGER.debug("finished async acknowledgment pipeline"))
                    .exceptionally(throwable -> {
                        LOGGER.error(
                                "could not execute async acknowledgment pipeline {}",
                                ErrorUtils.aggregate(throwable),
                                throwable);
                        return null;
                    });
            LOGGER.debug("called async acknowledgment pipeline");
        }

        return messageContext;
    }
}
