package com.evolvsys.nexuse2e.core.message.dispatcher;

import com.evolvsys.nexuse2e.core.exception.NexusException;
import com.evolvsys.nexuse2e.core.message.MessageStatus;
import com.evolvsys.nexuse2e.core.message.context.MessageContext;
import com.evolvsys.nexuse2e.internal.MessageType;
import java.util.EnumSet;
import lombok.RequiredArgsConstructor;
import lombok.extern.slf4j.Slf4j;
import org.springframework.stereotype.Component;

/**
 * @author SvenBarden
 * @since 21.01.2024
 */
@Component
@RequiredArgsConstructor
@Slf4j
class InboundNormalDispatcher implements Dispatcher {
    private final OutboundMessageDispatcher outboundMessageDispatcher;
    private final OutboundAcknowledgmentDispatcher outboundAcknowledgmentDispatcher;

    @Override
    public MessageContext dispatch(MessageContext messageContext) throws NexusException {
        outboundAcknowledgmentDispatcher.dispatch(messageContext);

        if (EnumSet.of(MessageStatus.RESOLVED, MessageStatus.CREATED)
                        .contains(messageContext.getSourceMessage().getMessageStatus())
                == false) {
            LOGGER.info(
                    "message {} was already received (current state: {}), skip outbound dispatcher",
                    messageContext.getSourceMessage().getId(),
                    messageContext.getSourceMessage().getMessageStatus());
        } else {
            LOGGER.debug(
                    "dispatch normal message {} to outbound dispatcher",
                    messageContext.getSourceMessage().getId());
            messageContext =
                    outboundMessageDispatcher.dispatch(messageContext.createExecutionContext(MessageType.NORMAL));
        }

        return messageContext;
    }
}
