package com.evolvsys.nexuse2e.core.logging.repository;

import com.evolvsys.nexuse2e.core.logging.model.LogEntity;
import com.evolvsys.nexuse2e.core.logging.model.LogFilter;
import com.evolvsys.nexuse2e.core.logging.model.QLogEntity;
import com.querydsl.core.types.dsl.BooleanExpression;
import org.springframework.data.domain.Page;
import org.springframework.data.domain.Pageable;
import org.springframework.data.jpa.repository.JpaRepository;
import org.springframework.data.jpa.repository.Modifying;
import org.springframework.data.jpa.repository.Query;
import org.springframework.data.querydsl.QuerydslPredicateExecutor;
import org.springframework.lang.Nullable;
import org.springframework.transaction.annotation.Transactional;

/**
 * @author SvenBarden
 * @since 04.01.2024
 */
public interface LogRepository extends JpaRepository<LogEntity, Long>, QuerydslPredicateExecutor<LogEntity> {

    default Page<LogEntity> findAllToRead(
            Pageable pageable, @Nullable LogFilter filter, String timeZone, BooleanExpression securityPredicate) {
        return findAll(securityPredicate.and(QLogEntity.logEntity.filter(filter, timeZone)), pageable);
    }

    @Transactional
    @Modifying
    @Query(
            "update LogEntity l set l.messageId = :messageId where l.mdcId = :mdcId and l.inMessageContext = true and (l.messageId is null or l.messageId != :messageId)")
    int updateMessageIdByMdcId(String messageId, String mdcId);

    @Transactional
    @Modifying
    @Query(
            "update LogEntity l set l.messageId = NULL, l.conversationId = NULL where l.mdcId = :mdcId and l.messageId = :messageId")
    int removeMessageIdByMdcId(String mdcId, String messageId);
}
