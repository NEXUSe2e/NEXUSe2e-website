package com.evolvsys.nexuse2e.core.logging.repository;

import com.evolvsys.nexuse2e.core.logging.model.FilterType;
import com.evolvsys.nexuse2e.core.logging.model.LogEntity;
import com.evolvsys.nexuse2e.core.logging.model.LogFilter;
import com.evolvsys.nexuse2e.core.logging.model.QLogEntity;
import com.evolvsys.nexuse2e.core.utils.DateRangeFilterUtils;
import com.querydsl.core.BooleanBuilder;
import com.querydsl.core.annotations.QueryDelegate;
import com.querydsl.core.types.Predicate;
import com.querydsl.core.types.dsl.StringPath;
import java.util.Set;
import org.apache.commons.collections4.CollectionUtils;
import org.apache.commons.lang3.StringUtils;
import org.springframework.lang.Nullable;

/**
 * @author SvenBarden
 * @since 09.01.2024
 */
public final class LogQueryDelegate {
    private LogQueryDelegate() {}

    private static void filterStringContains(BooleanBuilder builder, String filter, StringPath stringPath) {
        if (StringUtils.isNotBlank(filter)) {
            builder.and(stringPath.contains(filter));
        }
    }

    private static void filterStringEquals(BooleanBuilder builder, String filter, StringPath stringPath) {
        if (StringUtils.isNotBlank(filter)) {
            builder.and(stringPath.eq(filter));
        }
    }

    private static void filterStringIn(BooleanBuilder builder, Set<String> filters, StringPath stringPath) {
        if (CollectionUtils.isNotEmpty(filters)) {
            builder.and(stringPath.in(filters));
        }
    }

    @QueryDelegate(LogEntity.class)
    public static Predicate filter(QLogEntity logEntity, @Nullable LogFilter logFilter, String timeZone) {
        BooleanBuilder builder = new BooleanBuilder();

        if (logFilter != null) {
            filterStringContains(builder, logFilter.getLogger(), logEntity.logger);
            filterStringEquals(builder, logFilter.getChoreographyId(), logEntity.choreographyId);
            filterStringEquals(builder, logFilter.getActionId(), logEntity.actionId);
            filterStringEquals(builder, logFilter.getConversationId(), logEntity.conversationId);
            filterStringEquals(builder, logFilter.getMessageId(), logEntity.messageId);
            filterStringEquals(builder, logFilter.getMdcId(), logEntity.mdcId);
            filterStringContains(builder, logFilter.getMessage(), logEntity.message);
            filterStringIn(builder, logFilter.getLevels(), logEntity.level);
            if (logFilter.getType() != null && logFilter.getType() != FilterType.ALL) {
                switch (logFilter.getType()) {
                    case WITH_MESSAGE_CONTEXT -> builder.and(logEntity.conversationId.isNotNull());
                    case WITHOUT_MESSAGE_CONTEXT -> builder.and(logEntity.conversationId.isNull());
                    default -> {
                        // do nothing, hello spotbugs
                    }
                }
            }
            Predicate predicate = DateRangeFilterUtils.filter(
                    logEntity.created,
                    timeZone,
                    logFilter.getCreatedRange(),
                    logFilter.getCreatedFrom(),
                    logFilter.getCreatedTill());
            if (predicate != null) {
                builder.and(predicate);
            }
        }

        return builder.getValue();
    }
}
