package com.evolvsys.nexuse2e.core.action.action;

import com.evolvsys.app.core.process.*;
import com.evolvsys.nexuse2e.core.action.model.ActionEntity;
import com.evolvsys.nexuse2e.core.action.repository.ActionRepository;
import lombok.RequiredArgsConstructor;
import org.springframework.stereotype.Component;

/**
 * <p>Action to remove a route from an action.</p>
 *
 * @author Florian Drees
 * @since 09.04.2024
 */
@Component
@RequiredArgsConstructor
public class ActionRemoveRouteAction implements ProcessAction<ActionEntity> {
    public static final ProcessParameter<Long> ROUTE_ID_PARAMETER = new ProcessParameter<>() {};

    private final ActionRepository actionRepository;

    @Override
    public ActionEntity execute(ActionEntity entity, ProcessActionChain chain, ProcessParameters processParameters) {
        long routeId = processParameters.get(ROUTE_ID_PARAMETER);

        entity.getRoutes().removeIf(routeEntity -> routeEntity.getId() == routeId);

        return actionRepository.save(entity);
    }

    @Override
    public ProcessActionConditions conditions(ActionEntity entity, ProcessParameters processParameters) {
        return ProcessActionConditions.builder()
                .condition(processParameters.contains(ROUTE_ID_PARAMETER), "Route id parameter must not be null")
                .build();
    }
}
