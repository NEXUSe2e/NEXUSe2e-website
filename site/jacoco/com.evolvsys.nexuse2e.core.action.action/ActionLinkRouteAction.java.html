package com.evolvsys.nexuse2e.core.action.action;

import com.evolvsys.app.core.process.*;
import com.evolvsys.nexuse2e.core.action.model.ActionEntity;
import com.evolvsys.nexuse2e.core.action.repository.ActionRepository;
import com.evolvsys.nexuse2e.core.exception.NexusError;
import com.evolvsys.nexuse2e.core.exception.NexusRuntimeException;
import com.evolvsys.nexuse2e.core.model.entity.RouteEntity;
import com.evolvsys.nexuse2e.core.route.repository.RouteRepository;
import java.util.List;
import lombok.RequiredArgsConstructor;
import org.apache.commons.collections4.CollectionUtils;
import org.springframework.stereotype.Component;

/**
 * @author Florian Drees
 * @since 10.04.2024
 */
@Component
@RequiredArgsConstructor
public class ActionLinkRouteAction implements ProcessAction<ActionEntity> {
    public static final ProcessParameter<List<Long>> ROUTE_ID_PARAMETER = new ProcessParameter<>() {};

    private final ActionRepository actionRepository;
    private final RouteRepository routeRepository;

    @Override
    public ActionEntity execute(ActionEntity entity, ProcessActionChain chain, ProcessParameters processParameters) {
        List<Long> routeIds = processParameters.get(ROUTE_ID_PARAMETER);

        for (Long routeId : routeIds) {
            RouteEntity routeEntity = routeRepository
                    .findById(routeId)
                    .orElseThrow(() -> new NexusRuntimeException(NexusError.builder()
                            .message("route could not be found")
                            .extension("id", routeId)
                            .build()));

            entity.getRoutes().add(routeEntity);
        }

        return actionRepository.save(entity);
    }

    @Override
    public ProcessActionConditions conditions(ActionEntity entity, ProcessParameters processParameters) {
        return ProcessActionConditions.builder()
                .condition(
                        CollectionUtils.isNotEmpty(processParameters.get(ROUTE_ID_PARAMETER)),
                        "Route ids parameter must not be empty")
                .build();
    }
}
