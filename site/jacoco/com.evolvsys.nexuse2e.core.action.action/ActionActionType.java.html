package com.evolvsys.nexuse2e.core.action.action;

import com.evolvsys.app.core.process.ProcessActionType;
import com.evolvsys.app.core.process.ProcessDefinition;
import com.evolvsys.nexuse2e.core.action.model.ActionDto;
import com.evolvsys.nexuse2e.core.action.model.ActionEntity;
import lombok.Getter;

/**
 * @author SvenBarden
 * @since 04.03.2024
 */
@Getter
public enum ActionActionType implements ProcessActionType<ActionEntity, ActionDto> {
    ADD_ROUTE(ProcessDefinition.<ActionEntity, ActionDto>builder()
            .action(ActionAddRouteAction.class)
            .build()),
    REMOVE_ROUTE(ProcessDefinition.<ActionEntity, ActionDto>builder()
            .action(ActionRemoveRouteAction.class)
            .build()),
    LINK_ROUTE(ProcessDefinition.<ActionEntity, ActionDto>builder()
            .action(ActionLinkRouteAction.class)
            .build());

    private final ProcessDefinition<ActionEntity, ActionDto> process;

    ActionActionType(ProcessDefinition<ActionEntity, ActionDto> process) {
        this.process = process;
    }
}
