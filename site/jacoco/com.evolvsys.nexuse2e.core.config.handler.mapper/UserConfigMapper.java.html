package com.evolvsys.nexuse2e.core.config.handler.mapper;

import com.evolvsys.app.core.role.model.RoleEntity;
import com.evolvsys.app.core.tenant.model.TenantEntity;
import com.evolvsys.app.core.tenant.model.TenantRoleEntity;
import com.evolvsys.app.core.tenant.repository.TenantRepository;
import com.evolvsys.app.core.tenant.repository.TenantRoleRepository;
import com.evolvsys.app.core.user.model.UserEntity;
import com.evolvsys.nexuse2e.core.config.model.UserConfig;
import java.util.List;
import java.util.stream.Collectors;
import org.mapstruct.*;
import org.springframework.beans.factory.annotation.Autowired;

/**
 * @author Florian Drees
 * @since 11.03.2024
 */
@Mapper(config = DefaultConfigMapperConfig.class)
public abstract class UserConfigMapper {
    private static final String SYSTEM_TENANT_ID = "1";

    private TenantRepository tenantRepository;
    private TenantRoleRepository tenantRoleRepository;

    @Autowired
    protected void config(TenantRepository tenantRepository, TenantRoleRepository tenantRoleRepository) {
        this.tenantRepository = tenantRepository;
        this.tenantRoleRepository = tenantRoleRepository;
    }

    @Mapping(target = "roles", source = "userEntity")
    public abstract UserConfig toConfig(UserEntity userEntity);

    protected List<String> userRoles(UserEntity userEntity) {
        return tenantRoleRepository.findByUser_id(userEntity.getId()).stream()
                .map(TenantRoleEntity::getRole)
                .map(RoleEntity::getInternalName)
                .distinct()
                .collect(Collectors.toList());
    }

    @Mapping(target = "superior", ignore = true)
    @Mapping(target = "salutation", ignore = true)
    @Mapping(target = "profileImage", ignore = true)
    @Mapping(target = "phoneNumber", ignore = true)
    @Mapping(target = "personellNumber", ignore = true)
    @Mapping(target = "department", ignore = true)
    @Mapping(target = "deleted", ignore = true)
    @Mapping(target = "tenant", constant = SYSTEM_TENANT_ID, qualifiedByName = "idToTenantEntity")
    @InheritConfiguration(name = "updateCreatedEntity")
    public abstract void updateEntity(UserConfig userConfig, @MappingTarget UserEntity userEntity);

    @Named("idToTenantEntity")
    public TenantEntity idToTenantEntity(Long tenantId) {
        return tenantRepository.findById(tenantId).orElse(null);
    }

    public abstract void merge(UserConfig newUserConfig, @MappingTarget UserConfig oldUserConfig);
}
