package com.evolvsys.nexuse2e.core.config.handler.mapper;

import com.evolvsys.nexuse2e.core.config.model.ServiceConfig;
import com.evolvsys.nexuse2e.core.model.entity.ServiceEntity;
import com.evolvsys.nexuse2e.core.service.ServiceUtils;
import com.evolvsys.nexuse2e.core.service.repository.ServiceRepository;
import java.util.Optional;
import org.mapstruct.*;
import org.springframework.beans.factory.annotation.Autowired;

/**
 * @author SvenBarden
 * @since 27.11.2023
 */
@Mapper(
        config = DefaultConfigMapperConfig.class,
        uses = {PipelineConfigMapper.class, ParameterConfigMapper.class})
public abstract class ServiceConfigMapper {
    private ParameterConfigMapper parameterConfigMapper;
    private ServiceRepository serviceRepository;
    private ServiceUtils serviceUtils;

    @Autowired
    protected void config(
            ParameterConfigMapper parameterConfigMapper,
            ServiceRepository serviceRepository,
            ServiceUtils serviceUtils) {
        this.parameterConfigMapper = parameterConfigMapper;
        this.serviceRepository = serviceRepository;
        this.serviceUtils = serviceUtils;
    }

    @Mapping(target = "parameters", ignore = true)
    @Mapping(target = "serviceType", ignore = true)
    @Mapping(target = "description", source = "description", qualifiedByName = "toTrimmedString_service")
    @InheritConfiguration
    public abstract void updateEntity(ServiceConfig serviceConfig, @MappingTarget ServiceEntity serviceEntity);

    @Named("toTrimmedString_service")
    protected String toTrimmedString(String stringToTrim) {
        return Optional.ofNullable(stringToTrim).map(String::trim).orElse(stringToTrim);
    }

    @AfterMapping
    public void afterUpdateEntity(ServiceConfig serviceConfig, @MappingTarget ServiceEntity serviceEntity) {
        serviceEntity.setServiceType(serviceUtils.getServiceType(serviceEntity.getComponentId()));
        parameterConfigMapper.updateEntities(
                serviceConfig.getParameters(), serviceEntity.getParameters(), serviceConfig.getComponentId());
    }

    public abstract ServiceConfig toConfig(ServiceEntity serviceEntity);

    public abstract void merge(ServiceConfig newConfig, @MappingTarget ServiceConfig oldConfig);

    public ServiceEntity idToEntity(String serviceId) {
        return serviceRepository.findByServiceId(serviceId).orElse(null);
    }

    public String entityToId(ServiceEntity entity) {
        return Optional.ofNullable(entity).map(ServiceEntity::getServiceId).orElse(null);
    }
}
