package com.evolvsys.nexuse2e.core.config.handler.mapper;

import com.evolvsys.nexuse2e.core.component.ComponentRegistry;
import com.evolvsys.nexuse2e.core.component.Configurable;
import com.evolvsys.nexuse2e.core.component.RegisteredNexusComponent;
import com.evolvsys.nexuse2e.core.config.handler.HandlerUtils;
import com.evolvsys.nexuse2e.core.config.model.ParameterConfig;
import com.evolvsys.nexuse2e.core.exception.NexusError;
import com.evolvsys.nexuse2e.core.exception.NexusRuntimeException;
import com.evolvsys.nexuse2e.core.model.entity.ParameterEntity;
import com.evolvsys.nexuse2e.core.parameter.ParameterDefinition;
import java.util.Collection;
import java.util.Comparator;
import java.util.List;
import java.util.Optional;
import org.apache.commons.collections4.comparators.TransformingComparator;
import org.apache.commons.lang3.StringUtils;
import org.mapstruct.*;
import org.springframework.beans.factory.annotation.Autowired;

/**
 * @author SvenBarden
 * @since 28.11.2023
 */
@Mapper(config = DefaultConfigMapperConfig.class)
public abstract class ParameterConfigMapper {
    private ComponentRegistry componentRegistry;
    private HandlerUtils handlerUtils;

    private static final Comparator<ParameterConfig> COMPARATOR =
            new TransformingComparator<>(ParameterConfig::getName);

    @Autowired
    protected void config(ComponentRegistry componentRegistry, HandlerUtils handlerUtils) {
        this.componentRegistry = componentRegistry;
        this.handlerUtils = handlerUtils;
    }

    public void updateEntities(
            Collection<ParameterConfig> parameterConfigs,
            @MappingTarget Collection<ParameterEntity> parameterEntities) {
        handlerUtils.mapParametersToEntities(parameterConfigs, parameterEntities);
    }

    public void updateEntities(
            Collection<ParameterConfig> parameterConfigs,
            @MappingTarget Collection<ParameterEntity> parameterEntities,
            @Context String componentName) {

        if (StringUtils.isNotEmpty(componentName)) {
            Optional<RegisteredNexusComponent> optionalComponent = componentRegistry.getComponent(componentName);
            if (optionalComponent.isPresent()) {
                RegisteredNexusComponent component = optionalComponent.orElseThrow();
                if (Configurable.class.isAssignableFrom(component.getLoadedClass())) {
                    List<ParameterDefinition> parameterDefinitions = component.getParameterDefinitions();
                    handlerUtils.mapParametersToEntities(parameterConfigs, parameterEntities, parameterDefinitions);
                }
            }
        } else {
            throw new NexusRuntimeException(NexusError.builder()
                    .message("Missing component name for parameter config mapping")
                    .build());
        }
    }

    @Mapping(target = "values", ignore = true)
    public abstract ParameterConfig toConfig(ParameterEntity parameterEntity);

    @AfterMapping
    protected void afterToConfig(ParameterEntity parameterEntity, @MappingTarget ParameterConfig parameterConfig) {
        handlerUtils.mapEntityValueToConfig(parameterEntity, parameterConfig);
    }

    @AfterMapping
    protected void afterToConfigs(
            Collection<ParameterEntity> parameterEntities, @MappingTarget List<ParameterConfig> parameterConfigs) {
        parameterConfigs.sort(COMPARATOR);
    }
}
