package com.evolvsys.nexuse2e.core.config.handler.mapper;

import com.evolvsys.nexuse2e.core.config.model.PipelineConfig;
import com.evolvsys.nexuse2e.core.pipeline.model.PipelineEntity;
import com.evolvsys.nexuse2e.core.pipeline.repository.PipelineRepository;
import java.util.Optional;
import org.mapstruct.*;
import org.springframework.beans.factory.annotation.Autowired;

/**
 * @author SvenBarden
 * @since 27.11.2023
 */
@Mapper(config = DefaultConfigMapperConfig.class, uses = PipeletConfigMapper.class)
public abstract class PipelineConfigMapper {
    private PipelineRepository pipelineRepository;

    @Autowired
    protected void config(PipelineRepository pipelineRepository) {
        this.pipelineRepository = pipelineRepository;
    }

    public PipelineEntity idToEntity(String pipelineId) {
        return pipelineRepository.findByPipelineId(pipelineId).orElse(null);
    }

    public String entityToString(PipelineEntity pipelineEntity) {
        return pipelineEntity == null ? null : pipelineEntity.getPipelineId();
    }

    @Mapping(target = "pipelets", ignore = true)
    @Mapping(target = "description", source = "description", qualifiedByName = "toTrimmedString_pipeline")
    @InheritConfiguration
    public abstract void updateEntity(PipelineConfig pipelineConfig, @MappingTarget PipelineEntity pipelineEntity);

    @Named("toTrimmedString_pipeline")
    protected String toTrimmedString(String stringToTrim) {
        return Optional.ofNullable(stringToTrim).map(String::trim).orElse(stringToTrim);
    }

    public String configToString(PipelineConfig pipelineConfig) {
        return pipelineConfig == null ? null : pipelineConfig.getPipelineId();
    }

    public PipelineConfig stringToConfig(String pipelineId) {
        return pipelineRepository
                .findByPipelineId(pipelineId)
                .map(this::toConfig)
                .orElse(null);
    }

    public abstract PipelineConfig toConfig(PipelineEntity pipelineEntity);

    public abstract void merge(PipelineConfig newConfig, @MappingTarget PipelineConfig oldConfig);
}
