package com.evolvsys.nexuse2e.core.config.handler.mapper;

import com.evolvsys.nexuse2e.core.config.model.ConnectionConfig;
import com.evolvsys.nexuse2e.core.model.entity.PartnerConnectionEntity;
import org.mapstruct.*;
import org.springframework.beans.factory.annotation.Autowired;

/**
 * @author SvenBarden
 * @since 27.11.2023
 */
@Mapper(
        config = DefaultConfigMapperConfig.class,
        uses = {ParameterConfigMapper.class, PartnerConfigMapper.class, ServiceConfigMapper.class})
public abstract class PartnerConnectionConfigMapper {
    private ParameterConfigMapper parameterConfigMapper;

    @Autowired
    protected void config(ParameterConfigMapper parameterConfigMapper) {
        this.parameterConfigMapper = parameterConfigMapper;
    }

    @Mapping(target = "certificateContainer", ignore = true)
    @Mapping(target = "partner", ignore = true)
    @Mapping(target = "parameters", ignore = true)
    @InheritConfiguration
    public abstract void updateEntity(
            ConnectionConfig connectionConfig, @MappingTarget PartnerConnectionEntity partnerConnectionEntity);

    @Mapping(target = "certificateContainer", source = "certificateContainer.name")
    public abstract ConnectionConfig toConfig(PartnerConnectionEntity partnerConnectionEntity);

    @AfterMapping
    protected void afterUpdateEntity(
            ConnectionConfig connectionConfig, @MappingTarget PartnerConnectionEntity partnerConnectionEntity) {
        if (partnerConnectionEntity.getOutflowService() != null) {
            parameterConfigMapper.updateEntities(
                    connectionConfig.getParameters(),
                    partnerConnectionEntity.getParameters(),
                    partnerConnectionEntity.getOutflowService().getComponentId());
        }
    }

    public String toString(PartnerConnectionEntity partnerConnectionEntity) {
        return partnerConnectionEntity == null ? null : partnerConnectionEntity.getConnectionId();
    }
}
