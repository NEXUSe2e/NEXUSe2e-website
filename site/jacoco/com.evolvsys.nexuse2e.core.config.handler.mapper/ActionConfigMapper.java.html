package com.evolvsys.nexuse2e.core.config.handler.mapper;

import com.evolvsys.nexuse2e.core.action.model.ActionEntity;
import com.evolvsys.nexuse2e.core.config.model.ActionConfig;
import org.mapstruct.InheritConfiguration;
import org.mapstruct.Mapper;
import org.mapstruct.Mapping;
import org.mapstruct.MappingTarget;

/**
 * @author SvenBarden
 * @since 29.11.2023
 */
@Mapper(
        config = DefaultConfigMapperConfig.class,
        uses = {PipelineConfigMapper.class, RouteConfigMapper.class})
public abstract class ActionConfigMapper {

    protected String entityToString(ActionEntity actionEntity) {
        return actionEntity == null ? null : actionEntity.getActionId();
    }

    @Mapping(target = "choreography", ignore = true)
    @Mapping(target = "validNextActions", ignore = true)
    @Mapping(target = "routes", ignore = true)
    @InheritConfiguration
    public abstract void updateEntity(ActionConfig actionConfig, @MappingTarget ActionEntity actionEntity);

    public abstract ActionConfig toConfig(ActionEntity actionEntity);
}
