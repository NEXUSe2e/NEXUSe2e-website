package com.evolvsys.nexuse2e.core.config.handler.mapper;

import com.evolvsys.nexuse2e.core.config.model.KeyValuePropertyConfig;
import com.evolvsys.nexuse2e.core.keyvalue.persistence.model.KeyValuePropertyEntity;
import org.mapstruct.InheritConfiguration;
import org.mapstruct.Mapper;
import org.mapstruct.MappingTarget;

/**
 * @author SvenBarden
 * @since 17.06.2024
 */
@Mapper(config = DefaultConfigMapperConfig.class)
public abstract class KeyValuePropertyConfigMapper {

    public abstract KeyValuePropertyConfig toConfig(KeyValuePropertyEntity entity);

    @InheritConfiguration
    public abstract void updateEntity(KeyValuePropertyConfig config, @MappingTarget KeyValuePropertyEntity entity);

    public abstract void merge(KeyValuePropertyConfig newConfig, @MappingTarget KeyValuePropertyConfig oldConfig);
}
