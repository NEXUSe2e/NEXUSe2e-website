package com.evolvsys.nexuse2e.core.config.handler.mapper;

import com.evolvsys.nexuse2e.core.certificate.mapper.CertificateContainerMapper;
import com.evolvsys.nexuse2e.core.config.model.RouteConfig;
import com.evolvsys.nexuse2e.core.model.entity.RouteEntity;
import org.mapstruct.InheritConfiguration;
import org.mapstruct.Mapper;
import org.mapstruct.Mapping;
import org.mapstruct.MappingTarget;

/**
 * @author SvenBarden
 * @since 29.11.2023
 */
@Mapper(
        config = DefaultConfigMapperConfig.class,
        uses = {
            PartnerConnectionConfigMapper.class,
            PartnerConfigMapper.class,
            CertificateContainerMapper.class,
            PipelineConfigMapper.class
        })
public abstract class RouteConfigMapper {
    @Mapping(target = "firstPartner", ignore = true)
    @Mapping(target = "firstPartnerConnection", ignore = true)
    @Mapping(target = "firstNexusCertificateContainer", ignore = true)
    @Mapping(target = "secondPartner", ignore = true)
    @Mapping(target = "secondPartnerConnection", ignore = true)
    @Mapping(target = "secondNexusCertificateContainer", ignore = true)
    @Mapping(target = "allowSamePartnerType", ignore = true)
    @InheritConfiguration
    public abstract void updateEntity(RouteConfig routeConfig, @MappingTarget RouteEntity routeEntity);

    public abstract RouteConfig toConfig(RouteEntity routeEntity);

    public abstract void merge(RouteConfig newConfig, @MappingTarget RouteConfig oldConfig);
}
