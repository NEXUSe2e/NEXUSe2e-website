package com.evolvsys.nexuse2e.core.config.handler.mapper;

import com.evolvsys.nexuse2e.core.certificate.model.CertificateContainerEntity;
import com.evolvsys.nexuse2e.core.certificate.repository.CertificateContainerRepository;
import com.evolvsys.nexuse2e.core.config.model.CertificateContainerConfig;
import com.evolvsys.nexuse2e.core.config.model.ClientKeyContainerConfig;
import com.evolvsys.nexuse2e.core.model.entity.PartnerEntity;
import java.util.List;
import java.util.stream.Collectors;
import org.mapstruct.Mapper;
import org.mapstruct.Mapping;
import org.mapstruct.MappingTarget;
import org.springframework.beans.factory.annotation.Autowired;

/**
 * @author SvenBarden
 * @since 13.02.2024
 */
@Mapper(config = DefaultConfigMapperConfig.class, uses = CertificateConfigMapper.class)
public abstract class CertificateContainerConfigMapper {
    private CertificateContainerRepository certificateContainerRepository;

    @Autowired
    protected void config(CertificateContainerRepository certificateContainerRepository) {
        this.certificateContainerRepository = certificateContainerRepository;
    }

    public abstract void merge(ClientKeyContainerConfig newConfig, @MappingTarget ClientKeyContainerConfig oldConfig);

    @Mapping(target = "certificates", source = "certificateContainerEntity")
    public abstract CertificateContainerConfig toConfig(CertificateContainerEntity certificateContainerEntity);

    @Mapping(target = "clientKeys", source = "certificateContainerEntity")
    public abstract ClientKeyContainerConfig toClientKeyConfig(CertificateContainerEntity certificateContainerEntity);

    public List<CertificateContainerConfig> mapContainersFromPartner(PartnerEntity partnerEntity) {
        return certificateContainerRepository.findByPartner(partnerEntity).stream()
                .map(this::toConfig)
                .collect(Collectors.toList());
    }
}
