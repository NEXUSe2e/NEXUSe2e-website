package com.evolvsys.nexuse2e.core.config.handler.mapper;

import com.evolvsys.app.core.file.model.FileContentEntity;
import com.evolvsys.app.core.file.model.FileEntity;
import com.evolvsys.app.core.file.model.FileStatus;
import com.evolvsys.nexuse2e.core.certificate.mapper.CertificateInfoMapper;
import com.evolvsys.nexuse2e.core.certificate.model.CertificateContainerEntity;
import com.evolvsys.nexuse2e.core.certificate.model.CertificateEntity;
import com.evolvsys.nexuse2e.core.certificate.repository.CertificateRepository;
import com.evolvsys.nexuse2e.core.certificate.service.CertificateEntityFactory;
import com.evolvsys.nexuse2e.core.certificate.service.CertificateHelper;
import com.evolvsys.nexuse2e.core.certificate.service.CertificateKeystoreUtils;
import com.evolvsys.nexuse2e.core.config.model.BaseCertConfig;
import com.evolvsys.nexuse2e.core.config.model.CaCertConfig;

import com.evolvsys.nexuse2e.core.exception.NexusRuntimeException;
import edu.umd.cs.findbugs.annotations.SuppressFBWarnings;
import java.io.ByteArrayInputStream;
import java.io.IOException;
import java.io.InputStream;
import java.security.KeyStore;
import java.security.cert.X509Certificate;
import java.sql.SQLException;

import org.mapstruct.*;
import org.springframework.beans.factory.annotation.Autowired;

/**
 * @author SvenBarden
 * @since 13.02.2024
 */
@Mapper(config = DefaultConfigMapperConfig.class)
public abstract class CertificateConfigMapper {
    private static final String FILE = "file";
    private CertificateRepository certificateRepository;
    private CertificateInfoMapper certificateInfoMapper;

    @Autowired
    protected void config(CertificateRepository certificateRepository, CertificateInfoMapper certificateInfoMapper) {
        this.certificateRepository = certificateRepository;
        this.certificateInfoMapper = certificateInfoMapper;
    }

    @Mapping(target = "password", ignore = true)
    @Mapping(target = "type", ignore = true)
    @Mapping(target = "info", ignore = true)
    @Mapping(target = FILE, ignore = true)
    @Mapping(target = "disabled", ignore = true)
    @Mapping(target = "enabled", ignore = true)
    @Mapping(target = "certificateContainer", ignore = true)
    @InheritConfiguration(name = "updateCreatedEntity")
    protected abstract void updateBaseCertEntity(
            BaseCertConfig certificateConfig, @MappingTarget CertificateEntity certificateEntity);

    @InheritConfiguration(name = "updateBaseCertEntity")
    public abstract void updateEntity(
            CaCertConfig certificateConfig, @MappingTarget CertificateEntity certificateEntity);

    @InheritConfiguration(name = "updateBaseCertEntity")
    public abstract void updateEntity(
            CertificateConfig certificateConfig, @MappingTarget CertificateEntity certificateEntity);

    @Mapping(target = "type", ignore = true)
    @Mapping(target = "info", ignore = true)
    @Mapping(target = FILE, ignore = true)
    @Mapping(target = "disabled", ignore = true)
    @Mapping(target = "enabled", ignore = true)
    @Mapping(target = "certificateContainer", ignore = true)
    @InheritConfiguration(name = "updateCreatedEntity")
    public abstract void updateEntity(
            ClientKeyConfig certificateConfig, @MappingTarget CertificateEntity certificateEntity);

    @Mapping(target = "content", source = FILE)
    @InheritConfiguration
    public abstract CertificateConfig entityToConfig(CertificateEntity certificateEntity);

    @Mapping(target = "content", source = FILE)
    @InheritConfiguration
    public abstract ClientKeyConfig entityToClientKey(CertificateEntity certificateEntity);

    @Mapping(target = "content", source = FILE)
    @InheritConfiguration
    public abstract CaCertConfig entityToCaCert(CertificateEntity certificateEntity);

    public abstract void merge(CertificateConfig certificateConfig, @MappingTarget CertificateConfig oldConfig);

    public abstract void merge(CaCertConfig certificateConfig, @MappingTarget CaCertConfig oldConfig);

    public abstract void merge(ClientKeyConfig certificateConfig, @MappingTarget ClientKeyConfig oldConfig);

    public List<CertificateConfig> mapCertificatesFromContainer(CertificateContainerEntity certificateContainerEntity) {
        return certificateRepository.findByCertificateContainer(certificateContainerEntity).stream()
                .map(this::entityToConfig)
                .collect(Collectors.toList());
    }

    public List<CaCertConfig> mapCaCertificatesFromContainer(CertificateContainerEntity certificateContainerEntity) {
        return certificateRepository.findByCertificateContainer(certificateContainerEntity).stream()
                .map(this::entityToCaCert)
                .collect(Collectors.toList());
    }

    public List<ClientKeyConfig> mapClientKeysFromContainer(CertificateContainerEntity certificateContainerEntity) {
        return certificateRepository.findByCertificateContainer(certificateContainerEntity).stream()
                .map(this::entityToClientKey)
                .collect(Collectors.toList());
    }

    protected String fileToBase64String(FileEntity fileEntity) throws SQLException, IOException {
        return Base64.getEncoder()
                .encodeToString(fileEntity
                        .getFileContent()
                        .getContent()
                        .getBinaryStream()
                        .readAllBytes());
    }

    private InputStream baseUpdateContent(
            BaseCertConfig certificateConfig, @MappingTarget CertificateEntity certificateEntity) {
        byte[] decoded =
                Base64.getDecoder().decode(certificateConfig.getContent().trim());
        ByteArrayInputStream byteArrayInputStream = new ByteArrayInputStream(decoded);

        FileEntity file = certificateEntity.getFile();
        if (file == null) {
            certificateEntity.setFile(file = new FileEntity());
        }

        file.setMimeType(CertificateEntityFactory.typeToMimeType(certificateEntity.getType()));
        file.setSize(decoded.length);
        file.setStatus(FileStatus.SAVED);
        file.setFileName(certificateEntity.getName());

        FileContentEntity fileContent = file.getFileContent();
        if (fileContent == null) {
            file.setFileContent(fileContent = new FileContentEntity());
        }

        fileContent.setContent(BlobProxy.generateProxy(decoded));

        return byteArrayInputStream;
    }

    @AfterMapping
    protected void updateContent(
            CertificateConfig certificateConfig, @MappingTarget CertificateEntity certificateEntity) {
        updateX509CertificateContent(certificateConfig, certificateEntity);
    }

    @AfterMapping
    protected void updateContent(CaCertConfig certificateConfig, @MappingTarget CertificateEntity certificateEntity) {
        updateX509CertificateContent(certificateConfig, certificateEntity);
    }

    @SuppressFBWarnings("EXS_EXCEPTION_SOFTENING_NO_CONSTRAINTS")
    private void updateX509CertificateContent(BaseCertConfig config, @MappingTarget CertificateEntity entity) {
        try {
            X509Certificate certificate = CertificateHelper.getCertificate(baseUpdateContent(config, entity));
            entity.setInfo(certificateInfoMapper.certificateToInfo(certificate));
        } catch (Exception e) {
            throw new NexusRuntimeException(
                    NexusError.builder()
                            .message("content from config is not a valid certificate file " + config.getName())
                            .extension("name", config.getName())
                            .build(),
                    e);
        }
    }

    @AfterMapping
    @SuppressFBWarnings("EXS_EXCEPTION_SOFTENING_NO_CONSTRAINTS")
    protected void updateContent(
            ClientKeyConfig certificateConfig, @MappingTarget CertificateEntity certificateEntity) {
        try {
            // verify valid cert or keystore
            KeyStore keyStore = CertificateKeystoreUtils.toKeystore(
                    baseUpdateContent(certificateConfig, certificateEntity), certificateConfig.getPassword());
            certificateEntity.setInfo(certificateInfoMapper.privateKeyToInfo(keyStore));
        } catch (Exception e) {
            throw new NexusRuntimeException(
                    NexusError.builder()
                            .message("content from config is not a valid certificate file")
                            .extension("name", certificateConfig.getName())
                            .build(),
                    e);
        }
    }
}
