package com.evolvsys.nexuse2e.core.logging.model;

import com.evolvsys.nexuse2e.core.model.entity.MessageEntity;
import lombok.Builder;
import lombok.Value;

/**
 * @author SvenBarden
 * @since 02.08.2024
 */
@Value
@Builder
public class ForMessageLoggingContext {
    public static final String KEY_NAME = "ForMessageLoggingContext";
    String messageId;
    String conversationId;
    String choreographyId;
    String actionId;

    public static ForMessageLoggingContext forEntity(MessageEntity messageEntity) {
        return ForMessageLoggingContext.builder()
                .messageId(messageEntity.getMessageId())
                .conversationId(messageEntity.getConversation().getConversationId())
                .choreographyId(
                        messageEntity.getConversation().getChoreography().getChoreographyId())
                .actionId(messageEntity.getConversation().getCurrentAction().getActionId())
                .build();
    }
}
