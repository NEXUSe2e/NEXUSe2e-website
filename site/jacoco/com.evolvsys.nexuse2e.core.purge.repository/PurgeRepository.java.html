package com.evolvsys.nexuse2e.core.purge.repository;

import com.evolvsys.nexuse2e.core.logging.model.LogFilter;
import com.evolvsys.nexuse2e.core.logging.model.QLogEntity;
import com.evolvsys.nexuse2e.core.manualaction.model.MessageActionManualActionEntity;
import com.evolvsys.nexuse2e.core.manualaction.model.QMessageActionManualActionEntity;
import com.evolvsys.nexuse2e.core.message.ConversationFilter;
import com.evolvsys.nexuse2e.core.message.payload.model.PayloadEntity;
import com.evolvsys.nexuse2e.core.message.payload.model.QBinaryEntity;
import com.evolvsys.nexuse2e.core.message.payload.model.QPayloadEntity;
import com.evolvsys.nexuse2e.core.model.entity.*;
import com.evolvsys.nexuse2e.core.purge.model.PurgeResult;
import com.evolvsys.nexuse2e.core.settings.service.ApplicationSettingsService;
import com.querydsl.core.Tuple;
import com.querydsl.jpa.JPAExpressions;
import com.querydsl.jpa.JPQLQueryFactory;
import com.querydsl.jpa.JPQLSubQuery;
import jakarta.persistence.EntityManager;
import java.util.List;
import lombok.RequiredArgsConstructor;
import lombok.extern.slf4j.Slf4j;
import org.springframework.stereotype.Repository;
import org.springframework.transaction.annotation.Transactional;

/**
 * @author SvenBarden
 * @since 25.03.2024
 */
@Repository
@RequiredArgsConstructor
@Slf4j
public class PurgeRepository {
    private final JPQLQueryFactory jpqlQueryFactory;
    private final ApplicationSettingsService applicationSettingsService;
    private final EntityManager entityManager;

    @Transactional
    public PurgeResult purgeLogs(LogFilter logFilter) {
        long deletedLogs = jpqlQueryFactory
                .delete(QLogEntity.logEntity)
                .where(QLogEntity.logEntity.filter(
                        logFilter,
                        applicationSettingsService.getSystemSettingsOrDefault().getTimeZone()))
                .execute();
        LOGGER.debug("deleted {} filtered logs", deletedLogs);

        return PurgeResult.builder().deletedLogs(deletedLogs).build();
    }

    @Transactional
    public PurgeResult purgeAllLogs() {
        long deletedLogs = jpqlQueryFactory.delete(QLogEntity.logEntity).execute();
        LOGGER.debug("deleted all {} logs", deletedLogs);

        return PurgeResult.builder().deletedLogs(deletedLogs).build();
    }

    @Transactional
    public PurgeResult purgeAllConversations() {
        long deletedManualActions = jpqlQueryFactory
                .delete(QMessageActionManualActionEntity.messageActionManualActionEntity)
                .execute();
        LOGGER.debug("deleted all {} manual message actions", deletedManualActions);

        long deletedMessages =
                jpqlQueryFactory.delete(QMessageEntity.messageEntity).execute();
        LOGGER.debug("deleted all {} messages", deletedMessages);

        // remove payload reference from metaData, so we can safely delete all payloads and do not need two metaData
        // delete statements
        long clearedMetaData = jpqlQueryFactory
                .update(QMetaDataEntity.metaDataEntity)
                .setNull(QMetaDataEntity.metaDataEntity.binaryValue)
                .where(QMetaDataEntity.metaDataEntity.binaryValue.isNotNull())
                .execute();
        LOGGER.debug("removed payload reference from {} metaData", clearedMetaData);

        long deletedPayloads =
                jpqlQueryFactory.delete(QPayloadEntity.payloadEntity).execute();
        LOGGER.debug("deleted all {} payloads", deletedPayloads);

        long deletedBinaries =
                jpqlQueryFactory.delete(QBinaryEntity.binaryEntity).execute();
        LOGGER.debug("deleted all {} binary contents", deletedBinaries);

        long deletedMetaDatas =
                jpqlQueryFactory.delete(QMetaDataEntity.metaDataEntity).execute();
        LOGGER.debug("deleted all {} metaDatas", deletedMetaDatas);

        long deletedConversations =
                jpqlQueryFactory.delete(QConversationEntity.conversationEntity).execute();
        LOGGER.debug("deleted all {} conversations", deletedConversations);

        long deletedLogs = jpqlQueryFactory.delete(QLogEntity.logEntity).execute();
        LOGGER.debug("deleted all conversation {} logs", deletedLogs);

        return PurgeResult.builder()
                .deletedMessages(deletedMessages)
                .deletedConversations(deletedConversations)
                .deletedMetaDatas(deletedMetaDatas)
                .deletedPayloads(deletedPayloads)
                .deletedBinaries(deletedBinaries)
                .deletedLogs(deletedLogs)
                .deletedManualActions(deletedManualActions)
                .build();
    }

    @Transactional
    public PurgeResult purgeConversations(ConversationFilter conversationFilter) {
        JPQLSubQuery<ConversationEntity> conversationSelect = JPAExpressions.selectFrom(
                        QConversationEntity.conversationEntity)
                .where(QConversationEntity.conversationEntity.filter(
                        conversationFilter,
                        applicationSettingsService.getSystemSettingsOrDefault().getTimeZone()));

        JPQLSubQuery<PayloadEntity> messagePayloads = JPAExpressions.select(QPayloadEntity.payloadEntity)
                .from(QMessageEntity.messageEntity)
                .join(QMessageEntity.messageEntity.payloads, QPayloadEntity.payloadEntity);

        JPQLSubQuery<PayloadEntity> metaDataPayloads = JPAExpressions.select(QMetaDataEntity.metaDataEntity.binaryValue)
                .from(QMetaDataEntity.metaDataEntity)
                .where(QMetaDataEntity.metaDataEntity.binaryValue.isNotNull());

        JPQLSubQuery<MetaDataEntity> messageMetaDatas = JPAExpressions.select(QMetaDataEntity.metaDataEntity)
                .from(QMessageEntity.messageEntity)
                .join(QMessageEntity.messageEntity.metaDatas, QMetaDataEntity.metaDataEntity);
        JPQLSubQuery<MetaDataEntity> payloadMetaDatas = JPAExpressions.select(QMetaDataEntity.metaDataEntity)
                .from(QPayloadEntity.payloadEntity)
                .join(QPayloadEntity.payloadEntity.metaDatas, QMetaDataEntity.metaDataEntity);

        // delete manual actions

        QMessageActionManualActionEntity mainAction = new QMessageActionManualActionEntity("main");

        QMessageActionManualActionEntity allActions = new QMessageActionManualActionEntity("all");
        QMessageEntity allActionsMessages = new QMessageEntity("actions_all");
        JPQLSubQuery<Long> allSubquery = JPAExpressions.select(allActionsMessages.count())
                .from(allActions)
                .join(allActions.messages, allActionsMessages)
                .where(allActions.eq(mainAction));

        QMessageActionManualActionEntity filteredActions = new QMessageActionManualActionEntity("filtered");
        QMessageEntity filteredActionsMessages = new QMessageEntity("actions_filtered");
        JPQLSubQuery<Long> filteredSubquery = JPAExpressions.select(filteredActionsMessages.count())
                .from(filteredActions)
                .join(filteredActions.messages, filteredActionsMessages)
                .on(filteredActionsMessages.conversation.in(conversationSelect))
                .where(filteredActions.eq(mainAction));

        long deletedManualActions = jpqlQueryFactory
                .delete(mainAction)
                .where(allSubquery.eq(filteredSubquery))
                .execute();
        LOGGER.debug("deleted {} manual actions", deletedManualActions);

        // sadly we cannot directly delete entries from the join table with jpa queries, so we need to remove the
        // references manually

        List<Tuple> manualActions = jpqlQueryFactory
                .select(
                        QMessageActionManualActionEntity.messageActionManualActionEntity,
                        QMessageEntity.messageEntity.id)
                .from(QMessageActionManualActionEntity.messageActionManualActionEntity)
                .join(
                        QMessageActionManualActionEntity.messageActionManualActionEntity.messages,
                        QMessageEntity.messageEntity)
                .on(QMessageEntity.messageEntity.conversation.in(conversationSelect))
                .fetch();
        for (Tuple tuple : manualActions) {
            Long messageEntityId = tuple.get(QMessageEntity.messageEntity.id);
            MessageActionManualActionEntity messageActionManualActionEntity =
                    tuple.get(QMessageActionManualActionEntity.messageActionManualActionEntity);
            if (messageActionManualActionEntity != null) {
                messageActionManualActionEntity.getMessages().removeIf(message -> message.getId()
                        .equals(messageEntityId));
                LOGGER.info(
                        "manually removed message {} reference from manual action {}",
                        messageEntityId,
                        messageActionManualActionEntity.getId());
                entityManager.persist(messageActionManualActionEntity);
                // we need do flush the changes to the database, so the following delete statements can be executed
                entityManager.flush();
            }
        }
        long deletedManualActionsPartly = manualActions.size();
        LOGGER.debug("deleted {} manual actions references manually", manualActions.size());

        // delete messages
        long deletedMessages = jpqlQueryFactory
                .delete(QMessageEntity.messageEntity)
                .where(QMessageEntity.messageEntity.conversation.in(conversationSelect))
                .execute();
        LOGGER.debug("deleted {} messages", deletedMessages);

        // delete metaDatas without ref. Mapping table was removed by hibernate with delete messages statement, so we
        // need to find metaDatas without reference in the database.
        long deletedMetaDatas = jpqlQueryFactory
                .delete(QMetaDataEntity.metaDataEntity)
                .where(
                        QMetaDataEntity.metaDataEntity.notIn(messageMetaDatas),
                        QMetaDataEntity.metaDataEntity.notIn(payloadMetaDatas))
                .execute();
        LOGGER.debug("deleted {} metaDatas", deletedMetaDatas);

        // delete payload without ref. Mapping table was removed by hibernate with delete messages and delete metaData
        // statements, so we need to find payloads without reference in the database
        long deletedPayloads = jpqlQueryFactory
                .delete(QPayloadEntity.payloadEntity)
                .where(
                        QPayloadEntity.payloadEntity.notIn(messagePayloads),
                        QPayloadEntity.payloadEntity.notIn(metaDataPayloads))
                .execute();
        LOGGER.debug("deleted {} payloads", deletedPayloads);

        // delete metaDatas again, because they may be references from payloads
        long deletedSecondMetaDatas = jpqlQueryFactory
                .delete(QMetaDataEntity.metaDataEntity)
                .where(
                        QMetaDataEntity.metaDataEntity.notIn(messageMetaDatas),
                        QMetaDataEntity.metaDataEntity.notIn(payloadMetaDatas))
                .execute();
        LOGGER.debug("deleted {} metaDatas (second iteration)", deletedSecondMetaDatas);

        // delete binaries
        long deletedBinaries = jpqlQueryFactory
                .delete(QBinaryEntity.binaryEntity)
                .where(QBinaryEntity.binaryEntity.notIn(JPAExpressions.select(QPayloadEntity.payloadEntity.content)
                        .from(QPayloadEntity.payloadEntity)
                        .where(QPayloadEntity.payloadEntity.content.isNotNull())))
                .execute();
        LOGGER.debug("deleted {} binary contents", deletedBinaries);

        // delete conversations
        long deletedConversations = jpqlQueryFactory
                .delete(QConversationEntity.conversationEntity)
                .where(QConversationEntity.conversationEntity.filter(
                        conversationFilter,
                        applicationSettingsService.getSystemSettingsOrDefault().getTimeZone()))
                .execute();
        LOGGER.debug("deleted {} conversations", deletedConversations);

        long deletedLogs = jpqlQueryFactory
                .delete(QLogEntity.logEntity)
                .where(
                        QLogEntity.logEntity.conversationId.isNotNull(),
                        QLogEntity.logEntity.conversationId.notIn(
                                JPAExpressions.select(QConversationEntity.conversationEntity.conversationId)
                                        .from(QConversationEntity.conversationEntity)))
                .execute();
        LOGGER.debug("deleted filtered conversation {} logs", deletedLogs);

        return PurgeResult.builder()
                .deletedMessages(deletedMessages)
                .deletedConversations(deletedConversations)
                .deletedMetaDatas(deletedMetaDatas + deletedSecondMetaDatas)
                .deletedPayloads(deletedPayloads)
                .deletedBinaries(deletedBinaries)
                .deletedLogs(deletedLogs)
                .deletedManualActions(deletedManualActions)
                .deletedManualActionsPartly(deletedManualActionsPartly)
                .build();
    }
}
