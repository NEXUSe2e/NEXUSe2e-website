package com.evolvsys.nexuse2e.core.security.filter;

import com.evolvsys.app.core.util.RequestUtils;
import com.evolvsys.common.monitoring.DontSkipWhenMonitoring;
import com.evolvsys.nexuse2e.core.config.NEXUSe2eConfigurationProperties;
import jakarta.servlet.FilterChain;
import jakarta.servlet.ServletException;
import jakarta.servlet.http.HttpServletRequest;
import jakarta.servlet.http.HttpServletResponse;
import java.io.IOException;
import lombok.RequiredArgsConstructor;
import lombok.extern.slf4j.Slf4j;
import org.springframework.http.HttpStatus;
import org.springframework.lang.NonNull;
import org.springframework.stereotype.Component;
import org.springframework.web.filter.OncePerRequestFilter;

/**
 * @author Florian Drees
 * @since 21.05.2024
 */
@Component
@DontSkipWhenMonitoring
@RequiredArgsConstructor
@Slf4j
public class HttpsFilter extends OncePerRequestFilter {
    private final NEXUSe2eConfigurationProperties nexuSe2eConfigurationProperties;

    @Override
    protected void doFilterInternal(
            @NonNull HttpServletRequest request,
            @NonNull HttpServletResponse response,
            @NonNull FilterChain filterChain)
            throws ServletException, IOException {
        boolean valid;
        if (RequestUtils.isFrontendOrApiRequest(request)) {
            LOGGER.trace("Validating protocol '{}' for request '{}'", request.getScheme(), request.getRequestURI());
            if (nexuSe2eConfigurationProperties.isRequireHttpsForUi()) {
                if (request.isSecure()) {
                    valid = true;
                } else {
                    valid = false;
                    LOGGER.trace("Https is required for request '{}'", request.getRequestURI());
                }
            } else {
                valid = true;
            }
        } else {
            valid = true;
        }
        if (valid) {
            filterChain.doFilter(request, response);
        } else {
            response.sendError(HttpStatus.FORBIDDEN.value(), "Https is required for request");
        }
    }
}
