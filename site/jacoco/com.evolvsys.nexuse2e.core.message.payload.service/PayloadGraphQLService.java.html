package com.evolvsys.nexuse2e.core.message.payload.service;

import com.evolvsys.app.core.binary.service.BinaryType;
import com.evolvsys.app.core.graphql.annotation.GraphQLService;
import com.evolvsys.app.core.graphql.dataloader.DataLoaderService;
import com.evolvsys.nexuse2e.core.message.model.MetaDataDto;
import com.evolvsys.nexuse2e.core.message.payload.MetaDataDtoComparator;
import com.evolvsys.nexuse2e.core.message.payload.PayloadBinaryContentType;
import com.evolvsys.nexuse2e.core.message.payload.model.PayloadDto;
import io.leangen.graphql.annotations.GraphQLContext;
import io.leangen.graphql.annotations.GraphQLEnvironment;
import io.leangen.graphql.annotations.GraphQLNonNull;
import io.leangen.graphql.annotations.GraphQLQuery;
import io.leangen.graphql.execution.ResolutionEnvironment;
import java.util.Set;
import java.util.SortedSet;
import java.util.TreeSet;
import java.util.concurrent.CompletableFuture;
import lombok.RequiredArgsConstructor;
import org.apache.commons.collections4.CollectionUtils;

/**
 * @author Florian Drees
 * @since 29.02.2024
 */
@GraphQLService
@RequiredArgsConstructor
public class PayloadGraphQLService {
    private final DataLoaderService dataLoaderService;

    @GraphQLQuery
    public String downloadPath(
            @GraphQLContext PayloadDto payloadDto, @GraphQLEnvironment ResolutionEnvironment resolutionEnvironment) {
        return BinaryType.getPath(PayloadBinaryContentType.NAME, String.valueOf(payloadDto.getId()));
    }

    @GraphQLQuery
    public CompletableFuture<Set<@GraphQLNonNull MetaDataDto>> metaDatas(
            @GraphQLContext PayloadDto payloadDto, @GraphQLEnvironment ResolutionEnvironment resolutionEnvironment) {
        return dataLoaderService
                .loadRelation(resolutionEnvironment, PayloadMetaDataBatchLoader.class, payloadDto)
                .thenApply(PayloadGraphQLService::updateMetaDatas);
    }

    private static Set<MetaDataDto> updateMetaDatas(Set<MetaDataDto> metaDataDtos) {
        if (CollectionUtils.isNotEmpty(metaDataDtos)) {
            SortedSet<MetaDataDto> sorted = new TreeSet<>(MetaDataDtoComparator.INSTANCE);
            sorted.addAll(metaDataDtos);
            metaDataDtos = sorted;
        }
        return metaDataDtos;
    }
}
