package com.evolvsys.nexuse2e.core.service.mapper;

import com.evolvsys.app.core.mapper.BaseDtoEntityMapper;
import com.evolvsys.app.core.mapper.DefaultMapperConfiguration;
import com.evolvsys.app.core.mapper.interceptor.InterceptingMappingContext;
import com.evolvsys.nexuse2e.core.model.entity.ServiceEntity;
import com.evolvsys.nexuse2e.core.parameter.mapper.ParameterMapper;
import com.evolvsys.nexuse2e.core.pipeline.mapper.PipelineMapper;
import com.evolvsys.nexuse2e.core.schedule.SchedulingService;
import com.evolvsys.nexuse2e.core.schedule.model.SchedulingDataDto;
import com.evolvsys.nexuse2e.core.service.model.ServiceDto;
import com.evolvsys.nexuse2e.core.service.repository.ServiceRepository;
import org.apache.commons.lang3.StringUtils;
import org.mapstruct.*;
import org.springframework.beans.factory.annotation.Autowired;

/**
 * @author Florian Drees
 * @since 13.12.2023
 */
@Mapper(
        config = DefaultMapperConfiguration.class,
        uses = {PipelineMapper.class, ParameterMapper.class})
public abstract class ServiceMapper extends BaseDtoEntityMapper<ServiceDto, ServiceEntity> {
    private ServiceRepository serviceRepository;
    private SchedulingService schedulingService;

    @Autowired
    protected void config(ServiceRepository serviceRepository, SchedulingService schedulingService) {
        this.serviceRepository = serviceRepository;
        this.schedulingService = schedulingService;
    }

    @Override
    public ServiceEntity idToEntity(Long id, @Context InterceptingMappingContext interceptingMappingContext) {
        return idToEntity(id, interceptingMappingContext, serviceRepository::findById);
    }

    @Mapping(target = "resolverPipelineId", source = "resolverPipeline.id")
    @Mapping(target = "returnPipelineId", source = "returnPipeline.id")
    @Mapping(target = "errorPipelineId", source = "errorPipeline.id")
    @Mapping(target = "parameters", ignore = true)
    @Mapping(target = "schedulingData", source = "entity")
    @InheritConfiguration
    @Override
    protected abstract ServiceDto entityToDto(
            ServiceEntity entity, @Context InterceptingMappingContext interceptingMappingContext);

    protected SchedulingDataDto toSchedulingData(ServiceEntity entity) {
        return StringUtils.isNoneBlank(entity.getServiceId(), entity.getComponentId())
                ? schedulingService.toSchedulingData(entity.getServiceId(), entity.getComponentId())
                : null;
    }

    @Mapping(target = "resolverPipeline", source = "resolverPipelineId")
    @Mapping(target = "returnPipeline", source = "returnPipelineId")
    @Mapping(target = "errorPipeline", source = "errorPipelineId")
    @InheritConfiguration
    @Override
    protected abstract void updateEntity(
            ServiceDto dto,
            @MappingTarget ServiceEntity entity,
            @Context InterceptingMappingContext interceptingMappingContext);
}
