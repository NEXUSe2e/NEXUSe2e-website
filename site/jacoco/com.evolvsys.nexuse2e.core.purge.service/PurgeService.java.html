package com.evolvsys.nexuse2e.core.purge.service;

import com.evolvsys.nexuse2e.core.logging.model.LogFilter;
import com.evolvsys.nexuse2e.core.message.ConversationFilter;
import com.evolvsys.nexuse2e.core.purge.model.PurgeResult;
import com.evolvsys.nexuse2e.core.purge.repository.PurgeRepository;
import lombok.RequiredArgsConstructor;
import lombok.extern.slf4j.Slf4j;
import org.springframework.stereotype.Service;

/**
 * @author SvenBarden
 * @since 06.02.2024
 */
@Service
@RequiredArgsConstructor
@Slf4j
public class PurgeService {
    private final PurgeRepository purgeRepository;

    public PurgeResult purgeAllLogs() {
        LOGGER.debug("deleting all logs");
        return purgeRepository.purgeAllLogs();
    }

    public PurgeResult purgeAllConversations() {
        LOGGER.debug("deleting all conversations");
        return purgeRepository.purgeAllConversations();
    }

    public PurgeResult purgeConversations(ConversationFilter conversationFilter) {
        LOGGER.debug("deleting conversations {}", conversationFilter);
        return purgeRepository.purgeConversations(conversationFilter);
    }

    public PurgeResult purgeLogs(LogFilter logFilter) {
        LOGGER.debug("deleting logs {}", logFilter);
        return purgeRepository.purgeLogs(logFilter);
    }
}
