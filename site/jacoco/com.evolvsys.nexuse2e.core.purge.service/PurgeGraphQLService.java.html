package com.evolvsys.nexuse2e.core.purge.service;

import com.evolvsys.app.core.graphql.annotation.GraphQLService;
import com.evolvsys.nexuse2e.core.exception.NexusException;
import com.evolvsys.nexuse2e.core.logging.model.LogFilter;
import com.evolvsys.nexuse2e.core.manualaction.service.PurgeManualActionService;
import com.evolvsys.nexuse2e.core.message.ConversationFilter;
import com.evolvsys.nexuse2e.core.purge.model.PurgeResult;
import io.leangen.graphql.annotations.GraphQLMutation;
import io.leangen.graphql.annotations.GraphQLNonNull;
import jakarta.validation.Valid;
import lombok.RequiredArgsConstructor;

/**
 * @author SvenBarden
 * @since 06.02.2024
 */
@GraphQLService
@RequiredArgsConstructor
public class PurgeGraphQLService {
    private final PurgeService purgeService;
    private final PurgeManualActionService purgeManualActionService;

    @GraphQLMutation(name = "purge_allLogs")
    @GraphQLNonNull
    public PurgeResult purgeAllLogs() throws NexusException {
        PurgeResult purgeResult = purgeService.purgeAllLogs();
        purgeManualActionService.purgeLogs(null, purgeResult);
        return purgeResult;
    }

    @GraphQLMutation(name = "purge_logs", description = "Purge logs with optional selection")
    @GraphQLNonNull
    public PurgeResult purgeLogs(@Valid LogFilter logFilter) throws NexusException {
        PurgeResult purgeResult = purgeService.purgeLogs(logFilter);
        purgeManualActionService.purgeLogs(logFilter, purgeResult);
        return purgeResult;
    }

    @GraphQLMutation(name = "purge_conversations")
    @GraphQLNonNull
    public PurgeResult purgeConversations(@GraphQLNonNull ConversationFilter conversationFilter) throws NexusException {
        PurgeResult purgeResult = purgeService.purgeConversations(conversationFilter);
        purgeManualActionService.purgeConversations(conversationFilter, purgeResult);
        return purgeResult;
    }

    @GraphQLMutation(name = "purge_allConversations")
    @GraphQLNonNull
    public PurgeResult purgeAllConversations() throws NexusException {
        PurgeResult purgeResult = purgeService.purgeAllConversations();
        purgeManualActionService.purgeConversations(null, purgeResult);
        return purgeResult;
    }
}
