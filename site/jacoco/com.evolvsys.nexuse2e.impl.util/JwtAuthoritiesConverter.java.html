package com.evolvsys.nexuse2e.impl.util;

import java.util.*;
import java.util.stream.Collectors;
import lombok.Getter;
import lombok.RequiredArgsConstructor;
import lombok.Setter;
import org.springframework.core.convert.converter.Converter;
import org.springframework.security.core.GrantedAuthority;
import org.springframework.security.core.authority.SimpleGrantedAuthority;
import org.springframework.security.oauth2.core.ClaimAccessor;
import org.springframework.security.oauth2.jwt.Jwt;
import org.springframework.util.StringUtils;

/**
 * @author SvenBarden
 * @since 13.08.2024
 */
@Setter
@Getter
@RequiredArgsConstructor
public class JwtAuthoritiesConverter implements Converter<Jwt, Collection<GrantedAuthority>> {
    private static final String DEFAULT_AUTHORITIES_CLAIM_DELIMITER = " ";
    private final List<String> claimNames;

    private String authoritiesClaimDelimiter = DEFAULT_AUTHORITIES_CLAIM_DELIMITER;

    @Override
    public Collection<GrantedAuthority> convert(Jwt jwt) {
        Collection<String> authorities = getAuthorities(jwt);
        Collection<GrantedAuthority> grantedAuthorities = new ArrayList<>(authorities.size());
        for (String authority : authorities) {
            grantedAuthorities.add(new SimpleGrantedAuthority(authority));
        }
        return grantedAuthorities;
    }

    @SuppressWarnings("unchecked")
    private Collection<String> getAuthorities(ClaimAccessor jwt, String claimName) {
        Collection<String> result;
        Object authorities = jwt.getClaim(claimName);
        if (authorities instanceof String authorityString) {
            if (StringUtils.hasText(authorityString)) {
                result = Arrays.asList(authorityString.split(this.authoritiesClaimDelimiter));
            } else {
                result = Collections.emptyList();
            }
        } else if (authorities instanceof Collection<?> authoritiesCollection) {
            result = (Collection<String>) authoritiesCollection;
        } else {
            result = Collections.emptyList();
        }
        return result;
    }

    private Collection<String> getAuthorities(Jwt jwt) {
        return claimNames.stream().flatMap(s -> getAuthorities(jwt, s).stream()).collect(Collectors.toSet());
    }
}
