package com.evolvsys.nexuse2e.impl.util;

import com.evolvsys.nexuse2e.core.exception.NexusError;
import com.evolvsys.nexuse2e.core.exception.NexusRuntimeException;
import com.evolvsys.nexuse2e.core.parameter.Parameters;
import com.evolvsys.nexuse2e.core.schedule.SchedulingService;
import com.evolvsys.nexuse2e.core.schedule.SchedulingServiceAware;
import com.evolvsys.nexuse2e.core.schedule.provider.ScheduleProvider;
import com.evolvsys.nexuse2e.core.service.BaseInflowNexusService;
import lombok.experimental.UtilityClass;
import lombok.extern.slf4j.Slf4j;

/**
 * @author SvenBarden
 * @since 16.09.2024
 */
@UtilityClass
@Slf4j
public class SchedulingUtils {

    public static <T extends BaseInflowNexusService & SchedulingServiceAware> void start(
            T schedulingServiceAware, SchedulingService schedulingService, Parameters parameters) {
        ScheduleProvider<?> scheduleProvider =
                schedulingServiceAware.createScheduleProvider(parameters).orElse(null);

        if (scheduleProvider != null) {
            schedulingService.register(schedulingServiceAware, scheduleProvider);
        } else {
            LOGGER.error(
                    "Unable to start service '{}' ({}) as no ScheduleProvider supports the given schedule.",
                    schedulingServiceAware.getServiceId(),
                    schedulingServiceAware.getClass().getSimpleName());
            throw new NexusRuntimeException(NexusError.builder()
                    .message(String.format(
                            "Unable to start service '%s' as no ScheduleProvider supports the given schedule.",
                            schedulingServiceAware.getServiceId()))
                    .build());
        }
    }
}
