package com.evolvsys.nexuse2e.impl.util;

import com.evolvsys.nexuse2e.core.parameter.ParameterDefinition;
import com.evolvsys.nexuse2e.core.parameter.ParameterType;

/**
 *
 */
public interface FileSaveAware {
    String PARAM_TARGET_DIRECTORY = "target_directory";
    String PARAM_TARGET_FILENAME = "target_filename";
    String PARAM_CREATE_DIRECTORY = "create_dir";

    default ParameterDefinition getCreateMissingDirectories() {
        return ParameterDefinition.builder()
                .name(PARAM_CREATE_DIRECTORY)
                .displayName("Create missing directories")
                .description("Automatically create missing directories before saving payloads")
                .type(ParameterType.BOOLEAN)
                .required(true)
                .defaultValue(Boolean.FALSE)
                .build();
    }

    default ParameterDefinition getTargetFilename() {
        return ParameterDefinition.builder()
                .name(PARAM_TARGET_FILENAME)
                .displayName("Target Filename")
                .description("The filename or pattern used when storing the message payload.")
                .defaultValue("{messageId}_{sequenceNumber}{extension}")
                .required(true)
                .type(ParameterType.TRIMMED_STRING)
                .build();
    }

    default ParameterDefinition getTargetDirectory() {
        return ParameterDefinition.builder()
                .name(PARAM_TARGET_DIRECTORY)
                .displayName("Target Directory")
                .description("The target directory where the file is stored.")
                .required(true)
                .type(ParameterType.TRIMMED_STRING)
                .build();
    }
}
