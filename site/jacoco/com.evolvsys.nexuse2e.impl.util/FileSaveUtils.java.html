package com.evolvsys.nexuse2e.impl.util;

import com.evolvsys.nexuse2e.core.exception.NexusError;
import com.evolvsys.nexuse2e.core.exception.NexusException;
import com.evolvsys.nexuse2e.core.message.context.ExecutionContext;
import com.evolvsys.nexuse2e.core.utils.NEXUSe2eFileUtils;
import com.evolvsys.nexuse2e.core.utils.PatternReplaceUtils;
import com.evolvsys.nexuse2e.internal.ContextPayload;
import com.evolvsys.nexuse2e.internal.Message;
import edu.umd.cs.findbugs.annotations.SuppressFBWarnings;
import java.io.File;
import java.io.IOException;
import java.io.InputStream;
import java.io.OutputStream;
import java.nio.file.FileSystemNotFoundException;
import java.nio.file.Files;
import lombok.experimental.UtilityClass;
import lombok.extern.slf4j.Slf4j;
import org.springframework.util.ResourceUtils;

/**
 *
 */
@Slf4j
@UtilityClass
public class FileSaveUtils {

    @SuppressFBWarnings("PATH_TRAVERSAL_IN")
    private static File resolvePath(File directory, String filename) throws NexusException {
        File file;
        try {
            file = new File(directory, filename);
        } catch (FileSystemNotFoundException e) {
            throw new NexusException(
                    NexusError.builder()
                            .message("Error while resolving path with filename")
                            .extension("reason", e.getMessage())
                            .build(),
                    e);
        }
        return file;
    }

    public static void persistPayloads(
            ExecutionContext executionContext,
            Message message,
            String directoryString,
            String filePattern,
            boolean createDirectories)
            throws NexusException, IOException {

        directoryString = PatternReplaceUtils.replaceContextTokens(directoryString, executionContext);

        LOGGER.trace("saving message payloads to directory {}", directoryString);
        File directory = ResourceUtils.getFile(directoryString);

        try {

            if (directory.exists() == false) {
                if (createDirectories) {
                    if (!directory.mkdirs()) {
                        throw new NexusException(NexusError.builder()
                                .message("Directory >" + directoryString + "< could not be created.")
                                .build());
                    }
                } else {
                    throw new NexusException(NexusError.builder()
                            .message("Directory >" + directoryString
                                    + "< does not exist, please create or enable automatic directory creating")
                            .build());
                }
            }
        } catch (RuntimeException e) {
            throw new NexusException(
                    NexusError.builder()
                            .message("Error while checking directory: " + directoryString)
                            .extension("reason", e.getMessage())
                            .build(),
                    e);
        }

        for (ContextPayload payload : message.getPayloads()) {

            String filename = PatternReplaceUtils.replaceTokens(filePattern, executionContext, payload);
            filename = NEXUSe2eFileUtils.cleanFileName(filename, "_");
            LOGGER.trace(
                    "applying filePattern {} to payload {} resulted in fileName {}", filePattern, payload, filename);
            File file = resolvePath(directory, filename);

            LOGGER.trace("writing file: >{}<", file);
            try (OutputStream fos = Files.newOutputStream(file.toPath());
                    InputStream inputStream = payload.getContent().getInputStream()) {
                inputStream.transferTo(fos);
            }
            LOGGER.info("file {} saved.", file);
        }
    }
}
