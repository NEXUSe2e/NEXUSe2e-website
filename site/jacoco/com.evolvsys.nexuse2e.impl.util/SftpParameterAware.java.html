package com.evolvsys.nexuse2e.impl.util;

import com.evolvsys.nexuse2e.core.parameter.ParameterDefinition;
import com.evolvsys.nexuse2e.core.parameter.ParameterType;
import java.time.Duration;
import java.util.List;

public interface SftpParameterAware {

    String PARAM_LOGIN_USERNAME = "login_username";
    String PARAM_SSH_KEY_DATA = "ssh_key_data";
    String PARAM_SSH_KEY_PASSWORD = "ssh_key_password";
    String PARAM_LOGIN_PASSWORD = "login_password";
    String PARAM_HOST = "host";
    String PARAM_PORT = "port";
    String PARAM_HOST_KEY_ALGORITHM = "host_key_algorithm";
    String PARAM_HOST_KEY_DATA = "host_key_data";
    String PARAM_SERVER_PATH = "server_path";
    String PARAM_SERVER_CONNECT_TIMEOUT = "server_connect_timeout";
    String PARAM_ENABLE_DEPRECATED_ALGORITHMS = "enable_deprecated_algorithms";

    default void addSftpParameters(List<ParameterDefinition> parameterDefinitions) {

        parameterDefinitions.addAll(List.of(
                ParameterDefinition.builder()
                        .name(PARAM_LOGIN_USERNAME)
                        .displayName("Login User Name")
                        .description("The Server Login Username.")
                        .type(ParameterType.TRIMMED_STRING)
                        .required(true)
                        .build(),
                ParameterDefinition.builder()
                        .name(PARAM_LOGIN_PASSWORD)
                        .displayName("Login Password")
                        .description("The Login Password. Its optional, if the ssh key is provided.")
                        .type(ParameterType.PASSWORD)
                        .required(false)
                        .build(),
                ParameterDefinition.builder()
                        .name(PARAM_SSH_KEY_DATA)
                        .displayName("SSH Key PEM Data")
                        .description(
                                "Optional ssh key, used for authentication in combination with the proper username.")
                        .type(ParameterType.LARGE_STRING)
                        .required(false)
                        .build(),
                ParameterDefinition.builder()
                        .name(PARAM_SSH_KEY_PASSWORD)
                        .displayName("SSH Key password")
                        .description("Optional password for the also optional ssh key.")
                        .type(ParameterType.PASSWORD)
                        .required(false)
                        .build(),
                ParameterDefinition.builder()
                        .name(PARAM_HOST)
                        .displayName("Hostname")
                        .description("The DNS Hostname or IP for the server.")
                        .type(ParameterType.TRIMMED_STRING)
                        .required(true)
                        .build(),
                ParameterDefinition.builder()
                        .name(PARAM_PORT)
                        .displayName("Port Number")
                        .description("The tcp port number used to connect to the server.")
                        .type(ParameterType.LONG)
                        .defaultValue(22L)
                        .required(true)
                        .build(),
                ParameterDefinition.builder()
                        .name(PARAM_HOST_KEY_ALGORITHM)
                        .displayName("Host Key Algorithm")
                        .description("The algorithm for the optional host key validation.")
                        .type(ParameterType.TRIMMED_STRING)
                        .required(false)
                        .build(),
                ParameterDefinition.builder()
                        .name(PARAM_HOST_KEY_DATA)
                        .displayName("Host Key Data")
                        .description("The base64 encoded key data for the optional host key validation.")
                        .type(ParameterType.TRIMMED_LARGE_STRING)
                        .required(false)
                        .build(),
                ParameterDefinition.builder()
                        .name(PARAM_SERVER_PATH)
                        .displayName("Server Path")
                        .description(
                                "The root path. Must exist as its not created on runtime. Patterns are not allowed.")
                        .type(ParameterType.TRIMMED_STRING)
                        .defaultValue("/")
                        .required(true)
                        .build(),
                ParameterDefinition.builder()
                        .name(PARAM_SERVER_CONNECT_TIMEOUT)
                        .displayName("Connect Timeout")
                        .description("The initial connect and auth validation timeout.")
                        .type(ParameterType.DURATION)
                        .defaultValue(Duration.ofSeconds(10))
                        .required(true)
                        .build(),
                ParameterDefinition.builder()
                        .name(PARAM_ENABLE_DEPRECATED_ALGORITHMS)
                        .displayName("Enable deprecated algorithms")
                        .description("Optionally enable deprecated/insecure algorithms for interoperability reasons.")
                        .type(ParameterType.BOOLEAN)
                        .defaultValue(Boolean.FALSE)
                        .build()));
    }
}
