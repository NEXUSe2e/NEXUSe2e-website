package com.evolvsys.nexuse2e.core.utils;

import com.evolvsys.nexuse2e.core.exception.NexusError;
import com.evolvsys.nexuse2e.core.exception.NexusException;
import java.io.IOException;
import java.util.ArrayList;
import java.util.List;
import java.util.regex.Pattern;
import lombok.experimental.UtilityClass;
import lombok.extern.slf4j.Slf4j;
import org.springframework.core.io.ClassPathResource;
import org.springframework.core.io.FileSystemResource;
import org.springframework.core.io.Resource;
import org.springframework.core.io.support.PathMatchingResourcePatternResolver;

/**
 * @author SvenBarden
 * @since 15.11.2024
 */
@UtilityClass
@Slf4j
public class ResourceUtils {
    private static final Pattern VALID_RESOURCE_PATTERN = Pattern.compile("^.+\\.(?!class$|$)");

    private static String getPath(Resource resource) {
        return switch (resource) {
            case FileSystemResource fileSystemResource -> fileSystemResource.getPath();
            case ClassPathResource classPathResource -> classPathResource.getPath();
            default -> resource.getFilename();
        };
    }

    public static List<ExtendedResource> list(String path) throws NexusException {
        PathMatchingResourcePatternResolver patternResolver = new PathMatchingResourcePatternResolver();

        String finalPath;
        if (path.replace("classpath*", "").contains("*")) {
            finalPath = path;
        } else if (path.endsWith("/")) {
            finalPath = path + "*";
        } else {
            finalPath = path + "/*";
        }

        LOGGER.trace("list resources: {}", finalPath);

        List<ExtendedResource> result = new ArrayList<>();
        try {
            Resource[] resources = patternResolver.getResources(finalPath);
            for (Resource resource : resources) {
                if (resource.getFilename() == null
                        || VALID_RESOURCE_PATTERN
                                .matcher(resource.getFilename())
                                .find()) {
                    LOGGER.trace("add resource {}: {}", resource.getFilename(), resource);

                    String rawPath = getPath(resource);

                    result.add(new ExtendedResource(resource, rawPath));
                } else {
                    LOGGER.trace("skip resource {}: {}", resource.getFilename(), resource);
                }
            }
        } catch (IOException e) {
            throw new NexusException(
                    NexusError.builder()
                            .message("Could not load resource list")
                            .extension("path", finalPath)
                            .build(),
                    e);
        }
        return result;
    }

    public static String getParent(String path) {
        String finalPath;
        if (path.endsWith("/")) {
            finalPath = path.substring(0, path.length() - 1);
        } else {
            finalPath = path;
        }
        int lastSlash = finalPath.lastIndexOf('/');
        String result;
        if (lastSlash == -1) {
            result = "";
        } else {
            result = finalPath.substring(0, lastSlash);
            int secondLastSlash = result.lastIndexOf('/');
            if (secondLastSlash != -1) {
                result = result.substring(result.lastIndexOf('/') + 1);
            }
        }
        return result;
    }
}
