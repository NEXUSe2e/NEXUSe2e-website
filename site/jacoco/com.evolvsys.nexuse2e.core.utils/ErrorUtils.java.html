package com.evolvsys.nexuse2e.core.utils;

import com.evolvsys.app.core.exception.Error;
import com.evolvsys.nexuse2e.core.exception.NexusErrorContainer;
import java.util.Map;
import lombok.experimental.UtilityClass;
import org.apache.commons.lang3.StringUtils;

/**
 * @author SvenBarden
 * @since 14.03.2024
 */
@UtilityClass
public class ErrorUtils {

    public static String toString(Error error) {
        StringBuilder s = new StringBuilder(27);
        s.append("message: '").append(error.getMessage()).append('\'');
        if (StringUtils.isNotBlank(error.getCode())) {
            s.append(", code: '").append(error.getCode()).append('\'');
        }
        for (Map.Entry<String, Object> entry : error.getExtensions().entrySet()) {
            s.append(", ")
                    .append(entry.getKey())
                    .append(": '")
                    .append(entry.getValue())
                    .append('\'');
        }
        return s.toString();
    }

    public static String toString(Throwable throwable) {
        StringBuilder s = new StringBuilder("type: '")
                .append(throwable.getClass().getSimpleName())
                .append('\'');
        if (throwable instanceof NexusErrorContainer errorContainer) {
            s.append(", ").append(toString(errorContainer.getError()));
        } else {
            s.append(", message: '").append(throwable.getMessage()).append('\'');
        }
        return s.toString();
    }

    @SuppressWarnings("PMD.CompareObjectsWithEquals")
    public static String aggregate(Throwable throwable) {
        StringBuilder s = new StringBuilder();

        Throwable next = throwable;
        while (next != null) {
            if (next != throwable) {
                s.append(", ");
            }
            s.append('{').append(toString(next)).append('}');
            next = next.getCause();
        }

        return s.toString();
    }
}
