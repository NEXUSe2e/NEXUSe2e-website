package com.evolvsys.nexuse2e.core.utils;

import java.io.File;
import java.net.URL;
import java.util.Collection;
import java.util.Map;
import java.util.function.Function;
import java.util.regex.Pattern;
import lombok.experimental.UtilityClass;
import org.apache.commons.io.FileUtils;
import org.apache.commons.io.filefilter.FalseFileFilter;
import org.apache.commons.io.filefilter.IOFileFilter;
import org.apache.commons.io.filefilter.TrueFileFilter;
import org.apache.commons.lang3.Validate;

/**
 * @author SvenBarden
 * @since 20.11.2023
 */
@UtilityClass
public class NEXUSe2eFileUtils {
    private static final Pattern INVALID_FILENAME_PATTERN = Pattern.compile("[^a-zA-Z0-9.\\-_ ]");
    private static final Function<String, File> CLASSPATH_DIRECTORY_FINDER = directory -> {
        if (directory.startsWith("/") == false) {
            throw new IllegalArgumentException("classpath path " + directory + " must be absolute");
        }
        URL resource = NEXUSe2eFileUtils.class.getResource(directory);
        return FileUtils.toFile(resource);
    };
    private static final Function<String, File> FILE_DIRECTORY_FINDER = File::new;
    private static final Function<String, File> DEFAULT_DIRECTORY_FINDER = FILE_DIRECTORY_FINDER;

    private static final Map<String, Function<String, File>> FILE_FINDERS =
            Map.of("classpath:", CLASSPATH_DIRECTORY_FINDER, "file:", FILE_DIRECTORY_FINDER);

    public static Collection<File> getFiles(String directory) {
        return getFiles(directory, true);
    }

    public static Collection<File> getFiles(String directory, boolean recursive) {
        Validate.notNull(directory, "directory must not be null");
        File fileDirectory = FILE_FINDERS.entrySet().stream()
                .filter(entry -> directory.startsWith(entry.getKey()))
                .findFirst()
                .map(entry -> entry.getValue()
                        .apply(directory.substring(entry.getKey().length())))
                .orElseGet(() -> DEFAULT_DIRECTORY_FINDER.apply(directory));

        IOFileFilter dirFilter = recursive ? TrueFileFilter.TRUE : FalseFileFilter.FALSE;
        return FileUtils.listFiles(fileDirectory, TrueFileFilter.TRUE, dirFilter);
    }

    public static String cleanFileName(String fileName, String replacement) {
        return INVALID_FILENAME_PATTERN.matcher(fileName).replaceAll(replacement);
    }

    public static String cleanFileName(String fileName) {
        return cleanFileName(fileName, "_");
    }
}
