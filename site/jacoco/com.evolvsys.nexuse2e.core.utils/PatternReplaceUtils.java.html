package com.evolvsys.nexuse2e.core.utils;

import com.evolvsys.nexuse2e.core.message.context.ExecutionContext;
import com.evolvsys.nexuse2e.internal.ContextPayload;
import lombok.experimental.UtilityClass;

/**
 * @author GuidoEsch
 * @since 06.02.2024
 */
@UtilityClass
public class PatternReplaceUtils {

    public static String replaceTokens(String targetValue, ExecutionContext executionContext, ContextPayload payload) {
        if (targetValue == null) {
            return null;
        }
        return replacePayloadTokens(replaceContextTokens(targetValue, executionContext), payload);
    }

    public static String replaceContextTokens(String targetValue, ExecutionContext executionContext) {
        String result = targetValue;
        for (ContextPatternToken patternToken : ContextPatternToken.values()) {
            result = patternToken.applyPattern(result, executionContext);
        }
        return result;
    }

    public static String replacePayloadTokens(String targetValue, ContextPayload payload) {
        String result = targetValue;
        for (PayloadPatternToken patternToken : PayloadPatternToken.values()) {
            result = patternToken.applyPattern(result, payload);
        }
        return result;
    }
}
