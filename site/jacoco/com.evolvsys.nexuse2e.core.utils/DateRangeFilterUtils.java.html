package com.evolvsys.nexuse2e.core.utils;

import com.querydsl.core.BooleanBuilder;
import com.querydsl.core.types.Predicate;
import com.querydsl.core.types.dsl.DateTimePath;
import java.time.*;
import java.util.function.BiFunction;
import lombok.experimental.UtilityClass;
import org.springframework.lang.Nullable;

/**
 * @author SvenBarden
 * @since 27.03.2024
 */
@UtilityClass
public class DateRangeFilterUtils {

    @Nullable
    public static Predicate filter(
            DateTimePath<ZonedDateTime> dateTimePath,
            String timezone,
            @Nullable Range range,
            @Nullable ZonedDateTime from,
            @Nullable ZonedDateTime till) {
        BooleanBuilder builder = new BooleanBuilder();
        if (range != null) {
            builder.and(range.rangeFunction.apply(dateTimePath, ZoneId.of(timezone)));
        }

        if (from != null) {
            builder.and(dateTimePath.goe(from));
        }
        if (till != null) {
            builder.and(dateTimePath.loe(till));
        }
        return builder.getValue();
    }

    public enum Range {
        LAST_24_H((dateTimePath, zoneId) ->
                dateTimePath.goe(ZonedDateTime.now(zoneId).minusHours(24))),
        TODAY((dateTimePath, zoneId) -> dateTimePath.goe(LocalDate.now(zoneId).atStartOfDay(zoneId))),
        YESTERDAY((dateTimePath, zoneId) -> dateTimePath.between(
                LocalDate.now(zoneId).minusDays(1).atStartOfDay(zoneId),
                LocalDate.now(zoneId).minusDays(1).atTime(LocalTime.MAX).atZone(zoneId))),
        THIS_WEEK((dateTimePath, zoneId) ->
                dateTimePath.goe(LocalDate.now(zoneId).atStartOfDay(zoneId).with(DayOfWeek.MONDAY))),
        LAST_WEEK((dateTimePath, zoneId) -> dateTimePath.between(
                LocalDate.now(zoneId)
                        .atStartOfDay(zoneId)
                        .with(DayOfWeek.MONDAY)
                        .minusWeeks(1),
                LocalDate.now(zoneId)
                        .atTime(LocalTime.MAX)
                        .atZone(zoneId)
                        .with(DayOfWeek.SUNDAY)
                        .minusWeeks(1))),
        THIS_MONTH((dateTimePath, zoneId) ->
                dateTimePath.goe(LocalDate.now(zoneId).atStartOfDay(zoneId).withDayOfMonth(1))),
        LAST_7_DAYS((dateTimePath, zoneId) ->
                dateTimePath.goe(LocalDate.now(zoneId).atStartOfDay(zoneId).minusDays(6))),
        LAST_30_DAYS((dateTimePath, zoneId) ->
                dateTimePath.goe(LocalDate.now(zoneId).atStartOfDay(zoneId).minusDays(29)));

        private final BiFunction<DateTimePath<ZonedDateTime>, ZoneId, Predicate> rangeFunction;

        Range(BiFunction<DateTimePath<ZonedDateTime>, ZoneId, Predicate> rangeFunction) {
            this.rangeFunction = rangeFunction;
        }
    }
}
