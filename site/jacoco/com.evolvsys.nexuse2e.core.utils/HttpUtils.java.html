package com.evolvsys.nexuse2e.core.utils;

import java.util.List;
import java.util.regex.Matcher;
import java.util.regex.Pattern;
import lombok.experimental.UtilityClass;
import lombok.extern.slf4j.Slf4j;
import org.apache.commons.lang3.StringUtils;
import org.springframework.http.HttpStatusCode;

/**
 * @author SvenBarden
 * @since 30.04.2024
 */
@UtilityClass
@Slf4j
public class HttpUtils {
    private static final Pattern VALID_STATUS_PATTERN = Pattern.compile("[\\dx]{3}");
    private static final Pattern CONTENT_TYPE_FIX_PATTERN =
            Pattern.compile("(?!\\s+;)([A-Za-z0-9]+=)([^\"]+?)($|(;\\s+))");

    public static String fixContentType(String contentType) {
        StringBuilder result = new StringBuilder();
        if (StringUtils.isNotEmpty(contentType)) {
            Matcher matcher = CONTENT_TYPE_FIX_PATTERN.matcher(contentType);
            while (matcher.find()) {
                matcher.appendReplacement(result, matcher.group(1) + "\"" + matcher.group(2) + "\"" + matcher.group(3));
            }
            matcher.appendTail(result);
        }
        return result.toString();
    }

    public static boolean validResponseStatus(HttpStatusCode httpStatusCode, List<String> validCodes) {
        return validCodes.stream().anyMatch(code -> validResponseStatus(httpStatusCode, code));
    }

    public static boolean validResponseStatus(HttpStatusCode httpStatusCode, String validStatus) {
        boolean valid;
        if (VALID_STATUS_PATTERN.matcher(validStatus).matches()) {
            valid = Pattern.compile(validStatus.replace("x", "\\d"))
                    .matcher(String.valueOf(httpStatusCode.value()))
                    .matches();
        } else {
            valid = validStatus.equals(String.valueOf(httpStatusCode.value()));
        }
        return valid;
    }
}
