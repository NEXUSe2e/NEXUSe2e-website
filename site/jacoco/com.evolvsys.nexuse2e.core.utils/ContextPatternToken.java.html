package com.evolvsys.nexuse2e.core.utils;

import com.evolvsys.nexuse2e.core.message.context.ExecutionContext;
import java.time.ZonedDateTime;
import java.time.format.DateTimeFormatter;
import java.util.function.BiFunction;
import lombok.Getter;
import org.apache.commons.lang3.ObjectUtils;

/**
 * @author GuidoEsch
 * @since 06.02.2024
 */
@Getter
public enum ContextPatternToken {
    CHOREOGRAPHY_ID(
            "{choreographyId}",
            (targetString, executionContext) -> targetString.replace(
                    "{choreographyId}", ObjectUtils.defaultIfNull(executionContext.getChoreographyId(), ""))),
    ACTION_ID(
            "{actionId}",
            (targetString, executionContext) ->
                    targetString.replace("{actionId}", ObjectUtils.defaultIfNull(executionContext.getActionId(), ""))),
    TO_PARTNER_ID(
            "{toPartnerId}",
            (targetString, executionContext) -> targetString.replace(
                    "{toPartnerId}", ObjectUtils.defaultIfNull(executionContext.getTargetPartnerId(), ""))),
    FROM_PARTNER_ID(
            "{fromPartnerId}",
            (targetString, executionContext) -> targetString.replace(
                    "{fromPartnerId}", ObjectUtils.defaultIfNull(executionContext.getSourcePartnerId(), ""))),
    MESSAGE_ID(
            "{messageId}",
            (targetString, executionContext) -> targetString.replace(
                    "{messageId}",
                    ObjectUtils.defaultIfNull(executionContext.getMessage().getId(), ""))),
    CONVERSATION_ID(
            "{conversationId}",
            (targetString, executionContext) -> targetString.replace(
                    "{conversationId}",
                    ObjectUtils.defaultIfNull(executionContext.getMessage().getConversationId(), ""))),
    TIMESTAMP_NOW_EBXML("{tsEbxml}", (targetString, executionContext) -> {
        ZonedDateTime date = ZonedDateTime.now();
        return targetString.replace("{tsEbxml}", date.format(DateTimeFormatter.ISO_INSTANT));
    }),
    TIMESTAMP_NOW_SHORT("{tsShort}", (targetString, executionContext) -> {
        ZonedDateTime date = ZonedDateTime.now();
        return targetString.replace("{tsShort}", date.format(DateTimeFormatter.ofPattern("yyyyMMddHHmmssSSS")));
    }),
    TIMESTAMP_NOW_ISO_TZ("{tsIsoTZ}", (targetString, executionContext) -> {
        ZonedDateTime date = ZonedDateTime.now();
        return targetString.replace("{tsIsoTZ}", date.format(DateTimeFormatter.ISO_OFFSET_DATE_TIME));
    });

    private final String pattern;
    private final BiFunction<String, ExecutionContext, String> function;

    ContextPatternToken(String pattern, BiFunction<String, ExecutionContext, String> function) {
        this.pattern = pattern;
        this.function = function;
    }

    public String applyPattern(String targetString, ExecutionContext executionContext) {
        return function.apply(targetString, executionContext);
    }
}
