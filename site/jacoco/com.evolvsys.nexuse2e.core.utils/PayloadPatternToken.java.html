package com.evolvsys.nexuse2e.core.utils;

import com.evolvsys.nexuse2e.internal.ContextPayload;
import java.util.function.BiFunction;
import lombok.Getter;
import lombok.extern.slf4j.Slf4j;
import org.apache.commons.lang3.ObjectUtils;
import org.apache.tika.config.TikaConfig;
import org.apache.tika.mime.MimeTypeException;

/**
 * @author SvenBarden
 * @since 24.04.2024
 */
@Getter
@Slf4j
public enum PayloadPatternToken {
    CONTENT_ID(
            "{contentId}",
            (s, payload) -> s.replace("{contentId}", ObjectUtils.defaultIfNull(payload.getContentId(), ""))),
    SEQUENCE_NUMBER(
            "{sequenceNumber}",
            (s, payload) -> s.replace("{sequenceNumber}", String.valueOf(payload.getSequenceNumber()))),
    EXTENSION("{extension}", (s, payload) -> {
        String extension;
        try {
            extension = TikaConfig.getDefaultConfig()
                    .getMimeRepository()
                    .forName(payload.getContentType())
                    .getExtension();
        } catch (MimeTypeException e) {
            LOGGER.warn(
                    "could not detect extension for content type {} in payload {} {}",
                    payload.getContentType(),
                    payload.getContentId(),
                    payload.getSequenceNumber(),
                    e);
            extension = ".dat";
        }
        return s.replace("{extension}", extension);
    });

    private final String pattern;
    private final BiFunction<String, ContextPayload, String> function;

    PayloadPatternToken(String pattern, BiFunction<String, ContextPayload, String> function) {
        this.pattern = pattern;
        this.function = function;
    }

    public String applyPattern(String targetString, ContextPayload payload) {
        return function.apply(targetString, payload);
    }
}
