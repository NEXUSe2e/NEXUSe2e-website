package com.evolvsys.nexuse2e.impl.database;

import com.evolvsys.nexuse2e.core.database.DatabaseUtils;
import com.evolvsys.nexuse2e.core.database.DynamicDatabaseValue;
import com.evolvsys.nexuse2e.core.exception.NexusError;
import com.evolvsys.nexuse2e.core.exception.NexusException;
import com.evolvsys.nexuse2e.core.message.context.ExecutionContext;
import java.sql.Connection;
import java.sql.SQLException;
import java.sql.Statement;
import java.util.Optional;
import javax.sql.DataSource;
import lombok.RequiredArgsConstructor;
import org.apache.commons.lang3.StringUtils;
import org.springframework.jdbc.core.namedparam.MapSqlParameterSource;
import org.springframework.jdbc.core.namedparam.NamedParameterJdbcTemplate;
import org.springframework.lang.NonNull;

/**
 * @author Florian Drees
 * @since 21.06.2024
 */
@RequiredArgsConstructor
public class IncrementPrimaryKeyIdDbValue extends DynamicDatabaseValue<Long> {
    private final String tableName;
    private final String incrementField;

    @Override
    public Long getValue(
            @NonNull NamedParameterJdbcTemplate namedParameterJdbcTemplate, @NonNull ExecutionContext executionContext)
            throws NexusException {
        if (StringUtils.isBlank(tableName) || StringUtils.isBlank(incrementField)) {
            throw new NexusException(NexusError.builder()
                    .message("context is missing required fields 'table_name' and/or 'increment_field'")
                    .build());
        }
        DataSource dataSource = namedParameterJdbcTemplate.getJdbcTemplate().getDataSource();
        if (dataSource == null) {
            throw new NexusException(NexusError.builder()
                    .message("Could not get data source from named parameter jdbc template")
                    .build());
        }
        try (Connection connection = dataSource.getConnection();
                Statement statement = connection.createStatement()) {
            String sql = String.format(
                    "SELECT MAX(%s) FROM %s",
                    DatabaseUtils.enquoteIdentifier(statement, incrementField),
                    DatabaseUtils.enquoteIdentifier(statement, tableName));
            return Optional.ofNullable(
                            namedParameterJdbcTemplate.queryForObject(sql, new MapSqlParameterSource(), Long.class))
                    .map(result -> result + 1L)
                    .orElse(1L);
        } catch (SQLException e) {
            throw new NexusException(
                    NexusError.builder()
                            .message("Could not open jdbc database connection from datasource")
                            .build(),
                    e);
        }
    }
}
