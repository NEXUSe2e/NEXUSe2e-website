package com.evolvsys.nexuse2e.core.config.model;

import com.evolvsys.nexuse2e.core.component.ComponentType;
import com.evolvsys.nexuse2e.core.config.validation.*;
import com.fasterxml.jackson.dataformat.xml.annotation.JacksonXmlCData;
import edu.umd.cs.findbugs.annotations.SuppressFBWarnings;
import io.leangen.graphql.annotations.GraphQLNonNull;
import io.leangen.graphql.annotations.GraphQLQuery;
import jakarta.validation.Valid;
import jakarta.validation.constraints.NotBlank;
import jakarta.validation.groups.Default;
import jakarta.xml.bind.annotation.XmlAttribute;
import jakarta.xml.bind.annotation.XmlElement;
import jakarta.xml.bind.annotation.XmlElementWrapper;
import jakarta.xml.bind.annotation.XmlType;
import java.util.ArrayList;
import java.util.List;
import lombok.Data;

/**
 * @author SvenBarden
 * @since 26.10.2023
 */
@Data
@XmlType(
        name = "service",
        propOrder = {})
@ValidServiceParameters(payload = ValidationLevel.Warn.class)
public class ServiceConfig implements ConfigurableConfig {
    @XmlAttribute(required = true)
    @NotBlank(payload = ValidationLevel.Fatal.class, groups = Default.class)
    private String serviceId;

    @XmlAttribute(required = true)
    @NotBlank(payload = ValidationLevel.Fatal.class, groups = Default.class)
    @ValidConfigComponentId(
            types = {ComponentType.INFLOW_SERVICE, ComponentType.OUTFLOW_SERVICE, ComponentType.SUPPORT_SERVICE},
            payload = ValidationLevel.Error.class)
    private String componentId;

    @XmlAttribute
    private Boolean enabled = Boolean.FALSE;

    @XmlAttribute
    @ValidConfigPipelineId(payload = ValidationLevel.Error.class)
    private String resolverPipeline;

    @XmlAttribute
    @ValidConfigPipelineId(payload = ValidationLevel.Error.class)
    private String returnPipeline;

    @XmlAttribute
    @ValidConfigPipelineId(payload = ValidationLevel.Error.class)
    private String errorPipeline;

    @JacksonXmlCData
    @XmlElement
    private String description;

    @XmlElement(name = "parameter")
    @XmlElementWrapper
    @GraphQLQuery(name = "parameters")
    @Valid
    @ValidConfigUniqueIdentifier(propertyNames = "name", payload = ValidationLevel.Fatal.class)
    private List<@GraphQLNonNull ParameterConfig> parameters = new ArrayList<>();

    @Override // required for jaxb schema generation
    @SuppressFBWarnings(value = "EI_EXPOSE_REP", justification = "intentional modification")
    public List<@GraphQLNonNull ParameterConfig> getParameters() {
        return parameters;
    }

    @Override // required for jaxb schema generation
    public String getComponentId() {
        return componentId;
    }
}
