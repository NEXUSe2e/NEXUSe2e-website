package com.evolvsys.nexuse2e.core.route.service;

import com.evolvsys.app.core.graphql.annotation.GraphQLService;
import com.evolvsys.app.core.graphql.dataloader.DataLoaderService;
import com.evolvsys.nexuse2e.core.certificate.model.CertificateContainerDto;
import com.evolvsys.nexuse2e.core.certificate.service.CertificateContainerBatchLoader;
import com.evolvsys.nexuse2e.core.config.service.ConfigGraphQLService;
import com.evolvsys.nexuse2e.core.partner.model.PartnerConnectionDto;
import com.evolvsys.nexuse2e.core.partner.model.PartnerDto;
import com.evolvsys.nexuse2e.core.partner.service.PartnerBatchLoader;
import com.evolvsys.nexuse2e.core.pipeline.PipelineBatchLoader;
import com.evolvsys.nexuse2e.core.pipeline.model.PipelineDto;
import com.evolvsys.nexuse2e.core.route.RouteFilter;
import com.evolvsys.nexuse2e.core.route.model.RouteDto;
import com.evolvsys.nexuse2e.core.route.repository.PartnerConnectionBatchLoader;
import com.evolvsys.nexuse2e.core.route.validation.RouteValidationGroups;
import com.evolvsys.nexuse2e.core.route.validation.ValidRouteHasActions;
import com.evolvsys.nexuse2e.core.route.validation.ValidRouteHasChoreographies;
import io.leangen.graphql.annotations.*;
import io.leangen.graphql.execution.ResolutionEnvironment;
import jakarta.validation.Valid;
import jakarta.validation.groups.Default;
import java.util.List;
import java.util.Optional;
import java.util.concurrent.CompletableFuture;
import lombok.RequiredArgsConstructor;
import lombok.extern.slf4j.Slf4j;
import org.springframework.validation.annotation.Validated;

/**
 * @author Florian Drees
 * @since 29.12.2023
 */
@GraphQLService
@RequiredArgsConstructor
@Slf4j
public class RouteGraphQLService {
    private final DataLoaderService dataLoaderService;
    private final RouteService routeService;

    @GraphQLMutation(
            name = "route_update",
            description = "Updates an existing or creates a new route for the choreography")
    @GraphQLNonNull
    @Validated({Default.class, RouteValidationGroups.UpdateAction.class})
    public RouteDto update(@Valid @GraphQLNonNull RouteDto routeDto) {
        return ConfigGraphQLService.notifySubscribers(routeService.save(routeDto));
    }

    @GraphQLMutation(
            name = "route_delete",
            description = "Deletes a route. The configuration of partners and connections are not affected")
    public boolean delete(@ValidRouteHasChoreographies @ValidRouteHasActions long routeId) {
        routeService.delete(routeId);
        ConfigGraphQLService.notifySubscribers();
        return true;
    }

    @GraphQLQuery(name = "route_list", description = "List all routes")
    public List<@GraphQLNonNull RouteDto> list(RouteFilter routeFilter) {
        return routeService.list(routeFilter);
    }

    @GraphQLQuery(name = "route_getById", description = "Gets a route")
    public Optional<RouteDto> getById(long routeId) {
        return routeService.getById(routeId);
    }

    @GraphQLQuery
    public CompletableFuture<PartnerConnectionDto> firstPartnerConnection(
            @GraphQLContext RouteDto routeDto, @GraphQLEnvironment ResolutionEnvironment resolutionEnvironment) {
        return dataLoaderService.load(
                resolutionEnvironment,
                PartnerConnectionBatchLoader.class,
                routeDto,
                "firstPartnerConnectionId",
                RouteDto::getFirstPartnerConnectionId);
    }

    @GraphQLQuery
    public CompletableFuture<PartnerDto> firstPartner(
            @GraphQLContext RouteDto routeDto, @GraphQLEnvironment ResolutionEnvironment resolutionEnvironment) {
        return dataLoaderService.load(
                resolutionEnvironment,
                PartnerBatchLoader.class,
                routeDto,
                "firstPartnerId",
                RouteDto::getFirstPartnerId);
    }

    @GraphQLQuery
    public CompletableFuture<CertificateContainerDto> firstNexusCertificateContainer(
            @GraphQLContext RouteDto routeDto, @GraphQLEnvironment ResolutionEnvironment resolutionEnvironment) {
        return dataLoaderService.load(
                resolutionEnvironment,
                CertificateContainerBatchLoader.class,
                routeDto,
                "firstNexusCertificateContainerId",
                RouteDto::getFirstNexusCertificateContainerId);
    }

    @GraphQLQuery
    public CompletableFuture<PipelineDto> firstInflowPipeline(
            @GraphQLContext RouteDto routeDto, @GraphQLEnvironment ResolutionEnvironment resolutionEnvironment) {
        return dataLoaderService.load(
                resolutionEnvironment,
                PipelineBatchLoader.class,
                routeDto,
                "firstInflowPipelineId",
                RouteDto::getFirstInflowPipelineId);
    }

    @GraphQLQuery
    public CompletableFuture<PartnerConnectionDto> secondPartnerConnection(
            @GraphQLContext RouteDto routeDto, @GraphQLEnvironment ResolutionEnvironment resolutionEnvironment) {
        return dataLoaderService.load(
                resolutionEnvironment,
                PartnerConnectionBatchLoader.class,
                routeDto,
                "secondPartnerConnectionId",
                RouteDto::getSecondPartnerConnectionId);
    }

    @GraphQLQuery
    public CompletableFuture<PartnerDto> secondPartner(
            @GraphQLContext RouteDto routeDto, @GraphQLEnvironment ResolutionEnvironment resolutionEnvironment) {
        return dataLoaderService.load(
                resolutionEnvironment,
                PartnerBatchLoader.class,
                routeDto,
                "secondPartnerId",
                RouteDto::getSecondPartnerId);
    }

    @GraphQLQuery
    public CompletableFuture<CertificateContainerDto> secondNexusCertificateContainer(
            @GraphQLContext RouteDto routeDto, @GraphQLEnvironment ResolutionEnvironment resolutionEnvironment) {
        return dataLoaderService.load(
                resolutionEnvironment,
                CertificateContainerBatchLoader.class,
                routeDto,
                "secondNexusCertificateContainerId",
                RouteDto::getSecondNexusCertificateContainerId);
    }

    @GraphQLQuery
    public CompletableFuture<PipelineDto> secondInflowPipeline(
            @GraphQLContext RouteDto routeDto, @GraphQLEnvironment ResolutionEnvironment resolutionEnvironment) {
        return dataLoaderService.load(
                resolutionEnvironment,
                PipelineBatchLoader.class,
                routeDto,
                "secondInflowPipelineId",
                RouteDto::getSecondInflowPipelineId);
    }
}
