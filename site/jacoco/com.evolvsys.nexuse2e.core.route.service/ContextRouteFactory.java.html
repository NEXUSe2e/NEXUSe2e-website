package com.evolvsys.nexuse2e.core.route.service;

import com.evolvsys.app.core.persistence.model.IdEntity;
import com.evolvsys.nexuse2e.core.certificate.mapper.CertificateContainerKeyMapper;
import com.evolvsys.nexuse2e.core.certificate.model.CertificateContainerEntity;
import com.evolvsys.nexuse2e.core.certificate.repository.CertificateContainerRepository;
import com.evolvsys.nexuse2e.core.exception.NexusError;
import com.evolvsys.nexuse2e.core.exception.NexusRuntimeException;
import com.evolvsys.nexuse2e.core.message.context.ContextObjectFactory;
import com.evolvsys.nexuse2e.core.message.context.RouteKey;
import com.evolvsys.nexuse2e.core.model.entity.PartnerConnectionEntity;
import com.evolvsys.nexuse2e.core.model.entity.PartnerEntity;
import com.evolvsys.nexuse2e.core.model.entity.RouteEntity;
import com.evolvsys.nexuse2e.core.partner.mapper.ContextConnectionMapper;
import com.evolvsys.nexuse2e.core.pipeline.PipelineFactory;
import com.evolvsys.nexuse2e.core.pipeline.model.Pipeline;
import com.evolvsys.nexuse2e.core.pipeline.model.PipelineEntity;
import com.evolvsys.nexuse2e.core.route.model.ContextRoute;
import com.evolvsys.nexuse2e.core.route.repository.RouteRepository;
import java.security.cert.X509Certificate;
import java.util.Collection;
import java.util.List;
import java.util.Optional;
import java.util.function.Function;
import lombok.RequiredArgsConstructor;
import one.util.streamex.StreamEx;
import org.apache.commons.lang3.StringUtils;
import org.springframework.data.util.Lazy;
import org.springframework.stereotype.Component;
import org.springframework.transaction.annotation.Transactional;

/**
 * @author SvenBarden
 * @since 12.02.2024
 */
@Component
@RequiredArgsConstructor
public class ContextRouteFactory implements ContextObjectFactory<RouteKey, ContextRoute> {
    private final RouteRepository routeRepository;
    private final ContextConnectionMapper contextConnectionMapper;
    private final CertificateContainerKeyMapper certificateContainerKeyMapper;
    private final PipelineFactory pipelineFactory;
    private final CertificateContainerRepository certificateContainerRepository;

    private Long getSaveId(IdEntity idEntity) {
        return idEntity == null ? null : idEntity.getId();
    }

    private Long getContainerIdFromConnection(PartnerConnectionEntity partnerConnectionEntity) {
        return partnerConnectionEntity == null || partnerConnectionEntity.getCertificateContainer() == null
                ? null
                : partnerConnectionEntity.getCertificateContainer().getId();
    }

    private List<Long> getAllContainerIdsFromPartner(PartnerEntity partnerEntity) {
        return partnerEntity == null
                ? List.of()
                : certificateContainerRepository.findByPartner(partnerEntity).stream()
                        .map(CertificateContainerEntity::getId)
                        .toList();
    }

    private List<X509Certificate> containerIdsToCerts(Collection<Long> containerIds) {
        return StreamEx.of(containerIds)
                .map(certificateContainerKeyMapper::idToCertificates)
                .toFlatList(Function.identity());
    }

    private Pipeline getPipeline(PipelineEntity pipelineEntity) {
        return pipelineEntity == null ? null : pipelineFactory.get(pipelineEntity);
    }

    private ContextRoute createContextRoute(
            RouteEntity routeEntity,
            Function<RouteEntity, PartnerEntity> sourcePartner,
            Function<RouteEntity, PartnerEntity> targetPartner,
            Function<RouteEntity, PartnerConnectionEntity> sourceConnection,
            Function<RouteEntity, PartnerConnectionEntity> targetConnection,
            Function<RouteEntity, CertificateContainerEntity> sourceKeyContainer,
            Function<RouteEntity, CertificateContainerEntity> targetKeyContainer,
            Function<RouteEntity, PipelineEntity> inflowPipeline) {

        return ContextRoute.builder()
                .id(routeEntity.getId())
                .routeId(routeEntity.getRouteId())
                .sourceConnection(contextConnectionMapper.entityToConnection(sourceConnection.apply(routeEntity)))
                .sourceLocalKey(Lazy.of(() ->
                        certificateContainerKeyMapper.idToKeyStore(getSaveId(sourceKeyContainer.apply(routeEntity)))))
                .sourceConnectionTrustCertificates(Lazy.of(() -> certificateContainerKeyMapper.idToCertificates(
                        getContainerIdFromConnection(sourceConnection.apply(routeEntity)))))
                .sourcePartnerTrustCertificates(Lazy.of(
                        () -> containerIdsToCerts(getAllContainerIdsFromPartner(sourcePartner.apply(routeEntity)))))
                .targetConnection(contextConnectionMapper.entityToConnection(targetConnection.apply(routeEntity)))
                .targetLocalKey(Lazy.of(() ->
                        certificateContainerKeyMapper.idToKeyStore(getSaveId(targetKeyContainer.apply(routeEntity)))))
                .targetConnectionTrustCertificates(Lazy.of(() -> certificateContainerKeyMapper.idToCertificates(
                        getContainerIdFromConnection(targetConnection.apply(routeEntity)))))
                .targetPartnerTrustCertificates(Lazy.of(
                        () -> containerIdsToCerts(getAllContainerIdsFromPartner(targetPartner.apply(routeEntity)))))
                .allowSamePartnerType(routeEntity.isAllowSamePartnerType())
                .inflowPipeline(getPipeline(inflowPipeline.apply(routeEntity)))
                .build();
    }

    private ContextRoute toRoute(String sourcePartner, RouteEntity routeEntity) {
        ContextRoute contextRoute;
        if (routeEntity.getFirstPartner().getPartnerId().equals(sourcePartner)) {
            contextRoute = createContextRoute(
                    routeEntity,
                    RouteEntity::getFirstPartner,
                    RouteEntity::getSecondPartner,
                    RouteEntity::getFirstPartnerConnection,
                    RouteEntity::getSecondPartnerConnection,
                    RouteEntity::getFirstNexusCertificateContainer,
                    RouteEntity::getSecondNexusCertificateContainer,
                    RouteEntity::getFirstInflowPipeline);
        } else if (routeEntity.getSecondPartner().getPartnerId().equals(sourcePartner)) {
            contextRoute = createContextRoute(
                    routeEntity,
                    RouteEntity::getSecondPartner,
                    RouteEntity::getFirstPartner,
                    RouteEntity::getSecondPartnerConnection,
                    RouteEntity::getFirstPartnerConnection,
                    RouteEntity::getSecondNexusCertificateContainer,
                    RouteEntity::getFirstNexusCertificateContainer,
                    RouteEntity::getSecondInflowPipeline);
        } else {
            throw new NexusRuntimeException(NexusError.builder()
                    .message("sourcePartner is neither first nor second partner in route")
                    .extension("routeId", routeEntity.getId())
                    .extension("sourcePartner", sourcePartner)
                    .extension("firstPartnerId", routeEntity.getFirstPartner().getPartnerId())
                    .extension("secondPartnerId", routeEntity.getSecondPartner().getPartnerId())
                    .build());
        }
        return contextRoute;
    }

    @Transactional(readOnly = true)
    public Optional<ContextRoute> getRoute(RouteKey key) {
        Optional<ContextRoute> result;
        if (StringUtils.isAnyBlank(key.getSourcePartnerId(), key.getTargetPartnerId())
                || key.getChoreographyId() == null
                || key.getActionId() == null) {
            result = Optional.empty();
        } else {
            result = routeRepository
                    .findForAction(
                            key.getSourcePartnerId(),
                            key.getTargetPartnerId(),
                            key.getChoreographyId(),
                            key.getActionId())
                    .or(() -> routeRepository.findForChoreography(
                            key.getSourcePartnerId(), key.getTargetPartnerId(), key.getChoreographyId()))
                    .map(routeEntity -> toRoute(key.getSourcePartnerId(), routeEntity));
        }
        return result;
    }

    @Override
    @Transactional(readOnly = true)
    public ContextRoute create(RouteKey key) {
        ContextRoute route;
        if (StringUtils.isNoneBlank(key.getSourcePartnerId(), key.getTargetPartnerId())
                || key.getChoreographyId() == null
                || key.getActionId() == null) {
            route = getRoute(key).orElse(null);
        } else {
            route = null;
        }
        return route;
    }

    @Transactional(readOnly = true)
    public ContextRoute create(Long routeId, String sourcePartnerId) {
        return routeRepository
                .findById(routeId)
                .map(routeEntity -> toRoute(sourcePartnerId, routeEntity))
                .orElse(null);
    }
}
