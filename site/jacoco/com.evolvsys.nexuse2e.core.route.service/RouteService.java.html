package com.evolvsys.nexuse2e.core.route.service;

import com.evolvsys.app.core.auth.service.CurrentUserService;
import com.evolvsys.app.core.service.BaseService;
import com.evolvsys.app.core.service.ListService;
import com.evolvsys.nexuse2e.core.model.entity.RouteEntity;
import com.evolvsys.nexuse2e.core.route.RouteFilter;
import com.evolvsys.nexuse2e.core.route.mapper.RouteMapper;
import com.evolvsys.nexuse2e.core.route.model.RouteDto;
import com.evolvsys.nexuse2e.core.route.repository.RouteRepository;
import java.util.Comparator;
import java.util.List;
import lombok.extern.slf4j.Slf4j;
import org.springframework.lang.Nullable;
import org.springframework.stereotype.Service;

/**
 * @author Florian Drees
 * @since 29.12.2023
 */
@Service
@Slf4j
public class RouteService extends BaseService<RouteDto, RouteEntity, RouteRepository> {
    private final ListService listService;

    public RouteService(
            RouteMapper mapper,
            RouteRepository repository,
            CurrentUserService currentUserService,
            ListService listService) {
        super(mapper, repository, currentUserService);
        this.listService = listService;
    }

    @Override
    protected RouteEntity createEntity() {
        return new RouteEntity();
    }

    public List<RouteDto> list(@Nullable RouteFilter routeFilter) {
        return listService.list(routeFilter, RouteDto.class, repository::findAll).stream()
                .sorted(Comparator.comparing(RouteEntity::getRouteId))
                .map(this::mapEntityToDto)
                .toList();
    }
}
