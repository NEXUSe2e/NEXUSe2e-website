package com.evolvsys.nexuse2e.impl.service.support.transformator;

import com.evolvsys.nexuse2e.core.exception.NexusError;
import com.evolvsys.nexuse2e.core.exception.NexusException;
import com.evolvsys.nexuse2e.internal.ContextPayload;
import com.fasterxml.jackson.databind.JsonNode;
import com.fasterxml.jackson.databind.ObjectMapper;
import java.io.IOException;
import java.math.BigDecimal;
import java.util.Iterator;
import java.util.List;
import java.util.Map;
import java.util.Optional;
import java.util.function.Function;
import lombok.RequiredArgsConstructor;
import one.util.streamex.StreamEx;
import org.apache.commons.collections4.KeyValue;
import org.jspecify.annotations.NonNull;

/**
 * Reader for the {@link JsonToCsvTransformerService}. Reads the context payload's json and procudes
 * multiple {@link Map}, consisting of the key (json field name) and the optional value as a generic {@link Object}.
 *
 * @author Julian Drees
 * @since 09.09.2025
 */
@RequiredArgsConstructor
public class JsonToCsvReader {
    private final ObjectMapper objectMapper;
    private final List<KeyValue<String, String>> fieldColumnMappings;

    @NonNull
    public Iterator<Map<String, Optional<Object>>> iterator(ContextPayload originalPayload) throws NexusException {
        try {
            JsonNode root = objectMapper.readTree(originalPayload.getContent().getInputStream());
            if (!root.isArray()) {
                throw new NexusException(NexusError.builder()
                        .message("Payload '%s' does not start with a root array."
                                .formatted(originalPayload.getContentId()))
                        .build());
            }

            return StreamEx.of(root.elements())
                    .map(jsonNode -> StreamEx.of(fieldColumnMappings)
                            .map(KeyValue::getKey)
                            .toMap(Function.identity(), key -> getTypedValue(jsonNode.get(key))))
                    .iterator();
        } catch (IOException ex) {
            throw new NexusException(
                    NexusError.builder()
                            .message("Failed to parse json content for payload '%s'."
                                    .formatted(originalPayload.getContentId()))
                            .build(),
                    ex);
        }
    }

    @NonNull
    private Optional<Object> getTypedValue(JsonNode jsonNode) {
        if (jsonNode == null) {
            return Optional.empty();
        }
        Object result =
                switch (jsonNode.getNodeType()) {
                    case STRING -> jsonNode.asText();
                    case NUMBER -> {
                        if (jsonNode.isDouble()) {
                            yield BigDecimal.valueOf(jsonNode.asDouble());
                        }
                        yield jsonNode.asLong();
                    }
                    case BOOLEAN -> jsonNode.asBoolean();
                    case NULL -> null;
                    default ->
                        throw new UnsupportedOperationException("Unsupported node type: " + jsonNode.getNodeType());
                };
        return Optional.ofNullable(result);
    }
}
