package com.evolvsys.nexuse2e.impl.service.support.transformator;

import com.fasterxml.jackson.core.JsonGenerator;
import java.io.IOException;
import java.text.NumberFormat;
import java.text.ParseException;
import java.util.Locale;
import lombok.extern.slf4j.Slf4j;
import org.apache.commons.lang3.StringUtils;

@Slf4j
public enum JsonFieldRenderType {
    NUMBER {
        @Override
        public void render(JsonGenerator currentJsonGenerator, String key, String value) {
            try {
                if (StringUtils.isBlank(value)) {
                    currentJsonGenerator.writeNullField(key);
                } else {
                    parseAndWriteNumber(currentJsonGenerator, key, value, Locale.US);
                }

            } catch (IOException e) {
                LOGGER.error("failed to render number field", e);
            }
        }
    },
    NUMBER_DE {
        @Override
        public void render(JsonGenerator currentJsonGenerator, String key, String value) {
            try {
                if (StringUtils.isBlank(value)) {
                    currentJsonGenerator.writeNullField(key);
                } else {
                    parseAndWriteNumber(currentJsonGenerator, key, value, Locale.GERMANY);
                }
            } catch (IOException e) {
                LOGGER.error("failed to render number field", e);
            }
        }
    },
    STRING {
        @Override
        public void render(JsonGenerator currentJsonGenerator, String key, String value) {
            try {
                currentJsonGenerator.writeStringField(key, value);
            } catch (IOException e) {
                LOGGER.error("failed to render string field", e);
            }
        }
    },
    BOOLEAN {
        @Override
        public void render(JsonGenerator currentJsonGenerator, String key, String value) {
            try {
                currentJsonGenerator.writeBooleanField(key, Boolean.parseBoolean(value));
            } catch (IOException e) {
                LOGGER.error("failed to render boolean field", e);
            }
        }
    };

    private static void parseAndWriteNumber(
            JsonGenerator currentJsonGenerator, String key, String value, Locale locale) {
        try {
            Number number = NumberFormat.getInstance(locale).parse(value);
            if (number instanceof Long) {
                long result = number.longValue();
                currentJsonGenerator.writeNumberField(key, result);
            } else {
                double result = number.doubleValue();
                currentJsonGenerator.writeNumberField(key, result);
            }
        } catch (ParseException | IOException e) {
            LOGGER.error("failed to render number field, not a valid number: >{}<", value, e);
        }
    }

    public abstract void render(JsonGenerator currentJsonGenerator, String key, String value);
}
