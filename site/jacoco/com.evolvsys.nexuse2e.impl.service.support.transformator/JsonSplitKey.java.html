package com.evolvsys.nexuse2e.impl.service.support.transformator;

import java.util.LinkedHashMap;
import java.util.List;
import java.util.Map;
import lombok.EqualsAndHashCode;
import org.apache.commons.collections4.CollectionUtils;
import org.apache.commons.collections4.MapUtils;

@EqualsAndHashCode
public class JsonSplitKey {
    final Map<String, String> keys = new LinkedHashMap<>();

    public JsonSplitKey(List<String> keySegmentNames, Map<String, String> line) {
        if (CollectionUtils.isNotEmpty(keySegmentNames) && MapUtils.isNotEmpty(line)) {

            for (String keySegmentName : keySegmentNames) {
                String keySegmentValue = line.get(keySegmentName);
                if (keySegmentValue != null) {
                    keys.put(keySegmentName, keySegmentValue);
                }
            }
        }
    }
}
