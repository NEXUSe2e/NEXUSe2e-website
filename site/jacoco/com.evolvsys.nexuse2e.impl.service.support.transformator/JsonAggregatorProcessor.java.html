package com.evolvsys.nexuse2e.impl.service.support.transformator;

import static com.evolvsys.nexuse2e.impl.service.support.transformator.JsonAggregatorTransformatorService.*;

import com.evolvsys.nexuse2e.core.component.Configurable;
import com.evolvsys.nexuse2e.core.exception.NexusError;
import com.evolvsys.nexuse2e.core.exception.NexusException;
import com.evolvsys.nexuse2e.core.message.MetaData;
import com.evolvsys.nexuse2e.core.message.payload.content.PayloadContent;
import com.evolvsys.nexuse2e.core.message.payload.content.PayloadContentOutputStream;
import com.evolvsys.nexuse2e.core.parameter.Parameters;
import com.evolvsys.nexuse2e.internal.ContextPayload;
import com.fasterxml.jackson.core.*;
import java.io.IOException;
import java.io.InputStream;
import java.nio.charset.StandardCharsets;
import java.util.List;
import java.util.Map;
import javax.annotation.concurrent.NotThreadSafe;
import lombok.AccessLevel;
import lombok.NoArgsConstructor;
import lombok.ToString;
import lombok.extern.slf4j.Slf4j;
import one.util.streamex.StreamEx;
import org.apache.commons.collections4.KeyValue;
import org.apache.commons.lang3.ObjectUtils;
import org.apache.commons.lang3.StringUtils;
import org.apache.commons.lang3.Validate;

/**
 * Processor for {@link JsonAggregatorTransformatorService} for aggregating JSON payloads into one JSON payloads. It can also add a header to the
 * generated JSON.
 *
 * @author SvenBarden
 * @see JsonAggregatorTransformatorService
 * @since 09.04.2025
 */
@Slf4j
@NotThreadSafe
@ToString
@NoArgsConstructor(access = AccessLevel.PRIVATE)
class JsonAggregatorProcessor {
    private static final JsonFactory JSON_FACTORY = new JsonFactory();
    private boolean useContainer;
    private boolean addHeader;
    private String payloadContainerName;
    private boolean useInnerContainer;
    private String innerContainerName;
    private List<KeyValue<String, String>> headerAttributes;
    private Map<String, String> attributesToMetaDataMap;
    private boolean copyPayloadMetaData;

    @ToString.Exclude
    private JsonGenerator generator;

    // processing variables
    private MetaData metaData;
    private String headerName;

    public static JsonAggregatorProcessor create(JsonAggregatorTransformatorService jsonAggregatorTransformatorService)
            throws NexusException {
        Validate.notNull(jsonAggregatorTransformatorService, "jsonAggregatorTransformatorService must not be null");
        JsonAggregatorProcessor processor = new JsonAggregatorProcessor();
        processor.init(jsonAggregatorTransformatorService);
        processor.validate();
        return processor;
    }

    private void reset() {
        generator = null;
        metaData = null;
    }

    private void init(Configurable jsonAggregatorTransformatorService) {
        Parameters parameters = jsonAggregatorTransformatorService.getParameters();
        addHeader = parameters.value(PARAM_ADD_HEADER);
        useContainer = parameters.value(PARAM_USE_CONTAINER);
        payloadContainerName = parameters.value(PARAM_PAYLOAD_CONTAINER_NAME);
        useInnerContainer = parameters.value(PARAM_USE_INNER_CONTAINER);
        innerContainerName = parameters.value(PARAM_INNER_CONTAINER_NAME);
        copyPayloadMetaData = parameters.value(PARAM_COPY_PAYLOAD_META_DATA);
        headerName = parameters.value(PARAM_HEADER_NAME);

        headerAttributes = ObjectUtils.defaultIfNull(parameters.value(PARAM_HEADER_ATTRIBUTES), List.of());
        List<KeyValue<String, String>> attributesToMetaData =
                ObjectUtils.defaultIfNull(parameters.value(PARAM_ATTRIBUTES_TO_META_DATA), List.of());
        attributesToMetaDataMap = StreamEx.of(attributesToMetaData).toMap(KeyValue::getKey, KeyValue::getValue);
    }

    private void validate() throws NexusException {
        if (addHeader && !useContainer) {
            throw new NexusException(NexusError.builder()
                    .message("addHeader requires useContainer to be activated")
                    .build());
        }

        if (useContainer && StringUtils.isBlank(payloadContainerName)) {
            throw new NexusException(NexusError.builder()
                    .message("Payload container name must not be blank if useContainer is enabled")
                    .build());
        }

        if (useInnerContainer && StringUtils.isBlank(innerContainerName)) {
            throw new NexusException(NexusError.builder()
                    .message("Inner container name must not be blank if useInnerContainer is enabled")
                    .build());
        }
    }

    private void writeHeader() throws IOException {
        LOGGER.trace("adding header with name {}", headerName);

        generator.writeFieldName(headerName);
        generator.writeStartObject(); // open header root
        /*
         {
         THIS PROPERTY NAME --> "headerName": { <-- THIS TOKEN
               ... header content ...
           },
           ... payloads ...
         }
        */
        for (KeyValue<String, String> headerAttribute : headerAttributes) {
            LOGGER.trace("adding header attribute {}: {}", headerAttribute.getKey(), headerAttribute.getValue());

            generator.writeStringField(headerAttribute.getKey(), headerAttribute.getValue());
            /*
             {
               "headerName": {
                   ... header content ... <-- THIS CONTENT
               },
               ... payloads ...
             }
            */
        }
        generator.writeEndObject(); // close header root
        /*
         {
           "headerName": {
               ... header content ...
           } <-- THIS TOKEN
           ,
           ... payloads ...
         }
        */
    }

    private void writePayloadContent(ContextPayload contextPayload) throws IOException {
        try (InputStream contentStream = contextPayload.getContent().getInputStream();
                JsonParser parser = JSON_FACTORY.createParser(contentStream)) {

            String lastFieldName = null;
            while (parser.nextToken() != null) {
                LOGGER.trace("processing token {}", parser.getCurrentToken());

                // to copy the metadata from the payload we need to check if the current token is a field name and
                // remember it
                if (parser.currentToken() == JsonToken.FIELD_NAME) {
                    lastFieldName = parser.getText();
                } else if (lastFieldName != null) {
                    String metaDataKey = attributesToMetaDataMap.get(lastFieldName);
                    if (metaDataKey != null) {
                        switch (parser.currentToken()) {
                            case VALUE_NUMBER_INT -> metaData.add(metaDataKey, parser.getValueAsInt());
                            case VALUE_STRING -> metaData.add(metaDataKey, parser.getValueAsString());
                            case VALUE_NUMBER_FLOAT -> metaData.add(metaDataKey, parser.getDecimalValue());
                            case VALUE_TRUE -> metaData.add(metaDataKey, Boolean.TRUE);
                            case VALUE_FALSE -> metaData.add(metaDataKey, Boolean.FALSE);
                            default -> {} // other token types are not copied to metadata
                        }
                    }
                }

                writeToken(parser);

                /*
                EITHER

                {
                    "optionalHeader": {
                      ... optional header content ...
                    },
                    "payloadContainerName": [
                      {
                        "innerContainerName": {
                          ... payload content ... <-- THIS CONTENT
                        }
                      }
                    ]
                }

                OR

                [
                  {
                    "innerContainerName": {
                      ... payload content ... <-- THIS CONTENT
                    }
                  }
                 ]

                 OR

                 [
                  {
                    ... payload content ... <-- THIS CONTENT
                  }
                 ]
                 */

                LOGGER.trace("processed token {}", parser.getText());
            }
        }
    }

    private void writePayload(ContextPayload contextPayload) throws IOException {
        if (LOGGER.isTraceEnabled()) {
            LOGGER.trace("processing payload {}", contextPayload.getContent().getString(StandardCharsets.UTF_8));
        }
        if (contextPayload.getContentId() != null) {
            metaData.add("originalContentId", contextPayload.getContentId());
        }

        if (copyPayloadMetaData) {
            metaData.add(contextPayload.getMetaData());
        }

        if (useInnerContainer) {
            generator.writeStartObject(); // open payload root
            generator.writeFieldName(innerContainerName);
            /*
               EITHER

                {
                   "optionalHeader": {
                     ... optional header content ...
                   },
                   "payloadContainerName": [
                     { <-- THIS TOKEN
                       THIS PROPERTY NAME --> "innerContainerName": {
                         ... payload content ...
                       }
                     }
                   ]
                }

                OR

               [
                 { <-- THIS TOKEN
                   THIS PROPERTY NAME --> "innerContainerName": {
                     ... payload content ...
                   }
                 }
               ]
            */
        }

        writePayloadContent(contextPayload);

        if (useInnerContainer) {
            generator.writeEndObject(); // close payload root
            /*
               EITHER

                {
                   "optionalHeader": {
                     ... optional header content ...
                   },
                   "payloadContainerName": [
                     {
                       "innerContainerName": {
                         ... payload content ...
                       }
                     } <-- THIS TOKEN
                   ]
                }

                OR

               [
                 {
                   "innerContainerName": {
                     ... payload content ...
                   }
                 } <-- THIS TOKEN
               ]
            */
        }
    }

    private void writeToken(JsonParser parser) throws IOException {
        // small hack to force exhaustive switch to get compile errors if new token types are added
        // jackson has no method to directly write the parsed token into the generator, so we need to catch each
        // individual token type manually
        IoConsumer<JsonGenerator> consumer =
                switch (parser.currentToken()) {
                    case NOT_AVAILABLE -> jsonGenerator -> {};
                    case START_OBJECT -> JsonGenerator::writeStartObject;
                    case END_OBJECT -> JsonGenerator::writeEndObject;
                    case START_ARRAY -> JsonGenerator::writeStartArray;
                    case END_ARRAY -> JsonGenerator::writeEndArray;
                    case FIELD_NAME -> jsonGenerator -> jsonGenerator.writeFieldName(parser.currentName());
                    case VALUE_EMBEDDED_OBJECT ->
                        jsonGenerator -> jsonGenerator.writeEmbeddedObject(parser.getEmbeddedObject());
                    case VALUE_STRING -> jsonGenerator -> jsonGenerator.writeString(parser.getValueAsString());
                    case VALUE_NUMBER_INT -> jsonGenerator -> jsonGenerator.writeNumber(parser.getValueAsInt());
                    case VALUE_NUMBER_FLOAT -> jsonGenerator -> jsonGenerator.writeNumber(parser.getDecimalValue());
                    case VALUE_TRUE -> jsonGenerator -> jsonGenerator.writeBoolean(true);
                    case VALUE_FALSE -> jsonGenerator -> jsonGenerator.writeBoolean(false);
                    case VALUE_NULL -> JsonGenerator::writeNull;
                };
        consumer.accept(generator);
    }

    public ContextPayload process(List<ContextPayload> contextPayloads) throws NexusException {
        LOGGER.trace("processing {} payloads", contextPayloads.size());
        ContextPayload result;
        try (PayloadContentOutputStream outputStream = PayloadContent.createOutputStream();
                JsonGenerator generator = JSON_FACTORY.createGenerator(outputStream, JsonEncoding.UTF8)) {
            // small hack to use the autoclose feature for the class instance variable
            this.generator = generator;

            metaData = new MetaData();

            // open root container or array
            if (useContainer) {
                generator.writeStartObject();
                /*
                { <-- THIS TOKEN
                  ... following content ...
                }
                */
            } else {
                generator.writeStartArray();
                /*
                [ <-- THIS TOKEN
                  ... following content ...
                ]
                 */
            }

            if (addHeader) {
                writeHeader();
            }

            if (useContainer) {
                generator.writeFieldName(payloadContainerName); // set payload container name
                generator.writeStartArray(); // open payloads array

                /*
                {
                  "optionalHeader": {
                    ... optional header content ...
                  },

                  THIS PROPERTY NAME --> "payloadContainerName": [ <-- THIS TOKEN
                    ... following content ...
                  ]
                 }
                 */
            }

            for (ContextPayload contextPayload : contextPayloads) {
                writePayload(contextPayload);
            }

            generator.writeEndArray(); // close payloads array
            /*
            EITHER

                [
                  ... content ...
                ] <-- THIS TOKEN

            OR

                {
                  "optionalHeader": {
                    ... optional header content ...
                  },

                  "payloadContainerName": [
                    ... content ...
                  ] <-- THIS TOKEN
                }
             */

            // close root container
            if (useContainer) {
                generator.writeEndObject();
                /*
                {
                  "optionalHeader": {
                    ... optional header content ...
                  },

                  "payloadContainerName": [
                    ... content ...
                  ]
                }  <-- THIS TOKEN
                 */
            }

            // finish json generation and closing the output stream
            generator.close();
            PayloadContent payloadContent = outputStream.finish();

            result = ContextPayload.builder()
                    .content(payloadContent)
                    .size(payloadContent.getSize())
                    .contentType("application/json")
                    .encoding("UTF-8")
                    .metaData(metaData)
                    .build();
        } catch (IOException e) {
            throw new NexusException(
                    NexusError.builder()
                            .message("Error while aggregating json content")
                            .build(),
                    e);
        } finally {
            reset();
        }

        return result;
    }

    @FunctionalInterface
    private interface IoConsumer<T> {
        void accept(T t) throws IOException;
    }
}
