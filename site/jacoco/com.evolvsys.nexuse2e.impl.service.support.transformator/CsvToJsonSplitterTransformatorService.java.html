package com.evolvsys.nexuse2e.impl.service.support.transformator;

import com.evolvsys.nexuse2e.core.exception.NexusException;
import com.evolvsys.nexuse2e.core.parameter.ParameterDefinition;
import com.evolvsys.nexuse2e.core.parameter.ParameterType;
import com.evolvsys.nexuse2e.internal.ContextPayload;
import com.fasterxml.jackson.core.JsonFactory;
import com.fasterxml.jackson.dataformat.csv.CsvMapper;
import com.fasterxml.jackson.dataformat.csv.CsvSchema;
import java.util.ArrayList;
import java.util.Collection;
import java.util.LinkedHashMap;
import java.util.List;
import java.util.Map;
import lombok.extern.slf4j.Slf4j;
import org.apache.commons.collections4.CollectionUtils;
import org.apache.commons.collections4.KeyValue;

/**
 *
 */
@Slf4j
public class CsvToJsonSplitterTransformatorService extends TransformatorSupportService {

    private static final String PARAM_TRANSFORM_BEFORE = "transform_before";
    private static final String PARAM_TRANSFORM_AFTER = "transform_after";
    private static final String PARAM_COLUMN_MAPPING = "column_mapping";
    private static final String PARAM_KEY_COLUMNS = "key_columns";
    private static final String PARAM_HAS_HEADER = "has_header";
    private static final String PARAM_BLACK_LIST = "black_list";
    private static final String PARAM_WHITE_LIST = "white_list";
    private static final String PARAM_SEPARATOR_CHAR = "separator_char";
    private static final String PARAM_COPY_METADATA = "copy_metadata";
    private static final String PARAM_GENERATE_TOC = "generate_toc";

    private final List<String> keys = new ArrayList<>();
    private final List<String> blackListFilter = new ArrayList<>();
    private final List<String> whiteListFilter = new ArrayList<>();
    private final Map<String, JsonFieldRenderType> columnTypeMapping = new LinkedHashMap<>();

    @Override
    public List<ParameterDefinition> parameterDefinitions() {
        return List.of(
                ParameterDefinition.builder()
                        .name(PARAM_TRANSFORM_BEFORE)
                        .required(false)
                        .displayName("Pre Transformer")
                        .description("An optional Transformer Service which is executed before this transformer.")
                        .type(ParameterType.SUPPORT_SERVICE)
                        .build(),
                ParameterDefinition.builder()
                        .name(PARAM_TRANSFORM_AFTER)
                        .required(false)
                        .displayName("Post Transformer")
                        .description("An optional Transformer Service which is executed after this transformer.")
                        .type(ParameterType.SUPPORT_SERVICE)
                        .build(),
                ParameterDefinition.builder()
                        .name(PARAM_SEPARATOR_CHAR)
                        .displayName("Separator Character")
                        .description("The character used for column separation. The default is ';'.")
                        .type(ParameterType.TRIMMED_STRING)
                        .required(true)
                        .defaultValue(";")
                        .build(),
                ParameterDefinition.builder()
                        .name(PARAM_COLUMN_MAPPING)
                        .required(true)
                        .displayName("Column Mapping")
                        .description(
                                "Defines the columns available in the json and their type. (currently allowed is STRING, NUMBER and BOOLEAN)")
                        .type(ParameterType.KEY_VALUE_PAIR_LIST_TRIMMED)
                        .build(),
                ParameterDefinition.builder()
                        .name(PARAM_KEY_COLUMNS)
                        .required(false)
                        .displayName("Key Columns")
                        .description(
                                "Keys are used to group the lines by this field. For every group a new json document is created.")
                        .type(ParameterType.STRING_LIST)
                        .build(),
                ParameterDefinition.builder()
                        .name(PARAM_HAS_HEADER)
                        .required(true)
                        .defaultValue(Boolean.TRUE)
                        .displayName("CSV header row")
                        .description(
                                "This flag is used to skip the first row in case you provided a file with a header row in it.")
                        .type(ParameterType.BOOLEAN)
                        .build(),
                ParameterDefinition.builder()
                        .name(PARAM_BLACK_LIST)
                        .required(false)
                        .displayName("Column Black List")
                        .description(
                                "All values added in this list are excluded from the final JSON structure. The values must match the header mappings.")
                        .type(ParameterType.STRING_LIST)
                        .build(),
                ParameterDefinition.builder()
                        .name(PARAM_WHITE_LIST)
                        .required(false)
                        .displayName("Column White List")
                        .description(
                                "Only values in this list are included in the final JSON Structure. The values must match the header mappings.")
                        .type(ParameterType.STRING_LIST)
                        .build(),
                ParameterDefinition.builder()
                        .name(PARAM_COPY_METADATA)
                        .required(true)
                        .displayName("Copy Metadata")
                        .description("Copy all Metadata from original payload to all generated payloads.")
                        .type(ParameterType.BOOLEAN)
                        .defaultValue(Boolean.TRUE)
                        .build(),
                ParameterDefinition.builder()
                        .name(PARAM_GENERATE_TOC)
                        .required(true)
                        .displayName("Create Table of Content")
                        .description(
                                "Optionally create an additional payload containing a table of content with all keys found.")
                        .type(ParameterType.BOOLEAN)
                        .defaultValue(Boolean.FALSE)
                        .build());
    }

    @Override
    public List<ContextPayload> transform(Collection<ContextPayload> payloads) throws NexusException {

        if (payloads == null) {
            return List.of();
        }

        TransformatorSupportService preTransformer = getParameters().value(PARAM_TRANSFORM_BEFORE);
        TransformatorSupportService postTransformer = getParameters().value(PARAM_TRANSFORM_AFTER);
        List<KeyValue<String, String>> rawColumnTypeMapping = getParameters().value(PARAM_COLUMN_MAPPING);
        List<String> rawKeys = getParameters().value(PARAM_KEY_COLUMNS);
        boolean hasHeader = getParameters().value(PARAM_HAS_HEADER);
        List<String> blackList = getParameters().value(PARAM_BLACK_LIST);
        List<String> whiteList = getParameters().value(PARAM_WHITE_LIST);
        String separatorChar = getParameters().value(PARAM_SEPARATOR_CHAR);
        boolean copyPayloads = getParameters().value(PARAM_COPY_METADATA);
        boolean generateToC = getParameters().value(PARAM_GENERATE_TOC);

        if (preTransformer != null) {
            payloads = preTransformer.transform(payloads);
        }

        if (CollectionUtils.isNotEmpty(rawColumnTypeMapping)) {
            columnTypeMapping.clear();
            for (KeyValue<String, String> entry : rawColumnTypeMapping) {
                columnTypeMapping.put(
                        entry.getKey(),
                        JsonFieldRenderType.valueOf(entry.getValue().toUpperCase()));
            }
        }

        if (rawKeys != null) {
            keys.clear();
            keys.addAll(rawKeys);
        }

        if (blackList != null) {
            blackListFilter.clear();
            blackListFilter.addAll(blackList);
        }
        if (whiteList != null) {
            whiteListFilter.clear();
            whiteListFilter.addAll(whiteList);
        }

        List<ContextPayload> result = new ArrayList<>();

        CsvSchema.Builder csvSchemaBuilder = CsvSchema.builder()
                .setColumnSeparator(separatorChar.charAt(0))
                .setUseHeader(false)
                .setSkipFirstDataRow(hasHeader);

        columnTypeMapping.keySet().forEach(csvSchemaBuilder::addColumn);

        CsvSchema csvSchema = csvSchemaBuilder.build();

        CsvMapper csvMapper = new CsvMapper();
        JsonFactory jsonFactory = new JsonFactory();

        for (ContextPayload payload : payloads) {
            CsvPayloadProcessor csvPayloadProcessor =
                    getCsvPayloadProcessor(payload, csvMapper, csvSchema, jsonFactory, generateToC);

            List<ContextPayload> transformedPayloads = csvPayloadProcessor.processPayload();

            if (copyPayloads) {
                for (ContextPayload transformedPayload : transformedPayloads) {
                    transformedPayload.getMetaData().add(payload.getMetaData());
                }
            }
            result.addAll(transformedPayloads);
        }

        if (postTransformer != null) {
            result = postTransformer.transform(result);
        }
        return result;
    }

    protected CsvPayloadProcessor getCsvPayloadProcessor(
            ContextPayload payload,
            CsvMapper csvMapper,
            CsvSchema csvSchema,
            JsonFactory jsonFactory,
            boolean generateToC) {
        CsvProcessorConfig csvProcessorConfig = CsvProcessorConfig.builder()
                .payload(payload)
                .csvMapper(csvMapper)
                .csvSchema(csvSchema)
                .jsonFactory(jsonFactory)
                .keys(keys)
                .columnTypeMapping(columnTypeMapping)
                .blackListFilter(blackListFilter)
                .whiteListFilter(whiteListFilter)
                .generateTableOfContents(generateToC)
                .build();
        return new CsvPayloadProcessor(csvProcessorConfig);
    }
}
