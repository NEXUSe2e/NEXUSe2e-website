package com.evolvsys.nexuse2e.impl.service.support.transformator;

import com.evolvsys.nexuse2e.core.exception.NexusError;
import com.evolvsys.nexuse2e.core.exception.NexusException;
import com.evolvsys.nexuse2e.core.message.MetaData;
import com.evolvsys.nexuse2e.core.message.payload.content.PayloadContent;
import com.evolvsys.nexuse2e.core.message.payload.content.PayloadContentOutputStream;
import com.evolvsys.nexuse2e.core.parameter.ParameterDefinition;
import com.evolvsys.nexuse2e.core.parameter.ParameterType;
import com.evolvsys.nexuse2e.internal.ContextPayload;
import java.io.IOException;
import java.util.ArrayList;
import java.util.Collection;
import java.util.List;
import lombok.extern.slf4j.Slf4j;
import net.lingala.zip4j.io.inputstream.ZipInputStream;
import net.lingala.zip4j.model.LocalFileHeader;
import org.apache.commons.lang3.ObjectUtils;

/**
 *
 */
@Slf4j
public class UnzipTransformatorService extends TransformatorSupportService {

    private static final String PARAM_PASSWORD = "password";

    private static final String PARAM_TRANSFORM_AFTER = "transform_after";
    private static final String PARAM_MULTIPLE_FILES = "multiple_files";
    private static final String PARAM_ENCODING_HINT = "encoding_hint";

    @Override
    public List<ParameterDefinition> parameterDefinitions() {
        return List.of(
                ParameterDefinition.builder()
                        .name(PARAM_PASSWORD)
                        .required(false)
                        .displayName("Encryption Password")
                        .description("An optional password required for encrypted zip files.")
                        .type(ParameterType.PASSWORD)
                        .build(),
                ParameterDefinition.builder()
                        .name(PARAM_TRANSFORM_AFTER)
                        .required(false)
                        .displayName("Post Transformer")
                        .description("An optional Transformer Service which is executed after this transformer.")
                        .type(ParameterType.SUPPORT_SERVICE)
                        .build(),
                ParameterDefinition.builder()
                        .name(PARAM_MULTIPLE_FILES)
                        .required(true)
                        .defaultValue(Boolean.FALSE)
                        .displayName("Allow multiple files")
                        .description(
                                "By default, only one file in the zip file is allowed, this switch will allow multiple files within the zip file.")
                        .type(ParameterType.BOOLEAN)
                        .build(),
                ParameterDefinition.builder()
                        .name(PARAM_ENCODING_HINT)
                        .required(false)
                        .displayName("Content Encoding Hint")
                        .description("Hint for the expected Content Encoding inside the zip file.")
                        .type(ParameterType.STRING)
                        .build());
    }

    @Override
    public List<ContextPayload> transform(Collection<ContextPayload> payloads) throws NexusException {

        List<ContextPayload> result = new ArrayList<>();
        if (payloads != null) {

            for (ContextPayload payload : payloads) {
                processPayload(payload, result);
            }
        }
        return result;
    }

    private void processPayload(ContextPayload payload, List<ContextPayload> result) throws NexusException {
        try (ZipInputStream zipInputStream =
                new ZipInputStream(payload.getContent().getInputStream())) {
            String password = parameters.value(PARAM_PASSWORD);

            if (password != null) {
                zipInputStream.setPassword(password.toCharArray());
            }
            LocalFileHeader localFileHeader = zipInputStream.getNextEntry();
            boolean first = true;
            while (localFileHeader != null) {
                if (first) {
                    first = false;
                } else {
                    Boolean multipleFiles = parameters.value(PARAM_MULTIPLE_FILES);
                    if (!multipleFiles) {
                        throw new NexusException(NexusError.builder()
                                .message("only one file is expected in zip file.")
                                .build());
                    }
                }

                try (PayloadContentOutputStream payloadContentOutputStream = PayloadContent.createOutputStream()) {
                    long count = zipInputStream.transferTo(payloadContentOutputStream);
                    MetaData metaData = new MetaData();
                    metaData.add("zip-entry-name", localFileHeader.getFileName());
                    metaData.add("copied-bytes", count);
                    final String encoding;

                    if (parameters.value(PARAM_ENCODING_HINT) != null) {
                        encoding = parameters.value(PARAM_ENCODING_HINT);
                    } else {
                        encoding = ObjectUtils.defaultIfNull(payload.getEncoding(), "UTF-8");
                    }
                    PayloadContent payloadContent = payloadContentOutputStream.finish();

                    ContextPayload contextPayload = ContextPayload.builder()
                            .metaData(metaData)
                            .encoding(encoding)
                            .content(payloadContent)
                            .size(payloadContent.getSize())
                            .build();

                    if (parameters.value(PARAM_TRANSFORM_AFTER) != null) {
                        TransformatorSupportService transformatorSupportService =
                                parameters.value(PARAM_TRANSFORM_AFTER);
                        result.addAll(transformatorSupportService.transform(List.of(contextPayload)));
                    } else {
                        result.add(contextPayload);
                    }
                }

                localFileHeader = zipInputStream.getNextEntry();
            }
        } catch (IOException e) {
            throw new NexusException(
                    NexusError.builder().message("failed to unzip content.").build(), e);
        }
    }
}
