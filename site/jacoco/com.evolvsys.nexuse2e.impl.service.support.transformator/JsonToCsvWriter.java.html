package com.evolvsys.nexuse2e.impl.service.support.transformator;

import com.evolvsys.nexuse2e.core.message.payload.content.PayloadContent;
import com.evolvsys.nexuse2e.core.message.payload.content.PayloadContentOutputStream;
import com.evolvsys.nexuse2e.core.utils.StreamUtils;
import com.fasterxml.jackson.databind.SequenceWriter;
import com.fasterxml.jackson.dataformat.csv.CsvMapper;
import com.fasterxml.jackson.dataformat.csv.CsvSchema;
import java.io.IOException;
import java.io.OutputStreamWriter;
import java.nio.charset.Charset;
import java.util.Collection;
import java.util.Iterator;
import java.util.List;
import java.util.Map;
import java.util.Optional;
import java.util.stream.Collectors;
import lombok.extern.slf4j.Slf4j;
import one.util.streamex.EntryStream;
import org.apache.commons.collections4.KeyValue;
import org.jspecify.annotations.NonNull;

/**
 * Csv writer for the {@link JsonToCsvTransformerService}.
 *
 * @author Julian Drees
 * @since 09.09.2025
 */
@Slf4j
public class JsonToCsvWriter {
    private final List<String> headers;
    private final Map<String, String> fieldColumnMapping;
    private final boolean includeHeaders;
    private final Charset charset;
    private final char columnSeparator;

    public JsonToCsvWriter(
            Collection<KeyValue<String, String>> fieldColumnMappings,
            boolean includeHeaders,
            Charset charset,
            char columnSeparator) {
        super();
        this.headers = fieldColumnMappings.stream().map(KeyValue::getValue).toList();
        this.fieldColumnMapping =
                fieldColumnMappings.stream().collect(Collectors.toMap(KeyValue::getKey, KeyValue::getValue));
        this.includeHeaders = includeHeaders;
        this.charset = charset;
        this.columnSeparator = columnSeparator;
    }

    @NonNull
    public PayloadContent write(Iterator<Map<String, Optional<Object>>> iterator) throws IOException {
        CsvMapper csvMapper = new CsvMapper();
        CsvSchema csvSchema = createSchema();

        try (PayloadContentOutputStream outputStream = PayloadContent.createOutputStream();
                OutputStreamWriter outputStreamWriter = new OutputStreamWriter(outputStream, charset);
                SequenceWriter sequenceWriter = csvMapper.writer(csvSchema).writeValues(outputStreamWriter)) {
            if (includeHeaders) {
                sequenceWriter.write(headers);
            }

            while (iterator.hasNext()) {
                Map<String, Object> row = EntryStream.of(iterator.next().entrySet().stream())
                        .collect(StreamUtils.toMapOfNullables(
                                entry -> fieldColumnMapping.get(entry.getKey()),
                                entry -> entry.getValue().orElse(null)));

                if (row.isEmpty()) {
                    LOGGER.warn("Tried to write empty csv row.");
                    continue;
                }

                LOGGER.trace("Writing csv row: {}", row);
                sequenceWriter.write(row);
            }
            return outputStream.finish();
        }
    }

    private CsvSchema createSchema() {
        CsvSchema.Builder csvSchemaBuilder = CsvSchema.builder().setColumnSeparator(columnSeparator);
        for (String column : headers) {
            csvSchemaBuilder.addColumn(column);
        }
        return csvSchemaBuilder.build();
    }
}
