package com.evolvsys.nexuse2e.impl.service.support.transformator;

import com.evolvsys.nexuse2e.core.exception.NexusError;
import com.evolvsys.nexuse2e.core.exception.NexusException;
import com.evolvsys.nexuse2e.internal.ContextPayload;
import com.fasterxml.jackson.core.FormatSchema;
import com.fasterxml.jackson.databind.MappingIterator;
import com.fasterxml.jackson.dataformat.csv.CsvMapper;
import java.io.IOException;
import java.io.InputStream;
import java.io.InputStreamReader;
import java.nio.charset.Charset;
import java.util.Iterator;
import java.util.Map;
import lombok.AccessLevel;
import lombok.RequiredArgsConstructor;
import lombok.extern.slf4j.Slf4j;
import org.apache.commons.lang3.ObjectUtils;

@RequiredArgsConstructor(access = AccessLevel.PRIVATE)
@Slf4j
class IteratorWrapper implements AutoCloseable, Iterator<Map<String, String>> {

    private final InputStream inputStream;
    private final InputStreamReader inputStreamReader;
    private final MappingIterator<Map<String, String>> lines;

    public static IteratorWrapper createInstance(ContextPayload payload, CsvMapper csvMapper, FormatSchema csvSchema)
            throws IOException {

        InputStream inputStream = payload.getContent().getInputStream();
        InputStreamReader inputStreamReader =
                new InputStreamReader(inputStream, Charset.forName(payload.getEncoding()));

        try {
            MappingIterator<Map<String, String>> lines =
                    csvMapper.readerForMapOf(String.class).with(csvSchema).readValues(inputStreamReader);

            return new IteratorWrapper(inputStream, inputStreamReader, lines);
        } catch (IOException e) {
            inputStreamReader.close();
            inputStream.close();
            throw e;
        }
    }

    @Override
    public void close() throws NexusException {
        Exception e1 = null;
        Exception e2 = null;
        Exception e3 = null;
        try {
            lines.close();
        } catch (IOException e) {
            e1 = new NexusException(
                    NexusError.builder()
                            .message("failed to close lines iterator")
                            .build(),
                    e);
        }
        try {
            inputStreamReader.close();
        } catch (IOException e) {
            e2 = new NexusException(
                    NexusError.builder()
                            .message("failed to close inputStreamReader")
                            .build(),
                    e);
        }
        try {
            inputStream.close();
        } catch (IOException e) {
            e3 = new NexusException(
                    NexusError.builder().message("failed to close inputStream").build(), e);
        }
        if (ObjectUtils.anyNotNull(e1, e2, e3)) {
            NexusException e = new NexusException(NexusError.builder()
                    .message("failed to close IteratorWrapper")
                    .build());
            if (e1 != null) {
                e.addSuppressed(e1);
            }
            if (e2 != null) {
                e.addSuppressed(e2);
            }
            if (e3 != null) {
                e.addSuppressed(e3);
            }
            throw e;
        }
    }

    @Override
    public boolean hasNext() {
        return lines.hasNext();
    }

    @Override
    public Map<String, String> next() {
        return lines.next();
    }
}
