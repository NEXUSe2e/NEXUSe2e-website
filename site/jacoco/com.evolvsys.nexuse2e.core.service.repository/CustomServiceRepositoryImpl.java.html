package com.evolvsys.nexuse2e.core.service.repository;

import com.evolvsys.nexuse2e.core.model.entity.QParameterEntity;
import com.evolvsys.nexuse2e.core.model.entity.QServiceEntity;
import com.evolvsys.nexuse2e.core.model.entity.ServiceEntity;
import com.evolvsys.nexuse2e.core.service.ServiceType;
import com.querydsl.jpa.JPQLQueryFactory;
import java.util.List;
import lombok.RequiredArgsConstructor;
import org.springframework.stereotype.Repository;

/**
 * @author SvenBarden
 * @since 21.02.2024
 */
@Repository
@RequiredArgsConstructor
class CustomServiceRepositoryImpl implements CustomServiceRepository {
    private final JPQLQueryFactory queryFactory;

    @Override
    public List<ServiceEntity> getByTypeAndParameterValue(
            ServiceType serviceType, String parameterName, String parameterValue) {
        QServiceEntity serviceEntity = QServiceEntity.serviceEntity;
        QParameterEntity parameterEntity = QParameterEntity.parameterEntity;

        return queryFactory
                .selectFrom(serviceEntity)
                .join(serviceEntity.parameters, parameterEntity)
                .on(parameterEntity.name.eq(parameterName), parameterEntity.stringValue.eq(parameterValue))
                .where(serviceEntity.enabled.isTrue())
                .fetch();
    }
}
