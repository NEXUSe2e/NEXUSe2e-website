package com.evolvsys.nexuse2e.core.service.repository;

import com.evolvsys.nexuse2e.core.model.entity.QServiceEntity;
import com.evolvsys.nexuse2e.core.model.entity.ServiceEntity;
import com.evolvsys.nexuse2e.core.service.ServiceFilter;
import com.querydsl.core.BooleanBuilder;
import com.querydsl.core.annotations.QueryDelegate;
import com.querydsl.core.types.Predicate;
import org.apache.commons.collections4.CollectionUtils;
import org.apache.commons.lang3.StringUtils;
import org.springframework.lang.Nullable;

/**
 * @author Florian Drees
 * @since 14.12.2023
 */
public final class ServiceQueryDelegate {
    private ServiceQueryDelegate() {}

    @QueryDelegate(ServiceEntity.class)
    public static Predicate filter(QServiceEntity serviceEntity, @Nullable ServiceFilter serviceFilter) {
        BooleanBuilder builder = new BooleanBuilder();
        if (serviceFilter != null) {
            if (CollectionUtils.isNotEmpty(serviceFilter.getTypes())) {
                builder.and(serviceEntity.serviceType.in(serviceFilter.getTypes()));
            }
            if (StringUtils.isNotBlank(serviceFilter.getServiceId())) {
                builder.and(serviceEntity.serviceId.contains(serviceFilter.getServiceId()));
            }
            if (StringUtils.isNotBlank(serviceFilter.getComponentId())) {
                builder.and(serviceEntity.componentId.contains(serviceFilter.getComponentId()));
            }
        }
        return builder.getValue();
    }
}
