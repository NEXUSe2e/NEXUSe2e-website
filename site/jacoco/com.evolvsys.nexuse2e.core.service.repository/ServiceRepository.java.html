package com.evolvsys.nexuse2e.core.service.repository;

import com.evolvsys.nexuse2e.core.model.entity.QServiceEntity;
import com.evolvsys.nexuse2e.core.model.entity.ServiceEntity;
import com.evolvsys.nexuse2e.core.service.ServiceFilter;
import com.querydsl.core.types.Predicate;
import com.querydsl.core.types.dsl.BooleanExpression;
import java.util.List;
import java.util.Optional;
import org.apache.commons.collections4.IterableUtils;
import org.springframework.data.jpa.repository.EntityGraph;
import org.springframework.data.jpa.repository.JpaRepository;
import org.springframework.data.querydsl.QuerydslPredicateExecutor;

/**
 * @author SvenBarden
 * @since 11.10.2023
 */
public interface ServiceRepository
        extends JpaRepository<ServiceEntity, Long>, QuerydslPredicateExecutor<ServiceEntity>, CustomServiceRepository {
    Optional<ServiceEntity> findByServiceId(String serviceId);

    @EntityGraph(attributePaths = {"parameters"})
    Optional<ServiceEntity> getEagerById(long id);

    boolean existsByServiceId(String serviceId);

    boolean existsByServiceIdAndIdNot(String serviceId, long id);

    default List<ServiceEntity> findAll(ServiceFilter serviceFilter, BooleanExpression securityPredicate) {
        Predicate predicate = QServiceEntity.serviceEntity.filter(serviceFilter);
        if (predicate != null) {
            securityPredicate = securityPredicate.and(predicate);
        }
        return IterableUtils.toList(findAll(securityPredicate));
    }

    List<ServiceEntity> findByEnabledTrue();

    boolean existsByServiceIdAndEnabledTrue(String serviceId);

    Optional<ServiceEntity> findByServiceIdAndEnabledTrue(String serviceId);
}
