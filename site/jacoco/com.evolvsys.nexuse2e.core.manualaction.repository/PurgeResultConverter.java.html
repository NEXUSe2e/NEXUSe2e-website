package com.evolvsys.nexuse2e.core.manualaction.repository;

import com.evolvsys.nexuse2e.core.exception.NexusError;
import com.evolvsys.nexuse2e.core.exception.NexusRuntimeException;
import com.evolvsys.nexuse2e.core.purge.model.PurgeResult;
import com.fasterxml.jackson.core.JsonProcessingException;
import com.fasterxml.jackson.databind.ObjectMapper;
import edu.umd.cs.findbugs.annotations.SuppressFBWarnings;
import jakarta.persistence.AttributeConverter;
import jakarta.persistence.Converter;
import lombok.RequiredArgsConstructor;
import org.springframework.stereotype.Component;

/**
 * @author SvenBarden
 * @since 12.08.2024
 */
@Component
@RequiredArgsConstructor
@Converter(autoApply = true)
@SuppressFBWarnings(
        value = "EXS_EXCEPTION_SOFTENING_NO_CONSTRAINTS",
        justification = "API requires no checked exceptions")
public class PurgeResultConverter implements AttributeConverter<PurgeResult, String> {
    private final ObjectMapper objectMapper;

    @Override
    public String convertToDatabaseColumn(PurgeResult attribute) {
        try {
            return attribute == null ? null : objectMapper.writeValueAsString(attribute);
        } catch (JsonProcessingException e) {
            throw new NexusRuntimeException(
                    NexusError.builder()
                            .message("Could not serialize purge result")
                            .build(),
                    e);
        }
    }

    @Override
    public PurgeResult convertToEntityAttribute(String dbData) {
        try {
            return dbData == null ? null : objectMapper.readValue(dbData, PurgeResult.class);
        } catch (JsonProcessingException e) {
            throw new NexusRuntimeException(
                    NexusError.builder()
                            .message("Could not deserialize purge result")
                            .build(),
                    e);
        }
    }
}
