package com.evolvsys.nexuse2e.core.manualaction.repository;

import com.evolvsys.nexuse2e.core.manualaction.model.ManualActionEntity;
import com.evolvsys.nexuse2e.core.manualaction.model.ManualActionFilter;
import com.evolvsys.nexuse2e.core.manualaction.model.QManualActionEntity;
import com.evolvsys.nexuse2e.core.manualaction.model.QMessageActionManualActionEntity;
import com.querydsl.core.BooleanBuilder;
import com.querydsl.core.annotations.QueryDelegate;
import com.querydsl.core.types.Predicate;
import org.apache.commons.collections4.CollectionUtils;
import org.springframework.lang.Nullable;

/**
 * @author SvenBarden
 * @since 05.08.2024
 */
public final class ManualActionQueryDelegate {
    private ManualActionQueryDelegate() {}

    @QueryDelegate(ManualActionEntity.class)
    public static Predicate filter(
            QManualActionEntity manualActionEntity, @Nullable ManualActionFilter manualActionFilter) {
        BooleanBuilder builder = new BooleanBuilder();
        if (manualActionFilter != null) {
            if (manualActionFilter.getUserId() != null) {
                builder.and(manualActionEntity.user.id.eq(manualActionFilter.getUserId()));
            }
            if (CollectionUtils.isNotEmpty(manualActionFilter.getTypes())) {
                builder.and(manualActionEntity.type.in(manualActionFilter.getTypes()));
            }
            if (manualActionFilter.getFrom() != null) {
                builder.and(manualActionEntity.created.goe(manualActionFilter.getFrom()));
            }
            if (manualActionFilter.getTo() != null) {
                builder.and(manualActionEntity.created.loe(manualActionFilter.getTo()));
            }
            if (manualActionFilter.getMessageId() != null) {
                builder.and(manualActionEntity
                        .as(QMessageActionManualActionEntity.class)
                        .messages
                        .any()
                        .id
                        .in(manualActionFilter.getMessageId()));
            }
        }

        return builder.getValue();
    }
}
