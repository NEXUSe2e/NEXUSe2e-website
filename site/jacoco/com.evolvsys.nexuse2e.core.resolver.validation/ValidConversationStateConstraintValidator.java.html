package com.evolvsys.nexuse2e.core.resolver.validation;

import com.evolvsys.nexuse2e.core.message.context.MessageContext;
import com.evolvsys.nexuse2e.core.message.repository.ConversationRepository;
import com.evolvsys.nexuse2e.core.message.repository.MessageRepository;
import com.evolvsys.nexuse2e.core.model.entity.ConversationEntity;
import com.evolvsys.nexuse2e.core.model.entity.ConversationStatus;
import com.evolvsys.nexuse2e.core.model.entity.MessageEntity;
import com.evolvsys.nexuse2e.internal.Message;
import com.evolvsys.nexuse2e.internal.MessageType;
import jakarta.validation.ConstraintValidator;
import jakarta.validation.ConstraintValidatorContext;
import java.util.EnumSet;
import lombok.RequiredArgsConstructor;
import lombok.extern.slf4j.Slf4j;
import org.hibernate.validator.constraintvalidation.HibernateConstraintValidatorContext;
import org.springframework.stereotype.Component;

/**
 * @author SvenBarden
 * @since 04.04.2024
 */
@Component
@RequiredArgsConstructor
@Slf4j
class ValidConversationStateConstraintValidator implements ConstraintValidator<ValidConversationState, MessageContext> {
    private final ConversationRepository conversationRepository;
    private final MessageRepository messageRepository;

    @Override
    public boolean isValid(MessageContext value, ConstraintValidatorContext context) {
        Message sourceMessage = value.getSourceMessage();

        boolean isValid;
        if (sourceMessage != null && sourceMessage.getType() == MessageType.NORMAL) {
            MessageEntity messageEntity =
                    messageRepository.findByMessageId(sourceMessage.getId()).orElse(null);

            if (messageEntity == null) {
                ConversationEntity conversationEntity = conversationRepository
                        .findByConversationId(sourceMessage.getConversationId())
                        .orElse(null);
                if (conversationEntity == null) {
                    isValid = true;
                    LOGGER.trace("conversation does not exist, no state validation required");
                } else {
                    isValid = EnumSet.of(ConversationStatus.IDLE, ConversationStatus.WAITING_FOR_ACKNOWLEDGMENT)
                            .contains(conversationEntity.getStatus());
                    LOGGER.trace("conversation in state {} is valid: {}", conversationEntity.getStatus(), isValid);

                    if (isValid == false) {
                        HibernateConstraintValidatorContext hibernateConstraintValidatorContext =
                                context.unwrap(HibernateConstraintValidatorContext.class);
                        hibernateConstraintValidatorContext.addMessageParameter(
                                "conversation", sourceMessage.getConversationId());
                        hibernateConstraintValidatorContext.addMessageParameter(
                                "conversationState", conversationEntity.getStatus());
                    }
                }
            } else {
                isValid = true;
                LOGGER.trace("message already exists, no conversation state validation required");
            }
        } else {
            isValid = true;
        }

        return isValid;
    }
}
