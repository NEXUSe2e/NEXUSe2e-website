package com.evolvsys.nexuse2e.core.resolver.validation;

import com.evolvsys.nexuse2e.core.message.MessageStatus;
import com.evolvsys.nexuse2e.core.message.context.MessageContext;
import com.evolvsys.nexuse2e.core.message.repository.MessageRepository;
import com.evolvsys.nexuse2e.core.model.entity.MessageEntity;
import com.evolvsys.nexuse2e.core.utils.BeanValidationUtils;
import com.evolvsys.nexuse2e.internal.Message;
import com.evolvsys.nexuse2e.internal.MessageType;
import jakarta.validation.ConstraintValidator;
import jakarta.validation.ConstraintValidatorContext;
import java.util.EnumSet;
import java.util.Set;
import lombok.RequiredArgsConstructor;
import lombok.extern.slf4j.Slf4j;
import org.apache.commons.collections4.keyvalue.DefaultKeyValue;
import org.apache.commons.lang3.StringUtils;
import org.springframework.stereotype.Component;

/**
 * @author SvenBarden
 * @since 18.04.2024
 */
@Component
@RequiredArgsConstructor
@Slf4j
class ValidAcknowledgmentConstraintValidator implements ConstraintValidator<ValidAcknowledgment, MessageContext> {
    private static final Set<MessageStatus> VALID_MESSAGE_STATUS =
            EnumSet.of(MessageStatus.FAILED_MISSING_ACKNOWLEDGMENT, MessageStatus.WAITING_FOR_ACKNOWLEDGMENT);

    private final MessageRepository messageRepository;

    private void emptyReferenceId(ConstraintValidatorContext context, Message message) {
        context.disableDefaultConstraintViolation();
        BeanValidationUtils.addMessageParameters(
                context,
                new DefaultKeyValue<>("acknowledgmentId", message.getId()),
                new DefaultKeyValue<>("conversationId", message.getConversationId()));
        context.buildConstraintViolationWithTemplate(
                        "{com.evolvsys.nexuse2e.core.resolver.validation.ValidAcknowledgment.missingReferenceId.message}")
                .addConstraintViolation();
    }

    private void noReferenceMessage(ConstraintValidatorContext context, Message message) {
        context.disableDefaultConstraintViolation();
        BeanValidationUtils.addMessageParameters(
                context,
                new DefaultKeyValue<>("acknowledgmentId", message.getId()),
                new DefaultKeyValue<>("referenceMessageId", message.getReferenceMessageId()),
                new DefaultKeyValue<>("conversationId", message.getConversationId()));
        context.buildConstraintViolationWithTemplate(
                        "{com.evolvsys.nexuse2e.core.resolver.validation.ValidAcknowledgment.noReferenceMessage.message}")
                .addConstraintViolation();
    }

    private void invalidState(ConstraintValidatorContext context, Message message, MessageStatus messageStatus) {
        context.disableDefaultConstraintViolation();
        BeanValidationUtils.addMessageParameters(
                context,
                new DefaultKeyValue<>("acknowledgmentId", message.getId()),
                new DefaultKeyValue<>("referenceMessageId", message.getReferenceMessageId()),
                new DefaultKeyValue<>("conversationId", message.getConversationId()),
                new DefaultKeyValue<>("messageStatus", messageStatus));
        context.buildConstraintViolationWithTemplate(
                        "{com.evolvsys.nexuse2e.core.resolver.validation.ValidAcknowledgment.invalidStatus.message}")
                .addConstraintViolation();
    }

    @Override
    public boolean isValid(MessageContext value, ConstraintValidatorContext context) {
        Message message = value.getSourceMessage();
        boolean valid;
        if (message.getType() == MessageType.ACKNOWLEDGMENT) {
            String referenceMessageId = message.getReferenceMessageId();
            if (StringUtils.isBlank(referenceMessageId)) {
                valid = false;
                LOGGER.debug("acknowledgment {} has no referenceId", message.getId());
                emptyReferenceId(context, message);
            } else {
                MessageEntity messageEntity =
                        messageRepository.findByMessageId(referenceMessageId).orElse(null);
                if (messageEntity == null) {
                    valid = false;
                    LOGGER.debug(
                            "reference {} for acknowledgment {} does not exist",
                            message.getReferenceMessageId(),
                            message.getId());
                    noReferenceMessage(context, message);
                } else {
                    if (VALID_MESSAGE_STATUS.contains(messageEntity.getStatus())) {
                        valid = true;
                        LOGGER.debug(
                                "got acknowledgment {} for message {} in state {}",
                                message.getId(),
                                message.getReferenceMessageId(),
                                messageEntity.getStatus());
                    } else {
                        valid = false;
                        LOGGER.warn(
                                "got acknowledgment {} for message {}, but message was in state {}, valid states are {}",
                                message.getId(),
                                message.getReferenceMessageId(),
                                messageEntity.getStatus(),
                                VALID_MESSAGE_STATUS);
                        invalidState(context, message, messageEntity.getStatus());
                    }
                }
            }
        } else {
            valid = true;
        }
        return valid;
    }
}
