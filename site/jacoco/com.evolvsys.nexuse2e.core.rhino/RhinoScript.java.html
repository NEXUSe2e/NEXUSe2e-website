package com.evolvsys.nexuse2e.core.rhino;

import java.lang.reflect.Method;
import java.lang.reflect.Modifier;
import lombok.Value;
import lombok.extern.slf4j.Slf4j;
import org.mozilla.javascript.Context;
import org.mozilla.javascript.FunctionObject;
import org.mozilla.javascript.ScriptableObject;

/**
 * @author SvenBarden
 * @since 15.11.2023
 */
@Slf4j
public class RhinoScript {

    private void registerDefaultFunctions(ScriptableObject scriptableObject) {
        for (Method method : DefaultFunctions.class.getMethods()) {
            if (Modifier.isPublic(method.getModifiers()) && Modifier.isStatic(method.getModifiers())) {
                FunctionObject function = new FunctionObject(method.getName(), method, scriptableObject);
                scriptableObject.put(function.getFunctionName(), scriptableObject, function);
                LOGGER.trace("registering function {}", function.getFunctionName());
            }
        }
    }

    @SuppressWarnings("unchecked")
    public <T> T evaluate(String script, Class<T> targetType, Parameter... parameters) {
        try (Context context = Context.enter()) {
            context.getWrapFactory().setJavaPrimitiveWrap(false);

            ScriptableObject scriptableObject = context.initSafeStandardObjects();

            registerDefaultFunctions(scriptableObject);

            if (parameters != null) {
                for (Parameter parameter : parameters) {
                    ScriptableObject.putProperty(
                            scriptableObject,
                            parameter.getName(),
                            Context.javaToJS(parameter.getValue(), scriptableObject));
                }
            }

            return (T)
                    Context.jsToJava(context.evaluateString(scriptableObject, script, "script", 1, null), targetType);
        }
    }

    @Value
    public static class Parameter {
        String name;
        Object value;
    }
}
