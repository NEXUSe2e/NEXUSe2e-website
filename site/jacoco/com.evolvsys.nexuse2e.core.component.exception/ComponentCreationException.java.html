package com.evolvsys.nexuse2e.core.component.exception;

import com.evolvsys.nexuse2e.core.exception.NexusError;
import com.evolvsys.nexuse2e.core.exception.NexusRuntimeException;

/**
 * @author SvenBarden
 * @since 24.11.2023
 */
public class ComponentCreationException extends NexusRuntimeException {

    public ComponentCreationException(Class<?> type, Throwable cause) {
        super(
                NexusError.builder()
                        .message("could not create component with type " + type)
                        .code("ComponentCreation")
                        .extension("type", type.getName())
                        .build(),
                cause);
    }
}
