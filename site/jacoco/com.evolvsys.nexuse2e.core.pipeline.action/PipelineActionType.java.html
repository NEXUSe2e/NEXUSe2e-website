package com.evolvsys.nexuse2e.core.pipeline.action;

import com.evolvsys.app.core.process.ProcessActionType;
import com.evolvsys.app.core.process.ProcessDefinition;
import com.evolvsys.nexuse2e.core.pipeline.model.PipelineDto;
import com.evolvsys.nexuse2e.core.pipeline.model.PipelineEntity;
import lombok.Getter;

/**
 * @author SvenBarden
 * @since 08.04.2024
 */
@Getter
public enum PipelineActionType implements ProcessActionType<PipelineEntity, PipelineDto> {
    ADD_PIPELET(ProcessDefinition.<PipelineEntity, PipelineDto>builder()
            .action(PipelineAddPipeletAction.class)
            .build()),
    REMOVE_PIPELET(ProcessDefinition.<PipelineEntity, PipelineDto>builder()
            .action(PipelineRemovePipeletAction.class)
            .build()),
    REARRANGE_PIPELETS(ProcessDefinition.<PipelineEntity, PipelineDto>builder()
            .action(PipelineRearrangePipeletsAction.class)
            .build());

    private final ProcessDefinition<PipelineEntity, PipelineDto> process;

    PipelineActionType(ProcessDefinition<PipelineEntity, PipelineDto> process) {
        this.process = process;
    }
}
