package com.evolvsys.nexuse2e.core.pipeline.action;

import com.evolvsys.app.core.auth.service.CurrentUserService;
import com.evolvsys.app.core.process.*;
import com.evolvsys.nexuse2e.core.model.entity.PipeletEntity;
import com.evolvsys.nexuse2e.core.pipelet.mapper.PipeletMapper;
import com.evolvsys.nexuse2e.core.pipelet.model.PipeletDto;
import com.evolvsys.nexuse2e.core.pipeline.model.PipelineEntity;
import com.evolvsys.nexuse2e.core.pipeline.repository.PipelineRepository;
import lombok.RequiredArgsConstructor;
import org.springframework.stereotype.Component;

/**
 * @author SvenBarden
 * @since 08.04.2024
 */
@Component
@RequiredArgsConstructor
public class PipelineAddPipeletAction implements ProcessAction<PipelineEntity> {
    public static final ProcessParameter<PipeletDto> PIPELET = new ProcessParameter<>() {};
    private final PipeletMapper pipeletMapper;
    private final PipelineRepository pipelineRepository;
    private final CurrentUserService currentUserService;

    @Override
    public PipelineEntity execute(
            PipelineEntity entity, ProcessActionChain chain, ProcessParameters processParameters) {
        PipeletEntity pipeletEntity = new PipeletEntity();
        pipeletMapper.updateEntity(
                processParameters.get(PIPELET), pipeletEntity, currentUserService.getCurrentUserData());
        pipeletEntity.setOrder(entity.getPipelets().size());
        entity.getPipelets().add(pipeletEntity);

        return pipelineRepository.save(entity);
    }

    @Override
    public ProcessActionConditions conditions(PipelineEntity entity, ProcessParameters processParameters) {
        return ProcessActionConditions.builder()
                .condition(processParameters.containsValue(PIPELET), "Pipelet parameter is required")
                .build();
    }
}
