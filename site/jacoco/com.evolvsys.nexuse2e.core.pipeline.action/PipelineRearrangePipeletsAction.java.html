package com.evolvsys.nexuse2e.core.pipeline.action;

import com.evolvsys.app.core.process.*;
import com.evolvsys.nexuse2e.core.model.entity.PipeletEntity;
import com.evolvsys.nexuse2e.core.pipeline.model.PipelineEntity;
import com.evolvsys.nexuse2e.core.pipeline.repository.PipelineRepository;
import java.util.List;
import lombok.RequiredArgsConstructor;
import org.springframework.stereotype.Component;

/**
 * @author SvenBarden
 * @since 08.04.2024
 */
@Component
@RequiredArgsConstructor
public class PipelineRearrangePipeletsAction implements ProcessAction<PipelineEntity> {
    public static final ProcessParameter<List<Long>> PIPELET_IDS = new ProcessParameter<>() {};
    private final PipelineRepository pipelineRepository;

    @Override
    public PipelineEntity execute(
            PipelineEntity entity, ProcessActionChain chain, ProcessParameters processParameters) {
        List<Long> pipeletIdsInOrder = processParameters.get(PIPELET_IDS);
        for (PipeletEntity pipeletEntity : entity.getPipelets()) {
            pipeletEntity.setOrder(pipeletIdsInOrder.indexOf(pipeletEntity.getId()));
        }

        return pipelineRepository.save(entity);
    }

    @Override
    public ProcessActionConditions conditions(PipelineEntity entity, ProcessParameters processParameters) {
        return ProcessActionConditions.builder()
                .condition(processParameters.containsValue(PIPELET_IDS), "PipeletIds parameter is required")
                .condition(
                        () -> entity.getPipelets().size()
                                == processParameters.get(PIPELET_IDS).size(),
                        "All pipelets must be provided in the order")
                .build();
    }
}
