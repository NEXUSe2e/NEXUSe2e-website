package com.evolvsys.nexuse2e.core.certificate.repository;

import com.evolvsys.nexuse2e.core.certificate.model.CertificateContainerEntity;
import com.evolvsys.nexuse2e.core.certificate.model.CertificateContainerFilter;
import com.evolvsys.nexuse2e.core.certificate.model.QCertificateContainerEntity;
import com.querydsl.core.BooleanBuilder;
import com.querydsl.core.annotations.QueryDelegate;
import com.querydsl.core.types.Predicate;
import org.apache.commons.collections4.CollectionUtils;
import org.apache.commons.lang3.StringUtils;

/**
 * @author SvenBarden
 * @since 07.02.2024
 */
public final class CertificateContainerQueryDelegate {
    private CertificateContainerQueryDelegate() {}

    @QueryDelegate(CertificateContainerEntity.class)
    public static Predicate filter(
            QCertificateContainerEntity certificateContainerEntity,
            CertificateContainerFilter certificateContainerFilter) {
        BooleanBuilder builder = new BooleanBuilder();
        if (certificateContainerFilter != null) {
            if (certificateContainerFilter.getPartnerId() != null) {
                builder.and(certificateContainerEntity.partner.id.eq(certificateContainerFilter.getPartnerId()));
            }
            if (StringUtils.isNotBlank(certificateContainerFilter.getName())) {
                builder.and(certificateContainerEntity.name.contains(certificateContainerFilter.getName()));
            }
            if (CollectionUtils.isNotEmpty(certificateContainerFilter.getTypes())) {
                builder.and(certificateContainerEntity.type.in(certificateContainerFilter.getTypes()));
            }
        }

        return builder.getValue();
    }
}
