package com.evolvsys.nexuse2e.core.certificate.repository;

import com.evolvsys.nexuse2e.core.certificate.model.CertificateEntity;
import com.evolvsys.nexuse2e.core.certificate.model.CertificateFilter;
import com.evolvsys.nexuse2e.core.certificate.model.QCertificateEntity;
import com.evolvsys.nexuse2e.core.certificate.model.QCertificateInfo;
import com.evolvsys.nexuse2e.core.filter.BooleanFilterType;
import com.evolvsys.nexuse2e.core.filter.BooleanQueryDelegateUtils;
import com.querydsl.core.BooleanBuilder;
import com.querydsl.core.annotations.QueryDelegate;
import com.querydsl.core.types.Predicate;
import com.querydsl.core.types.dsl.StringPath;
import java.time.ZonedDateTime;
import org.apache.commons.collections4.CollectionUtils;
import org.apache.commons.lang3.StringUtils;

/**
 * @author SvenBarden
 * @since 09.02.2024
 */
public final class CertificateQueryDelegate {
    private CertificateQueryDelegate() {}

    @QueryDelegate(CertificateEntity.class)
    public static Predicate filter(QCertificateEntity certificateEntity, CertificateFilter certificateFilter) {
        BooleanBuilder builder = new BooleanBuilder();

        if (certificateFilter != null) {
            filterInternal(certificateEntity, certificateFilter, builder);
        }
        return builder.getValue();
    }

    private static void filterInternal(
            QCertificateEntity certificateEntity, CertificateFilter certificateFilter, BooleanBuilder builder) {
        QCertificateInfo certificateInfo = QCertificateEntity.certificateEntity.info;

        if (certificateFilter.getValid() != null) {
            ZonedDateTime now = ZonedDateTime.now();
            if (certificateFilter.getValid() == BooleanFilterType.TRUE) {
                builder.and(certificateInfo.validFrom.before(now).and(certificateInfo.validUntil.after(now)));
            } else if (certificateFilter.getValid() == BooleanFilterType.FALSE) {
                builder.and(certificateInfo.validFrom.after(now).or(certificateInfo.validUntil.before(now)));
            }
        }
        if (CollectionUtils.isNotEmpty(certificateFilter.getTypes())) {
            builder.and(certificateEntity.type.in(certificateFilter.getTypes()));
        }
        if (certificateFilter.getActive() == BooleanFilterType.TRUE) {
            builder.and(certificateEntity.active.isTrue());
        } else if (certificateFilter.getActive() == BooleanFilterType.FALSE) {
            builder.and(certificateEntity.active.isFalse());
        }

        filterStringValue(builder, certificateInfo.sha1Fingerprint, certificateFilter.getSha1Fingerprint());
        filterStringValue(builder, certificateInfo.sha256Fingerprint, certificateFilter.getSha256Fingerprint());
        filterStringValue(builder, certificateInfo.commonName, certificateFilter.getCommonName());
        filterStringValue(builder, certificateInfo.organisation, certificateFilter.getOrganisation());
        filterStringValue(builder, certificateInfo.organisationUnit, certificateFilter.getOrganisationUnit());
        filterStringValue(builder, certificateInfo.country, certificateFilter.getCountry());
        filterStringValue(builder, certificateInfo.state, certificateFilter.getState());
        filterStringValue(builder, certificateInfo.location, certificateFilter.getLocation());
        filterStringValue(builder, certificateInfo.email, certificateFilter.getEmail());
        filterStringValue(builder, certificateInfo.issuer, certificateFilter.getIssuer());

        if (certificateFilter.getSelfSigned() != null) {
            Predicate selfSignedPredicate =
                    BooleanQueryDelegateUtils.filter(certificateInfo.selfSigned, certificateFilter.getSelfSigned());
            if (selfSignedPredicate != null) {
                builder.and(selfSignedPredicate);
            }
        }
    }

    private static void filterStringValue(BooleanBuilder builder, StringPath filterPath, String value) {
        if (StringUtils.isNotEmpty(value)) {
            builder.and(filterPath.contains(value));
        }
    }
}
