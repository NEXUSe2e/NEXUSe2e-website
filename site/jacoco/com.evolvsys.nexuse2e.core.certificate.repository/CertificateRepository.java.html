package com.evolvsys.nexuse2e.core.certificate.repository;

import com.evolvsys.app.core.repository.ListProviderRepository;
import com.evolvsys.nexuse2e.core.certificate.model.*;
import com.querydsl.core.types.dsl.BooleanExpression;
import java.util.Collection;
import java.util.List;
import java.util.Optional;
import org.springframework.data.domain.Page;
import org.springframework.data.domain.Pageable;
import org.springframework.data.jpa.repository.EntityGraph;
import org.springframework.data.jpa.repository.JpaRepository;
import org.springframework.data.jpa.repository.Query;
import org.springframework.data.querydsl.QuerydslPredicateExecutor;
import org.springframework.transaction.annotation.Transactional;

/**
 * @author SvenBarden
 * @since 06.02.2024
 */
public interface CertificateRepository
        extends JpaRepository<CertificateEntity, Long>,
                QuerydslPredicateExecutor<CertificateEntity>,
                ListProviderRepository<CertificateEntity, Long, CertificateFilter> {
    List<CertificateEntity> findByType(CertificateType type);

    List<CertificateEntity> findByCertificateContainerAndActive(
            CertificateContainerEntity certificateContainer, boolean active);

    Optional<CertificateEntity> findFirstByCertificateContainerAndActive(
            CertificateContainerEntity certificateContainer, boolean active);

    @Override
    default Page<CertificateEntity> findAllToRead(
            Pageable pageable, CertificateFilter filter, BooleanExpression securityPredicate) {
        return findAll(securityPredicate.and(QCertificateEntity.certificateEntity.filter(filter)), pageable);
    }

    boolean existsByCertificateContainerAndInfo_sha1Fingerprint(
            CertificateContainerEntity certificateContainer, String sha1Fingerprint);

    boolean existsByCertificateContainerNullAndInfo_sha1Fingerprint(String sha1Fingerprint);

    List<CertificateEntity> findByTypeAndActive(CertificateType type, boolean active);

    @EntityGraph(attributePaths = "file.fileContent")
    List<CertificateEntity> getByTypeAndActive(CertificateType type, boolean active);

    Optional<CertificateEntity> findByNameAndType(String name, CertificateType type);

    List<CertificateEntity> findByCertificateContainer(CertificateContainerEntity certificateContainer);

    Optional<CertificateEntity> findByCertificateContainerAndName(
            CertificateContainerEntity certificateContainer, String name);

    @Transactional
    long deleteByCertificateContainerNotNull();

    @Query(
            """
                    select outerCert from CertificateEntity outerCert where
                    outerCert.id in (
                        select
                            first_value(innerCert.id) over (
                                partition by innerCert.certificateContainer.id
                                order by innerCert.info.validUntil desc
                            )
                        from CertificateEntity innerCert
                        where innerCert.active = :active and innerCert.certificateContainer.id IN(:certificateContainerIds)
                    )
                """)
    List<CertificateEntity> findFirstByActiveAndCertificateContainer_Id(
            boolean active, Collection<Long> certificateContainerIds);

    boolean existsByCertificateContainer_Id(Long id);
}
