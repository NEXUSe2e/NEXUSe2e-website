package com.evolvsys.nexuse2e.core.certificate.repository;

import com.evolvsys.app.core.repository.ListProviderRepository;
import com.evolvsys.nexuse2e.core.certificate.model.CertificateContainerEntity;
import com.evolvsys.nexuse2e.core.certificate.model.CertificateContainerFilter;
import com.evolvsys.nexuse2e.core.certificate.model.CertificateContainerType;
import com.evolvsys.nexuse2e.core.certificate.model.QCertificateContainerEntity;
import com.evolvsys.nexuse2e.core.model.entity.PartnerEntity;
import com.querydsl.core.types.dsl.BooleanExpression;
import java.util.List;
import java.util.Optional;
import org.springframework.data.domain.Page;
import org.springframework.data.domain.Pageable;
import org.springframework.data.jpa.repository.JpaRepository;
import org.springframework.data.querydsl.QuerydslPredicateExecutor;

/**
 * @author SvenBarden
 * @since 07.02.2024
 */
public interface CertificateContainerRepository
        extends JpaRepository<CertificateContainerEntity, Long>,
                QuerydslPredicateExecutor<CertificateContainerEntity>,
                ListProviderRepository<CertificateContainerEntity, Long, CertificateContainerFilter> {

    @Override
    default Page<CertificateContainerEntity> findAllToRead(
            Pageable pageable, CertificateContainerFilter filter, BooleanExpression securityPredicate) {
        return findAll(
                securityPredicate.and(QCertificateContainerEntity.certificateContainerEntity.filter(filter)), pageable);
    }

    List<CertificateContainerEntity> findByPartner(PartnerEntity partner);

    Optional<CertificateContainerEntity> findByPartnerAndName(PartnerEntity partner, String name);

    Optional<CertificateContainerEntity> findByNameAndType(String name, CertificateContainerType type);

    List<CertificateContainerEntity> findByType(CertificateContainerType type);
}
