package com.evolvsys.nexuse2e.core.parameter.converter;

import com.evolvsys.nexuse2e.core.keyvalue.KeyValueLookup;
import com.evolvsys.nexuse2e.core.model.entity.ParameterEntity;
import com.evolvsys.nexuse2e.core.parameter.ParameterDefinition;
import com.evolvsys.nexuse2e.core.parameter.model.ParameterDto;
import java.util.ArrayList;
import java.util.List;
import java.util.function.BiFunction;
import java.util.function.Function;
import org.apache.commons.collections4.CollectionUtils;
import org.springframework.stereotype.Component;

/**
 * @author Florian Drees
 * @since 15.01.2024
 */
@Component
public class StringListTypeConverter implements TypeConverter<List<String>, List<String>> {

    @Override
    public Function<ParameterDto, List<String>> getDtoGetter() {
        return ParameterDto::getStringValues;
    }

    @Override
    public BiFunction<ParameterDto, List<String>, ParameterDto> getDtoSetter() {
        return ParameterDto::setStringValues;
    }

    @Override
    public BiFunction<ParameterDto, List<String>, ParameterDto> getDtoDefaultSetter() {
        return ParameterDto::setDefaultStringValues;
    }

    @Override
    public Function<ParameterDefinition, List<String>> getDefinitionDefaultGetter() {
        //noinspection unchecked
        return parameterDefinition -> (List<String>) parameterDefinition.getDefaultValue();
    }

    @Override
    public List<String> toDtoValue(ParameterEntity parameterEntity) {
        return CollectionUtils.isEmpty(parameterEntity.getStringValues())
                ? new ArrayList<>()
                : new ArrayList<>(
                        parameterEntity
                                .getStringValues()); // entity may contain a persistence proxy, so we copy to be safe
    }

    @Override
    public List<String> toConfigValue(ParameterEntity parameterEntity) {
        return toDtoValue(parameterEntity);
    }

    @Override
    public List<String> toContextValue(ParameterEntity parameterEntity) {
        return toDtoValue(parameterEntity).stream().map(KeyValueLookup::lookup).toList();
    }

    @Override
    public List<String> toDtoValue(List<String> value) {
        return value;
    }

    // CPD-OFF
    @Override
    public void updateFromDto(ParameterEntity parameterEntity, List<String> dtoValue) {
        if (parameterEntity.getStringValues() == null) {
            parameterEntity.setStringValues(new ArrayList<>());
        }
        parameterEntity.getStringValues().clear();
        if (dtoValue != null) {
            parameterEntity.getStringValues().addAll(dtoValue);
        }
    }

    @Override
    public void updateFromConfig(ParameterEntity parameterEntity, List<String> configValue) {
        updateFromDto(parameterEntity, configValue);
    }

    @Override
    public Class<?> getDtoValueType() {
        return List.class;
    }
    // CPD-ON
}
