package com.evolvsys.nexuse2e.core.parameter.converter;

import com.evolvsys.nexuse2e.core.keyvalue.KeyValueLookup;
import com.evolvsys.nexuse2e.core.model.entity.ParameterEntity;
import java.util.ArrayList;
import java.util.List;
import org.apache.commons.lang3.StringUtils;
import org.springframework.stereotype.Component;

/**
 * @author Florian Drees
 * @since 27.02.2024
 */
@Component
public class PasswordTypeConverter extends StringTypeConverter {
    private static final String CENSORED = "$-CENSORED-$";

    @Override
    public void updateFromDto(ParameterEntity parameterEntity, String dtoValue) {
        if (CENSORED.equals(dtoValue) == false) {
            super.updateFromDto(parameterEntity, dtoValue);
        }
    }

    @Override
    public String toDtoValue(ParameterEntity parameterEntity) {
        String value = super.toDtoValue(parameterEntity);
        if (StringUtils.isNotEmpty(value) && KeyValueLookup.lookup(value).equals(value)) {
            value = CENSORED;
        }
        return value;
    }

    @Override
    public String toContextValue(ParameterEntity parameterEntity) {
        String value = parameterEntity.getStringValue();
        if (StringUtils.isNotBlank(value)) {
            value = KeyValueLookup.lookup(value);
        }
        return value;
    }

    @Override
    public List<String> toConfigValue(ParameterEntity parameterEntity) {
        String dtoValue = super.toDtoValue(parameterEntity);
        List<String> result = new ArrayList<>();
        if (dtoValue != null) {
            result.add(dtoValue);
        }
        return result;
    }
}
