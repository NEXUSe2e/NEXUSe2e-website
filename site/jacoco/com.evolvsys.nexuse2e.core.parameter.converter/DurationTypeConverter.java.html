package com.evolvsys.nexuse2e.core.parameter.converter;

import com.evolvsys.nexuse2e.core.model.entity.ParameterEntity;
import java.time.Duration;
import org.springframework.stereotype.Component;

/**
 * @author SvenBarden
 * @since 19.03.2024
 */
@Component
public class DurationTypeConverter extends BaseStringDtoValueTypeConverter<Duration> {

    @Override
    public String toDtoValue(ParameterEntity parameterEntity) {
        return parameterEntity.getStringValue();
    }

    @Override
    public Duration toContextValue(ParameterEntity parameterEntity) {
        return parameterEntity.getStringValue() == null ? null : Duration.parse(parameterEntity.getStringValue());
    }

    @Override
    public String toDtoValue(Duration contextValue) {
        return contextValue == null ? null : String.valueOf(contextValue);
    }

    @Override
    public void updateFromDto(ParameterEntity parameterEntity, String dtoValue) {
        parameterEntity.setStringValue(dtoValue);
    }
}
