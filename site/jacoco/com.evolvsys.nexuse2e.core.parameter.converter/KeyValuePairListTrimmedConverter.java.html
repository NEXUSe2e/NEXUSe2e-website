package com.evolvsys.nexuse2e.core.parameter.converter;

import com.evolvsys.nexuse2e.core.exception.NexusError;
import com.evolvsys.nexuse2e.core.exception.NexusRuntimeException;
import com.evolvsys.nexuse2e.core.keyvalue.KeyValueLookup;
import com.evolvsys.nexuse2e.core.model.entity.ParameterEntity;
import com.evolvsys.nexuse2e.core.parameter.ParameterDefinition;
import com.evolvsys.nexuse2e.core.parameter.model.ParameterDto;
import java.util.ArrayList;
import java.util.Arrays;
import java.util.List;
import java.util.function.BiFunction;
import java.util.function.Function;
import java.util.regex.Pattern;
import lombok.extern.slf4j.Slf4j;
import org.apache.commons.collections4.CollectionUtils;
import org.apache.commons.collections4.KeyValue;
import org.apache.commons.collections4.keyvalue.DefaultKeyValue;
import org.apache.commons.lang3.StringUtils;
import org.springframework.stereotype.Component;

/**
 * @author SvenBarden
 * @since 23.05.2024
 */
@Component
@Slf4j
public class KeyValuePairListTrimmedConverter
        implements TypeConverter<List<? extends KeyValue<String, String>>, List<String>> {
    public static final String SEPARATOR = "~#*#~";

    public static String pairToString(KeyValue<?, ?> keyValue) {
        return keyValue.getKey() + SEPARATOR + keyValue.getValue();
    }

    public static KeyValue<String, String> createKeyValue(String string) {
        KeyValue<String, String> result;
        if (StringUtils.isBlank(string)) {
            result = null;
        } else {
            String[] split = string.split(Pattern.quote(SEPARATOR));
            if (split.length != 2) {
                throw new NexusRuntimeException(NexusError.builder()
                        .message("KeyValuePar string '" + string + "' was invalid")
                        .extension("string", string)
                        .extension("separator", SEPARATOR)
                        .extension("tokens", Arrays.toString(split))
                        .build());
            }
            result = new DefaultKeyValue<>(
                    KeyValueLookup.lookup(split[0].trim()), KeyValueLookup.lookup(split[1].trim()));
        }
        return result;
    }

    @Override
    public Function<ParameterDto, List<String>> getDtoGetter() {
        return ParameterDto::getStringValues;
    }

    @Override
    public BiFunction<ParameterDto, List<String>, ParameterDto> getDtoSetter() {
        return ParameterDto::setStringValues;
    }

    @Override
    public BiFunction<ParameterDto, List<String>, ParameterDto> getDtoDefaultSetter() {
        return ParameterDto::setDefaultStringValues;
    }

    @Override
    public Function<ParameterDefinition, List<? extends KeyValue<String, String>>> getDefinitionDefaultGetter() {
        return parameterDefinition -> null;
    }

    @Override
    public List<String> toDtoValue(ParameterEntity parameterEntity) {
        return CollectionUtils.isEmpty(parameterEntity.getStringValues())
                ? new ArrayList<>()
                : new ArrayList<>(
                        parameterEntity
                                .getStringValues()); // entity may contain a persistence proxy, so we copy to be safe
    }

    @Override
    public List<String> toConfigValue(ParameterEntity parameterEntity) {
        return toDtoValue(parameterEntity);
    }

    @Override
    public List<KeyValue<String, String>> toContextValue(ParameterEntity parameterEntity) {
        return parameterEntity.getStringValues() == null
                ? null
                : parameterEntity.getStringValues().stream()
                        .map(KeyValuePairListTrimmedConverter::createKeyValue)
                        .toList();
    }

    @Override
    public List<String> toDtoValue(List<? extends KeyValue<String, String>> contextValue) {
        return contextValue == null
                ? null
                : contextValue.stream()
                        .map(KeyValuePairListTrimmedConverter::pairToString)
                        .toList();
    }

    // CPD-OFF
    @Override
    public void updateFromDto(ParameterEntity parameterEntity, List<String> dtoValue) {
        if (parameterEntity.getStringValues() == null) {
            parameterEntity.setStringValues(new ArrayList<>());
        }
        parameterEntity.getStringValues().clear();
        if (dtoValue != null) {
            parameterEntity.getStringValues().addAll(dtoValue);
        }
    }

    @Override
    public void updateFromConfig(ParameterEntity parameterEntity, List<String> configValue) {
        updateFromDto(parameterEntity, configValue);
    }

    @Override
    public Class<?> getDtoValueType() {
        return List.class;
    }
    // CPD-ON
}
