package com.evolvsys.nexuse2e.core.parameter.converter;

import com.evolvsys.nexuse2e.core.model.entity.ParameterEntity;
import java.math.BigDecimal;
import java.util.Optional;
import org.springframework.stereotype.Component;

/**
 * @author Florian Drees
 * @since 11.01.2024
 */
@Component
public class BigDecimalTypeConverter extends BaseStringDtoValueTypeConverter<BigDecimal> {

    @Override
    public String toDtoValue(ParameterEntity parameterEntity) {
        return toDtoValue(parameterEntity.getDecimalValue());
    }

    @Override
    public BigDecimal toContextValue(ParameterEntity parameterEntity) {
        return parameterEntity.getDecimalValue();
    }

    @Override
    public String toDtoValue(BigDecimal value) {
        return Optional.ofNullable(value).map(BigDecimal::toString).orElse(null);
    }

    @Override
    public void updateFromDto(ParameterEntity parameterEntity, String dtoValue) {
        parameterEntity.setDecimalValue(dtoValue == null ? null : new BigDecimal(dtoValue));
    }
}
