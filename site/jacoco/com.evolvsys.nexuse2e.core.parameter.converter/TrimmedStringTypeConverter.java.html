package com.evolvsys.nexuse2e.core.parameter.converter;

import com.evolvsys.nexuse2e.core.model.entity.ParameterEntity;
import org.springframework.stereotype.Component;

/**
 * @author Florian Drees
 * @since 12.01.2024
 */
@Component
public class TrimmedStringTypeConverter extends StringTypeConverter {
    @Override
    public String toDtoValue(ParameterEntity parameterEntity) {
        String dtoValue = super.toDtoValue(parameterEntity);
        if (dtoValue != null) {
            dtoValue = dtoValue.trim();
        }
        return dtoValue;
    }

    @Override
    public void updateFromDto(ParameterEntity parameterEntity, String dtoValue) {
        super.updateFromDto(parameterEntity, dtoValue == null ? null : dtoValue.trim());
    }

    @Override
    public String toDtoValue(String contextValue) {
        return super.toDtoValue(contextValue == null ? null : contextValue.trim());
    }
}
