package com.evolvsys.nexuse2e.core.parameter.converter;

import com.evolvsys.app.core.util.GenericUtils;
import com.evolvsys.nexuse2e.core.model.entity.ParameterEntity;
import com.evolvsys.nexuse2e.core.parameter.ParameterDefinition;
import com.evolvsys.nexuse2e.core.parameter.model.ParameterDto;
import java.util.List;
import java.util.function.BiFunction;
import java.util.function.Function;

/**
 * <p>Interface to convert parameter values between strings and their given data type.</p>
 *
 * @author Florian Drees
 * @since 10.01.2024
 */
public interface TypeConverter<ContextValueT, DtoValueT> {

    Function<ParameterDto, DtoValueT> getDtoGetter();

    BiFunction<ParameterDto, DtoValueT, ParameterDto> getDtoSetter();

    BiFunction<ParameterDto, DtoValueT, ParameterDto> getDtoDefaultSetter();

    Function<ParameterDefinition, ContextValueT> getDefinitionDefaultGetter();

    DtoValueT toDtoValue(ParameterEntity parameterEntity);

    List<String> toConfigValue(ParameterEntity parameterEntity);

    ContextValueT toContextValue(ParameterEntity parameterEntity);

    DtoValueT toDtoValue(ContextValueT contextValue);

    void updateFromDto(ParameterEntity parameterEntity, DtoValueT dtoValue);

    default void updateFromDto(ParameterEntity parameterEntity, ParameterDto parameterDto) {
        updateFromDto(parameterEntity, getDtoGetter().apply(parameterDto));
    }

    void updateFromConfig(ParameterEntity parameterEntity, List<String> configValue);

    default Class<?> getDtoValueType() {
        return GenericUtils.getGenericType(getClass(), TypeConverter.class, 1);
    }
}
