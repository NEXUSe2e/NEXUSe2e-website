package com.evolvsys.nexuse2e.core.parameter.converter;

import com.evolvsys.nexuse2e.core.model.entity.ParameterEntity;
import org.apache.commons.lang3.StringUtils;
import org.springframework.stereotype.Component;

/**
 * @author Florian Drees
 * @since 11.01.2024
 */
@Component
public class LongTypeConverter extends BaseStringDtoValueTypeConverter<Long> {
    @Override
    public String toDtoValue(ParameterEntity parameterEntity) {
        return parameterEntity.getLongValue() == null ? null : String.valueOf(parameterEntity.getLongValue());
    }

    @Override
    public Long toContextValue(ParameterEntity parameterEntity) {
        return parameterEntity.getLongValue();
    }

    @Override
    public String toDtoValue(Long contextValue) {
        return contextValue == null ? null : String.valueOf(contextValue);
    }

    @Override
    public void updateFromDto(ParameterEntity parameterEntity, String dtoValue) {
        parameterEntity.setLongValue(StringUtils.isBlank(dtoValue) ? null : Long.valueOf(dtoValue));
    }
}
