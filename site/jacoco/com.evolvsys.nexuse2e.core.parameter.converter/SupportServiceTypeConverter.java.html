package com.evolvsys.nexuse2e.core.parameter.converter;

import com.evolvsys.nexuse2e.core.component.SupportService;
import com.evolvsys.nexuse2e.core.exception.NexusError;
import com.evolvsys.nexuse2e.core.exception.NexusRuntimeException;
import com.evolvsys.nexuse2e.core.model.entity.ParameterEntity;
import com.evolvsys.nexuse2e.core.model.entity.ServiceEntity;
import com.evolvsys.nexuse2e.core.service.ServiceFactory;
import com.evolvsys.nexuse2e.core.service.repository.ServiceRepository;
import java.util.List;
import java.util.Optional;
import lombok.RequiredArgsConstructor;
import lombok.extern.slf4j.Slf4j;
import org.apache.commons.collections4.CollectionUtils;
import org.springframework.stereotype.Component;

/**
 * @author SvenBarden
 * @since 10.06.2024
 */
@Component
@RequiredArgsConstructor
@Slf4j
public class SupportServiceTypeConverter extends BaseStringDtoValueTypeConverter<SupportService> {
    private final ServiceFactory serviceFactory;
    private final ServiceRepository serviceRepository;

    private SupportService getService(long id) {
        try {
            return serviceFactory.get(id);
        } catch (IllegalArgumentException e) {
            LOGGER.error("support service with id {} could not be found", id, e);
            return null;
        }
    }

    private Optional<ServiceEntity> getServiceEntity(long id) {
        return serviceRepository.findById(id);
    }

    @Override
    public String toDtoValue(ParameterEntity parameterEntity) {
        return parameterEntity.getLongValue() == null ? null : String.valueOf(parameterEntity.getLongValue());
    }

    @Override
    public List<String> toConfigValue(ParameterEntity parameterEntity) {
        return Optional.ofNullable(parameterEntity.getLongValue())
                .flatMap(this::getServiceEntity)
                .map(ServiceEntity::getServiceId)
                .stream()
                .toList();
    }

    @Override
    public void updateFromConfig(ParameterEntity parameterEntity, List<String> configValue) {
        if (CollectionUtils.isEmpty(configValue)) {
            parameterEntity.setLongValue(null);
        } else {
            parameterEntity.setLongValue(serviceRepository
                    .findByServiceId(configValue.getFirst())
                    .map(ServiceEntity::getId)
                    .orElse(null));
        }
    }

    @Override
    public SupportService toContextValue(ParameterEntity parameterEntity) {
        return parameterEntity.getLongValue() == null ? null : getService(parameterEntity.getLongValue());
    }

    @Override
    public String toDtoValue(SupportService contextValue) {
        if (contextValue != null) {
            throw new NexusRuntimeException(NexusError.builder()
                    .message(
                            "Conversion from context service to dto is not supported. You cannot assign a default value to SupportService types")
                    .build());
        }
        return null;
    }

    @Override
    public void updateFromDto(ParameterEntity parameterEntity, String dtoValue) {
        parameterEntity.setLongValue(dtoValue == null ? null : Long.valueOf(dtoValue));
    }
}
