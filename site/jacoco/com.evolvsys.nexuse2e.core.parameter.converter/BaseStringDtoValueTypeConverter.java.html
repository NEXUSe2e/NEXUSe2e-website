package com.evolvsys.nexuse2e.core.parameter.converter;

import com.evolvsys.nexuse2e.core.model.entity.ParameterEntity;
import com.evolvsys.nexuse2e.core.parameter.ParameterDefinition;
import com.evolvsys.nexuse2e.core.parameter.model.ParameterDto;
import edu.umd.cs.findbugs.annotations.SuppressFBWarnings;
import java.util.List;
import java.util.function.BiFunction;
import java.util.function.Function;
import org.apache.commons.collections4.CollectionUtils;

/**
 * <p>Interface to convert parameter values between strings and their given data type of the entity.</p>
 * <p>Overrides basic methods which are equal across all types with dto type string.</p>
 *
 * @author Florian Drees
 * @since 15.01.2024
 */
public abstract class BaseStringDtoValueTypeConverter<ContextValueT> implements TypeConverter<ContextValueT, String> {

    @Override
    public Function<ParameterDto, String> getDtoGetter() {
        return ParameterDto::getValue;
    }

    @Override
    public BiFunction<ParameterDto, String, ParameterDto> getDtoSetter() {
        return ParameterDto::setValue;
    }

    @Override
    public BiFunction<ParameterDto, String, ParameterDto> getDtoDefaultSetter() {
        return ParameterDto::setDefaultValue;
    }

    @Override
    @SuppressFBWarnings(value = "FII_USE_METHOD_REFERENCE", justification = "casting required")
    public Function<ParameterDefinition, ContextValueT> getDefinitionDefaultGetter() {
        //noinspection unchecked
        return parameterDefinition -> (ContextValueT) parameterDefinition.getDefaultValue();
    }

    @Override
    public List<String> toConfigValue(ParameterEntity parameterEntity) {
        String dtoValue = toDtoValue(parameterEntity);
        return dtoValue == null ? List.of() : List.of(dtoValue);
    }

    @Override
    public void updateFromConfig(ParameterEntity parameterEntity, List<String> configValue) {
        updateFromDto(parameterEntity, CollectionUtils.isEmpty(configValue) ? null : configValue.getFirst());
    }
}
