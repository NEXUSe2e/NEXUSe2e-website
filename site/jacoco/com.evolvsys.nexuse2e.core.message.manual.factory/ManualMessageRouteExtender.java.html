package com.evolvsys.nexuse2e.core.message.manual.factory;

import com.evolvsys.nexuse2e.core.exception.NexusError;
import com.evolvsys.nexuse2e.core.exception.NexusException;
import com.evolvsys.nexuse2e.core.exception.NexusRuntimeException;
import com.evolvsys.nexuse2e.core.message.context.MessageContext;
import com.evolvsys.nexuse2e.core.message.context.RouteKey;
import com.evolvsys.nexuse2e.core.message.manual.model.ManualMessageDto;
import com.evolvsys.nexuse2e.core.route.model.ContextRoute;
import com.evolvsys.nexuse2e.core.route.service.ContextRouteFactory;
import lombok.RequiredArgsConstructor;
import org.springframework.core.annotation.Order;
import org.springframework.stereotype.Component;

/**
 * @author SvenBarden
 * @since 25.07.2024
 */
@Component
@RequiredArgsConstructor
@Order(500)
class ManualMessageRouteExtender implements ManualMessageContextExtender {
    private final ContextRouteFactory contextRouteFactory;

    @Override
    public void extend(MessageContext messageContext, ManualMessageDto manualMessageDto) throws NexusException {
        ContextRoute contextRoute = contextRouteFactory
                .getRoute(new RouteKey(
                        manualMessageDto.getChoreographyId(),
                        manualMessageDto.getActionId(),
                        messageContext.getSourcePartnerId(),
                        messageContext.getTargetPartnerId()))
                .orElseThrow(() -> new NexusRuntimeException(
                        NexusError.builder().message("no route found").build()));
        messageContext.setRoute(contextRoute);
    }
}
