package com.evolvsys.nexuse2e.core.message.manual.factory;

import com.evolvsys.common.mdc.MDCIdGenerator;
import com.evolvsys.common.mdc.MDCService;
import com.evolvsys.nexuse2e.core.exception.NexusErrorContainer;
import com.evolvsys.nexuse2e.core.message.context.ContextFactory;
import com.evolvsys.nexuse2e.core.message.context.MessageContext;
import com.evolvsys.nexuse2e.core.message.dispatcher.OutboundMessageDispatcher;
import com.evolvsys.nexuse2e.core.message.manual.model.ManualMessageResponseDto;
import com.evolvsys.nexuse2e.core.message.manual.model.ManualMessageStatus;
import com.evolvsys.nexuse2e.core.message.state.MessageEventService;
import com.evolvsys.nexuse2e.core.resolver.ManualMessageResolver;
import com.evolvsys.nexuse2e.core.resolver.MessageResolverValidator;
import com.evolvsys.nexuse2e.core.utils.ErrorUtils;
import com.evolvsys.nexuse2e.internal.MessageType;
import lombok.RequiredArgsConstructor;
import org.springframework.stereotype.Component;

/**
 * @author SvenBarden
 * @since 25.07.2024
 */
@Component
@RequiredArgsConstructor
public class ManualMessageDispatcher {
    private final OutboundMessageDispatcher outboundMessageDispatcher;
    private final MessageEventService messageEventService;
    private final MessageResolverValidator messageResolverValidator;
    private final ContextFactory contextFactory;
    private final MDCService mdcService;
    private final MDCIdGenerator mdcIdGenerator;

    public ManualMessageResponseDto send(MessageContext messageContext) {
        ManualMessageResponseDto.ManualMessageResponseDtoBuilder responseBuilder = ManualMessageResponseDto.builder();

        String originalMdcId = mdcService.getId();
        try {
            mdcService.setId(mdcIdGenerator.newId());

            ManualMessageResolver manualMessageResolver = new ManualMessageResolver(
                    MessageType.NORMAL, messageResolverValidator, messageEventService, contextFactory);
            manualMessageResolver.resolve(messageContext);

            outboundMessageDispatcher.dispatch(messageContext.createExecutionContext(MessageType.NORMAL));

            responseBuilder.status(ManualMessageStatus.SUCCESS);
            responseBuilder.message("success");
        } catch (Exception e) {
            responseBuilder.status(ManualMessageStatus.ERROR);
            if (e instanceof NexusErrorContainer errorContainer) {
                responseBuilder.message(errorContainer.getError().getMessage());
            } else {
                responseBuilder.message(e.getMessage());
            }
            responseBuilder.messageAggregated(ErrorUtils.aggregate(e));
        } finally {
            if (originalMdcId != null) {
                mdcService.setId(originalMdcId);
            }
        }
        return responseBuilder.build();
    }
}
