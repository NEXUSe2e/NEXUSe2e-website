package com.evolvsys.nexuse2e.core.message.manual.factory;

import com.evolvsys.nexuse2e.core.exception.NexusException;
import com.evolvsys.nexuse2e.core.message.context.MessageContext;
import com.evolvsys.nexuse2e.core.message.manual.model.ManualMessageDto;
import org.apache.commons.lang3.StringUtils;
import org.springframework.core.annotation.Order;
import org.springframework.stereotype.Component;

/**
 * @author SvenBarden
 * @since 25.07.2024
 */
@Component
@Order(300)
class ManualMessageIdExtender implements ManualMessageContextExtender {
    @Override
    public void extend(MessageContext messageContext, ManualMessageDto manualMessageDto) throws NexusException {
        if (StringUtils.isNotBlank(manualMessageDto.getMessageId())) {
            messageContext.getSourceMessage().setId(manualMessageDto.getMessageId());
        }
        if (StringUtils.isNotBlank(manualMessageDto.getConversationId())) {
            messageContext.getSourceMessage().setConversationId(manualMessageDto.getConversationId());
        }
    }
}
