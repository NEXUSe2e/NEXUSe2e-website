package com.evolvsys.nexuse2e.core.message.manual.factory;

import com.evolvsys.app.core.exception.GeneralErrorException;
import com.evolvsys.nexuse2e.core.exception.NexusException;
import com.evolvsys.nexuse2e.core.message.context.ContextFactory;
import com.evolvsys.nexuse2e.core.message.context.MessageContext;
import com.evolvsys.nexuse2e.core.message.manual.model.ManualMessageDto;
import com.evolvsys.nexuse2e.internal.ContextPayload;
import com.evolvsys.nexuse2e.internal.Message;
import com.evolvsys.nexuse2e.internal.MessageType;
import java.io.IOException;
import java.sql.SQLException;
import java.util.List;
import java.util.UUID;
import lombok.RequiredArgsConstructor;
import org.apache.commons.lang3.StringUtils;
import org.springframework.stereotype.Component;

/**
 * @author SvenBarden
 * @since 25.07.2024
 */
@RequiredArgsConstructor
@Component
public class ManualMessageFactory {
    private final ManualMessagePayloadFactory payloadFactory;
    private final ContextFactory contextFactory;
    private final List<ManualMessageContextExtender> extenders;

    private MessageContext buildMessageContext(ContextPayload contextPayload) {
        MessageContext messageContext = contextFactory.createNew();

        messageContext.setSourceMessage(buildMessage(contextPayload));
        messageContext.setDoReturnPipelines(true);

        return messageContext;
    }

    private Message buildMessage(ContextPayload contextPayload) {
        Message message = new Message();
        message.setId(UUID.randomUUID().toString());
        message.getPayloads().add(contextPayload);
        message.setConversationId(UUID.randomUUID().toString());
        message.setType(MessageType.NORMAL);
        message.getMetaData().set("Manual-Message", Boolean.TRUE);
        return message;
    }

    public MessageContext create(ManualMessageDto manualMessageDto) throws SQLException, IOException, NexusException {

        ContextPayload contextPayload;
        if (StringUtils.isNotBlank(manualMessageDto.getDefaultPayload())) {
            contextPayload = payloadFactory.buildContextPayload(manualMessageDto.getDefaultPayload());
        } else if (manualMessageDto.getPayloadFileId() != null) {
            contextPayload = payloadFactory.buildContextPayload(manualMessageDto.getPayloadFileId());
        } else {
            throw new GeneralErrorException("either default payload or payload file id must be provided");
        }

        MessageContext messageContext = buildMessageContext(contextPayload);

        for (ManualMessageContextExtender manualMessageContextExtender : extenders) {
            manualMessageContextExtender.extend(messageContext, manualMessageDto);
        }

        return messageContext;
    }
}
