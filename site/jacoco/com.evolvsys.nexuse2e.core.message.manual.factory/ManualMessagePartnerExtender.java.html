package com.evolvsys.nexuse2e.core.message.manual.factory;

import com.evolvsys.nexuse2e.core.config.model.PartnerType;
import com.evolvsys.nexuse2e.core.exception.NexusError;
import com.evolvsys.nexuse2e.core.exception.NexusException;
import com.evolvsys.nexuse2e.core.exception.NexusRuntimeException;
import com.evolvsys.nexuse2e.core.message.context.MessageContext;
import com.evolvsys.nexuse2e.core.message.manual.model.ManualMessageDto;
import com.evolvsys.nexuse2e.core.partner.repository.PartnerRepository;
import lombok.RequiredArgsConstructor;
import org.springframework.core.annotation.Order;
import org.springframework.stereotype.Component;

/**
 * @author SvenBarden
 * @since 25.07.2024
 */
@Component
@RequiredArgsConstructor
@Order(100)
class ManualMessagePartnerExtender implements ManualMessageContextExtender {
    private final PartnerRepository partnerRepository;

    private String getSavePartnerId(Long id) {
        String partnerId;
        if (id == null) {
            partnerId = partnerRepository
                    .getDefaultLocalIdentityName(PartnerType.INTERN)
                    .orElseThrow(() -> new NexusRuntimeException(NexusError.builder()
                            .message("no default local identity found")
                            .build()));
        } else {
            partnerId = partnerRepository
                    .findById(id)
                    .orElseThrow(() -> new NexusRuntimeException(NexusError.builder()
                            .message("partner not found")
                            .extension("id", id)
                            .build()))
                    .getPartnerId();
        }

        return partnerId;
    }

    @Override
    public void extend(MessageContext messageContext, ManualMessageDto manualMessageDto) throws NexusException {
        messageContext.setSourcePartnerId(getSavePartnerId(manualMessageDto.getSourcePartnerId()));
        messageContext.setTargetPartnerId(getSavePartnerId(manualMessageDto.getTargetPartnerId()));
    }
}
