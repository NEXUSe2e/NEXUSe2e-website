package com.evolvsys.nexuse2e.core.message.manual.factory;

import com.evolvsys.nexuse2e.core.action.model.ActionEntity;
import com.evolvsys.nexuse2e.core.action.repository.ActionRepository;
import com.evolvsys.nexuse2e.core.choreography.repository.ChoreographyRepository;
import com.evolvsys.nexuse2e.core.exception.NexusError;
import com.evolvsys.nexuse2e.core.exception.NexusRuntimeException;
import com.evolvsys.nexuse2e.core.message.context.MessageContext;
import com.evolvsys.nexuse2e.core.message.manual.model.ManualMessageDto;
import com.evolvsys.nexuse2e.core.model.entity.ChoreographyEntity;
import lombok.RequiredArgsConstructor;
import org.springframework.core.annotation.Order;
import org.springframework.stereotype.Component;

/**
 * @author SvenBarden
 * @since 25.07.2024
 */
@Component
@RequiredArgsConstructor
@Order(200)
class ManualMessageChoreographyExtender implements ManualMessageContextExtender {
    private final ChoreographyRepository choreographyRepository;
    private final ActionRepository actionRepository;

    @Override
    public void extend(MessageContext messageContext, ManualMessageDto manualMessageDto) {
        ChoreographyEntity choreographyEntity = choreographyRepository
                .findById(manualMessageDto.getChoreographyId())
                .orElseThrow(() -> new NexusRuntimeException(NexusError.builder()
                        .message("Could not find choreography for manual message")
                        .extension("id", manualMessageDto.getChoreographyId())
                        .build()));
        ActionEntity actionEntity = actionRepository
                .findById(manualMessageDto.getActionId())
                .orElseThrow(() -> new NexusRuntimeException(NexusError.builder()
                        .message("Could not find action for manual message")
                        .extension("id", manualMessageDto.getActionId())
                        .build()));

        messageContext.setChoreographyId(choreographyEntity.getChoreographyId());
        messageContext.setActionId(actionEntity.getActionId());
    }
}
