package com.evolvsys.nexuse2e.core.message.manual.factory;

import com.evolvsys.app.core.exception.GeneralErrorException;
import com.evolvsys.app.core.exception.ObjectNotFoundException;
import com.evolvsys.app.core.file.model.FileDto;
import com.evolvsys.app.core.file.model.FileEntity;
import com.evolvsys.app.core.file.repository.FileRepository;
import com.evolvsys.nexuse2e.core.config.NEXUSe2eConfigurationProperties;
import com.evolvsys.nexuse2e.core.exception.NexusException;
import com.evolvsys.nexuse2e.core.message.payload.content.PayloadContent;
import com.evolvsys.nexuse2e.core.utils.ResourceUtils;
import com.evolvsys.nexuse2e.internal.ContextPayload;
import java.io.IOException;
import java.sql.SQLException;
import lombok.RequiredArgsConstructor;
import lombok.extern.slf4j.Slf4j;
import org.apache.tika.Tika;
import org.springframework.core.io.Resource;
import org.springframework.stereotype.Component;

/**
 * @author SvenBarden
 * @since 25.07.2024
 */
@Component
@RequiredArgsConstructor
@Slf4j
class ManualMessagePayloadFactory {
    private final FileRepository fileRepository;
    private final NEXUSe2eConfigurationProperties configurationProperties;

    public ContextPayload buildContextPayload(Long fileId) throws SQLException {
        FileEntity fileEntity = fileRepository
                .getFileEntityById(fileId)
                .orElseThrow(() -> new ObjectNotFoundException(FileDto.class, fileId));

        ContextPayload contextPayload = new ContextPayload();
        contextPayload.setContent(
                PayloadContent.of(fileEntity.getFileContent().getContent().getBinaryStream(), fileEntity.getSize()));
        contextPayload.setEncoding("utf-8");
        contextPayload.setSize(fileEntity.getSize());
        contextPayload.setContentType(fileEntity.getMimeType());
        contextPayload.setContentId(fileEntity.getFileName());

        return contextPayload;
    }

    public ContextPayload buildContextPayload(String fileName) throws IOException, NexusException {
        LOGGER.debug("filter manual files for {}", fileName);
        Resource manualMessageFile = ResourceUtils.list(configurationProperties.getDefaultManualMessageFiles()).stream()
                .filter(resource -> fileName.equals(resource.getFilename()))
                .findFirst()
                .orElse(null);
        if (manualMessageFile == null) {
            LOGGER.error(
                    "file {} could not be found in directory {}",
                    fileName,
                    configurationProperties.getDefaultManualMessageFiles());
            throw new GeneralErrorException("no manual message files found");
        }

        long fileLength = manualMessageFile.contentLength();

        ContextPayload contextPayload = new ContextPayload();
        contextPayload.setContent(PayloadContent.of(manualMessageFile.getInputStream(), fileLength));
        contextPayload.setEncoding("utf-8");
        contextPayload.setSize(fileLength);
        contextPayload.setContentType(new Tika().detect(fileName));
        contextPayload.setContentId(fileName);

        return contextPayload;
    }
}
