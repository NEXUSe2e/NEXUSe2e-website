package com.evolvsys.nexuse2e.core.keyvalue.persistence;

import java.util.LinkedHashMap;
import java.util.Map;
import java.util.Optional;
import lombok.Value;

/**
 * @author SvenBarden
 * @since 18.06.2024
 */
public class InMemoryPersistentKeyValueRepositoryImpl implements PersistentKeyValueRepository {
    private final Map<Key, String> values = new LinkedHashMap<>();

    @Override
    public void save(String category, String key, String value) {
        values.put(new Key(category, key), value);
    }

    @Override
    public Optional<String> get(String category, String key) {
        return Optional.ofNullable(values.get(new Key(category, key)));
    }

    @Value
    private static class Key {
        String category;
        String name;
    }
}
