package com.evolvsys.nexuse2e.core.keyvalue.persistence;

import com.evolvsys.nexuse2e.core.keyvalue.persistence.model.KeyValuePropertyEntity;
import com.evolvsys.nexuse2e.core.keyvalue.persistence.repository.KeyValuePropertyRepository;
import java.util.Optional;
import lombok.RequiredArgsConstructor;
import org.springframework.stereotype.Component;

/**
 * @author SvenBarden
 * @since 18.06.2024
 */
@RequiredArgsConstructor
@Component
public class JpaPersistentKeyValueRepositoryImpl implements PersistentKeyValueRepository {
    private final KeyValuePropertyRepository keyValuePropertyRepository;

    private KeyValuePropertyEntity create(String category, String key) {
        KeyValuePropertyEntity entity = new KeyValuePropertyEntity();
        entity.setCategory(category);
        entity.setKey(key);
        return entity;
    }

    @Override
    public void save(String category, String key, String value) {
        KeyValuePropertyEntity entity =
                keyValuePropertyRepository.findByCategoryAndKey(category, key).orElseGet(() -> create(category, key));
        entity.setValue(value);
        keyValuePropertyRepository.save(entity);
    }

    @Override
    public Optional<String> get(String category, String key) {
        return keyValuePropertyRepository.findByCategoryAndKey(category, key).map(KeyValuePropertyEntity::getValue);
    }
}
