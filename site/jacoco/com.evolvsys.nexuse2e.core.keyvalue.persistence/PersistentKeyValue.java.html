package com.evolvsys.nexuse2e.core.keyvalue.persistence;

import com.evolvsys.nexuse2e.core.exception.NexusError;
import com.evolvsys.nexuse2e.core.exception.NexusRuntimeException;
import java.time.ZonedDateTime;
import java.time.format.DateTimeFormatter;
import java.util.Optional;
import lombok.RequiredArgsConstructor;
import org.apache.commons.lang3.StringUtils;

/**
 * @author SvenBarden
 * @since 18.06.2024
 */
@RequiredArgsConstructor
public class PersistentKeyValue {
    private final PersistentKeyValueRepository keyValueRepository;

    public void set(String category, String key, String value) {
        if (StringUtils.isBlank(category)) {
            throw new NexusRuntimeException(NexusError.builder()
                    .message("Persistent Category Name must not be blank")
                    .build());
        }
        if (StringUtils.isBlank(key)) {
            throw new NexusRuntimeException(NexusError.builder()
                    .message("Persistent Key must not be blank")
                    .build());
        }
        keyValueRepository.save(category, key, value);
    }

    public void setLong(String category, String key, Long value) {
        set(category, key, value == null ? null : value.toString());
    }

    public void setDateTime(String category, String key, ZonedDateTime dateTime) {
        set(category, key, dateTime == null ? null : dateTime.format(DateTimeFormatter.ISO_ZONED_DATE_TIME));
    }

    public Optional<String> get(String category, String key) {
        return keyValueRepository.get(category, key);
    }

    public Optional<Long> getAsLong(String category, String key) {
        return get(category, key).map(Long::valueOf);
    }

    public Optional<ZonedDateTime> getAsDatetime(String category, String key) {
        return get(category, key).map(ZonedDateTime::parse);
    }
}
