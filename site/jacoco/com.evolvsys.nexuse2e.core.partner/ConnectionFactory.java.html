package com.evolvsys.nexuse2e.core.partner;

import com.evolvsys.nexuse2e.core.action.Action;
import com.evolvsys.nexuse2e.core.message.context.RouteKey;
import com.evolvsys.nexuse2e.core.route.model.ContextRoute;
import com.evolvsys.nexuse2e.core.route.service.ContextRouteFactory;
import java.util.Optional;
import lombok.RequiredArgsConstructor;
import lombok.extern.slf4j.Slf4j;
import org.springframework.stereotype.Service;
import org.springframework.transaction.annotation.Transactional;

/**
 * @author SvenBarden
 * @since 23.10.2023
 */
@Service
@RequiredArgsConstructor
@Slf4j
public class ConnectionFactory {
    private final ContextRouteFactory contextRouteFactory;

    @Transactional(readOnly = true)
    public Optional<ContextRoute> getRoute(String sourcePartnerId, String targetPartnerId, Action action) {
        return contextRouteFactory.getRoute(
                new RouteKey(action.getChoreographyRefId(), action.getId(), sourcePartnerId, targetPartnerId));
    }
}
