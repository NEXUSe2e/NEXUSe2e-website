package com.evolvsys.nexuse2e.core.partner;

import com.evolvsys.nexuse2e.core.message.context.ContextObjectFactory;
import com.evolvsys.nexuse2e.core.partner.mapper.ContextPartnerMapper;
import com.evolvsys.nexuse2e.core.partner.model.ContextPartner;
import com.evolvsys.nexuse2e.core.partner.repository.PartnerRepository;
import lombok.RequiredArgsConstructor;
import lombok.extern.slf4j.Slf4j;
import org.springframework.stereotype.Component;

/**
 * @author SvenBarden
 * @since 24.01.2024
 */
@Component
@RequiredArgsConstructor
@Slf4j
public class PartnerFactory implements ContextObjectFactory<String, ContextPartner> {
    private final PartnerRepository partnerRepository;
    private final ContextPartnerMapper contextPartnerMapper;

    @Override
    public ContextPartner create(String partnerId) {
        return partnerRepository
                .findByPartnerId(partnerId)
                .map(contextPartnerMapper::entityToPartner)
                .orElseThrow(() -> new IllegalArgumentException("no partner for id " + partnerId + " found"));
    }
}
