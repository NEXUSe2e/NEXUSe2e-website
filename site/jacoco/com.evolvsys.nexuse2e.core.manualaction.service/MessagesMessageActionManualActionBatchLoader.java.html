package com.evolvsys.nexuse2e.core.manualaction.service;

import com.evolvsys.app.core.graphql.dataloader.BaseJoinBatchLoader;
import com.evolvsys.app.core.graphql.repository.BatchLoaderRepository;
import com.evolvsys.nexuse2e.core.manualaction.model.QMessageActionManualActionEntity;
import com.evolvsys.nexuse2e.core.message.mapper.MessageMapper;
import com.evolvsys.nexuse2e.core.message.model.MessageDto;
import com.evolvsys.nexuse2e.core.model.entity.MessageEntity;
import com.evolvsys.nexuse2e.core.model.entity.QMessageEntity;
import com.querydsl.core.types.OrderSpecifier;
import org.springframework.stereotype.Component;

/**
 * @author SvenBarden
 * @since 05.08.2024
 */
@Component
public class MessagesMessageActionManualActionBatchLoader
        extends BaseJoinBatchLoader<MessageDto, MessageEntity, QMessageEntity> {
    public MessagesMessageActionManualActionBatchLoader(
            MessageMapper mapper, BatchLoaderRepository batchLoaderRepository) {
        super(mapper, batchLoaderRepository, QMessageActionManualActionEntity.messageActionManualActionEntity.messages);
    }

    @Override
    protected OrderSpecifier<?>[] orderSpecifiers() {
        return new OrderSpecifier[] {new QMessageEntity("rel").created.asc()};
    }
}
