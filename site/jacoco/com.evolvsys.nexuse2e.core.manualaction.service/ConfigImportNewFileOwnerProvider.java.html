package com.evolvsys.nexuse2e.core.manualaction.service;

import com.evolvsys.app.core.file.model.DefaultFileObjectType;
import com.evolvsys.app.core.file.model.FileDto;
import com.evolvsys.app.core.file.model.FileObjectType;
import com.evolvsys.app.core.file.service.FileOwnerProvider;
import com.evolvsys.nexuse2e.core.manualaction.model.ConfigImportManualActionDto;
import java.util.Collection;
import lombok.RequiredArgsConstructor;
import org.springframework.stereotype.Component;

/**
 * @author SvenBarden
 * @since 08.08.2024
 */
@Component
@RequiredArgsConstructor
public class ConfigImportNewFileOwnerProvider implements FileOwnerProvider<ConfigImportManualActionDto> {
    public static final FileObjectType NEW_CONFIG = new DefaultFileObjectType("NEW_CONFIG", "newConfig");

    private final ConfigImportManualActionCrudService configImportCrudService;

    @Override
    public Collection<FileDto> getFiles(ConfigImportManualActionDto ownerDto) {
        return getSingleFile(ownerDto.getNewConfig());
    }

    @Override
    public ConfigImportManualActionDto getOwnerDto(Long ownerId) {
        return configImportCrudService.getOrThrow(ownerId);
    }

    @Override
    public FileObjectType getType() {
        return NEW_CONFIG;
    }
}
