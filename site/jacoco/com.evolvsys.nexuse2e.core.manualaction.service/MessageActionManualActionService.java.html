package com.evolvsys.nexuse2e.core.manualaction.service;

import com.evolvsys.app.core.auth.service.CurrentUserService;
import com.evolvsys.nexuse2e.core.exception.NexusError;
import com.evolvsys.nexuse2e.core.exception.NexusRuntimeException;
import com.evolvsys.nexuse2e.core.logging.model.ForMessageLoggingContext;
import com.evolvsys.nexuse2e.core.manualaction.model.MessageActionManualActionEntity;
import com.evolvsys.nexuse2e.core.manualaction.model.MessageActionManualActionType;
import com.evolvsys.nexuse2e.core.manualaction.repository.MessageActionManualActionRepository;
import com.evolvsys.nexuse2e.core.model.entity.MessageEntity;
import com.evolvsys.nexuse2e.core.utils.LoggingUtils;
import lombok.RequiredArgsConstructor;
import lombok.extern.slf4j.Slf4j;
import org.springframework.stereotype.Service;

/**
 * @author SvenBarden
 * @since 05.08.2024
 */
@Service
@RequiredArgsConstructor
@Slf4j
public class MessageActionManualActionService {
    private final MessageActionManualActionRepository manualActionRepository;
    private final CurrentUserService currentUserService;

    private Long saveEntity(Long manualActionId, MessageEntity messageEntity, MessageActionManualActionType type) {
        MessageActionManualActionEntity manualAction;
        if (manualActionId == null) {
            manualAction = new MessageActionManualActionEntity();
        } else {
            manualAction = manualActionRepository
                    .findById(manualActionId)
                    .orElseThrow(() -> new NexusRuntimeException(NexusError.builder()
                            .message("Manual action not found")
                            .extension("manualActionId", manualActionId)
                            .build()));
        }
        manualAction.setActionType(type);
        manualAction.getMessages().add(messageEntity);
        manualAction = manualActionRepository.save(manualAction);

        return manualAction.getId();
    }

    public Long requeue(Long manualActionId, MessageEntity messageEntity) {
        manualActionId = saveEntity(manualActionId, messageEntity, MessageActionManualActionType.REQUEUE);
        LoggingUtils.forMessageWrapper(LOGGER, ForMessageLoggingContext.forEntity(messageEntity))
                .info(
                        "{} {} requeued message {}",
                        currentUserService.getCurrentUser().getFirstName(),
                        currentUserService.getCurrentUser().getLastName(),
                        messageEntity.getMessageId());
        return manualActionId;
    }

    public Long stop(Long manualActionId, MessageEntity messageEntity) {
        manualActionId = saveEntity(manualActionId, messageEntity, MessageActionManualActionType.STOP);
        LoggingUtils.forMessageWrapper(LOGGER, ForMessageLoggingContext.forEntity(messageEntity))
                .info(
                        "{} {} stopped message {}",
                        currentUserService.getCurrentUser().getFirstName(),
                        currentUserService.getCurrentUser().getLastName(),
                        messageEntity.getMessageId());
        return manualActionId;
    }
}
