package com.evolvsys.nexuse2e.core.manualaction.service;

import com.evolvsys.app.core.auth.service.CurrentUserService;
import com.evolvsys.app.core.security.service.SecurityQueryService;
import com.evolvsys.nexuse2e.core.manualaction.mapper.ManualActionMapper;
import com.evolvsys.nexuse2e.core.manualaction.model.ManualActionDto;
import com.evolvsys.nexuse2e.core.manualaction.model.ManualActionEntity;
import com.evolvsys.nexuse2e.core.manualaction.model.ManualActionFilter;
import com.evolvsys.nexuse2e.core.manualaction.repository.ManualActionRepository;
import com.querydsl.core.types.dsl.BooleanExpression;
import lombok.RequiredArgsConstructor;
import org.springframework.data.domain.Page;
import org.springframework.data.domain.Pageable;
import org.springframework.stereotype.Service;

/**
 * @author SvenBarden
 * @since 05.08.2024
 */
@Service
@RequiredArgsConstructor
class ManualActionCrudService {
    private final ManualActionRepository manualActionRepository;
    private final ManualActionMapper manualActionMapper;
    private final CurrentUserService currentUserService;
    private final SecurityQueryService securityQueryService;

    public Page<ManualActionDto> list(Pageable pageable, ManualActionFilter manualActionFilter) {
        BooleanExpression predicate =
                securityQueryService.getPredicate(ManualActionDto.class, currentUserService.getCurrentUserData());
        Page<ManualActionEntity> manualActionPage =
                manualActionRepository.findActionsToRead(pageable, manualActionFilter, predicate);
        return manualActionPage.map(manualActionEntity ->
                manualActionMapper.entityToDto(manualActionEntity, currentUserService.getCurrentUserData()));
    }
}
