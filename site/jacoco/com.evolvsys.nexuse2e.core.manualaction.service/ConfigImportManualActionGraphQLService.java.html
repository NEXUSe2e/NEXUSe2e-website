package com.evolvsys.nexuse2e.core.manualaction.service;

import com.evolvsys.app.core.graphql.annotation.GraphQLService;
import com.evolvsys.nexuse2e.core.manualaction.model.ConfigImportManualActionDto;
import io.leangen.graphql.annotations.GraphQLQuery;
import java.util.Optional;
import lombok.RequiredArgsConstructor;

/**
 * @author SvenBarden
 * @since 07.08.2024
 */
@GraphQLService
@RequiredArgsConstructor
public class ConfigImportManualActionGraphQLService {
    private final ConfigImportManualActionCrudService configImportManualActionCrudService;

    @GraphQLQuery(name = "configImportManualAction_getById")
    public Optional<ConfigImportManualActionDto> getById(long id) {
        return configImportManualActionCrudService.getById(id);
    }
}
