package com.evolvsys.nexuse2e.core.manualaction.service;

import com.evolvsys.app.core.graphql.annotation.GraphQLService;
import com.evolvsys.app.core.graphql.dataloader.DataLoaderService;
import com.evolvsys.nexuse2e.core.manualaction.model.MessageActionManualActionDto;
import com.evolvsys.nexuse2e.core.message.model.MessageDto;
import io.leangen.graphql.annotations.GraphQLContext;
import io.leangen.graphql.annotations.GraphQLEnvironment;
import io.leangen.graphql.annotations.GraphQLNonNull;
import io.leangen.graphql.annotations.GraphQLQuery;
import io.leangen.graphql.execution.ResolutionEnvironment;
import java.util.Optional;
import java.util.Set;
import java.util.concurrent.CompletableFuture;
import lombok.RequiredArgsConstructor;

/**
 * @author SvenBarden
 * @since 05.08.2024
 */
@GraphQLService
@RequiredArgsConstructor
public class MessageActionManualActionGraphQLService {
    private final MessageActionManualActionCrudService messageActionManualActionCrudService;
    private final DataLoaderService dataLoaderService;

    @GraphQLQuery(name = "messageActionManualAction_getById")
    public Optional<MessageActionManualActionDto> getById(long id) {
        return messageActionManualActionCrudService.getById(id);
    }

    @GraphQLQuery
    public CompletableFuture<@GraphQLNonNull Set<@GraphQLNonNull MessageDto>> messages(
            @GraphQLContext MessageActionManualActionDto messageActionManualActionDto,
            @GraphQLEnvironment ResolutionEnvironment resolutionEnvironment) {
        return dataLoaderService.loadRelation(
                resolutionEnvironment,
                MessagesMessageActionManualActionBatchLoader.class,
                messageActionManualActionDto);
    }

    @GraphQLQuery
    public CompletableFuture<Long> messageCount(
            @GraphQLContext MessageActionManualActionDto messageActionManualActionDto,
            @GraphQLEnvironment ResolutionEnvironment resolutionEnvironment) {
        return dataLoaderService.load(
                resolutionEnvironment,
                MessageCountMessageActionManualActionBatchLoader.class,
                messageActionManualActionDto,
                MessageActionManualActionDto::getId);
    }
}
