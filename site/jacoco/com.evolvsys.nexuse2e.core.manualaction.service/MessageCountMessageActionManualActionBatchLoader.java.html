package com.evolvsys.nexuse2e.core.manualaction.service;

import com.evolvsys.app.core.graphql.dataloader.GraphQLBatchLoader;
import com.evolvsys.nexuse2e.core.manualaction.model.QMessageActionManualActionEntity;
import com.evolvsys.nexuse2e.core.utils.BatchLoaderUtils;
import com.querydsl.core.group.GroupBy;
import com.querydsl.jpa.JPQLQueryFactory;
import java.util.Map;
import java.util.Set;
import java.util.concurrent.CompletionStage;
import lombok.RequiredArgsConstructor;
import org.dataloader.BatchLoaderEnvironment;
import org.springframework.stereotype.Component;

/**
 * @author SvenBarden
 * @since 12.08.2024
 */
@Component
@RequiredArgsConstructor
public class MessageCountMessageActionManualActionBatchLoader implements GraphQLBatchLoader<Long, Long> {
    private final JPQLQueryFactory queryFactory;

    @Override
    public CompletionStage<Map<Long, Long>> load(Set<Long> keys, BatchLoaderEnvironment environment) {
        return BatchLoaderUtils.load(keys, this::loadInternal);
    }

    private Map<Long, Long> loadInternal(Set<Long> keys) {
        return queryFactory
                .select(
                        QMessageActionManualActionEntity.messageActionManualActionEntity.id,
                        QMessageActionManualActionEntity.messageActionManualActionEntity
                                .messages
                                .size()
                                .longValue())
                .from(QMessageActionManualActionEntity.messageActionManualActionEntity)
                .where(QMessageActionManualActionEntity.messageActionManualActionEntity.id.in(keys))
                .transform(GroupBy.groupBy(QMessageActionManualActionEntity.messageActionManualActionEntity.id)
                        .as(QMessageActionManualActionEntity.messageActionManualActionEntity
                                .messages
                                .size()
                                .longValue()));
    }
}
