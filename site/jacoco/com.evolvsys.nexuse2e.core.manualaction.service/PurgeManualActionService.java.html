package com.evolvsys.nexuse2e.core.manualaction.service;

import com.evolvsys.app.core.auth.service.CurrentUserService;
import com.evolvsys.app.core.user.model.UserEntity;
import com.evolvsys.nexuse2e.core.exception.NexusError;
import com.evolvsys.nexuse2e.core.exception.NexusException;
import com.evolvsys.nexuse2e.core.logging.model.LogFilter;
import com.evolvsys.nexuse2e.core.manualaction.model.PurgeManualActionEntity;
import com.evolvsys.nexuse2e.core.manualaction.model.PurgeManualActionType;
import com.evolvsys.nexuse2e.core.manualaction.repository.PurgeManualActionRepository;
import com.evolvsys.nexuse2e.core.message.ConversationFilter;
import com.evolvsys.nexuse2e.core.purge.model.PurgeResult;
import com.fasterxml.jackson.core.JsonProcessingException;
import com.fasterxml.jackson.databind.ObjectMapper;
import lombok.RequiredArgsConstructor;
import lombok.extern.slf4j.Slf4j;
import org.springframework.stereotype.Service;

/**
 * @author SvenBarden
 * @since 12.08.2024
 */
@Service
@RequiredArgsConstructor
@Slf4j
public class PurgeManualActionService {
    private final PurgeManualActionRepository purgeManualActionRepository;
    private final CurrentUserService currentUserService;
    private final ObjectMapper objectMapper;

    private void purgeConversationsInternal(
            UserEntity currentUser, ConversationFilter conversationFilter, PurgeResult purgeResult)
            throws NexusException {
        PurgeManualActionEntity purgeManualActionEntity = new PurgeManualActionEntity();
        purgeManualActionEntity.setPurgeType(PurgeManualActionType.CONVERSATIONS);
        purgeManualActionEntity.setResult(purgeResult);
        if (conversationFilter != null) {
            try {
                purgeManualActionEntity.setFilter(objectMapper.writeValueAsString(conversationFilter));
            } catch (JsonProcessingException e) {
                throw new NexusException(
                        NexusError.builder()
                                .message("Could not deserialize conversation filter")
                                .extension("filter", conversationFilter)
                                .build(),
                        e);
            }
        }
        purgeManualActionRepository.save(purgeManualActionEntity);
        LOGGER.info(
                "{} {} purged conversations: {} with filter {}",
                currentUser.getFirstName(),
                currentUser.getLastName(),
                purgeResult,
                conversationFilter);
    }

    public void purgeConversations(ConversationFilter conversationFilter, PurgeResult purgeResult)
            throws NexusException {
        try {
            UserEntity currentUser = currentUserService.getCurrentUser();
            purgeConversationsInternal(currentUser, conversationFilter, purgeResult);
        } catch (IllegalStateException e) {
            // no user currently logged in, e.g. during startup or internal calls
        }
    }

    private void purgeLogsInternal(UserEntity currentUser, LogFilter logFilter, PurgeResult purgeResult)
            throws NexusException {
        PurgeManualActionEntity purgeManualActionEntity = new PurgeManualActionEntity();
        purgeManualActionEntity.setPurgeType(PurgeManualActionType.LOGS);
        purgeManualActionEntity.setResult(purgeResult);
        if (logFilter != null) {
            try {
                purgeManualActionEntity.setFilter(objectMapper.writeValueAsString(logFilter));
            } catch (JsonProcessingException e) {
                throw new NexusException(
                        NexusError.builder()
                                .message("Could not deserialize log filter")
                                .extension("filter", logFilter)
                                .build(),
                        e);
            }
        }
        purgeManualActionRepository.save(purgeManualActionEntity);
        LOGGER.info(
                "{} {} purged logs: {} with filter {}",
                currentUser.getFirstName(),
                currentUser.getLastName(),
                purgeResult,
                logFilter);
    }

    public void purgeLogs(LogFilter logFilter, PurgeResult purgeResult) throws NexusException {
        try {
            UserEntity currentUser = currentUserService.getCurrentUser();
            purgeLogsInternal(currentUser, logFilter, purgeResult);
        } catch (IllegalStateException e) {
            // no user currently logged in, e.g. during startup or internal calls
        }
    }
}
