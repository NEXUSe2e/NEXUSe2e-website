package com.evolvsys.nexuse2e.core.service.http.client;

import com.evolvsys.nexuse2e.core.certificate.model.CertificateType;
import com.evolvsys.nexuse2e.core.certificate.model.ContextKeyStore;
import com.evolvsys.nexuse2e.core.certificate.repository.CertificateRepository;
import com.evolvsys.nexuse2e.core.certificate.service.CertificateHelper;
import com.evolvsys.nexuse2e.core.exception.NexusError;
import com.evolvsys.nexuse2e.core.exception.NexusException;
import com.evolvsys.nexuse2e.core.exception.NexusRuntimeException;
import com.evolvsys.nexuse2e.core.route.model.ContextRoute;
import com.evolvsys.nexuse2e.core.service.http.WebClientContext;
import java.security.cert.X509Certificate;
import java.sql.SQLException;
import java.util.Collections;
import java.util.List;
import lombok.RequiredArgsConstructor;
import org.springframework.data.util.Lazy;
import org.springframework.stereotype.Component;

/**
 * @author SvenBarden
 * @since 23.07.2024
 */
@RequiredArgsConstructor
@Component
public class HttpContextFactory {
    private final CertificateRepository certificateRepository;

    private List<X509Certificate> getCaCertificates() {
        return certificateRepository.getByTypeAndActive(CertificateType.CA, true).stream()
                .map(certificateEntity -> {
                    try {
                        return CertificateHelper.getCertificate(certificateEntity
                                .getFile()
                                .getFileContent()
                                .getContent()
                                .getBinaryStream());
                    } catch (NexusException | SQLException e) {
                        throw new NexusRuntimeException(
                                NexusError.builder()
                                        .message("could not load ca certificate for entity")
                                        .extension("id", certificateEntity.getId())
                                        .build(),
                                e);
                    }
                })
                .toList();
    }

    public WebClientContext.WebClientContextBuilder createContextBuilder() {
        return WebClientContext.builder().trustedCerts(getCaCertificates());
    }

    public WebClientContext.WebClientContextBuilder createContextBuilder(ContextRoute route) {
        return createContextBuilder(route, true);
    }

    public WebClientContext.WebClientContextBuilder createContextBuilder(ContextRoute route, boolean target) {
        Lazy<ContextKeyStore> keyStore = target ? route.getTargetLocalKey() : route.getSourceLocalKey();
        Lazy<List<X509Certificate>> certs =
                target ? route.getTargetConnectionTrustCertificates() : route.getSourceConnectionTrustCertificates();

        return WebClientContext.builder()
                .keyStore(
                        keyStore.getOptional().map(ContextKeyStore::getKeyStore).orElse(null))
                .keyStorePassword(
                        keyStore.getOptional().map(ContextKeyStore::getPassword).orElse(null))
                .trustedCerts(getCaCertificates())
                .partnerCertificates(
                        certs == null
                                ? Collections.emptyList()
                                : certs.getOptional().orElse(Collections.emptyList()));
    }
}
