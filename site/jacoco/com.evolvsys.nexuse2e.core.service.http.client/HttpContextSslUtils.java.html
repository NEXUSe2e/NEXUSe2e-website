package com.evolvsys.nexuse2e.core.service.http.client;

import com.evolvsys.nexuse2e.core.certificate.service.CertificateHelper;
import com.evolvsys.nexuse2e.core.exception.NexusError;
import com.evolvsys.nexuse2e.core.exception.NexusException;
import com.evolvsys.nexuse2e.core.service.http.WebClientContext;
import java.security.Security;
import java.security.cert.CertificateExpiredException;
import java.security.cert.CertificateNotYetValidException;
import java.security.cert.X509Certificate;
import java.util.*;
import java.util.regex.Pattern;
import java.util.stream.Collectors;
import lombok.experimental.UtilityClass;
import lombok.extern.slf4j.Slf4j;
import org.apache.commons.collections4.CollectionUtils;
import org.apache.commons.lang3.StringUtils;

/**
 * @author SvenBarden
 * @since 23.07.2024
 */
@UtilityClass
@Slf4j
public class HttpContextSslUtils {
    private static final List<String> ALL_SSL_PROTOCOLS = List.of("TLSv1", "TLSv1.1", "TLSv1.2", "TLSv1.3");
    private static final Pattern PROPERTY_LIST_PATTERN = Pattern.compile("\\s*,\\s*");
    private static final List<String> SUPPORTED_SSL_PROTOCOLS;

    static {
        SUPPORTED_SSL_PROTOCOLS = supportedSslProtocols(Security.getProperty("jdk.tls.disabledAlgorithms"));
    }

    static List<String> supportedSslProtocols(String property) {
        Set<String> values = Arrays.stream(property.split("\\s*,\\s*")).collect(Collectors.toSet());
        return ALL_SSL_PROTOCOLS.stream()
                .filter(s -> values.contains(s) == false)
                .toList();
    }

    static void verifyProtocols(Collection<String> supportedProtocols, Collection<String> protocols)
            throws NexusException {
        if (CollectionUtils.containsAll(supportedProtocols, protocols) == false) {
            throw new NexusException(NexusError.builder()
                    .message("Unsupported tls protocols configured")
                    .extension("supported", supportedProtocols)
                    .extension("configured", protocols)
                    .build());
        }
    }

    public static void verifyProtocols(Collection<String> protocols) throws NexusException {
        verifyProtocols(SUPPORTED_SSL_PROTOCOLS, protocols);
    }

    private static Collection<String> getOrSystemProperty(Collection<String> collection, String propertyName) {
        Collection<String> result;
        if (CollectionUtils.isEmpty(collection)) {
            String property = System.getProperty(propertyName);
            if (StringUtils.isNotBlank(property)) {
                result = Arrays.stream(PROPERTY_LIST_PATTERN.split(property)).toList();
            } else {
                result = Collections.emptyList();
            }
        } else {
            result = collection;
        }

        return result;
    }

    public static Collection<String> getProtocols(WebClientContext context) {
        return getOrSystemProperty(context.getProtocols(), "https.protocols");
    }

    public static Collection<String> getCiphers(WebClientContext context) {
        return getOrSystemProperty(context.getCiphers(), "https.cipherSuites");
    }

    public static List<X509Certificate> getTrustedCertificates(WebClientContext context) throws NexusException {
        Collection<X509Certificate> certificates;
        if (CollectionUtils.isNotEmpty(context.getPartnerCertificates())) {
            certificates = getTrustedPartnerCertificates(context);
        } else if (CollectionUtils.isNotEmpty(context.getTrustedCerts())) {
            LOGGER.debug(
                    "using {} ca certificates for trustStore",
                    context.getTrustedCerts().size());
            certificates = context.getTrustedCerts();
        } else {
            throw new NexusException(NexusError.builder()
                    .message("Neither ca or trusted partner certificates found")
                    .build());
        }
        return new ArrayList<>(certificates);
    }

    private static List<X509Certificate> getTrustedPartnerCertificates(WebClientContext context) throws NexusException {
        LOGGER.debug(
                "using {} partner certificates for trustStore",
                context.getPartnerCertificates().size());
        List<X509Certificate> certificates;
        if (context.isValidateCertChain()) {
            LOGGER.trace("check trusted partner certs for valid chain");
            List<X509Certificate> finalPartnerCertificates = new ArrayList<>();
            for (X509Certificate partnerCertificate : context.getPartnerCertificates()) {
                List<X509Certificate> partnerChainCerts =
                        CertificateHelper.buildChain(partnerCertificate, new ArrayList<>(context.getTrustedCerts()));
                boolean allValid = true;
                for (X509Certificate partnerChainCert : partnerChainCerts) {
                    try {
                        partnerChainCert.checkValidity();
                    } catch (CertificateExpiredException | CertificateNotYetValidException e) {
                        allValid = false;
                        LOGGER.warn(
                                "partner cert chain validation is enabled, but cert chain for one of the partner certs ({}) contains an invalid cert ({})",
                                partnerCertificate.getSubjectX500Principal(),
                                partnerChainCert.getSubjectX500Principal());
                        break;
                    }
                }
                if (allValid) {
                    X509Certificate lastInChain = partnerChainCerts.getLast();
                    if (lastInChain.getSubjectX500Principal().equals(lastInChain.getIssuerX500Principal())) {
                        LOGGER.trace("last cert in chain is self signed, chain is complete");
                        finalPartnerCertificates.add(partnerCertificate);
                    } else {
                        LOGGER.warn(
                                "partner cert chain validation is enabled, but the chain for cert ({}) is not complete",
                                partnerCertificate.getSubjectX500Principal());
                    }
                }
            }
            LOGGER.trace(
                    "{} out of {} partner certs have a valid chain",
                    finalPartnerCertificates.size(),
                    context.getPartnerCertificates().size());
            if (CollectionUtils.isEmpty(finalPartnerCertificates)) {
                throw new NexusException(NexusError.builder()
                        .message("No trusted partner certificates with valid chain found")
                        .build());
            }
            certificates = finalPartnerCertificates;
        } else {
            LOGGER.trace("skip trusted partner certs chain validation");
            certificates = context.getPartnerCertificates();
        }
        return certificates;
    }
}
