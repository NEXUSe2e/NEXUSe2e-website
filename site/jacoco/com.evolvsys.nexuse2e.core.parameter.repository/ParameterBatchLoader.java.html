package com.evolvsys.nexuse2e.core.parameter.repository;

import com.evolvsys.app.core.auth.model.CurrentUserData;
import com.evolvsys.app.core.graphql.dataloader.BatchLoaderContext;
import com.evolvsys.app.core.graphql.dataloader.GraphQLBatchLoader;
import com.evolvsys.app.core.graphql.repository.BatchLoaderRepository;
import com.evolvsys.nexuse2e.core.component.ComponentRegistry;
import com.evolvsys.nexuse2e.core.component.RegisteredNexusComponent;
import com.evolvsys.nexuse2e.core.model.entity.ParameterEntity;
import com.evolvsys.nexuse2e.core.model.entity.QParameterEntity;
import com.evolvsys.nexuse2e.core.parameter.ParameterBatchLoaderHelper;
import com.evolvsys.nexuse2e.core.parameter.ParameterBatchLoaderKeyProvider;
import com.evolvsys.nexuse2e.core.parameter.model.ParameterDto;
import com.querydsl.core.types.dsl.SetPath;
import java.util.*;
import java.util.concurrent.CompletableFuture;
import java.util.concurrent.CompletionStage;
import java.util.stream.Collectors;
import lombok.RequiredArgsConstructor;
import lombok.extern.slf4j.Slf4j;
import org.apache.commons.collections4.CollectionUtils;
import org.dataloader.BatchLoaderEnvironment;

/**
 * @author Florian Drees
 * @since 09.01.2024
 */
@RequiredArgsConstructor
@Slf4j
public abstract class ParameterBatchLoader
        implements GraphQLBatchLoader<ParameterBatchLoaderKeyProvider, Set<ParameterDto>> {
    private final ComponentRegistry componentRegistry;
    private final BatchLoaderRepository batchLoaderRepository;
    private final ParameterBatchLoaderHelper parameterBatchLoaderHelper;
    private final SetPath<ParameterEntity, QParameterEntity> setPath;

    @Override
    public CompletionStage<Map<ParameterBatchLoaderKeyProvider, Set<ParameterDto>>> load(
            Set<ParameterBatchLoaderKeyProvider> keys, BatchLoaderEnvironment environment) {
        BatchLoaderContext context = environment.getContext();
        CurrentUserData currentUserData = context.getCurrentUserData();

        Set<ParameterBatchLoaderKeyProvider> notNullKeys =
                keys.stream().filter(Objects::nonNull).collect(Collectors.toSet());

        CompletionStage<Map<ParameterBatchLoaderKeyProvider, Set<ParameterDto>>> result;
        if (CollectionUtils.isEmpty(notNullKeys)) {
            result = CompletableFuture.completedStage(new HashMap<>());
        } else {
            result = CompletableFuture.supplyAsync(() -> loadInternal(notNullKeys, currentUserData));
        }
        return result;
    }

    private Map<ParameterBatchLoaderKeyProvider, Set<ParameterDto>> loadInternal(
            Set<ParameterBatchLoaderKeyProvider> keys, CurrentUserData currentUserData) {

        Map<ParameterBatchLoaderKeyProvider, Set<ParameterDto>> map = new HashMap<>();

        List<Long> idKeys =
                keys.stream().map(ParameterBatchLoaderKeyProvider::getId).toList();

        Map<Long, Set<ParameterEntity>> entityMap = batchLoaderRepository.loadEntitiesByJoin(idKeys, setPath, false);

        Map<Long, List<String>> parameterStringLists = parameterBatchLoaderHelper.getStringList(entityMap);

        for (ParameterBatchLoaderKeyProvider keyProvider : keys) {
            Optional<RegisteredNexusComponent> optionalComponent =
                    componentRegistry.getComponent(keyProvider.getComponentId());

            if (optionalComponent.isPresent()) {
                Set<ParameterDto> parameterDtos = parameterBatchLoaderHelper.loadInternalById(
                        optionalComponent.orElseThrow(),
                        currentUserData,
                        entityMap.computeIfAbsent(keyProvider.getId(), id -> new HashSet<>()),
                        parameterStringLists);
                map.put(keyProvider, parameterDtos);
            }
        }
        return map;
    }
}
