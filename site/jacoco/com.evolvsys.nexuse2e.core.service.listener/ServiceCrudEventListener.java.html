package com.evolvsys.nexuse2e.core.service.listener;

import com.evolvsys.app.core.service.listener.TypeSafeCrudEventListener;
import com.evolvsys.nexuse2e.core.model.entity.ServiceEntity;
import com.evolvsys.nexuse2e.core.service.ServiceFactory;
import lombok.RequiredArgsConstructor;
import lombok.extern.slf4j.Slf4j;
import org.springframework.stereotype.Component;

/**
 * @author SvenBarden
 * @since 09.09.2024
 */
@Component
@RequiredArgsConstructor
@Slf4j
class ServiceCrudEventListener extends TypeSafeCrudEventListener<ServiceEntity> {
    private final ServiceFactory serviceFactory;

    @Override
    public void preDeleteInternal(ServiceEntity entity) {
        LOGGER.debug("stopping service {} before deletion", entity.getServiceId());
        serviceFactory.get(entity).stop();
    }

    @Override
    public void preSaveInternal(ServiceEntity entity) {
        if (entity.getId() != null) {
            LOGGER.debug("stopping service {} before update", entity.getServiceId());
            serviceFactory.get(entity).stop();
        }
    }

    @Override
    public void postSaveInternal(ServiceEntity entity) {
        if (entity.isEnabled()) {
            LOGGER.debug("starting service {} after update", entity.getServiceId());
            serviceFactory.get(entity).start();
        }
    }
}
