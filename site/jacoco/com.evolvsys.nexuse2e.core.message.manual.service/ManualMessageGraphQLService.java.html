package com.evolvsys.nexuse2e.core.message.manual.service;

import com.evolvsys.app.core.graphql.annotation.GraphQLService;
import com.evolvsys.nexuse2e.core.exception.NexusException;
import com.evolvsys.nexuse2e.core.message.manual.model.ManualMessageDto;
import com.evolvsys.nexuse2e.core.message.manual.model.ManualMessageResponseListDto;
import com.evolvsys.nexuse2e.core.message.manual.model.ManualMessageRoutingInfoDto;
import com.evolvsys.nexuse2e.core.message.manual.repository.CustomManualMessageRepository;
import io.leangen.graphql.annotations.GraphQLMutation;
import io.leangen.graphql.annotations.GraphQLNonNull;
import io.leangen.graphql.annotations.GraphQLQuery;
import jakarta.validation.Valid;
import java.util.Collection;
import java.util.List;
import lombok.RequiredArgsConstructor;
import lombok.extern.slf4j.Slf4j;

/**
 * @author SvenBarden
 * @since 07.12.2023
 */
@GraphQLService
@RequiredArgsConstructor
@Slf4j
public class ManualMessageGraphQLService {
    private final ManualMessageService manualMessageService;
    private final CustomManualMessageRepository customManualMessageRepository;

    @GraphQLQuery(name = "manualMessage_getRoutingInfos")
    @GraphQLNonNull
    public Collection<@GraphQLNonNull ManualMessageRoutingInfoDto> getRoutingInfos(
            long sourcePartnerId, long targetPartnerId) {
        return customManualMessageRepository.getRoutingInfos(sourcePartnerId, targetPartnerId);
    }

    @GraphQLQuery(name = "manualMessage_defaultPayloads")
    @GraphQLNonNull
    public List<@GraphQLNonNull String> getDefaultPayloads() throws NexusException {
        return manualMessageService.getDefaultFiles();
    }

    @GraphQLMutation(name = "manualMessage_send")
    @GraphQLNonNull
    public ManualMessageResponseListDto send(@GraphQLNonNull @Valid ManualMessageDto manualMessage) {
        LOGGER.info("sending new message {}", manualMessage);
        return manualMessageService.send(manualMessage);
    }
}
