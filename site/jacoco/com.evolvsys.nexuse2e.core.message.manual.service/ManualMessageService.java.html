package com.evolvsys.nexuse2e.core.message.manual.service;

import com.evolvsys.app.core.auth.service.CurrentUserService;
import com.evolvsys.nexuse2e.core.config.NEXUSe2eConfigurationProperties;
import com.evolvsys.nexuse2e.core.exception.NexusErrorContainer;
import com.evolvsys.nexuse2e.core.exception.NexusException;
import com.evolvsys.nexuse2e.core.logging.MessageLogContextHolder;
import com.evolvsys.nexuse2e.core.message.context.MessageContext;
import com.evolvsys.nexuse2e.core.message.manual.factory.ManualMessageDispatcher;
import com.evolvsys.nexuse2e.core.message.manual.factory.ManualMessageFactory;
import com.evolvsys.nexuse2e.core.message.manual.model.ManualMessageDto;
import com.evolvsys.nexuse2e.core.message.manual.model.ManualMessageResponseDto;
import com.evolvsys.nexuse2e.core.message.manual.model.ManualMessageResponseListDto;
import com.evolvsys.nexuse2e.core.message.manual.model.ManualMessageStatus;
import com.evolvsys.nexuse2e.core.utils.ErrorUtils;
import com.evolvsys.nexuse2e.core.utils.ResourceUtils;
import edu.umd.cs.findbugs.annotations.SuppressFBWarnings;
import java.util.List;
import lombok.RequiredArgsConstructor;
import lombok.extern.slf4j.Slf4j;
import org.springframework.core.io.Resource;
import org.springframework.stereotype.Service;

/**
 * @author SvenBarden
 * @since 11.12.2023
 */
@Service
@RequiredArgsConstructor
@Slf4j
public class ManualMessageService {
    private final NEXUSe2eConfigurationProperties configurationProperties;
    private final CurrentUserService currentUserService;

    private final ManualMessageFactory manualMessageFactory;
    private final ManualMessageDispatcher manualMessageDispatcher;

    @SuppressFBWarnings("EXS_EXCEPTION_SOFTENING_NO_CONSTRAINTS")
    public List<String> getDefaultFiles() throws NexusException {
        return ResourceUtils.list(configurationProperties.getDefaultManualMessageFiles()).stream()
                .filter(resource -> resource.getFilename() != null)
                .map(Resource::getFilename)
                .sorted()
                .toList();
    }

    public ManualMessageResponseListDto send(ManualMessageDto manualMessageDto) {
        currentUserService.checkSystemPermissions("NEXUS.SEND_MANUAL_MESSAGE");
        LOGGER.debug("prepare sending manual message {}", manualMessageDto);

        ManualMessageResponseListDto.ManualMessageResponseListDtoBuilder responseListBuilder =
                ManualMessageResponseListDto.builder().status(ManualMessageStatus.SUCCESS);
        for (int i = 1; i <= manualMessageDto.getRepeat(); i++) {
            LOGGER.trace("sending message number {}", i);
            try {
                MessageContext messageContext = manualMessageFactory.create(manualMessageDto);

                ManualMessageResponseDto messageResponseDto = manualMessageDispatcher.send(messageContext);
                responseListBuilder.messageResponse(messageResponseDto);

                if (messageResponseDto.getStatus() != ManualMessageStatus.SUCCESS) {
                    responseListBuilder.status(ManualMessageStatus.FAILURE);
                }
            } catch (Exception e) {
                ManualMessageResponseDto.ManualMessageResponseDtoBuilder responseBuilder =
                        ManualMessageResponseDto.builder();
                responseBuilder.status(ManualMessageStatus.ERROR);
                if (e instanceof NexusErrorContainer errorContainer) {
                    responseBuilder.message(errorContainer.getError().getMessage());
                } else {
                    responseBuilder.message(e.getMessage());
                }
                responseBuilder.messageAggregated(ErrorUtils.aggregate(e));
                responseListBuilder.messageResponse(responseBuilder.build());
                responseListBuilder.status(ManualMessageStatus.FAILURE);
            } finally {
                MessageLogContextHolder.clearContext();
            }
        }

        return responseListBuilder.build();
    }
}
