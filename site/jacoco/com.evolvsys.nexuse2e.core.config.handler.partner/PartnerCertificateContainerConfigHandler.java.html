package com.evolvsys.nexuse2e.core.config.handler.partner;

import com.evolvsys.nexuse2e.core.certificate.model.CertificateContainerEntity;
import com.evolvsys.nexuse2e.core.certificate.model.CertificateContainerType;
import com.evolvsys.nexuse2e.core.certificate.model.CertificateEntity;
import com.evolvsys.nexuse2e.core.certificate.model.CertificateType;
import com.evolvsys.nexuse2e.core.certificate.repository.CertificateContainerRepository;
import com.evolvsys.nexuse2e.core.certificate.repository.CertificateRepository;
import com.evolvsys.nexuse2e.core.config.handler.mapper.CertificateConfigMapper;
import com.evolvsys.nexuse2e.core.config.model.CertificateConfig;
import com.evolvsys.nexuse2e.core.config.model.CertificateContainerConfig;
import com.evolvsys.nexuse2e.core.exception.NexusError;
import com.evolvsys.nexuse2e.core.exception.NexusRuntimeException;
import com.evolvsys.nexuse2e.core.model.entity.PartnerEntity;
import com.evolvsys.nexuse2e.core.partner.repository.PartnerRepository;
import java.time.ZonedDateTime;
import lombok.RequiredArgsConstructor;
import lombok.extern.slf4j.Slf4j;
import org.apache.commons.collections4.CollectionUtils;
import org.springframework.stereotype.Component;
import org.springframework.transaction.annotation.Transactional;

/**
 * @author SvenBarden
 * @since 10.04.2024
 */
@Component
@RequiredArgsConstructor
@Slf4j
class PartnerCertificateContainerConfigHandler {
    private final CertificateContainerRepository certificateContainerRepository;
    private final PartnerRepository partnerRepository;
    private final CertificateRepository certificateRepository;
    private final CertificateConfigMapper certificateConfigMapper;

    @Transactional
    public void update(String partnerId, CertificateContainerConfig certificateContainerConfig) {
        PartnerEntity partnerEntity = partnerRepository
                .findByPartnerId(partnerId)
                .orElseThrow(() -> new NexusRuntimeException(NexusError.builder()
                        .message("Partner for connection not found")
                        .extension("partnerId", partnerId)
                        .build()));

        CertificateContainerEntity certificateContainerEntity = certificateContainerRepository
                .findByPartnerAndName(partnerEntity, certificateContainerConfig.getName())
                .orElseGet(CertificateContainerEntity::new);

        certificateContainerEntity.setName(certificateContainerConfig.getName());
        certificateContainerEntity.setDescription(certificateContainerConfig.getDescription());
        certificateContainerEntity.setPartner(partnerEntity);
        certificateContainerEntity.setType(CertificateContainerType.PUBLIC_KEY);

        certificateContainerEntity = certificateContainerRepository.save(certificateContainerEntity);

        if (CollectionUtils.isNotEmpty(certificateContainerConfig.getCertificates())) {
            for (CertificateConfig certificate : certificateContainerConfig.getCertificates()) {
                CertificateEntity certificateEntity = certificateRepository
                        .findByCertificateContainerAndName(certificateContainerEntity, certificate.getName())
                        .orElseGet(CertificateEntity::new);

                certificateEntity.setCertificateContainer(certificateContainerEntity);
                certificateEntity.setType(CertificateType.PARTNER);

                boolean wasDisabled =
                        certificateEntity.isActive() && (Boolean.TRUE.equals(certificate.getActive()) == false);

                certificateConfigMapper.updateEntity(certificate, certificateEntity);

                if (certificateEntity.isActive() && certificateEntity.getEnabled() == null) {
                    certificateEntity.setEnabled(ZonedDateTime.now());
                }
                if (wasDisabled) {
                    certificateEntity.setDisabled(ZonedDateTime.now());
                }

                certificateRepository.save(certificateEntity);
            }
        }
    }
}
