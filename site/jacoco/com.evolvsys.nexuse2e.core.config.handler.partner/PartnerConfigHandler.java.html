package com.evolvsys.nexuse2e.core.config.handler.partner;

import com.evolvsys.nexuse2e.core.certificate.repository.CertificateContainerRepository;
import com.evolvsys.nexuse2e.core.certificate.repository.CertificateRepository;
import com.evolvsys.nexuse2e.core.config.handler.ConfigHandler;
import com.evolvsys.nexuse2e.core.config.handler.mapper.PartnerConfigMapper;
import com.evolvsys.nexuse2e.core.config.model.*;
import com.evolvsys.nexuse2e.core.model.entity.PartnerEntity;
import com.evolvsys.nexuse2e.core.partner.repository.PartnerConnectionRepository;
import com.evolvsys.nexuse2e.core.partner.repository.PartnerRepository;
import java.util.Collection;
import java.util.Comparator;
import java.util.List;
import lombok.RequiredArgsConstructor;
import lombok.extern.slf4j.Slf4j;
import org.apache.commons.collections4.CollectionUtils;
import org.springframework.core.annotation.Order;
import org.springframework.stereotype.Component;

/**
 * @author SvenBarden
 * @since 26.10.2023
 */
@Component
@RequiredArgsConstructor
@Slf4j
@Order(3)
public class PartnerConfigHandler implements ConfigHandler<PartnerConfig> {
    private final PartnerRepository partnerRepository;
    private final PartnerConnectionRepository partnerConnectionRepository;
    private final PartnerConfigMapper partnerConfigMapper;
    private final CertificateContainerRepository certificateContainerRepository;
    private final CertificateRepository certificateRepository;
    private final PartnerCertificateContainerConfigHandler partnerCertificateContainerConfigHandler;
    private final PartnerConnectionConfigHandler partnerConnectionConfigHandler;

    private PartnerEntity create(PartnerConfig partnerConfig) {
        LOGGER.debug("creating new partner for {}", partnerConfig);
        PartnerEntity partnerEntity = new PartnerEntity();
        partnerEntity.setPartnerId(partnerConfig.getPartnerId());
        return partnerEntity;
    }

    @Override
    public void clear() {
        partnerConnectionRepository.deleteAll();
        certificateRepository.deleteByCertificateContainerNotNull();
        certificateContainerRepository.deleteAll();
        partnerRepository.deleteAll();
    }

    @Override
    public void importConfig(ConfigContext configContext, PartnerConfig partnerConfig) {
        String id = partnerConfig.getPartnerId();
        LOGGER.trace("handling partner with id {}", id);

        PartnerEntity partnerEntity = partnerRepository.findByPartnerId(id).orElseGet(() -> create(partnerConfig));

        partnerConfigMapper.updateEntity(partnerConfig, partnerEntity);

        PartnerEntity savedPartnerEntity = partnerRepository.save(partnerEntity);

        LOGGER.trace("saved partner {}", savedPartnerEntity.getId());

        if (CollectionUtils.isNotEmpty(partnerConfig.getCertificateContainers())) {
            for (CertificateContainerConfig certificateContainer : partnerConfig.getCertificateContainers()) {
                partnerCertificateContainerConfigHandler.update(partnerConfig.getPartnerId(), certificateContainer);
            }
        }
        if (CollectionUtils.isNotEmpty(partnerConfig.getConnections())) {
            for (ConnectionConfig connection : partnerConfig.getConnections()) {
                partnerConnectionConfigHandler.update(partnerConfig.getPartnerId(), connection);
            }
        }
    }

    @Override
    public void exportConfig(ConfigExportContext configExportContext) {
        partnerRepository.findAll().stream()
                .map(partnerConfigMapper::toConfig)
                .forEach(partnerConfig ->
                        configExportContext.getConfig().getPartners().add(partnerConfig));
    }

    @Override
    public void sort(NexusConfig nexusConfig) {
        nexusConfig.getPartners().sort(Comparator.comparing(PartnerConfig::getPartnerId));
        nexusConfig.getPartners().forEach(this::sort);
    }

    @Override
    public void sort(PartnerConfig config) {
        config.getConnections().sort(Comparator.comparing(ConnectionConfig::getConnectionId));
        config.getConnections().forEach(connection -> {
            connection.getParameters().sort(Comparator.comparing(ParameterConfig::getName));
        });
        config.getCertificateContainers().sort(Comparator.comparing(CertificateContainerConfig::getName));
        config.getCertificateContainers().forEach(certificateContainer -> {
            certificateContainer.getCertificates().sort(Comparator.comparing(CertificateConfig::getName));
        });
    }

    @Override
    public List<PartnerConfig> merge(Collection<PartnerConfig> oldConfigs, Collection<PartnerConfig> newConfigs) {
        return defaultMerge(oldConfigs, newConfigs, PartnerConfig::getPartnerId, partnerConfigMapper::merge);
    }
}
