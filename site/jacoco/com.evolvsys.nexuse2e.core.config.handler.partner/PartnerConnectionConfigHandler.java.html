package com.evolvsys.nexuse2e.core.config.handler.partner;

import com.evolvsys.nexuse2e.core.certificate.model.CertificateContainerEntity;
import com.evolvsys.nexuse2e.core.certificate.repository.CertificateContainerRepository;
import com.evolvsys.nexuse2e.core.config.handler.mapper.PartnerConnectionConfigMapper;
import com.evolvsys.nexuse2e.core.config.model.ConnectionConfig;
import com.evolvsys.nexuse2e.core.exception.NexusError;
import com.evolvsys.nexuse2e.core.exception.NexusRuntimeException;
import com.evolvsys.nexuse2e.core.model.entity.PartnerConnectionEntity;
import com.evolvsys.nexuse2e.core.model.entity.PartnerEntity;
import com.evolvsys.nexuse2e.core.partner.repository.PartnerRepository;
import java.util.List;
import lombok.RequiredArgsConstructor;
import lombok.extern.slf4j.Slf4j;
import org.apache.commons.lang3.StringUtils;
import org.springframework.stereotype.Component;
import org.springframework.transaction.annotation.Transactional;

/**
 * @author SvenBarden
 * @since 10.04.2024
 */
@Component
@RequiredArgsConstructor
@Slf4j
class PartnerConnectionConfigHandler {
    private final PartnerRepository partnerRepository;
    private final PartnerConnectionConfigMapper partnerConnectionConfigMapper;
    private final CertificateContainerRepository certificateContainerRepository;

    protected PartnerConnectionEntity create(PartnerEntity partnerEntity) {
        LOGGER.debug("creating new connection for partner {}", partnerEntity.getPartnerId());
        PartnerConnectionEntity partnerConnectionEntity = new PartnerConnectionEntity();
        partnerConnectionEntity.setPartner(partnerEntity);
        return partnerConnectionEntity;
    }

    @Transactional
    public void update(String partnerId, ConnectionConfig connectionConfig) {
        LOGGER.trace("handling connection {}", connectionConfig);

        PartnerEntity partnerEntity = partnerRepository
                .findByPartnerId(partnerId)
                .orElseThrow(() -> new NexusRuntimeException(NexusError.builder()
                        .message("Partner for connection not found")
                        .extension("partnerId", partnerId)
                        .build()));

        PartnerConnectionEntity partnerConnectionEntity = partnerEntity.getConnections().stream()
                .filter(c -> c.getConnectionId().equals(connectionConfig.getConnectionId()))
                .findFirst()
                .orElseGet(() -> create(partnerEntity));

        partnerConnectionConfigMapper.updateEntity(connectionConfig, partnerConnectionEntity);

        if (partnerConnectionEntity.getId() == null) {
            partnerEntity.getConnections().add(partnerConnectionEntity);
        }

        List<CertificateContainerEntity> certificateContainerEntities =
                certificateContainerRepository.findByPartner(partnerEntity);

        if (StringUtils.isNotBlank(connectionConfig.getCertificateContainer())) {
            CertificateContainerEntity containerEntity = certificateContainerEntities.stream()
                    .filter(certificateContainerEntity ->
                            certificateContainerEntity.getName().equals(connectionConfig.getCertificateContainer()))
                    .findFirst()
                    .orElseThrow(() -> new NexusRuntimeException(NexusError.builder()
                            .message("No certificate container for partner connection found")
                            .extension("container", connectionConfig.getCertificateContainer())
                            .extension("partner", partnerId)
                            .build()));
            partnerConnectionEntity.setCertificateContainer(containerEntity);
        }

        partnerRepository.save(partnerEntity);
    }
}
