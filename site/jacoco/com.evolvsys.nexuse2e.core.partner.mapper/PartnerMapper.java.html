package com.evolvsys.nexuse2e.core.partner.mapper;

import com.evolvsys.app.core.auth.model.CurrentUserData;
import com.evolvsys.app.core.mapper.BaseDtoEntityMapper;
import com.evolvsys.app.core.mapper.DefaultMapperConfiguration;
import com.evolvsys.app.core.mapper.interceptor.InterceptingMappingContext;
import com.evolvsys.app.core.persistence.model.IdDto;
import com.evolvsys.nexuse2e.core.model.entity.PartnerConnectionEntity;
import com.evolvsys.nexuse2e.core.model.entity.PartnerEntity;
import com.evolvsys.nexuse2e.core.partner.model.PartnerConnectionDto;
import com.evolvsys.nexuse2e.core.partner.model.PartnerDto;
import com.evolvsys.nexuse2e.core.partner.repository.PartnerRepository;
import java.util.List;
import java.util.Objects;
import java.util.Set;
import java.util.stream.Collectors;
import org.mapstruct.*;
import org.springframework.beans.factory.annotation.Autowired;

/**
 * <p>{@link Mapper} f√ºr {@link PartnerEntity} und {@link PartnerDto}.</p>
 *
 * @author Florian Drees
 * @since 02.01.2024
 */
@Mapper(config = DefaultMapperConfiguration.class)
public abstract class PartnerMapper extends BaseDtoEntityMapper<PartnerDto, PartnerEntity> {
    private PartnerRepository partnerRepository;
    private PartnerConnectionMapper partnerConnectionMapper;

    @Autowired
    protected void config(PartnerRepository partnerRepository, PartnerConnectionMapper partnerConnectionMapper) {
        this.partnerRepository = partnerRepository;
        this.partnerConnectionMapper = partnerConnectionMapper;
    }

    @Override
    public PartnerEntity idToEntity(Long id, @Context InterceptingMappingContext interceptingMappingContext) {
        return idToEntity(id, interceptingMappingContext, partnerRepository::findById);
    }

    @InheritConfiguration
    @Override
    protected abstract PartnerDto entityToDto(
            PartnerEntity entity, @Context InterceptingMappingContext interceptingMappingContext);

    @Mapping(target = "connections", ignore = true)
    @InheritConfiguration
    @Override
    protected abstract void updateEntity(
            PartnerDto dto,
            @MappingTarget PartnerEntity entity,
            @Context InterceptingMappingContext interceptingMappingContext);

    public void updateConnections(
            List<PartnerConnectionDto> connections, PartnerEntity entity, CurrentUserData currentUserData) {
        Set<Long> connectionIds =
                connections.stream().map(IdDto::getId).filter(Objects::nonNull).collect(Collectors.toSet());

        entity.getConnections().removeIf(connectionEntity -> !connectionIds.contains(connectionEntity.getId()));

        for (PartnerConnectionDto partnerConnectionDto : connections) {
            PartnerConnectionEntity partnerConnectionEntity = entity.getConnections().stream()
                    .filter(c -> c.getId() != null && c.getId().equals(partnerConnectionDto.getId()))
                    .findFirst()
                    .orElseGet(PartnerConnectionEntity::new);

            if (partnerConnectionEntity.getId() == null) {
                entity.getConnections().add(partnerConnectionEntity);
            }

            partnerConnectionMapper.updateEntity(partnerConnectionDto, partnerConnectionEntity, currentUserData);
            partnerConnectionEntity.setPartner(entity);
        }
    }
}
