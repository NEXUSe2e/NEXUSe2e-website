package com.evolvsys.nexuse2e.core.partner.mapper;

import com.evolvsys.app.core.mapper.BaseDtoEntityMapper;
import com.evolvsys.app.core.mapper.DefaultMapperConfiguration;
import com.evolvsys.app.core.mapper.interceptor.InterceptingMappingContext;
import com.evolvsys.nexuse2e.core.model.entity.PartnerEntity;
import com.evolvsys.nexuse2e.core.partner.model.PartnerDto;
import com.evolvsys.nexuse2e.core.partner.repository.PartnerRepository;
import org.mapstruct.*;
import org.springframework.beans.factory.annotation.Autowired;

/**
 * @author Florian Drees
 * @since 29.01.2024
 */
@Mapper(config = DefaultMapperConfiguration.class)
public abstract class BasePartnerMapper extends BaseDtoEntityMapper<PartnerDto, PartnerEntity> {
    private PartnerRepository partnerRepository;

    @Autowired
    protected void config(PartnerRepository partnerRepository) {
        this.partnerRepository = partnerRepository;
    }

    @Override
    public PartnerEntity idToEntity(Long id, @Context InterceptingMappingContext interceptingMappingContext) {
        return idToEntity(id, interceptingMappingContext, partnerRepository::findById);
    }

    @InheritConfiguration
    @Override
    protected abstract PartnerDto entityToDto(
            PartnerEntity entity, @Context InterceptingMappingContext interceptingMappingContext);

    @Mapping(target = "connections", ignore = true)
    @InheritConfiguration
    @Override
    protected abstract void updateEntity(
            PartnerDto dto,
            @MappingTarget PartnerEntity entity,
            @Context InterceptingMappingContext interceptingMappingContext);
}
