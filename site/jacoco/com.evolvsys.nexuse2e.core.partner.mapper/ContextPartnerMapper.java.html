package com.evolvsys.nexuse2e.core.partner.mapper;

import com.evolvsys.app.core.mapper.DefaultMapperConfiguration;
import com.evolvsys.nexuse2e.core.model.entity.PartnerEntity;
import com.evolvsys.nexuse2e.core.partner.model.ContextPartner;
import com.evolvsys.nexuse2e.core.partner.repository.PartnerRepository;
import org.apache.commons.lang3.StringUtils;
import org.mapstruct.Mapper;
import org.springframework.beans.factory.annotation.Autowired;

/**
 * @author SvenBarden
 * @since 24.01.2024
 */
@Mapper(config = DefaultMapperConfiguration.class)
public abstract class ContextPartnerMapper {
    private PartnerRepository partnerRepository;

    @Autowired
    protected void config(PartnerRepository partnerRepository) {
        this.partnerRepository = partnerRepository;
    }

    public abstract ContextPartner entityToPartner(PartnerEntity partnerEntity);

    public PartnerEntity stringToPartner(String name) {
        return StringUtils.isBlank(name)
                ? null
                : partnerRepository.findByPartnerId(name).orElse(null);
    }
}
