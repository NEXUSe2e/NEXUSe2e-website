package com.evolvsys.nexuse2e.core.partner.mapper;

import com.evolvsys.app.core.mapper.BaseDtoEntityMapper;
import com.evolvsys.app.core.mapper.DefaultMapperConfiguration;
import com.evolvsys.app.core.mapper.interceptor.InterceptingMappingContext;
import com.evolvsys.nexuse2e.core.certificate.mapper.CertificateContainerMapper;
import com.evolvsys.nexuse2e.core.model.entity.PartnerConnectionEntity;
import com.evolvsys.nexuse2e.core.parameter.mapper.ParameterMapper;
import com.evolvsys.nexuse2e.core.partner.model.PartnerConnectionDto;
import com.evolvsys.nexuse2e.core.partner.repository.PartnerConnectionRepository;
import com.evolvsys.nexuse2e.core.service.mapper.ServiceMapper;
import org.mapstruct.*;
import org.springframework.beans.factory.annotation.Autowired;

/**
 * <p>Mapper f√ºr {@link PartnerConnectionEntity} und {@link PartnerConnectionDto}.</p>
 *
 * @author Florian Drees
 * @since 22.12.2023
 */
@Mapper(
        config = DefaultMapperConfiguration.class,
        uses = {ParameterMapper.class, BasePartnerMapper.class, CertificateContainerMapper.class, ServiceMapper.class})
public abstract class PartnerConnectionMapper
        extends BaseDtoEntityMapper<PartnerConnectionDto, PartnerConnectionEntity> {
    private PartnerConnectionRepository partnerConnectionRepository;

    @Autowired
    protected void config(PartnerConnectionRepository partnerConnectionRepository) {
        this.partnerConnectionRepository = partnerConnectionRepository;
    }

    @Override
    public PartnerConnectionEntity idToEntity(Long id, @Context InterceptingMappingContext interceptingMappingContext) {
        return idToEntity(id, interceptingMappingContext, partnerConnectionRepository::findById);
    }

    @Mapping(target = "certificateContainerId", source = "certificateContainer.id")
    @Mapping(target = "parameters", ignore = true)
    @Mapping(target = "partnerId", source = "partner.id")
    @Mapping(target = "outflowServiceId", source = "outflowService.id")
    @Mapping(target = "requeueIntervals", ignore = true)
    @InheritConfiguration
    @Override
    protected abstract PartnerConnectionDto entityToDto(
            PartnerConnectionEntity entity, @Context InterceptingMappingContext interceptingMappingContext);

    @Mapping(target = "certificateContainer", source = "certificateContainerId")
    @Mapping(target = "partner", source = "partnerId")
    @Mapping(target = "outflowService", source = "outflowServiceId")
    @InheritConfiguration
    @Override
    protected abstract void updateEntity(
            PartnerConnectionDto dto,
            @MappingTarget PartnerConnectionEntity entity,
            @Context InterceptingMappingContext interceptingMappingContext);
}
