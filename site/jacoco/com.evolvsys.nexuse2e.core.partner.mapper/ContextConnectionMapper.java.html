package com.evolvsys.nexuse2e.core.partner.mapper;

import com.evolvsys.app.core.mapper.DefaultMapperConfiguration;
import com.evolvsys.nexuse2e.core.model.entity.PartnerConnectionEntity;
import com.evolvsys.nexuse2e.core.partner.model.ContextConnection;
import com.evolvsys.nexuse2e.core.partner.repository.PartnerConnectionRepository;
import java.time.Duration;
import java.util.*;
import java.util.stream.Collectors;
import lombok.extern.slf4j.Slf4j;
import org.apache.commons.collections4.CollectionUtils;
import org.apache.commons.lang3.StringUtils;
import org.mapstruct.Mapper;
import org.mapstruct.Mapping;
import org.springframework.beans.factory.annotation.Autowired;

/**
 * @author SvenBarden
 * @since 30.01.2024
 */
@Mapper(config = DefaultMapperConfiguration.class)
@Slf4j
public abstract class ContextConnectionMapper {
    private PartnerConnectionRepository partnerConnectionRepository;

    @Autowired
    protected void config(PartnerConnectionRepository partnerConnectionRepository) {
        this.partnerConnectionRepository = partnerConnectionRepository;
    }

    public PartnerConnectionEntity connectionToEntity(ContextConnection connection) {
        return connection == null
                ? null
                : partnerConnectionRepository
                        .findByConnectionIdAndPartner_PartnerId(connection.getConnectionId(), connection.getPartnerId())
                        .orElse(null);
    }

    @Mapping(target = "partnerId", source = "partner.partnerId")
    @Mapping(target = "outflowService", source = "outflowService.serviceId")
    public abstract ContextConnection entityToConnection(PartnerConnectionEntity partnerConnectionEntity);

    protected List<Duration> secondsToDuration(Collection<Long> collection) {
        return CollectionUtils.isEmpty(collection)
                ? Collections.emptyList()
                : collection.stream().map(Duration::ofSeconds).toList();
    }

    protected Set<String> stringToList(String s) {
        return s == null
                ? Collections.emptySet()
                : Arrays.stream(s.split("\\s*,\\s*"))
                        .filter(StringUtils::isNotBlank)
                        .collect(Collectors.toSet());
    }
}
