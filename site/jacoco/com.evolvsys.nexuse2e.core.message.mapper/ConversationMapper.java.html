package com.evolvsys.nexuse2e.core.message.mapper;

import com.evolvsys.app.core.mapper.BaseDtoEntityMapper;
import com.evolvsys.app.core.mapper.DefaultMapperConfiguration;
import com.evolvsys.app.core.mapper.interceptor.InterceptingMappingContext;
import com.evolvsys.nexuse2e.core.action.mapper.ActionMapper;
import com.evolvsys.nexuse2e.core.choreography.mapper.ChoreographyMapper;
import com.evolvsys.nexuse2e.core.message.model.ConversationDto;
import com.evolvsys.nexuse2e.core.message.repository.ConversationRepository;
import com.evolvsys.nexuse2e.core.model.entity.ConversationEntity;
import com.evolvsys.nexuse2e.core.partner.mapper.PartnerMapper;
import org.mapstruct.*;
import org.springframework.beans.factory.annotation.Autowired;

/**
 * @author Florian Drees
 * @since 27.12.2023
 */
@Mapper(
        config = DefaultMapperConfiguration.class,
        uses = {ActionMapper.class, ChoreographyMapper.class, PartnerMapper.class})
public abstract class ConversationMapper extends BaseDtoEntityMapper<ConversationDto, ConversationEntity> {
    private ConversationRepository conversationRepository;

    @Autowired
    protected void config(ConversationRepository conversationRepository) {
        this.conversationRepository = conversationRepository;
    }

    @Override
    public ConversationEntity idToEntity(Long id, InterceptingMappingContext interceptingMappingContext) {
        return idToEntity(id, interceptingMappingContext, conversationRepository::findById);
    }

    @Mapping(target = "lastTargetPartnerId", source = "lastTargetPartner.id")
    @Mapping(target = "lastSourcePartnerId", source = "lastSourcePartner.id")
    @Mapping(target = "currentActionId", source = "currentAction.id")
    @Mapping(target = "choreographyId", source = "choreography.id")
    @InheritConfiguration
    @Override
    protected abstract ConversationDto entityToDto(
            ConversationEntity entity, @Context InterceptingMappingContext interceptingMappingContext);

    @Mapping(target = "lastTargetPartner", source = "lastTargetPartnerId")
    @Mapping(target = "lastSourcePartner", source = "lastSourcePartnerId")
    @Mapping(target = "currentAction", source = "currentActionId")
    @Mapping(target = "choreography", source = "choreographyId")
    @Mapping(target = "created", ignore = true)
    @Mapping(target = "lastUpdated", ignore = true)
    @InheritConfiguration
    @Override
    protected abstract void updateEntity(
            ConversationDto dto,
            @MappingTarget ConversationEntity entity,
            @Context InterceptingMappingContext interceptingMappingContext);
}
