package com.evolvsys.nexuse2e.core.message.mapper;

import com.evolvsys.app.core.mapper.DefaultMapperConfiguration;
import com.evolvsys.nexuse2e.core.action.mapper.ContextActionMapper;
import com.evolvsys.nexuse2e.core.choreography.mapper.ContextChoreographyMapper;
import com.evolvsys.nexuse2e.core.message.model.ContextConversation;
import com.evolvsys.nexuse2e.core.message.repository.ConversationRepository;
import com.evolvsys.nexuse2e.core.model.entity.ConversationEntity;
import org.mapstruct.AfterMapping;
import org.mapstruct.Mapper;
import org.mapstruct.Mapping;
import org.mapstruct.MappingTarget;
import org.springframework.beans.factory.annotation.Autowired;

/**
 * @author SvenBarden
 * @since 24.01.2024
 */
@Mapper(
        config = DefaultMapperConfiguration.class,
        uses = {ContextActionMapper.class, ContextChoreographyMapper.class})
public abstract class ContextConversationMapper {
    private ConversationRepository conversationRepository;
    private ContextActionMapper contextActionMapper;

    @Autowired
    protected void config(ConversationRepository conversationRepository, ContextActionMapper contextActionMapper) {
        this.conversationRepository = conversationRepository;
        this.contextActionMapper = contextActionMapper;
    }

    public ConversationEntity stringToEntity(String conversationName) {
        return conversationRepository.findByConversationId(conversationName).orElse(null);
    }

    @Mapping(target = "choreographyId", source = "choreography.choreographyId")
    @Mapping(target = "currentAction", source = "currentAction.actionId")
    public abstract ContextConversation entityToConversation(ConversationEntity conversationEntity);

    @Mapping(target = "lastTargetPartner", ignore = true)
    @Mapping(target = "lastSourcePartner", ignore = true)
    @Mapping(target = "lastDirection", ignore = true)
    @Mapping(target = "version", ignore = true)
    @Mapping(target = "lastUpdated", ignore = true)
    @Mapping(target = "id", ignore = true)
    @Mapping(target = "status", ignore = true)
    @Mapping(target = "finished", ignore = true)
    @Mapping(target = "created", ignore = true)
    @Mapping(target = "choreography", source = "choreographyId")
    @Mapping(target = "acknowledgmentStatus", ignore = true)
    @Mapping(target = "currentAction", ignore = true)
    public abstract void updateEntity(
            ContextConversation conversation, @MappingTarget ConversationEntity conversationEntity);

    @AfterMapping
    protected void afterUpdateEntity(
            ContextConversation conversation, @MappingTarget ConversationEntity conversationEntity) {
        conversationEntity.setCurrentAction(
                contextActionMapper.namesToEntity(conversation.getChoreographyId(), conversation.getCurrentAction()));
    }
}
