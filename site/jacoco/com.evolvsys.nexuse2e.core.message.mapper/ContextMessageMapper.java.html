package com.evolvsys.nexuse2e.core.message.mapper;

import com.evolvsys.app.core.mapper.DefaultMapperConfiguration;
import com.evolvsys.nexuse2e.core.action.mapper.ContextActionMapper;
import com.evolvsys.nexuse2e.core.message.context.MessageContext;
import com.evolvsys.nexuse2e.core.message.repository.MessageRepository;
import com.evolvsys.nexuse2e.core.model.MessageDirection;
import com.evolvsys.nexuse2e.core.model.entity.MessageEntity;
import com.evolvsys.nexuse2e.core.partner.mapper.ContextConnectionMapper;
import com.evolvsys.nexuse2e.core.partner.mapper.ContextPartnerMapper;
import com.evolvsys.nexuse2e.internal.ContextPayload;
import com.evolvsys.nexuse2e.internal.Message;
import java.time.ZonedDateTime;
import org.apache.commons.lang3.StringUtils;
import org.mapstruct.*;
import org.springframework.beans.factory.annotation.Autowired;

/**
 * @author SvenBarden
 * @since 30.01.2024
 */
@Mapper(
        config = DefaultMapperConfiguration.class,
        uses = {
            ContextConversationMapper.class,
            ContextMetaDataMapper.class,
            ContextConnectionMapper.class,
            ContextActionMapper.class,
            ContextPayloadMapper.class,
            ContextPartnerMapper.class
        },
        imports = ZonedDateTime.class)
public abstract class ContextMessageMapper {
    private static final String UPDATE_ONLY_CONDITION = "updateOnly";
    private MessageRepository messageRepository;

    @Autowired
    protected void config(MessageRepository messageRepository) {
        this.messageRepository = messageRepository;
    }

    public MessageEntity stringToEntity(String messageId) {
        return messageRepository.findByMessageId(messageId).orElse(null);
    }

    protected MessageDirection calculateDirection(MessageContext messageContext) {
        return switch (messageContext.getSourcePartner().getPartnerType()) {
            case EXTERN ->
                switch (messageContext.getTargetPartner().getPartnerType()) {
                    case EXTERN -> MessageDirection.EXTERN_TO_EXTERN;
                    case INTERN -> MessageDirection.EXTERN_TO_INTERN;
                };
            case INTERN ->
                switch (messageContext.getTargetPartner().getPartnerType()) {
                    case EXTERN -> MessageDirection.INTERN_TO_EXTERN;
                    case INTERN -> MessageDirection.INTERN_TO_INTERN;
                };
        };
    }

    @Condition
    @Named(UPDATE_ONLY_CONDITION)
    protected boolean updateOnlyCondition(@MappingTarget MessageEntity messageEntity) {
        return messageEntity.getId() == null;
    }

    @Mapping(target = "finished", ignore = true)
    @Mapping(target = "nextRequeue", ignore = true)
    @Mapping(target = "maxRetries", ignore = true)
    @Mapping(target = "targetPartner", ignore = true)
    @Mapping(target = "sourcePartner", ignore = true)
    @Mapping(target = "outboundAcknowledgment", ignore = true)
    @Mapping(target = "id", ignore = true)
    @Mapping(target = "version", ignore = true)
    @Mapping(target = "status", source = "messageStatus")
    @Mapping(target = "sourceComponent", ignore = true)
    @Mapping(target = "sent", ignore = true)
    @Mapping(target = "referenceMessage", source = "referenceMessageId")
    @Mapping(target = "metaDatas", source = "metaData")
    @Mapping(
            target = "messageId",
            source = "id",
            conditionQualifiedByName = UPDATE_ONLY_CONDITION,
            nullValuePropertyMappingStrategy = NullValuePropertyMappingStrategy.IGNORE)
    @Mapping(target = "lastUpdated", ignore = true)
    @Mapping(target = "lastTry", ignore = true)
    @Mapping(
            target = "conversation",
            source = "conversationId") // TODO the messageQueue test fails if the condition here returns false at any
    // point, needs further investigation
    @Mapping(target = "component", ignore = true)
    @Mapping(target = "action", ignore = true)
    @Mapping(target = "created", ignore = true)
    public abstract void updateEntity(Message message, @MappingTarget MessageEntity messageEntity);

    @Mapping(target = "payloads", ignore = true)
    @Mapping(target = "finished", ignore = true)
    @Mapping(target = "nextRequeue", ignore = true)
    @Mapping(target = "maxRetries", ignore = true)
    @Mapping(target = "outboundAcknowledgment", ignore = true)
    @Mapping(
            target = "targetPartner",
            source = "context.targetPartnerId",
            conditionQualifiedByName = UPDATE_ONLY_CONDITION,
            nullValuePropertyMappingStrategy = NullValuePropertyMappingStrategy.IGNORE)
    @Mapping(
            target = "sourcePartner",
            source = "context.sourcePartnerId",
            conditionQualifiedByName = UPDATE_ONLY_CONDITION,
            nullValuePropertyMappingStrategy = NullValuePropertyMappingStrategy.IGNORE)
    @Mapping(target = "id", ignore = true)
    @Mapping(target = "version", ignore = true)
    @Mapping(target = "status", source = "message.messageStatus")
    @Mapping(target = "sourceComponent", ignore = true)
    @Mapping(target = "sent", ignore = true)
    @Mapping(target = "referenceMessage", source = "message.referenceMessageId")
    @Mapping(target = "metaDatas", source = "message.metaData")
    @Mapping(
            target = "messageId",
            source = "message.id",
            conditionQualifiedByName = UPDATE_ONLY_CONDITION,
            nullValuePropertyMappingStrategy = NullValuePropertyMappingStrategy.IGNORE)
    @Mapping(target = "lastUpdated", ignore = true)
    @Mapping(target = "lastTry", ignore = true)
    @Mapping(
            target = "conversation",
            source =
                    "message.conversationId") // TODO the messageQueue test fails if the condition here returns false at
    // any point, needs further investigation
    @Mapping(target = "component", ignore = true)
    @Mapping(
            target = "direction",
            source = "message.direction",
            defaultExpression = "java(calculateDirection(context))")
    @Mapping(
            target = "action",
            conditionQualifiedByName = UPDATE_ONLY_CONDITION,
            nullValuePropertyMappingStrategy = NullValuePropertyMappingStrategy.IGNORE)
    @Mapping(target = "created", ignore = true)
    public abstract void updateEntity(
            Message message, MessageContext context, @MappingTarget MessageEntity messageEntity);

    @InheritConfiguration
    @Mapping(target = "payloads", source = "message.payloads")
    public abstract void updateEntityWithPayloads(
            Message message, MessageContext context, @MappingTarget MessageEntity messageEntity);

    @InheritConfiguration(name = "entityToMessage")
    public abstract void updateMessage(MessageEntity messageEntity, @MappingTarget Message message);

    @Mapping(target = "id", source = "messageId")
    @Mapping(target = "referenceMessageId", source = "referenceMessage.messageId")
    @Mapping(target = "metaData", source = "metaDatas")
    @Mapping(target = "messageStatus", source = "status")
    @Mapping(target = "conversationId", source = "conversation.conversationId")
    public abstract Message entityToMessage(MessageEntity messageEntity);

    @Mapping(target = "newContext", ignore = true)
    @Mapping(target = "clientCertificate", ignore = true)
    @Mapping(target = "route", ignore = true)
    @Mapping(target = "targetPartnerId", source = "targetPartner.partnerId")
    @Mapping(target = "sourcePartnerId", source = "sourcePartner.partnerId")
    @Mapping(target = "sourcePartner", ignore = true)
    @Mapping(target = "sourceMessage", ignore = true)
    @Mapping(target = "sourceData", ignore = true)
    @Mapping(target = "sender", ignore = true)
    @Mapping(target = "returnMessage", ignore = true)
    @Mapping(target = "returnData", ignore = true)
    @Mapping(target = "messageErrors", ignore = true)
    @Mapping(target = "errorMessage", ignore = true)
    @Mapping(target = "errorData", ignore = true)
    @Mapping(target = "doReturnPipelines", ignore = true)
    @Mapping(target = "choreographyId", source = "conversation.choreography.choreographyId")
    @Mapping(target = "actionId", source = "action.actionId")
    @Mapping(target = "acknowledgmentMessage", ignore = true)
    @Mapping(target = "acknowledgmentData", ignore = true)
    public abstract void updateContext(MessageEntity messageEntity, @MappingTarget MessageContext messageContext);

    @BeforeMapping
    protected void beforeUpdateEntity(
            Message message, MessageContext context, @MappingTarget MessageEntity messageEntity) {
        for (ContextPayload payload : message.getPayloads()) {
            if (StringUtils.isBlank(payload.getContentId())) {
                payload.setContentId(message.getId() + "_" + payload.getSequenceNumber());
            }
        }
    }
}
