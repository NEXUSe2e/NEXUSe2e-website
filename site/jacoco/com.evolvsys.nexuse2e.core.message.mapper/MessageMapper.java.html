package com.evolvsys.nexuse2e.core.message.mapper;

import com.evolvsys.app.core.mapper.BaseDtoEntityMapper;
import com.evolvsys.app.core.mapper.DefaultMapperConfiguration;
import com.evolvsys.app.core.mapper.interceptor.InterceptingMappingContext;
import com.evolvsys.nexuse2e.core.action.mapper.ActionMapper;
import com.evolvsys.nexuse2e.core.message.model.MessageDto;
import com.evolvsys.nexuse2e.core.message.repository.MessageRepository;
import com.evolvsys.nexuse2e.core.model.entity.MessageEntity;
import org.mapstruct.*;
import org.springframework.beans.factory.annotation.Autowired;

/**
 * @author Florian Drees
 * @since 27.12.2023
 */
@Mapper(config = DefaultMapperConfiguration.class, uses = ActionMapper.class)
public abstract class MessageMapper extends BaseDtoEntityMapper<MessageDto, MessageEntity> {
    private MessageRepository messageRepository;

    @Autowired
    protected void config(MessageRepository messageRepository) {
        this.messageRepository = messageRepository;
    }

    @Override
    public MessageEntity idToEntity(Long id, @Context InterceptingMappingContext interceptingMappingContext) {
        return idToEntity(id, interceptingMappingContext, messageRepository::findById);
    }

    @Mapping(target = "targetPartnerId", source = "targetPartner.id")
    @Mapping(target = "sourcePartnerId", source = "sourcePartner.id")
    @Mapping(target = "referenceMessageId", source = "referenceMessage.id")
    @Mapping(target = "conversationId", source = "conversation.id")
    @Mapping(target = "actionId", source = "action.id")
    @InheritConfiguration
    @Override
    protected abstract MessageDto entityToDto(
            MessageEntity entity, @Context InterceptingMappingContext interceptingMappingContext);

    @Mapping(target = "targetPartner", ignore = true)
    @Mapping(target = "sourcePartner", ignore = true)
    @Mapping(target = "referenceMessage", source = "referenceMessageId")
    @Mapping(target = "component", ignore = true)
    @Mapping(target = "metaDatas", ignore = true)
    @Mapping(target = "sourceComponent", ignore = true)
    @Mapping(target = "payloads", ignore = true)
    @Mapping(target = "lastUpdated", ignore = true)
    @Mapping(target = "created", ignore = true)
    @Mapping(target = "conversation", ignore = true)
    @Mapping(target = "action", source = "actionId")
    @InheritConfiguration
    @Override
    protected abstract void updateEntity(
            MessageDto dto,
            @MappingTarget MessageEntity entity,
            @Context InterceptingMappingContext interceptingMappingContext);
}
