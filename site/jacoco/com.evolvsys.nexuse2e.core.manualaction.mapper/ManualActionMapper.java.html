package com.evolvsys.nexuse2e.core.manualaction.mapper;

import com.evolvsys.app.core.mapper.BaseDtoEntityMapper;
import com.evolvsys.app.core.mapper.DefaultMapperConfiguration;
import com.evolvsys.app.core.mapper.interceptor.InterceptingMappingContext;
import com.evolvsys.app.core.user.mapper.UserMapper;
import com.evolvsys.nexuse2e.core.manualaction.model.*;
import org.apache.commons.lang3.NotImplementedException;
import org.hibernate.Hibernate;
import org.mapstruct.*;

/**
 * @author SvenBarden
 * @since 05.08.2024
 */
@Mapper(
        config = DefaultMapperConfiguration.class,
        uses = {
            MessageActionManualActionMapper.class,
            ConfigImportManualActionMapper.class,
            PurgeManualActionMapper.class,
            UserMapper.class
        },
        subclassExhaustiveStrategy = SubclassExhaustiveStrategy.RUNTIME_EXCEPTION)
public abstract class ManualActionMapper extends BaseDtoEntityMapper<ManualActionDto, ManualActionEntity> {

    @Override
    public ManualActionEntity idToEntity(Long id, InterceptingMappingContext interceptingMappingContext) {
        throw new NotImplementedException();
    }

    @Override
    protected ManualActionDto entityToDto(
            ManualActionEntity entity, @Context InterceptingMappingContext interceptingMappingContext) {
        return internalEntityToDto(Hibernate.unproxy(entity, ManualActionEntity.class), interceptingMappingContext);
    }

    @InheritConfiguration
    @ManualActionToDto
    @SubclassMapping(source = MessageActionManualActionEntity.class, target = MessageActionManualActionDto.class)
    @SubclassMapping(source = ConfigImportManualActionEntity.class, target = ConfigImportManualActionDto.class)
    @SubclassMapping(source = PurgeManualActionEntity.class, target = PurgeManualActionDto.class)
    protected abstract ManualActionDto internalEntityToDto(
            ManualActionEntity entity, @Context InterceptingMappingContext interceptingMappingContext);

    @InheritConfiguration
    @ManualActionToEntity
    @Override
    protected abstract void updateEntity(
            ManualActionDto dto,
            @MappingTarget ManualActionEntity entity,
            @Context InterceptingMappingContext interceptingMappingContext);
}
