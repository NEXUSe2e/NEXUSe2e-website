package com.evolvsys.nexuse2e.core.manualaction.mapper;

import com.evolvsys.app.core.file.mapper.FileMapper;
import com.evolvsys.app.core.mapper.BaseDtoEntityMapper;
import com.evolvsys.app.core.mapper.DefaultMapperConfiguration;
import com.evolvsys.app.core.mapper.interceptor.InterceptingMappingContext;
import com.evolvsys.app.core.user.mapper.UserMapper;
import com.evolvsys.nexuse2e.core.manualaction.model.ConfigImportManualActionDto;
import com.evolvsys.nexuse2e.core.manualaction.model.ConfigImportManualActionEntity;
import com.evolvsys.nexuse2e.core.manualaction.repository.ConfigImportManualActionRepository;
import com.evolvsys.nexuse2e.core.manualaction.service.ConfigImportNewFileOwnerProvider;
import com.evolvsys.nexuse2e.core.manualaction.service.ConfigImportOldFileOwnerProvider;
import org.mapstruct.*;
import org.springframework.beans.factory.annotation.Autowired;

/**
 * @author SvenBarden
 * @since 05.08.2024
 */
@Mapper(
        config = DefaultMapperConfiguration.class,
        uses = {UserMapper.class})
public abstract class ConfigImportManualActionMapper
        extends BaseDtoEntityMapper<ConfigImportManualActionDto, ConfigImportManualActionEntity> {
    private ConfigImportManualActionRepository configImportManualActionRepository;
    private FileMapper fileMapper;

    @Autowired
    protected void config(
            ConfigImportManualActionRepository configImportManualActionRepository, FileMapper fileMapper) {
        this.configImportManualActionRepository = configImportManualActionRepository;
        this.fileMapper = fileMapper;
    }

    @Override
    public ConfigImportManualActionEntity idToEntity(
            Long id, @Context InterceptingMappingContext interceptingMappingContext) {
        return idToEntity(id, interceptingMappingContext, configImportManualActionRepository::findById);
    }

    @InheritConfiguration
    @ManualActionToDto
    @Mapping(target = "oldConfig", ignore = true)
    @Mapping(target = "newConfig", ignore = true)
    @Override
    protected abstract ConfigImportManualActionDto entityToDto(
            ConfigImportManualActionEntity entity, @Context InterceptingMappingContext interceptingMappingContext);

    @InheritConfiguration
    @ManualActionToEntity
    @Mapping(target = "oldConfig", ignore = true)
    @Mapping(target = "newConfig", ignore = true)
    @Override
    protected abstract void updateEntity(
            ConfigImportManualActionDto dto,
            @MappingTarget ConfigImportManualActionEntity entity,
            @Context InterceptingMappingContext interceptingMappingContext);

    @AfterMapping
    protected void fileToDto(ConfigImportManualActionEntity entity, @MappingTarget ConfigImportManualActionDto dto) {
        if (entity.getOldConfig() != null) {
            dto.setOldConfig(fileMapper.entityToDto(
                    entity.getOldConfig(), ConfigImportOldFileOwnerProvider.OLD_CONFIG, entity.getId()));
        }
        if (entity.getNewConfig() != null) {
            dto.setNewConfig(fileMapper.entityToDto(
                    entity.getNewConfig(), ConfigImportNewFileOwnerProvider.NEW_CONFIG, entity.getId()));
        }
    }

    @AfterMapping
    protected void fileToEntity(ConfigImportManualActionDto dto, @MappingTarget ConfigImportManualActionEntity entity) {
        fileMapper.dtoToEntity(dto.getOldConfig(), entity::getOldConfig, entity::setOldConfig);
        fileMapper.dtoToEntity(dto.getNewConfig(), entity::getNewConfig, entity::setNewConfig);
    }
}
