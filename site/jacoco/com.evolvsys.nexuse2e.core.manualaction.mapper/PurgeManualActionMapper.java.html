package com.evolvsys.nexuse2e.core.manualaction.mapper;

import com.evolvsys.app.core.mapper.BaseDtoEntityMapper;
import com.evolvsys.app.core.mapper.DefaultMapperConfiguration;
import com.evolvsys.app.core.mapper.interceptor.InterceptingMappingContext;
import com.evolvsys.app.core.user.mapper.UserMapper;
import com.evolvsys.nexuse2e.core.manualaction.model.PurgeManualActionDto;
import com.evolvsys.nexuse2e.core.manualaction.model.PurgeManualActionEntity;
import com.evolvsys.nexuse2e.core.manualaction.repository.PurgeManualActionRepository;
import org.mapstruct.Context;
import org.mapstruct.InheritConfiguration;
import org.mapstruct.Mapper;
import org.mapstruct.MappingTarget;
import org.springframework.beans.factory.annotation.Autowired;

/**
 * @author SvenBarden
 * @since 12.08.2024
 */
@Mapper(
        config = DefaultMapperConfiguration.class,
        uses = {UserMapper.class})
public abstract class PurgeManualActionMapper
        extends BaseDtoEntityMapper<PurgeManualActionDto, PurgeManualActionEntity> {
    private PurgeManualActionRepository purgeManualActionRepository;

    @Autowired
    protected void config(PurgeManualActionRepository purgeManualActionRepository) {
        this.purgeManualActionRepository = purgeManualActionRepository;
    }

    @Override
    public PurgeManualActionEntity idToEntity(Long id, @Context InterceptingMappingContext interceptingMappingContext) {
        return idToEntity(id, interceptingMappingContext, purgeManualActionRepository::findById);
    }

    @InheritConfiguration
    @ManualActionToDto
    @Override
    protected abstract PurgeManualActionDto entityToDto(
            PurgeManualActionEntity entity, @Context InterceptingMappingContext interceptingMappingContext);

    @InheritConfiguration
    @ManualActionToEntity
    @Override
    protected abstract void updateEntity(
            PurgeManualActionDto dto,
            @MappingTarget PurgeManualActionEntity entity,
            @Context InterceptingMappingContext interceptingMappingContext);
}
