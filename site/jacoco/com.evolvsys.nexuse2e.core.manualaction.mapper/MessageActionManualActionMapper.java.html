package com.evolvsys.nexuse2e.core.manualaction.mapper;

import com.evolvsys.app.core.mapper.BaseDtoEntityMapper;
import com.evolvsys.app.core.mapper.DefaultMapperConfiguration;
import com.evolvsys.app.core.mapper.interceptor.InterceptingMappingContext;
import com.evolvsys.app.core.user.mapper.UserMapper;
import com.evolvsys.nexuse2e.core.manualaction.model.MessageActionManualActionDto;
import com.evolvsys.nexuse2e.core.manualaction.model.MessageActionManualActionEntity;
import com.evolvsys.nexuse2e.core.manualaction.repository.MessageActionManualActionRepository;
import org.mapstruct.*;
import org.springframework.beans.factory.annotation.Autowired;

/**
 * @author SvenBarden
 * @since 05.08.2024
 */
@Mapper(config = DefaultMapperConfiguration.class, uses = UserMapper.class)
public abstract class MessageActionManualActionMapper
        extends BaseDtoEntityMapper<MessageActionManualActionDto, MessageActionManualActionEntity> {
    private MessageActionManualActionRepository messageActionManualActionRepository;

    @Autowired
    protected void config(MessageActionManualActionRepository messageActionManualActionRepository) {
        this.messageActionManualActionRepository = messageActionManualActionRepository;
    }

    @Override
    public MessageActionManualActionEntity idToEntity(
            Long id, @Context InterceptingMappingContext interceptingMappingContext) {
        return idToEntity(id, interceptingMappingContext, messageActionManualActionRepository::findById);
    }

    @InheritConfiguration
    @ManualActionToDto
    @Override
    protected abstract MessageActionManualActionDto entityToDto(
            MessageActionManualActionEntity entity, @Context InterceptingMappingContext interceptingMappingContext);

    @Mapping(target = "messages", ignore = true)
    @InheritConfiguration
    @ManualActionToEntity
    @Override
    protected abstract void updateEntity(
            MessageActionManualActionDto dto,
            @MappingTarget MessageActionManualActionEntity entity,
            @Context InterceptingMappingContext interceptingMappingContext);
}
