package com.evolvsys.nexuse2e.core.service.security;

import com.evolvsys.app.core.auth.model.ContextPermissions;
import com.evolvsys.app.core.auth.model.CurrentUserData;
import com.evolvsys.app.core.persistence.model.IdDto;
import com.evolvsys.nexuse2e.core.service.model.ServiceDto;
import com.evolvsys.nexuse2e.core.system.security.BaseConfigAccessInterceptor;
import java.util.HashSet;
import java.util.Optional;
import java.util.Set;
import java.util.function.Function;
import org.springframework.stereotype.Component;

/**
 * @author SvenBarden
 * @since 05.04.2024
 */
@Component
public class ServiceAccessInterceptor extends BaseConfigAccessInterceptor<ServiceDto> {
    public static final String WRITE_COMPONENT = "NEXUS.SERVICE.WRITE_COMPONENT";

    @Override
    public Set<String> getRevokedPermissions(
            ServiceDto target,
            CurrentUserData currentUserData,
            Function<Optional<? extends IdDto>, ContextPermissions> contextPermissionsFunction) {
        Set<String> revokedPermissions = new HashSet<>();
        if (target.getId() != null) {
            revokedPermissions.add(WRITE_COMPONENT);
        }
        return revokedPermissions;
    }
}
