package com.evolvsys.nexuse2e.core.keyvalue;

import com.evolvsys.nexuse2e.core.component.ComponentRegistry;
import com.evolvsys.nexuse2e.core.component.ComponentType;
import com.evolvsys.nexuse2e.core.component.RegisteredNexusComponent;
import com.evolvsys.nexuse2e.core.exception.NexusError;
import com.evolvsys.nexuse2e.core.exception.NexusRuntimeException;
import com.evolvsys.nexuse2e.core.keyvalue.persistence.service.KeyValuePropertyService;
import com.evolvsys.nexuse2e.core.lifecycle.LifecycleConstants;
import com.evolvsys.nexuse2e.core.model.entity.QServiceEntity;
import com.evolvsys.nexuse2e.core.model.entity.ServiceEntity;
import com.evolvsys.nexuse2e.core.service.ServiceFactory;
import com.evolvsys.nexuse2e.core.service.repository.ServiceRepository;
import com.evolvsys.nexuse2e.impl.service.support.KeyValueSupportService;
import java.util.ArrayList;
import java.util.LinkedHashMap;
import java.util.List;
import java.util.Map;
import java.util.concurrent.TimeUnit;
import java.util.concurrent.atomic.AtomicBoolean;
import lombok.RequiredArgsConstructor;
import lombok.extern.slf4j.Slf4j;
import org.apache.commons.collections4.CollectionUtils;
import org.apache.commons.text.lookup.StringLookup;
import org.apache.commons.text.lookup.StringLookupFactory;
import org.springframework.context.SmartLifecycle;
import org.springframework.data.repository.query.FluentQuery;
import org.springframework.scheduling.annotation.Scheduled;
import org.springframework.stereotype.Service;
import org.springframework.transaction.annotation.Transactional;

/**
 * @author SvenBarden
 * @since 14.06.2024
 */
@Service
@RequiredArgsConstructor
@Slf4j
public class KeyValueService implements SmartLifecycle {
    private final KeyValuePropertyService keyValuePropertyService;
    private final ComponentRegistry componentRegistry;
    private final ServiceRepository serviceRepository;
    private final ServiceFactory serviceFactory;
    private final List<String> componentIds = new ArrayList<>();
    private final AtomicBoolean started = new AtomicBoolean(false);

    StringLookup createStringLookup() {
        Map<String, StringLookup> lookupMap = new LinkedHashMap<>(keyValuePropertyService.loadLookups());

        List<ServiceEntity> serviceEntities = serviceRepository.findBy(
                QServiceEntity.serviceEntity
                        .componentId
                        .in(componentIds)
                        .and(QServiceEntity.serviceEntity.enabled.isTrue()),
                FluentQuery.FetchableFluentQuery::all);

        // since services may have dependencies that trigger errors we need to collect them and try again later
        List<ServiceEntity> errorServices = new ArrayList<>();

        for (ServiceEntity serviceEntity : serviceEntities) {
            try {
                addServiceToMap(serviceEntity, lookupMap);
            } catch (Throwable e) {
                LOGGER.debug("initial error for key value provider service {}", serviceEntity.getServiceId(), e);
                errorServices.add(serviceEntity);
            }
        }

        int lastSize = 0;
        while (CollectionUtils.size(errorServices) != lastSize) {
            lastSize = errorServices.size();
            KeyValueLookup.setStringLookup(StringLookupFactory.INSTANCE.interpolatorStringLookup(
                    lookupMap, new DefaultKeyValueProvider(), false));
            List<ServiceEntity> tempServices = new ArrayList<>(errorServices);
            for (ServiceEntity errorService : tempServices) {
                try {
                    addServiceToMap(errorService, lookupMap);
                    errorServices.remove(errorService);
                } catch (Throwable e) {
                    LOGGER.debug("additional error for key value provider service {}", errorService.getServiceId(), e);
                }
            }
        }

        for (ServiceEntity errorService : errorServices) {
            LOGGER.error("error while reading key value service {}", errorService.getServiceId());
        }

        return StringLookupFactory.INSTANCE.interpolatorStringLookup(lookupMap, new DefaultKeyValueProvider(), false);
    }

    private void addServiceToMap(ServiceEntity serviceEntity, Map<String, StringLookup> lookupMap) {
        KeyValueSupportService service = serviceFactory.get(serviceEntity);
        KeyValueProvider keyValueProvider = service.getKeyValueProvider();
        if (keyValueProvider == null) {
            throw new NexusRuntimeException(NexusError.builder()
                    .message(String.format("Service '%s' returned no keyValueProvider", serviceEntity.getServiceId()))
                    .build());
        } else {
            lookupMap.put(keyValueProvider.getName(), keyValueProvider.getLookup());
        }
    }

    @Scheduled(fixedRate = 1, timeUnit = TimeUnit.MINUTES)
    @Transactional(readOnly = true)
    public void update() {
        if (isRunning()) {
            LOGGER.debug("updating keyValue map");
            KeyValueLookup.setStringLookup(createStringLookup());
        } else {
            LOGGER.debug("skipping keyValue map update, not running");
        }
    }

    @Override
    public void start() {
        try {
            if (started.compareAndSet(false, true)) {
                componentIds.clear();
                componentIds.addAll(componentRegistry.getRegisteredComponents(ComponentType.SUPPORT_SERVICE).stream()
                        .filter(registeredNexusComponent -> KeyValueSupportService.class.isAssignableFrom(
                                registeredNexusComponent.getLoadedClass()))
                        .map(RegisteredNexusComponent::getComponentId)
                        .toList());
            }
        } catch (Exception e) {
            throw new NexusRuntimeException(
                    NexusError.builder()
                            .message("Could not initialize key value service")
                            .build(),
                    e);
        }
    }

    @Override
    public void stop() {
        started.set(false);
    }

    @Override
    public boolean isRunning() {
        return started.get();
    }

    @Override
    public int getPhase() {
        return LifecycleConstants.KEY_VALUE_INITIALIZE;
    }
}
