package com.evolvsys.nexuse2e.core.keyvalue;

import lombok.experimental.UtilityClass;
import org.apache.commons.text.StringSubstitutor;
import org.apache.commons.text.lookup.StringLookup;

/**
 * @author SvenBarden
 * @since 17.06.2024
 */
@UtilityClass
public class KeyValueLookup {
    private static StringSubstitutor STRING_SUBSTITUTOR;

    static void setStringLookup(StringLookup stringLookup) {
        STRING_SUBSTITUTOR = new StringSubstitutor(stringLookup);
    }

    public static String lookup(String s) {
        return STRING_SUBSTITUTOR == null ? s : STRING_SUBSTITUTOR.replace(s);
    }
}
