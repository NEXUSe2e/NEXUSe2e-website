package com.evolvsys.nexuse2e.core.keyvalue;

import java.time.Instant;
import java.time.LocalDate;
import java.time.ZonedDateTime;
import java.time.format.DateTimeFormatter;
import java.util.Map;
import java.util.function.Supplier;
import lombok.extern.slf4j.Slf4j;
import org.apache.commons.text.lookup.StringLookup;

/**
 * Implements
 *
 * @author SvenBarden
 * @since 11.04.2025
 */
@Slf4j
class DefaultKeyValueProvider implements StringLookup {
    private static final Map<String, Supplier<String>> DEFAULT_LOOKUPS = Map.of(
            "NOW",
            () -> ZonedDateTime.now().format(DateTimeFormatter.ISO_DATE_TIME),
            "NOW_MILLIS",
            () -> String.valueOf(Instant.now().toEpochMilli()),
            "NOW_DATE",
            () -> LocalDate.now().format(DateTimeFormatter.ISO_DATE));

    @Override
    public String lookup(String key) {
        LOGGER.trace("KeyValueProvider lookup for key {}", key);
        Supplier<String> supplier = DEFAULT_LOOKUPS.get(key);
        String result;
        if (supplier == null) {
            LOGGER.warn("Unknown keyValue key '{}'", key);
            result = "";
        } else {
            result = supplier.get();
        }
        return result;
    }
}
