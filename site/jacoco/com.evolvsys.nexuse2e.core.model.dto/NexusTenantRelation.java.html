package com.evolvsys.nexuse2e.core.model.dto;

import com.evolvsys.app.core.tenant.model.TenantRelation;
import io.leangen.graphql.annotations.GraphQLIgnore;

/**
 * <p>Interface to be implemented by DTO's when tenant permissions are to be used.</p>
 *
 * @author Florian Drees
 * @since 15.03.2024
 */
public interface NexusTenantRelation extends TenantRelation {
    @GraphQLIgnore
    long SYSTEM_TENANT_ID = 1;

    @Override
    default Long getTenantId() {
        return SYSTEM_TENANT_ID;
    }
}
