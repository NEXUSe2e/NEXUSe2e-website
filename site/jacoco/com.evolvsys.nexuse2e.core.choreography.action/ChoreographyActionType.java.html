package com.evolvsys.nexuse2e.core.choreography.action;

import com.evolvsys.app.core.process.ProcessActionType;
import com.evolvsys.app.core.process.ProcessDefinition;
import com.evolvsys.nexuse2e.core.choreography.model.ChoreographyDto;
import com.evolvsys.nexuse2e.core.model.entity.ChoreographyEntity;
import lombok.Getter;

/**
 * @author SvenBarden
 * @since 04.03.2024
 */
@Getter
public enum ChoreographyActionType implements ProcessActionType<ChoreographyEntity, ChoreographyDto> {
    ADD_ROUTE(ProcessDefinition.<ChoreographyEntity, ChoreographyDto>builder()
            .action(ChoreographyAddRouteAction.class)
            .build()),
    REMOVE_ROUTE(ProcessDefinition.<ChoreographyEntity, ChoreographyDto>builder()
            .action(ChoreographyRemoveRouteAction.class)
            .build()),
    LINK_ROUTE(ProcessDefinition.<ChoreographyEntity, ChoreographyDto>builder()
            .action(ChoreographyLinkRouteAction.class)
            .build());

    private final ProcessDefinition<ChoreographyEntity, ChoreographyDto> process;

    ChoreographyActionType(ProcessDefinition<ChoreographyEntity, ChoreographyDto> process) {
        this.process = process;
    }
}
