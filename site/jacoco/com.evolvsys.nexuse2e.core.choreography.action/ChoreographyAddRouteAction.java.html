package com.evolvsys.nexuse2e.core.choreography.action;

import com.evolvsys.app.core.process.*;
import com.evolvsys.nexuse2e.core.choreography.repository.ChoreographyRepository;
import com.evolvsys.nexuse2e.core.model.entity.ChoreographyEntity;
import com.evolvsys.nexuse2e.core.model.entity.RouteEntity;
import com.evolvsys.nexuse2e.core.route.model.RouteDto;
import com.evolvsys.nexuse2e.core.route.service.RouteService;
import java.util.Set;
import java.util.UUID;
import lombok.RequiredArgsConstructor;
import lombok.extern.slf4j.Slf4j;
import org.apache.tika.utils.StringUtils;
import org.springframework.stereotype.Component;

/**
 * @author SvenBarden
 * @since 04.03.2024
 */
@Component
@RequiredArgsConstructor
@Slf4j
public class ChoreographyAddRouteAction implements ProcessAction<ChoreographyEntity> {
    public static final ProcessParameter<RouteDto> ROUTE_PARAMETER = new ProcessParameter<>() {};

    private final RouteService routeService;
    private final ChoreographyRepository choreographyRepository;

    @Override
    public ChoreographyEntity execute(
            ChoreographyEntity entity, ProcessActionChain chain, ProcessParameters processParameters) {
        Set<RouteEntity> routes = entity.getRoutes();

        RouteDto routeDto = processParameters.get(ROUTE_PARAMETER);

        if (routeDto.getId() == null && StringUtils.isBlank(routeDto.getRouteId())) {
            routeDto.setRouteId(UUID.randomUUID().toString());
            LOGGER.trace("Generated routeId '{}' for route {}", routeDto.getRouteId(), routeDto);
        }

        RouteEntity routeEntity = routeService.getEntity(routeDto);

        routes.add(routeEntity);

        return choreographyRepository.save(entity);
    }

    @Override
    public ProcessActionConditions conditions(ChoreographyEntity entity, ProcessParameters processParameters) {
        return ProcessActionConditions.builder()
                .condition(processParameters.contains(ROUTE_PARAMETER), "Route parameter must not be empty")
                .build();
    }
}
