package com.evolvsys.nexuse2e.core.parameter;

import java.util.LinkedHashMap;
import java.util.Map;
import lombok.EqualsAndHashCode;
import lombok.Singular;
import lombok.ToString;
import lombok.extern.slf4j.Slf4j;

/**
 * @author SvenBarden
 * @since 06.10.2023
 */
@ToString
@EqualsAndHashCode
@Slf4j
public class Parameters {
    @Singular
    private final Map<String, Object> parameters = new LinkedHashMap<>();

    public <T> Parameters put(String name, T value) {
        parameters.put(name, value);
        return this;
    }

    public boolean containsKey(String key) {
        return parameters.containsKey(key);
    }

    @SuppressWarnings("unchecked")
    public <T> T value(String name) {
        return (T) parameters.get(name);
    }

    @SuppressWarnings("unchecked")
    public <T> T value(String name, Class<T> type) {
        return (T) parameters.get(name);
    }

    public int size() {
        return parameters.size();
    }

    /**
     * Merges this and the given {@link Parameters} into a new {@link Parameters} object. The parameters from the given
     * object will overwrite the values in this object.
     *
     * <h3>Example:</h3>
     * <pre>
     * {@code
     *  Parameters p1 = new Parameters().put("param1", "p1-v1").put("param2", "p1-v2");
     *  Parameters p2 = new Parameters().put("param1", "p2-v1").put("param3", "p2-v3");
     *
     *  Parameters p3 = p1.merge(p2);
     * }
     * </pre>
     *
     * Has the new Parameters {@code p3} object with parameters
     * <pre>
     * <b>param1</b>: p2-v1
     * <b>param2</b>: p1-v2
     * <b>param3</b>: p2-v3</pre>
     *
     * @param parameters parameters to merge into a new object
     * @return the new merge parameters
     */
    public Parameters merge(Parameters parameters) {
        if (LOGGER.isTraceEnabled()) {
            LOGGER.trace("merging parameters {} into {}", parameters.parameters.keySet(), this.parameters.keySet());
        }

        Parameters merged = new Parameters();
        merged.parameters.putAll(this.parameters);
        merged.parameters.putAll(parameters.parameters);

        return merged;
    }
}
