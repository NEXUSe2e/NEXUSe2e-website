package com.evolvsys.nexuse2e.core.parameter;

import com.evolvsys.nexuse2e.core.parameter.converter.*;
import lombok.Getter;

/**
 * @author SvenBarden
 * @since 09.10.2023
 */
@Getter
public enum ParameterType {
    STRING(StringTypeConverter.class),
    TRIMMED_STRING(TrimmedStringTypeConverter.class),
    LARGE_STRING(LargeStringTypeConverter.class),
    TRIMMED_LARGE_STRING(TrimmedLargeStringTypeConverter.class),
    CRON_PATTERN(StringTypeConverter.class),
    REGEX_PATTERN(StringTypeConverter.class),
    LONG(LongTypeConverter.class),
    DECIMAL(BigDecimalTypeConverter.class),
    ACTION_NAME(ActionInfoTypeConverter.class),
    CHOREOGRAPHY_NAME(StringTypeConverter.class),
    BOOLEAN(BooleanTypeConverter.class),
    SPRING_BEAN(StringTypeConverter.class),
    STRING_LIST(StringListTypeConverter.class),
    DROPDOWN_LIST(StringTypeConverter.class),
    DURATION(DurationTypeConverter.class),
    PASSWORD(PasswordTypeConverter.class),
    KEY_VALUE_PAIR_LIST_TRIMMED(KeyValuePairListTrimmedConverter.class),
    SUPPORT_SERVICE(SupportServiceTypeConverter.class);

    final Class<? extends TypeConverter<?, ?>> converterClass;

    ParameterType(Class<? extends TypeConverter<?, ?>> converterClass) {
        this.converterClass = converterClass;
    }
}
