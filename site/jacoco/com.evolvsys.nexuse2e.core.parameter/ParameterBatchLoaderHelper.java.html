package com.evolvsys.nexuse2e.core.parameter;

import com.evolvsys.app.core.auth.model.CurrentUserData;
import com.evolvsys.app.core.graphql.repository.BatchLoaderRepository;
import com.evolvsys.app.core.persistence.model.IdEntity;
import com.evolvsys.nexuse2e.core.component.Configurable;
import com.evolvsys.nexuse2e.core.component.RegisteredNexusComponent;
import com.evolvsys.nexuse2e.core.model.entity.ParameterEntity;
import com.evolvsys.nexuse2e.core.model.entity.QParameterEntity;
import com.evolvsys.nexuse2e.core.parameter.mapper.ParameterMapper;
import com.evolvsys.nexuse2e.core.parameter.model.ParameterDto;
import java.util.*;
import java.util.stream.Collectors;
import lombok.RequiredArgsConstructor;
import lombok.extern.slf4j.Slf4j;
import org.apache.commons.collections4.CollectionUtils;
import org.springframework.stereotype.Component;

/**
 * @author SvenBarden
 * @since 19.03.2024
 */
@Slf4j
@Component
@RequiredArgsConstructor
public class ParameterBatchLoaderHelper {
    private final ParameterMapper parameterMapper;
    private final BatchLoaderRepository batchLoaderRepository;

    private ParameterDto entityToDto(
            ParameterEntity parameterEntity,
            ParameterDefinition definition,
            CurrentUserData currentUserData,
            Map<Long, List<String>> parameterStringLists) {
        List<String> strings = parameterStringLists.get(parameterEntity.getId());
        if (strings == null) {
            parameterEntity.setStringValues(Collections.emptyList());
        } else {
            parameterEntity.setStringValues(strings);
        }
        return parameterMapper.entityToDto(parameterEntity, definition, currentUserData);
    }

    public Set<ParameterDto> loadInternalById(
            RegisteredNexusComponent component,
            CurrentUserData currentUserData,
            Set<ParameterEntity> parameterEntities,
            Map<Long, List<String>> parameterStringLists) {
        return loadInternalById(component, currentUserData, parameterEntities, parameterStringLists, true);
    }

    public Set<ParameterDto> loadInternalById(
            RegisteredNexusComponent component,
            CurrentUserData currentUserData,
            Set<ParameterEntity> parameterEntities,
            Map<Long, List<String>> parameterStringLists,
            boolean addDefaults) {

        Set<ParameterDto> parameterDtos = new LinkedHashSet<>();
        if (Configurable.class.isAssignableFrom(component.getLoadedClass())
                && CollectionUtils.isNotEmpty(component.getParameterDefinitions())) {

            for (ParameterDefinition definition : component.getParameterDefinitions()) {
                LOGGER.trace("Processing parameter '{}'", definition.getName());

                ParameterDto parameterDto = parameterEntities.stream()
                        .filter(entity -> entity.getName().equalsIgnoreCase(definition.getName()))
                        .findFirst()
                        .map(entity -> entityToDto(entity, definition, currentUserData, parameterStringLists))
                        .orElseGet(() -> parameterMapper.definitionToDto(definition, currentUserData));

                if (addDefaults || parameterDto.getId() != null) {
                    parameterDtos.add(parameterDto);
                }
            }
        }
        return parameterDtos;
    }

    public Map<Long, List<String>> getStringList(Map<Long, Set<ParameterEntity>> entityMap) {
        Set<Long> parameterIds = entityMap.values().stream()
                .flatMap(parameterEntities -> parameterEntities.stream()
                        .filter(entity -> entity.getType() == ParameterType.STRING_LIST)
                        .map(IdEntity::getId)
                        .distinct())
                .collect(Collectors.toSet());
        return batchLoaderRepository.loadList(parameterIds, QParameterEntity.parameterEntity.stringValues);
    }
}
