package com.evolvsys.nexuse2e.core.service.validation;

import java.util.function.BiFunction;
import java.util.function.Function;
import lombok.experimental.UtilityClass;

/**
 * @author Florian Drees
 * @since 03.01.2024
 */
@UtilityClass
public class ServiceValidationUtils {

    public static boolean validateServiceName(
            Long id,
            String name,
            Function<String, Boolean> existsByNameFunction,
            BiFunction<String, Long, Boolean> existsByNameIdNotFunction) {
        if (id == null) {
            return !existsByNameFunction.apply(name);
        } else {
            return !existsByNameIdNotFunction.apply(name, id);
        }
    }
}
