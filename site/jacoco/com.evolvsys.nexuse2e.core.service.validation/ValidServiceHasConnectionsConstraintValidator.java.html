package com.evolvsys.nexuse2e.core.service.validation;

import com.evolvsys.nexuse2e.core.partner.repository.PartnerConnectionRepository;
import jakarta.validation.ConstraintValidator;
import jakarta.validation.ConstraintValidatorContext;
import lombok.RequiredArgsConstructor;
import org.springframework.stereotype.Component;

/**
 * @author SvenBarden
 * @since 22.04.2024
 */
@Component
@RequiredArgsConstructor
class ValidServiceHasConnectionsConstraintValidator implements ConstraintValidator<ValidServiceHasConnections, Long> {
    private final PartnerConnectionRepository partnerConnectionRepository;

    @Override
    public boolean isValid(Long value, ConstraintValidatorContext context) {
        boolean isValid;
        if (value == null) {
            isValid = true;
        } else {
            isValid = partnerConnectionRepository.existsByOutflowService_Id(value) == false;
        }
        return isValid;
    }
}
