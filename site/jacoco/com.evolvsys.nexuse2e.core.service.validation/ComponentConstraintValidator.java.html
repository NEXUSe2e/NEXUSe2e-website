package com.evolvsys.nexuse2e.core.service.validation;

import com.evolvsys.nexuse2e.core.component.ComponentReference;
import com.evolvsys.nexuse2e.core.component.ComponentRegistry;
import jakarta.validation.ConstraintValidator;
import jakarta.validation.ConstraintValidatorContext;
import lombok.RequiredArgsConstructor;
import org.springframework.stereotype.Component;

/**
 * @author Florian Drees
 * @since 03.01.2024
 */
@Component
@RequiredArgsConstructor
public class ComponentConstraintValidator implements ConstraintValidator<ValidComponent, ComponentReference> {
    private final ComponentRegistry componentRegistry;

    @Override
    public boolean isValid(ComponentReference value, ConstraintValidatorContext context) {
        return componentRegistry.exists(value.getComponentId());
    }
}
