package com.evolvsys.nexuse2e.core.service.validation;

import com.evolvsys.nexuse2e.core.service.model.ServiceDto;
import com.evolvsys.nexuse2e.core.service.repository.ServiceRepository;
import jakarta.validation.ConstraintValidator;
import jakarta.validation.ConstraintValidatorContext;
import lombok.RequiredArgsConstructor;
import org.springframework.stereotype.Component;

/**
 * @author Florian Drees
 * @since 03.01.2024
 */
@Component
@RequiredArgsConstructor
public class ValidServiceConstraintValidator implements ConstraintValidator<ValidService, ServiceDto> {
    private final ServiceRepository serviceRepository;

    @Override
    public boolean isValid(ServiceDto value, ConstraintValidatorContext context) {
        return ServiceValidationUtils.validateServiceName(
                value.getId(),
                value.getServiceId(),
                serviceRepository::existsByServiceId,
                serviceRepository::existsByServiceIdAndIdNot);
    }
}
