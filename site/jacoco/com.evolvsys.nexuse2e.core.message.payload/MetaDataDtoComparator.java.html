package com.evolvsys.nexuse2e.core.message.payload;

import com.evolvsys.nexuse2e.core.message.model.MetaDataDto;
import com.evolvsys.nexuse2e.core.model.MetaDataType;
import java.util.Comparator;
import java.util.List;
import lombok.AccessLevel;
import lombok.NoArgsConstructor;
import org.apache.commons.collections4.comparators.ComparatorChain;
import org.apache.commons.collections4.comparators.FixedOrderComparator;
import org.apache.commons.collections4.comparators.TransformingComparator;
import org.apache.commons.lang3.ObjectUtils;

/**
 * @author SvenBarden
 * @since 04.04.2024
 */
@NoArgsConstructor(access = AccessLevel.PRIVATE)
public class MetaDataDtoComparator implements Comparator<MetaDataDto> {
    public static final MetaDataDtoComparator INSTANCE = new MetaDataDtoComparator();
    private final Comparator<MetaDataDto> delegate = new ComparatorChain<>(List.of(
            new TransformingComparator<>(MetaDataDto::getType, typeComparator()),
            new TransformingComparator<>(
                    input -> ObjectUtils.defaultIfNull(input.getName(), "").toLowerCase()),
            new TransformingComparator<>(MetaDataDto::getIndex)));

    private static Comparator<MetaDataType> typeComparator() {
        FixedOrderComparator<MetaDataType> comparator = new FixedOrderComparator<>(MetaDataType.PAYLOAD);
        comparator.setUnknownObjectBehavior(FixedOrderComparator.UnknownObjectBehavior.AFTER);
        return comparator;
    }

    @Override
    public int compare(MetaDataDto o1, MetaDataDto o2) {
        return delegate.compare(o1, o2);
    }
}
