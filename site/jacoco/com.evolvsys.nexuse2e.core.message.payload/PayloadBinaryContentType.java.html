package com.evolvsys.nexuse2e.core.message.payload;

import com.evolvsys.app.core.binary.model.BinaryResult;
import com.evolvsys.app.core.binary.service.BinaryType;
import com.evolvsys.nexuse2e.core.exception.NexusError;
import com.evolvsys.nexuse2e.core.exception.NexusRuntimeException;
import com.evolvsys.nexuse2e.core.message.payload.model.PayloadEntity;
import com.evolvsys.nexuse2e.core.message.payload.repository.PayloadRepository;
import edu.umd.cs.findbugs.annotations.SuppressFBWarnings;
import java.sql.SQLException;
import java.util.Optional;
import lombok.RequiredArgsConstructor;
import lombok.extern.slf4j.Slf4j;
import org.apache.commons.lang3.StringUtils;
import org.apache.tika.config.TikaConfig;
import org.apache.tika.mime.MimeTypeException;
import org.springframework.core.io.InputStreamResource;
import org.springframework.stereotype.Component;

/**
 * @author SvenBarden
 * @since 25.03.2024
 */
@Component
@RequiredArgsConstructor
@Slf4j
public class PayloadBinaryContentType implements BinaryType {
    public static final String NAME = "PAYLOAD_CONTENT";

    private final PayloadRepository payloadRepository;

    @Override
    public String name() {
        return NAME;
    }

    @SuppressFBWarnings("EXS_EXCEPTION_SOFTENING_NO_CONSTRAINTS")
    private BinaryResult entityToResult(PayloadEntity payloadEntity) {
        String fileName = payloadEntity.getName();
        if (fileName == null) {
            fileName = String.valueOf(payloadEntity.getId());
        }
        try {
            String extension = TikaConfig.getDefaultConfig()
                    .getMimeRepository()
                    .forName(payloadEntity.getContentType())
                    .getExtension();
            LOGGER.trace("detected extension {} for mime type {}", extension, payloadEntity.getContentType());
            if (StringUtils.endsWithIgnoreCase(fileName, extension)) {
                LOGGER.trace("file already has extension in name, do not add additional suffix");
            } else {
                fileName += extension;
            }
        } catch (MimeTypeException e) {
            LOGGER.warn(
                    "could not detect suffix for mimeType {} in payload {}",
                    payloadEntity.getContentType(),
                    payloadEntity.getName(),
                    e);
        }
        try {
            return BinaryResult.builder()
                    .size(payloadEntity.getSize())
                    .mimeType(payloadEntity.getContentType())
                    .fileName(fileName)
                    .resource(new InputStreamResource(
                            payloadEntity.getContent().getContent().getBinaryStream()))
                    .build();
        } catch (SQLException e) {
            throw new NexusRuntimeException(
                    NexusError.builder()
                            .message("could not load content for payload")
                            .extension("id", payloadEntity.getId())
                            .build(),
                    e);
        }
    }

    @Override
    public Optional<BinaryResult> get(String id) {
        long payloadId = Long.parseLong(id);
        return payloadRepository.getEntityById(payloadId).map(this::entityToResult);
    }
}
