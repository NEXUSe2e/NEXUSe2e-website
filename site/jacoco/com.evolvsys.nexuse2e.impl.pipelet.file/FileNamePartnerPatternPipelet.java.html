package com.evolvsys.nexuse2e.impl.pipelet.file;

import com.evolvsys.nexuse2e.core.component.Pipelet;
import com.evolvsys.nexuse2e.core.constants.FileConstants;
import com.evolvsys.nexuse2e.core.exception.NexusException;
import com.evolvsys.nexuse2e.core.message.context.ExecutionContext;
import com.evolvsys.nexuse2e.core.parameter.ParameterDefinition;
import com.evolvsys.nexuse2e.core.parameter.ParameterType;
import java.util.List;
import java.util.Optional;
import java.util.regex.Matcher;
import java.util.regex.Pattern;
import lombok.extern.slf4j.Slf4j;

/**
 * @author SvenBarden
 * @since 16.01.2024
 */
@Slf4j
public class FileNamePartnerPatternPipelet extends Pipelet {
    private static final String PARAMETER_SOURCE_PARTNER_GROUP_NUMBER = "source_partner_group_number";
    private static final String PARAMETER_TARGET_PARTNER_GROUP_NUMBER = "target_partner_group_number";

    @Override
    public List<ParameterDefinition> parameterDefinitions() {
        return List.of(
                ParameterDefinition.builder()
                        .name("pattern")
                        .displayName("The regexp pattern")
                        .description(
                                "The regular expression to match the file name pattern. Group numbers can be referenced by additional parameters.")
                        .type(ParameterType.REGEX_PATTERN)
                        .required(true)
                        .defaultValue("(.+)_(.+)_.+")
                        .build(),
                ParameterDefinition.builder()
                        .name(PARAMETER_SOURCE_PARTNER_GROUP_NUMBER)
                        .displayName("Source partner group number")
                        .description(
                                "The number of the regexp group in the pattern, that matches the source partner id.")
                        .type(ParameterType.LONG)
                        .defaultValue(1L)
                        .build(),
                ParameterDefinition.builder()
                        .name(PARAMETER_TARGET_PARTNER_GROUP_NUMBER)
                        .displayName("Target partner group number")
                        .description(
                                "The number of the regexp group in the pattern, that matches the target partner id.")
                        .type(ParameterType.LONG)
                        .defaultValue(2L)
                        .build());
    }

    @Override
    public ExecutionContext execute(ExecutionContext executionContext) throws NexusException {
        Pattern pattern = Pattern.compile(parameters.value("pattern"));
        String fileName = executionContext.getMetaData().getFirstOrThrows(FileConstants.FILE_NAME);

        LOGGER.debug("applying pattern {} on fileName {}", pattern, fileName);

        Matcher matcher = pattern.matcher(fileName);

        if (matcher.matches()) {
            Long sourcePartnerGroupNumber = parameters.value(PARAMETER_SOURCE_PARTNER_GROUP_NUMBER);
            Optional.ofNullable(sourcePartnerGroupNumber)
                    .map(aLong -> matcher.group(aLong.intValue()))
                    .ifPresent(executionContext::setSourcePartnerId);

            Long targetPartnerGroupNumber = parameters.value(PARAMETER_TARGET_PARTNER_GROUP_NUMBER);
            Optional.ofNullable(targetPartnerGroupNumber)
                    .map(aLong -> matcher.group(aLong.intValue()))
                    .ifPresent(executionContext::setTargetPartnerId);
        } else {
            LOGGER.debug("pattern did not match");
        }

        return executionContext;
    }
}
