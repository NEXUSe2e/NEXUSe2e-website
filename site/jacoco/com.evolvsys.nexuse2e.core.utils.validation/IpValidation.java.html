package com.evolvsys.nexuse2e.core.utils.validation;

import com.github.jgonian.ipmath.AbstractIp;
import com.github.jgonian.ipmath.AbstractIpRange;
import java.util.Collection;
import java.util.function.Function;
import org.apache.commons.collections4.CollectionUtils;

/**
 * @author Florian Drees
 * @since 02.04.2024
 */
public abstract class IpValidation<
        IpType extends AbstractIp<IpType, ?>, IpRangeType extends AbstractIpRange<IpType, ?>> {
    protected static final String SLASH = "/";
    protected static final String DASH = "-";

    abstract IpType toIp(String ipAddress);

    public abstract boolean isValid(String ipAddress);

    public boolean isValidRange(String ipAddressRange) {
        try {
            buildRange(ipAddressRange);
            return true;
        } catch (IllegalArgumentException e) {
            // do nothing
        }
        return false;
    }

    public boolean isValid(String ipAddress, Function<String, AbstractIp<?, ?>> ipFunction) {
        try {
            ipFunction.apply(ipAddress);
            return true;
        } catch (IllegalArgumentException e) {
            // do nothing
        }
        return false;
    }

    public boolean isInRange(String ipAddress, String ipAddressRange) {
        return buildRange(ipAddressRange).contains(toIp(ipAddress));
    }

    public boolean isInAnyRange(String ipAddress, Collection<String> ipAddressRanges) {
        return CollectionUtils.emptyIfNull(ipAddressRanges).stream()
                .anyMatch(ipAddressRange -> isInRange(ipAddress, ipAddressRange));
    }

    abstract IpRangeType buildRange(String rawIpAddressRange);
}
