package com.evolvsys.nexuse2e.core.utils.validation;

import com.github.jgonian.ipmath.Ipv4;
import com.github.jgonian.ipmath.Ipv4Range;
import org.apache.commons.lang3.StringUtils;

/**
 * @author Florian Drees
 * @since 02.04.2024
 */
public class Ipv4Validation extends IpValidation<Ipv4, Ipv4Range> {
    @Override
    Ipv4 toIp(String ipAddress) {
        return Ipv4.of(ipAddress);
    }

    @Override
    public boolean isValid(String ipAddress) {
        return isValid(ipAddress, Ipv4::of);
    }

    @Override
    Ipv4Range buildRange(String rawIpAddressRange) {
        if (StringUtils.containsAny(rawIpAddressRange, DASH, SLASH)) {
            return Ipv4Range.parse(rawIpAddressRange);
        } else {
            return Ipv4Range.from(rawIpAddressRange).to(rawIpAddressRange);
        }
    }
}
