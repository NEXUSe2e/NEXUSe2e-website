package com.evolvsys.nexuse2e.core.utils.validation;

import java.util.ArrayList;
import java.util.Collection;
import java.util.List;
import lombok.experimental.UtilityClass;
import lombok.extern.slf4j.Slf4j;
import org.apache.commons.collections4.CollectionUtils;

/**
 * @author Florian Drees
 * @since 26.03.2024
 */
@UtilityClass
@Slf4j
public class IpUtils {
    private static final Ipv4Validation IPV_4_VALIDATION = new Ipv4Validation();
    private static final Ipv6Validation IPV_6_VALIDATION = new Ipv6Validation();

    public static boolean isValid(
            String ipAddress, Collection<String> validIpv4AddressRanges, Collection<String> validIpv6AddressRanges) {
        List<String> mergedCollection = new ArrayList<>();
        if (CollectionUtils.isNotEmpty(validIpv4AddressRanges)) {
            mergedCollection.addAll(validIpv4AddressRanges);
        }
        if (CollectionUtils.isNotEmpty(validIpv6AddressRanges)) {
            mergedCollection.addAll(validIpv6AddressRanges);
        }
        return isValid(ipAddress, mergedCollection);
    }

    public static boolean isValid(String ipAddress, Collection<String> validIpAddressRanges) {
        return CollectionUtils.isEmpty(validIpAddressRanges)
                || validIpAddressRanges.stream().anyMatch(ipAddressRange -> isValid(ipAddress, ipAddressRange));
    }

    public static boolean isValid(String ipAddress, String ipAddressRange) {
        boolean result;
        if (IPV_4_VALIDATION.isValidRange(ipAddressRange) && IPV_4_VALIDATION.isValid(ipAddress)) {
            result = IPV_4_VALIDATION.isInRange(ipAddress, ipAddressRange);
        } else if (IPV_6_VALIDATION.isValidRange(ipAddressRange) && IPV_6_VALIDATION.isValid(ipAddress)) {
            result = IPV_6_VALIDATION.isInRange(ipAddress, ipAddressRange);
        } else {
            LOGGER.trace("ip address '{}' and range '{}' do not match ipv4 and ipv6", ipAddress, ipAddressRange);
            result = false;
        }
        LOGGER.trace("ip validation for {} against {} results in {}", ipAddress, ipAddressRange, result);
        return result;
    }
}
