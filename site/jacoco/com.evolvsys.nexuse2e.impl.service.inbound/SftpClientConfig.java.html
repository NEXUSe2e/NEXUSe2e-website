package com.evolvsys.nexuse2e.impl.service.inbound;

import java.time.Duration;
import lombok.Builder;
import lombok.ToString;
import lombok.Value;

/**
 *
 */
@Value
@Builder
public class SftpClientConfig {

    String serverRoot;
    boolean enableDeprecatedAlgorithms;
    String hostKeyData;
    String hostKeyAlgorithm;
    String host;
    int port;
    String user;

    @ToString.Exclude
    String password;

    Duration serverConnectTimeout;

    @ToString.Exclude
    String privateKeyData;

    @ToString.Exclude
    String privateKeyPassword;

    boolean createDirectory;

    public static class SftpClientConfigBuilder {
        public SftpClientConfigBuilder serverRoot(String serverRoot) {
            this.serverRoot = SftpUtils.sanitizePath(serverRoot);
            return this;
        }
    }
}
