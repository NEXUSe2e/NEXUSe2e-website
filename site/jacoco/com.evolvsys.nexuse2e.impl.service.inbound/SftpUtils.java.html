package com.evolvsys.nexuse2e.impl.service.inbound;

import com.evolvsys.nexuse2e.core.exception.NexusError;
import com.evolvsys.nexuse2e.core.exception.NexusException;
import java.io.ByteArrayInputStream;
import java.io.IOException;
import java.nio.charset.StandardCharsets;
import java.security.GeneralSecurityException;
import java.security.KeyPair;
import java.security.PublicKey;
import lombok.experimental.UtilityClass;
import lombok.extern.slf4j.Slf4j;
import org.apache.commons.lang3.StringUtils;
import org.apache.sshd.client.SshClient;
import org.apache.sshd.client.config.hosts.KnownHostEntry;
import org.apache.sshd.client.keyverifier.AcceptAllServerKeyVerifier;
import org.apache.sshd.client.keyverifier.RequiredServerKeyVerifier;
import org.apache.sshd.client.keyverifier.ServerKeyVerifier;
import org.apache.sshd.client.session.ClientSession;
import org.apache.sshd.common.config.keys.AuthorizedKeyEntry;
import org.apache.sshd.common.config.keys.PublicKeyEntryResolver;
import org.apache.sshd.common.util.security.SecurityUtils;

@Slf4j
@UtilityClass
public class SftpUtils {

    public static void applyHostKeyValidation(SshClient client, String algorithm, String data)
            throws IOException, GeneralSecurityException {

        ServerKeyVerifier serverKeyVerifier;

        if (StringUtils.isAnyBlank(data, algorithm)) {
            LOGGER.warn("no host key validation applied");
            serverKeyVerifier = AcceptAllServerKeyVerifier.INSTANCE;
        } else {
            LOGGER.trace("host key validation is applied");
            KnownHostEntry knownHostEntry =
                    KnownHostEntry.parseKnownHostEntry("unknown %s %s".formatted(algorithm, data));
            AuthorizedKeyEntry authEntry = knownHostEntry.getKeyEntry();
            PublicKey key = authEntry.resolvePublicKey(null, PublicKeyEntryResolver.UNSUPPORTED);
            serverKeyVerifier = new RequiredServerKeyVerifier(key);
        }

        client.setServerKeyVerifier(serverKeyVerifier);
    }

    public static void authenticateSessionViaSshKey(ClientSession session, String privateKeyData, String password)
            throws NexusException {
        LOGGER.trace("authenticating via SSH Key");
        try {
            Iterable<KeyPair> keyPairs = SecurityUtils.loadKeyPairIdentities(
                    session,
                    null,
                    new ByteArrayInputStream(privateKeyData.getBytes(StandardCharsets.UTF_8)),
                    (session1, resourceKey, retryIndex) -> password);
            session.addPublicKeyIdentity(keyPairs.iterator().next());
        } catch (Exception e) {
            throw new NexusException(
                    NexusError.builder()
                            .message("An error occurred during ssh key authentication")
                            .build(),
                    e);
        }
    }

    public static void authenticateSessionViaPassword(ClientSession session, String password) throws NexusException {
        LOGGER.trace("authenticating via password");
        try {
            session.addPasswordIdentity(password);
        } catch (Exception e) {
            throw new NexusException(
                    NexusError.builder()
                            .message("An error occurred during password authentication")
                            .build(),
                    e);
        }
    }

    public static String sanitizePath(String path) {
        String sanitizedPath;
        if (StringUtils.isBlank(path)) {
            sanitizedPath = "/";
        } else {
            sanitizedPath = StringUtils.appendIfMissing(path, "/");
        }
        return sanitizedPath;
    }

    public static String cleanupAndValidateRelativePath(String path) throws NexusException {

        String resultingPath = "";
        if (path != null) {

            if (path.contains(".")) {
                throw new NexusException(NexusError.builder()
                        .message("possible path traversal issue, dots are not expected in path, please fix the path: "
                                + path)
                        .build());
            }

            resultingPath = path;
            if ("/".equals(resultingPath)) {
                resultingPath = "";
            } else {
                if (resultingPath.startsWith("/")) {
                    resultingPath = resultingPath.substring(1);
                }
                if (!resultingPath.endsWith("/")) {
                    resultingPath = resultingPath + "/";
                }
            }
        }

        return resultingPath;
    }

    public static void validateFilename(String filename) throws NexusException {
        if (filename.contains("/")) {
            throw new NexusException(NexusError.builder()
                    .message(
                            "possible traversal issue, / is not allowed in filename, please change the filename or filename pattern: "
                                    + filename)
                    .build());
        }
    }
}
