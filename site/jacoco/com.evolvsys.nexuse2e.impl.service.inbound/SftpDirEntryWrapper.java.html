package com.evolvsys.nexuse2e.impl.service.inbound;

import lombok.AccessLevel;
import lombok.Getter;
import lombok.NoArgsConstructor;
import lombok.Setter;
import lombok.experimental.Delegate;
import lombok.extern.slf4j.Slf4j;
import org.apache.sshd.sftp.client.SftpClient;

/**
 *
 */
@Slf4j
@Getter
@Setter
@NoArgsConstructor(access = AccessLevel.PRIVATE)
public class SftpDirEntryWrapper {

    @Delegate
    private SftpClient.DirEntry entry;

    private ProcessingBehavior processingBehavior = ProcessingBehavior.DELETE_AFTER_PROCESSING;

    private String target;
    private String source;
    private String serverPath;
    private int processedCount;

    public static SftpDirEntryWrapper of(SftpClient.DirEntry entry, String serverPath) {
        SftpDirEntryWrapper wrapper = new SftpDirEntryWrapper();
        wrapper.setEntry(entry);
        wrapper.serverPath = serverPath;
        wrapper.setSource(serverPath + entry.getFilename());

        return wrapper;
    }
}
