package com.evolvsys.nexuse2e.impl.service.inbound;

import com.evolvsys.nexuse2e.core.exception.NexusError;
import com.evolvsys.nexuse2e.core.exception.NexusException;
import java.io.IOException;
import lombok.AllArgsConstructor;
import org.apache.sshd.sftp.client.SftpClient;

/**
 *
 */
@AllArgsConstructor
public enum ProcessingBehavior {
    DELETE_WITHOUT_PROCESSING(Timing.PREPROCESSING) {
        @Override
        public void processEntry(SftpClient sftpClient, SftpDirEntryWrapper sftpDirEntryWrapper) throws NexusException {
            deleteFile(sftpClient, sftpDirEntryWrapper);
        }
    },
    DELETE_AFTER_PROCESSING(Timing.POSTPROCESSING) {
        @Override
        public void processEntry(SftpClient sftpClient, SftpDirEntryWrapper sftpDirEntryWrapper) throws NexusException {
            deleteFile(sftpClient, sftpDirEntryWrapper);
        }
    },
    RENAME_WITHOUT_PROCESSING(Timing.PREPROCESSING) {
        @Override
        public void processEntry(SftpClient sftpClient, SftpDirEntryWrapper sftpDirEntryWrapper) throws NexusException {
            renameFile(sftpClient, sftpDirEntryWrapper);
        }
    },
    RENAME_AFTER_PROCESSING(Timing.POSTPROCESSING) {
        @Override
        public void processEntry(SftpClient sftpClient, SftpDirEntryWrapper sftpDirEntryWrapper) throws NexusException {
            renameFile(sftpClient, sftpDirEntryWrapper);
        }
    },
    DO_NOTHING_AFTER_PROCESSING(Timing.POSTPROCESSING) {
        @Override
        public void processEntry(SftpClient sftpClient, SftpDirEntryWrapper sftpDirEntryWrapper) {}
    };

    public final Timing timing;

    private static void renameFile(SftpClient sftpClient, SftpDirEntryWrapper sftpDirEntryWrapper)
            throws NexusException {
        try {
            sftpClient.rename(sftpDirEntryWrapper.getSource(), sftpDirEntryWrapper.getTarget());
        } catch (IOException e) {
            throw new NexusException(
                    NexusError.builder()
                            .message(String.format("Could not rename remote file %s", sftpDirEntryWrapper.getSource()))
                            .code("FILE_ERROR")
                            .build(),
                    e);
        }
    }

    private static void deleteFile(SftpClient sftpClient, SftpDirEntryWrapper sftpDirEntryWrapper)
            throws NexusException {
        try {
            sftpClient.remove(sftpDirEntryWrapper.getSource());
        } catch (IOException e) {
            throw new NexusException(
                    NexusError.builder()
                            .message(String.format("Could not remove remote file %s", sftpDirEntryWrapper.getSource()))
                            .code("FILE_ERROR")
                            .build(),
                    e);
        }
    }

    public abstract void processEntry(SftpClient sftpClient, SftpDirEntryWrapper sftpDirEntryWrapper)
            throws NexusException;

    public enum Timing {
        PREPROCESSING,
        POSTPROCESSING
    }
}
