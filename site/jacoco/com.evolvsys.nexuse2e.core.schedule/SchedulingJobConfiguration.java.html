package com.evolvsys.nexuse2e.core.schedule;

import org.quartz.*;
import org.springframework.context.annotation.Bean;
import org.springframework.context.annotation.Configuration;

/**
 * @author SvenBarden
 * @since 26.03.2024
 */
@Configuration
public class SchedulingJobConfiguration {
    public static final JobKey SERVICE_CLEANUP_JOB_KEY = new JobKey("cleanup", "scheduling_service");
    public static final TriggerKey SERVICE_CLEANUP_TRIGGER_KEY = new TriggerKey("cleanup", "scheduling_service");

    @Bean
    JobDetail schedulingCleanupJob() {
        return JobBuilder.newJob(SchedulingServiceCleanupJob.class)
                .withIdentity(SERVICE_CLEANUP_JOB_KEY)
                .storeDurably()
                .build();
    }

    @Bean
    Trigger schedulingCleanupTrigger() {
        return TriggerBuilder.newTrigger()
                .withIdentity(SERVICE_CLEANUP_TRIGGER_KEY)
                .forJob(SERVICE_CLEANUP_JOB_KEY)
                .withSchedule(SimpleScheduleBuilder.repeatHourlyForever(1))
                .build();
    }
}
