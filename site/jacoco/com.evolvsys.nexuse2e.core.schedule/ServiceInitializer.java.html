package com.evolvsys.nexuse2e.core.schedule;

import com.evolvsys.nexuse2e.core.component.Service;
import com.evolvsys.nexuse2e.core.model.entity.ServiceEntity;
import com.evolvsys.nexuse2e.core.service.ServiceFactory;
import com.evolvsys.nexuse2e.core.service.repository.ServiceRepository;
import com.evolvsys.nexuse2e.core.utils.ErrorUtils;
import lombok.RequiredArgsConstructor;
import lombok.extern.slf4j.Slf4j;
import org.springframework.beans.factory.InitializingBean;
import org.springframework.stereotype.Component;

/**
 * <p>Start all enabled services on application startup.</p>
 *
 * @author Florian Drees
 * @since 31.01.2024
 */
@Component
@RequiredArgsConstructor
@Slf4j
public class ServiceInitializer implements InitializingBean {
    private final ServiceRepository serviceRepository;
    private final ServiceFactory serviceFactory;

    @Override
    public void afterPropertiesSet() {
        for (ServiceEntity serviceEntity : serviceRepository.findByEnabledTrue()) {
            LOGGER.debug(
                    "registering scheduled service '{}' with component '{}'",
                    serviceEntity.getServiceId(),
                    serviceEntity.getComponentId());
            try {
                Service service = serviceFactory.get(serviceEntity.getServiceId());
                service.start();
            } catch (Throwable e) {
                LOGGER.error(
                        "Unexpected error occurred while starting service {}: {}",
                        serviceEntity.getServiceId(),
                        ErrorUtils.aggregate(e),
                        e);
                serviceEntity.setEnabled(false);
                serviceRepository.save(serviceEntity);
            }
        }
    }
}
