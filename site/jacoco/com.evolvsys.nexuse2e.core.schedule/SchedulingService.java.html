package com.evolvsys.nexuse2e.core.schedule;

import com.evolvsys.nexuse2e.core.schedule.model.SchedulingDataDto;
import com.evolvsys.nexuse2e.core.schedule.provider.ScheduleProvider;
import com.evolvsys.nexuse2e.core.service.BaseInflowNexusService;
import com.evolvsys.nexuse2e.core.utils.ErrorUtils;
import java.time.ZoneId;
import java.time.ZonedDateTime;
import java.util.Date;
import java.util.Optional;
import java.util.Set;
import lombok.RequiredArgsConstructor;
import lombok.extern.slf4j.Slf4j;
import org.quartz.*;
import org.quartz.impl.matchers.GroupMatcher;
import org.springframework.stereotype.Component;

/**
 * <p>Service for {@link BaseInflowNexusService} implementing {@link SchedulingServiceAware}.</p>
 *
 * @author Florian Drees
 * @since 30.01.2024
 */
@Component
@RequiredArgsConstructor
@Slf4j
public class SchedulingService {
    public static final String QUARTZ_GROUP_ID = "scheduled_service";
    private final Scheduler scheduler;

    public void register(BaseInflowNexusService nexusService, ScheduleProvider<? extends ScheduleBuilder<?>> provider) {
        Trigger trigger = TriggerBuilder.newTrigger()
                .forJob(getJobKey(nexusService))
                .withIdentity(getTriggerKey(nexusService))
                .withSchedule(provider.getScheduleBuilder())
                .build();

        JobDetail jobDetail = JobBuilder.newJob(BasicServiceSchedulingJob.class)
                .withIdentity(trigger.getJobKey())
                .build();
        jobDetail.getJobDataMap().put("service", nexusService.getServiceId());

        try {
            LOGGER.debug(
                    "Scheduling quartz job for nexus service '{}' of component '{}'",
                    nexusService.getServiceId(),
                    nexusService.getComponentId());
            scheduler.scheduleJob(jobDetail, Set.of(trigger), true);
        } catch (SchedulerException e) {
            LOGGER.error("Unexpected exception while scheduling quartz job: {}", ErrorUtils.aggregate(e), e);
        }
    }

    public boolean isScheduled(BaseInflowNexusService nexusService) {
        return isScheduled(getJobKey(nexusService));
    }

    private boolean isScheduled(JobKey jobKey) {
        try {
            return scheduler.checkExists(jobKey);
        } catch (SchedulerException e) {
            LOGGER.error("Unable to check if job {} is enabled: {}", jobKey, ErrorUtils.aggregate(e));
        }
        return false;
    }

    public void deregister(BaseInflowNexusService nexusService) {
        deregister(getJobKey(nexusService));
    }

    private void deregister(JobKey jobKey) {
        LOGGER.debug("Remove job '{}' from the scheduler", jobKey);
        try {
            if (!scheduler.deleteJob(jobKey)) {
                LOGGER.error("Unable to remove job '{}' from the quartz scheduler.", jobKey);
            }
        } catch (SchedulerException e) {
            LOGGER.error(
                    "Internal scheduler error while trying to delete job '{}': {}", jobKey, ErrorUtils.aggregate(e));
        }
    }

    public void deleteAll() {
        try {
            scheduler.getJobKeys(GroupMatcher.jobGroupEquals(QUARTZ_GROUP_ID)).forEach(this::deregister);
        } catch (SchedulerException e) {
            LOGGER.error("Clearing all scheduled service jobs failed", e);
        }
    }

    private static String getQuartzName(String serviceId, String componentId) {
        return componentId + "_AS_" + serviceId;
    }

    private static JobKey getJobKey(BaseInflowNexusService nexusService) {
        return new JobKey(getQuartzName(nexusService.getServiceId(), nexusService.getComponentId()), QUARTZ_GROUP_ID);
    }

    private static TriggerKey getTriggerKey(BaseInflowNexusService nexusService) {
        return getTriggerKey(nexusService.getServiceId(), nexusService.getComponentId());
    }

    private static TriggerKey getTriggerKey(String serviceId, String componentId) {
        return new TriggerKey(getQuartzName(serviceId, componentId), QUARTZ_GROUP_ID);
    }

    public SchedulingDataDto toSchedulingData(String serviceId, String componentId) {
        Trigger trigger;
        try {
            trigger = scheduler.getTrigger(getTriggerKey(serviceId, componentId));
        } catch (SchedulerException e) {
            LOGGER.error(
                    "Internal scheduler error while trying to get trigger for service {}: {}",
                    serviceId,
                    ErrorUtils.aggregate(e),
                    e);
            return null;
        }

        if (trigger == null) {
            return null;
        }

        SchedulingDataDto schedulingDataDto = new SchedulingDataDto();
        schedulingDataDto.setLastRun(toZonedDateTime(trigger.getPreviousFireTime()));

        Date nextFireTime = trigger.getNextFireTime();
        schedulingDataDto.setNextRun(toZonedDateTime(nextFireTime));

        if (nextFireTime != null) {
            Date runAfterNextTime = trigger.getFireTimeAfter(nextFireTime);
            schedulingDataDto.setRunAfterNext(toZonedDateTime(runAfterNextTime));
        }
        return schedulingDataDto;
    }

    private ZonedDateTime toZonedDateTime(Date date) {
        return Optional.ofNullable(date)
                .map(aDate -> aDate.toInstant().atZone(ZoneId.systemDefault()))
                .orElse(null);
    }
}
