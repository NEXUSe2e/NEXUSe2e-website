package com.evolvsys.nexuse2e.core.schedule;

import com.evolvsys.nexuse2e.core.component.ComponentRegistry;
import com.evolvsys.nexuse2e.core.model.entity.ServiceEntity;
import com.evolvsys.nexuse2e.core.service.repository.ServiceRepository;
import com.evolvsys.nexuse2e.core.utils.ErrorUtils;
import edu.umd.cs.findbugs.annotations.SuppressFBWarnings;
import java.util.Set;
import lombok.RequiredArgsConstructor;
import lombok.extern.slf4j.Slf4j;
import org.quartz.*;
import org.quartz.impl.matchers.GroupMatcher;
import org.springframework.scheduling.quartz.QuartzJobBean;

/**
 * @author SvenBarden
 * @since 26.03.2024
 */
@RequiredArgsConstructor
@Slf4j
@DisallowConcurrentExecution
public class SchedulingServiceCleanupJob extends QuartzJobBean {
    private final ServiceRepository serviceRepository;
    private final ComponentRegistry componentRegistry;

    @Override
    @SuppressFBWarnings("CLI_CONSTANT_LIST_INDEX")
    protected void executeInternal(JobExecutionContext context) throws JobExecutionException {
        Scheduler scheduler = context.getScheduler();
        try {
            Set<JobKey> serviceJobKeys =
                    scheduler.getJobKeys(GroupMatcher.jobGroupEquals(SchedulingService.QUARTZ_GROUP_ID));
            for (JobKey serviceJobKey : serviceJobKeys) {
                String[] strings = extractServiceName(serviceJobKey.getName());
                if (strings.length == 2) {
                    String componentId = strings[0];
                    String serviceId = strings[1];

                    ServiceEntity serviceEntity =
                            serviceRepository.findByServiceId(serviceId).orElse(null);
                    if (serviceEntity == null) {
                        LOGGER.error(
                                "service with id {} for job {} could not be found, job will be removed",
                                serviceId,
                                serviceJobKey);
                        scheduler.deleteJob(serviceJobKey);
                    } else {
                        if (serviceEntity.getComponentId().equals(componentId)) {
                            if (componentRegistry.exists(componentId)) {
                                LOGGER.trace("service {} for job {} is valid", serviceId, serviceJobKey);
                            } else {
                                LOGGER.error(
                                        "service {} has an invalid componentId {}. Job will be deleted and service will be disabled",
                                        serviceId,
                                        serviceJobKey);
                                serviceEntity.setEnabled(false);
                                serviceRepository.save(serviceEntity);
                                scheduler.deleteJob(serviceJobKey);
                            }
                        } else {
                            LOGGER.error(
                                    "component id for service {} does not match for job {}. Job will be deleted and service will be disabled",
                                    serviceId,
                                    serviceJobKey);
                            serviceEntity.setEnabled(false);
                            serviceRepository.save(serviceEntity);
                            scheduler.deleteJob(serviceJobKey);
                        }
                    }
                } else {
                    LOGGER.error("job key {} has invalid format for service jobs and will be removed", serviceJobKey);
                    scheduler.deleteJob(serviceJobKey);
                }
            }
        } catch (SchedulerException e) {
            LOGGER.error("scheduling error while cleaning up service jobs: {}", ErrorUtils.aggregate(e), e);
        }
    }

    private String[] extractServiceName(String jobName) {
        return jobName.split("_AS_");
    }
}
