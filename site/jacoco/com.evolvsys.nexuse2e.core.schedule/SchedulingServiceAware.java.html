package com.evolvsys.nexuse2e.core.schedule;

import com.evolvsys.nexuse2e.core.parameter.ParameterDefinition;
import com.evolvsys.nexuse2e.core.parameter.ParameterType;
import com.evolvsys.nexuse2e.core.parameter.Parameters;
import com.evolvsys.nexuse2e.core.schedule.provider.*;
import com.evolvsys.nexuse2e.core.service.RunnableService;
import java.time.Duration;
import java.util.Optional;
import org.apache.commons.lang3.StringUtils;

/**
 * @author Florian Drees
 * @since 30.01.2024
 */
@SuppressWarnings("PMD.ImplicitFunctionalInterface")
public interface SchedulingServiceAware extends RunnableService {
    String PARAM_SCHEDULE_CRON = "schedule_cron";
    String PARAM_SCHEDULE_DURATION = "schedule_duration";
    String PARAM_SCHEDULE_DAILY_AT_HOUR_AND_MINUTE = "schedule_daily_at_hour_and_minute";
    String PARAM_SCHEDULE_MISFIRE = "schedule_misfire";

    void setSchedulingService(SchedulingService schedulingService);

    default ParameterDefinition createCronPatternScheduleParameter() {
        return ParameterDefinition.builder()
                .name(PARAM_SCHEDULE_CRON)
                .displayName("Cron expression")
                .description("Cron expression to schedule this service")
                .type(ParameterType.CRON_PATTERN)
                .required(false)
                .build();
    }

    default ParameterDefinition createHourMinuteScheduleParameter() {
        return ParameterDefinition.builder()
                .name(PARAM_SCHEDULE_DAILY_AT_HOUR_AND_MINUTE)
                .displayName("Hour/Minute pattern")
                .description("Pattern in format HH:mm to schedule this service")
                .type(ParameterType.STRING)
                .required(false)
                .build();
    }

    default ParameterDefinition createDurationScheduleParameter() {
        return ParameterDefinition.builder()
                .name(PARAM_SCHEDULE_DURATION)
                .displayName("Interval")
                .description("Interval to schedule this service")
                .type(ParameterType.DURATION)
                .required(false)
                .build();
    }

    default ParameterDefinition createMisfireScheduleParameter() {
        return ParameterDefinition.builder()
                .name(PARAM_SCHEDULE_MISFIRE)
                .displayName("Misfire instruction")
                .description(
                        "Controls if the service should run immediate if it missed its scheduled time (FIRE_ONCE) or wait for the next scheduled time (NEXT_SCHEDULE)")
                .type(ParameterType.DROPDOWN_LIST)
                .option(MisfireInstruction.NEXT_SCHEDULE.name())
                .option(MisfireInstruction.FIRE_ONCE.name())
                .defaultValue(MisfireInstruction.NEXT_SCHEDULE.name())
                .build();
    }

    default Optional<ScheduleProvider<?>> createScheduleProvider(Parameters parameters) {
        ScheduleProvider<?> scheduleProvider = null;
        if (StringUtils.isNotEmpty(parameters.value(PARAM_SCHEDULE_CRON))) {
            scheduleProvider = CronScheduleProvider.create(
                    parameters.value(PARAM_SCHEDULE_CRON),
                    MisfireInstruction.valueOf(parameters.value(PARAM_SCHEDULE_MISFIRE)));
        }
        if (scheduleProvider == null) {
            if (StringUtils.isNotEmpty(parameters.value(PARAM_SCHEDULE_DAILY_AT_HOUR_AND_MINUTE))) {
                scheduleProvider = HourAndMinuteIntervalScheduleProvider.create(
                        parameters.value(PARAM_SCHEDULE_DAILY_AT_HOUR_AND_MINUTE),
                        MisfireInstruction.valueOf(parameters.value(PARAM_SCHEDULE_MISFIRE)));
            }
        }
        if (scheduleProvider == null) {
            if (parameters.value(PARAM_SCHEDULE_DURATION) != null) {
                scheduleProvider = MillisecondScheduleProvider.create(
                        parameters
                                .value(PARAM_SCHEDULE_DURATION, Duration.class)
                                .toMillis(),
                        MisfireInstruction.valueOf(parameters.value(PARAM_SCHEDULE_MISFIRE)));
            }
        }
        return Optional.ofNullable(scheduleProvider);
    }
}
