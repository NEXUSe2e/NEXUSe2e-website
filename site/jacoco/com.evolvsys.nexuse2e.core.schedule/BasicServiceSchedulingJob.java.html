package com.evolvsys.nexuse2e.core.schedule;

import com.evolvsys.nexuse2e.core.model.entity.ServiceEntity;
import com.evolvsys.nexuse2e.core.service.BaseInflowNexusService;
import com.evolvsys.nexuse2e.core.service.ServiceFactory;
import com.evolvsys.nexuse2e.core.service.repository.ServiceRepository;
import com.evolvsys.nexuse2e.core.utils.ErrorUtils;
import lombok.RequiredArgsConstructor;
import lombok.extern.slf4j.Slf4j;
import org.quartz.JobExecutionContext;
import org.quartz.JobKey;
import org.springframework.scheduling.quartz.QuartzJobBean;

/**
 * @author Florian Drees
 * @since 30.01.2024
 */
@RequiredArgsConstructor
@Slf4j
public class BasicServiceSchedulingJob extends QuartzJobBean {
    private final ServiceFactory serviceFactory;
    private final ServiceRepository serviceRepository;

    @Override
    protected void executeInternal(JobExecutionContext context) {
        BaseInflowNexusService service =
                serviceFactory.get((String) context.getMergedJobDataMap().get("service"));

        JobKey jobKey = context.getJobDetail().getKey();

        LOGGER.debug(
                "executing job: {} at startTime {} and previousFireTime {}",
                jobKey,
                context.getTrigger().getStartTime(),
                context.getTrigger().getPreviousFireTime());

        if (!serviceRepository.existsByServiceIdAndEnabledTrue(service.getServiceId())) {
            LOGGER.warn("Service '{}' is not enabled so the scheduling is skipped!", service.getServiceId());
            ServiceEntity serviceEntity =
                    serviceRepository.findByServiceId(service.getServiceId()).orElseThrow();
            serviceEntity.setEnabled(false);
            serviceRepository.save(serviceEntity);
            return;
        }

        try {
            service.execute();
        } catch (Throwable e) {
            LOGGER.error(
                    "Unexpected error occurred while executing service {}: {}",
                    service.getServiceId(),
                    ErrorUtils.aggregate(e),
                    e);
        }

        LOGGER.debug(
                "next execution date of job '{}' is {}",
                jobKey,
                context.getTrigger().getNextFireTime());
    }
}
