package com.evolvsys.nexuse2e.core;

import com.azure.core.credential.TokenCredential;
import com.azure.identity.ClientSecretCredentialBuilder;
import com.azure.identity.DefaultAzureCredentialBuilder;
import com.evolvsys.nexuse2e.core.parameter.ParameterDefinition;
import com.evolvsys.nexuse2e.core.parameter.ParameterType;
import com.evolvsys.nexuse2e.core.parameter.Parameters;
import org.apache.commons.lang3.StringUtils;

/**
 * <p>Interface to provide azure authentication for Nexus services-</p>
 *
 * @author Florian Drees
 * @since 02.07.2024
 */
public interface AzureAuthenticationProvider {
    String PARAM_CLIENT_ID = "client_id";
    String PARAM_CLIENT_SECRET = "client_secret";
    String PARAM_TENANT_ID = "tenant_id";

    default TokenCredential getTokenCredential(Parameters parameters) {
        String clientId = parameters.value(PARAM_CLIENT_ID);
        String clientSecret = parameters.value(PARAM_CLIENT_SECRET);
        String tenantId = parameters.value(PARAM_TENANT_ID);
        TokenCredential tokenCredential;
        if (StringUtils.isNoneBlank(clientId, clientSecret, tenantId)) {
            tokenCredential = new ClientSecretCredentialBuilder()
                    .clientId(clientId)
                    .clientSecret(clientSecret)
                    .tenantId(tenantId)
                    .build();
        } else {
            tokenCredential = new DefaultAzureCredentialBuilder().build();
        }
        return tokenCredential;
    }

    default ParameterDefinition createClientIdParameter() {
        return ParameterDefinition.builder()
                .name(PARAM_CLIENT_ID)
                .displayName("Client id")
                .description("The client id to authenticate against the keyvault")
                .type(ParameterType.TRIMMED_STRING)
                .build();
    }

    default ParameterDefinition createClientSecretParameter() {
        return ParameterDefinition.builder()
                .name(PARAM_CLIENT_SECRET)
                .displayName("Client secret")
                .description("The client secret to authenticate against the keyvault")
                .type(ParameterType.PASSWORD)
                .build();
    }

    default ParameterDefinition createTenantIdParameter() {
        return ParameterDefinition.builder()
                .name(PARAM_TENANT_ID)
                .displayName("Tenant id")
                .description(
                        "The tenant id to authenticate against the keyvault. Only Required with clientId and clientSecret.")
                .type(ParameterType.TRIMMED_STRING)
                .build();
    }
}
