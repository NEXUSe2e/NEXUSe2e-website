package com.evolvsys.nexuse2e.core.action;

import com.evolvsys.nexuse2e.core.pipeline.model.Pipeline;
import java.util.Set;
import lombok.Builder;
import lombok.Singular;
import lombok.Value;

/**
 * @author SvenBarden
 * @since 06.10.2023
 */
@Value
@Builder
public class Action {
    Long id;
    Long choreographyRefId;
    String actionId;
    String choreographyId;
    Pipeline forwardPipeline;
    Pipeline returnPipeline;
    Pipeline acknowledgmentPipeline;
    boolean validStartAction;
    boolean validTerminationAction;

    @Singular
    Set<String> validNextActions;

    public boolean hasReturnPipeline() {
        return returnPipeline != null;
    }

    public boolean hasAcknowledgmentPipeline() {
        return acknowledgmentPipeline != null;
    }
}
