package com.evolvsys.nexuse2e.core.action;

import com.evolvsys.nexuse2e.core.action.mapper.ContextActionMapper;
import com.evolvsys.nexuse2e.core.action.model.ActionEntity;
import com.evolvsys.nexuse2e.core.action.repository.ActionRepository;
import com.evolvsys.nexuse2e.core.message.context.ActionKey;
import com.evolvsys.nexuse2e.core.message.context.ContextObjectFactory;
import lombok.RequiredArgsConstructor;
import lombok.extern.slf4j.Slf4j;
import org.springframework.stereotype.Component;
import org.springframework.transaction.annotation.Transactional;

/**
 * @author SvenBarden
 * @since 17.10.2023
 */
@Component
@RequiredArgsConstructor
@Slf4j
public class ActionFactory implements ContextObjectFactory<ActionKey, Action> {
    private final ActionRepository actionRepository;
    private final ContextActionMapper contextActionMapper;

    private Action entityToAction(ActionEntity actionEntity) {
        return contextActionMapper.entityToAction(actionEntity);
    }

    @Transactional(readOnly = true)
    public Action getAction(String choreographyId, String actionId) {
        return actionRepository.findByActionIdAndChoreography_ChoreographyId(actionId, choreographyId).stream()
                .filter(actionEntity ->
                        actionEntity.getChoreography().getChoreographyId().equals(choreographyId)
                                && actionEntity.getActionId().equals(actionId))
                .findFirst()
                .map(this::entityToAction)
                .orElseThrow(() -> new IllegalArgumentException(
                        "no action for choreography " + choreographyId + " and action " + actionId));
    }

    @Override
    @Transactional(readOnly = true)
    public Action create(ActionKey actionKey) {
        return actionKey.getActionId() == null || actionKey.getChoreographyId() == null
                ? null
                : getAction(actionKey.getChoreographyId(), actionKey.getActionId());
    }
}
