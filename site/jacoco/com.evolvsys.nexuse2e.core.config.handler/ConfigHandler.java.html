package com.evolvsys.nexuse2e.core.config.handler;

import com.evolvsys.app.core.util.GenericUtils;
import com.evolvsys.nexuse2e.core.config.model.ConfigContext;
import com.evolvsys.nexuse2e.core.config.model.ConfigExportContext;
import com.evolvsys.nexuse2e.core.config.model.NexusConfig;
import java.util.ArrayList;
import java.util.Collection;
import java.util.List;
import java.util.function.BiConsumer;
import java.util.function.BiFunction;
import java.util.function.Function;
import org.apache.commons.lang3.StringUtils;

/**
 * @author SvenBarden
 * @since 26.10.2023
 */
public interface ConfigHandler<T> {

    default Class<T> type() {
        return GenericUtils.getGenericType(getClass(), ConfigHandler.class, 0);
    }

    void clear();

    void importConfig(ConfigContext configContext, T config);

    default void postImportConfig(ConfigContext configContext, T config) {
        // no default implementation
    }

    void exportConfig(ConfigExportContext configExportContext);

    List<T> merge(Collection<T> oldConfigs, Collection<T> newConfigs);

    static <T, IT> List<T> defaultCollectionMerge(
            Collection<T> oldConfigs,
            Collection<T> newConfigs,
            Function<T, IT> idFunction,
            BiConsumer<T, T> mergeFunction,
            BiFunction<IT, IT, Boolean> equalsFunction) {
        List<T> mergedConfigs = new ArrayList<>(oldConfigs);

        for (T oldConfig : oldConfigs) {
            newConfigs.stream()
                    .filter(config -> equalsFunction.apply(idFunction.apply(config), idFunction.apply(oldConfig)))
                    .findFirst()
                    .ifPresent(newConfig -> mergeFunction.accept(newConfig, oldConfig));
        }

        for (T newConfig : newConfigs) {
            if (mergedConfigs.stream()
                    .noneMatch(config -> equalsFunction.apply(idFunction.apply(config), idFunction.apply(newConfig)))) {
                mergedConfigs.add(newConfig);
            }
        }

        return mergedConfigs;
    }

    default <IT> List<T> defaultMerge(
            Collection<T> oldConfigs,
            Collection<T> newConfigs,
            Function<T, IT> idFunction,
            BiConsumer<T, T> mergeFunction,
            BiFunction<IT, IT, Boolean> equalsFunction) {
        return defaultCollectionMerge(oldConfigs, newConfigs, idFunction, mergeFunction, equalsFunction);
    }

    default List<T> defaultMerge(
            Collection<T> oldConfigs,
            Collection<T> newConfigs,
            Function<T, String> idFunction,
            BiConsumer<T, T> mergeFunction) {
        return defaultCollectionMerge(oldConfigs, newConfigs, idFunction, mergeFunction, StringUtils::equalsIgnoreCase);
    }

    default void sort(NexusConfig nexusConfig) {}

    default void sort(T config) {}
}
