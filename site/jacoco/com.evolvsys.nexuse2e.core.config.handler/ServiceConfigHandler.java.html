package com.evolvsys.nexuse2e.core.config.handler;

import com.evolvsys.nexuse2e.core.config.handler.mapper.ServiceConfigMapper;
import com.evolvsys.nexuse2e.core.config.model.*;
import com.evolvsys.nexuse2e.core.model.entity.ServiceEntity;
import com.evolvsys.nexuse2e.core.service.listener.ServiceChangeListenerService;
import com.evolvsys.nexuse2e.core.service.repository.ServiceRepository;
import java.util.Collection;
import java.util.Comparator;
import java.util.List;
import lombok.RequiredArgsConstructor;
import lombok.extern.slf4j.Slf4j;
import org.springframework.core.annotation.Order;
import org.springframework.stereotype.Component;
import org.springframework.transaction.annotation.Transactional;

/**
 * @author SvenBarden
 * @since 26.10.2023
 */
@Component
@RequiredArgsConstructor
@Slf4j
@Order(4)
public class ServiceConfigHandler implements ConfigHandler<ServiceConfig> {
    private final ServiceRepository serviceRepository;
    private final ServiceConfigMapper serviceConfigMapper;
    private final ServiceChangeListenerService serviceChangeListenerService;

    private ServiceEntity create() {
        return new ServiceEntity();
    }

    @Override
    public void clear() {
        serviceChangeListenerService.preDeleteAll();
        serviceRepository.deleteAll();
    }

    @Override
    @Transactional
    public void postImportConfig(ConfigContext configContext, ServiceConfig config) {
        importConfig(configContext, config);
    }

    @Override
    @Transactional
    public void importConfig(ConfigContext configContext, ServiceConfig config) {
        ServiceEntity serviceEntity =
                serviceRepository.findByServiceId(config.getServiceId()).orElseGet(this::create);

        serviceConfigMapper.updateEntity(config, serviceEntity);

        serviceChangeListenerService.preSave(serviceEntity);

        serviceRepository.save(serviceEntity);

        serviceChangeListenerService.postSave(serviceEntity);
    }

    @Override
    public void exportConfig(ConfigExportContext configExportContext) {
        serviceRepository.findAll().stream()
                .map(serviceConfigMapper::toConfig)
                .forEach(serviceConfig ->
                        configExportContext.getConfig().getServices().add(serviceConfig));
    }

    @Override
    public void sort(NexusConfig config) {
        config.getServices().sort(Comparator.comparing(ServiceConfig::getServiceId));
        config.getServices().forEach(this::sort);
    }

    @Override
    public void sort(ServiceConfig config) {
        config.getParameters().sort(Comparator.comparing(ParameterConfig::getName));
    }

    @Override
    public List<ServiceConfig> merge(Collection<ServiceConfig> oldConfigs, Collection<ServiceConfig> newConfigs) {
        return defaultMerge(oldConfigs, newConfigs, ServiceConfig::getServiceId, serviceConfigMapper::merge);
    }
}
