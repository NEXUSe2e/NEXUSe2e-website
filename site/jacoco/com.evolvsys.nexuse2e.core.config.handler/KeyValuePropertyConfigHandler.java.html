package com.evolvsys.nexuse2e.core.config.handler;

import com.evolvsys.nexuse2e.core.config.handler.mapper.KeyValuePropertyConfigMapper;
import com.evolvsys.nexuse2e.core.config.model.ConfigContext;
import com.evolvsys.nexuse2e.core.config.model.ConfigExportContext;
import com.evolvsys.nexuse2e.core.config.model.KeyValuePropertyConfig;
import com.evolvsys.nexuse2e.core.config.model.NexusConfig;
import com.evolvsys.nexuse2e.core.keyvalue.persistence.model.KeyValuePropertyEntity;
import com.evolvsys.nexuse2e.core.keyvalue.persistence.repository.KeyValuePropertyRepository;
import java.util.Collection;
import java.util.Comparator;
import java.util.List;
import lombok.RequiredArgsConstructor;
import lombok.Value;
import org.apache.commons.collections4.comparators.ComparatorChain;
import org.apache.commons.collections4.comparators.TransformingComparator;
import org.springframework.stereotype.Component;

/**
 * @author SvenBarden
 * @since 17.06.2024
 */
@Component
@RequiredArgsConstructor
public class KeyValuePropertyConfigHandler implements ConfigHandler<KeyValuePropertyConfig> {
    private static final Comparator<KeyValuePropertyConfig> COMPARATOR = new ComparatorChain<>(List.of(
            new TransformingComparator<>(KeyValuePropertyConfig::getCategory),
            new TransformingComparator<>(KeyValuePropertyConfig::getKey)));
    private final KeyValuePropertyRepository keyValuePropertyRepository;
    private final KeyValuePropertyConfigMapper keyValuePropertyConfigMapper;

    @Override
    public void clear() {
        keyValuePropertyRepository.deleteAllInBatch();
    }

    @Override
    public void importConfig(ConfigContext configContext, KeyValuePropertyConfig config) {
        KeyValuePropertyEntity keyValuePropertyEntity = keyValuePropertyRepository
                .findByCategoryAndKey(config.getCategory(), config.getKey())
                .orElseGet(KeyValuePropertyEntity::new);

        keyValuePropertyConfigMapper.updateEntity(config, keyValuePropertyEntity);
        keyValuePropertyRepository.save(keyValuePropertyEntity);
    }

    @Override
    public void exportConfig(ConfigExportContext configExportContext) {
        keyValuePropertyRepository.findAll().stream()
                .map(keyValuePropertyConfigMapper::toConfig)
                .forEach(keyValuePropertyConfig ->
                        configExportContext.getConfig().getKeyValueProperties().add(keyValuePropertyConfig));
    }

    @Override
    public List<KeyValuePropertyConfig> merge(
            Collection<KeyValuePropertyConfig> oldConfigs, Collection<KeyValuePropertyConfig> newConfigs) {
        return defaultMerge(
                oldConfigs,
                newConfigs,
                keyValuePropertyConfig ->
                        new KeyValueKey(keyValuePropertyConfig.getCategory(), keyValuePropertyConfig.getKey()),
                keyValuePropertyConfigMapper::merge,
                (keyValueKey, keyValueKey2) -> keyValueKey.name.equalsIgnoreCase(keyValueKey2.name)
                        && keyValueKey.key.equalsIgnoreCase(keyValueKey2.key));
    }

    @Override
    public void sort(NexusConfig nexusConfig) {
        nexusConfig.getKeyValueProperties().sort(COMPARATOR);
    }

    @Value
    private static class KeyValueKey {
        String name;
        String key;
    }
}
