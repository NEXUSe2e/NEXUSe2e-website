package com.evolvsys.nexuse2e.core.config.handler;

import com.evolvsys.nexuse2e.core.certificate.model.CertificateContainerType;
import com.evolvsys.nexuse2e.core.certificate.repository.CertificateContainerRepository;
import com.evolvsys.nexuse2e.core.config.handler.mapper.RouteConfigMapper;
import com.evolvsys.nexuse2e.core.config.model.ConfigContext;
import com.evolvsys.nexuse2e.core.config.model.ConfigExportContext;
import com.evolvsys.nexuse2e.core.config.model.NexusConfig;
import com.evolvsys.nexuse2e.core.config.model.RouteConfig;
import com.evolvsys.nexuse2e.core.exception.NexusError;
import com.evolvsys.nexuse2e.core.exception.NexusRuntimeException;
import com.evolvsys.nexuse2e.core.model.entity.RouteEntity;
import com.evolvsys.nexuse2e.core.partner.repository.PartnerConnectionRepository;
import com.evolvsys.nexuse2e.core.partner.repository.PartnerRepository;
import com.evolvsys.nexuse2e.core.route.repository.RouteRepository;
import java.util.Collection;
import java.util.Comparator;
import java.util.List;
import lombok.RequiredArgsConstructor;
import lombok.extern.slf4j.Slf4j;
import org.apache.commons.lang3.StringUtils;
import org.springframework.core.annotation.Order;
import org.springframework.stereotype.Component;

/**
 * @author Florian Drees
 * @since 08.04.2024
 */
@Component
@RequiredArgsConstructor
@Slf4j
@Order(2)
public class RouteConfigHandler implements ConfigHandler<RouteConfig> {
    private static final String PARTNER = "partner";
    private final RouteRepository routeRepository;
    private final RouteConfigMapper routeConfigMapper;
    private final PartnerConnectionRepository partnerConnectionRepository;
    private final PartnerRepository partnerRepository;
    private final CertificateContainerRepository certificateContainerRepository;

    @Override
    public void clear() {
        routeRepository.deleteAll();
    }

    @Override
    public void importConfig(ConfigContext configContext, RouteConfig config) {
        LOGGER.trace("handling route with name {}", config.getRouteId());

        RouteEntity routeEntity =
                routeRepository.findByRouteId(config.getRouteId()).orElseGet(RouteEntity::new);

        routeConfigMapper.updateEntity(config, routeEntity);

        routeEntity.setFirstPartner(partnerRepository
                .findByPartnerId(config.getFirstPartner())
                .orElseThrow(() -> new NexusRuntimeException(NexusError.builder()
                        .message("first partner could not be found")
                        .extension(PARTNER, config.getFirstPartner())
                        .build())));
        if (StringUtils.isNotBlank(config.getFirstPartnerConnection())) {
            routeEntity.setFirstPartnerConnection(partnerConnectionRepository
                    .findByConnectionIdAndPartner_PartnerId(
                            config.getFirstPartnerConnection(), config.getFirstPartner())
                    .orElseThrow(() -> new NexusRuntimeException(NexusError.builder()
                            .message("connection for first partner not found")
                            .extension("connection", config.getFirstPartnerConnection())
                            .extension(PARTNER, config.getFirstPartner())
                            .build())));
        } else {
            routeEntity.setFirstPartnerConnection(null);
        }

        if (StringUtils.isNotBlank(config.getFirstNexusCertificateContainer())) {
            routeEntity.setFirstNexusCertificateContainer(certificateContainerRepository
                    .findByNameAndType(config.getFirstNexusCertificateContainer(), CertificateContainerType.PRIVATE_KEY)
                    .orElseThrow(() -> new NexusRuntimeException(NexusError.builder()
                            .message("first certificate container not found")
                            .extension(PARTNER, config.getFirstPartner())
                            .extension("container", config.getFirstNexusCertificateContainer())
                            .build())));
        } else {
            routeEntity.setFirstNexusCertificateContainer(null);
        }

        routeEntity.setSecondPartner(partnerRepository
                .findByPartnerId(config.getSecondPartner())
                .orElseThrow(() -> new NexusRuntimeException(NexusError.builder()
                        .message("second partner could not be found")
                        .extension(PARTNER, config.getSecondPartner())
                        .build())));
        if (StringUtils.isNotBlank(config.getSecondPartnerConnection())) {
            routeEntity.setSecondPartnerConnection(partnerConnectionRepository
                    .findByConnectionIdAndPartner_PartnerId(
                            config.getSecondPartnerConnection(), config.getSecondPartner())
                    .orElseThrow(() -> new NexusRuntimeException(NexusError.builder()
                            .message("connection for second partner not found")
                            .extension("connection", config.getSecondPartnerConnection())
                            .extension(PARTNER, config.getSecondPartner())
                            .build())));
        } else {
            routeEntity.setSecondPartnerConnection(null);
        }

        if (StringUtils.isNotBlank(config.getSecondNexusCertificateContainer())) {
            routeEntity.setSecondNexusCertificateContainer(certificateContainerRepository
                    .findByNameAndType(
                            config.getSecondNexusCertificateContainer(), CertificateContainerType.PRIVATE_KEY)
                    .orElseThrow(() -> new NexusRuntimeException(NexusError.builder()
                            .message("second certificate container not found")
                            .extension(PARTNER, config.getSecondPartner())
                            .extension("container", config.getSecondNexusCertificateContainer())
                            .build())));
        } else {
            routeEntity.setSecondNexusCertificateContainer(null);
        }

        routeEntity.setAllowSamePartnerType(Boolean.TRUE.equals(config.getAllowSamePartnerType()));

        routeRepository.save(routeEntity);
    }

    @Override
    public void exportConfig(ConfigExportContext configExportContext) {
        routeRepository.findAll().stream()
                .map(routeConfigMapper::toConfig)
                .forEach(config -> configExportContext.getConfig().getRoutes().add(config));
    }

    @Override
    public List<RouteConfig> merge(Collection<RouteConfig> oldConfigs, Collection<RouteConfig> newConfigs) {
        return defaultMerge(oldConfigs, newConfigs, RouteConfig::getRouteId, routeConfigMapper::merge);
    }

    @Override
    public void sort(NexusConfig nexusConfig) {
        nexusConfig.getRoutes().sort(Comparator.comparing(RouteConfig::getRouteId));
    }
}
