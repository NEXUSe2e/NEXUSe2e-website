package com.evolvsys.nexuse2e.core.config.handler;

import com.evolvsys.nexuse2e.core.certificate.model.CertificateContainerEntity;
import com.evolvsys.nexuse2e.core.certificate.model.CertificateContainerType;
import com.evolvsys.nexuse2e.core.certificate.model.CertificateEntity;
import com.evolvsys.nexuse2e.core.certificate.model.CertificateType;
import com.evolvsys.nexuse2e.core.certificate.repository.CertificateContainerRepository;
import com.evolvsys.nexuse2e.core.certificate.repository.CertificateRepository;
import com.evolvsys.nexuse2e.core.config.handler.mapper.CertificateConfigMapper;
import com.evolvsys.nexuse2e.core.config.handler.mapper.CertificateContainerConfigMapper;
import com.evolvsys.nexuse2e.core.config.model.*;
import java.time.ZonedDateTime;
import java.util.Collection;
import java.util.Comparator;
import java.util.List;
import lombok.RequiredArgsConstructor;
import lombok.extern.slf4j.Slf4j;
import org.apache.commons.collections4.CollectionUtils;
import org.springframework.stereotype.Component;
import org.springframework.transaction.annotation.Transactional;

/**
 * @author SvenBarden
 * @since 13.02.2024
 */
@Component
@RequiredArgsConstructor
@Slf4j
public class ClientKeyConfigHandler implements ConfigHandler<ClientKeyContainerConfig> {
    private final CertificateRepository certificateRepository;
    private final CertificateConfigMapper certificateConfigMapper;
    private final CertificateContainerRepository certificateContainerRepository;
    private final CertificateContainerConfigMapper certificateContainerConfigMapper;

    private CertificateContainerEntity create() {
        return new CertificateContainerEntity();
    }

    @Override
    public void clear() {
        certificateContainerRepository.deleteAll();
    }

    @Override
    @Transactional
    public void importConfig(ConfigContext configContext, ClientKeyContainerConfig config) {
        CertificateContainerEntity certificateContainerEntity = certificateContainerRepository
                .findByNameAndType(config.getName(), CertificateContainerType.PRIVATE_KEY)
                .orElseGet(this::create);

        certificateContainerEntity.setName(config.getName());
        certificateContainerEntity.setDescription(config.getDescription());
        certificateContainerEntity.setType(CertificateContainerType.PRIVATE_KEY);

        certificateContainerEntity = certificateContainerRepository.save(certificateContainerEntity);

        if (CollectionUtils.isNotEmpty(config.getClientKeys())) {
            for (ClientKeyConfig clientKey : config.getClientKeys()) {
                CertificateEntity certificateEntity = certificateRepository
                        .findByCertificateContainerAndName(certificateContainerEntity, clientKey.getName())
                        .orElseGet(CertificateEntity::new);

                certificateEntity.setCertificateContainer(certificateContainerEntity);
                certificateEntity.setType(CertificateType.LOCAL);

                boolean wasDisabled =
                        certificateEntity.isActive() && (Boolean.TRUE.equals(clientKey.getActive()) == false);

                certificateConfigMapper.updateEntity(clientKey, certificateEntity);

                if (certificateEntity.isActive() && certificateEntity.getEnabled() == null) {
                    certificateEntity.setEnabled(ZonedDateTime.now());
                }
                if (wasDisabled) {
                    certificateEntity.setDisabled(ZonedDateTime.now());
                }

                certificateRepository.save(certificateEntity);
            }
        }
    }

    @Override
    public void exportConfig(ConfigExportContext configExportContext) {
        certificateContainerRepository.findByType(CertificateContainerType.PRIVATE_KEY).stream()
                .map(certificateContainerConfigMapper::toClientKeyConfig)
                .forEach(certificateContainerConfig ->
                        configExportContext.getConfig().getClientKeyContainers().add(certificateContainerConfig));
    }

    @Override
    public List<ClientKeyContainerConfig> merge(
            Collection<ClientKeyContainerConfig> oldConfigs, Collection<ClientKeyContainerConfig> newConfigs) {
        return defaultMerge(
                oldConfigs, newConfigs, ClientKeyContainerConfig::getName, certificateContainerConfigMapper::merge);
    }

    @Override
    public void sort(NexusConfig nexusConfig) {
        nexusConfig.getClientKeyContainers().sort(Comparator.comparing(ClientKeyContainerConfig::getName));
        nexusConfig.getClientKeyContainers().forEach(this::sort);
    }

    @Override
    public void sort(ClientKeyContainerConfig config) {
        config.getClientKeys().sort(Comparator.comparing(ClientKeyConfig::getName));
    }
}
