package com.evolvsys.nexuse2e.core.config.handler;

import com.evolvsys.nexuse2e.core.action.model.ActionEntity;
import com.evolvsys.nexuse2e.core.choreography.repository.ChoreographyRepository;
import com.evolvsys.nexuse2e.core.config.handler.mapper.ChoreographyConfigMapper;
import com.evolvsys.nexuse2e.core.config.model.*;
import com.evolvsys.nexuse2e.core.model.entity.ChoreographyEntity;
import com.evolvsys.nexuse2e.core.pipeline.repository.PipelineRepository;
import com.evolvsys.nexuse2e.core.route.repository.RouteRepository;
import java.util.*;
import java.util.stream.Collectors;
import lombok.RequiredArgsConstructor;
import lombok.extern.slf4j.Slf4j;
import org.apache.commons.collections4.CollectionUtils;
import org.apache.commons.lang3.ObjectUtils;
import org.springframework.core.annotation.Order;
import org.springframework.stereotype.Component;
import org.springframework.transaction.annotation.Transactional;

/**
 * @author SvenBarden
 * @since 26.10.2023
 */
@Component
@RequiredArgsConstructor
@Slf4j
@Order(1)
public class ChoreographyConfigHandler implements ConfigHandler<ChoreographyConfig> {
    private final ChoreographyRepository choreographyRepository;
    private final ChoreographyConfigMapper choreographyConfigMapper;
    private final PipelineRepository pipelineRepository;
    private final RouteRepository routeRepository;

    private ChoreographyEntity create() {
        return new ChoreographyEntity();
    }

    protected void updateAction(ActionConfig config, ActionEntity actionEntity) {
        actionEntity.setActionId(config.getActionId());
        actionEntity.setValidStartAction(ObjectUtils.defaultIfNull(config.getValidStartAction(), Boolean.FALSE));
        actionEntity.setValidTerminationAction(
                ObjectUtils.defaultIfNull(config.getValidTerminationAction(), Boolean.FALSE));

        actionEntity.setReturnPipeline(
                pipelineRepository.findByPipelineId(config.getReturnPipeline()).orElse(null));
        actionEntity.setForwardPipeline(
                pipelineRepository.findByPipelineId(config.getForwardPipeline()).orElse(null));
        actionEntity.setAcknowledgmentPipeline(pipelineRepository
                .findByPipelineId(config.getAcknowledgmentPipeline())
                .orElse(null));

        actionEntity.getRoutes().clear();
        if (CollectionUtils.isNotEmpty(config.getRoutes())) {
            for (RouteLinkConfig route : config.getRoutes()) {
                LOGGER.trace("adding route to action {}", route.getRouteId());
                routeRepository
                        .findByRouteId(route.getRouteId())
                        .ifPresent(routeEntity -> actionEntity.getRoutes().add(routeEntity));
            }
        }
    }

    @Override
    public void clear() {
        choreographyRepository.deleteAll();
    }

    @Override
    @Transactional
    public void importConfig(ConfigContext configContext, ChoreographyConfig config) {
        LOGGER.trace("handling choreography with name {}", config.getChoreographyId());
        ChoreographyEntity choreographyEntity =
                choreographyRepository.findByChoreographyId(config.getChoreographyId()).stream()
                        .filter(entity -> entity.getChoreographyId().equals(config.getChoreographyId()))
                        .findFirst()
                        .orElseGet(this::create);

        choreographyConfigMapper.updateEntity(config, choreographyEntity);

        if (CollectionUtils.isNotEmpty(config.getActions())) {
            Map<String, ActionEntity> actionEntities =
                    HashMap.newHashMap(config.getActions().size());
            for (ActionConfig action : config.getActions()) {
                LOGGER.trace("adding action {}", action);
                ActionEntity actionEntity = choreographyEntity.getActions().stream()
                        .filter(a -> a.getActionId().equals(action.getActionId()))
                        .findFirst()
                        .orElseGet(ActionEntity::new);

                actionEntity.setChoreography(choreographyEntity);

                updateAction(action, actionEntity);

                choreographyEntity.getActions().add(actionEntity);
                actionEntities.put(action.getActionId(), actionEntity);
            }

            for (ActionConfig action : config.getActions()) {
                if (CollectionUtils.isNotEmpty(action.getValidNextActions())) {
                    ActionEntity actionEntity = actionEntities.get(action.getActionId());
                    action.getValidNextActions().stream()
                            .map(actionEntities::get)
                            .filter(Objects::nonNull)
                            .forEach((nextEntity) ->
                                    actionEntity.getValidNextActions().add(nextEntity));
                }
            }
        }

        choreographyEntity.getRoutes().clear();
        if (CollectionUtils.isNotEmpty(config.getRoutes())) {
            for (RouteLinkConfig route : config.getRoutes()) {
                LOGGER.trace("adding route to choreography {}", route.getRouteId());
                routeRepository
                        .findByRouteId(route.getRouteId())
                        .ifPresent(routeEntity -> choreographyEntity.getRoutes().add(routeEntity));
            }
        }

        choreographyRepository.save(choreographyEntity);
    }

    @Override
    public void exportConfig(ConfigExportContext configExportContext) {
        choreographyRepository.findAll().stream()
                .map(choreographyConfigMapper::toConfig)
                .forEach(config ->
                        configExportContext.getConfig().getChoreographies().add(config));
    }

    @Override
    public List<ChoreographyConfig> merge(
            Collection<ChoreographyConfig> oldConfigs, Collection<ChoreographyConfig> newConfigs) {
        return defaultMerge(
                oldConfigs, newConfigs, ChoreographyConfig::getChoreographyId, choreographyConfigMapper::merge);
    }

    @Override
    public void sort(NexusConfig nexusConfig) {
        nexusConfig.getChoreographies().sort(Comparator.comparing(ChoreographyConfig::getChoreographyId));
        nexusConfig.getChoreographies().forEach(this::sort);
    }

    @Override
    public void sort(ChoreographyConfig config) {
        config.getActions().sort(Comparator.comparing(ActionConfig::getActionId));
        config.getRoutes().sort(Comparator.comparing(RouteLinkConfig::getRouteId));
        for (ActionConfig actionConfig : config.getActions()) {
            actionConfig.getRoutes().sort(Comparator.comparing(RouteLinkConfig::getRouteId));
            Set<String> sortedValidNextActions = actionConfig.getValidNextActions().stream()
                    .sorted()
                    .collect(Collectors.toCollection(LinkedHashSet::new));
            actionConfig.setValidNextActions(sortedValidNextActions);
        }
    }
}
