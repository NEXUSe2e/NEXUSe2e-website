package com.evolvsys.nexuse2e.core.config.handler;

import com.evolvsys.nexuse2e.core.config.model.ParameterConfig;
import com.evolvsys.nexuse2e.core.model.entity.ParameterEntity;
import com.evolvsys.nexuse2e.core.parameter.ParameterDefinition;
import com.evolvsys.nexuse2e.core.parameter.converter.TypeConverter;
import com.evolvsys.nexuse2e.core.parameter.service.ParameterConverterRegistry;
import java.util.Collection;
import lombok.RequiredArgsConstructor;
import lombok.extern.slf4j.Slf4j;
import org.apache.commons.collections4.CollectionUtils;
import org.springframework.lang.Nullable;
import org.springframework.stereotype.Component;

/**
 * @author SvenBarden
 * @since 26.10.2023
 */
@Component
@RequiredArgsConstructor
@Slf4j
public class HandlerUtils {
    private final ParameterConverterRegistry parameterConverterRegistry;

    public void mapParametersToEntities(
            @Nullable Collection<ParameterConfig> parameterConfigs, Collection<ParameterEntity> parameterEntities) {
        mapParametersToEntities(parameterConfigs, parameterEntities, null);
    }

    public void mapParametersToEntities(
            @Nullable Collection<ParameterConfig> parameterConfigs,
            Collection<ParameterEntity> parameterEntities,
            Collection<ParameterDefinition> parameterDefinitions) {
        parameterEntities.clear();
        if (CollectionUtils.isNotEmpty(parameterConfigs)) {
            for (ParameterDefinition parameterDefinition : parameterDefinitions) {
                parameterConfigs.stream()
                        .filter(parameterConfig -> parameterConfig.getName().equals(parameterDefinition.getName()))
                        .findFirst()
                        .ifPresent(parameterConfig -> parameterEntities.add(
                                mapConfigWithDefinitionToEntity(parameterConfig, parameterDefinition)));
            }
        }
    }

    private ParameterEntity mapConfigWithDefinitionToEntity(ParameterConfig config, ParameterDefinition definition) {
        LOGGER.trace(
                "use parameter definition for '{}' with type '{}'",
                definition.getName(),
                definition.getType().name());

        ParameterEntity entity = new ParameterEntity();
        entity.setName(definition.getName());
        entity.setType(definition.getType());

        TypeConverter<?, ?> typeConverter = parameterConverterRegistry.getTypeConverter(definition.getType());

        typeConverter.updateFromConfig(entity, config.getValues());
        return entity;
    }

    public void mapEntityValueToConfig(ParameterEntity entity, ParameterConfig config) {
        TypeConverter<?, ?> typeConverter = parameterConverterRegistry.getTypeConverter(entity.getType());
        config.setValues(typeConverter.toConfigValue(entity));
    }
}
