package com.evolvsys.nexuse2e.core.config.handler;

import com.evolvsys.nexuse2e.core.config.handler.mapper.PipeletConfigMapper;
import com.evolvsys.nexuse2e.core.config.handler.mapper.PipelineConfigMapper;
import com.evolvsys.nexuse2e.core.config.model.*;
import com.evolvsys.nexuse2e.core.model.entity.PipeletEntity;
import com.evolvsys.nexuse2e.core.pipeline.model.PipelineEntity;
import com.evolvsys.nexuse2e.core.pipeline.repository.PipelineRepository;
import java.util.Collection;
import java.util.Comparator;
import java.util.List;
import lombok.RequiredArgsConstructor;
import lombok.extern.slf4j.Slf4j;
import org.apache.commons.collections4.CollectionUtils;
import org.springframework.core.annotation.Order;
import org.springframework.stereotype.Component;
import org.springframework.transaction.annotation.Transactional;

/**
 * @author SvenBarden
 * @since 26.10.2023
 */
@Component
@RequiredArgsConstructor
@Slf4j
@Order(5)
public class PipelineConfigHandler implements ConfigHandler<PipelineConfig> {
    private final PipelineRepository pipelineRepository;
    private final PipelineConfigMapper pipelineConfigMapper;
    private final PipeletConfigMapper pipeletConfigMapper;

    private PipelineEntity create() {
        return new PipelineEntity();
    }

    private PipeletEntity create(PipeletConfig config, PipelineEntity pipelineEntity) {
        LOGGER.trace("create pipelet {}", config);
        PipeletEntity pipeletEntity = new PipeletEntity();
        pipelineEntity.getPipelets().add(pipeletEntity);
        return pipeletEntity;
    }

    @Override
    public void clear() {
        pipelineRepository.deleteAll();
    }

    @Override
    @Transactional
    public void postImportConfig(ConfigContext configContext, PipelineConfig config) {
        importConfig(configContext, config);
    }

    @Override
    @Transactional
    public void importConfig(ConfigContext configContext, PipelineConfig config) {
        LOGGER.trace("handling pipeline with pipelineId {}", config.getPipelineId());

        PipelineEntity pipelineEntity =
                pipelineRepository.findByPipelineId(config.getPipelineId()).orElseGet(this::create);

        pipelineConfigMapper.updateEntity(config, pipelineEntity);

        pipelineEntity.getPipelets().clear();

        List<PipeletConfig> pipelets = config.getPipelets();
        if (CollectionUtils.isNotEmpty(pipelets)) {
            for (int i = 0; i < pipelets.size(); i++) {
                PipeletConfig pipelet = pipelets.get(i);
                LOGGER.trace("adding pipelet {}", pipelet.getComponentId());

                PipeletEntity pipeletEntity = create(pipelet, pipelineEntity);
                pipeletEntity.setOrder(i);

                pipeletConfigMapper.updateEntity(pipelet, pipeletEntity);
            }
        }

        pipelineRepository.save(pipelineEntity);
    }

    @Override
    public void exportConfig(ConfigExportContext configExportContext) {
        pipelineRepository.findAll().stream()
                .map(pipelineConfigMapper::toConfig)
                .forEach(
                        config -> configExportContext.getConfig().getPipelines().add(config));
    }

    @Override
    public List<PipelineConfig> merge(Collection<PipelineConfig> oldConfigs, Collection<PipelineConfig> newConfigs) {
        return defaultMerge(oldConfigs, newConfigs, PipelineConfig::getPipelineId, pipelineConfigMapper::merge);
    }

    @Override
    public void sort(NexusConfig nexusConfig) {
        nexusConfig.getPipelines().sort(Comparator.comparing(PipelineConfig::getPipelineId));
        nexusConfig.getPipelines().forEach(this::sort);
    }

    @Override
    public void sort(PipelineConfig config) {
        config.getPipelets().forEach(pipeletConfig -> {
            pipeletConfig.getParameters().sort(Comparator.comparing(ParameterConfig::getName));
        });
    }
}
