package com.evolvsys.nexuse2e.core.route.mapper;

import com.evolvsys.app.core.mapper.DefaultMapperConfiguration;
import com.evolvsys.nexuse2e.core.certificate.mapper.CertificateContainerKeyMapper;
import com.evolvsys.nexuse2e.core.certificate.model.ContextKeyStore;
import com.evolvsys.nexuse2e.core.model.entity.RouteEntity;
import com.evolvsys.nexuse2e.core.route.model.ContextRoute;
import org.mapstruct.Mapper;
import org.mapstruct.Mapping;
import org.springframework.beans.factory.annotation.Autowired;
import org.springframework.data.util.Lazy;

/**
 * @author SvenBarden
 * @since 12.02.2024
 */
@Mapper(config = DefaultMapperConfiguration.class)
public abstract class ContextRouteMapper {
    private CertificateContainerKeyMapper certificateContainerKeyMapper;

    @Autowired
    protected void config(CertificateContainerKeyMapper certificateContainerKeyMapper) {
        this.certificateContainerKeyMapper = certificateContainerKeyMapper;
    }

    @Mapping(target = "sourcePartnerTrustCertificates", ignore = true)
    @Mapping(target = "targetPartnerTrustCertificates", ignore = true)
    @Mapping(target = "sourceConnectionTrustCertificates", ignore = true)
    @Mapping(target = "targetConnectionTrustCertificates", ignore = true)
    @Mapping(target = "targetLocalKey", ignore = true)
    @Mapping(target = "targetConnection", ignore = true)
    @Mapping(target = "sourceLocalKey", ignore = true)
    @Mapping(target = "sourceConnection", ignore = true)
    @Mapping(target = "inflowPipeline", ignore = true)
    public abstract ContextRoute entityToContext(RouteEntity routeEntity);

    protected Lazy<ContextKeyStore> containerIdToLazyKeystore(Long id) {
        return Lazy.of(() -> certificateContainerKeyMapper.idToKeyStore(id));
    }
}
