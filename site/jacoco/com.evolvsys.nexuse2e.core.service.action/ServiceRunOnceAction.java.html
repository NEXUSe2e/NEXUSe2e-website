package com.evolvsys.nexuse2e.core.service.action;

import com.evolvsys.app.core.process.ProcessAction;
import com.evolvsys.app.core.process.ProcessActionChain;
import com.evolvsys.app.core.process.ProcessActionConditions;
import com.evolvsys.app.core.process.ProcessParameters;
import com.evolvsys.nexuse2e.core.exception.NexusException;
import com.evolvsys.nexuse2e.core.model.entity.ServiceEntity;
import com.evolvsys.nexuse2e.core.service.BaseInflowNexusService;
import com.evolvsys.nexuse2e.core.service.ServiceFactory;
import com.evolvsys.nexuse2e.core.utils.ErrorUtils;
import lombok.RequiredArgsConstructor;
import lombok.extern.slf4j.Slf4j;
import org.springframework.stereotype.Component;

/**
 * @author Florian Drees
 * @since 21.02.2024
 */
@Component
@RequiredArgsConstructor
@Slf4j
public class ServiceRunOnceAction implements ProcessAction<ServiceEntity> {
    private final ServiceFactory serviceFactory;

    @Override
    public ServiceEntity execute(ServiceEntity entity, ProcessActionChain chain, ProcessParameters processParameters) {
        BaseInflowNexusService service = serviceFactory.get(entity);
        LOGGER.debug("executing service '{}' with component '{}'", service.getServiceId(), service.getComponentId());
        try {
            service.execute();
        } catch (NexusException e) {
            LOGGER.error("Error while executing service: {}", ErrorUtils.aggregate(e), e);
        }
        return entity;
    }

    @Override
    public ProcessActionConditions conditions(ServiceEntity entity, ProcessParameters processParameters) {
        return ProcessActionConditions.EMPTY;
    }
}
