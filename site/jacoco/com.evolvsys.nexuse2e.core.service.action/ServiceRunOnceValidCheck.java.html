package com.evolvsys.nexuse2e.core.service.action;

import com.evolvsys.app.core.process.ProcessValidCheck;
import com.evolvsys.nexuse2e.core.component.ComponentRegistry;
import com.evolvsys.nexuse2e.core.model.entity.ServiceEntity;
import com.evolvsys.nexuse2e.core.service.RunnableService;
import com.evolvsys.nexuse2e.core.service.model.ServiceDto;
import lombok.RequiredArgsConstructor;
import org.springframework.stereotype.Component;

/**
 * @author Florian Drees
 * @since 21.02.2024
 */
@Component
@RequiredArgsConstructor
public class ServiceRunOnceValidCheck implements ProcessValidCheck<ServiceEntity, ServiceDto> {
    private final ComponentRegistry componentRegistry;

    @Override
    public boolean isValid(ServiceEntity entity) {
        return isValid(entity.getComponentId());
    }

    @Override
    public boolean isValid(ServiceDto dto) {
        return isValid(dto.getComponentId());
    }

    protected boolean isValid(String componentName) {
        return componentRegistry
                .getComponent(componentName)
                .map(component -> RunnableService.class.isAssignableFrom(component.getLoadedClass()))
                .orElse(Boolean.FALSE);
    }
}
