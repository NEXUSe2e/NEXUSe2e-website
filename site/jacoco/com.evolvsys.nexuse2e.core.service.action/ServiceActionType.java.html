package com.evolvsys.nexuse2e.core.service.action;

import com.evolvsys.app.core.process.ProcessActionType;
import com.evolvsys.app.core.process.ProcessDefinition;
import com.evolvsys.nexuse2e.core.model.entity.ServiceEntity;
import com.evolvsys.nexuse2e.core.service.model.ServiceDto;
import lombok.Getter;

/**
 * @author Florian Drees
 * @since 21.02.2024
 */
@Getter
public enum ServiceActionType implements ProcessActionType<ServiceEntity, ServiceDto> {
    RUN_ONCE(ProcessDefinition.<ServiceEntity, ServiceDto>builder()
            .action(ServiceRunOnceAction.class)
            .processValidCheck(ServiceRunOnceValidCheck.class)
            .build()),
    START(ProcessDefinition.<ServiceEntity, ServiceDto>builder()
            .action(ServiceStartAction.class)
            .isValidDto(dto -> !dto.isEnabled())
            .isValidEntity(entity -> !entity.isEnabled())
            .build()),
    STOP(ProcessDefinition.<ServiceEntity, ServiceDto>builder()
            .action(ServiceStopAction.class)
            .isValidDto(ServiceDto::isEnabled)
            .isValidEntity(ServiceEntity::isEnabled)
            .build());

    private final ProcessDefinition<ServiceEntity, ServiceDto> process;

    ServiceActionType(ProcessDefinition<ServiceEntity, ServiceDto> process) {
        this.process = process;
    }
}
