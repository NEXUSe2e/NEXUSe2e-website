package com.evolvsys.nexuse2e.core.choreography.mapper;

import com.evolvsys.app.core.mapper.BaseDtoEntityMapper;
import com.evolvsys.app.core.mapper.DefaultMapperConfiguration;
import com.evolvsys.app.core.mapper.interceptor.InterceptingMappingContext;
import com.evolvsys.nexuse2e.core.choreography.model.ChoreographyDto;
import com.evolvsys.nexuse2e.core.model.entity.ChoreographyEntity;
import com.evolvsys.nexuse2e.core.route.mapper.RouteMapper;
import lombok.extern.slf4j.Slf4j;
import org.mapstruct.Context;
import org.mapstruct.Mapper;
import org.mapstruct.MappingTarget;
import org.springframework.beans.factory.annotation.Autowired;

/**
 * <p>Mapper f√ºr {@link ChoreographyEntity} und {@link ChoreographyDto}.</p>
 *
 * @author Florian Drees
 * @since 18.12.2023
 */
@Mapper(config = DefaultMapperConfiguration.class, uses = RouteMapper.class)
@Slf4j
public abstract class ChoreographyMapper extends BaseDtoEntityMapper<ChoreographyDto, ChoreographyEntity> {
    private BaseChoreographyMapper baseChoreographyMapper;

    @Autowired
    protected void config(BaseChoreographyMapper baseChoreographyMapper) {
        this.baseChoreographyMapper = baseChoreographyMapper;
    }

    @Override
    public ChoreographyEntity idToEntity(Long id, @Context InterceptingMappingContext interceptingMappingContext) {
        return baseChoreographyMapper.idToEntity(id, interceptingMappingContext);
    }

    @Override
    protected ChoreographyDto entityToDto(
            ChoreographyEntity entity, @Context InterceptingMappingContext interceptingMappingContext) {
        return baseChoreographyMapper.entityToDto(entity, interceptingMappingContext);
    }

    @Override
    protected void updateEntity(
            ChoreographyDto dto,
            @MappingTarget ChoreographyEntity entity,
            @Context InterceptingMappingContext interceptingMappingContext) {
        baseChoreographyMapper.updateEntity(dto, entity, interceptingMappingContext);
    }
}
