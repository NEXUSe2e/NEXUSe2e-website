package com.evolvsys.nexuse2e.core.choreography.mapper;

import com.evolvsys.app.core.mapper.DefaultMapperConfiguration;
import com.evolvsys.nexuse2e.core.choreography.model.ContextChoreography;
import com.evolvsys.nexuse2e.core.choreography.repository.ChoreographyRepository;
import com.evolvsys.nexuse2e.core.model.entity.ChoreographyEntity;
import org.mapstruct.Mapper;
import org.springframework.beans.factory.annotation.Autowired;

/**
 * @author SvenBarden
 * @since 24.01.2024
 */
@Mapper(config = DefaultMapperConfiguration.class)
public abstract class ContextChoreographyMapper {
    private ChoreographyRepository choreographyRepository;

    @Autowired
    protected void config(ChoreographyRepository choreographyRepository) {
        this.choreographyRepository = choreographyRepository;
    }

    public ChoreographyEntity nameToEntity(String choreographyId) {
        return choreographyRepository.findByChoreographyId(choreographyId).stream()
                .filter(choreographyEntity ->
                        choreographyEntity.getChoreographyId().equals(choreographyId))
                .findFirst()
                .orElse(null);
    }

    public abstract ContextChoreography entityToChoreography(ChoreographyEntity choreographyEntity);
}
