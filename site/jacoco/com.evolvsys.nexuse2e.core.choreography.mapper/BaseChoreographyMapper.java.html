package com.evolvsys.nexuse2e.core.choreography.mapper;

import com.evolvsys.app.core.mapper.BaseDtoEntityMapper;
import com.evolvsys.app.core.mapper.DefaultMapperConfiguration;
import com.evolvsys.app.core.mapper.interceptor.InterceptingMappingContext;
import com.evolvsys.nexuse2e.core.choreography.model.ChoreographyDto;
import com.evolvsys.nexuse2e.core.choreography.repository.ChoreographyRepository;
import com.evolvsys.nexuse2e.core.model.entity.ChoreographyEntity;
import org.mapstruct.*;
import org.springframework.beans.factory.annotation.Autowired;

/**
 * @author SvenBarden
 * @since 26.01.2024
 */
@Mapper(config = DefaultMapperConfiguration.class)
public abstract class BaseChoreographyMapper extends BaseDtoEntityMapper<ChoreographyDto, ChoreographyEntity> {
    private ChoreographyRepository choreographyRepository;

    @Autowired
    protected void config(ChoreographyRepository choreographyRepository) {
        this.choreographyRepository = choreographyRepository;
    }

    @Override
    public ChoreographyEntity idToEntity(Long id, @Context InterceptingMappingContext interceptingMappingContext) {
        return idToEntity(id, interceptingMappingContext, choreographyRepository::findById);
    }

    @InheritConfiguration
    @Override
    protected abstract ChoreographyDto entityToDto(
            ChoreographyEntity entity, @Context InterceptingMappingContext interceptingMappingContext);

    @Mapping(target = "routes", ignore = true)
    @Mapping(target = "actions", ignore = true)
    @InheritConfiguration
    @Override
    protected abstract void updateEntity(
            ChoreographyDto dto,
            @MappingTarget ChoreographyEntity entity,
            @Context InterceptingMappingContext interceptingMappingContext);
}
