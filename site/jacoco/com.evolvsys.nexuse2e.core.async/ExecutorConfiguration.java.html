package com.evolvsys.nexuse2e.core.async;

import com.evolvsys.nexuse2e.core.logging.MessageLogContextHolder;
import java.util.concurrent.Executor;
import lombok.extern.slf4j.Slf4j;
import org.springframework.context.annotation.Bean;
import org.springframework.context.annotation.Configuration;
import org.springframework.scheduling.concurrent.ThreadPoolTaskExecutor;

/**
 * @author SvenBarden
 * @since 18.10.2023
 */
@Configuration
@Slf4j
public class ExecutorConfiguration {

    /**
     * Default spring task executor used for @{@link org.springframework.scheduling.annotation.Async} annotated methods
     *
     * @return the default task executor
     */
    @Bean
    public Executor taskExecutor() {
        ThreadPoolTaskExecutor executor = new ThreadPoolTaskExecutor();
        executor.setCorePoolSize(10);
        executor.setMaxPoolSize(200);
        executor.setQueueCapacity(150);
        executor.setThreadNamePrefix("async-task-");
        executor.initialize();
        return executor;
    }

    /**
     * Task executor for async message transmission. Uses a special decorator for clearing {@link MessageLogContextHolder}.
     * Can be used with the @{@link MessageExecutor} annotation.
     * <pre>
     * {@code @MessageExecutor
     * private final Executor messageExecutor}
     * </pre>
     *
     * @return the message task executor
     */
    @Bean
    @MessageExecutor
    public Executor messageExecutor() {
        ThreadPoolTaskExecutor executor = new ThreadPoolTaskExecutor();
        executor.setCorePoolSize(5);
        executor.setMaxPoolSize(200);
        executor.setQueueCapacity(150);
        executor.setThreadNamePrefix("async-message-");
        executor.initialize();
        executor.setTaskDecorator(ExecutorConfiguration::logContextRunnable);
        return executor;
    }

    private static Runnable logContextRunnable(Runnable runnable) {
        return () -> {
            runnable.run();
            MessageLogContextHolder.clearContext();
            LOGGER.trace("cleared message log context");
        };
    }
}
