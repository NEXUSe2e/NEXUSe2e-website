package com.evolvsys.nexuse2e.core.partner.service;

import com.evolvsys.app.core.exception.GeneralErrorException;
import com.evolvsys.app.core.graphql.annotation.GraphQLService;
import com.evolvsys.app.core.graphql.dataloader.DataLoaderService;
import com.evolvsys.nexuse2e.core.certificate.model.CertificateContainerDto;
import com.evolvsys.nexuse2e.core.config.service.ConfigGraphQLService;
import com.evolvsys.nexuse2e.core.partner.PartnerFilter;
import com.evolvsys.nexuse2e.core.partner.model.PartnerConnectionDto;
import com.evolvsys.nexuse2e.core.partner.model.PartnerDto;
import com.evolvsys.nexuse2e.core.partner.validation.ValidPartnerHasMessages;
import com.evolvsys.nexuse2e.core.partner.validation.ValidPartnerHasRoutes;
import io.leangen.graphql.annotations.*;
import io.leangen.graphql.execution.ResolutionEnvironment;
import jakarta.validation.Valid;
import java.util.List;
import java.util.Optional;
import java.util.Set;
import java.util.concurrent.CompletableFuture;
import lombok.RequiredArgsConstructor;
import lombok.extern.slf4j.Slf4j;
import org.springframework.dao.DataIntegrityViolationException;

/**
 * @author Florian Drees
 * @since 15.01.2024
 */
@GraphQLService
@RequiredArgsConstructor
@Slf4j
public class PartnerGraphQLService {
    private final PartnerService partnerService;
    private final DataLoaderService dataLoaderService;

    @GraphQLMutation(name = "partner_update", description = "Update an existing or create a new partner")
    @GraphQLNonNull
    public PartnerDto update(@Valid @GraphQLNonNull PartnerDto partnerDto) {
        return ConfigGraphQLService.notifySubscribers(partnerService.save(partnerDto));
    }

    @GraphQLMutation(
            name = "partner_updateWithConnections",
            description = "Updates an existing or creates a new partner with the provided connections")
    @GraphQLNonNull
    public PartnerDto update(
            @GraphQLNonNull @Valid PartnerDto partnerDto,
            @Valid List<@GraphQLNonNull PartnerConnectionDto> connections) {
        try {
            return ConfigGraphQLService.notifySubscribers(partnerService.save(partnerDto, connections));
        } catch (DataIntegrityViolationException e) {
            throw new GeneralErrorException(
                    "At least one partner connection cannot be updated. Entities are referencing to these entities.",
                    e);
        }
    }

    @GraphQLMutation(name = "partner_delete", description = "Deletes a partner")
    public boolean delete(@ValidPartnerHasMessages @ValidPartnerHasRoutes long partnerId) {
        partnerService.delete(partnerId);
        ConfigGraphQLService.notifySubscribers();
        return true;
    }

    @GraphQLQuery(name = "partner_list", description = "List all partners")
    @GraphQLNonNull
    public List<@GraphQLNonNull PartnerDto> list(PartnerFilter partnerFilter) {
        return partnerService.list(partnerFilter);
    }

    @GraphQLQuery(name = "partner_getById", description = "Gets a partner")
    public Optional<PartnerDto> getById(long partnerId) {
        return partnerService.getById(partnerId);
    }

    @GraphQLQuery(description = "list of partner connections of the provided partner")
    public CompletableFuture<Set<@GraphQLNonNull PartnerConnectionDto>> partnerConnections(
            @GraphQLContext PartnerDto partnerDto, @GraphQLEnvironment ResolutionEnvironment resolutionEnvironment) {
        return dataLoaderService.loadRelation(
                resolutionEnvironment, PartnerPartnerConnectionBatchLoader.class, partnerDto);
    }

    @GraphQLQuery
    public CompletableFuture<Set<@GraphQLNonNull CertificateContainerDto>> certificateContainers(
            @GraphQLContext PartnerDto partnerDto, @GraphQLEnvironment ResolutionEnvironment resolutionEnvironment) {
        return dataLoaderService.loadRelation(
                resolutionEnvironment, PartnerCertificateContainerBatchLoader.class, partnerDto);
    }
}
