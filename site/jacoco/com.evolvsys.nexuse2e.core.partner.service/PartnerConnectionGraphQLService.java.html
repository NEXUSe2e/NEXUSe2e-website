package com.evolvsys.nexuse2e.core.partner.service;

import com.evolvsys.app.core.exception.GeneralErrorException;
import com.evolvsys.app.core.graphql.annotation.GraphQLService;
import com.evolvsys.app.core.graphql.dataloader.DataLoaderService;
import com.evolvsys.app.core.persistence.model.IdDto;
import com.evolvsys.nexuse2e.core.certificate.model.CertificateContainerDto;
import com.evolvsys.nexuse2e.core.certificate.service.CertificateContainerBatchLoader;
import com.evolvsys.nexuse2e.core.config.service.ConfigGraphQLService;
import com.evolvsys.nexuse2e.core.parameter.model.ParameterDto;
import com.evolvsys.nexuse2e.core.partner.model.PartnerConnectionDto;
import com.evolvsys.nexuse2e.core.partner.model.PartnerDto;
import com.evolvsys.nexuse2e.core.partner.validation.ValidConnectionHasRoutes;
import com.evolvsys.nexuse2e.core.service.ServiceBatchLoader;
import com.evolvsys.nexuse2e.core.service.model.ServiceDto;
import io.leangen.graphql.annotations.*;
import io.leangen.graphql.execution.ResolutionEnvironment;
import jakarta.validation.Valid;
import java.util.List;
import java.util.Optional;
import java.util.Set;
import java.util.concurrent.CompletableFuture;
import java.util.function.Function;
import lombok.RequiredArgsConstructor;
import lombok.extern.slf4j.Slf4j;
import org.springframework.dao.DataIntegrityViolationException;

/**
 * @author Florian Drees
 * @since 02.01.2024
 */
@GraphQLService
@RequiredArgsConstructor
@Slf4j
public class PartnerConnectionGraphQLService {
    private final PartnerConnectionService partnerConnectionService;
    private final DataLoaderService dataLoaderService;

    @GraphQLMutation(
            name = "partnerConnection_update",
            description = "Updates an existing or creates a new partner connection")
    @GraphQLNonNull
    public PartnerConnectionDto update(@Valid @GraphQLNonNull PartnerConnectionDto partnerConnectionDto) {
        return ConfigGraphQLService.notifySubscribers(partnerConnectionService.save(partnerConnectionDto));
    }

    @GraphQLMutation(name = "partnerConnection_delete", description = "Deletes a connection")
    public boolean delete(@ValidConnectionHasRoutes long partnerConnectionId) {
        try {
            partnerConnectionService.delete(partnerConnectionId);
            ConfigGraphQLService.notifySubscribers();
        } catch (DataIntegrityViolationException e) {
            throw new GeneralErrorException(
                    "Cannot delete partner connection. Entities are referencing to this entity.", e);
        }
        return true;
    }

    @GraphQLQuery(name = "partnerConnection_getById", description = "Gets a partner connection")
    public Optional<PartnerConnectionDto> getById(long partnerConnectionId) {
        return partnerConnectionService.getById(partnerConnectionId);
    }

    @GraphQLQuery(description = "Gets the partner of the provided connection")
    public CompletableFuture<PartnerDto> partner(
            @GraphQLContext PartnerConnectionDto partnerConnectionDto,
            @GraphQLEnvironment ResolutionEnvironment resolutionEnvironment) {
        return dataLoaderService.load(
                resolutionEnvironment,
                PartnerBatchLoader.class,
                partnerConnectionDto,
                "partnerId",
                PartnerConnectionDto::getPartnerId);
    }

    @GraphQLQuery(description = "Gets the outflow service of the provided connection")
    public CompletableFuture<ServiceDto> outflowService(
            @GraphQLContext PartnerConnectionDto partnerConnectionDto,
            @GraphQLEnvironment ResolutionEnvironment resolutionEnvironment) {
        return dataLoaderService.load(
                resolutionEnvironment,
                ServiceBatchLoader.class,
                partnerConnectionDto,
                "outflowServiceId",
                PartnerConnectionDto::getOutflowServiceId);
    }

    @GraphQLQuery
    public CompletableFuture<CertificateContainerDto> certificateContainer(
            @GraphQLContext PartnerConnectionDto partnerConnectionDto,
            @GraphQLEnvironment ResolutionEnvironment resolutionEnvironment) {
        return dataLoaderService.load(
                resolutionEnvironment,
                CertificateContainerBatchLoader.class,
                partnerConnectionDto,
                "certificateContainerId",
                PartnerConnectionDto::getCertificateContainerId);
    }

    @GraphQLQuery
    public CompletableFuture<Set<@GraphQLNonNull ParameterDto>> parameters(
            @GraphQLContext PartnerConnectionDto partnerConnectionDto,
            @GraphQLEnvironment ResolutionEnvironment resolutionEnvironment) {
        return dataLoaderService.load(
                resolutionEnvironment,
                PartnerConnectionParameterBatchLoader.class,
                partnerConnectionDto,
                Function.identity());
    }

    @GraphQLQuery
    public CompletableFuture<List<@GraphQLNonNull Long>> requeueIntervals(
            @GraphQLContext PartnerConnectionDto partnerConnectionDto,
            @GraphQLEnvironment ResolutionEnvironment resolutionEnvironment) {
        return dataLoaderService.load(
                resolutionEnvironment,
                PartnerConnectionRequeueIntervalsBatchLoader.class,
                partnerConnectionDto,
                "requeueIntervals",
                IdDto::getId);
    }
}
