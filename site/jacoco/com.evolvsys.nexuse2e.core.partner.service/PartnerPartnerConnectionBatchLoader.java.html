package com.evolvsys.nexuse2e.core.partner.service;

import com.evolvsys.app.core.graphql.dataloader.BaseRelationBatchLoader;
import com.evolvsys.app.core.graphql.repository.BatchLoaderRepository;
import com.evolvsys.app.core.mapper.BaseDtoEntityMapper;
import com.evolvsys.nexuse2e.core.model.entity.PartnerConnectionEntity;
import com.evolvsys.nexuse2e.core.model.entity.PartnerEntity;
import com.evolvsys.nexuse2e.core.model.entity.QPartnerConnectionEntity;
import com.evolvsys.nexuse2e.core.partner.model.PartnerConnectionDto;
import com.querydsl.core.types.OrderSpecifier;
import org.springframework.stereotype.Component;

/**
 * <p>Batch-Loader zum Laden bestehender {@link PartnerConnectionEntity} einer {@link PartnerEntity}.</p>
 *
 * @author Florian Drees
 * @since 17.01.2024
 */
@Component
public class PartnerPartnerConnectionBatchLoader
        extends BaseRelationBatchLoader<PartnerConnectionDto, PartnerConnectionEntity, PartnerEntity> {

    public PartnerPartnerConnectionBatchLoader(
            BaseDtoEntityMapper<PartnerConnectionDto, PartnerConnectionEntity> mapper,
            BatchLoaderRepository batchLoaderRepository) {
        super(mapper, batchLoaderRepository, QPartnerConnectionEntity.partnerConnectionEntity.partner);
    }

    @Override
    protected OrderSpecifier<?>[] orderSpecifiers() {
        return new OrderSpecifier[] {QPartnerConnectionEntity.partnerConnectionEntity.connectionId.asc()};
    }
}
