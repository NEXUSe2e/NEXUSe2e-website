package com.evolvsys.nexuse2e.core.partner.service;

import com.evolvsys.app.core.auth.service.CurrentUserService;
import com.evolvsys.app.core.service.BaseService;
import com.evolvsys.app.core.service.ListService;
import com.evolvsys.nexuse2e.core.model.entity.PartnerEntity;
import com.evolvsys.nexuse2e.core.partner.PartnerFilter;
import com.evolvsys.nexuse2e.core.partner.mapper.PartnerMapper;
import com.evolvsys.nexuse2e.core.partner.model.PartnerConnectionDto;
import com.evolvsys.nexuse2e.core.partner.model.PartnerDto;
import com.evolvsys.nexuse2e.core.partner.repository.PartnerRepository;
import java.util.List;
import org.springframework.stereotype.Service;
import org.springframework.transaction.annotation.Transactional;

/**
 * @author Florian Drees
 * @since 15.01.2024
 */
@Service
public class PartnerService extends BaseService<PartnerDto, PartnerEntity, PartnerRepository> {
    private final ListService listService;

    public PartnerService(
            PartnerMapper mapper,
            PartnerRepository repository,
            CurrentUserService currentUserService,
            ListService listService) {
        super(mapper, repository, currentUserService);
        this.listService = listService;
    }

    @Override
    protected PartnerEntity createEntity() {
        return new PartnerEntity();
    }

    public List<PartnerDto> list(PartnerFilter partnerFilter) {
        return listService.list(partnerFilter, PartnerDto.class, repository::findAll).stream()
                .map(this::mapEntityToDto)
                .toList();
    }

    @Transactional
    public PartnerDto save(PartnerDto partnerDto, List<PartnerConnectionDto> connections) {
        PartnerEntity entity = getEntity(partnerDto);

        ((PartnerMapper) mapper).updateConnections(connections, entity, currentUserService.getCurrentUserData());

        entity = save(entity);

        return mapEntityToDto(entity);
    }
}
