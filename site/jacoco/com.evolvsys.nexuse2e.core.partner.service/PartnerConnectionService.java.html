package com.evolvsys.nexuse2e.core.partner.service;

import com.evolvsys.app.core.auth.service.CurrentUserService;
import com.evolvsys.app.core.service.BaseService;
import com.evolvsys.nexuse2e.core.model.entity.PartnerConnectionEntity;
import com.evolvsys.nexuse2e.core.partner.mapper.PartnerConnectionMapper;
import com.evolvsys.nexuse2e.core.partner.model.PartnerConnectionDto;
import com.evolvsys.nexuse2e.core.partner.repository.PartnerConnectionRepository;
import org.springframework.stereotype.Service;

/**
 * <p>Service for partner connections.</p>
 *
 * @author Florian Drees
 * @since 05.01.2024
 */
@Service
public class PartnerConnectionService
        extends BaseService<PartnerConnectionDto, PartnerConnectionEntity, PartnerConnectionRepository> {
    public PartnerConnectionService(
            PartnerConnectionMapper mapper,
            PartnerConnectionRepository repository,
            CurrentUserService currentUserService) {
        super(mapper, repository, currentUserService);
    }

    @Override
    protected PartnerConnectionEntity createEntity() {
        return new PartnerConnectionEntity();
    }
}
