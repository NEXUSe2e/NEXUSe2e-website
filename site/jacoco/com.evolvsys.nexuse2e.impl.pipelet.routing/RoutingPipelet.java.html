package com.evolvsys.nexuse2e.impl.pipelet.routing;

import com.evolvsys.nexuse2e.core.component.Pipelet;
import com.evolvsys.nexuse2e.core.message.context.ExecutionContext;
import com.evolvsys.nexuse2e.core.parameter.ParameterDefinition;
import com.evolvsys.nexuse2e.core.parameter.ParameterType;
import java.util.List;
import lombok.extern.slf4j.Slf4j;

/**
 * @author SvenBarden
 * @since 06.10.2023
 */
@Slf4j
public class RoutingPipelet extends Pipelet {
    public static final String ACTION_ID = "action";
    public static final String CHOREOGRAPHY_ID = "choreography";

    @Override
    public List<ParameterDefinition> parameterDefinitions() {
        return List.of(
                ParameterDefinition.builder()
                        .name(CHOREOGRAPHY_ID)
                        .displayName("Choreography ID")
                        .description("Static choreography id used for every message")
                        .type(ParameterType.STRING)
                        .required(true)
                        .build(),
                ParameterDefinition.builder()
                        .name(ACTION_ID)
                        .displayName("Action ID")
                        .description("Static action id used for every message")
                        .type(ParameterType.STRING)
                        .required(true)
                        .build());
    }

    @Override
    public ExecutionContext execute(ExecutionContext executionContext) {

        executionContext.setChoreographyId(parameters.value(CHOREOGRAPHY_ID));

        LOGGER.trace("setting choreographyId to {}", executionContext.getChoreographyId());

        executionContext.setActionId(parameters.value(ACTION_ID));

        LOGGER.trace("setting actionId to {}", executionContext.getActionId());

        return executionContext;
    }
}
