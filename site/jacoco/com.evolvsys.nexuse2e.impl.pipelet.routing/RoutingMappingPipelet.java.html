package com.evolvsys.nexuse2e.impl.pipelet.routing;

import com.evolvsys.nexuse2e.core.component.Pipelet;
import com.evolvsys.nexuse2e.core.exception.NexusException;
import com.evolvsys.nexuse2e.core.message.context.ExecutionContext;
import com.evolvsys.nexuse2e.core.parameter.ParameterDefinition;
import com.evolvsys.nexuse2e.core.parameter.ParameterType;
import com.evolvsys.nexuse2e.impl.service.support.StringMapperSupportService;
import java.util.List;
import java.util.function.Consumer;
import java.util.function.Supplier;
import lombok.extern.slf4j.Slf4j;

/**
 * Uses StringMapperSupportServices to map the routing information (choreographyId, actionId, sourcePartnerId and targetPartnerId) of a message.
 *
 * @author SvenBarden
 * @since 11.08.2025
 */
@Slf4j
public class RoutingMappingPipelet extends Pipelet {
    private static final String PARAM_CHOREOGRAPHY_MAPPER_SERVICE = "choreography_mapper_service";
    private static final String PARAM_ACTION_MAPPER_SERVICE = "action_mapper_service";
    private static final String SOURCE_PARTNER_MAPPER_SERVICE = "source_partner_mapper_service";
    private static final String TARGET_PARTNER_MAPPER_SERVICE = "target_partner_mapper_service";

    @Override
    public List<ParameterDefinition> parameterDefinitions() {
        return List.of(
                ParameterDefinition.builder()
                        .name(PARAM_CHOREOGRAPHY_MAPPER_SERVICE)
                        .displayName("Choreography mapper")
                        .description(
                                "Service to map the choreographyId to a different value. Skipped if no service is provided.")
                        .type(ParameterType.SUPPORT_SERVICE)
                        .build(),
                ParameterDefinition.builder()
                        .name(PARAM_ACTION_MAPPER_SERVICE)
                        .displayName("Action mapper")
                        .description(
                                "Service to map the actionId to a different value. Skipped if no service is provided.")
                        .type(ParameterType.SUPPORT_SERVICE)
                        .build(),
                ParameterDefinition.builder()
                        .name(SOURCE_PARTNER_MAPPER_SERVICE)
                        .displayName("Source partner mapper")
                        .description(
                                "Service to map the sourcePartnerId to a different value. Skipped if no service is provided.")
                        .type(ParameterType.SUPPORT_SERVICE)
                        .build(),
                ParameterDefinition.builder()
                        .name(TARGET_PARTNER_MAPPER_SERVICE)
                        .displayName("Target partner mapper")
                        .description(
                                "Service to map the targetPartnerId to a different value. Skipped if no service is provided.")
                        .type(ParameterType.SUPPORT_SERVICE)
                        .build());
    }

    @Override
    public ExecutionContext execute(ExecutionContext executionContext) throws NexusException {

        mapChoreography(executionContext);
        mapAction(executionContext);
        mapSourcePartner(executionContext);
        mapTargetPartner(executionContext);

        return executionContext;
    }

    private void map(String label, String paramName, Supplier<String> valueSupplier, Consumer<String> valueConsumer)
            throws NexusException {
        StringMapperSupportService mapper = parameters.value(paramName);

        if (mapper == null) {
            LOGGER.trace("{} mapper service is not defined, skipping mapping.", label);
        } else {
            String oldValue = valueSupplier.get();
            String newValue = mapper.map(oldValue);
            LOGGER.info("mapping {} '{}' to '{}'", label, oldValue, newValue);
            valueConsumer.accept(newValue);
        }
    }

    private void mapChoreography(ExecutionContext executionContext) throws NexusException {
        map(
                "choreographyId",
                PARAM_CHOREOGRAPHY_MAPPER_SERVICE,
                executionContext::getChoreographyId,
                executionContext::setChoreographyId);
    }

    private void mapAction(ExecutionContext executionContext) throws NexusException {
        map("actionId", PARAM_ACTION_MAPPER_SERVICE, executionContext::getActionId, executionContext::setActionId);
    }

    private void mapSourcePartner(ExecutionContext executionContext) throws NexusException {
        map(
                "sourcePartnerId",
                SOURCE_PARTNER_MAPPER_SERVICE,
                executionContext::getSourcePartnerId,
                executionContext::setSourcePartnerId);
    }

    private void mapTargetPartner(ExecutionContext executionContext) throws NexusException {
        map(
                "targetPartnerId",
                TARGET_PARTNER_MAPPER_SERVICE,
                executionContext::getTargetPartnerId,
                executionContext::setTargetPartnerId);
    }
}
