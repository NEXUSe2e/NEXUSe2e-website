package com.evolvsys.nexuse2e.core.filter;

import com.querydsl.core.types.Predicate;
import com.querydsl.core.types.dsl.BooleanPath;
import lombok.experimental.UtilityClass;

/**
 * @author Florian Drees
 * @since 11.04.2024
 */
@UtilityClass
public class BooleanQueryDelegateUtils {
    public static Predicate filter(BooleanPath path, BooleanFilterType value) {
        if (value != null) {
            return switch (value) {
                case TRUE -> path.isTrue();
                case FALSE -> path.isFalse();
                default -> null;
            };
        }
        return null;
    }
}
