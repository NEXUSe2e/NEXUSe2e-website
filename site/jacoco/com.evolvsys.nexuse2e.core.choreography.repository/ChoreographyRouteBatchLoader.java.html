package com.evolvsys.nexuse2e.core.choreography.repository;

import com.evolvsys.app.core.graphql.dataloader.BaseJoinBatchLoader;
import com.evolvsys.app.core.graphql.repository.BatchLoaderRepository;
import com.evolvsys.nexuse2e.core.model.entity.ChoreographyEntity;
import com.evolvsys.nexuse2e.core.model.entity.QChoreographyEntity;
import com.evolvsys.nexuse2e.core.model.entity.QRouteEntity;
import com.evolvsys.nexuse2e.core.model.entity.RouteEntity;
import com.evolvsys.nexuse2e.core.route.mapper.RouteMapper;
import com.evolvsys.nexuse2e.core.route.model.RouteDto;
import com.querydsl.core.types.OrderSpecifier;
import org.springframework.stereotype.Component;

/**
 * <p>Batch-Loader zum Laden bestehender {@link RouteEntity} einer {@link ChoreographyEntity}.</p>
 *
 * @author Florian Drees
 * @since 22.12.2023
 */
@Component
public class ChoreographyRouteBatchLoader extends BaseJoinBatchLoader<RouteDto, RouteEntity, QRouteEntity> {

    public ChoreographyRouteBatchLoader(RouteMapper mapper, BatchLoaderRepository batchLoaderRepository) {
        super(mapper, batchLoaderRepository, QChoreographyEntity.choreographyEntity.routes);
    }

    @Override
    protected OrderSpecifier<?>[] orderSpecifiers() {
        return new OrderSpecifier[] {new QRouteEntity("rel").id.asc()};
    }
}
