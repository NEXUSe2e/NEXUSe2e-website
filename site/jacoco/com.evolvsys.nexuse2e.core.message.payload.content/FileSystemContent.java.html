package com.evolvsys.nexuse2e.core.message.payload.content;

import com.evolvsys.nexuse2e.core.exception.NexusError;
import com.evolvsys.nexuse2e.core.exception.NexusRuntimeException;
import edu.umd.cs.findbugs.annotations.SuppressFBWarnings;
import java.io.File;
import java.io.IOException;
import java.io.InputStream;
import java.nio.file.Files;
import lombok.AccessLevel;
import lombok.AllArgsConstructor;
import org.apache.commons.io.FileUtils;

/**
 * @author SvenBarden
 * @since 22.02.2024
 */
@AllArgsConstructor(access = AccessLevel.PRIVATE)
@SuppressFBWarnings({"EXS_EXCEPTION_SOFTENING_NO_CONSTRAINTS", "EXS_EXCEPTION_SOFTENING_NO_CHECKED"})
final class FileSystemContent implements PayloadContent {
    @SuppressWarnings("PMD.MutableStaticState")
    static String TEMP_FILE_PREFIX = "NEXUSe2e";

    @SuppressWarnings("PMD.MutableStaticState")
    static String TEMP_FILE_SUFFIX = "payload";

    private final File file;

    static File createFile() throws IOException {
        return Files.createTempFile(TEMP_FILE_PREFIX, TEMP_FILE_SUFFIX).toFile();
    }

    public static FileSystemContent of(File file) {
        return new FileSystemContent(file);
    }

    public static FileSystemContent of(byte[] content) {
        try {
            File file = createFile();
            FileUtils.writeByteArrayToFile(file, content);
            return new FileSystemContent(file);
        } catch (IOException e) {
            throw new NexusRuntimeException(
                    NexusError.builder()
                            .message("could not generate content from byte array")
                            .build(),
                    e);
        }
    }

    public static FileSystemContent of(InputStream inputStream) throws IOException {
        try (inputStream) {
            File file = createFile();
            FileUtils.copyInputStreamToFile(inputStream, file);
            return new FileSystemContent(file);
        }
    }

    @Override
    public InputStream getInputStream() {
        try {
            return FileUtils.openInputStream(file);
        } catch (IOException e) {
            throw new NexusRuntimeException(
                    NexusError.builder()
                            .message("failed to open input stream for file")
                            .extension("file", file)
                            .build(),
                    e);
        }
    }

    @Override
    public byte[] getByteArray() {
        try {
            return Files.readAllBytes(file.toPath());
        } catch (IOException e) {
            throw new NexusRuntimeException(
                    NexusError.builder()
                            .message("failed to get byte array for file")
                            .extension("file", file)
                            .build(),
                    e);
        }
    }

    @Override
    public long getSize() {
        return file.length();
    }
}
