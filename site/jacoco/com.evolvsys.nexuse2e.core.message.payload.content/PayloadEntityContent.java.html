package com.evolvsys.nexuse2e.core.message.payload.content;

import com.evolvsys.nexuse2e.core.exception.NexusError;
import com.evolvsys.nexuse2e.core.exception.NexusRuntimeException;
import com.evolvsys.nexuse2e.core.message.payload.model.PayloadEntity;
import com.evolvsys.nexuse2e.core.message.payload.repository.BinaryRepository;
import edu.umd.cs.findbugs.annotations.SuppressFBWarnings;
import java.io.IOException;
import java.io.InputStream;
import java.sql.SQLException;
import lombok.AccessLevel;
import lombok.RequiredArgsConstructor;

/**
 * @author SvenBarden
 * @since 22.02.2024
 */
@RequiredArgsConstructor(access = AccessLevel.PROTECTED)
@SuppressFBWarnings({"EXS_EXCEPTION_SOFTENING_NO_CONSTRAINTS", "EXS_EXCEPTION_SOFTENING_NO_CHECKED"})
sealed class PayloadEntityContent implements PayloadContent permits CachingFileEntityContent {
    private final Long binaryId;
    private final long size;
    private final BinaryRepository binaryRepository;

    static PayloadEntityContent of(PayloadEntity payloadEntity, BinaryRepository binaryRepository) {
        return new PayloadEntityContent(payloadEntity.getContent().getId(), payloadEntity.getSize(), binaryRepository);
    }

    private InputStream loadContent() {
        try {
            return binaryRepository
                    .findById(binaryId)
                    .orElseThrow(() -> new NexusRuntimeException(NexusError.builder()
                            .message("binary file for payload could not be found")
                            .extension("id", binaryId)
                            .build()))
                    .getContent()
                    .getBinaryStream();
        } catch (SQLException e) {
            throw new NexusRuntimeException(
                    NexusError.builder()
                            .message("could not load payload binary entity")
                            .extension("id", binaryId)
                            .build(),
                    e);
        }
    }

    @Override
    public Long getId() {
        return binaryId;
    }

    @Override
    public InputStream getInputStream() {
        return loadContent();
    }

    @Override
    public byte[] getByteArray() {
        try (InputStream inputStream = loadContent()) {
            return inputStream.readAllBytes();
        } catch (IOException e) {
            throw new NexusRuntimeException(
                    NexusError.builder()
                            .message("could not get byte array from binary content")
                            .extension("id", binaryId)
                            .build(),
                    e);
        }
    }

    @Override
    public long getSize() {
        return size;
    }
}
