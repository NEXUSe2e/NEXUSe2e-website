package com.evolvsys.nexuse2e.core.message.payload.content;

import com.evolvsys.nexuse2e.core.message.payload.model.PayloadEntity;
import com.evolvsys.nexuse2e.core.message.payload.repository.BinaryRepository;
import java.io.InputStream;

/**
 * @author SvenBarden
 * @since 22.02.2024
 */
final class CachingFileEntityContent extends PayloadEntityContent {
    private InMemoryContent cache;

    private CachingFileEntityContent(Long binaryId, long size, BinaryRepository binaryRepository) {
        super(binaryId, size, binaryRepository);
    }

    static CachingFileEntityContent of(PayloadEntity payloadEntity, BinaryRepository binaryRepository) {
        return new CachingFileEntityContent(
                payloadEntity.getContent().getId(), payloadEntity.getSize(), binaryRepository);
    }

    @Override
    public byte[] getByteArray() {
        if (cache == null) {
            cache = InMemoryContent.of(super.getByteArray());
        }
        return cache.getByteArray();
    }

    @Override
    public InputStream getInputStream() {
        if (cache == null) {
            cache = InMemoryContent.of(super.getByteArray());
        }
        return cache.getInputStream();
    }
}
