package com.evolvsys.nexuse2e.core.message.payload.content;

import com.evolvsys.nexuse2e.core.exception.NexusError;
import com.evolvsys.nexuse2e.core.exception.NexusRuntimeException;
import edu.umd.cs.findbugs.annotations.SuppressFBWarnings;
import java.io.IOException;
import java.io.InputStream;
import java.nio.charset.Charset;
import org.apache.commons.io.IOUtils;

/**
 * @author SvenBarden
 * @since 22.02.2024
 */
@SuppressFBWarnings("EXS_EXCEPTION_SOFTENING_NO_CONSTRAINTS")
public sealed interface PayloadContent permits PayloadEntityContent, FileSystemContent, InMemoryContent {
    default Long getId() {
        return null;
    }

    InputStream getInputStream();

    byte[] getByteArray();

    long getSize();

    default String getString(Charset charset) {
        try {
            return IOUtils.toString(getInputStream(), charset);
        } catch (IOException e) {
            throw new NexusRuntimeException(
                    NexusError.builder()
                            .message("could not convert payloadContent to string")
                            .build(),
                    e);
        }
    }

    static PayloadContent of(byte[] content) {
        return PayloadContentFactory.create(content);
    }

    static PayloadContent of(InputStream content, long length) {
        return PayloadContentFactory.create(content, length);
    }

    static PayloadContent of(InputStream content) {
        return PayloadContentFactory.create(content);
    }

    static PayloadContentOutputStream createOutputStream() {
        return PayloadContentOutputStream.create();
    }
}
