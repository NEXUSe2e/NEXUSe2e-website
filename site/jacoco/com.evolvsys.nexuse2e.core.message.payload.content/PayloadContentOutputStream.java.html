package com.evolvsys.nexuse2e.core.message.payload.content;

import com.evolvsys.nexuse2e.core.exception.NexusError;
import com.evolvsys.nexuse2e.core.exception.NexusRuntimeException;
import edu.umd.cs.findbugs.annotations.SuppressFBWarnings;
import java.io.File;
import java.io.FileOutputStream;
import java.io.IOException;
import java.io.OutputStream;
import lombok.RequiredArgsConstructor;
import org.apache.commons.io.FileUtils;

/**
 * @author SvenBarden
 * @since 23.02.2024
 */
@RequiredArgsConstructor
@SuppressFBWarnings("EXS_EXCEPTION_SOFTENING_NO_CONSTRAINTS")
public final class PayloadContentOutputStream extends OutputStream {
    private final FileOutputStream fileOutputStream;
    private final File file;

    static PayloadContentOutputStream create() {
        try {
            File file = FileSystemContent.createFile();
            return new PayloadContentOutputStream(FileUtils.openOutputStream(file), file);
        } catch (IOException e) {
            throw new NexusRuntimeException(
                    NexusError.builder()
                            .message("could not create file outputStream for payload")
                            .build(),
                    e);
        }
    }

    @Override
    public void write(int b) throws IOException {
        fileOutputStream.write(b);
    }

    @Override
    public void write(byte[] b) throws IOException {
        fileOutputStream.write(b);
    }

    @Override
    public void write(byte[] b, int off, int len) throws IOException {
        fileOutputStream.write(b, off, len);
    }

    @Override
    public void flush() throws IOException {
        fileOutputStream.flush();
    }

    @Override
    public void close() throws IOException {
        super.close();
        fileOutputStream.close();
    }

    public PayloadContent finish() {
        try {
            close();
        } catch (IOException e) {
            throw new NexusRuntimeException(
                    NexusError.builder()
                            .message("could not close payload output stream")
                            .build(),
                    e);
        }
        return FileSystemContent.of(file);
    }
}
