package com.evolvsys.nexuse2e.core.message.payload.content;

import com.evolvsys.nexuse2e.core.config.NEXUSe2eConfigurationProperties;
import com.evolvsys.nexuse2e.core.exception.NexusError;
import com.evolvsys.nexuse2e.core.exception.NexusRuntimeException;
import com.evolvsys.nexuse2e.core.message.payload.model.PayloadEntity;
import com.evolvsys.nexuse2e.core.message.payload.repository.BinaryRepository;
import edu.umd.cs.findbugs.annotations.SuppressFBWarnings;
import java.io.IOException;
import java.io.InputStream;
import lombok.RequiredArgsConstructor;
import lombok.extern.slf4j.Slf4j;
import org.springframework.beans.factory.InitializingBean;
import org.springframework.stereotype.Component;
import org.springframework.util.unit.DataSize;

/**
 * @author SvenBarden
 * @since 23.02.2024
 */
@Component
@RequiredArgsConstructor
@Slf4j
@SuppressFBWarnings("EXS_EXCEPTION_SOFTENING_NO_CONSTRAINTS")
public class PayloadContentFactory implements InitializingBean {
    private static DataSize IN_MEMORY_THRESHOLD = DataSize.ofKilobytes(10);
    private final NEXUSe2eConfigurationProperties nexuSe2eConfigurationProperties;
    private final BinaryRepository binaryRepository;

    private static boolean doInMemory(long size) {
        return size < IN_MEMORY_THRESHOLD.toBytes();
    }

    public PayloadContent createFromEntity(PayloadEntity payloadEntity) {
        PayloadContent payloadContent;
        if (doInMemory(payloadEntity.getSize())) {
            LOGGER.trace(
                    "create new caching entity payload content from entity {} and binaryId {} with size {}",
                    payloadEntity.getId(),
                    payloadEntity.getContent().getId(),
                    DataSize.ofBytes(payloadEntity.getSize()));
            payloadContent = CachingFileEntityContent.of(payloadEntity, binaryRepository);
        } else {
            LOGGER.trace(
                    "create new entity payload content from entity {} and binaryId {} with size {}",
                    payloadEntity.getId(),
                    payloadEntity.getContent().getId(),
                    payloadEntity.getSize());
            payloadContent = PayloadEntityContent.of(payloadEntity, binaryRepository);
        }
        return payloadContent;
    }

    static PayloadContent create(InputStream inputStream) {
        PayloadContent payloadContent;
        try (inputStream) {
            try {
                LOGGER.trace("create new fileSystem payload content from input stream");
                payloadContent = FileSystemContent.of(inputStream);
            } catch (IOException e) {
                throw new NexusRuntimeException(
                        NexusError.builder()
                                .message("could not create file payload from inputStream")
                                .build(),
                        e);
            }
        } catch (IOException e) {
            throw new NexusRuntimeException(
                    NexusError.builder()
                            .message("could not close input stream for payloadContent")
                            .build(),
                    e);
        }
        return payloadContent;
    }

    static PayloadContent create(InputStream inputStream, long length) {
        PayloadContent payloadContent;
        try (inputStream) {
            if (doInMemory(length)) {
                try {
                    LOGGER.trace(
                            "create new inMemory payload content from input stream with length {}",
                            DataSize.ofBytes(length));
                    payloadContent = InMemoryContent.of(inputStream.readAllBytes());
                } catch (IOException e) {
                    throw new NexusRuntimeException(
                            NexusError.builder()
                                    .message("could not create inMemory payload from inputStream")
                                    .extension("size", length)
                                    .build(),
                            e);
                }
            } else {
                try {
                    LOGGER.trace(
                            "create new fileSystem payload content from input stream with length {}",
                            DataSize.ofBytes(length));
                    payloadContent = FileSystemContent.of(inputStream);
                } catch (IOException e) {
                    throw new NexusRuntimeException(
                            NexusError.builder()
                                    .message("could not create file payload from inputStream")
                                    .extension("size", length)
                                    .build(),
                            e);
                }
            }
        } catch (IOException e) {
            throw new NexusRuntimeException(
                    NexusError.builder()
                            .message("could not close input stream for payloadContent")
                            .build(),
                    e);
        }
        return payloadContent;
    }

    static PayloadContent create(byte[] content) {
        PayloadContent payloadContent;
        if (doInMemory(content.length)) {
            LOGGER.trace(
                    "create new inMemory payload content from byte array with length {}",
                    DataSize.ofBytes(content.length));
            payloadContent = InMemoryContent.of(content);
        } else {
            LOGGER.trace(
                    "create new fileSystem payload content from byte array with length {}",
                    DataSize.ofBytes(content.length));
            payloadContent = FileSystemContent.of(content);
        }
        return payloadContent;
    }

    static void setInMemoryThreshold(DataSize dataSize) {
        IN_MEMORY_THRESHOLD = dataSize;
    }

    @Override
    public void afterPropertiesSet() throws Exception {
        setInMemoryThreshold(nexuSe2eConfigurationProperties.getInMemoryPayloadThreshold());
    }
}
