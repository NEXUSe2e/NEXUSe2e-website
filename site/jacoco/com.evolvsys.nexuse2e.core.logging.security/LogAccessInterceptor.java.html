package com.evolvsys.nexuse2e.core.logging.security;

import com.evolvsys.app.core.auth.model.CurrentUserData;
import com.evolvsys.app.core.security.service.AccessInterceptor;
import com.evolvsys.nexuse2e.core.logging.model.LogDto;
import com.evolvsys.nexuse2e.core.model.dto.NexusTenantRelation;
import com.querydsl.core.types.dsl.BooleanExpression;
import com.querydsl.core.types.dsl.Expressions;
import org.springframework.stereotype.Component;

/**
 * @author SvenBarden
 * @since 05.04.2024
 */
@Component
public class LogAccessInterceptor implements AccessInterceptor<LogDto> {
    public static final String READ_LOG = "NEXUS.READ_LOG";
    public static final String WRITE_LOG = "NEXUS.WRITE_LOG";

    @Override
    public BooleanExpression getPredicate(CurrentUserData currentUserData) {
        boolean contains = currentUserData
                .getPermissionsForTenant(NexusTenantRelation.SYSTEM_TENANT_ID)
                .contains(READ_LOG);
        BooleanExpression booleanExpression;
        if (contains) {
            booleanExpression = null;
        } else {
            booleanExpression = Expressions.TRUE.isFalse();
        }
        return booleanExpression;
    }
}
