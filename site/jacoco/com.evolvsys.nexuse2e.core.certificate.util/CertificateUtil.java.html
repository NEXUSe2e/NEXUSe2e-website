package com.evolvsys.nexuse2e.core.certificate.util;

import java.nio.charset.StandardCharsets;
import java.security.cert.CertificateEncodingException;
import java.security.cert.X509Certificate;
import lombok.experimental.UtilityClass;
import org.bouncycastle.jcajce.provider.digest.MD5;
import org.bouncycastle.jcajce.provider.digest.SHA1;
import org.bouncycastle.jcajce.provider.digest.SHA256;
import org.bouncycastle.util.encoders.Hex;

/**
 * @author SvenBarden
 * @since 06.02.2024
 */
@UtilityClass
public class CertificateUtil {

    public static String getMD5Fingerprint(X509Certificate certificate) throws CertificateEncodingException {
        MD5.Digest digest = new MD5.Digest();

        digest.update(certificate.getEncoded(), 0, certificate.getEncoded().length);

        return new String(Hex.encode(digest.engineDigest()), StandardCharsets.UTF_8);
    }

    public static String getSha1Fingerprint(X509Certificate certificate) throws CertificateEncodingException {
        SHA1.Digest digest = new SHA1.Digest();

        digest.update(certificate.getEncoded(), 0, certificate.getEncoded().length);

        return new String(Hex.encode(digest.engineDigest()), StandardCharsets.UTF_8);
    }

    public static String getSha256Fingerprint(X509Certificate certificate) throws CertificateEncodingException {
        SHA256.Digest digest = new SHA256.Digest();

        digest.update(certificate.getEncoded(), 0, certificate.getEncoded().length);

        return new String(Hex.encode(digest.engineDigest()), StandardCharsets.UTF_8);
    }
}
