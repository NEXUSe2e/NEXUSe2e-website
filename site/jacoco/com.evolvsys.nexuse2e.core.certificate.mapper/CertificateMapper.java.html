package com.evolvsys.nexuse2e.core.certificate.mapper;

import com.evolvsys.app.core.binary.model.BinaryObjectDto;
import com.evolvsys.app.core.binary.service.BinaryType;
import com.evolvsys.app.core.file.mapper.FileMapper;
import com.evolvsys.app.core.mapper.BaseDtoEntityMapper;
import com.evolvsys.app.core.mapper.DefaultMapperConfiguration;
import com.evolvsys.app.core.mapper.interceptor.InterceptingMappingContext;
import com.evolvsys.nexuse2e.core.certificate.model.CertificateDto;
import com.evolvsys.nexuse2e.core.certificate.model.CertificateEntity;
import com.evolvsys.nexuse2e.core.certificate.repository.CertificateRepository;
import com.evolvsys.nexuse2e.core.certificate.service.CertificateBinaryType;
import com.evolvsys.nexuse2e.core.certificate.service.CertificateChainBinaryType;
import com.evolvsys.nexuse2e.core.certificate.service.CertificateFileOwnerProvider;
import com.evolvsys.nexuse2e.core.certificate.service.ClientCertificateKeystoreBinaryType;
import org.mapstruct.*;
import org.springframework.beans.factory.annotation.Autowired;

/**
 * @author SvenBarden
 * @since 07.02.2024
 */
@Mapper(
        config = DefaultMapperConfiguration.class,
        uses = {CertificateContainerMapper.class, CertificateInfoMapper.class})
public abstract class CertificateMapper extends BaseDtoEntityMapper<CertificateDto, CertificateEntity> {
    private CertificateRepository certificateRepository;
    private FileMapper fileMapper;

    @Autowired
    protected void config(CertificateRepository certificateRepository, FileMapper fileMapper) {
        this.certificateRepository = certificateRepository;
        this.fileMapper = fileMapper;
    }

    @Override
    public CertificateEntity idToEntity(Long id, @Context InterceptingMappingContext interceptingMappingContext) {
        return idToEntity(id, interceptingMappingContext, certificateRepository::findById);
    }

    @Mapping(target = "certificateContainerId", source = "certificateContainer.id")
    @Mapping(target = "certificateBinaryObject", ignore = true)
    @Mapping(target = "certificateChainBinaryObject", ignore = true)
    @Mapping(target = "certificateKeystoreBinaryObject", ignore = true)
    @Override
    @InheritConfiguration
    @Mapping(target = "file", ignore = true)
    protected abstract CertificateDto entityToDto(
            CertificateEntity entity, @Context InterceptingMappingContext interceptingMappingContext);

    @AfterMapping
    protected void afterEntityToDto(@MappingTarget CertificateDto certificateDto) {
        certificateDto.setCertificateBinaryObject(BinaryObjectDto.builder()
                .fileName(CertificateBinaryType.getFileName(
                        certificateDto.getInfo().getCommonName()))
                .path(BinaryType.getPath(
                        CertificateBinaryType.NAME, certificateDto.getId().toString()))
                .mimeType("application/zip")
                .size(-1)
                .build());
        certificateDto.setCertificateChainBinaryObject(BinaryObjectDto.builder()
                .fileName(CertificateChainBinaryType.getFileName(
                        certificateDto.getInfo().getCommonName()))
                .path(BinaryType.getPath(
                        CertificateChainBinaryType.NAME, certificateDto.getId().toString()))
                .mimeType("application/x-pem-file")
                .size(-1)
                .build());
        certificateDto.setCertificateKeystoreBinaryObject(BinaryObjectDto.builder()
                .fileName(ClientCertificateKeystoreBinaryType.getFileName(certificateDto.getName()))
                .path(BinaryType.getPath(
                        ClientCertificateKeystoreBinaryType.NAME,
                        certificateDto.getId().toString()))
                .mimeType("application/zip")
                .size(-1)
                .build());
    }

    @Mapping(target = "info", ignore = true)
    @Mapping(target = "certificateContainer", source = "certificateContainerId")
    @Override
    @InheritConfiguration
    @Mapping(target = "file", ignore = true)
    protected abstract void updateEntity(
            CertificateDto dto,
            @MappingTarget CertificateEntity entity,
            @Context InterceptingMappingContext interceptingMappingContext);

    @AfterMapping
    protected void fileToDto(CertificateEntity certificateEntity, @MappingTarget CertificateDto certificateDto) {
        if (certificateEntity.getFile() != null) {
            certificateDto.setFile(fileMapper.entityToDto(
                    certificateEntity.getFile(), CertificateFileOwnerProvider.FILE, certificateEntity.getId()));
        }
    }

    @AfterMapping
    protected void fileToEntity(CertificateDto certificateDto, @MappingTarget CertificateEntity certificateEntity) {
        fileMapper.dtoToEntity(certificateDto.getFile(), certificateEntity::getFile, certificateEntity::setFile);
    }
}
