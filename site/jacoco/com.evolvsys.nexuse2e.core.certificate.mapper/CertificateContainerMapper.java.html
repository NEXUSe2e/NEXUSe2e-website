package com.evolvsys.nexuse2e.core.certificate.mapper;

import com.evolvsys.app.core.mapper.BaseDtoEntityMapper;
import com.evolvsys.app.core.mapper.DefaultMapperConfiguration;
import com.evolvsys.app.core.mapper.interceptor.InterceptingMappingContext;
import com.evolvsys.nexuse2e.core.certificate.model.CertificateContainerDto;
import com.evolvsys.nexuse2e.core.certificate.model.CertificateContainerEntity;
import com.evolvsys.nexuse2e.core.certificate.repository.CertificateContainerRepository;
import com.evolvsys.nexuse2e.core.partner.mapper.BasePartnerMapper;
import org.mapstruct.*;
import org.springframework.beans.factory.annotation.Autowired;

/**
 * @author SvenBarden
 * @since 07.02.2024
 */
@Mapper(config = DefaultMapperConfiguration.class, uses = BasePartnerMapper.class)
public abstract class CertificateContainerMapper
        extends BaseDtoEntityMapper<CertificateContainerDto, CertificateContainerEntity> {
    private CertificateContainerRepository certificateContainerRepository;

    @Autowired
    protected void config(CertificateContainerRepository certificateContainerRepository) {
        this.certificateContainerRepository = certificateContainerRepository;
    }

    public String toString(CertificateContainerEntity certificateContainerEntity) {
        return certificateContainerEntity == null ? null : certificateContainerEntity.getName();
    }

    @Override
    public CertificateContainerEntity idToEntity(
            Long id, @Context InterceptingMappingContext interceptingMappingContext) {
        return idToEntity(id, interceptingMappingContext, certificateContainerRepository::findById);
    }

    @Mapping(target = "partnerId", source = "partner.id")
    @Override
    @InheritConfiguration
    protected abstract CertificateContainerDto entityToDto(
            CertificateContainerEntity entity, @Context InterceptingMappingContext interceptingMappingContext);

    @Mapping(target = "partner", source = "partnerId")
    @Override
    @InheritConfiguration
    protected abstract void updateEntity(
            CertificateContainerDto dto,
            @MappingTarget CertificateContainerEntity entity,
            @Context InterceptingMappingContext interceptingMappingContext);
}
