package com.evolvsys.nexuse2e.core.certificate.mapper;

import com.evolvsys.nexuse2e.core.certificate.model.CertificateContainerEntity;
import com.evolvsys.nexuse2e.core.certificate.model.CertificateEntity;
import com.evolvsys.nexuse2e.core.certificate.model.CertificateType;
import com.evolvsys.nexuse2e.core.certificate.model.ContextKeyStore;
import com.evolvsys.nexuse2e.core.certificate.repository.CertificateContainerRepository;
import com.evolvsys.nexuse2e.core.certificate.repository.CertificateRepository;
import com.evolvsys.nexuse2e.core.certificate.service.CertificateKeystoreUtils;
import com.evolvsys.nexuse2e.core.exception.NexusError;
import com.evolvsys.nexuse2e.core.exception.NexusException;
import com.evolvsys.nexuse2e.core.exception.NexusRuntimeException;
import edu.umd.cs.findbugs.annotations.SuppressFBWarnings;
import java.security.KeyStore;
import java.security.cert.X509Certificate;
import java.util.Collections;
import java.util.List;
import lombok.RequiredArgsConstructor;
import org.springframework.stereotype.Component;
import org.springframework.transaction.annotation.Transactional;

/**
 * @author SvenBarden
 * @since 12.02.2024
 */
@Component
@RequiredArgsConstructor
public class CertificateContainerKeyMapper {
    private final CertificateContainerRepository certificateContainerRepository;
    private final CertificateRepository certificateRepository;
    private final CertificateInfoMapper certificateInfoMapper;

    @SuppressFBWarnings("EXS_EXCEPTION_SOFTENING_NO_CONSTRAINTS")
    private ContextKeyStore certificateEntityToStore(CertificateEntity certificateEntity) {
        try {
            KeyStore keyStore = CertificateKeystoreUtils.toKeyStore(certificateEntity);
            return ContextKeyStore.builder()
                    .keyStore(keyStore)
                    .password(certificateEntity.getPassword())
                    .build();
        } catch (NexusException e) {
            throw new NexusRuntimeException(e.getError(), e);
        }
    }

    private ContextKeyStore entityToKeyStore(CertificateContainerEntity certificateContainerEntity) {
        return certificateRepository
                .findFirstByCertificateContainerAndActive(certificateContainerEntity, true)
                .map(this::certificateEntityToStore)
                .orElse(null);
    }

    @Transactional(readOnly = true)
    public ContextKeyStore idToKeyStore(Long id) {
        return id == null
                ? null
                : certificateContainerRepository
                        .findById(id)
                        .map(this::entityToKeyStore)
                        .orElse(null);
    }

    @SuppressFBWarnings("EXS_EXCEPTION_SOFTENING_NO_CONSTRAINTS")
    private X509Certificate entityToCert(CertificateEntity certificateEntity) {
        try {
            return certificateEntity.getType() == CertificateType.LOCAL
                    ? certificateInfoMapper.getPrivateCert(certificateEntity)
                    : certificateInfoMapper.getPublicCert(certificateEntity);
        } catch (NexusException e) {
            throw new NexusRuntimeException(
                    NexusError.builder()
                            .message("could not get cert from entity")
                            .build(),
                    e);
        }
    }

    private List<X509Certificate> entityToCertificates(CertificateContainerEntity certificateContainerEntity) {
        return certificateRepository.findByCertificateContainerAndActive(certificateContainerEntity, true).stream()
                .map(this::entityToCert)
                .toList();
    }

    @Transactional(readOnly = true)
    public List<X509Certificate> idToCertificates(Long id) {
        return id == null
                ? Collections.emptyList()
                : certificateContainerRepository
                        .findById(id)
                        .map(this::entityToCertificates)
                        .orElse(Collections.emptyList());
    }
}
