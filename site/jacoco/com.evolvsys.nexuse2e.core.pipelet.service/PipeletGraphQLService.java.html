package com.evolvsys.nexuse2e.core.pipelet.service;

import com.evolvsys.app.core.graphql.annotation.GraphQLService;
import com.evolvsys.app.core.graphql.dataloader.DataLoaderService;
import com.evolvsys.nexuse2e.core.config.service.ConfigGraphQLService;
import com.evolvsys.nexuse2e.core.parameter.model.ParameterDto;
import com.evolvsys.nexuse2e.core.pipelet.PipeletParameterBatchLoader;
import com.evolvsys.nexuse2e.core.pipelet.model.PipeletDto;
import io.leangen.graphql.annotations.*;
import io.leangen.graphql.execution.ResolutionEnvironment;
import jakarta.validation.Valid;
import java.util.Optional;
import java.util.Set;
import java.util.concurrent.CompletableFuture;
import java.util.function.Function;
import lombok.RequiredArgsConstructor;
import lombok.extern.slf4j.Slf4j;

/**
 * @author Florian Drees
 * @since 04.12.2023
 */
@GraphQLService
@RequiredArgsConstructor
@Slf4j
public class PipeletGraphQLService {
    private final PipeletService pipeletService;
    private final DataLoaderService dataLoaderService;

    @GraphQLMutation(name = "pipelet_update", description = "save parameters of a pipelet")
    @GraphQLNonNull
    public PipeletDto update(@GraphQLNonNull @Valid PipeletDto pipeletDto) {
        return ConfigGraphQLService.notifySubscribers(pipeletService.save(pipeletDto));
    }

    @GraphQLQuery(name = "pipelet_getById", description = "Gets a pipelet")
    public Optional<PipeletDto> getById(long pipeletId) {
        return pipeletService.getById(pipeletId);
    }

    @GraphQLQuery(description = "list of parameters of the provided pipelet")
    public CompletableFuture<Set<@GraphQLNonNull ParameterDto>> parameters(
            @GraphQLContext PipeletDto pipeletDto, @GraphQLEnvironment ResolutionEnvironment resolutionEnvironment) {
        LOGGER.trace("getting parameters for pipelet {}", pipeletDto);
        return dataLoaderService.load(
                resolutionEnvironment, PipeletParameterBatchLoader.class, pipeletDto, Function.identity());
    }
}
