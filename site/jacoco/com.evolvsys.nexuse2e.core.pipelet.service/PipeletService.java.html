package com.evolvsys.nexuse2e.core.pipelet.service;

import com.evolvsys.app.core.auth.service.CurrentUserService;
import com.evolvsys.app.core.service.BaseService;
import com.evolvsys.nexuse2e.core.model.entity.PipeletEntity;
import com.evolvsys.nexuse2e.core.pipelet.PipeletRepository;
import com.evolvsys.nexuse2e.core.pipelet.mapper.PipeletMapper;
import com.evolvsys.nexuse2e.core.pipelet.model.PipeletDto;
import org.springframework.stereotype.Service;

/**
 * @author Florian Drees
 * @since 04.12.2023
 */
@Service
public class PipeletService extends BaseService<PipeletDto, PipeletEntity, PipeletRepository> {
    public PipeletService(PipeletMapper mapper, PipeletRepository repository, CurrentUserService currentUserService) {
        super(mapper, repository, currentUserService);
    }

    @Override
    public PipeletEntity getEntity(PipeletDto dto) {
        return super.getEntity(dto, getRepository()::getEagerById);
    }

    @Override
    protected PipeletEntity createEntity() {
        return new PipeletEntity();
    }
}
