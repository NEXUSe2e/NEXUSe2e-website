package com.evolvsys.nexuse2e.core.pipelet.service;

import com.evolvsys.app.core.graphql.dataloader.BaseJoinBatchLoader;
import com.evolvsys.app.core.graphql.repository.BatchLoaderRepository;
import com.evolvsys.nexuse2e.core.model.entity.PipeletEntity;
import com.evolvsys.nexuse2e.core.model.entity.QPipeletEntity;
import com.evolvsys.nexuse2e.core.pipelet.mapper.PipeletMapper;
import com.evolvsys.nexuse2e.core.pipelet.model.PipeletDto;
import com.evolvsys.nexuse2e.core.pipeline.model.PipelineEntity;
import com.evolvsys.nexuse2e.core.pipeline.model.QPipelineEntity;
import com.querydsl.core.types.OrderSpecifier;
import org.springframework.stereotype.Component;

/**
 * <p>Batch-Loader zum Laden bestehender {@link PipeletEntity} einer {@link PipelineEntity}.</p>
 *
 * @author Florian Drees
 * @since 27.11.2023
 */
@Component
public class PipeletBatchLoader extends BaseJoinBatchLoader<PipeletDto, PipeletEntity, QPipeletEntity> {
    public PipeletBatchLoader(PipeletMapper mapper, BatchLoaderRepository batchLoaderRepository) {
        super(mapper, batchLoaderRepository, QPipelineEntity.pipelineEntity.pipelets);
    }

    @Override
    protected OrderSpecifier<?>[] orderSpecifiers() {
        return new OrderSpecifier[] {new QPipeletEntity("rel").order.asc()};
    }
}
