package com.evolvsys.nexuse2e.core.config.validation;

import jakarta.validation.ConstraintValidator;
import jakarta.validation.ConstraintValidatorContext;
import java.util.Objects;
import lombok.extern.slf4j.Slf4j;
import org.apache.commons.lang3.StringUtils;

/**
 * @author SvenBarden
 * @since 26.04.2024
 */
@Slf4j
class ValidConfigClientKeyNameConstraintValidator implements ConstraintValidator<ValidConfigClientKeyName, String> {
    @Override
    public boolean isValid(String value, ConstraintValidatorContext context) {
        boolean isValid;
        if (ValidationContextHolder.exists()) {
            if (StringUtils.isBlank(value)) {
                isValid = true;
            } else {
                isValid = ValidationContextHolder.getContext().getConfig().getClientKeyContainers().stream()
                        .anyMatch(
                                clientKeyContainerConfig -> Objects.equals(clientKeyContainerConfig.getName(), value));
            }
        } else {
            LOGGER.warn("no context found to validate client key container name for {}", value);
            isValid = false;
        }
        return isValid;
    }
}
