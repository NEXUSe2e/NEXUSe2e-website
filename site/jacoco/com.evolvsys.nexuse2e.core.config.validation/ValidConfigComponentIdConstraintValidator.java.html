package com.evolvsys.nexuse2e.core.config.validation;

import com.evolvsys.nexuse2e.core.component.ComponentRegistry;
import com.evolvsys.nexuse2e.core.component.ComponentType;
import jakarta.validation.ConstraintValidator;
import jakarta.validation.ConstraintValidatorContext;
import lombok.RequiredArgsConstructor;
import org.apache.commons.lang3.ArrayUtils;
import org.apache.commons.lang3.ObjectUtils;
import org.apache.commons.lang3.StringUtils;
import org.springframework.beans.factory.config.ConfigurableBeanFactory;
import org.springframework.context.annotation.Scope;
import org.springframework.stereotype.Component;

/**
 * @author SvenBarden
 * @since 25.04.2024
 */
@Component
@RequiredArgsConstructor
@Scope(ConfigurableBeanFactory.SCOPE_PROTOTYPE)
class ValidConfigComponentIdConstraintValidator implements ConstraintValidator<ValidConfigComponentId, String> {
    private final ComponentRegistry componentRegistry;
    private ComponentType[] types;

    @Override
    public void initialize(ValidConfigComponentId constraintAnnotation) {
        synchronized (this) {
            types = ObjectUtils.defaultIfNull(constraintAnnotation.types(), ComponentType.values());
        }
    }

    @Override
    public boolean isValid(String value, ConstraintValidatorContext context) {
        boolean isValid;
        if (StringUtils.isBlank(value)) {
            isValid = true;
        } else {
            isValid = componentRegistry
                    .getComponent(value)
                    .map(registeredNexusComponent ->
                            ArrayUtils.contains(types, registeredNexusComponent.getComponentType()))
                    .orElse(Boolean.FALSE);
        }
        return isValid;
    }
}
