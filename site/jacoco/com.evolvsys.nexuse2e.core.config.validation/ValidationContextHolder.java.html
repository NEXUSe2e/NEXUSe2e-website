package com.evolvsys.nexuse2e.core.config.validation;

import lombok.experimental.UtilityClass;

/**
 * @author SvenBarden
 * @since 25.04.2024
 */
@UtilityClass
public class ValidationContextHolder {
    private static final ThreadLocal<ValidationContext> CONTEXT_HOLDER = new ThreadLocal<>();

    public static void clearContext() {
        CONTEXT_HOLDER.remove();
    }

    public static boolean exists() {
        return CONTEXT_HOLDER.get() != null;
    }

    public static ValidationContext getContext() {
        return CONTEXT_HOLDER.get();
    }

    public static void setContext(ValidationContext context) {
        CONTEXT_HOLDER.set(context);
    }
}
