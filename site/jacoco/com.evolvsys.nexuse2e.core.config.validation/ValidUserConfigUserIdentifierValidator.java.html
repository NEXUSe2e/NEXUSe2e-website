package com.evolvsys.nexuse2e.core.config.validation;

import com.evolvsys.nexuse2e.core.config.model.UserConfig;
import jakarta.validation.ConstraintValidator;
import jakarta.validation.ConstraintValidatorContext;
import lombok.extern.slf4j.Slf4j;
import org.apache.commons.lang3.StringUtils;

@Slf4j
class ValidUserConfigUserIdentifierValidator implements ConstraintValidator<ValidUserConfigUserIdentifier, UserConfig> {
    @Override
    public boolean isValid(UserConfig value, ConstraintValidatorContext context) {
        return value == null || !StringUtils.isAllBlank(value.getUserName(), value.getEmail());
    }
}
