package com.evolvsys.nexuse2e.core.config.validation;

import com.evolvsys.nexuse2e.core.config.model.NexusConfig;
import com.evolvsys.nexuse2e.core.config.model.RouteConfig;
import jakarta.validation.ConstraintValidator;
import jakarta.validation.ConstraintValidatorContext;
import java.util.Objects;
import lombok.extern.slf4j.Slf4j;
import org.apache.commons.lang3.StringUtils;

/**
 * @author SvenBarden
 * @since 26.04.2024
 */
@Slf4j
class ValidConfigRouteConnectionIdConstraintValidator
        implements ConstraintValidator<ValidConfigRouteConnectionId, RouteConfig> {
    private ValidConfigRouteConnectionId.Target target;

    private boolean isValid(String partnerId, String connectionId) {
        NexusConfig nexusConfig = ValidationContextHolder.getContext().getConfig();

        boolean isValid;
        if (StringUtils.isAllBlank(partnerId, connectionId)) {
            isValid = true;
        } else if (StringUtils.isBlank(partnerId)) {
            isValid = false;
        } else if (StringUtils.isBlank(connectionId)) {
            isValid = true;
        } else {
            isValid = nexusConfig.getPartners().stream()
                    .filter(partnerConfig -> Objects.equals(partnerConfig.getPartnerId(), partnerId))
                    .flatMap(partnerConfig -> partnerConfig.getConnections().stream())
                    .anyMatch(connectionConfig -> Objects.equals(connectionConfig.getConnectionId(), connectionId));
        }
        return isValid;
    }

    @Override
    public void initialize(ValidConfigRouteConnectionId constraintAnnotation) {
        target = constraintAnnotation.target();
    }

    @Override
    public boolean isValid(RouteConfig value, ConstraintValidatorContext context) {
        boolean isValid;
        if (ValidationContextHolder.exists()) {
            isValid = switch (target) {
                case FIRST_PARTNER -> isValid(value.getFirstPartner(), value.getFirstPartnerConnection());
                case SECOND_PARTNER -> isValid(value.getSecondPartner(), value.getSecondPartnerConnection());
            };
        } else {
            LOGGER.warn("no context found to validate connectionId for route {}", value);
            isValid = true;
        }
        return isValid;
    }
}
