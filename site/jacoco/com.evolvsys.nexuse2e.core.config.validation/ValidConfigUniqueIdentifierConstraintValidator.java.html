package com.evolvsys.nexuse2e.core.config.validation;

import com.evolvsys.nexuse2e.core.exception.NexusError;
import com.evolvsys.nexuse2e.core.exception.NexusRuntimeException;
import edu.umd.cs.findbugs.annotations.SuppressFBWarnings;
import jakarta.validation.ConstraintValidator;
import jakarta.validation.ConstraintValidatorContext;
import java.util.Collection;
import java.util.LinkedHashMap;
import java.util.List;
import java.util.Map;
import java.util.function.Function;
import one.util.streamex.StreamEx;
import org.apache.commons.beanutils.BeanUtils;
import org.apache.commons.collections4.CollectionUtils;

/**
 * @author SvenBarden
 * @since 25.04.2024
 */
class ValidConfigUniqueIdentifierConstraintValidator
        implements ConstraintValidator<ValidConfigUniqueIdentifier, Collection<?>> {

    private String[] propertyNames;
    private boolean ignoreNullValues;

    @Override
    public void initialize(ValidConfigUniqueIdentifier constraintAnnotation) {
        propertyNames = constraintAnnotation.propertyNames();
        ignoreNullValues = constraintAnnotation.ignoreNullValues();
    }

    @Override
    public boolean isValid(Collection<?> values, ConstraintValidatorContext context) {
        boolean isValid;

        if (CollectionUtils.isEmpty(values)) {
            isValid = true;
        } else {
            List<?> filteredValues = values.stream()
                    .filter(ignoreNullValues ? this::hasNonNullPropertyValue : v -> true)
                    .toList();
            isValid = StreamEx.of(filteredValues).distinct(distinctFunction()).count() == filteredValues.size();
        }

        return isValid;
    }

    private boolean hasNonNullPropertyValue(Object value) {
        boolean allPropertiesNull = true;
        for (String propertyName : propertyNames) {
            if (getPropertyValue(value, propertyName) != null) {
                allPropertiesNull = false;
                break;
            }
        }
        return !allPropertiesNull;
    }

    private Function<Object, Map<String, String>> distinctFunction() {
        return o -> {
            Map<String, String> map = new LinkedHashMap<>(propertyNames.length);
            for (String propertyName : propertyNames) {
                map.put(propertyName, getPropertyValue(o, propertyName));
            }
            return map;
        };
    }

    @SuppressFBWarnings("EXS_EXCEPTION_SOFTENING_NO_CONSTRAINTS")
    private String getPropertyValue(Object o, String propertyName) {
        String propertyValue;
        try {
            propertyValue = BeanUtils.getProperty(o, propertyName);
        } catch (Throwable e) {
            throw new NexusRuntimeException(
                    NexusError.builder()
                            .message("Invalid unique id property found")
                            .extension("propertyName", propertyName)
                            .extension("value", o)
                            .build(),
                    e);
        }
        return propertyValue;
    }
}
