package com.evolvsys.nexuse2e.core.config.validation;

import com.evolvsys.nexuse2e.core.component.ComponentRegistry;
import com.evolvsys.nexuse2e.core.config.model.ConnectionConfig;
import com.evolvsys.nexuse2e.core.config.model.ParameterConfig;
import com.evolvsys.nexuse2e.core.parameter.ParameterDefinition;
import jakarta.validation.ConstraintValidator;
import jakarta.validation.ConstraintValidatorContext;
import java.util.List;
import java.util.Objects;
import lombok.RequiredArgsConstructor;
import lombok.extern.slf4j.Slf4j;
import org.apache.commons.collections4.CollectionUtils;
import org.apache.commons.lang3.StringUtils;
import org.springframework.stereotype.Component;

/**
 * @author SvenBarden
 * @since 25.04.2024
 */
@Component
@RequiredArgsConstructor
@Slf4j
class ValidPartnerConnectionParametersConstraintValidator
        implements ConstraintValidator<ValidPartnerConnectionParameters, ConnectionConfig> {
    private final ComponentRegistry componentRegistry;

    @Override
    public boolean isValid(ConnectionConfig value, ConstraintValidatorContext context) {
        boolean isValid;
        if (ValidationContextHolder.exists()) {
            ValidationContext validationContext = ValidationContextHolder.getContext();
            List<ParameterConfig> parameters = value.getParameters();
            if (CollectionUtils.isEmpty(parameters)) {
                isValid = true;
            } else {
                if (StringUtils.isBlank(value.getOutflowService())) {
                    isValid = false;
                } else {
                    List<String> parameterNames = validationContext.getConfig().getServices().stream()
                            .filter(serviceConfig ->
                                    Objects.equals(serviceConfig.getServiceId(), value.getOutflowService()))
                            .flatMap(serviceConfig ->
                                    componentRegistry.getComponent(serviceConfig.getComponentId()).stream())
                            .flatMap(registeredNexusComponent ->
                                    registeredNexusComponent.getParameterDefinitions().stream())
                            .map(ParameterDefinition::getName)
                            .toList();
                    isValid = parameters.stream()
                            .filter(parameterConfig -> StringUtils.isNotBlank(parameterConfig.getName()))
                            .allMatch(parameterConfig -> parameterNames.contains(parameterConfig.getName()));
                }
            }
        } else {
            LOGGER.warn("no context found to validate parameters for connection {}", value);
            isValid = true;
        }
        return isValid;
    }
}
