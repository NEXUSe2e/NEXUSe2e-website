package com.evolvsys.nexuse2e.core.config.validation;

import com.evolvsys.nexuse2e.core.component.ComponentRegistry;
import com.evolvsys.nexuse2e.core.config.model.ConfigurableConfig;
import com.evolvsys.nexuse2e.core.config.model.ParameterConfig;
import com.evolvsys.nexuse2e.core.parameter.ParameterDefinition;
import jakarta.validation.ConstraintValidator;
import jakarta.validation.ConstraintValidatorContext;
import java.lang.annotation.Annotation;
import java.util.List;
import lombok.RequiredArgsConstructor;
import org.apache.commons.collections4.CollectionUtils;
import org.apache.commons.lang3.StringUtils;

/**
 * @author SvenBarden
 * @since 16.09.2024
 */
@RequiredArgsConstructor
abstract class ValidParametersConstraintValidator<A extends Annotation, T extends ConfigurableConfig>
        implements ConstraintValidator<A, T> {
    private final ComponentRegistry componentRegistry;

    @Override
    public boolean isValid(T value, ConstraintValidatorContext context) {
        boolean isValid;
        List<ParameterConfig> parameters = value.getParameters();
        if (StringUtils.isBlank(value.getComponentId()) || CollectionUtils.isEmpty(parameters)) {
            isValid = true;
        } else {
            List<String> parameterNames = componentRegistry.getComponent(value.getComponentId()).stream()
                    .flatMap(registeredNexusComponent -> registeredNexusComponent.getParameterDefinitions().stream())
                    .map(ParameterDefinition::getName)
                    .toList();
            isValid = parameters.stream()
                    .filter(parameterConfig -> StringUtils.isNotBlank(parameterConfig.getName()))
                    .allMatch(parameterConfig -> parameterNames.contains(parameterConfig.getName()));
        }
        return isValid;
    }
}
