package com.evolvsys.nexuse2e.core.config.validation;

import com.evolvsys.app.core.role.model.RoleEntity;
import com.evolvsys.app.core.role.repository.RoleRepository;
import jakarta.validation.ConstraintValidator;
import jakarta.validation.ConstraintValidatorContext;
import lombok.RequiredArgsConstructor;
import org.apache.commons.lang3.StringUtils;
import org.springframework.data.domain.Example;
import org.springframework.data.domain.ExampleMatcher;
import org.springframework.stereotype.Component;

/**
 * @author SvenBarden
 * @since 29.05.2024
 */
@Component
@RequiredArgsConstructor
public class ValidRoleNameConstraintValidator implements ConstraintValidator<ValidRoleName, String> {
    private final RoleRepository roleRepository;

    @Override
    public boolean isValid(String value, ConstraintValidatorContext context) {
        boolean valid;
        if (StringUtils.isBlank(value)) {
            valid = true;
        } else {
            RoleEntity search = new RoleEntity();
            search.setInternalName(value);
            valid = roleRepository.exists(
                    Example.of(search, ExampleMatcher.matching().withIgnorePaths("hierarchical")));
        }
        return valid;
    }
}
