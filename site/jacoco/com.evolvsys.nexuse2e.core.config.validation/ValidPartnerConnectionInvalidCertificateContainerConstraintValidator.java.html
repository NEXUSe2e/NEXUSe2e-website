package com.evolvsys.nexuse2e.core.config.validation;

import com.evolvsys.nexuse2e.core.config.model.ConnectionConfig;
import jakarta.validation.ConstraintValidator;
import jakarta.validation.ConstraintValidatorContext;
import lombok.extern.slf4j.Slf4j;
import org.apache.commons.lang3.StringUtils;

/**
 * @author SvenBarden
 * @since 25.04.2024
 */
@Slf4j
class ValidPartnerConnectionInvalidCertificateContainerConstraintValidator
        implements ConstraintValidator<ValidPartnerConnectionInvalidCertificateContainer, ConnectionConfig> {
    @Override
    public boolean isValid(ConnectionConfig value, ConstraintValidatorContext context) {
        boolean isValid;
        if (ValidationContextHolder.exists()) {
            if (StringUtils.isBlank(value.getCertificateContainer())) {
                isValid = true;
            } else {
                ValidationContext validationContext = ValidationContextHolder.getContext();
                isValid = validationContext.getConfig().getPartners().stream()
                        .filter(partnerConfig -> partnerConfig.getConnections().contains(value))
                        .flatMap(partnerConfig -> partnerConfig.getCertificateContainers().stream())
                        .anyMatch(certificateContainerConfig ->
                                certificateContainerConfig.getName().equals(value.getCertificateContainer()));
            }
        } else {
            LOGGER.warn("no context found to validate certificate container for connection {}", value);
            isValid = true;
        }
        return isValid;
    }
}
