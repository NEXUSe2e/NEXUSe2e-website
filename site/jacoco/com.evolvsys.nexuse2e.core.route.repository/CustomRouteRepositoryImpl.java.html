package com.evolvsys.nexuse2e.core.route.repository;

import com.evolvsys.nexuse2e.core.action.model.QActionEntity;
import com.evolvsys.nexuse2e.core.choreography.repository.ChoreographyRepository;
import com.evolvsys.nexuse2e.core.model.entity.QChoreographyEntity;
import com.evolvsys.nexuse2e.core.model.entity.QRouteEntity;
import com.evolvsys.nexuse2e.core.model.entity.RouteEntity;
import com.querydsl.jpa.JPQLQueryFactory;
import java.util.Optional;
import lombok.RequiredArgsConstructor;
import org.springframework.stereotype.Repository;

/**
 * @author SvenBarden
 * @since 15.02.2024
 */
@Repository
@RequiredArgsConstructor
class CustomRouteRepositoryImpl implements CustomRouteRepository {
    private final JPQLQueryFactory queryFactory;
    private final ChoreographyRepository choreographyRepository;

    @Override
    public Optional<RouteEntity> findForChoreography(String firstPartner, String secondPartner, Long choreographyId) {
        QRouteEntity routeEntity = QRouteEntity.routeEntity;
        QChoreographyEntity choreographyEntity = QChoreographyEntity.choreographyEntity;

        return Optional.ofNullable(queryFactory
                .select(routeEntity)
                .from(choreographyEntity)
                .join(choreographyEntity.routes, routeEntity)
                .on(routeEntity.forPartner(firstPartner, secondPartner))
                .where(choreographyEntity.id.eq(choreographyId))
                .fetchOne());
    }

    @Override
    public Optional<RouteEntity> findForAction(
            String firstPartner, String secondPartner, Long choreographyId, Long actionId) {
        QRouteEntity routeEntity = QRouteEntity.routeEntity;
        QChoreographyEntity choreographyEntity = QChoreographyEntity.choreographyEntity;
        QActionEntity actionEntity = QActionEntity.actionEntity;

        return Optional.ofNullable(queryFactory
                .select(routeEntity)
                .from(actionEntity)
                .join(actionEntity.routes, routeEntity)
                .on(routeEntity.forPartner(firstPartner, secondPartner))
                .join(actionEntity.choreography, choreographyEntity)
                .on(choreographyEntity.id.eq(choreographyId))
                .where(actionEntity.id.eq(actionId))
                .fetchOne());
    }
}
