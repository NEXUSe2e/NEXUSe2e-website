package com.evolvsys.nexuse2e.core.route.repository;

import com.evolvsys.nexuse2e.core.model.entity.QRouteEntity;
import com.evolvsys.nexuse2e.core.model.entity.RouteEntity;
import com.evolvsys.nexuse2e.core.route.RouteFilter;
import com.querydsl.core.BooleanBuilder;
import com.querydsl.core.annotations.QueryDelegate;
import com.querydsl.core.types.Predicate;
import com.querydsl.core.types.dsl.Expressions;
import org.apache.commons.lang3.StringUtils;
import org.springframework.lang.Nullable;

/**
 * @author Florian Drees
 * @since 09.04.2024
 */
public final class RouteQueryDelegate {
    private RouteQueryDelegate() {}

    @QueryDelegate(RouteEntity.class)
    public static Predicate forPartner(QRouteEntity routeEntity, String firstPartnerId, String secondPartnerId) {
        return Expressions.anyOf(
                Expressions.allOf(
                        routeEntity.firstPartner.partnerId.eq(firstPartnerId),
                        routeEntity.secondPartner.partnerId.eq(secondPartnerId)),
                Expressions.allOf(
                        routeEntity.firstPartner.partnerId.eq(secondPartnerId),
                        routeEntity.secondPartner.partnerId.eq(firstPartnerId)));
    }

    @QueryDelegate(RouteEntity.class)
    public static Predicate filter(QRouteEntity qRouteEntity, @Nullable RouteFilter routeFilter) {
        BooleanBuilder builder = new BooleanBuilder();
        if (routeFilter != null) {
            if (StringUtils.isNotEmpty(routeFilter.getRouteId())) {
                builder.and(qRouteEntity.routeId.contains(routeFilter.getRouteId()));
            }
            if (StringUtils.isNotEmpty(routeFilter.getDescription())) {
                builder.and(qRouteEntity.description.contains(routeFilter.getDescription()));
            }
            if (StringUtils.isNotEmpty(routeFilter.getPartnerId())) {
                builder.and(Expressions.anyOf(
                        qRouteEntity.firstPartner.partnerId.contains(routeFilter.getPartnerId()),
                        qRouteEntity.secondPartner.partnerId.contains(routeFilter.getPartnerId())));
            }
        }
        return builder.getValue();
    }
}
